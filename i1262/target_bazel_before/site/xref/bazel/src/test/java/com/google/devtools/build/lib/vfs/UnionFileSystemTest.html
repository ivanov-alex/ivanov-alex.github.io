54    private void setupDelegateFileSystems() {
55      inDelegate = new XAttrInMemoryFs(BlazeClock.instance());
56      outDelegate = new XAttrInMemoryFs(BlazeClock.instance());
57      defaultDelegate = new XAttrInMemoryFs(BlazeClock.instance());
58  
59      unionfs = createDefaultUnionFileSystem();
60    }

62    private UnionFileSystem createDefaultUnionFileSystem() {
63      return createDefaultUnionFileSystem(false);
64    }

66    private UnionFileSystem createDefaultUnionFileSystem(boolean readOnly) {
67      return new UnionFileSystem(ImmutableMap.<PathFragment, FileSystem>of(
68          new PathFragment("/in"), inDelegate,
69          new PathFragment("/out"), outDelegate),
70          defaultDelegate, readOnly);
71    }

87    @Test
88    public void testBasicDelegation() throws Exception {
89      unionfs = createDefaultUnionFileSystem();
90      Path fooPath = unionfs.getPath("/foo");
91      Path inPath = unionfs.getPath("/in");
92      Path outPath = unionfs.getPath("/out/in.txt");
93      assertSame(inDelegate, unionfs.getDelegate(inPath));
94      assertSame(outDelegate, unionfs.getDelegate(outPath));
95      assertSame(defaultDelegate, unionfs.getDelegate(fooPath));
96    }

99    public void testBasicXattr() throws Exception {
100     Path fooPath = unionfs.getPath("/foo");
101     Path inPath = unionfs.getPath("/in");
102     Path outPath = unionfs.getPath("/out/in.txt");
103 
104     assertArrayEquals(XATTR_VAL.getBytes(UTF_8), inPath.getxattr(XATTR_KEY));
105     assertArrayEquals(XATTR_VAL.getBytes(UTF_8), outPath.getxattr(XATTR_KEY));
106     assertArrayEquals(XATTR_VAL.getBytes(UTF_8), fooPath.getxattr(XATTR_KEY));
107     assertNull(inPath.getxattr("not_key"));
108     assertNull(outPath.getxattr("not_key"));
109     assertNull(fooPath.getxattr("not_key"));
110   }

122   // Check for appropriate registration and lookup of delegate filesystems based

137   // Checks that files cannot be modified when the filesystem is created
138   // read-only, even if the delegate filesystems are read/write.
139   @Test
140   public void testModificationFlag() throws Exception {
141     assertTrue(unionfs.supportsModifications());
142     Path outPath = unionfs.getPath("/out/foo.txt");
143     assertTrue(unionfs.createDirectory(outPath.getParentDirectory()));
144     OutputStream outFile = unionfs.getOutputStream(outPath);
145     outFile.write('b');
146     outFile.close();
147 
148     unionfs.setExecutable(outPath, true);
149 
150     // Note that this does not destroy the underlying filesystems;
151     // UnionFileSystem is just a view.
152     unionfs = createDefaultUnionFileSystem(true);
153     assertFalse(unionfs.supportsModifications());
154 
155     InputStream outFileInput = unionfs.getInputStream(outPath);
156     int outFileByte = outFileInput.read();
157     outFileInput.close();
158     assertEquals('b', outFileByte);
159 
160     assertTrue(unionfs.isExecutable(outPath));
161 
162     // Modifying files through the unionfs isn't permitted, even if the
163     // delegates are read/write.
164     try {
165       unionfs.setExecutable(outPath, false);
166       fail("Modification to a read-only UnionFileSystem succeeded.");
167     } catch (UnsupportedOperationException expected) {
168       // OK - should fail.
169     }
170   }

172   // Checks that roots of delegate filesystems are created outside of the
173   // delegate filesystems; i.e. they can be seen from the filesystem of the parent.
174   @Test
175   public void testDelegateRootDirectoryCreation() throws Exception {
176     Path foo = unionfs.getPath("/foo");
177     Path bar = unionfs.getPath("/bar");
178     Path out = unionfs.getPath("/out");
179     assertTrue(unionfs.createDirectory(foo));
180     assertTrue(unionfs.createDirectory(bar));
181     assertTrue(unionfs.createDirectory(out));
182     Path outFile = unionfs.getPath("/out/in");
183     FileSystemUtils.writeContentAsLatin1(outFile, "Out");
184 
185     // FileSystemTest.setUp() silently creates the test root on the filesystem...
186     Path testDirUnderRoot = unionfs.getPath(workingDir.asFragment().subFragment(0, 1));
187     assertThat(unionfs.getDirectoryEntries(unionfs.getRootDirectory())).containsExactly(foo, bar,
188         out, testDirUnderRoot);
189     assertThat(unionfs.getDirectoryEntries(out)).containsExactly(outFile);
190 
191     assertSame(unionfs.getDelegate(foo), defaultDelegate);
192     assertEquals(foo.asFragment(), unionfs.adjustPath(foo, defaultDelegate).asFragment());
193     assertSame(unionfs.getDelegate(bar), defaultDelegate);
194     assertSame(unionfs.getDelegate(outFile), outDelegate);
195     assertSame(unionfs.getDelegate(out), outDelegate);
196 
197     // As a fragment (i.e. without filesystem or root info), the path name should be preserved.
198     assertEquals(outFile.asFragment(), unionfs.adjustPath(outFile, outDelegate).asFragment());
199   }

210   // Basic *explicit* cross-filesystem symlink check.
211   // Note: This does not work implicitly yet, as the next test illustrates.
212   @Test
213   public void testCrossDeviceSymlinks() throws Exception {
214     assertTrue(unionfs.createDirectory(unionfs.getPath("/out")));
215 
216     // Create an "/in" directory directly on the output delegate to bypass the
217     // UnionFileSystem's mapping.
218     assertTrue(inDelegate.getPath("/in").createDirectory());
219     OutputStream outStream = inDelegate.getPath("/in/bar.txt").getOutputStream();
220     outStream.write('i');
221     outStream.close();
222 
223     Path outFoo = unionfs.getPath("/out/foo");
224     unionfs.createSymbolicLink(outFoo, new PathFragment("../in/bar.txt"));
225     assertTrue(unionfs.stat(outFoo, false).isSymbolicLink());
226 
227     try {
228       unionfs.stat(outFoo, true).isFile();
229       fail("Stat on cross-device symlink succeeded!");
230     } catch (FileNotFoundException expected) {
231       // OK
232     }
233 
234     Path resolved = unionfs.resolveSymbolicLinks(outFoo);
235     assertSame(unionfs, resolved.getFileSystem());
236     InputStream barInput = resolved.getInputStream();
237     int barChar = barInput.read();
238     barInput.close();
239     assertEquals('i', barChar);
240   }

253   // Prefix mappings can apply to files starting with a prefix within a directory.
254   @Test
255   public void testWithinDirectoryMapping() throws Exception {
256     unionfs = new UnionFileSystem(ImmutableMap.<PathFragment, FileSystem>of(
257         new PathFragment("/fruit/a"), inDelegate,
258         new PathFragment("/fruit/b"), outDelegate), defaultDelegate);
259     assertTrue(unionfs.createDirectory(unionfs.getPath("/fruit")));
260     assertTrue(defaultDelegate.getPath("/fruit").isDirectory());
261     assertTrue(inDelegate.getPath("/fruit").createDirectory());
262     assertTrue(outDelegate.getPath("/fruit").createDirectory());
263 
264     Path apple = unionfs.getPath("/fruit/apple");
265     Path banana = unionfs.getPath("/fruit/banana");
266     Path cherry = unionfs.getPath("/fruit/cherry");
267     unionfs.createDirectory(apple);
268     unionfs.createDirectory(banana);
269     assertSame(inDelegate, unionfs.getDelegate(apple));
270     assertSame(outDelegate, unionfs.getDelegate(banana));
271     assertSame(defaultDelegate, unionfs.getDelegate(cherry));
272 
273     FileSystemUtils.writeContentAsLatin1(apple.getRelative("table"), "penny");
274     FileSystemUtils.writeContentAsLatin1(banana.getRelative("nana"), "nanana");
275     FileSystemUtils.writeContentAsLatin1(cherry, "garcia");
276 
277     assertEquals("penny", new String(
278         FileSystemUtils.readContentAsLatin1(inDelegate.getPath("/fruit/apple/table"))));
279     assertEquals("nanana", new String(
280         FileSystemUtils.readContentAsLatin1(outDelegate.getPath("/fruit/banana/nana"))));
281     assertEquals("garcia", new String(
282         FileSystemUtils.readContentAsLatin1(defaultDelegate.getPath("/fruit/cherry"))));
283   }

285   // Write using the VFS through a UnionFileSystem and check that the file can
286   // be read back in the same location using standard Java IO.
287   // There is a similar test in UnixFileSystem, but this is essential to ensure
288   // that paths aren't being remapped in some nasty way on the underlying FS.
289   @Test
290   public void testDelegateOperationsReflectOnLocalFilesystem() throws Exception {
291     unionfs = new UnionFileSystem(ImmutableMap.<PathFragment, FileSystem>of(
292         workingDir.getParentDirectory().asFragment(), new UnixFileSystem()),
293         defaultDelegate, false);
294     // This is a child of the current tmpdir, and doesn't exist on its own.
295     // It would be created in setup(), but of course, that didn't use a UnixFileSystem.
296     unionfs.createDirectory(workingDir);
297     Path testFile = unionfs.getPath(workingDir.getRelative("test_file").asFragment());
298     assertTrue(testFile.asFragment().startsWith(workingDir.asFragment()));
299     String testString = "This is a test file";
300     FileSystemUtils.writeContentAsLatin1(testFile, testString);
301     try {
302       assertEquals(testString, new String(FileSystemUtils.readContentAsLatin1(testFile)));
303     } finally {
304       testFile.delete();
305       assertTrue(unionfs.delete(workingDir));
306     }
307   }

319   private static class XAttrInMemoryFs extends InMemoryFileSystem {
320     public XAttrInMemoryFs(Clock clock) {
321       super(clock);
322     }
323 
324     @Override
325     protected byte[] getxattr(Path path, String name, boolean followSymlinks) {
326       assertSame(this, path.getFileSystem());
327       return (name.equals(XATTR_KEY)) ? XATTR_VAL.getBytes(UTF_8) : null;
328     }
