49    private static FilesetEntry makeFilesetEntry() {
50      try {
51        return new FilesetEntry(Label.parseAbsolute("//foo:bar"),
52                                Lists.<Label>newArrayList(), Lists.newArrayList("xyz"), "",
53                                FilesetEntry.SymlinkBehavior.COPY, ".");
54      } catch (Label.SyntaxException e) {
55        throw new RuntimeException("Bad label: ", e);
56      }
57    }

59    @Test
60    public void testPrinter() throws Exception {
61      // Note that prettyPrintValue and printValue only differ on behaviour of
62      // labels and strings at toplevel.
63      assertEquals("foo\nbar", Printer.str("foo\nbar"));
64      assertEquals("\"foo\\nbar\"", Printer.repr("foo\nbar"));
65      assertEquals("'", Printer.str("'"));
66      assertEquals("\"'\"", Printer.repr("'"));
67      assertEquals("\"", Printer.str("\""));
68      assertEquals("\"\\\"\"", Printer.repr("\""));
69      assertEquals("3", Printer.str(3));
70      assertEquals("3", Printer.repr(3));
71      assertEquals("None", Printer.repr(Environment.NONE));
72  
73      assertEquals("//x:x", Printer.str(Label.parseAbsolute("//x")));
74      assertEquals("\"//x:x\"", Printer.repr(Label.parseAbsolute("//x")));
75  
76      List<?> list = makeList("foo", "bar");
77      List<?> tuple = makeTuple("foo", "bar");
78  
79      assertEquals("(1, [\"foo\", \"bar\"], 3)",
80                   Printer.str(makeTuple(1, list, 3)));
81      assertEquals("(1, [\"foo\", \"bar\"], 3)",
82                   Printer.repr(makeTuple(1, list, 3)));
83      assertEquals("[1, (\"foo\", \"bar\"), 3]",
84                   Printer.str(makeList(1, tuple, 3)));
85      assertEquals("[1, (\"foo\", \"bar\"), 3]",
86                   Printer.repr(makeList(1, tuple, 3)));
87  
88      Map<Object, Object> dict = ImmutableMap.<Object, Object>of(
89          1, tuple,
90          2, list,
91          "foo", makeList());
92      assertEquals("{1: (\"foo\", \"bar\"), 2: [\"foo\", \"bar\"], \"foo\": []}",
93                  Printer.str(dict));
94      assertEquals("{1: (\"foo\", \"bar\"), 2: [\"foo\", \"bar\"], \"foo\": []}",
95                  Printer.repr(dict));
96      assertEquals("FilesetEntry(srcdir = \"//foo:bar\", files = [], "
97                 + "excludes = [\"xyz\"], destdir = \"\", "
98                 + "strip_prefix = \".\", symlinks = \"copy\")",
99                   Printer.repr(makeFilesetEntry()));
100   }

102   private void checkFormatPositionalFails(String errorMessage, String format, Object... arguments) {
103     try {
104       Printer.format(format, arguments);
105       fail();
106     } catch (IllegalFormatException e) {
107       assertThat(e).hasMessage(errorMessage);
108     }
109   }

111   @Test
112   public void testFormatPositional() throws Exception {
113     assertEquals("foo 3", Printer.formatString("%s %d", makeTuple("foo", 3)));
114     assertEquals("foo 3", Printer.format("%s %d", "foo", 3));
115 
116     // Note: formatString doesn't perform scalar x -> (x) conversion;
117     // The %-operator is responsible for that.
118     assertThat(Printer.formatString("", makeTuple())).isEmpty();
119     assertEquals("foo", Printer.format("%s", "foo"));
120     assertEquals("3.14159", Printer.format("%s", 3.14159));
121     checkFormatPositionalFails("not all arguments converted during string formatting",
122         "%s", 1, 2, 3);
123     assertEquals("%foo", Printer.format("%%%s", "foo"));
124     checkFormatPositionalFails("not all arguments converted during string formatting",
125         "%%s", "foo");
126     checkFormatPositionalFails("unsupported format character \" \" at index 1 in \"% %s\"",
127         "% %s", "foo");
128     assertEquals("[1, 2, 3]", Printer.format("%s", makeList(1, 2, 3)));
129     assertEquals("(1, 2, 3)", Printer.format("%s", makeTuple(1, 2, 3)));
130     assertEquals("[]", Printer.format("%s", makeList()));
131     assertEquals("()", Printer.format("%s", makeTuple()));
132     assertEquals("% 1 \"2\" 3", Printer.format("%% %d %r %s", 1, "2", "3"));
133 
134     checkFormatPositionalFails(
135         "invalid argument \"1\" for format pattern %d",
136         "%d", "1");
137     checkFormatPositionalFails("unsupported format character \".\" at index 1 in \"%.3g\"",
138         "%.3g");
139     checkFormatPositionalFails("unsupported format character \".\" at index 1 in \"%.3g\"",
140         "%.3g", 1, 2);
141     checkFormatPositionalFails("unsupported format character \".\" at index 1 in \"%.s\"",
142         "%.s");
143   }

155   private FilesetEntry createStripPrefixFilesetEntry(String stripPrefix)  throws Exception {
156     Label label = Label.parseAbsolute("//x");
157     return new FilesetEntry(
158         label,
159         Arrays.asList(label),
160         Arrays.<String>asList(),
161         "",
162         FilesetEntry.SymlinkBehavior.DEREFERENCE,
163         stripPrefix);
164   }

166   @Test
167   public void testFilesetEntryStripPrefixAttr() throws Exception {
168     FilesetEntry withoutStripPrefix = createStripPrefixFilesetEntry(".");
169     FilesetEntry withStripPrefix = createStripPrefixFilesetEntry("orange");
170 
171     String prettyWithout = Printer.repr(withoutStripPrefix);
172     String prettyWith = Printer.repr(withStripPrefix);
173 
174     assertThat(prettyWithout).contains("strip_prefix = \".\"");
175     assertThat(prettyWith).contains("strip_prefix = \"orange\"");
176   }

178   @Test
179   public void testRegressionCrashInPrettyPrintValue() throws Exception {
180     // Would cause crash in code such as this:
181     //  Fileset(name='x', entries=[], out=[FilesetEntry(files=['a'])])
182     // While formatting the "expected x, got y" message for the 'out'
183     // attribute, prettyPrintValue(FilesetEntry) would be recursively called
184     // with a List<Label> even though this isn't a valid datatype in the
185     // interpreter.
186     // Fileset isn't part of bazel, even though FilesetEntry is.
187     assertEquals(createExpectedFilesetEntryString('"'), Printer.repr(createTestFilesetEntry()));
188   }

190   @Test
191   public void testSingleQuotes() throws Exception {
192     assertThat(Printer.str("test", '\'')).isEqualTo("test");
193     assertThat(Printer.repr("test", '\'')).isEqualTo("'test'");
194 
195     assertEquals("'\\''", Printer.repr("'", '\''));
196     assertEquals("\"", Printer.str("\"", '\''));
197     assertEquals("'\"'", Printer.repr("\"", '\''));
198 
199     List<?> list = makeList("foo", "bar");
200     List<?> tuple = makeTuple("foo", "bar");
201 
202     assertThat(Printer.str(makeTuple(1, list, 3), '\'')).isEqualTo("(1, ['foo', 'bar'], 3)");
203     assertThat(Printer.repr(makeTuple(1, list, 3), '\'')).isEqualTo("(1, ['foo', 'bar'], 3)");
204     assertThat(Printer.str(makeList(1, tuple, 3), '\'')).isEqualTo("[1, ('foo', 'bar'), 3]");
205     assertThat(Printer.repr(makeList(1, tuple, 3), '\'')).isEqualTo("[1, ('foo', 'bar'), 3]");
206 
207     Map<Object, Object> dict =
208         ImmutableMap.<Object, Object>of(1, tuple, 2, list, "foo", makeList());
209 
210     assertThat(Printer.str(dict, '\''))
211         .isEqualTo("{1: ('foo', 'bar'), 2: ['foo', 'bar'], 'foo': []}");
212     assertThat(Printer.repr(dict, '\''))
213         .isEqualTo("{1: ('foo', 'bar'), 2: ['foo', 'bar'], 'foo': []}");
214 
215     assertThat(Printer.repr(createTestFilesetEntry(), '\''))
216         .isEqualTo(createExpectedFilesetEntryString('\''));
217   }

219   private String createExpectedFilesetEntryString(
220       FilesetEntry.SymlinkBehavior symlinkBehavior, char quotationMark) {
221     return String.format(
222         "FilesetEntry(srcdir = %1$c//x:x%1$c,"
223         + " files = [%1$c//x:x%1$c],"
224         + " excludes = [],"
225         + " destdir = %1$c%1$c,"
226         + " strip_prefix = %1$c.%1$c,"
227         + " symlinks = %1$c%2$s%1$c)",
228         quotationMark, symlinkBehavior.toString().toLowerCase());
229   }
230 
231   private String createExpectedFilesetEntryString(char quotationMark) {
232     return createExpectedFilesetEntryString(FilesetEntry.SymlinkBehavior.COPY, quotationMark);
233   }

235   private FilesetEntry createTestFilesetEntry(FilesetEntry.SymlinkBehavior symlinkBehavior)
236       throws SyntaxException {
237     Label label = Label.parseAbsolute("//x");
238     return new FilesetEntry(
239         label, Arrays.asList(label), Arrays.<String>asList(), "", symlinkBehavior, ".");
240   }

242   private FilesetEntry createTestFilesetEntry() throws SyntaxException {
243     return createTestFilesetEntry(FilesetEntry.SymlinkBehavior.COPY);
244   }
