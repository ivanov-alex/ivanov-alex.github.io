27  /**
28   * Factory class for generating artifacts which are used as intermediate output.
29   */
30  // TODO(bazel-team): This should really be named DerivedArtifacts as it contains methods for
31  // final as well as intermediate artifacts.
32  final class IntermediateArtifacts {
33  
34    /**
35     * Extension used on the temporary dsym bundle location. Must end in {@code .dSYM} for dsymutil
36     * to generate a plist file.
37     */
38    static final String TMP_DSYM_BUNDLE_SUFFIX = ".temp.app.dSYM";
39  
40    private final AnalysisEnvironment analysisEnvironment;
41    private final Root binDirectory;
42    private final Label ownerLabel;
43    private final String archiveFileNameSuffix;
44  
45    /**
46     * Label to scope the output paths of generated artifacts, in addition to {@link #ownerLabel}.
47     */
48    private final Optional<Label> scopingLabel;
49  
50    IntermediateArtifacts(
51        AnalysisEnvironment analysisEnvironment, Root binDirectory, Label ownerLabel,
52        String archiveFileNameSuffix) {
53      this.analysisEnvironment = Preconditions.checkNotNull(analysisEnvironment);
54      this.binDirectory = Preconditions.checkNotNull(binDirectory);
55      this.ownerLabel = Preconditions.checkNotNull(ownerLabel);
56      this.scopingLabel = Optional.<Label>absent();
57      this.archiveFileNameSuffix = Preconditions.checkNotNull(archiveFileNameSuffix);
58    }
59  
60    IntermediateArtifacts(
61        AnalysisEnvironment analysisEnvironment, Root binDirectory, Label ownerLabel,
62        Label scopingLabel, String archiveFileNameSuffix) {
63      this.analysisEnvironment = Preconditions.checkNotNull(analysisEnvironment);
64      this.binDirectory = Preconditions.checkNotNull(binDirectory);
65      this.ownerLabel = Preconditions.checkNotNull(ownerLabel);
66      this.scopingLabel = Optional.of(Preconditions.checkNotNull(scopingLabel));
67      this.archiveFileNameSuffix = Preconditions.checkNotNull(archiveFileNameSuffix);
68    }

70    /**
71     * Returns a derived artifact in the bin directory obtained by appending some extension to the end
72     * of the given {@link PathFragment}.
73     */
74    private Artifact appendExtension(PathFragment original, String extension) {
75      return analysisEnvironment.getDerivedArtifact(
76          FileSystemUtils.appendExtension(original, extension), binDirectory);
77    }
78  
79    /**
80     * Returns a derived artifact in the bin directory obtained by appending some extension to the end
81     * of the {@link PathFragment} corresponding to the owner {@link Label}.
82     */
83    private Artifact appendExtension(String extension) {
84      return appendExtension(labelScopedDir(), extension);
85    }

131    * The artifact which is the binary (or library) which is comprised of one or more .a files linked
132    * together. It also contains full debug symbol information, compared to the artifact returned
133    * by {@link #strippedSingleArchitectureBinary}. This artifact will serve as input for the symbol

149   /**
150    * The {@code .a} file which contains all the compiled sources for a rule.
151    */
152   public Artifact archive() {
153     PathFragment labelPath = labelScopedDir();
154     PathFragment rootRelative = labelPath
155         .getParentDirectory()
156         .getRelative(String.format("lib%s%s.a", labelPath.getBaseName(), archiveFileNameSuffix));
157     return analysisEnvironment.getDerivedArtifact(rootRelative, binDirectory);
158   }
159 
160   /**
161    * The debug symbol bundle file which contains debug symbols generated by dsymutil.
162    */
163   public Artifact dsymBundle() {
164     return appendExtension(TMP_DSYM_BUNDLE_SUFFIX);
165   }
166 
167   /**
168    * Returns a unique directory scoped by {@code ownerLabel} and {@code scopingLabel}. Normally
169    * when {@code scopingLabel} is absent, the returned directory is just a path fragment
170    * containing the package and the name of the ownerLabel. If scopingLabel is present, the
171    * returned directory is scoped by scopingLabel first. For example, if ownerLabel is
172    * //a/b:c and scopingLabel is //d/e:f, the returned directory will be: d/e/a/b/c/f/.
173    */
174   private PathFragment labelScopedDir() {
175     if (scopingLabel.isPresent()) {
176       return AnalysisUtils.getUniqueDirectory(scopingLabel.get(), ownerLabel.toPathFragment());
177     } else {
178       return ownerLabel.toPathFragment();
179     }
180   }
181 
182   private PathFragment inUniqueObjsDir(Artifact source, String extension) {
183     PathFragment labelPath = labelScopedDir();
184     PathFragment uniqueDir = labelPath.getParentDirectory().getRelative(new PathFragment("_objs"))
185         .getRelative(labelPath.getBaseName());
186     PathFragment sourceFile = uniqueDir.getRelative(source.getRootRelativePath());
187     return FileSystemUtils.replaceExtension(sourceFile, extension);
188   }

196   }
