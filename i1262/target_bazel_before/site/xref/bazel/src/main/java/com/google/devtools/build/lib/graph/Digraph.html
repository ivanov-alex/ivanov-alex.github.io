33  /**
34   * <p> {@code Digraph} a generic directed graph or "digraph", suitable for
35   * modeling asymmetric binary relations. </p>
36   *
37   * <p> An instance <code>G = &lt;V,E&gt;</code> consists of a set of nodes or
38   * vertices <code>V</code>, and a set of directed edges <code>E</code>, which
39   * is a subset of <code>V &times; V</code>.  This permits self-edges but does
40   * not represent multiple edges between the same pair of nodes. </p>
41   *
42   * <p> Nodes may be labeled with values of any type (type parameter
43   * T).  All nodes within a graph have distinct labels.  The null
44   * pointer is not a valid label.</p>
45   *
46   * <p> The package supports various operations for modeling partial order
47   * relations, and supports input/output in AT&amp;T's 'dot' format.  See
48   * http://www.research.att.com/sw/tools/graphviz/. </p>
49   *
50   * <p> Some invariants: </p>
51   * <ul>
52   *
53   * <li> Each graph instances "owns" the nodes is creates.  The behaviour of
54   * operations on nodes a graph does not own is undefined.
55   *
56   * <li> {@code Digraph} assumes immutability of node labels, much like {@link
57   * HashMap} assumes it for keys.
58   *
59   * <li> Mutating the underlying graph invalidates any sets and iterators backed
60   * by it.
61   *
62   * </ul>
63   *
64   * <p>Each node stores successor and predecessor adjacency sets using a
65   * representation that dynamically changes with size: small sets are stored as
66   * arrays, large sets using hash tables.  This representation provides
67   * significant space and time performance improvements upon two prior versions:
68   * the earliest used only HashSets; a later version used linked lists, as
69   * described in Cormen, Leiserson &amp; Rivest.
70   */
71  public final class Digraph<T> implements Cloneable {

88    /**
89     * Sanity-check: assert that a node is indeed a member of this graph and not
90     * another one.  Perform this check whenever a function is supplied a node by
91     * the user.
92     */
93    private final void checkNode(Node<T> node) {
94      if (getNode(node.getLabel()) != node) {
95        throw new IllegalArgumentException("node " + node
96                                           + " is not a member of this graph");
97      }
98    }

100   /**
101    * Adds a directed edge between the nodes labelled 'from' and 'to', creating
102    * them if necessary.
103    *
104    * @return true iff the edge was not already present.
105    */
106   public boolean addEdge(T from, T to) {
107     Node<T> fromNode = createNode(from);
108     Node<T> toNode   = createNode(to);
109     return addEdge(fromNode, toNode);
110   }
111 
112   /**
113    * Adds a directed edge between the specified nodes, which must exist and
114    * belong to this graph.
115    *
116    * @return true iff the edge was not already present.
117    *
118    * Note: multi-edges are ignored.  Self-edges are permitted.
119    */
120   public boolean addEdge(Node<T> fromNode, Node<T> toNode) {
121     checkNode(fromNode);
122     checkNode(toNode);
123     boolean isNewSuccessor = fromNode.addSuccessor(toNode);
124     boolean isNewPredecessor = toNode.addPredecessor(fromNode);
125     if (isNewPredecessor != isNewSuccessor) {
126       throw new IllegalStateException();
127     }
128     return isNewSuccessor;
129   }
130 
131   /**
132    * Returns true iff the graph contains an edge between the
133    * specified nodes, which must exist and belong to this graph.
134    */
135   public boolean containsEdge(Node<T> fromNode, Node<T> toNode) {
136     checkNode(fromNode);
137     checkNode(toNode);
138     // TODO(bazel-team): (2009) iterate only over the shorter of from.succs, to.preds.
139     return fromNode.getSuccessors().contains(toNode);
140   }

142   /**
143    * Removes the edge between the specified nodes.  Idempotent: attempts to
144    * remove non-existent edges have no effect.
145    *
146    * @return true iff graph changed.
147    */
148   public boolean removeEdge(Node<T> fromNode, Node<T> toNode) {
149     checkNode(fromNode);
150     checkNode(toNode);
151     boolean changed = fromNode.removeSuccessor(toNode);
152     if (changed) {
153       toNode.removePredecessor(fromNode);
154     }
155     return changed;
156   }

175   /**
176    * Returns true iff the two graphs are equivalent, i.e. have the same set
177    * of node labels, with the same connectivity relation.
178    *
179    * O(n^2) in the worst case, i.e. equivalence.  The algorithm could be speed up by
180    * close to a factor 2 in the worst case by a more direct implementation instead
181    * of using isSubgraph twice.
182    */
183   @Override
184   public boolean equals(Object thatObject) {
185     /* If this graph is a subgraph of thatObject, then we know that thatObject is of
186      * type Digraph<?> and thatObject can be cast to this type.
187      */
188     return isSubgraph(thatObject) && ((Digraph<?>) thatObject).isSubgraph(this);
189   }

191   /**
192    * Returns true iff this graph is a subgraph of the argument. This means that this graph's nodes
193    * are a subset of those of the argument; moreover, for each node of this graph the set of
194    * successors is a subset of those of the corresponding node in the argument graph.
195    *
196    * This algorithm is O(n^2), but linear in the total sizes of the graphs.
197    */
198   public boolean isSubgraph(Object thatObject) {
199     if (this == thatObject) {
200       return true;
201     }
202     if (!(thatObject instanceof Digraph)) {
203       return false;
204     }
205 
206     @SuppressWarnings("unchecked")
207     Digraph<T> that = (Digraph<T>) thatObject;
208     if (this.getNodeCount() > that.getNodeCount()) {
209       return false;
210     }
211     for (Node<T> n1: nodes.values()) {
212       Node<T> n2 = that.getNodeMaybe(n1.getLabel());
213       if (n2 == null) {
214         return false; // 'that' is missing a node
215       }
216 
217       // Now compare the successor relations.
218       // Careful:
219       // - We can't do simple equality on the succs-sets because the
220       //   nodes belong to two different graphs!
221       // - There's no need to check both predecessor and successor
222       //   relations, either one is sufficient.
223       Collection<Node<T>> n1succs = n1.getSuccessors();
224       Collection<Node<T>> n2succs = n2.getSuccessors();
225       if (n1succs.size() > n2succs.size()) {
226         return false;
227       }
228       // foreach successor of n1, ensure n2 has a similarly-labeled succ.
229       for (Node<T> succ1: n1succs) {
230         Node<T> succ2 = that.getNodeMaybe(succ1.getLabel());
231         if (succ2 == null) {
232           return false;
233         }
234         if (!n2succs.contains(succ2)) {
235           return false;
236         }
237       }
238     }
239     return true;
240   }

242   /**
243    * Returns a duplicate graph with the same set of node labels and the same
244    * connectivity relation.  The labels themselves are not cloned.
245    */
246   @Override
247   public Digraph<T> clone() {
248     final Digraph<T> that = new Digraph<T>();
249     visitNodesBeforeEdges(new AbstractGraphVisitor<T>() {
250       @Override
251       public void visitEdge(Node<T> lhs, Node<T> rhs) {
252         that.addEdge(lhs.getLabel(), rhs.getLabel());
253       }
254       @Override
255       public void visitNode(Node<T> node) {
256         that.createNode(node.getLabel());
257       }
258     });
259     return that;
260   }

263    * Returns a deterministic immutable view of the nodes of this graph.
264    */
265   public Collection<Node<T>> getNodes(final Comparator<T> comparator) {
266     Ordering<Node<T>> ordering = new Ordering<Node<T>>() {
267       @Override
268       public int compare(Node<T> o1, Node<T> o2) {
269         return comparator.compare(o1.getLabel(), o2.getLabel());
270       }
271     };
272     return ordering.immutableSortedCopy(nodes.values());
273   }

373   /**
374    * Find or create a node with the specified label.  This is the <i>only</i>
375    * factory of Nodes.  The null pointer is not a valid label.
376    */
377   public Node<T> createNode(T label) {
378     if (label == null) {
379       throw new NullPointerException();
380     }
381     Node<T> n = nodes.get(label);
382     if (n == null) {
383       nodes.put(label, n = new Node<T>(label, nextHashCode++));
384     }
385     return n;
386   }

398   /**
399    * Returns true iff the graph is cyclic.  Time: O(n).
400    */
401   public boolean isCyclic() {
402 
403     // To detect cycles, we use a colored depth-first search. All nodes are
404     // initially marked white.  When a node is encountered, it is marked grey,
405     // and when its descendants are completely visited, it is marked black.
406     // If a grey node is ever encountered, then there is a cycle.
407     final Object WHITE = null; // i.e. not present in nodeToColor, the default.
408     final Object GREY  = new Object();
409     final Object BLACK = new Object();
410     final Map<Node<T>, Object> nodeToColor = new HashMap<>(); // empty => all white
411 
412     class CycleDetector { /* defining a class gives us lexical scope */
413       boolean visit(Node<T> node) {
414         nodeToColor.put(node, GREY);
415         for (Node<T> succ: node.getSuccessors()) {
416           if (nodeToColor.get(succ) == GREY) {
417             return true;
418           } else if (nodeToColor.get(succ) == WHITE) {
419             if (visit(succ)) {
420               return true;
421             }
422           }
423         }
424         nodeToColor.put(node, BLACK);
425         return false;
426       }
427     }
428 
429     CycleDetector detector = new CycleDetector();
430     for (Node<T> node: nodes.values()) {
431       if (nodeToColor.get(node) == WHITE) {
432         if (detector.visit(node)) {
433           return true;
434         }
435       }
436     }
437     return false;
438   }

440   /**
441    * Returns the strong component graph of "this".  That is, returns a new
442    * acyclic graph in which all strongly-connected components in the original
443    * graph have been "fused" into a single node.
444    *
445    * @return a new graph, whose node labels are sets of nodes of the
446    * original graph.  (Do not get confused as to which graph each
447    * set of Nodes belongs!)
448    */
449   public Digraph<Set<Node<T>>> getStrongComponentGraph() {
450     Collection<Set<Node<T>>> sccs = getStronglyConnectedComponents();
451     Digraph<Set<Node<T>>> scGraph = createImageUnderPartition(sccs);
452     scGraph.removeSelfEdges(); // scGraph should be acyclic: no self-edges
453     return scGraph;
454   }

456   /**
457    * Returns a partition of the nodes of this graph into sets, each set being
458    * one strongly-connected component of the graph.
459    */
460   public Collection<Set<Node<T>>> getStronglyConnectedComponents() {
461     final List<Set<Node<T>>> sccs = new ArrayList<>();
462     NodeSetReceiver<T> r = new NodeSetReceiver<T>() {
463       @Override
464       public void accept(Set<Node<T>> scc) {
465         sccs.add(scc);
466       }
467     };
468     SccVisitor<T> v = new SccVisitor<T>();
469     for (Node<T> node : nodes.values()) {
470       v.visit(r, node);
471     }
472     return sccs;
473   }

475   /**
476    * <p> Given a partition of the graph into sets of nodes, returns the image
477    * of this graph under the function which maps each node to the
478    * partition-set in which it appears.  The labels of the new graph are the
479    * (immutable) sets of the partition, and the edges of the new graph are the
480    * edges of the original graph, mapped via the same function. </p>
481    *
482    * <p> Note: the resulting graph may contain self-edges.  If these are not
483    * wanted, call <code>removeSelfEdges()</code>> on the result. </p>
484    *
485    * <p> Interesting special case: if the partition is the set of
486    * strongly-connected components, the result of this function is the
487    * strong-component graph. </p>
488    */
489   public Digraph<Set<Node<T>>>
490     createImageUnderPartition(Collection<Set<Node<T>>> partition) {
491 
492     // Build mapping function: each node label is mapped to its equiv class:
493     Map<T, Set<Node<T>>> labelToImage = new HashMap<>();
494     for (Set<Node<T>> set: partition) {
495       // It's important to use immutable sets of node labels when sets are keys
496       // in a map; see ImmutableSet class for explanation.
497       Set<Node<T>> imageSet = ImmutableSet.copyOf(set);
498       for (Node<T> node: imageSet) {
499         labelToImage.put(node.getLabel(), imageSet);
500       }
501     }
502 
503     if (labelToImage.size() != getNodeCount()) {
504       throw new IllegalArgumentException(
505           "createImageUnderPartition(): argument is not a partition");
506     }
507 
508     return createImageUnderMapping(labelToImage);
509   }

511   /**
512    * Returns the image of this graph in a given function, expressed as a
513    * mapping from labels to some other domain.
514    */
515   public <IMAGE> Digraph<IMAGE>
516     createImageUnderMapping(Map<T, IMAGE> map) {
517     Digraph<IMAGE> imageGraph = new Digraph<>();
518 
519     for (Node<T> fromNode: nodes.values()) {
520       T fromLabel = fromNode.getLabel();
521 
522       IMAGE fromImage = map.get(fromLabel);
523       if (fromImage == null) {
524         throw new IllegalArgumentException(
525             "Incomplete function: undefined for " + fromLabel);
526       }
527       imageGraph.createNode(fromImage);
528 
529       for (Node<T> toNode: fromNode.getSuccessors()) {
530         T toLabel = toNode.getLabel();
531 
532         IMAGE toImage = map.get(toLabel);
533         if (toImage == null) {
534           throw new IllegalArgumentException(
535             "Incomplete function: undefined for " + toLabel);
536         }
537         imageGraph.addEdge(fromImage, toImage);
538       }
539     }
540 
541     return imageGraph;
542   }

553   /**
554    * Finds the shortest directed path from "fromNode" to "toNode".  The path is
555    * returned as an ordered list of nodes, including both endpoints.  Returns
556    * null if there is no path.  Uses breadth-first search.  Running time is
557    * O(n).
558    */
559   public List<Node<T>> getShortestPath(Node<T> fromNode,
560                                            Node<T> toNode) {
561     checkNode(fromNode);
562     checkNode(toNode);
563 
564     if (fromNode == toNode) {
565       return Collections.singletonList(fromNode);
566     }
567 
568     Map<Node<T>, Node<T>> pathPredecessor = new HashMap<>();
569 
570     Set<Node<T>> marked = new HashSet<>();
571 
572     LinkedList<Node<T>> queue = new LinkedList<>();
573     queue.addLast(fromNode);
574     marked.add(fromNode);
575 
576     while (!queue.isEmpty()) {
577       Node<T> u = queue.removeFirst();
578       for (Node<T> v: u.getSuccessors()) {
579         if (marked.add(v)) {
580           pathPredecessor.put(v, u);
581           if (v == toNode) {
582             return getPathToTreeNode(pathPredecessor, v); // found a path
583           }
584           queue.addLast(v);
585         }
586       }
587     }
588     return null; // no path
589   }

591   /**
592    * Given a tree (expressed as a map from each node to its parent), and a
593    * starting node, returns the path from the root of the tree to 'node' as a
594    * list.
595    */
596   public static <X> List<X> getPathToTreeNode(Map<X, X> tree, X node) {
597     List<X> path = new ArrayList<X>();
598     while (node != null) {
599       path.add(node);
600       node = tree.get(node); // get parent
601     }
602     Collections.reverse(path);
603     return path;
604   }

608    * [a.k.a "reverse post-order" of depth-first search.]
609    *
610    * A topological order is one such that, if (u, v) is a path in
611    * acyclic graph G, then u is before v in the topological order.
612    * In other words "tails before heads" or "roots before leaves".

614    * @return The nodes of the graph, in a topological order
615    */

618     Collections.reverse(order);

622   /**
623    * Returns the nodes of an acyclic graph in topological order
624    * [a.k.a "reverse post-order" of depth-first search.]
625    *
626    * A topological order is one such that, if (u, v) is a path in
627    * acyclic graph G, then u is before v in the topological order.
628    * In other words "tails before heads" or "roots before leaves".
629    *
630    * If an ordering is given, returns a specific topological order from the set
631    * of all topological orders; if no ordering given, returns an arbitrary
632    * (nondeterministic) one, but is a bit faster because no sorting needs to be
633    * done for each node.
634    *
635    * @param edgeOrder the ordering in which edges originating from the same node
636    *     are visited.
637    * @return The nodes of the graph, in a topological order
638    */
639   public List<Node<T>> getTopologicalOrder(
640       Comparator<T> edgeOrder) {
641     CollectingVisitor<T> visitor = new CollectingVisitor<T>();
642     DFS<T> visitation = new DFS<T>(DFS.Order.POSTORDER, edgeOrder, false);
643     visitor.beginVisit();
644     for (Node<T> node : getNodes(edgeOrder)) {
645       visitation.visit(node, visitor);
646     }
647     visitor.endVisit();
648 
649     List<Node<T>> order = visitor.getVisitedNodes();
650     Collections.reverse(order);
651     return order;
652   }

692   /**
693    * Returns the (immutable) set of nodes that reach some node in {@code
694    * startNodes} (reflexive transitive closure).
695    */
696   public Set<Node<T>> getBackReachable(Collection<Node<T>> startNodes) {
697     // This method is intentionally not static, to permit future expansion.
698     DFS<T> dfs = new DFS<T>(DFS.Order.PREORDER, true);
699     for (Node<T> n : startNodes) {
700       dfs.visit(n, new AbstractGraphVisitor<T>());
701     }
702     return dfs.getMarked();
703   }

705   /**
706    * Removes the node in the graph specified by the given label.  Optionally,
707    * preserves the graph order (by connecting up the broken edges) or drop them
708    * all.  If the specified label is not the label of any node in the graph,
709    * does nothing.
710    *
711    * @param label the label of the node to remove.
712    * @param preserveOrder if true, adds edges between the neighbours
713    *   of the removed node so as to maintain the graph ordering
714    *   relation between all pairs of such nodes.  If false, simply
715    *   discards all edges from the deleted node to its neighbours.
716    * @return true iff 'label' identifies a node (i.e. the graph was changed).
717    */
718   public boolean removeNode(T label, boolean preserveOrder) {
719     Node<T> node = getNodeMaybe(label);
720     if (node != null) {
721       removeNode(node, preserveOrder);
722       return true;
723     } else {
724       return false;
725     }
726   }

728   /**
729    * Removes the specified node in the graph.
730    *
731    * @param n the node to remove (must be in the graph).
732    * @param preserveOrder see removeNode(T, boolean).
733    */
734   public void removeNode(Node<T> n, boolean preserveOrder) {
735     checkNode(n);
736     for (Node<T> b:  n.getSuccessors()) { // edges from n
737       // exists: n -> b
738       if (preserveOrder) {
739         for (Node<T> a: n.getPredecessors()) { // edges to n
740           // exists: a -> n
741           // beware self edges: they prevent n's deletion!
742           if (a != n && b != n) {
743             addEdge(a, b); // concurrent mod?
744           }
745         }
746       }
747       b.removePredecessor(n); // remove edge n->b in b
748     }
749     for (Node<T> a: n.getPredecessors()) { // edges to n
750       a.removeSuccessor(n); // remove edge a->n in a
751     }
752 
753     n.removeAllEdges(); // remove edges b->n and a->n in n
754     Object del = nodes.remove(n.getLabel());
755     if (del != n) {
756       throw new IllegalStateException(del + " " + n);
757     }
758   }

774   /**
775    * Removes all nodes from this graph except those whose label is an element of {@code keepLabels}.
776    * Edges are added so as to preserve the <i>transitive</i> closure relation.
777    *
778    * @param keepLabels a subset of the labels of this graph; the resulting graph
779    * will have only the nodes with these labels.
780    */
781   public void subgraph(final Set<T> keepLabels) {
782     // This algorithm does the following:
783     // Let keep = nodes that have labels in keepLabels.
784     // Let toRemove = nodes \ keep. reachables = successors and predecessors of keep in nodes.
785     // reachables is the subset of nodes of remove that are an immediate neighbor of some node in
786     // keep.
787     //
788     // Removes all nodes of reachables from keepLabels.
789     // Until reachables is empty:
790     //   Takes n from reachables
791     //   for all s in succ(n)
792     //     for all p in pred(n)
793     //       add the edge (p, s)
794     //     add s to reachables
795     //   for all p in pred(n)
796     //     add p to reachables
797     //   Remove n and its edges
798     //
799     // A few adjustments are needed to do the whole computation.
800 
801     final Set<Node<T>> toRemove = new HashSet<>();
802     final Set<Node<T>> keepNeighbors = new HashSet<>();
803 
804     // Look for all nodes if they are to be kept or removed
805     for (Node<T> node : nodes.values()) {
806       if (keepLabels.contains(node.getLabel())) {
807         // Node is to be kept
808         keepNeighbors.addAll(node.getPredecessors());
809         keepNeighbors.addAll(node.getSuccessors());
810       } else {
811         // node is to be removed.
812         toRemove.add(node);
813       }
814     }
815 
816     if (toRemove.isEmpty()) {
817       // This premature return is needed to avoid 0-size priority queue creation.
818       return;
819     }
820 
821     // We use a priority queue to look for low-order nodes first so we don't propagate the high
822     // number of paths of high-order nodes making the time consumption explode.
823     // For perfect results we should reorder the set each time we add a new edge but this would
824     // be too expensive, so this is a good enough approximation.
825     final PriorityQueue<Node<T>> reachables = new PriorityQueue<>(toRemove.size(),
826         new Comparator<Node<T>>() {
827       @Override
828       public int compare(Node<T> o1, Node<T> o2) {
829         return Long.compare((long) o1.numPredecessors() * (long) o1.numSuccessors(),
830             (long) o2.numPredecessors() * (long) o2.numSuccessors());
831       }
832     });
833 
834     // Construct the reachables queue with the list of successors and predecessors of keep in
835     // toRemove.
836     keepNeighbors.retainAll(toRemove);
837     reachables.addAll(keepNeighbors);
838     toRemove.removeAll(reachables);
839 
840     // Remove nodes, least connected first, preserving reachability.
841     while (!reachables.isEmpty()) {
842       Node<T> node = reachables.poll();
843       for (Node<T> s : node.getSuccessors()) {
844         if (s == node) { continue; } // ignore self-edge
845 
846         for (Node<T> p : node.getPredecessors()) {
847           if (p == node) { continue; } // ignore self-edge
848           addEdge(p, s);
849         }
850 
851         // removes n -> s
852         s.removePredecessor(node);
853         if (toRemove.remove(s)) {
854           reachables.add(s);
855         }
856       }
857 
858       for (Node<T> p : node.getPredecessors()) {
859         if (p == node) { continue; } // ignore self-edge
860         p.removeSuccessor(node);
861         if (toRemove.remove(p)) {
862           reachables.add(p);
863         }
864       }
865 
866       // After the node deletion, the graph is again well-formed and the original topological order
867       // is preserved.
868       nodes.remove(node.getLabel());
869     }
870 
871     // Final cleanup for non-reachable nodes.
872     for (Node<T> node : toRemove) {
873       removeNode(node, false);
874     }
875   }

881   /**
882    * Find strongly connected components using path-based strong component
883    * algorithm. This has the advantage over the default method of returning
884    * the components in postorder.
885    *
886    * We visit nodes depth-first, keeping track of the order that
887    * we visit them in (preorder). Our goal is to find the smallest node (in
888    * this preorder of visitation) reachable from a given node. We keep track of the
889    * smallest node pointed to so far at the top of a stack. If we ever find an
890    * already-visited node, then if it is not already part of a component, we
891    * pop nodes from that stack until we reach this already-visited node's number
892    * or an even smaller one.
893    *
894    * Once the depth-first visitation of a node is complete, if this node's
895    * number is at the top of the stack, then it is the "first" element visited
896    * in its strongly connected component. Hence we pop all elements that were
897    * pushed onto the visitation stack and put them in a strongly connected
898    * component with this one, then send a passed-in {@link Digraph.NodeSetReceiver} this component.
899    */
900   private class SccVisitor<T> {
901     // Nodes already assigned to a strongly connected component.
902     private final Set<Node<T>> assigned = new HashSet<>();
903 
904     // The order each node was visited in.
905     private final Map<Node<T>, Integer> preorder = new HashMap<>();
906 
907     // Stack of all nodes visited whose SCC has not yet been determined. When an SCC is found,
908     // that SCC is an initial segment of this stack, and is popped off. Every time a new node is
909     // visited, it is put on this stack.
910     private final List<Node<T>> stack = new ArrayList<>();
911 
912     // Stack of visited indices for the first-visited nodes in each of their known-so-far
913     // strongly connected components. A node pushes its index on when it is visited. If any of
914     // its successors have already been visited and are not in an already-found strongly connected
915     // component, then, since the successor was already visited, it and this node must be part of a
916     // cycle. So every node visited since the successor is actually in the same strongly connected
917     // component. In this case, preorderStack is popped until the top is at most the successor's
918     // index.
919     //
920     // After all descendants of a node have been visited, if the top element of preorderStack is
921     // still the current node's index, then it was the first element visited of the current strongly
922     // connected component. So all nodes on {@code stack} down to the current node are in its
923     // strongly connected component. And the node's index is popped from preorderStack.
924     private final List<Integer> preorderStack = new ArrayList<>();
925 
926     // Index of node being visited.
927     private int counter = 0;
928 
929     private void visit(NodeSetReceiver<T> visitor, Node<T> node) {
930       if (preorder.containsKey(node)) {
931         // This can only happen if this was a non-recursive call, and a previous
932         // visit call had already visited node.
933         return;
934       }
935       preorder.put(node, counter);
936       stack.add(node);
937       preorderStack.add(counter++);
938       int preorderLength = preorderStack.size();
939       for (Node<T> succ : node.getSuccessors()) {
940         Integer succPreorder = preorder.get(succ);
941         if (succPreorder == null) {
942           visit(visitor, succ);
943         } else {
944           // Does succ not already belong to an SCC? If it doesn't, then it
945           // must be in the same SCC as node. The "starting node" of this SCC
946           // must have been visited before succ (or is succ itself).
947           if (!assigned.contains(succ)) {
948             while (preorderStack.get(preorderStack.size() - 1) > succPreorder) {
949               preorderStack.remove(preorderStack.size() - 1);
950             }
951           }
952         }
953       }
954       if (preorderLength == preorderStack.size()) {
955         // If the length of the preorderStack is unchanged, we did not find any earlier-visited
956         // nodes that were part of a cycle with this node. So this node is the first-visited
957         // element in its strongly connected component, and we collect the component.
958         preorderStack.remove(preorderStack.size() - 1);
959         Set<Node<T>> scc = new HashSet<>();
960         Node<T> compNode;
961         do {
962           compNode = stack.remove(stack.size() - 1);
963           assigned.add(compNode);
964           scc.add(compNode);
965         } while (!node.equals(compNode));
966         visitor.accept(scc);
967       }
968     }
969   }

1039    * such that each node is visited before any edge coming out of that node;
1040    * the order is otherwise unspecified.

1049       for (Node<T> toNode: fromNode.getSuccessors()) {
