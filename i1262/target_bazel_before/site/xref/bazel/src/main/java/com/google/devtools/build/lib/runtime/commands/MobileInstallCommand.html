27  import com.google.devtools.common.options.OptionsBase;
28  import com.google.devtools.common.options.OptionsParser;

30  import com.google.devtools.common.options.OptionsProvider;

32  import java.util.List;
33  
34  /**
35   * Implementation of the 'mobile-install' command.
36   */
37   @Command(name = "mobile-install",
38           builds = true,
39           options = { MobileInstallCommand.Options.class, WriteAdbArgsAction.Options.class },
40           inherits = { BuildCommand.class },
41           shortDescription = "Installs targets to mobile devices.",
42           completion = "label",
43           allowResidue = true,
44           help = "resource:mobile-install.txt")
45  public class MobileInstallCommand implements BlazeCommand {
46    /**
47     * Command line options for the 'mobile-install' command.
48     */
49    public static final class Options extends OptionsBase {
50      @Option(name = "split_apks",
51          defaultValue = "false",
52          category = "undocumented")
53      public boolean splitApks;
54  
55      @Option(name = "incremental",
56          category = "mobile-install",
57          defaultValue = "false",
58          help = "Whether to do an incremental install. If true, try to avoid unnecessary additional "
59              + "work by reading the state of the device the code is to be installed on and using "
60              + "that information to avoid unnecessary work. If false (the default), always do a "
61              + "full install.")
62      public boolean incremental;
63    }
64  
65    @Override
66    public ExitCode exec(BlazeRuntime runtime, OptionsProvider options) {
67      List<String> targets = ProjectFileSupport.getTargets(runtime, options);
68      BuildRequest request = BuildRequest.create(
69          this.getClass().getAnnotation(Command.class).name(), options,
70          runtime.getStartupOptionsProvider(), targets,
71          runtime.getReporter().getOutErr(), runtime.getCommandId(), runtime.getCommandStartTime());
72      return runtime.getBuildTool().processRequest(request, null).getExitCondition();
73    }
74  
75    @Override
76    public void editOptions(BlazeRuntime runtime, OptionsParser optionsParser)
77        throws AbruptExitException {
78      try {
79        String outputGroup =
80            optionsParser.getOptions(Options.class).splitApks ? "mobile_install_split"
81            : optionsParser.getOptions(Options.class).incremental ? "mobile_install_incremental"
82            : "mobile_install_full";
83  
84        optionsParser.parse(OptionPriority.COMMAND_LINE,
85            "Options required by the mobile-install command",
86            ImmutableList.of(
87                "--output_groups=-" + OutputGroupProvider.DEFAULT,
88                "--output_groups=-" + OutputGroupProvider.HIDDEN_TOP_LEVEL,
89                "--output_groups=" + outputGroup));
90      } catch (OptionsParsingException e) {
91        throw new IllegalStateException(e);
92      }
93    }
94  }
