38  /**
39   * Fetches external repositories. Which is so fetch.
40   */
41  @Command(name = FetchCommand.NAME,
42      options = {
43          PackageCacheOptions.class,
44          FetchOptions.class,
45          JavaOptions.class,
46      },
47      help = "resource:fetch.txt",
48      shortDescription = "Fetches external repositories that are prerequisites to the targets.",
49      allowResidue = true,
50      completion = "label")
51  public final class FetchCommand implements BlazeCommand {
52    // TODO(kchodorow): add an option to force-fetch targets, even if they're already downloaded.
53    // TODO(kchodorow): this would be a great time to check for difference and invalidate the upward
54    //                  transitive closure for local repositories.
55  
56    public static final String NAME = "fetch";
57  
58    @Override
59    public void editOptions(BlazeRuntime runtime, OptionsParser optionsParser) { }
60  
61    @Override
62    public ExitCode exec(BlazeRuntime runtime, OptionsProvider options) {
63      if (options.getResidue().isEmpty()) {
64        runtime.getReporter().handle(Event.error(String.format(
65            "missing fetch expression. Type '%s help fetch' for syntax and help",
66            Constants.PRODUCT_NAME)));
67        return ExitCode.COMMAND_LINE_ERROR;
68      }
69  
70      try {
71        runtime.setupPackageCache(
72            options.getOptions(PackageCacheOptions.class),
73            runtime.getDefaultsPackageContent());
74      } catch (InterruptedException e) {
75        runtime.getReporter().handle(Event.error("fetch interrupted"));
76        return ExitCode.INTERRUPTED;
77      } catch (AbruptExitException e) {
78        runtime.getReporter().handle(Event.error(null, "Unknown error: " + e.getMessage()));
79        return e.getExitCode();
80      }
81  
82      PackageCacheOptions pkgOptions = options.getOptions(PackageCacheOptions.class);
83      if (pkgOptions.fetch == false) {
84        runtime.getReporter().handle(Event.error(null, "You cannot run fetch with --fetch=false"));
85        return ExitCode.COMMAND_LINE_ERROR;
86      }
87  
88      // Querying for all of the dependencies of the targets has the side-effect of populating the
89      // Skyframe graph for external targets, which requires downloading them. The JDK is required to
90      // build everything but isn't counted as a dep in the build graph so we add it manually.
91      JavaOptions javaOptions = options.getOptions(JavaOptions.class);
92      ImmutableList.Builder<String> labelsToLoad = new ImmutableList.Builder<String>()
93          .addAll(options.getResidue());
94      if (String.valueOf(javaOptions.javaLangtoolsJar).equals(JavaOptions.DEFAULT_LANGTOOLS)) {
95        labelsToLoad.add(javaOptions.javaBase);
96      } else {
97        // TODO(kchodroow): Remove this when OS X isn't as hacky about finding the JVM. Our test
98        // framework currently doesn't set up the JDK normally on OS X, so attempting to fetch
99        // tools/jdk:jdk will cause errors.
100       labelsToLoad.add(String.valueOf(javaOptions.javaToolchain));
101     }
102     String query = Joiner.on(" union ").join(labelsToLoad.build());
103     query = "deps(" + query + ")";
104 
105     AbstractBlazeQueryEnvironment<Target> env = QueryCommand.newQueryEnvironment(
106         runtime, options.getOptions(FetchOptions.class).keepGoing, false,
107         Lists.<String>newArrayList(), 200, Sets.<Setting>newHashSet());
108 
109     // 1. Parse query:
110     QueryExpression expr;
111     try {
112       expr = QueryExpression.parse(query, env);
113     } catch (QueryException e) {
114       runtime.getReporter().handle(Event.error(
115           null, "Error while parsing '" + query + "': " + e.getMessage()));
116       return ExitCode.COMMAND_LINE_ERROR;
117     }
118 
119     // 2. Evaluate expression:
120     try {
121       env.evaluateQuery(expr);
122     } catch (QueryException | InterruptedException e) {
123       // Keep consistent with reportBuildFileError()
124       runtime.getReporter().handle(Event.error(e.getMessage()));
125       return ExitCode.COMMAND_LINE_ERROR;
126     }
127 
128     runtime.getReporter().handle(
129         Event.progress("All external dependencies fetched successfully."));
130     return ExitCode.SUCCESS;
131   }
132 
133 
134 }
