73    /**
74     * Generates the Skylark documentation to the given output directory.
75     */
76    public void generateDocumentation(String outputPath) throws IOException,
77        BuildEncyclopediaDocException {
78      File skylarkDocPath = new File(outputPath);
79      try (BufferedWriter bw = new BufferedWriter(
80          Files.newWriter(skylarkDocPath, StandardCharsets.UTF_8))) {
81        if (USE_TEMPLATE) {
82          bw.write(SourceFileReader.readTemplateContents(DocgenConsts.SKYLARK_BODY_TEMPLATE,
83              ImmutableMap.<String, String>of(
84                  DocgenConsts.VAR_SECTION_SKYLARK_BUILTIN, generateAllBuiltinDoc())));
85        } else {
86          bw.write(generateAllBuiltinDoc());
87        }
88        System.out.println("Skylark documentation generated: " + skylarkDocPath.getAbsolutePath());
89      }
90    }

92    @VisibleForTesting
93    Map<String, SkylarkModuleDoc> collectModules() {
94      Map<String, SkylarkModuleDoc> modules = new TreeMap<>();
95      Map<String, SkylarkModuleDoc> builtinModules = collectBuiltinModules();
96      Map<SkylarkModule, Class<?>> builtinJavaObjects = collectBuiltinJavaObjects();
97  
98      modules.putAll(builtinModules);
99      SkylarkJavaInterfaceExplorer explorer = new SkylarkJavaInterfaceExplorer();
100     for (SkylarkModuleDoc builtinObject : builtinModules.values()) {
101       // Check the return type for built-in functions, it can be a module previously not added.
102       for (SkylarkBuiltinMethod builtinMethod : builtinObject.getBuiltinMethods().values()) {
103         Class<?> type = builtinMethod.annotation.returnType();
104         if (type.isAnnotationPresent(SkylarkModule.class)) {
105           explorer.collect(type.getAnnotation(SkylarkModule.class), type, modules);
106         }
107       }
108       explorer.collect(builtinObject.getAnnotation(), builtinObject.getClassObject(), modules);
109     }
110     for (Entry<SkylarkModule, Class<?>> builtinModule : builtinJavaObjects.entrySet()) {
111       explorer.collect(builtinModule.getKey(), builtinModule.getValue(), modules);
112     }
113     return modules;
114   }

116   private String generateAllBuiltinDoc() {
117     Map<String, SkylarkModuleDoc> modules = collectModules();
118 
119     StringBuilder sb = new StringBuilder();
120     // Generate the top level module first in the doc
121     SkylarkModuleDoc topLevelModule = modules.remove(getTopLevelModule().name());
122     generateModuleDoc(topLevelModule, sb);
123     for (SkylarkModuleDoc module : modules.values()) {
124       if (module.getAnnotation().documented()) {
125         sb.append("<hr>");
126         generateModuleDoc(module, sb);
127       }
128     }
129     return sb.toString();
130   }

132   private void generateModuleDoc(SkylarkModuleDoc module, StringBuilder sb) {
133     SkylarkModule annotation = module.getAnnotation();
134     sb.append(String.format("<h2 id=\"modules.%s\">%s</h2>\n",
135           getModuleId(annotation),
136           annotation.name()))
137       .append(annotation.doc())
138       .append("\n");
139     sb.append("<ul>");
140     // Sort Java and Skylark builtin methods together. The map key is only used for sorting.
141     TreeMap<String, Object> methodMap = new TreeMap<>();
142     for (SkylarkJavaMethod method : module.getJavaMethods()) {
143       methodMap.put(method.name + method.method.getParameterTypes().length, method);
144     }
145     for (SkylarkBuiltinMethod builtin : module.getBuiltinMethods().values()) {
146       methodMap.put(builtin.annotation.name(), builtin);
147     }
148     for (Object object : methodMap.values()) {
149       if (object instanceof SkylarkJavaMethod) {
150         SkylarkJavaMethod method = (SkylarkJavaMethod) object;
151         generateDirectJavaMethodDoc(annotation.name(), method.name, method.method,
152             method.callable, sb);
153       }
154       if (object instanceof SkylarkBuiltinMethod) {
155         generateBuiltinItemDoc(getModuleId(annotation), (SkylarkBuiltinMethod) object, sb);
156       }
157     }
158     sb.append("</ul>");
159   }

161   private String getModuleId(SkylarkModule annotation) {
162     if (annotation == getTopLevelModule()) {
163       return TOP_LEVEL_ID;
164     } else {
165       return annotation.name();
166     }
167   }

169   private void generateBuiltinItemDoc(
170       String moduleId, SkylarkBuiltinMethod method, StringBuilder sb) {
171     SkylarkSignature annotation = method.annotation;
172     if (!annotation.documented()) {
173       return;
174     }
175     sb.append(String.format("<li><h3 id=\"modules.%s.%s\">%s</h3>\n",
176           moduleId,
177           annotation.name(),
178           annotation.name()));
179 
180     if (BaseFunction.class.isAssignableFrom(method.fieldClass)) {
181       sb.append(getSignature(moduleId, annotation));
182     } else {
183       if (!annotation.returnType().equals(Object.class)) {
184         sb.append("<code>" + getTypeAnchor(annotation.returnType()) + "</code><br>");
185       }
186     }
187 
188     sb.append(annotation.doc() + "\n");
189     printParams(moduleId, annotation, sb);
190   }

192   // Elide self parameter from mandatoryPositionals in class methods.
193   private static Param[] adjustedMandatoryPositionals(SkylarkSignature annotation) {
194     Param[] mandatoryPos = annotation.mandatoryPositionals();
195     if (mandatoryPos.length > 0
196         && annotation.objectType() != Object.class
197         && !FuncallExpression.isNamespace(annotation.objectType())) {
198       // Skip the self parameter, which is the first mandatory positional parameter.
199       return Arrays.copyOfRange(mandatoryPos, 1, mandatoryPos.length);
200     } else {
201       return mandatoryPos;
202     }
203   }
204 
205   private void printParams(String moduleId, SkylarkSignature annotation, StringBuilder sb) {
206     Param[] mandatoryPos = adjustedMandatoryPositionals(annotation);
207     Param[] optionalPos = annotation.optionalPositionals();
208     Param[] optionalKey = annotation.optionalNamedOnly();
209     Param[] mandatoryKey = annotation.mandatoryNamedOnly();
210     Param[] star = annotation.extraPositionals();
211     Param[] starStar = annotation.extraKeywords();
212 
213     if (mandatoryPos.length + optionalPos.length + optionalKey.length + mandatoryKey.length
214         + star.length + starStar.length > 0) {
215       sb.append("<h4>Parameters</h4>\n");
216       printParams(moduleId, annotation.name(), mandatoryPos, sb);
217       printParams(moduleId, annotation.name(), optionalPos, sb);
218       printParams(moduleId, annotation.name(), star, sb);
219       printParams(moduleId, annotation.name(), mandatoryKey, sb);
220       printParams(moduleId, annotation.name(), optionalKey, sb);
221       printParams(moduleId, annotation.name(), starStar, sb);
222     } else {
223       sb.append("<br/>\n");
224     }
225   }

227   private void generateDirectJavaMethodDoc(String objectName, String methodName,
228       Method method, SkylarkCallable annotation, StringBuilder sb) {
229     if (!annotation.documented()) {
230       return;
231     }
232     if (annotation.doc().isEmpty()) {
233       throw new RuntimeException(String.format(
234           "empty SkylarkCallable.doc() for object %s, method %s", objectName, methodName));
235     }
236 
237     sb.append(String.format("<li><h3 id=\"modules.%s.%s\">%s</h3>\n%s\n",
238             objectName,
239             methodName,
240             methodName,
241             getSignature(objectName, methodName, method)))
242         .append(annotation.doc())
243         .append(getReturnTypeExtraMessage(annotation))
244         .append("\n");
245   }

247   private String getReturnTypeExtraMessage(SkylarkCallable annotation) {
248     if (annotation.allowReturnNones()) {
249       return " May return <code>None</code>.\n";
250     }
251     return "";
252   }

254   private String getSignature(String objectName, String methodName, Method method) {
255     String args = method.getAnnotation(SkylarkCallable.class).structField()
256         ? "" : "(" + getParameterString(method) + ")";
257 
258     return String.format("<code>%s %s.%s%s</code><br>",
259         getTypeAnchor(method.getReturnType()), objectName, methodName, args);
260   }
261 
262   private String getSignature(String objectName, SkylarkSignature method) {
263     List<String> argList = new ArrayList<>();
264     for (Param param : adjustedMandatoryPositionals(method)) {
265       argList.add(param.name());
266     }
267     for (Param param : method.optionalPositionals()) {
268       argList.add(formatOptionalParameter(param));
269     }
270     for (Param param : method.extraPositionals()) {
271       argList.add("*" + param.name());
272     }
273     if (argList.size() > 0 && method.extraPositionals().length == 0
274         && (method.optionalNamedOnly().length > 0 || method.mandatoryNamedOnly().length > 0)) {
275       argList.add("*");
276     }
277     for (Param param : method.mandatoryNamedOnly()) {
278       argList.add(param.name());
279     }
280     for (Param param : method.optionalNamedOnly()) {
281       argList.add(formatOptionalParameter(param));
282     }
283     for (Param param : method.extraKeywords()) {
284       argList.add("**" + param.name());
285     }
286     String args = "(" + Joiner.on(", ").join(argList) + ")";
287     if (!objectName.equals(TOP_LEVEL_ID)) {
288       return String.format("<code>%s %s.%s%s</code><br>\n",
289           getTypeAnchor(method.returnType()), objectName, method.name(), args);
290     } else {
291       return String.format("<code>%s %s%s</code><br>\n",
292           getTypeAnchor(method.returnType()), method.name(), args);
293     }
294   }
295 
296   private String formatOptionalParameter(Param param) {
297     String defaultValue = param.defaultValue();
298 
299     return String.format("%s=%s", param.name(),
300         (defaultValue == null || defaultValue.isEmpty()) ? "&hellip;" : defaultValue);
301   }

303   private String getTypeAnchor(Class<?> returnType, Class<?> generic1) {
304     return getTypeAnchor(returnType) + " of " + getTypeAnchor(generic1) + "s";
305   }

307   private String getTypeAnchor(Class<?> type) {
308     if (type.equals(Boolean.class) || type.equals(boolean.class)) {
309       return "<a class=\"anchor\" href=\"#modules._top_level.bool\">bool</a>";
310     } else if (type.equals(String.class)) {
311       return "<a class=\"anchor\" href=\"#modules.string\">string</a>";
312     } else if (Map.class.isAssignableFrom(type)) {
313       return "<a class=\"anchor\" href=\"#modules.dict\">dict</a>";
314     } else if (List.class.isAssignableFrom(type) || SkylarkList.class.isAssignableFrom(type)
315         || type == HackHackEitherList.class) {
316       // Annotated Java methods can return simple java.util.Lists (which get auto-converted).
317       return "<a class=\"anchor\" href=\"#modules.list\">list</a>";
318     } else if (type.equals(Void.TYPE) || type.equals(NoneType.class)) {
319       return "<a class=\"anchor\" href=\"#modules." + TOP_LEVEL_ID + ".None\">None</a>";
320     } else if (type.isAnnotationPresent(SkylarkModule.class)) {
321       // TODO(bazel-team): this can produce dead links for types don't show up in the doc.
322       // The correct fix is to generate those types (e.g. SkylarkFileType) too.
323       String module = type.getAnnotation(SkylarkModule.class).name();
324       return "<a class=\"anchor\" href=\"#modules." + module + "\">" + module + "</a>";
325     } else {
326       return EvalUtils.getDataTypeNameFromClass(type);
327     }
328   }

330   private String getParameterString(Method method) {
331     return Joiner.on(", ").join(Iterables.transform(
332         ImmutableList.copyOf(method.getParameterTypes()), new Function<Class<?>, String>() {
333           @Override
334           public String apply(Class<?> input) {
335             return getTypeAnchor(input);
336           }
337         }));
338   }

340   private void printParams(String moduleId, String methodName,
341       Param[] params, StringBuilder sb) {
342     if (params.length > 0) {
343       sb.append("<ul>\n");
344       for (Param param : params) {
345         String paramType = param.type().equals(Object.class) ? ""
346             : (param.generic1().equals(Object.class)
347                 ? " (" + getTypeAnchor(param.type()) + ")"
348                 : " (" + getTypeAnchor(param.type(), param.generic1()) + ")");
349         sb.append(String.format("\t<li id=\"modules.%s.%s.%s\"><code>%s%s</code>: ",
350                 moduleId,
351                 methodName,
352                 param.name(),
353                 param.name(),
354                 paramType))
355             .append(param.doc())
356             .append("\n\t</li>\n");
357       }
358       sb.append("</ul>\n");
359     }
360   }

362   private Map<String, SkylarkModuleDoc> collectBuiltinModules() {
363     Map<String, SkylarkModuleDoc> modules = new HashMap<>();
364     collectBuiltinDoc(modules, Environment.class.getDeclaredFields());
365     collectBuiltinDoc(modules, MethodLibrary.class.getDeclaredFields());
366     for (Class<?> moduleClass : SkylarkModules.MODULES) {
367       collectBuiltinDoc(modules, moduleClass.getDeclaredFields());
368     }
369     return modules;
370   }

372   private Map<SkylarkModule, Class<?>> collectBuiltinJavaObjects() {
373     Map<SkylarkModule, Class<?>> modules = new HashMap<>();
374     collectBuiltinModule(modules, SkylarkRuleContext.class);
375     collectBuiltinModule(modules, TransitiveInfoCollection.class);
376     return modules;
377   }

379   /**
380    * Returns the top level modules and functions with their documentation in a command-line
381    * printable format.
382    */
383   public Map<String, String> collectTopLevelModules() {
384     Map<String, String> modules = new TreeMap<>();
385     for (SkylarkModuleDoc doc : collectBuiltinModules().values()) {
386       if (doc.getAnnotation() == getTopLevelModule()) {
387         for (Map.Entry<String, SkylarkBuiltinMethod> entry : doc.getBuiltinMethods().entrySet()) {
388           if (entry.getValue().annotation.documented()) {
389             modules.put(entry.getKey(),
390                 DocgenConsts.toCommandLineFormat(entry.getValue().annotation.doc()));
391           }
392         }
393       } else {
394         modules.put(doc.getAnnotation().name(),
395             DocgenConsts.toCommandLineFormat(doc.getAnnotation().doc()));
396       }
397     }
398     return modules;
399   }

401   /**
402    * Returns the API doc for the specified Skylark object in a command line printable format,
403    * params[0] identifies either a module or a top-level object, the optional params[1] identifies a
404    * method in the module.<br>
405    * Returns null if no Skylark object is found.
406    */
407   public String getCommandLineAPIDoc(String[] params) {
408     Map<String, SkylarkModuleDoc> modules = collectModules();
409     SkylarkModuleDoc toplevelModuleDoc = modules.get(getTopLevelModule().name());
410     if (modules.containsKey(params[0])) {
411       // Top level module
412       SkylarkModuleDoc module = modules.get(params[0]);
413       if (params.length == 1) {
414         String moduleName = module.getAnnotation().name();
415         StringBuilder sb = new StringBuilder();
416         sb.append(moduleName).append("\n\t").append(module.getAnnotation().doc()).append("\n");
417         // Print the signature of all built-in methods
418         for (SkylarkBuiltinMethod method : module.getBuiltinMethods().values()) {
419           printBuiltinFunctionDoc(moduleName, method.annotation, sb);
420         }
421         // Print all Java methods
422         for (SkylarkJavaMethod method : module.getJavaMethods()) {
423           printJavaFunctionDoc(moduleName, method, sb);
424         }
425         return DocgenConsts.toCommandLineFormat(sb.toString());
426       } else {
427         return getFunctionDoc(module.getAnnotation().name(), params[1], module);
428       }
429     } else if (toplevelModuleDoc.getBuiltinMethods().containsKey(params[0])){
430       // Top level object / function
431       return getFunctionDoc(null, params[0], toplevelModuleDoc);
432     }
433     return null;
434   }

436   private String getFunctionDoc(String moduleName, String methodName, SkylarkModuleDoc module) {
437     if (module.getBuiltinMethods().containsKey(methodName)) {
438       // Create the doc for the built-in function
439       SkylarkBuiltinMethod method = module.getBuiltinMethods().get(methodName);
440       StringBuilder sb = new StringBuilder();
441       printBuiltinFunctionDoc(moduleName, method.annotation, sb);
442       printParams(moduleName, method.annotation, sb);
443       return DocgenConsts.removeDuplicatedNewLines(DocgenConsts.toCommandLineFormat(sb.toString()));
444     } else {
445       // Search if there are matching Java functions
446       StringBuilder sb = new StringBuilder();
447       boolean foundMatchingMethod = false;
448       for (SkylarkJavaMethod method : module.getJavaMethods()) {
449         if (method.name.equals(methodName)) {
450           printJavaFunctionDoc(moduleName, method, sb);
451           foundMatchingMethod = true;
452         }
453       }
454       if (foundMatchingMethod) {
455         return DocgenConsts.toCommandLineFormat(sb.toString()); 
456       }
457     }
458     return null;
459   }

461   private void printBuiltinFunctionDoc(
462       String moduleName, SkylarkSignature annotation, StringBuilder sb) {
463     if (moduleName != null) {
464       sb.append(moduleName).append(".");
465     }
466     sb.append(annotation.name()).append("\n\t").append(annotation.doc()).append("\n");
467   }
468 
469   private void printJavaFunctionDoc(String moduleName, SkylarkJavaMethod method, StringBuilder sb) {
470     sb.append(getSignature(moduleName, method.name, method.method))
471       .append("\t").append(method.callable.doc()).append("\n");
472   }

474   private void collectBuiltinModule(
475       Map<SkylarkModule, Class<?>> modules, Class<?> moduleClass) {
476     if (moduleClass.isAnnotationPresent(SkylarkModule.class)) {
477       SkylarkModule skylarkModule = moduleClass.getAnnotation(SkylarkModule.class);
478       modules.put(skylarkModule, moduleClass);
479     }
480   }
481 
482   private void collectBuiltinDoc(Map<String, SkylarkModuleDoc> modules, Field[] fields) {
483     for (Field field : fields) {
484       if (field.isAnnotationPresent(SkylarkSignature.class)) {
485         SkylarkSignature skylarkSignature = field.getAnnotation(SkylarkSignature.class);
486         Class<?> moduleClass = skylarkSignature.objectType();
487         SkylarkModule skylarkModule = moduleClass.equals(Object.class)
488             ? getTopLevelModule()
489             : moduleClass.getAnnotation(SkylarkModule.class);
490         if (!modules.containsKey(skylarkModule.name())) {
491           modules.put(skylarkModule.name(), new SkylarkModuleDoc(skylarkModule, moduleClass));
492         }
493         modules.get(skylarkModule.name()).getBuiltinMethods()
494             .put(skylarkSignature.name(),
495                 new SkylarkBuiltinMethod(skylarkSignature, field.getType()));
496       }
497     }
