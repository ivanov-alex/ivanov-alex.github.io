15  package com.google.devtools.build.lib.rules.objc;
16  
17  import com.google.common.base.Objects;
18  import com.google.common.collect.ImmutableList;
19  import com.google.common.collect.Iterators;
20  import com.google.devtools.build.lib.actions.Artifact;
21  import com.google.devtools.build.lib.analysis.AnalysisUtils;
22  import com.google.devtools.build.lib.analysis.RuleContext;
23  import com.google.devtools.build.lib.syntax.Label;
24  import com.google.devtools.build.lib.vfs.FileSystemUtils;
25  import com.google.devtools.build.lib.vfs.PathFragment;
26  
27  /**
28   * An object that captures information of ObjC files generated by J2ObjC in a single target.
29   */
30  public class J2ObjcSource {
31  
32    /**
33     * Indicates the type of files from which the ObjC files included in {@link J2ObjcSource} are
34     * generated.
35     */
36    public enum SourceType {
37      /**
38       * Indicates the original file type is java source file.
39       */
40      JAVA,
41  
42      /**
43       * Indicates the original file type is proto file.
44       */
45      PROTO;
46    }
47  
48    private final Label targetLabel;
49    private final Iterable<Artifact> objcSrcs;
50    private final Iterable<Artifact> objcHdrs;
51    private final PathFragment objcFilePath;
52    private final SourceType sourceType;
53  
54    /**
55     * Constructs a J2ObjcSource containing target information for j2objc transpilation.
56     *
57     * @param targetLabel the @{code Label} of the associated target.
58     * @param objcSrcs the {@code Iterable} containing objc source files generated by J2ObjC
59     * @param objcHdrs the {@code Iterable} containing objc header files generated by J2ObjC
60     * @param objcFilePath the {@code PathFragment} under which all the generated objc files are. It
61     *     can be used as header search path for objc compilations.
62     * @param sourceType the type of files from which the ObjC files are generated.
63     */
64    public J2ObjcSource(Label targetLabel, Iterable<Artifact> objcSrcs,
65        Iterable<Artifact> objcHdrs, PathFragment objcFilePath, SourceType sourceType) {
66      this.targetLabel = targetLabel;
67      this.objcSrcs = objcSrcs;
68      this.objcHdrs = objcHdrs;
69      this.objcFilePath = objcFilePath;
70      this.sourceType = sourceType;
71    }
72  
73    /**
74     * Returns a corresponding {@link J2ObjcSource} with source artifacts replaced by the outputs of
75     * the J2objC dead code removal script, for use after that action has processed the originals.
76     *
77     * <p>The script in question builds a dependency graph with entry classes specified
78     * transitively on j2objc_library rules as roots. Translated files from this (original)
79     * {@link J2ObjcSource} which are reachable in the graph from the roots will be copied over to the
80     * source file paths in the returned pruned {@link J2ObjcSource} with full original contents.
81     * Unreachable files will not be copied over and the artifacts pointed to by the returned pruned 
82     * {@link J2ObjcSource} will only contain empty files.
83     *
84     * @param ruleContext the {@link RuleContext} of the current rule
85     */
86    public J2ObjcSource toPrunedSource(RuleContext ruleContext) {
87      ImmutableList.Builder<Artifact> prunedSourceArtifacts = ImmutableList.builder();
88  
89      for (Artifact sourceArtifact : getObjcSrcs()) {
90        PathFragment scopedPath = AnalysisUtils.getUniqueDirectory(
91            ruleContext.getRule().getLabel(), new PathFragment("_j2objc_pruned"));
92        PathFragment prunedSourceArtifactPath = FileSystemUtils.appendWithoutExtension(
93            scopedPath.getRelative(sourceArtifact.getRootRelativePath()), "_pruned");
94  
95        Artifact prunedArtifact = ruleContext.getAnalysisEnvironment().getDerivedArtifact(
96            prunedSourceArtifactPath, ruleContext.getBinOrGenfilesDirectory());
97        prunedSourceArtifacts.add(prunedArtifact);
98      }
99  
100     return new J2ObjcSource(
101         getTargetLabel(),
102         prunedSourceArtifacts.build(),
103         getObjcHdrs(),
104         getObjcFilePath(),
105         getSourceType());
106   }
107 
108   /**
109    * Returns the label of the associated target.
110    */
111   public Label getTargetLabel() {
112     return targetLabel;
113   }
114 
115   /**
116    * Returns the objc source files generated by J2ObjC.
117    */
118   public Iterable<Artifact> getObjcSrcs() {
119     return objcSrcs;
120   }
121 
122   /*
123    * Returns the objc header files generated by J2ObjC
124    */
125   public Iterable<Artifact> getObjcHdrs() {
126     return objcHdrs;
127   }
128 
129   /**
130    * Returns the {@code PathFragment} which represents a directory where the generated ObjC files
131    * reside and which can also be used as header search path in ObjC compilation.
132    */
133   public PathFragment getObjcFilePath() {
134     return objcFilePath;
135   }
136 
137   /**
138    * Returns the type of files from which the ObjC files inside this object are generated.
139    */
140   public SourceType getSourceType() {
141     return sourceType;
142   }
143 
144   @Override
145   public final boolean equals(Object other) {
146     if (!(other instanceof J2ObjcSource)) {
147       return false;
148     }
149 
150     J2ObjcSource that = (J2ObjcSource) other;
151     return Objects.equal(this.targetLabel, that.targetLabel)
152         && Iterators.elementsEqual(this.objcSrcs.iterator(), that.objcSrcs.iterator())
153         && Iterators.elementsEqual(this.objcHdrs.iterator(), that.objcHdrs.iterator())
154         && Objects.equal(this.objcFilePath, that.objcFilePath)
155         && this.sourceType == that.sourceType;
156   }
157 
158   @Override
159   public int hashCode() {
160     return Objects.hashCode(targetLabel, objcSrcs, objcHdrs, objcFilePath, sourceType);
161   }
162 }
