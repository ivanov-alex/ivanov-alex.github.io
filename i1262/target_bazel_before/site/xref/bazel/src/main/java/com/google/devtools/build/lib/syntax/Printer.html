70    // In absence of a Python naming tradition, the write() vs print() function names
71    // follow the Lisp tradition: print() displays the informal representation (as in Python str)
72    // whereas write() displays a readable representation (as in Python repr).
73    /**
74     * Print an informal representation of object x.
75     * Currently only differs from repr in the behavior for strings and labels at top-level,
76     * that are returned as is rather than quoted.
77     * @param buffer the Appendable to which to print the representation
78     * @param o the object
79     * @param quotationMark The quotation mark to be used (' or ")
80     * @return the buffer, in fluent style
81     */
82    public static Appendable print(Appendable buffer, Object o, char quotationMark) {
83      if (o instanceof Label) {
84        return append(buffer, o.toString());  // Pretty-print a label like a string
85      }
86      if (o instanceof String) {
87        return append(buffer, (String) o);
88      }
89      return write(buffer, o, quotationMark);
90    }
91    
92    public static Appendable print(Appendable buffer, Object o) {
93      return print(buffer, o, SKYLARK_QUOTATION_MARK);
94    }

96    /**
97     * Print an official representation of object x.
98     * For regular data structures, the value should be parsable back into an equal data structure.
99     * @param buffer the Appendable to write to.
100    * @param o the string a representation of which to write.
101    * @param quotationMark The quotation mark to be used (' or ")
102    * @return the Appendable, in fluent style.
103    */
104   public static Appendable write(Appendable buffer, Object o, char quotationMark) {
105     if (o == null) {
106       throw new NullPointerException(); // Java null is not a build language value.
107 
108     } else if (o instanceof String) {
109       writeString(buffer, (String) o, quotationMark);
110 
111     } else if (o instanceof Integer || o instanceof Double) {
112       append(buffer, o.toString());
113 
114     } else if (o == Environment.NONE) {
115       append(buffer, "None");
116 
117     } else if (o == Boolean.TRUE) {
118       append(buffer, "True");
119 
120     } else if (o == Boolean.FALSE) {
121       append(buffer, "False");
122 
123     } else if (o instanceof List<?>) {
124       List<?> seq = (List<?>) o;
125       printList(buffer, seq, EvalUtils.isImmutable(seq), quotationMark);
126 
127     } else if (o instanceof SkylarkList) {
128       SkylarkList list = (SkylarkList) o;
129       printList(buffer, list.toList(), list.isTuple(), quotationMark);
130 
131     } else if (o instanceof Map<?, ?>) {
132       Map<?, ?> dict = (Map<?, ?>) o;
133       printList(buffer, dict.entrySet(), "{", ", ", "}", null, quotationMark);
134 
135     } else if (o instanceof Map.Entry<?, ?>) {
136       Map.Entry<?, ?> entry = (Map.Entry<?, ?>) o;
137       write(buffer, entry.getKey(), quotationMark);
138       append(buffer, ": ");
139       write(buffer, entry.getValue(), quotationMark);
140 
141     } else if (o instanceof SkylarkNestedSet) {
142       SkylarkNestedSet set = (SkylarkNestedSet) o;
143       append(buffer, "set(");
144       printList(buffer, set, "[", ", ", "]", null, quotationMark);
145       Order order = set.getOrder();
146       if (order != Order.STABLE_ORDER) {
147         append(buffer, ", order = \"" + order.getName() + "\"");
148       }
149       append(buffer, ")");
150 
151     } else if (o instanceof BaseFunction) {
152       BaseFunction func = (BaseFunction) o;
153       append(buffer, "<function " + func.getName() + ">");
154 
155     } else if (o instanceof Label) {
156       write(buffer, o.toString(), quotationMark);
157 
158     } else if (o instanceof FilesetEntry) {
159            FilesetEntry entry = (FilesetEntry) o;
160       append(buffer, "FilesetEntry(srcdir = ");
161       write(buffer, entry.getSrcLabel().toString(), quotationMark);
162       append(buffer, ", files = ");
163       write(buffer, makeStringList(entry.getFiles()), quotationMark);
164       append(buffer, ", excludes = ");
165       write(buffer, makeList(entry.getExcludes()), quotationMark);
166       append(buffer, ", destdir = ");
167       write(buffer, entry.getDestDir().getPathString(), quotationMark);
168       append(buffer, ", strip_prefix = ");
169       write(buffer, entry.getStripPrefix(), quotationMark);
170       append(buffer, ", symlinks = ");
171       append(buffer, quotationMark);
172       append(buffer, entry.getSymlinkBehavior().toString());
173       append(buffer, quotationMark);
174       append(buffer, ")");
175 
176     } else if (o instanceof PathFragment) {
177       append(buffer, ((PathFragment) o).getPathString());
178 
179     } else {
180       append(buffer, o.toString());
181     }
182 
183     return buffer;
184   }

187     return write(buffer, o, SKYLARK_QUOTATION_MARK);

190   // Throughout this file, we transform IOException into AssertionError.
191   // During normal operations, we only use in-memory Appendable-s that
192   // cannot cause an IOException.
193   private static Appendable append(Appendable buffer, char c) {
194     try {
195       return buffer.append(c);
196     } catch (IOException e) {
197       throw new AssertionError(e);
198     }
199   }

201   private static Appendable append(Appendable buffer, CharSequence s) {
202     try {
203       return buffer.append(s);
204     } catch (IOException e) {
205       throw new AssertionError(e);
206     }
207   }

209   private static Appendable append(Appendable buffer, CharSequence s, int start, int end) {
210     try {
211       return buffer.append(s, start, end);
212     } catch (IOException e) {
213       throw new AssertionError(e);
214     }
215   }

217   private static Appendable backslashChar(Appendable buffer, char c) {
218     return append(append(buffer, '\\'), c);
219   }
220 
221   private static Appendable escapeCharacter(Appendable buffer, char c, char quote) {
222     if (c == quote) {
223       return backslashChar(buffer, c);
224     }
225     switch (c) {
226       case '\\':
227         return backslashChar(buffer, '\\');
228       case '\r':
229         return backslashChar(buffer, 'r');
230       case '\n':
231         return backslashChar(buffer, 'n');
232       case '\t':
233         return backslashChar(buffer, 't');
234       default:
235         if (c < 32) {
236           return append(buffer, String.format("\\x%02x", (int) c));
237         }
238         return append(buffer, c); // no need to support UTF-8
239     } // endswitch
240   }

242   /**
243    * Write a properly escaped Skylark representation of a string to a buffer.
244    *
245    * @param buffer the Appendable to write to.
246    * @param s the string a representation of which to write.
247    * @param quote the quote character to use, '"' or '\''.
248    * @return the Appendable, in fluent style.
249    */
250   public static Appendable writeString(Appendable buffer, String s, char quote) {
251     append(buffer, quote);
252     int len = s.length();
253     for (int i = 0; i < len; i++) {
254       char c = s.charAt(i);
255       escapeCharacter(buffer, c, quote);
256     }
257     return append(buffer, quote);
258   }

261    * Write a properly escaped Skylark representation of a string to a buffer.
262    * By default, standard Skylark convention is used, i.e., double-quoted single-line string,
263    * as opposed to standard Python convention, i.e. single-quoted single-line string.
264    *
265    * @param buffer the Appendable we're writing to.
266    * @param s the string a representation of which to write.
267    * @return the buffer, in fluent style.
268    */
269   public static Appendable writeString(Appendable buffer, String s) {
270     return writeString(buffer, s, SKYLARK_QUOTATION_MARK);

273   /**
274    * Print a list of object representations
275    * @param buffer an appendable buffer onto which to write the list.
276    * @param list the list of objects to write (each as with repr)
277    * @param before a string to print before the list
278    * @param separator a separator to print between each object
279    * @param after a string to print after the list
280    * @param singletonTerminator null or a string to print after the list if it is a singleton
281    * The singleton case is notably relied upon in python syntax to distinguish
282    * a tuple of size one such as ("foo",) from a merely parenthesized object such as ("foo").
283    * @param quotationMark The quotation mark to be used (' or ")
284    * @return the Appendable, in fluent style.
285    */
286   public static Appendable printList(Appendable buffer, Iterable<?> list, String before,
287       String separator, String after, String singletonTerminator, char quotationMark) {
288     boolean printSeparator = false; // don't print the separator before the first element
289     int len = 0;
290     append(buffer, before);
291     for (Object o : list) {
292       if (printSeparator) {
293         append(buffer, separator);
294       }
295       write(buffer, o, quotationMark);
296       printSeparator = true;
297       len++;
298     }
299     if (singletonTerminator != null && len == 1) {
300       append(buffer, singletonTerminator);
301     }
302     return append(buffer, after);
303   }

305   public static Appendable printList(Appendable buffer, Iterable<?> list, String before,
306       String separator, String after, String singletonTerminator) {
307     return printList(
308         buffer, list, before, separator, after, singletonTerminator, SKYLARK_QUOTATION_MARK);
309   }

351   private static List<?> makeList(Collection<?> list) {
352     return list == null ? Lists.newArrayList() : Lists.newArrayList(list);

355   private static List<String> makeStringList(List<Label> labels) {
356     if (labels == null) {
357       return Collections.emptyList();
358     }
359     List<String> strings = Lists.newArrayListWithCapacity(labels.size());
360     for (Label label : labels) {
361       strings.add(label.toString());
362     }
363     return strings;
364   }

366   /**
367    * Convert BUILD language objects to Formattable so JDK can render them correctly.
368    * Don't do this for numeric or string types because we want %d, %x, %s to work.
369    * This function is intended for use in assertions such as Precondition.checkArgument
370    * so that you only pay the cost of computing the string when the assertion passes.
371    */
372   public static Object strFormattable(final Object o, final char quotationMark) {
373     if (o instanceof Integer || o instanceof Double || o instanceof String) {
374       return o;
375     } else {
376       return new Formattable() {
377         @Override
378         public String toString() {
379           return str(o, quotationMark);
380         }
381 
382         @Override
383         public void formatTo(Formatter formatter, int flags, int width, int precision) {
384           print(formatter.out(), o, quotationMark);
385         }
386       };
387     }
388   }

391     return strFormattable(o, SKYLARK_QUOTATION_MARK);
392   }

451   /**
452    * Perform Python-style string formatting, as per pattern % tuple
453    * Limitations: only %d %s %r %% are supported.
454    *
455    * @param buffer an Appendable to output to.
456    * @param pattern a format string.
457    * @param arguments a list containing positional arguments.
458    * @return the buffer, in fluent style.
459    */
460   // TODO(bazel-team): support formatting arguments, and more complex Python patterns.
461   public static Appendable format(Appendable buffer, String pattern, List<?> arguments)
462       throws IllegalFormatException {
463     // N.B. MissingFormatWidthException is the only kind of IllegalFormatException
464     // whose constructor can take and display arbitrary error message, hence its use below.
465 
466     int length = pattern.length();
467     int argLength = arguments.size();
468     int i = 0; // index of next character in pattern
469     int a = 0; // index of next argument in arguments
470 
471     while (i < length) {
472       int p = pattern.indexOf('%', i);
473       if (p == -1) {
474         append(buffer, pattern, i, length);
475         break;
476       }
477       if (p > i) {
478         append(buffer, pattern, i, p);
479       }
480       if (p == length - 1) {
481         throw new MissingFormatWidthException(
482             "incomplete format pattern ends with %: " + repr(pattern));
483       }
484       char directive = pattern.charAt(p + 1);
485       i = p + 2;
486       switch (directive) {
487         case '%':
488           append(buffer, '%');
489           continue;
490         case 'd':
491         case 'r':
492         case 's':
493           if (a >= argLength) {
494             throw new MissingFormatWidthException("not enough arguments for format pattern "
495                 + repr(pattern) + ": "
496                 + repr(SkylarkList.tuple(arguments)));
497           }
498           Object argument = arguments.get(a++);
499           switch (directive) {
500             case 'd':
501               if (argument instanceof Integer) {
502                 append(buffer, argument.toString());
503                 continue;
504               } else {
505                 throw new MissingFormatWidthException(
506                     "invalid argument " + repr(argument) + " for format pattern %d");
507               }
508             case 'r':
509               write(buffer, argument);
510               continue;
511             case 's':
512               print(buffer, argument);
513               continue;
514           }
515         default:
516           throw new MissingFormatWidthException(
517               "unsupported format character " + repr(String.valueOf(directive))
518               + " at index " + (p + 1) + " in " + repr(pattern));
519       }
520     }
521     if (a < argLength) {
522       throw new MissingFormatWidthException(
523           "not all arguments converted during string formatting");
524     }
525     return buffer;
526   }
