46  /**
47   * The Skyframe function that generates aspects.
48   */
49  public final class AspectFunction implements SkyFunction {
50    private final BuildViewProvider buildViewProvider;
51  
52    public AspectFunction(BuildViewProvider buildViewProvider) {
53      this.buildViewProvider = buildViewProvider;
54    }
55  
56    @Nullable
57    @Override
58    public SkyValue compute(SkyKey skyKey, Environment env)
59        throws AspectFunctionException {
60      SkyframeBuildView view = buildViewProvider.getSkyframeBuildView();
61      AspectKey key = (AspectKey) skyKey.argument();
62      ConfiguredAspectFactory aspectFactory =
63          (ConfiguredAspectFactory) AspectFactory.Util.create(key.getAspect());
64  
65      PackageValue packageValue =
66          (PackageValue) env.getValue(PackageValue.key(key.getLabel().getPackageIdentifier()));
67      if (packageValue == null) {
68        return null;
69      }
70  
71      Target target;
72      try {
73        target = packageValue.getPackage().getTarget(key.getLabel().getName());
74      } catch (NoSuchTargetException e) {
75        throw new AspectFunctionException(skyKey, e);
76      }
77  
78      if (!(target instanceof Rule)) {
79        throw new AspectFunctionException(new AspectCreationException(
80            "aspects must be attached to rules"));
81      }
82  
83      RuleConfiguredTarget associatedTarget = (RuleConfiguredTarget)
84          ((ConfiguredTargetValue) env.getValue(ConfiguredTargetValue.key(
85              key.getLabel(), key.getConfiguration()))).getConfiguredTarget();
86  
87      if (associatedTarget == null) {
88        return null;
89      }
90  
91      SkyframeDependencyResolver resolver = view.createDependencyResolver(env);
92      if (resolver == null) {
93        return null;
94      }
95  
96      TargetAndConfiguration ctgValue =
97          new TargetAndConfiguration(target, key.getConfiguration());
98  
99      try {
100       // Get the configuration targets that trigger this rule's configurable attributes.
101       Set<ConfigMatchingProvider> configConditions =
102           ConfiguredTargetFunction.getConfigConditions(target, env, resolver, ctgValue);
103       if (configConditions == null) {
104         // Those targets haven't yet been resolved.
105         return null;
106       }
107 
108       ListMultimap<Attribute, ConfiguredTarget> depValueMap =
109           ConfiguredTargetFunction.computeDependencies(env, resolver, ctgValue,
110               aspectFactory.getDefinition(), configConditions);
111 
112       return createAspect(env, key, associatedTarget, configConditions, depValueMap);
113     } catch (DependencyEvaluationException e) {
114       throw new AspectFunctionException(e.getRootCauseSkyKey(), e.getCause());
115     }
116   }
117 
118   @Nullable
119   private AspectValue createAspect(Environment env, AspectKey key,
120       RuleConfiguredTarget associatedTarget, Set<ConfigMatchingProvider> configConditions,
121       ListMultimap<Attribute, ConfiguredTarget> directDeps)
122       throws AspectFunctionException {
123     SkyframeBuildView view = buildViewProvider.getSkyframeBuildView();
124     BuildConfiguration configuration = associatedTarget.getConfiguration();
125 
126     StoredEventHandler events = new StoredEventHandler();
127     CachingAnalysisEnvironment analysisEnvironment = view.createAnalysisEnvironment(
128         key, false, events, env, configuration);
129     if (env.valuesMissing()) {
130       return null;
131     }
132 
133     ConfiguredAspectFactory aspectFactory =
134         (ConfiguredAspectFactory) AspectFactory.Util.create(key.getAspect());
135     Aspect aspect = view.createAspect(
136         analysisEnvironment, associatedTarget, aspectFactory, directDeps, configConditions);
137 
138     events.replayOn(env.getListener());
139     if (events.hasErrors()) {
140       analysisEnvironment.disable(associatedTarget.getTarget());
141       throw new AspectFunctionException(new AspectCreationException(
142           "Analysis of target '" + associatedTarget.getLabel() + "' failed; build aborted"));
143     }
144     Preconditions.checkState(!analysisEnvironment.hasErrors(),
145         "Analysis environment hasError() but no errors reported");
146 
147     if (env.valuesMissing()) {
148       return null;
149     }
150 
151     analysisEnvironment.disable(associatedTarget.getTarget());
152     Preconditions.checkNotNull(aspect);
153 
154     return new AspectValue(
155         aspect, ImmutableList.copyOf(analysisEnvironment.getRegisteredActions()));
156   }
157 
158   @Nullable
159   @Override
160   public String extractTag(SkyKey skyKey) {
161     return null;
162   }
163   
164   /**
165    * An exception indicating that there was a problem creating an aspect.
166    */
167   public static final class AspectCreationException extends Exception {
168     public AspectCreationException(String message) {
169       super(message);
170     }
171   }
172 
173   /**
174    * Used to indicate errors during the computation of an {@link AspectValue}.
175    */
176   private static final class AspectFunctionException extends SkyFunctionException {
177     public AspectFunctionException(Exception e) {
178       super(e, Transience.PERSISTENT);
179     }
180 
181     /** Used to rethrow a child error that we cannot handle. */
182     public AspectFunctionException(SkyKey childKey, Exception transitiveError) {
183       super(transitiveError, childKey);
184     }
185   }
186 }
