293   @Override
294   public FragmentOptions getHost(boolean fallback) {
295     JavaOptions host = (JavaOptions) getDefault();
296 
297     host.javaBase = hostJavaBase;
298     host.jvmOpts = ImmutableList.of("-client", "-XX:ErrorFile=/dev/stderr");
299 
300     host.javacOpts = javacOpts;
301     host.javaLangtoolsJar = javaLangtoolsJar;
302     host.javacExtdir = javacExtdir;
303     host.javaBuilderTop = javaBuilderTop;
304     host.javaToolchain = javaToolchain;
305     host.singleJarTop = singleJarTop;
306     host.genClassTop = genClassTop;
307     host.iJarTop = iJarTop;
308 
309     // Java builds often contain complicated code generators for which
310     // incremental build performance is important.
311     host.useIjars = useIjars;
312 
313     host.javaDeps = javaDeps;
314     host.experimentalJavaClasspath = experimentalJavaClasspath;
315 
316     return host;
317   }
318 
319   @Override
320   public void addAllLabels(Multimap<String, Label> labelMap) {
321     addOptionalLabel(labelMap, "jdk", javaBase);
322     addOptionalLabel(labelMap, "jdk", hostJavaBase);
323     if (javaLauncher != null) {
324       labelMap.put("java_launcher", javaLauncher);
325     }
326     labelMap.put("javabuilder", javaBuilderTop);
327     labelMap.put("singlejar", singleJarTop);
328     labelMap.put("genclass", genClassTop);
329     labelMap.put("ijar", iJarTop);
330     labelMap.put("java_toolchain", javaToolchain);
331     labelMap.putAll("translation", getTranslationLabels());
332   }
333 
334   @Override
335   public Map<String, Set<Label>> getDefaultsLabels(BuildConfiguration.Options commonOptions) {
336     Set<Label> jdkLabels = new LinkedHashSet<>();
337     DefaultsPackage.parseAndAdd(jdkLabels, javaBase);
338     DefaultsPackage.parseAndAdd(jdkLabels, hostJavaBase);
339     Map<String, Set<Label>> result = new HashMap<>();
340     result.put("JDK", jdkLabels);
341     result.put("JAVA_LANGTOOLS", ImmutableSet.of(javaLangtoolsJar));
342     result.put("JAVAC_BOOTCLASSPATH", ImmutableSet.of(javacBootclasspath));
343     result.put("JAVAC_EXTDIR", ImmutableSet.of(javacExtdir));
344     result.put("JAVABUILDER", ImmutableSet.of(javaBuilderTop));
345     result.put("SINGLEJAR", ImmutableSet.of(singleJarTop));
346     result.put("GENCLASS", ImmutableSet.of(genClassTop));
347     result.put("IJAR", ImmutableSet.of(iJarTop));
348     result.put("JAVA_TOOLCHAIN", ImmutableSet.of(javaToolchain));
349 
350     return result;
351   }
352 
353   private Set<Label> getTranslationLabels() {
354     Set<Label> result = new LinkedHashSet<>();
355     for (String s : translationTargets) {
356       try {
357         Label label = Label.parseAbsolute(s);
358         result.add(label);
359       } catch (SyntaxException e) {
360         // We ignore this exception here - it will cause an error message at a later time.
361       }
362     }
363     return result;
364   }
