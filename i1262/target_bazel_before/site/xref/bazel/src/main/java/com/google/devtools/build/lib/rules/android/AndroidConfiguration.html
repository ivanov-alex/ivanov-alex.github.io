39  /**
40   * Configuration fragment for Android rules.
41   */
42  public class AndroidConfiguration extends BuildConfiguration.Fragment {
43    /**
44     * Value used to avoid multiple configurations from conflicting.
45     *
46     * <p>This is set to {@code ANDROID} in Android configurations and to {@code MAIN} otherwise. This
47     * influences the output directory name: if it didn't, an Android and a non-Android configuration
48     * would conflict if they had the same toolchain identifier.
49     *
50     * <p>Note that this is not just a theoretical concern: even if {@code --crosstool_top} and
51     * {@code --android_crosstool_top} point to different labels, they may end up being redirected to
52     * the same thing, and this is exactly what happens on OSX X.
53     */
54    public enum ConfigurationDistinguisher {
55      MAIN(null),
56      ANDROID("android");
57  
58      private final String suffix;
59  
60      private ConfigurationDistinguisher(String suffix) {
61        this.suffix = suffix;
62      };
63    }
64  
65    /**
66     * Converter for {@link com.google.devtools.build.lib.rules.android.AndroidConfiguration.ConfigurationDistinguisher}
67     */
68    public static final class ConfigurationDistinguisherConverter
69        extends EnumConverter<ConfigurationDistinguisher> {
70      public ConfigurationDistinguisherConverter() {
71        super(ConfigurationDistinguisher.class, "Android configuration distinguisher");
72      }
73    }

75    /**
76     * Android configuration options.
77     */
78    public static class Options extends FragmentOptions {
79      // Spaces make it impossible to specify this on the command line
80      @Option(name = "Android configuration distinguisher",
81          defaultValue = "MAIN",
82          converter = ConfigurationDistinguisherConverter.class,
83          category = "undocumented")
84      public ConfigurationDistinguisher configurationDistinguisher;
85  
86      // For deploying incremental installation of native libraries. Do not use on the command line.
87      // The idea is that once this option works, we'll flip the default value in a config file, then
88      // once it is proven that it works, remove it from Bazel and said config file.
89      @Option(name = "android_incremental_native_libs",
90          defaultValue = "false",
91          category = "undocumented")
92      public boolean incrementalNativeLibs;
93  
94      @Option(name = "android_crosstool_top",
95          defaultValue = "null",
96          category = "semantics",
97          converter = LabelConverter.class,
98          help = "The location of the C++ compiler used for Android builds.")
99      public Label androidCrosstoolTop;
100 
101     @Option(name = "android_cpu",
102         defaultValue = "armeabi",
103         category = "semantics",
104         help = "The Android target CPU.")
105     public String cpu;
106 
107     @Option(name = "strict_android_deps",
108         allowMultiple = false,
109         defaultValue = "default",
110         converter = StrictDepsConverter.class,
111         category = "semantics",
112         help = "If true, checks that an Android target explicitly declares all directly used "
113             + "targets as dependencies.")
114     public StrictDepsMode strictDeps;
115 
116     // Label of filegroup combining all Android tools used as implicit dependencies of
117     // android_* rules
118     @Option(name = "android_sdk",
119             defaultValue = "null",
120             category = "version",
121             converter = LabelConverter.class,
122             help = "Specifies Android SDK/platform that is used to build Android applications.")
123     public Label sdk;
124 
125     @Option(name = "proguard_top",
126         defaultValue = "null",
127         category = "version",
128         converter = LabelConverter.class,
129         help = "Specifies which version of ProGuard to use for code removal when building an "
130             + "Android binary.")
131     public Label proguard;
132 
133     @Option(name = "legacy_android_native_support",
134         defaultValue = "true",
135         category = "semantics",
136         help = "Switches back to old native support for android_binaries. Disable to link together "
137             + "native deps of android_binaries into a single .so by default.")
138     public boolean legacyNativeSupport;
139 
140     // TODO(bazel-team): Maybe merge this with --android_cpu above.
141     @Option(name = "fat_apk_cpu",
142             converter = Converters.CommaSeparatedOptionListConverter.class,
143             allowMultiple = true,
144             defaultValue = "",
145             category = "undocumented",
146             help = "Setting this option enables fat APKs, which contain native binaries for all "
147                 + "specified target architectures, e.g., --fat_apk_cpu=x86,armeabi-v7a. Note that "
148                 + "you will also at least need to select an Android-compatible crosstool. "
149                 + "If this flag is specified, then --android_cpu is ignored for dependencies of "
150                 + "android_binary rules.")
151     public List<String> fatApkCpus;
152 
153     @Option(name = "experimental_android_use_jack_for_dexing",
154         defaultValue = "false",
155         category = "semantics",
156         help = "Switches to the Jack and Jill toolchain for dexing instead of javac and dx.")
157     public boolean useJackForDexing;
158 
159     @Option(name = "experimental_android_jack_sanity_checks",
160         defaultValue = "false",
161         category = "semantics",
162         help = "Enables sanity checks for Jack and Jill compilation.")
163     public boolean jackSanityChecks;
164 
165     @Override
166     public void addAllLabels(Multimap<String, Label> labelMap) {
167       if (proguard != null) {
168         labelMap.put("android_proguard", proguard);
169       }
170 
171       if (realAndroidCrosstoolTop() != null) {
172         labelMap.put("android_crosstool_top", realAndroidCrosstoolTop());
173       }
174 
175       labelMap.put("android_sdk", realSdk());
176     }
177 
178     // This method is here because Constants.ANDROID_DEFAULT_SDK cannot be a constant, because we
179     // replace the class file in the .jar after compilation. However, that means that we cannot use
180     // it as an attribute value in an annotation.
181     private Label realSdk() {
182       return sdk == null
183           ? Label.parseAbsoluteUnchecked(Constants.ANDROID_DEFAULT_SDK)
184           : sdk;
185     }
186 
187     // This method is here because Constants.ANDROID_DEFAULT_CROSSTOOL cannot be a constant, because
188     // we replace the class file in the .jar after compilation. However, that means that we cannot
189     // use it as an attribute value in an annotation.
190     public Label realAndroidCrosstoolTop() {
191       if (androidCrosstoolTop != null) {
192         return androidCrosstoolTop;
193       }
194 
195       if (Constants.ANDROID_DEFAULT_CROSSTOOL.equals("null")) {
196         return null;
197       }
198 
199       return Label.parseAbsoluteUnchecked(Constants.ANDROID_DEFAULT_CROSSTOOL);
200     }
201 
202     @Override
203     public ImmutableList<String> getDefaultsRules() {
204       return ImmutableList.of("android_tools_defaults_jar(name = 'android_jar')");
205     }
206 
207     @Override
208     public List<SplitTransition<BuildOptions>> getPotentialSplitTransitions() {
209       return ImmutableList.of(AndroidRuleClasses.ANDROID_SPLIT_TRANSITION);
210     }
211   }

213   /**
214    * Configuration loader for the Android fragment.
215    */
216   public static class Loader implements ConfigurationFragmentFactory {
217     @Override
218     public Fragment create(ConfigurationEnvironment env, BuildOptions buildOptions)
219         throws InvalidConfigurationException {
220       return new AndroidConfiguration(buildOptions.get(Options.class));
221     }
222 
223     @Override
224     public Class<? extends Fragment> creates() {
225       return AndroidConfiguration.class;
226     }
227 
228     @Override
229     public ImmutableSet<Class<? extends FragmentOptions>> requiredOptions() {
230       return ImmutableSet.<Class<? extends FragmentOptions>>of(Options.class);
231     }
232   }

234   private final Label sdk;
235   private final StrictDepsMode strictDeps;
236   private final boolean legacyNativeSupport;
237   private final String cpu;
238   private final boolean incrementalNativeLibs;
239   private final boolean fatApk;
240   private final ConfigurationDistinguisher configurationDistinguisher;
241   private final Label proguard;
242   private final boolean useJackForDexing;
243   private final boolean jackSanityChecks;
244 
245   AndroidConfiguration(Options options) {
246     this.sdk = options.realSdk();
247     this.incrementalNativeLibs = options.incrementalNativeLibs;
248     this.strictDeps = options.strictDeps;
249     this.legacyNativeSupport = options.legacyNativeSupport;
250     this.cpu = options.cpu;
251     this.fatApk = !options.fatApkCpus.isEmpty();
252     this.configurationDistinguisher = options.configurationDistinguisher;
253     this.proguard = options.proguard;
254     this.useJackForDexing = options.useJackForDexing;
255     this.jackSanityChecks = options.jackSanityChecks;
256   }
