45  /**
46   * Build a post-processed ConfiguredTarget, vetting it for action conflict issues.
47   */
48  public class PostConfiguredTargetFunction implements SkyFunction {
49    private static final Function<Dependency, SkyKey> TO_KEYS =
50        new Function<Dependency, SkyKey>() {
51      @Override
52      public SkyKey apply(Dependency input) {
53        return PostConfiguredTargetValue.key(
54            new ConfiguredTargetKey(input.getLabel(), input.getConfiguration()));
55      }
56    };
57  
58    private final SkyframeExecutor.BuildViewProvider buildViewProvider;
59  
60    public PostConfiguredTargetFunction(
61        SkyframeExecutor.BuildViewProvider buildViewProvider) {
62      this.buildViewProvider = Preconditions.checkNotNull(buildViewProvider);
63    }
64  
65    @Nullable
66    @Override
67    public SkyValue compute(SkyKey skyKey, Environment env) throws SkyFunctionException {
68      ImmutableMap<Action, ConflictException> badActions = PrecomputedValue.BAD_ACTIONS.get(env);
69      ConfiguredTargetValue ctValue = (ConfiguredTargetValue)
70          env.getValue(ConfiguredTargetValue.key((ConfiguredTargetKey) skyKey.argument()));
71      SkyframeDependencyResolver resolver =
72          buildViewProvider.getSkyframeBuildView().createDependencyResolver(env);
73      if (env.valuesMissing()) {
74        return null;
75      }
76  
77      for (Action action : ctValue.getActions()) {
78        if (badActions.containsKey(action)) {
79          throw new ActionConflictFunctionException(badActions.get(action));
80        }
81      }
82  
83      ConfiguredTarget ct = ctValue.getConfiguredTarget();
84      TargetAndConfiguration ctgValue =
85          new TargetAndConfiguration(ct.getTarget(), ct.getConfiguration());
86  
87      Set<ConfigMatchingProvider> configConditions =
88          getConfigurableAttributeConditions(ctgValue, env);
89      if (configConditions == null) {
90        return null;
91      }
92  
93      Collection<Dependency> deps = resolver.dependentNodes(ctgValue, configConditions);
94      env.getValues(Iterables.transform(deps, TO_KEYS));
95      if (env.valuesMissing()) {
96        return null;
97      }
98  
99      return new PostConfiguredTargetValue(ct);
100   }
101 
102   /**
103    * Returns the configurable attribute conditions necessary to evaluate the given configured
104    * target, or null if not all dependencies have yet been SkyFrame-evaluated.
105    */
106   @Nullable
107   private Set<ConfigMatchingProvider> getConfigurableAttributeConditions(
108       TargetAndConfiguration ctg, Environment env) {
109     if (!(ctg.getTarget() instanceof Rule)) {
110       return ImmutableSet.of();
111     }
112     Rule rule = (Rule) ctg.getTarget();
113     RawAttributeMapper mapper = RawAttributeMapper.of(rule);
114     Set<SkyKey> depKeys = new LinkedHashSet<>();
115     for (Attribute attribute : rule.getAttributes()) {
116       for (Label label : mapper.getConfigurabilityKeys(attribute.getName(), attribute.getType())) {
117         if (!Type.Selector.isReservedLabel(label)) {
118           depKeys.add(ConfiguredTargetValue.key(label, ctg.getConfiguration()));
119         }
120       }
121     }
122     Map<SkyKey, SkyValue> cts = env.getValues(depKeys);
123     if (env.valuesMissing()) {
124       return null;
125     }
126     ImmutableSet.Builder<ConfigMatchingProvider> conditions = ImmutableSet.builder();
127     for (SkyValue ctValue : cts.values()) {
128       ConfiguredTarget ct = ((ConfiguredTargetValue) ctValue).getConfiguredTarget();
129       conditions.add(Preconditions.checkNotNull(ct.getProvider(ConfigMatchingProvider.class)));
130     }
131     return conditions.build();
132   }
133 
134   @Nullable
135   @Override
136   public String extractTag(SkyKey skyKey) {
137     return Label.print(((LabelAndConfiguration) skyKey.argument()).getLabel());
138   }
139 
140   private static class ActionConflictFunctionException extends SkyFunctionException {
141     public ActionConflictFunctionException(ConflictException e) {
142       super(e, Transience.PERSISTENT);
143     }
144   }
