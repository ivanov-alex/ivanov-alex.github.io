176   /**
177    * Creates a new action to compile C/C++ source files.
178    *
179    * @param owner the owner of the action, usually the configured target that
180    *        emitted it
181    * @param sourceFile the source file that should be compiled. {@code mandatoryInputs} must
182    *        contain this file
183    * @param sourceLabel the label of the rule the source file is generated by
184    * @param mandatoryInputs any additional files that need to be present for the
185    *        compilation to succeed, can be empty but not null, for example, extra sources for FDO.
186    * @param outputFile the object file that is written as result of the
187    *        compilation, or the fake object for {@link FakeCppCompileAction}s
188    * @param dotdFile the .d file that is generated as a side-effect of
189    *        compilation
190    * @param gcnoFile the coverage notes that are written in coverage mode, can
191    *        be null
192    * @param dwoFile the .dwo output file where debug information is stored for Fission
193    *        builds (null if Fission mode is disabled)
194    * @param optionalSourceFile an additional optional source file (null if unneeded)
195    * @param configuration the build configurations
196    * @param context the compilation context
197    * @param copts options for the compiler
198    * @param coptsFilter regular expression to remove options from {@code copts}
199    */
200   protected CppCompileAction(ActionOwner owner,
201       // TODO(bazel-team): Eventually we will remove 'features'; all functionality in 'features'
202       // will be provided by 'featureConfiguration'.
203       ImmutableList<String> features,
204       FeatureConfiguration featureConfiguration,
205       CcToolchainFeatures.Variables variables,
206       Artifact sourceFile,
207       Label sourceLabel,
208       NestedSet<Artifact> mandatoryInputs,
209       Artifact outputFile,
210       DotdFile dotdFile,
211       @Nullable Artifact gcnoFile,
212       @Nullable Artifact dwoFile,
213       Artifact optionalSourceFile,
214       BuildConfiguration configuration,
215       CppConfiguration cppConfiguration,
216       CppCompilationContext context,
217       Class<? extends CppCompileActionContext> actionContext,
218       ImmutableList<String> copts,
219       ImmutableList<String> pluginOpts,
220       Predicate<String> coptsFilter,
221       ImmutableList<PathFragment> extraSystemIncludePrefixes,
222       @Nullable String fdoBuildStamp,
223       IncludeResolver includeResolver,
224       Iterable<IncludeScannable> lipoScannables,
225       UUID actionClassId,
226       boolean usePic,
227       RuleContext ruleContext) {
228     super(owner,
229           createInputs(mandatoryInputs, context.getCompilationPrerequisites(), optionalSourceFile),
230           CollectionUtils.asListWithoutNulls(outputFile, dotdFile.artifact(),
231               gcnoFile, dwoFile));
232     this.configuration = configuration;
233     this.sourceLabel = sourceLabel;
234     this.outputFile = Preconditions.checkNotNull(outputFile);
235     this.dwoFile = dwoFile;
236     this.optionalSourceFile = optionalSourceFile;
237     this.context = context;
238     this.extraSystemIncludePrefixes = extraSystemIncludePrefixes;
239     this.includeResolver = includeResolver;
240     this.cppConfiguration = cppConfiguration;
241     if (cppConfiguration != null && !cppConfiguration.shouldScanIncludes()) {
242       inputsKnown = true;
243     }
244     this.cppCompileCommandLine = new CppCompileCommandLine(sourceFile, dotdFile,
245         copts, coptsFilter, pluginOpts, (gcnoFile != null), features,
246         featureConfiguration, variables, fdoBuildStamp);
247     this.actionContext = actionContext;
248     this.lipoScannables = lipoScannables;
249     this.actionClassId = actionClassId;
250     this.usePic = usePic;
251 
252     // We do not need to include the middleman artifact since it is a generated
253     // artifact and will definitely exist prior to this action execution.
254     this.mandatoryInputs = mandatoryInputs;
255     verifyIncludePaths(ruleContext);
256   }

258   /**
259    * Verifies that the include paths of this action are within the limits of the execution root.
260    */
261   private void verifyIncludePaths(RuleContext ruleContext) {
262     if (ruleContext == null) {
263       return;
264     }
265 
266     Iterable<PathFragment> ignoredDirs = getValidationIgnoredDirs();
267 
268     // We currently do not check the output of:
269     // - getQuoteIncludeDirs(): those only come from includes attributes, and are checked in
270     //   CcCommon.getIncludeDirsFromIncludesAttribute().
271     // - getBuiltinIncludeDirs(): while in practice this doesn't happen, bazel can be configured
272     //   to use an absolute system root, in which case the builtin include dirs might be absolute.
273     for (PathFragment include : Iterables.concat(getIncludeDirs(), getSystemIncludeDirs())) {
274 
275       // Ignore headers from built-in include directories.
276       if (FileSystemUtils.startsWithAny(include, ignoredDirs)) {
277         continue;
278       }
279 
280       if (include.isAbsolute()
281           || !PathFragment.EMPTY_FRAGMENT.getRelative(include).normalize().isNormalized()) {
282         ruleContext.ruleError(
283             "The include path '" + include + "' references a path outside of the execution root.");
284       }
285     }
286   }

288   private static NestedSet<Artifact> createInputs(
289       NestedSet<Artifact> mandatoryInputs,
290       Set<Artifact> prerequisites, Artifact optionalSourceFile) {
291     NestedSetBuilder<Artifact> builder = NestedSetBuilder.stableOrder();
292     if (optionalSourceFile != null) {
293       builder.add(optionalSourceFile);
294     }
295     builder.addAll(prerequisites);
296     builder.addTransitive(mandatoryInputs);
297     return builder.build();
298   }

345   @Nullable
346   @Override
347   public Collection<Artifact> discoverInputs(ActionExecutionContext actionExecutionContext)
348       throws ActionExecutionException, InterruptedException {
349     Executor executor = actionExecutionContext.getExecutor();
350     Collection<Artifact> initialResult;
351     try {
352       initialResult = executor.getContext(actionContext)
353           .findAdditionalInputs(this, actionExecutionContext);
354     } catch (ExecException e) {
355       throw e.toActionExecutionException("Include scanning of rule '" + getOwner().getLabel() + "'",
356           executor.getVerboseFailures(), this);
357     }
358     if (initialResult == null) {
359       // We will find inputs during execution. Store an empty list to show we did try to discover
360       // inputs and return null to inform the caller that inputs will be discovered later.
361       this.additionalInputs = ImmutableList.of();
362       return null;
363     }
364     this.additionalInputs = initialResult;
365     // In some cases, execution backends need extra files for each included file. Add them
366     // to the set of inputs the caller may need to be aware of.
367     Collection<Artifact> result = new HashSet<>();
368     ArtifactResolver artifactResolver =
369         executor.getContext(IncludeScanningContext.class).getArtifactResolver();
370     for (Artifact artifact : initialResult) {
371       result.addAll(includeResolver.getInputsForIncludedFile(artifact, artifactResolver));
372     }
373     for (Artifact artifact : getInputs()) {
374       result.addAll(includeResolver.getInputsForIncludedFile(artifact, artifactResolver));
375     }
376     if (result.isEmpty()) {
377       result = initialResult;
378     } else {
379       result.addAll(initialResult);
380     }
381     return result;
382   }

430   @Override
431   public Map<Artifact, Artifact> getLegalGeneratedScannerFileMap() {
432     Map<Artifact, Artifact> legalOuts = new HashMap<>();
433 
434     for (Artifact a : context.getDeclaredIncludeSrcs()) {
435       if (!a.isSourceArtifact()) {
436         legalOuts.put(a, null);
437       }
438     }
439     for (Pair<Artifact, Artifact> pregreppedSrcs : context.getPregreppedHeaders()) {
440       Artifact hdr = pregreppedSrcs.getFirst();
441       Preconditions.checkState(!hdr.isSourceArtifact(), hdr);
442       legalOuts.put(hdr, pregreppedSrcs.getSecond());
443     }
444     return Collections.unmodifiableMap(legalOuts);
445   }

470   @Override
471   public List<PathFragment> getIncludeDirs() {
472     ImmutableList.Builder<PathFragment> result = ImmutableList.builder();
473     result.addAll(context.getIncludeDirs());
474     for (String opt : cppCompileCommandLine.copts) {
475       if (opt.startsWith("-I") && opt.length() > 2) {
476         // We insist on the combined form "-Idir".
477         result.add(new PathFragment(opt.substring(2)));
478       }
479     }
480     return result.build();
481   }

483   @Override
484   public List<PathFragment> getSystemIncludeDirs() {
485     // TODO(bazel-team): parsing the command line flags here couples us to gcc-style compiler
486     // command lines; use a different way to specify system includes (for example through a
487     // system_includes attribute in cc_toolchain); note that that would disallow users from
488     // specifying system include paths via the copts attribute.
489     // Currently, this works together with the include_paths features because getCommandLine() will
490     // get the system include paths from the CppCompilationContext instead.
491     ImmutableList.Builder<PathFragment> result = ImmutableList.builder();
492     List<String> compilerOptions = getCompilerOptions();
493     for (int i = 0; i < compilerOptions.size(); i++) {
494       String opt = compilerOptions.get(i);
495       if (opt.startsWith("-isystem")) {
496         if (opt.length() > 8) {
497           result.add(new PathFragment(opt.substring(8).trim()));
498         } else if (i + 1 < compilerOptions.size()) {
499           i++;
500           result.add(new PathFragment(compilerOptions.get(i)));
501         } else {
502           System.err.println("WARNING: dangling -isystem flag in options for " + prettyPrint());
503         }
504       }
505     }
506     return result.build();
507   }

529   @Override
530   public Collection<Artifact> getIncludeScannerSources() {
531     NestedSetBuilder<Artifact> builder = NestedSetBuilder.stableOrder();
532     // For every header module we use for the build we need the set of sources that it can
533     // reference.
534     builder.addTransitive(context.getTransitiveHeaderModuleSrcs());
535     if (CppFileTypes.CPP_MODULE_MAP.matches(getSourceFile().getPath())) {
536       // If this is an action that compiles the header module itself, the source we build is the
537       // module map, and we need to include-scan all headers that are referenced in the module map.
538       // We need to do include scanning as long as we want to support building code bases that are
539       // not fully strict layering clean.
540       builder.addTransitive(context.getHeaderModuleSrcs());
541     } else {
542       builder.add(getSourceFile());
543     }
544     return builder.build().toCollection();
545   }

561   /**
562    * Returns an (immutable) map of environment key, value pairs to be
563    * provided to the C++ compiler.
564    */
565   public ImmutableMap<String, String> getEnvironment() {
566     Map<String, String> environment =
567         new LinkedHashMap<>(configuration.getDefaultShellEnvironment());
568     if (configuration.isCodeCoverageEnabled()) {
569       environment.put("PWD", "/proc/self/cwd");
570     }
571     if (OS.getCurrent() == OS.WINDOWS) {
572       // TODO(bazel-team): Both GCC and clang rely on their execution directories being on
573       // PATH, otherwise they fail to find dependent DLLs (and they fail silently...). On
574       // the other hand, Windows documentation says that the directory of the executable
575       // is always searched for DLLs first. Not sure what to make of it.
576       // Other options are to forward the system path (brittle), or to add a PATH field to
577       // the crosstool file.
578       environment.put("PATH", cppConfiguration.getToolPathFragment(Tool.GCC).getParentDirectory()
579           .getPathString());
580    }
581     return ImmutableMap.copyOf(environment);
582   }

625   /**
626    * Enforce that the includes actually visited during the compile were properly
627    * declared in the rules.
628    *
629    * <p>The technique is to walk through all of the reported includes that gcc
630    * emits into the .d file, and verify that they came from acceptable
631    * relative include directories. This is done in two steps:
632    *
633    * <p>First, each included file is stripped of any include path prefix from
634    * {@code quoteIncludeDirs} to produce an effective relative include dir+name.
635    *
636    * <p>Second, the remaining directory is looked up in {@code declaredIncludeDirs},
637    * a list of acceptable dirs. This list contains a set of dir fragments that
638    * have been calculated by the configured target to be allowable for inclusion
639    * by this source. If no match is found, an error is reported and an exception
640    * is thrown.
641    *
642    * @throws ActionExecutionException iff there was an undeclared dependency
643    */
644   @VisibleForTesting
645   public void validateInclusions(
646       MiddlemanExpander middlemanExpander, EventHandler eventHandler)
647       throws ActionExecutionException {
648     if (!cppConfiguration.shouldScanIncludes() || !inputsKnown()) {
649       return;
650     }
651 
652     IncludeProblems errors = new IncludeProblems();
653     IncludeProblems warnings = new IncludeProblems();
654     Set<Artifact> allowedIncludes = new HashSet<>();
655     for (Artifact input : mandatoryInputs) {
656       if (input.isMiddlemanArtifact()) {
657         middlemanExpander.expand(input, allowedIncludes);
658       }
659       allowedIncludes.add(input);
660     }
661 
662     if (optionalSourceFile != null) {
663       allowedIncludes.add(optionalSourceFile);
664     }
665     Iterable<PathFragment> ignoreDirs = getValidationIgnoredDirs();
666 
667     // Copy the sets to hash sets for fast contains checking.
668     // Avoid immutable sets here to limit memory churn.
669     Set<PathFragment> declaredIncludeDirs = Sets.newHashSet(context.getDeclaredIncludeDirs());
670     Set<PathFragment> warnIncludeDirs = Sets.newHashSet(context.getDeclaredIncludeWarnDirs());
671     Set<Artifact> declaredIncludeSrcs = Sets.newHashSet(context.getDeclaredIncludeSrcs());
672     for (Artifact input : getInputs()) {
673       if (context.getCompilationPrerequisites().contains(input)
674           || allowedIncludes.contains(input)) {
675         continue; // ignore our fixed source in mandatoryInput: we just want includes
676       }
677       // Ignore headers from built-in include directories.
678       if (FileSystemUtils.startsWithAny(input.getExecPath(), ignoreDirs)) {
679         continue;
680       }
681       if (!isDeclaredIn(input, declaredIncludeDirs, declaredIncludeSrcs)) {
682         // This call can never match the declared include sources (they would be matched above).
683         // There are no declared include sources we need to warn about, so use an empty set here.
684         if (isDeclaredIn(input, warnIncludeDirs, ImmutableSet.<Artifact>of())) {
685           warnings.add(input.getPath().toString());
686         } else {
687           errors.add(input.getPath().toString());
688         }
689       }
690     }
691     if (VALIDATION_DEBUG_WARN) {
692       synchronized (System.err) {
693         if (VALIDATION_DEBUG >= 2 || errors.hasProblems() || warnings.hasProblems()) {
694           if (errors.hasProblems()) {
695             System.err.println("ERROR: Include(s) were not in declared srcs:");
696           } else if (warnings.hasProblems()) {
697             System.err.println("WARN: Include(s) were not in declared srcs:");
698           } else {
699             System.err.println("INFO: Include(s) were OK for '" + getSourceFile()
700                 + "', declared srcs:");
701           }
702           for (Artifact a : context.getDeclaredIncludeSrcs()) {
703             System.err.println("  '" + a.toDetailString() + "'");
704           }
705           System.err.println(" or under declared dirs:");
706           for (PathFragment f : Sets.newTreeSet(context.getDeclaredIncludeDirs())) {
707             System.err.println("  '" + f + "'");
708           }
709           System.err.println(" or under declared warn dirs:");
710           for (PathFragment f : Sets.newTreeSet(context.getDeclaredIncludeWarnDirs())) {
711             System.err.println("  '" + f + "'");
712           }
713           System.err.println(" with prefixes:");
714           for (PathFragment dirpath : context.getQuoteIncludeDirs()) {
715             System.err.println("  '" + dirpath + "'");
716           }
717         }
718       }
719     }
720 
721     if (warnings.hasProblems()) {
722       eventHandler.handle(
723           new Event(EventKind.WARNING,
724               getOwner().getLocation(), warnings.getMessage(this, getSourceFile()),
725           Label.print(getOwner().getLabel())));
726     }
727     errors.assertProblemFree(this, getSourceFile());
728   }

736   /**
737    * Returns true if an included artifact is declared in a set of allowed
738    * include directories. The simple case is that the artifact's parent
739    * directory is contained in the set, or is empty.
740    *
741    * <p>This check also supports a wildcard suffix of '**' for the cases where the
742    * calculations are inexact.
743    *
744    * <p>It also handles unseen non-nested-package subdirs by walking up the path looking
745    * for matches.
746    */
747   private static boolean isDeclaredIn(
748       Artifact input, Set<PathFragment> declaredIncludeDirs, Set<Artifact> declaredIncludeSrcs) {
749     // First check if it's listed in "srcs". If so, then its declared & OK.
750     if (declaredIncludeSrcs.contains(input)) {
751       return true;
752     }
753     // If it's a derived artifact, then it MUST be listed in "srcs" as checked above.
754     // We define derived here as being not source and not under the include link tree.
755     if (!input.isSourceArtifact()
756         && !input.getRoot().getExecPath().getBaseName().equals("include")) {
757       return false;
758     }
759     // Need to do dir/package matching: first try a quick exact lookup.
760     PathFragment includeDir = input.getRootRelativePath().getParentDirectory();
761     if (includeDir.segmentCount() == 0 || declaredIncludeDirs.contains(includeDir)) {
762       return true;  // OK: quick exact match.
763     }
764     // Not found in the quick lookup: try the wildcards.
765     for (PathFragment declared : declaredIncludeDirs) {
766       if (declared.getBaseName().equals("**")) {
767         if (includeDir.startsWith(declared.getParentDirectory())) {
768           return true;  // OK: under a wildcard dir.
769         }
770       }
771     }
772     // Still not found: see if it is in a subdir of a declared package.
773     Path root = input.getRoot().getPath();
774     for (Path dir = input.getPath().getParentDirectory();;) {
775       if (dir.getRelative("BUILD").exists()) {
776         return false;  // Bad: this is a sub-package, not a subdir of a declared package.
777       }
778       dir = dir.getParentDirectory();
779       if (dir.equals(root)) {
780         return false;  // Bad: at the top, give up.
781       }
782       if (declaredIncludeDirs.contains(dir.relativeTo(root))) {
783         return true;  // OK: found under a declared dir.
784       }
785     }
786   }

835   /**
836    * Populates the given ordered collection with additional input artifacts
837    * relevant to the specific action implementation.
838    *
839    * <p>The default implementation updates this Action's input set by reading
840    * dynamically-discovered dependency information out of the .d file.
841    *
842    * <p>Artifacts are considered inputs but not "mandatory" inputs.
843    *
844    *
845    * @param reply the reply from the compilation.
846    * @param inputs the ordered collection of inputs to append to
847    * @throws ActionExecutionException iff the .d is missing, malformed or has
848    *         unresolvable included artifacts.
849    */
850   @ThreadCompatible
851   private void populateActionInputs(Path execRoot,
852       ArtifactResolver artifactResolver, CppCompileActionContext.Reply reply,
853       NestedSetBuilder<Artifact> inputs)
854       throws ActionExecutionException {
855     try {
856       // Read .d file.
857       DependencySet depSet = processDepset(execRoot, reply);
858 
859       // Determine prefixes of allowed absolute inclusions.
860       CppConfiguration toolchain = cppConfiguration;
861       List<PathFragment> systemIncludePrefixes = new ArrayList<>();
862       for (PathFragment includePath : toolchain.getBuiltInIncludeDirectories()) {
863         if (includePath.isAbsolute()) {
864           systemIncludePrefixes.add(includePath);
865         }
866       }
867       systemIncludePrefixes.addAll(extraSystemIncludePrefixes);
868 
869       // Check inclusions.
870       IncludeProblems problems = new IncludeProblems();
871       Map<PathFragment, Artifact> allowedDerivedInputsMap = getAllowedDerivedInputsMap();
872       for (PathFragment execPath : depSet.getDependencies()) {
873         if (execPath.isAbsolute()) {
874           // Absolute includes from system paths are ignored.
875           if (FileSystemUtils.startsWithAny(execPath, systemIncludePrefixes)) {
876             continue;
877           }
878           // Since gcc is given only relative paths on the command line,
879           // non-system include paths here should never be absolute. If they
880           // are, it's probably due to a non-hermetic #include, & we should stop
881           // the build with an error.
882           if (execPath.startsWith(execRoot.asFragment())) {
883             execPath = execPath.relativeTo(execRoot.asFragment()); // funky but tolerable path
884           } else {
885             problems.add(execPath.getPathString());
886             continue;
887           }
888         }
889         Artifact artifact = allowedDerivedInputsMap.get(execPath);
890         if (artifact == null) {
891           artifact = artifactResolver.resolveSourceArtifact(execPath);
892         }
893         if (artifact != null) {
894           inputs.add(artifact);
895           // In some cases, execution backends need extra files for each included file. Add them
896           // to the set of actual inputs.
897           inputs.addAll(includeResolver.getInputsForIncludedFile(artifact, artifactResolver));
898         } else {
899           // Abort if we see files that we can't resolve, likely caused by
900           // undeclared includes or illegal include constructs.
901           problems.add(execPath.getPathString());
902         }
903       }
904       problems.assertProblemFree(this, getSourceFile());
905     } catch (IOException e) {
906       // Some kind of IO or parse exception--wrap & rethrow it to stop the build.
907       throw new ActionExecutionException("error while parsing .d file", e, this, false);
908     }
909   }

911   @Override
912   public Iterable<Artifact> resolveInputsFromCache(
913       ArtifactResolver artifactResolver, PackageRootResolver resolver,
914       Collection<PathFragment> inputPaths) throws PackageRootResolutionException {
915     // Note that this method may trigger a violation of the desirable invariant that getInputs()
916     // is a superset of getMandatoryInputs(). See bug about an "action not in canonical form"
917     // error message and the integration test test_crosstool_change_and_failure().
918     Map<PathFragment, Artifact> allowedDerivedInputsMap = getAllowedDerivedInputsMap();
919     List<Artifact> inputs = new ArrayList<>();
920     List<PathFragment> unresolvedPaths = new ArrayList<>();
921     for (PathFragment execPath : inputPaths) {
922       Artifact artifact = allowedDerivedInputsMap.get(execPath);
923       if (artifact != null) {
924         inputs.add(artifact);
925       } else {
926         // Remember this execPath, we will try to resolve it as a source artifact.
927         unresolvedPaths.add(execPath);
928       }
929     }
930 
931     Map<PathFragment, Artifact> resolvedArtifacts =
932         artifactResolver.resolveSourceArtifacts(unresolvedPaths, resolver);
933     if (resolvedArtifacts == null) {
934       // We are missing some dependencies. We need to rerun this update later.
935       return null;
936     }
937 
938     for (PathFragment execPath : unresolvedPaths) {
939       Artifact artifact = resolvedArtifacts.get(execPath);
940       // If PathFragment cannot be resolved into the artifact - ignore it. This could happen if
941       // rule definition has changed and action no longer depends on, e.g., additional source file
942       // in the separate package and that package is no longer referenced anywhere else.
943       // It is safe to ignore such paths because dependency checker would identify change in inputs
944       // (ignored path was used before) and will force action execution.
945       if (artifact != null) {
946         inputs.add(artifact);
947       }
948     }
949     return inputs;
950   }

1214     private String getActionName() {
1215       PathFragment sourcePath = sourceFile.getExecPath();
1216       if (CppFileTypes.CPP_MODULE_MAP.matches(sourcePath)) {
1217         return CPP_MODULE_COMPILE;
1218       } else if (CppFileTypes.CPP_HEADER.matches(sourcePath)) {
1219         // TODO(bazel-team): Handle C headers that probably don't work in C++ mode.
1220         if (featureConfiguration.isEnabled(CppRuleClasses.PARSE_HEADERS)) {
1221           return CPP_HEADER_PARSING;
1222         } else if (featureConfiguration.isEnabled(CppRuleClasses.PREPROCESS_HEADERS)) {
1223           return CPP_HEADER_PREPROCESSING;
1224         } else {
1225           // CcCommon.collectCAndCppSources() ensures we do not add headers to
1226           // the compilation artifacts unless either 'parse_headers' or
1227           // 'preprocess_headers' is set.
1228           throw new IllegalStateException();
1229         }
1230       } else if (CppFileTypes.C_SOURCE.matches(sourcePath)) {
1231         return C_COMPILE;
1232       } else if (CppFileTypes.CPP_SOURCE.matches(sourcePath)) {
1233         return CPP_COMPILE;
1234       } else if (CppFileTypes.ASSEMBLER_WITH_C_PREPROCESSOR.matches(sourcePath)) {
1235         return PREPROCESS_ASSEMBLE;
1236       }
1237       // CcLibraryHelper ensures CppCompileAction only gets instantiated for supported file types.
1238       throw new IllegalStateException();

1249       // Enable instrumentation if requested.
1250       if (isInstrumented) {
1251         addFilteredOptions(options, ImmutableList.of("-fprofile-arcs", "-ftest-coverage"));
1252       }
1253 
1254       String sourceFilename = sourceFile.getExecPathString();
1255       if (CppFileTypes.C_SOURCE.matches(sourceFilename)) {
1256         addFilteredOptions(options, toolchain.getCOptions());
1257       }
1258       if (CppFileTypes.CPP_SOURCE.matches(sourceFilename)
1259           || CppFileTypes.CPP_HEADER.matches(sourceFilename)
1260           || CppFileTypes.CPP_MODULE_MAP.matches(sourceFilename)) {
1261         addFilteredOptions(options, toolchain.getCxxOptions(features));
1262       }
1263 
1264       for (String warn : cppConfiguration.getCWarns()) {
1265         options.add("-W" + warn);
1266       }
1267       for (String define : context.getDefines()) {
1268         options.add("-D" + define);
1269       }
1270 
1271       // Stamp FDO builds with FDO subtype string
1272       if (fdoBuildStamp != null) {
1273         options.add("-D" + CppConfiguration.FDO_STAMP_MACRO + "=\"" + fdoBuildStamp + "\"");
1274       }
1275 
1276       // TODO(bazel-team): This needs to be before adding getUnfilteredCompilerOptions() and after
1277       // adding the warning flags until all toolchains are migrated; currently toolchains use the
1278       // unfiltered compiler options to inject include paths, which is superseded by the feature
1279       // configuration; on the other hand toolchains switch off warnings for the layering check
1280       // that will be re-added by the feature flags.
1281       addFilteredOptions(options,
1282           featureConfiguration.getCommandLine(getActionName(), variables));
1283 
1284       // Users don't expect the explicit copts to be filtered by coptsFilter, add them verbatim.
1285       // Make sure these are added after the options from the feature configuration, so that
1286       // those options can be overriden.
1287       options.addAll(copts);
1288 
1289       // Unfiltered compiler options contain system include paths. These must be added after
1290       // the user provided options, otherwise users adding include paths will not pick up their
1291       // own include paths first.
1292       options.addAll(toolchain.getUnfilteredCompilerOptions(features));
1293 
1294       // GCC gives randomized names to symbols which are defined in
1295       // an anonymous namespace but have external linkage.  To make
1296       // computation of these deterministic, we want to override the
1297       // default seed for the random number generator.  It's safe to use
1298       // any value which differs for all translation units; we use the
1299       // path to the object file.
1300       options.add("-frandom-seed=" + outputFile.getExecPathString());
1301 
1302       // Add the options of --per_file_copt, if the label or the base name of the source file
1303       // matches the specified regular expression filter.
1304       for (PerLabelOptions perLabelOptions : cppConfiguration.getPerFileCopts()) {
1305         if ((sourceLabel != null && perLabelOptions.isIncluded(sourceLabel))
1306             || perLabelOptions.isIncluded(sourceFile)) {
1307           options.addAll(perLabelOptions.getOptions());
1308         }
1309       }
1310 
1311       // Enable <object>.d file generation.
1312       if (dotdFile != null) {
1313         // Gcc options:
1314         //  -MD turns on .d file output as a side-effect (doesn't imply -E)
1315         //  -MM[D] enables user includes only, not system includes
1316         //  -MF <name> specifies the dotd file name
1317         // Issues:
1318         //  -M[M] alone subverts actual .o output (implies -E)
1319         //  -M[M]D alone breaks some of the .d naming assumptions
1320         // This combination gets user and system includes with specified name:
1321         //  -MD -MF <name>
1322         options.add("-MD");
1323         options.add("-MF");
1324         options.add(dotdFile.getSafeExecPath().getPathString());
1325       }
1326 
1327       if (FileType.contains(outputFile, CppFileTypes.ASSEMBLER, CppFileTypes.PIC_ASSEMBLER)) {
1328         options.add("-S");
1329       } else if (FileType.contains(outputFile, CppFileTypes.PREPROCESSED_C,
1330           CppFileTypes.PREPROCESSED_CPP, CppFileTypes.PIC_PREPROCESSED_C,
1331           CppFileTypes.PIC_PREPROCESSED_CPP)) {
1332         options.add("-E");
1333       }
1334 
1335       if (cppConfiguration.useFission()) {
1336         options.add("-gsplit-dwarf");
1337       }
1338       if (usePic) {
1339         options.add("-fPIC");
1340       }
1341 
1342       return options;
1343     }
1344 
1345     // For each option in 'in', add it to 'out' unless it is matched by the 'coptsFilter' regexp.
1346     private void addFilteredOptions(List<String> out, List<String> in) {
1347       Iterables.addAll(out, Iterables.filter(in, coptsFilter));
1348     }
1349   }

1351   /**
1352    * A reference to a .d file. There are two modes:
1353    * <ol>
1354    *   <li>an Artifact that represents a real on-disk file
1355    *   <li>just an execPath that refers to a virtual .d file that is not written to disk
1356    * </ol>
1357    */
1358   public static class DotdFile {
1359     private final Artifact artifact;
1360     private final PathFragment execPath;
1361 
1362     public DotdFile(Artifact artifact) {
1363       this.artifact = artifact;
1364       this.execPath = null;
1365     }
1366 
1367     public DotdFile(PathFragment execPath) {
1368       this.artifact = null;
1369       this.execPath = execPath;
1370     }
1371 
1372     /**
1373      * @return the Artifact or null
1374      */
1375     public Artifact artifact() {
1376       return artifact;
1377     }
1378 
1379     /**
1380      * @return Gets the execPath regardless of whether this is a real Artifact
1381      */
1382     public PathFragment getSafeExecPath() {
1383       return execPath == null ? artifact.getExecPath() : execPath;
1384     }
1385 
1386     /**
1387      * @return the on-disk location of the .d file or null
1388      */
1389     public Path getPath() {
1390       return artifact.getPath();
1391     }
