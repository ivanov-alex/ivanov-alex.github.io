86  /**
87   * Instances of BuildConfiguration represent a collection of context
88   * information which may affect a build (for example: the target platform for
89   * compilation, or whether or not debug tables are required).  In fact, all
90   * "environmental" information (e.g. from the tool's command-line, as opposed
91   * to the BUILD file) that can affect the output of any build tool should be
92   * explicitly represented in the BuildConfiguration instance.
93   *
94   * <p>A single build may require building tools to run on a variety of
95   * platforms: when compiling a server application for production, we must build
96   * the build tools (like compilers) to run on the host platform, but cross-compile
97   * the application for the production environment.
98   *
99   * <p>There is always at least one BuildConfiguration instance in any build:
100  * the one representing the host platform. Additional instances may be created,
101  * in a cross-compilation build, for example.
102  *
103  * <p>Instances of BuildConfiguration are canonical:
104  * <pre>c1.equals(c2) <=> c1==c2.</pre>
105  */
106 @SkylarkModule(name = "configuration",
107     doc = "Data required for the analysis of a target that comes from targets that "
108         + "depend on it and not targets that it depends on.")

261   /**
262    * A converter from strings to Labels.
263    */
264   public static class LabelConverter implements Converter<Label> {
265     @Override
266     public Label convert(String input) throws OptionsParsingException {
267       try {
268         // Check if the input starts with '/'. We don't check for "//" so that
269         // we get a better error message if the user accidentally tries to use
270         // an absolute path (starting with '/') for a label.
271         if (!input.startsWith("/") && !input.startsWith("@")) {
272           input = "//" + input;
273         }
274         return Label.parseAbsolute(input);
275       } catch (SyntaxException e) {
276         throw new OptionsParsingException(e.getMessage());
277       }
278     }
279 
280     @Override
281     public String getTypeDescription() {
282       return "a build target label";
283     }
284   }

289     public Map.Entry<String, String> convert(String input) throws OptionsParsingException {

291       if (index == -1) {
292         throw new OptionsParsingException("Plugin option not in the plugin=option format");
293       }
294       String option = input.substring(0, index);
295       String value = input.substring(index + 1);
296       return Maps.immutableEntry(option, value);
297     }

305   /** TODO(bazel-team): document this */
306   public static class RunsPerTestConverter extends PerLabelOptions.PerLabelOptionsConverter {
307     @Override
308     public PerLabelOptions convert(String input) throws OptionsParsingException {
309       try {
310         return parseAsInteger(input);
311       } catch (NumberFormatException ignored) {
312         return parseAsRegex(input);
313       }
314     }
315 
316     private PerLabelOptions parseAsInteger(String input)
317         throws NumberFormatException, OptionsParsingException {
318       int numericValue = Integer.parseInt(input);
319       if (numericValue <= 0) {
320         throw new OptionsParsingException("'" + input + "' should be >= 1");
321       } else {
322         RegexFilter catchAll = new RegexFilter(Collections.singletonList(".*"),
323             Collections.<String>emptyList());
324         return new PerLabelOptions(catchAll, Collections.singletonList(input));
325       }
326     }
327 
328     private PerLabelOptions parseAsRegex(String input) throws OptionsParsingException {
329       PerLabelOptions testRegexps = super.convert(input);
330       if (testRegexps.getOptions().size() != 1) {
331         throw new OptionsParsingException(
332             "'" + input + "' has multiple runs for a single pattern");
333       }
334       String runsPerTest = Iterables.getOnlyElement(testRegexps.getOptions());
335       try {
336         int numericRunsPerTest = Integer.parseInt(runsPerTest);
337         if (numericRunsPerTest <= 0) {
338           throw new OptionsParsingException("'" + input + "' has a value < 1");
339         }
340       } catch (NumberFormatException e) {
341         throw new OptionsParsingException("'" + input + "' has a non-numeric value", e);
342       }
343       return testRegexps;
344     }
345 
346     @Override
347     public String getTypeDescription() {
348       return "a positive integer or test_regex@runs. This flag may be passed more than once";
349     }

377   /**
378    * Converter for default --host_cpu to the auto-detected host cpu.
379    *
380    * <p>This detects the host cpu of the Blaze's server but if the compilation happens in a
381    * compilation cluster then the host cpu of the compilation cluster might be different than
382    * the auto-detected one and the --host_cpu option must then be set explicitly.
383    */
384   public static class HostCpuConverter implements Converter<String> {
385     @Override
386     public String convert(String input) throws OptionsParsingException {
387       if (input.isEmpty()) {
388         // TODO(philwo) - replace these deprecated names with more logical ones (e.g. k8 becomes
389         // linux-x86_64, darwin includes the CPU architecture, ...).
390         switch (OS.getCurrent()) {
391           case DARWIN:
392             return "darwin";
393           case LINUX:
394             switch (CPU.getCurrent()) {
395               case X86_32:
396                 return "piii";
397               case X86_64:
398                 return "k8";
399             }
400         }
401         return "unknown";
402       }
403       return input;
404     }
405 
406     @Override
407     public String getTypeDescription() {
408       return "a string";
409     }

871   /**
872    * Directories in the output tree.
873    * 
874    * <p>The computation of the output directory should be a non-injective mapping from
875    * BuildConfiguration instances to strings. The result should identify the aspects of the
876    * configuration that should be reflected in the output file names.  Furthermore the
877    * returned string must not contain shell metacharacters.
878    *
879    * <p>For configuration settings which are NOT part of the output directory name,
880    * rebuilding with a different value of such a setting will build in
881    * the same output directory.  This means that any actions whose
882    * keys (see Action.getKey()) have changed will be rerun.  That
883    * may result in a lot of recompilation.
884    *
885    * <p>For configuration settings which ARE part of the output directory name,
886    * rebuilding with a different value of such a setting will rebuild
887    * in a different output directory; this will result in higher disk
888    * usage and more work the <i>first</i> time you rebuild with a different
889    * setting, but will result in less work if you regularly switch
890    * back and forth between different settings.
891    *
892    * <p>With one important exception, it's sound to choose any subset of the
893    * config's components for this string, it just alters the dimensionality
894    * of the cache.  In other words, it's a trade-off on the "injectiveness"
895    * scale: at one extreme (output directory name contains all data in the config, and is
896    * thus injective) you get extremely precise caching (no competition for the
897    * same output-file locations) but you have to rebuild for even the
898    * slightest change in configuration.  At the other extreme (the output
899    * (directory name is a constant) you have very high competition for
900    * output-file locations, but if a slight change in configuration doesn't
901    * affect a particular build step, you're guaranteed not to have to
902    * rebuild it. The important exception has to do with multiple configurations: every
903    * configuration in the build must have a different output directory name so that
904    * their artifacts do not conflict.
905    *
906    * <p>The host configuration is special-cased: in order to guarantee that its output directory
907    * is always separate from that of the target configuration, we simply pin it to "host". We do
908    * this so that the build works even if the two configurations are too close (which is common)
909    * and so that the path of artifacts in the host configuration is a bit more readable.
910    */
911   private final OutputRoots outputRoots;

1090   /**
1091    * Computes and returns the transitive optionName -> "option info" map for
1092    * this configuration.
1093    */
1094   private static Map<String, OptionDetails> computeOptionsMap(BuildOptions buildOptions,
1095       Iterable<Fragment> fragments) {
1096     // Collect from our fragments "alternative defaults" for options where the default
1097     // should be something other than what's specified in Option.defaultValue.
1098     Map<String, Object> lateBoundDefaults = Maps.newHashMap();
1099     for (Fragment fragment : fragments) {
1100       lateBoundDefaults.putAll(fragment.lateBoundOptionDefaults());
1101     }
1102 
1103     ImmutableMap.Builder<String, OptionDetails> map = ImmutableMap.builder();
1104     try {
1105       for (FragmentOptions options : buildOptions.getOptions()) {
1106         for (Field field : options.getClass().getFields()) {
1107           if (field.isAnnotationPresent(Option.class)) {
1108             Option option = field.getAnnotation(Option.class);
1109             Object value = field.get(options);
1110             if (value == null) {
1111               if (lateBoundDefaults.containsKey(option.name())) {
1112                 value = lateBoundDefaults.get(option.name());
1113               } else if (!option.defaultValue().equals("null")) {
1114                  // See {@link Option#defaultValue} for an explanation of default "null" strings.
1115                 value = option.defaultValue();
1116               }
1117             }
1118             map.put(option.name(),
1119                 new OptionDetails(options.getClass(), value, option.allowMultiple()));
1120           }
1121         }
1122       }
1123     } catch (IllegalAccessException e) {
1124       throw new IllegalStateException(
1125           "Unexpected illegal access trying to create this configuration's options map: ", e);
1126     }
1127     return map.build();
1128   }

1219   /**
1220    * Calculates the configurations of a direct dependency. If a rule in some BUILD file refers
1221    * to a target (like another rule or a source file) using a label attribute, that target needs
1222    * to have a configuration, too. This method figures out the proper configuration for the
1223    * dependency.
1224    *
1225    * @param fromRule the rule that's depending on some target
1226    * @param attribute the attribute using which the rule depends on that target (eg. "srcs")
1227    * @param toTarget the target that's dependeded on
1228    * @return the configuration that should be associated to {@code toTarget}
1229    */
1230   public Iterable<BuildConfiguration> evaluateTransition(final Rule fromRule,
1231       final Attribute attribute, final Target toTarget) {
1232     // Fantastic configurations and where to find them:
1233 
1234     // I. Input files and package groups have no configurations. We don't want to duplicate them.
1235     if (toTarget instanceof InputFile || toTarget instanceof PackageGroup) {
1236       return NULL_LIST;
1237     }
1238 
1239     // II. Host configurations never switch to another. All prerequisites of host targets have the
1240     // same host configuration.
1241     if (isHostConfiguration()) {
1242       return ImmutableList.of(this);
1243     }
1244 
1245     // Make sure config_setting dependencies are resolved in the referencing rule's configuration,
1246     // unconditionally. For example, given:
1247     //
1248     // genrule(
1249     //     name = 'myrule',
1250     //     tools = select({ '//a:condition': [':sometool'] })
1251     //
1252     // all labels in "tools" get resolved in the host configuration (since the "tools" attribute
1253     // declares a host configuration transition). We want to explicitly exclude configuration labels
1254     // from these transitions, since their *purpose* is to do computation on the owning
1255     // rule's configuration.
1256     // TODO(bazel-team): implement this more elegantly. This is far too hackish. Specifically:
1257     // don't reference the rule name explicitly and don't require special-casing here.
1258     if (toTarget instanceof Rule && ((Rule) toTarget).getRuleClass().equals("config_setting")) {
1259       return ImmutableList.of(this);
1260     }
1261 
1262     List<BuildConfiguration> toConfigurations;
1263     if (attribute.getConfigurationTransition() instanceof SplitTransition) {
1264       Preconditions.checkState(attribute.getConfigurator() == null);
1265       toConfigurations = getSplitConfigurations(
1266           (SplitTransition<?>) attribute.getConfigurationTransition());
1267     } else {
1268       // III. Attributes determine configurations. The configuration of a prerequisite is determined
1269       // by the attribute.
1270       @SuppressWarnings("unchecked")
1271       Configurator<BuildConfiguration, Rule> configurator =
1272           (Configurator<BuildConfiguration, Rule>) attribute.getConfigurator();
1273       toConfigurations = ImmutableList.of((configurator != null)
1274           ? configurator.apply(fromRule, this, attribute, toTarget)
1275           : getConfiguration(attribute.getConfigurationTransition()));
1276     }
1277 
1278     return Iterables.transform(toConfigurations,
1279         new Function<BuildConfiguration, BuildConfiguration>() {
1280       @Override
1281       public BuildConfiguration apply(BuildConfiguration input) {
1282         // IV. Allow the transition object to perform an arbitrary switch. Blaze modules can inject
1283         // configuration transition logic by extending the Transitions class.
1284         BuildConfiguration actual = getTransitions().configurationHook(
1285             fromRule, attribute, toTarget, input);
1286 
1287         // V. Allow rule classes to override their own configurations.
1288         Rule associatedRule = toTarget.getAssociatedRule();
1289         if (associatedRule != null) {
1290           @SuppressWarnings("unchecked")
1291           RuleClass.Configurator<BuildConfiguration, Rule> func =
1292               associatedRule.getRuleClassObject().<BuildConfiguration, Rule>getConfigurator();
1293           actual = func.apply(associatedRule, actual);
1294         }
1295 
1296         return actual;
1297       }
1298     });
1299   }
