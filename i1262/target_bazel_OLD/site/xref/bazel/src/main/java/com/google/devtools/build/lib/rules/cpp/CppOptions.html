60    public static class GccWarnConverter implements Converter<String> {
61      @Override
62      public String convert(String input) throws OptionsParsingException {
63        if (input.startsWith("no-") || input.startsWith("-W")) {
64          throw new OptionsParsingException("Not a valid gcc warning to enable");
65        }
66        return input;
67      }
68  
69      @Override
70      public String getTypeDescription() {
71        return "A gcc warning to enable";
72      }
73    }

75    /**
76     * Converts a comma-separated list of compilation mode settings to a properly typed List.
77     */
78    public static class FissionOptionConverter implements Converter<List<CompilationMode>> {
79      @Override
80      public List<CompilationMode> convert(String input) throws OptionsParsingException {
81        ImmutableSet.Builder<CompilationMode> modes = ImmutableSet.builder();
82        if (input.equals("yes")) { // Special case: enable all modes.
83          modes.add(CompilationMode.values());
84        } else if (!input.equals("no")) { // "no" is another special case that disables all modes.
85          CompilationMode.Converter modeConverter = new CompilationMode.Converter();
86          for (String mode : Splitter.on(',').split(input)) {
87            modes.add(modeConverter.convert(mode));
88          }
89        }
90        return modes.build().asList();
91      }
92  
93      @Override
94      public String getTypeDescription() {
95        return "a set of compilation modes";
96      }
97    }

124   /**
125    * Converts a String, which is an absolute path or label into a LibcTop
126    * object.
127    */
128   public static class LibcTopConverter implements Converter<LibcTop> {
129     @Override
130     public LibcTop convert(String input) throws OptionsParsingException {
131       if (!input.startsWith("//")) {
132         throw new OptionsParsingException("Not a label");
133       }
134       try {
135         Label label = Label.parseAbsolute(input).getRelative(LIBC_RELATIVE_LABEL);
136         return new LibcTop(label);
137       } catch (SyntaxException e) {
138         throw new OptionsParsingException(e.getMessage());
139       }
140     }
141 
142     @Override
143     public String getTypeDescription() {
144       return "a label";
145     }
146   }

505   @Override
506   public FragmentOptions getHost(boolean fallback) {
507     CppOptions host = (CppOptions) getDefault();
508 
509     host.commandLineDefinedVariables = commandLineDefinedVariables;
510 
511     // The crosstool options are partially copied from the target configuration.
512     if (!fallback) {
513       if (hostCrosstoolTop == null) {
514         host.cppCompiler = cppCompiler;
515         host.crosstoolTop = crosstoolTop;
516         host.glibc = glibc;
517       } else {
518         host.crosstoolTop = hostCrosstoolTop;
519       }
520     }
521 
522     if (hostLibcTop != null) {
523       host.libcTop = hostLibcTop;
524     } else if (hostCrosstoolTop == null) {
525       // Track libc in the host configuration if no host crosstool is set.
526       host.libcTop = libcTop;
527     }
528 
529     // -g0 is the default, but allowMultiple options cannot have default values so we just pass
530     // -g0 first and let the user options override it.
531     host.coptList = ImmutableList.<String>builder().add("-g0").addAll(hostCoptList).build();
532 
533     host.useThinArchives = useThinArchives;
534     host.useStartEndLib = useStartEndLib;
535     host.extractInclusions = extractInclusions;
536     host.stripBinaries = StripMode.ALWAYS;
537     host.fdoOptimize = null;
538     host.lipoMode = LipoMode.OFF;
539     host.scanIncludes = scanIncludes;
540     host.inmemoryDotdFiles = inmemoryDotdFiles;
541 
542     return host;
543   }

545   @Override
546   public void addAllLabels(Multimap<String, Label> labelMap) {
547     labelMap.put("crosstool", crosstoolTop);
548     if (hostCrosstoolTop != null) {
549       labelMap.put("crosstool", hostCrosstoolTop);
550     }
551 
552     if (libcTop != null) {
553       Label libcLabel = libcTop.getLabel();
554       if (libcLabel != null) {
555         labelMap.put("crosstool", libcLabel);
556       }
557     }
558     addOptionalLabel(labelMap, "fdo", fdoOptimize);
559 
560     if (stl != null) {
561       labelMap.put("STL", stl);
562     }
563 
564     if (customMalloc != null) {
565       labelMap.put("custom_malloc", customMalloc);
566     }
567 
568     if (getLipoContextLabel() != null) {
569       labelMap.put("lipo", getLipoContextLabel());
570     }
571   }
572 
573   @Override
574   public Map<String, Set<Label>> getDefaultsLabels(BuildConfiguration.Options commonOptions) {
575     Set<Label> crosstoolLabels = new LinkedHashSet<>();
576     crosstoolLabels.add(crosstoolTop);
577     if (hostCrosstoolTop != null) {
578       crosstoolLabels.add(hostCrosstoolTop);
579     }
580 
581     if (libcTop != null) {
582       Label libcLabel = libcTop.getLabel();
583       if (libcLabel != null) {
584         crosstoolLabels.add(libcLabel);
585       }
586     }
587 
588     return ImmutableMap.of(
589         "CROSSTOOL", crosstoolLabels,
590         "COVERAGE", ImmutableSet.<Label>of());
591   }

593   public boolean isFdo() {
594     return fdoOptimize != null || fdoInstrument != null;
595   }
596 
597   public boolean isLipoOptimization() {
598     return lipoMode == LipoMode.BINARY && fdoOptimize != null && lipoContext != null;
599   }
600 
601   public boolean isLipoOptimizationOrInstrumentation() {
602     return lipoMode == LipoMode.BINARY &&
603         ((fdoOptimize != null && lipoContext != null) || fdoInstrument != null);
604   }
605 
606   public Label getLipoContextLabel() {
607     return (lipoMode == LipoMode.BINARY && fdoOptimize != null)
608         ? lipoContext : null;
609   }
610 
611   public LipoMode getLipoMode() {
612     return lipoMode;
