14  package com.google.devtools.build.lib.actions;
15  
16  import com.google.common.base.Preconditions;
17  import com.google.common.collect.Iterables;
18  import com.google.devtools.build.lib.actions.Action.MiddlemanType;
19  import com.google.devtools.build.lib.concurrent.ThreadSafety.ThreadSafe;
20  import com.google.devtools.build.lib.syntax.Label;
21  import com.google.devtools.build.lib.util.Pair;
22  import com.google.devtools.build.lib.vfs.PathFragment;
23  
24  import java.util.Iterator;
25  
26  /**
27   * A factory to create middleman objects.
28   */
29  @ThreadSafe
30  public final class MiddlemanFactory {
31  
32    private final ArtifactFactory artifactFactory;
33    private final ActionRegistry actionRegistry;
34  
35    public MiddlemanFactory(
36        ArtifactFactory artifactFactory, ActionRegistry actionRegistry) {
37      this.artifactFactory = Preconditions.checkNotNull(artifactFactory);
38      this.actionRegistry = Preconditions.checkNotNull(actionRegistry);
39    }
40  
41    /**
42     * Creates a {@link MiddlemanType#AGGREGATING_MIDDLEMAN aggregating} middleman.
43     *
44     * @param owner the owner of the action that will be created; must not be null
45     * @param purpose the purpose for which this middleman is created. This should be a string which
46     *        is suitable for use as a filename. A single rule may have many middlemen with distinct
47     *        purposes.
48     * @param inputs the set of artifacts for which the created artifact is to be the middleman.
49     * @param middlemanDir the directory in which to place the middleman.
50     * @return null iff {@code inputs} is empty; the single element of {@code inputs} if there's only
51     *         one; a new aggregating middleman for the {@code inputs} otherwise
52     */
53    public Artifact createAggregatingMiddleman(
54        ActionOwner owner, String purpose, Iterable<Artifact> inputs, Root middlemanDir) {
55      if (hasExactlyOneInput(inputs)) { // Optimization: No middleman for just one input.
56        return Iterables.getOnlyElement(inputs);
57      }
58      Pair<Artifact, Action> result = createMiddleman(
59          owner, Label.print(owner.getLabel()), purpose, inputs, middlemanDir,
60          MiddlemanType.AGGREGATING_MIDDLEMAN);
61      return result == null ? null : result.getFirst();
62    }
63  
64    /**
65     * Returns <code>null</code> iff inputs is empty. Returns the sole element
66     * of inputs iff <code>inputs.size()==1</code>. Otherwise, returns a
67     * middleman artifact and creates a middleman action that generates that
68     * artifact.
69     *
70     * @param owner the owner of the action that will be created.
71     * @param owningArtifact the artifact of the file for which the runfiles
72     *        should be created. There may be at most one set of runfiles for
73     *        an owning artifact, unless the owning artifact is null. There
74     *        may be at most one set of runfiles per owner with a null
75     *        owning artifact.
76     *        Further, if the owning Artifact is non-null, the owning Artifacts'
77     *        root-relative path must be unique and the artifact must be part
78     *        of the runfiles tree for which this middleman is created. Usually
79     *        this artifact will be an executable program.
80     * @param inputs the set of artifacts for which the created artifact is to be
81     *        the middleman.
82     * @param middlemanDir the directory in which to place the middleman.
83     */
84    public Artifact createRunfilesMiddleman(
85        ActionOwner owner, Artifact owningArtifact, Iterable<Artifact> inputs, Root middlemanDir) {
86      if (hasExactlyOneInput(inputs)) { // Optimization: No middleman for just one input.
87        return Iterables.getOnlyElement(inputs);
88      }
89      String middlemanPath = owningArtifact == null
90         ? Label.print(owner.getLabel())
91         : owningArtifact.getRootRelativePath().getPathString();
92      return createMiddleman(owner, middlemanPath, "runfiles", inputs, middlemanDir,
93          MiddlemanType.RUNFILES_MIDDLEMAN).getFirst();
94    }
95  
96    private <T> boolean hasExactlyOneInput(Iterable<T> iterable) {
97      Iterator<T> it = iterable.iterator();
98      if (!it.hasNext()) {
99        return false;
100     }
101     it.next();
102     return !it.hasNext();
103   }
104 
105   /**
106    * Creates a {@link MiddlemanType#ERROR_PROPAGATING_MIDDLEMAN error-propagating} middleman.
107    *
108    * @param owner the owner of the action that will be created. May not be null.
109    * @param middlemanName a unique file name for the middleman artifact in the {@code middlemanDir};
110    *        in practice this is usually the owning rule's label (so it gets escaped as such)
111    * @param purpose the purpose for which this middleman is created. This should be a string which
112    *        is suitable for use as a filename. A single rule may have many middlemen with distinct
113    *        purposes.
114    * @param inputs the set of artifacts for which the created artifact is to be the middleman; must
115    *        not be null or empty
116    * @param middlemanDir the directory in which to place the middleman.
117    * @return a middleman that enforces scheduling order (just like a scheduling middleman) and
118    *         propagates errors, but is ignored by the dependency checker
119    * @throws IllegalArgumentException if {@code inputs} is null or empty
120    */
121   public Artifact createErrorPropagatingMiddleman(ActionOwner owner, String middlemanName,
122       String purpose, Iterable<Artifact> inputs, Root middlemanDir) {
123     Preconditions.checkArgument(inputs != null);
124     Preconditions.checkArgument(!Iterables.isEmpty(inputs));
125     // We must always create this middleman even if there is only one input.
126     return createMiddleman(owner, middlemanName, purpose, inputs, middlemanDir,
127         MiddlemanType.ERROR_PROPAGATING_MIDDLEMAN).getFirst();
128   }
129 
130   /**
131    * Returns the same artifact as {@code createErrorPropagatingMiddleman} would return,
132    * but doesn't create any action.
133    */
134   public Artifact getErrorPropagatingMiddlemanArtifact(String middlemanName, String purpose,
135       Root middlemanDir) {
136     return getStampFileArtifact(middlemanName, purpose, middlemanDir);
137   }
138 
139   /**
140    * Creates both normal and scheduling middlemen.
141    *
142    * <p>Note: there's no need to synchronize this method; the only use of a field is via a call to
143    * another synchronized method (getArtifact()).
144    *
145    * @return null iff {@code inputs} is null or empty; the middleman file and the middleman action
146    *         otherwise
147    */
148   private Pair<Artifact, Action> createMiddleman(
149       ActionOwner owner, String middlemanName, String purpose, Iterable<Artifact> inputs,
150       Root middlemanDir, MiddlemanType middlemanType) {
151     if (inputs == null || Iterables.isEmpty(inputs)) {
152       return null;
153     }
154 
155     Artifact stampFile = getStampFileArtifact(middlemanName, purpose, middlemanDir);
156     Action action = new MiddlemanAction(owner, inputs, stampFile, purpose, middlemanType);
157     actionRegistry.registerAction(action);
158     return Pair.of(stampFile, action);
159   }
160 
161   /**
162    * Creates a normal middleman.
163    *
164    * <p>If called multiple times, it always returns the same object depending on the {@code
165    * purpose}. It does not check that the list of inputs is identical. In contrast to other
166    * middleman methods, this one also returns an object if the list of inputs is empty.
167    *
168    * <p>Note: there's no need to synchronize this method; the only use of a field is via a call to
169    * another synchronized method (getArtifact()).
170    */
171   public Artifact createMiddlemanAllowMultiple(ActionRegistry registry,
172       ActionOwner owner, String purpose, Iterable<Artifact> inputs, Root middlemanDir) {
173     PathFragment stampName = new PathFragment("_middlemen/" + purpose);
174     Artifact stampFile = artifactFactory.getDerivedArtifact(stampName, middlemanDir,
175         actionRegistry.getOwner());
176     MiddlemanAction.create(
177         registry, owner, inputs, stampFile, purpose, MiddlemanType.AGGREGATING_MIDDLEMAN);
178     return stampFile;
179   }
180 
181   private Artifact getStampFileArtifact(String middlemanName, String purpose, Root middlemanDir) {
182     String escapedFilename = Actions.escapedPath(middlemanName);
183     PathFragment stampName = new PathFragment("_middlemen/" + escapedFilename + "-" + purpose);
184     Artifact stampFile = artifactFactory.getDerivedArtifact(stampName, middlemanDir,
185         actionRegistry.getOwner());
186     return stampFile;
187   }
188 }
