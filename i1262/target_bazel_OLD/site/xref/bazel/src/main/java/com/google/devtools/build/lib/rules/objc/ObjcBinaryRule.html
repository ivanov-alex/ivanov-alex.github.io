29  /**
30   * Rule definition for objc_binary.
31   */
32  // TODO(bazel-team): Remove bundling functionality (dependency on ApplicationRule, IPA output).
33  public class ObjcBinaryRule implements RuleDefinition {
34  
35    @Override
36    public RuleClass build(Builder builder, RuleDefinitionEnvironment env) {
37      return builder
38          /*<!-- #BLAZE_RULE(objc_binary).IMPLICIT_OUTPUTS -->
39          <ul>
40           <li><code><var>name</var>.ipa</code>: the application bundle as an <code>.ipa</code>
41               file</li>
42           <li><code><var>name</var>.xcodeproj/project.pbxproj</code>: An Xcode project file which
43               can be used to develop or build on a Mac.</li>
44          </ul>
45          <!-- #END_BLAZE_RULE.IMPLICIT_OUTPUTS -->*/
46          .setImplicitOutputsFunction(
47              ImplicitOutputsFunction.fromFunctions(ReleaseBundlingSupport.IPA, XcodeSupport.PBXPROJ))
48          // TODO(bazel-team): Remove these when this rule no longer produces a bundle.
49          .add(attr("$runner_script_template", LABEL).cfg(HOST)
50              .value(env.getLabel("//tools/objc:ios_runner.sh.mac_template")))
51          .add(attr("$is_executable", BOOLEAN).value(true)
52              .nonconfigurable("Called from RunCommand.isExecutable, which takes a Target"))
53          .build();
54    }
55  
56    @Override
57    public Metadata getMetadata() {
58      return RuleDefinition.Metadata.builder()
59          .name("objc_binary")
60          .factoryClass(ObjcBinary.class)
61          .ancestors(BaseRuleClasses.BaseRule.class, ObjcRuleClasses.LinkingRule.class,
62              ObjcRuleClasses.XcodegenRule.class, ObjcRuleClasses.ReleaseBundlingRule.class,
63              ObjcRuleClasses.SimulatorRule.class)
64          .build();
65    }
66  }
