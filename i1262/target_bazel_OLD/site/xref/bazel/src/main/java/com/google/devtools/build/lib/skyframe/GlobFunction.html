37  /**
38   * A {@link SkyFunction} for {@link GlobValue}s.
39   *
40   * <p>This code drives the glob matching process.
41   */
42  public final class GlobFunction implements SkyFunction {
43  
44    private final Cache<String, Pattern> regexPatternCache =
45        CacheBuilder.newBuilder().concurrencyLevel(4).build();
46  
47    @Override
48    public SkyValue compute(SkyKey skyKey, Environment env) throws GlobFunctionException {
49      GlobDescriptor glob = (GlobDescriptor) skyKey.argument();
50  
51      PackageLookupValue globPkgLookupValue = (PackageLookupValue)
52          env.getValue(PackageLookupValue.key(glob.getPackageId()));
53      if (globPkgLookupValue == null) {
54        return null;
55      }
56      Preconditions.checkState(globPkgLookupValue.packageExists(), "%s isn't an existing package",
57          glob.getPackageId());
58      // Note that this implies that the package's BUILD file exists which implies that the
59      // package's directory exists.
60  
61      PathFragment globSubdir = glob.getSubdir();
62      if (!globSubdir.equals(PathFragment.EMPTY_FRAGMENT)) {
63        PackageLookupValue globSubdirPkgLookupValue = (PackageLookupValue) env.getValue(
64            PackageLookupValue.key(new PackageIdentifier(
65                glob.getPackageId().getRepository(),
66                glob.getPackageId().getPackageFragment().getRelative(globSubdir))));
67        if (globSubdirPkgLookupValue == null) {
68          return null;
69        }
70        if (globSubdirPkgLookupValue.packageExists()) {
71          // We crossed the package boundary, that is, pkg/subdir contains a BUILD file and thus
72          // defines another package, so glob expansion should not descend into that subdir.
73          return GlobValue.EMPTY;
74        }
75      }
76  
77      String pattern = glob.getPattern();
78      // Split off the first path component of the pattern.
79      int slashPos = pattern.indexOf('/');
80      String patternHead;
81      String patternTail;
82      if (slashPos == -1) {
83        patternHead = pattern;
84        patternTail = null;
85      } else {
86        // Substrings will share the backing array of the original glob string. That should be fine.
87        patternHead = pattern.substring(0, slashPos);
88        patternTail = pattern.substring(slashPos + 1);
89      }
90  
91      NestedSetBuilder<PathFragment> matches = NestedSetBuilder.stableOrder();
92  
93      // "**" also matches an empty segment, so try the case where it is not present.
94      if ("**".equals(patternHead)) {
95        if (patternTail == null) {
96          if (!glob.excludeDirs()) {
97            matches.add(globSubdir);
98          }
99        } else {
100         SkyKey globKey = GlobValue.internalKey(
101             glob.getPackageId(), globSubdir, patternTail, glob.excludeDirs());
102         GlobValue globValue = (GlobValue) env.getValue(globKey);
103         if (globValue == null) {
104           return null;
105         }
106         matches.addTransitive(globValue.getMatches());
107       }
108     }
109 
110     PathFragment dirPathFragment = glob.getPackageId().getPackageFragment().getRelative(globSubdir);
111     RootedPath dirRootedPath = RootedPath.toRootedPath(globPkgLookupValue.getRoot(),
112         dirPathFragment);
113     if (containsGlobs(patternHead)) {
114       // Pattern contains globs, so a directory listing is required.
115       //
116       // Note that we have good reason to believe the directory exists: if this is the
117       // top-level directory of the package, the package's existence implies the directory's
118       // existence; if this is a lower-level directory in the package, then we got here from
119       // previous directory listings. Filesystem operations concurrent with build could mean the
120       // directory no longer exists, but DirectoryListingFunction handles that gracefully.
121       DirectoryListingValue listingValue = (DirectoryListingValue)
122           env.getValue(DirectoryListingValue.key(dirRootedPath));
123       if (listingValue == null) {
124         return null;
125       }
126 
127       for (Dirent dirent : listingValue.getDirents()) {
128         Type direntType = dirent.getType();
129         String fileName = dirent.getName();
130 
131         boolean isDirectory = (direntType == Dirent.Type.DIRECTORY);
132 
133         if (!UnixGlob.matches(patternHead, fileName, regexPatternCache)) {
134           continue;
135         }
136 
137         if (direntType == Dirent.Type.SYMLINK) {
138           // TODO(bazel-team): Consider extracting the symlink resolution logic.
139           // For symlinks, look up the corresponding FileValue. This ensures that if the symlink
140           // changes and "switches types" (say, from a file to a directory), this value will be
141           // invalidated.
142           RootedPath symlinkRootedPath = RootedPath.toRootedPath(globPkgLookupValue.getRoot(),
143               dirPathFragment.getRelative(fileName));
144           FileValue symlinkFileValue = (FileValue) env.getValue(FileValue.key(symlinkRootedPath));
145           if (symlinkFileValue == null) {
146             continue;
147           }
148           if (!symlinkFileValue.isSymlink()) {
149             throw new GlobFunctionException(new InconsistentFilesystemException(
150                 "readdir and stat disagree about whether " + symlinkRootedPath.asPath()
151                     + " is a symlink."), Transience.TRANSIENT);
152           }
153           isDirectory = symlinkFileValue.isDirectory();
154         }
155 
156         String subdirPattern = "**".equals(patternHead) ? glob.getPattern() : patternTail;
157         addFile(fileName, glob, subdirPattern, patternTail == null, isDirectory,
158             matches, env);
159       }
160     } else {
161       // Pattern does not contain globs, so a direct stat is enough.
162       String fileName = patternHead;
163       RootedPath fileRootedPath = RootedPath.toRootedPath(globPkgLookupValue.getRoot(),
164           dirPathFragment.getRelative(fileName));
165       FileValue fileValue = (FileValue) env.getValue(FileValue.key(fileRootedPath));
166       if (fileValue == null) {
167         return null;
168       }
169       if (fileValue.exists()) {
170         addFile(fileName, glob, patternTail, patternTail == null,
171             fileValue.isDirectory(), matches, env);
172       }
173     }
174 
175     if (env.valuesMissing()) {
176       return null;
177     }
178 
179     NestedSet<PathFragment> matchesBuilt = matches.build();
180     // Use the same value to represent that we did not match anything.
181     if (matchesBuilt.isEmpty()) {
182       return GlobValue.EMPTY;
183     }
184     return new GlobValue(matchesBuilt);
185   }
186 
187   /**
188    * Returns true if the given pattern contains globs.
189    */
190   private boolean containsGlobs(String pattern) {
191     return pattern.contains("*") || pattern.contains("?");
192   }
193 
194   /**
195    * Includes the given file/directory in the glob.
196    *
197    * <p>{@code fileName} must exist.
198    *
199    * <p>{@code isDirectory} must be true iff the file is a directory.
200    *
201    * <p>{@code directResult} must be set if the file should be included in the result set
202    * directly rather than recursed into if it is a directory.
203    */
204   private void addFile(String fileName, GlobDescriptor glob, String subdirPattern,
205       boolean directResult, boolean isDirectory, NestedSetBuilder<PathFragment> matches,
206       Environment env) {
207     if (isDirectory && subdirPattern != null) {
208       // This is a directory, and the pattern covers files under that directory.
209       SkyKey subdirGlobKey = GlobValue.internalKey(glob.getPackageId(),
210           glob.getSubdir().getRelative(fileName), subdirPattern, glob.excludeDirs());
211       GlobValue subdirGlob = (GlobValue) env.getValue(subdirGlobKey);
212       if (subdirGlob == null) {
213         return;
214       }
215       matches.addTransitive(subdirGlob.getMatches());
216     }
217 
218     if (directResult && !(isDirectory && glob.excludeDirs())) {
219       if (isDirectory) {
220         // TODO(bazel-team): Refactor. This is basically inlined code from the next recursion level.
221         // Ensure that subdirectories that contain other packages are not picked up.
222         PathFragment directory = glob.getPackageId().getPackageFragment()
223             .getRelative(glob.getSubdir()).getRelative(fileName);
224         PackageLookupValue pkgLookupValue = (PackageLookupValue)
225             env.getValue(PackageLookupValue.key(new PackageIdentifier(
226                 glob.getPackageId().getRepository(), directory)));
227         if (pkgLookupValue == null) {
228           return;
229         }
230         if (pkgLookupValue.packageExists()) {
231           // The file is a directory and contains another package.
232           return;
233         }
234       }
235       matches.add(glob.getSubdir().getRelative(fileName));
236     }
237   }

239   @Nullable
240   @Override
241   public String extractTag(SkyKey skyKey) {
242     return null;
243   }

245   /**
246    * Used to declare all the exception types that can be wrapped in the exception thrown by
247    * {@link GlobFunction#compute}.
248    */
249   private static final class GlobFunctionException extends SkyFunctionException {
250     public GlobFunctionException(InconsistentFilesystemException e, Transience transience) {
251       super(e, transience);
252     }
253   }
