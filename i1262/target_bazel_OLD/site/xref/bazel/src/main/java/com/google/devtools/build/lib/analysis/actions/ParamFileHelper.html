33  /**
34   * A command-line implementation that wraps another command line and puts the arguments in a
35   * parameter file if necessary
36   *
37   * <p>The Linux kernel has a limit for the command line length, and that can be easily reached
38   * if, for example, a command is listing all its inputs on the command line.
39   */
40  @Immutable
41  public final class ParamFileHelper {
42  
43    /**
44     * Returns a params file artifact or null for a given command description.
45     *
46     *  <p>Returns null if parameter files are not to be used according to paramFileInfo, or if the
47     * command line is short enough that a parameter file is not needed.
48     *
49     * <p>Make sure to add the returned artifact (if not null) as an input of the corresponding
50     * action.
51     *
52     * @param executableArgs leading arguments that should never be wrapped in a parameter file
53     * @param arguments arguments to the command (in addition to executableArgs), OR
54     * @param commandLine a {@link CommandLine} that provides the arguments (in addition to
55     *        executableArgs)
56     * @param paramFileInfo parameter file information
57     * @param configuration the configuration
58     * @param analysisEnvironment the analysis environment
59     * @param outputs outputs of the action (used to construct a filename for the params file)
60     */
61    public static Artifact getParamsFile(
62        List<String> executableArgs,
63        @Nullable Iterable<String> arguments,
64        @Nullable CommandLine commandLine,
65        @Nullable ParamFileInfo paramFileInfo,
66        BuildConfiguration configuration,
67        AnalysisEnvironment analysisEnvironment,
68        Iterable<Artifact> outputs) {
69      if (paramFileInfo == null ||
70          getParamFileSize(executableArgs, arguments, commandLine)
71              < configuration.getMinParamFileSize()) {
72        return null;
73      }
74  
75      PathFragment paramFilePath = ParameterFile.derivePath(
76          Iterables.getFirst(outputs, null).getRootRelativePath());
77      return analysisEnvironment.getDerivedArtifact(paramFilePath, configuration.getBinDirectory());
78    }
79  
80    /**
81     * Creates a command line using an external params file.
82     *
83     * <p>Call this with the result of {@link #getParamsFile} if it is not null.
84     *
85     * @param executableArgs leading arguments that should never be wrapped in a parameter file
86     * @param arguments arguments to the command (in addition to executableArgs), OR
87     * @param commandLine a {@link CommandLine} that provides the arguments (in addition to
88     *        executableArgs)
89     * @param isShellCommand true if this is a shell command
90     * @param owner owner of the action
91     * @param paramFileInfo parameter file information
92     */
93    public static CommandLine createWithParamsFile(
94        List<String> executableArgs,
95        @Nullable Iterable<String> arguments,
96        @Nullable CommandLine commandLine,
97        boolean isShellCommand,
98        ActionOwner owner,
99        List<Action> requiredActions,
100       ParamFileInfo paramFileInfo,
101       Artifact parameterFile) {
102     Preconditions.checkNotNull(parameterFile);
103     if (commandLine != null && arguments != null && !Iterables.isEmpty(arguments)) {
104       throw new IllegalStateException("must provide either commandLine or arguments: " + arguments);
105     }
106 
107     CommandLine paramFileContents =
108         (commandLine != null) ? commandLine : CommandLine.ofInternal(arguments, false);
109     requiredActions.add(new ParameterFileWriteAction(owner, parameterFile, paramFileContents,
110         paramFileInfo.getFileType(), paramFileInfo.getCharset()));
111 
112     String pathWithFlag = paramFileInfo.getFlag() + parameterFile.getExecPathString();
113     Iterable<String> commandArgv = Iterables.concat(executableArgs, ImmutableList.of(pathWithFlag));
114     return CommandLine.ofInternal(commandArgv, isShellCommand);
115   }
116 
117   /**
118    * Creates a command line without using a params file.
119    *
120    * <p>Call this if {@link #getParamsFile} returns null.
121    *
122    * @param executableArgs leading arguments that should never be wrapped in a parameter file
123    * @param arguments arguments to the command (in addition to executableArgs), OR
124    * @param commandLine a {@link CommandLine} that provides the arguments (in addition to
125    *        executableArgs)
126    * @param isShellCommand true if this is a shell command
127    */
128   public static CommandLine createWithoutParamsFile(List<String> executableArgs,
129       Iterable<String> arguments, CommandLine commandLine, boolean isShellCommand) {
130     if (commandLine == null) {
131       Iterable<String> commandArgv = Iterables.concat(executableArgs, arguments);
132       return CommandLine.ofInternal(commandArgv, isShellCommand);
133     }
134 
135     if (executableArgs.isEmpty()) {
136       return commandLine;
137     }
138 
139     return CommandLine.ofMixed(ImmutableList.copyOf(executableArgs), commandLine, isShellCommand);
140   }
141 
142   /**
143    * Estimates the params file size for the given arguments.
144    */
145   private static int getParamFileSize(
146       List<String> executableArgs, Iterable<String> arguments, CommandLine commandLine) {
147     Iterable<String> actualArguments = (commandLine != null) ? commandLine.arguments() : arguments;
148     return getParamFileSize(executableArgs) + getParamFileSize(actualArguments);
149   }
150 
151   private static int getParamFileSize(Iterable<String> args) {
152     int size = 0;
153     for (String s : args) {
154       size += s.length() + 1;
155     }
156     return size;
157   }
158 }
