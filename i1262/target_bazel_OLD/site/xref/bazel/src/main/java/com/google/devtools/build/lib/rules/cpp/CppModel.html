48  /**
49   * Representation of a C/C++ compilation. Its purpose is to share the code that creates compilation
50   * actions between all classes that need to do so. It follows the builder pattern - load up the
51   * necessary settings and then call {@link #createCcCompileActions}.
52   *
53   * <p>This class is not thread-safe, and it should only be used once for each set of source files,
54   * i.e. calling {@link #createCcCompileActions} will throw an Exception if called twice.
55   */
56  public final class CppModel {
57    private final CppSemantics semantics;
58    private final RuleContext ruleContext;
59    private final BuildConfiguration configuration;
60    private final CppConfiguration cppConfiguration;
61

63    private CppCompilationContext context;
64    private final List<Pair<Artifact, Label>> sourceFiles = new ArrayList<>();
65    private final List<String> copts = new ArrayList<>();
66    private final List<PathFragment> additionalIncludes = new ArrayList<>();
67    @Nullable private Pattern nocopts;
68    private boolean fake;
69    private boolean maySaveTemps;
70    private boolean onlySingleOutput;
71    private CcCompilationOutputs compilationOutputs;

249   /**
250    * @return whether this target needs to generate non-pic actions.
251    */
252   private boolean getGenerateNoPicActions() {
253     return
254         // If we always need pic for everything, then don't bother to create a no-pic action.
255         (!CppHelper.usePic(ruleContext, true) || !CppHelper.usePic(ruleContext, false))
256         // onlySingleOutput guarantees that the code is only ever linked into a dynamic library - so
257         // we don't need a no-pic action even if linking into a binary would require it.
258         && !((onlySingleOutput && getGeneratePicActions()));
259   }

278   /**
279    * Returns a {@code CppCompileActionBuilder} with the common fields for a C++ compile action
280    * being initialized.
281    */
282   private CppCompileActionBuilder initializeCompileAction(Artifact sourceArtifact,
283       Label sourceLabel) {
284     CppCompileActionBuilder builder = createCompileActionBuilder(sourceArtifact, sourceLabel);
285     if (nocopts != null) {
286       builder.addNocopts(nocopts);
287     }
288 
289     builder.setExtraSystemIncludePrefixes(additionalIncludes);
290     builder.setFdoBuildStamp(CppHelper.getFdoBuildStamp(cppConfiguration));
291     builder.setFeatureConfiguration(featureConfiguration);
292     return builder;
293   }

306   /**
307    * Select .pcm inputs to pass on the command line depending on whether we are in pic or non-pic
308    * mode.
309    */
310   private Collection<String> getHeaderModulePaths(CppCompileActionBuilder builder,
311       boolean usePic) {
312     Collection<String> result = new LinkedHashSet<>();
313     NestedSet<Artifact> artifacts = featureConfiguration.isEnabled(
314         CppRuleClasses.HEADER_MODULE_INCLUDES_DEPENDENCIES)
315         ? builder.getContext().getTopLevelHeaderModules()
316         : builder.getContext().getAdditionalInputs();
317     for (Artifact artifact : artifacts) {
318       String filename = artifact.getFilename();
319       if (!filename.endsWith(".pcm")) {
320         continue;
321       }
322       // Depending on whether this specific compile action is pic or non-pic, select the
323       // corresponding header modules. Note that the compilation context might give us both
324       // from targets that are built in both modes.
325       if (usePic == filename.endsWith(".pic.pcm")) {
326         result.add(artifact.getExecPathString());
327       }
328     }
329     return result;
330   }

332   private void setupBuildVariables(CppCompileActionBuilder builder,
333       boolean usePic, PathFragment ccRelativeName) {
334     CcToolchainFeatures.Variables.Builder buildVariables =
335         new CcToolchainFeatures.Variables.Builder();
336 
337     CppModuleMap cppModuleMap = context.getCppModuleMap();
338     if (featureConfiguration.isEnabled(CppRuleClasses.MODULE_MAPS) && cppModuleMap != null) {
339       // If the feature is enabled and cppModuleMap is null, we are about to fail during analysis
340       // in any case, but don't crash.
341       buildVariables.addVariable("module_name", cppModuleMap.getName());
342       buildVariables.addVariable("module_map_file",
343           cppModuleMap.getArtifact().getExecPathString());
344     }
345     if (featureConfiguration.isEnabled(CppRuleClasses.USE_HEADER_MODULES)) {
346       buildVariables.addSequenceVariable("module_files", getHeaderModulePaths(builder, usePic));
347     }
348     if (featureConfiguration.isEnabled(CppRuleClasses.INCLUDE_PATHS)) {
349       buildVariables.addSequenceVariable("include_paths",
350           getSafePathStrings(context.getIncludeDirs()));
351       buildVariables.addSequenceVariable("quote_include_paths",
352           getSafePathStrings(context.getQuoteIncludeDirs()));
353       buildVariables.addSequenceVariable("system_include_paths",
354           getSafePathStrings(context.getSystemIncludeDirs()));
355     }
356 
357     if (ccRelativeName != null) {
358       cppConfiguration.getFdoSupport().configureCompilation(builder, buildVariables, ruleContext,
359           ccRelativeName, usePic, featureConfiguration, cppConfiguration);
360     }
361 
362     CcToolchainFeatures.Variables variables = buildVariables.build();
363     builder.setVariables(variables);
364   }

366   /**
367    * Constructs the C++ compiler actions. It generally creates one action for every specified source
368    * file. It takes into account LIPO, fake-ness, coverage, and PIC, in addition to using the
369    * settings specified on the current object. This method should only be called once.
370    */
371   public CcCompilationOutputs createCcCompileActions() {
372     CcCompilationOutputs.Builder result = new CcCompilationOutputs.Builder();
373     Preconditions.checkNotNull(context);
374     AnalysisEnvironment env = ruleContext.getAnalysisEnvironment();
375     PathFragment objectDir = CppHelper.getObjDirectory(ruleContext.getLabel());
376     
377     if (featureConfiguration.isEnabled(CppRuleClasses.HEADER_MODULES)) {
378       Artifact moduleMapArtifact = context.getCppModuleMap().getArtifact();
379       Label moduleMapLabel = Label.parseAbsoluteUnchecked(context.getCppModuleMap().getName());
380       PathFragment outputName = getObjectOutputPath(moduleMapArtifact, objectDir);
381       CppCompileActionBuilder builder = initializeCompileAction(moduleMapArtifact, moduleMapLabel);
382 
383       // A header module compile action is just like a normal compile action, but:
384       // - the compiled source file is the module map
385       // - it creates a header module (.pcm file).
386       createSourceAction(outputName, result, env, moduleMapArtifact, builder, ".pcm", ".pcm.d",
387           /*addObject=*/false);
388     }
389 
390     for (Pair<Artifact, Label> source : sourceFiles) {
391       Artifact sourceArtifact = source.getFirst();
392       Label sourceLabel = source.getSecond();
393       PathFragment outputName = getObjectOutputPath(sourceArtifact, objectDir);
394       CppCompileActionBuilder builder = initializeCompileAction(sourceArtifact, sourceLabel);
395       
396       if (CppFileTypes.CPP_HEADER.matches(source.first.getExecPath())) {
397         createHeaderAction(outputName, result, env, builder);
398       } else {
399         createSourceAction(outputName, result, env, sourceArtifact, builder, ".o", ".d",
400             /*addObject=*/true);
401       }
402     }
403 
404     compilationOutputs = result.build();
405     return compilationOutputs;
406   }

408   private void createHeaderAction(PathFragment outputName, Builder result, AnalysisEnvironment env,
409       CppCompileActionBuilder builder) {
410     builder
411         .setOutputFile(ruleContext.getRelatedArtifact(outputName, ".h.processed"))
412         .setDotdFile(outputName, ".h.d")
413         // If we generate pic actions, we prefer the header actions to use the pic artifacts.
414         .setPicMode(this.getGeneratePicActions());
415     setupBuildVariables(builder, this.getGeneratePicActions(), null);
416     semantics.finalizeCompileActionBuilder(ruleContext, builder);
417     CppCompileAction compileAction = builder.build();
418     env.registerAction(compileAction);
419     Artifact tokenFile = compileAction.getOutputFile();
420     result.addHeaderTokenFile(tokenFile);
421   }

423   private void createSourceAction(PathFragment outputName,
424       CcCompilationOutputs.Builder result,
425       AnalysisEnvironment env,
426       Artifact sourceArtifact,
427       CppCompileActionBuilder builder,
428       String outputExtension,
429       String dependencyFileExtension,
430       boolean addObject) {
431     PathFragment ccRelativeName = semantics.getEffectiveSourcePath(sourceArtifact);
432     if (cppConfiguration.isLipoOptimization()) {
433       // TODO(bazel-team): we shouldn't be needing this, merging context with the binary
434       // is a superset of necessary information.
435       LipoContextProvider lipoProvider =
436           Preconditions.checkNotNull(CppHelper.getLipoContextProvider(ruleContext), outputName);
437       builder.setContext(CppCompilationContext.mergeForLipo(lipoProvider.getLipoContext(),
438           context));
439     }
440     if (fake) {
441       // For cc_fake_binary, we only create a single fake compile action. It's
442       // not necessary to use -fPIC for negative compilation tests, and using
443       // .pic.o files in cc_fake_binary would break existing uses of
444       // cc_fake_binary.
445       Artifact outputFile = ruleContext.getRelatedArtifact(outputName, outputExtension);
446       PathFragment tempOutputName =
447           FileSystemUtils.replaceExtension(outputFile.getExecPath(), ".temp" + outputExtension);
448       builder
449           .setOutputFile(outputFile)
450           .setDotdFile(outputName, dependencyFileExtension)
451           .setTempOutputFile(tempOutputName);
452       setupBuildVariables(builder, getGeneratePicActions(), ccRelativeName);
453       semantics.finalizeCompileActionBuilder(ruleContext, builder);
454       CppCompileAction action = builder.build();
455       env.registerAction(action);
456       if (addObject) {
457         result.addObjectFile(action.getOutputFile());
458       }
459     } else {
460       boolean generatePicAction = getGeneratePicActions();
461       // If we always need pic for everything, then don't bother to create a no-pic action.
462       boolean generateNoPicAction = getGenerateNoPicActions();
463       Preconditions.checkState(generatePicAction || generateNoPicAction);
464 
465       // Create PIC compile actions (same as non-PIC, but use -fPIC and
466       // generate .pic.o, .pic.d, .pic.gcno instead of .o, .d, .gcno.)
467       if (generatePicAction) {
468         CppCompileActionBuilder picBuilder =
469             copyAsPicBuilder(builder, outputName, outputExtension, dependencyFileExtension);
470         setupBuildVariables(picBuilder, /*usePic=*/true, ccRelativeName);
471 
472         if (maySaveTemps) {
473           result.addTemps(
474               createTempsActions(sourceArtifact, outputName, picBuilder, /*usePic=*/true,
475                 ccRelativeName));
476         }
477 
478         if (isCodeCoverageEnabled()) {
479           picBuilder.setGcnoFile(ruleContext.getRelatedArtifact(outputName, ".pic.gcno"));
480         }
481 
482         semantics.finalizeCompileActionBuilder(ruleContext, picBuilder);
483         CppCompileAction picAction = picBuilder.build();
484         env.registerAction(picAction);
485         if (addObject) {
486           result.addPicObjectFile(picAction.getOutputFile());          
487         }
488         if (picAction.getDwoFile() != null) {
489           // Host targets don't produce .dwo files.
490           result.addPicDwoFile(picAction.getDwoFile());
491         }
492         if (cppConfiguration.isLipoContextCollector() && !generateNoPicAction) {
493           result.addLipoScannable(picAction);
494         }
495       }
496 
497       if (generateNoPicAction) {
498         builder
499             .setOutputFile(ruleContext.getRelatedArtifact(outputName, outputExtension))
500             .setDotdFile(outputName, dependencyFileExtension);
501         // Create non-PIC compile actions
502         setupBuildVariables(builder, /*usePic=*/false, ccRelativeName);
503 
504         if (maySaveTemps) {
505           result.addTemps(
506               createTempsActions(sourceArtifact, outputName, builder, /*usePic=*/false,
507                 ccRelativeName));
508         }
509 
510         if (!cppConfiguration.isLipoOptimization() && isCodeCoverageEnabled()) {
511           builder.setGcnoFile(ruleContext.getRelatedArtifact(outputName, ".gcno"));
512         }
513 
514         semantics.finalizeCompileActionBuilder(ruleContext, builder);
515         CppCompileAction compileAction = builder.build();
516         env.registerAction(compileAction);
517         Artifact objectFile = compileAction.getOutputFile();
518         if (addObject) {
519           result.addObjectFile(objectFile);
520         }
521         if (compileAction.getDwoFile() != null) {
522           // Host targets don't produce .dwo files.
523           result.addDwoFile(compileAction.getDwoFile());
524         }
525         if (cppConfiguration.isLipoContextCollector()) {
526           result.addLipoScannable(compileAction);
527         }
528       }
529     }
530   }

532   /**
533    * Constructs the C++ linker actions. It generally generates two actions, one for a static library
534    * and one for a dynamic library. If PIC is required for shared libraries, but not for binaries,
535    * it additionally creates a third action to generate a PIC static library.
536    *
537    * <p>For dynamic libraries, this method can additionally create an interface shared library that
538    * can be used for linking, but doesn't contain any executable code. This increases the number of
539    * cache hits for link actions. Call {@link #setAllowInterfaceSharedObjects(boolean)} to enable
540    * this behavior.
541    */
542   public CcLinkingOutputs createCcLinkActions(CcCompilationOutputs ccOutputs) {
543     // For now only handle static links. Note that the dynamic library link below ignores linkType.
544     // TODO(bazel-team): Either support non-static links or move this check to setLinkType().
545     Preconditions.checkState(linkType.isStaticLibraryLink(), "can only handle static links");
546 
547     CcLinkingOutputs.Builder result = new CcLinkingOutputs.Builder();
548     if (cppConfiguration.isLipoContextCollector()) {
549       // Don't try to create LIPO link actions in collector mode,
550       // because it needs some data that's not available at this point.
551       return result.build();
552     }
553 
554     AnalysisEnvironment env = ruleContext.getAnalysisEnvironment();
555     boolean usePicForBinaries = CppHelper.usePic(ruleContext, true);
556     boolean usePicForSharedLibs = CppHelper.usePic(ruleContext, false);
557 
558     // Create static library (.a). The linkType only reflects whether the library is alwayslink or
559     // not. The PIC-ness is determined by whether we need to use PIC or not. There are three cases
560     // for (usePicForSharedLibs usePicForBinaries):
561     //
562     // (1) (false false) -> no pic code
563     // (2) (true false)  -> shared libraries as pic, but not binaries
564     // (3) (true true)   -> both shared libraries and binaries as pic
565     //
566     // In case (3), we always need PIC, so only create one static library containing the PIC object
567     // files. The name therefore does not match the content.
568     //
569     // Presumably, it is done this way because the .a file is an implicit output of every cc_library
570     // rule, so we can't use ".pic.a" that in the always-PIC case.
571     PathFragment linkedFileName = CppHelper.getLinkedFilename(ruleContext, linkType);
572     CppLinkAction maybePicAction = newLinkActionBuilder(linkedFileName)
573         .addNonLibraryInputs(ccOutputs.getObjectFiles(usePicForBinaries))
574         .addNonLibraryInputs(ccOutputs.getHeaderTokenFiles())
575         .setLinkType(linkType)
576         .setLinkStaticness(LinkStaticness.FULLY_STATIC)
577         .build();
578     env.registerAction(maybePicAction);
579     result.addStaticLibrary(maybePicAction.getOutputLibrary());
580 
581     // Create a second static library (.pic.a). Only in case (2) do we need both PIC and non-PIC
582     // static libraries. In that case, the first static library contains the non-PIC code, and this
583     // one contains the PIC code, so the names match the content.
584     if (!usePicForBinaries && usePicForSharedLibs) {
585       LinkTargetType picLinkType = (linkType == LinkTargetType.ALWAYS_LINK_STATIC_LIBRARY)
586           ? LinkTargetType.ALWAYS_LINK_PIC_STATIC_LIBRARY
587           : LinkTargetType.PIC_STATIC_LIBRARY;
588 
589       PathFragment picFileName = CppHelper.getLinkedFilename(ruleContext, picLinkType);
590       CppLinkAction picAction = newLinkActionBuilder(picFileName)
591           .addNonLibraryInputs(ccOutputs.getObjectFiles(true))
592           .addNonLibraryInputs(ccOutputs.getHeaderTokenFiles())
593           .setLinkType(picLinkType)
594           .setLinkStaticness(LinkStaticness.FULLY_STATIC)
595           .build();
596       env.registerAction(picAction);
597       result.addPicStaticLibrary(picAction.getOutputLibrary());
598     }
599 
600     if (!createDynamicLibrary) {
601       return result.build();
602     }
603 
604     // Create dynamic library.
605     if (soImplFilename == null) {
606       soImplFilename = CppHelper.getLinkedFilename(ruleContext, LinkTargetType.DYNAMIC_LIBRARY);
607     }
608     List<String> sonameLinkopts = ImmutableList.of();
609     PathFragment soInterfaceFilename = null;
610     if (cppConfiguration.useInterfaceSharedObjects() && allowInterfaceSharedObjects) {
611       soInterfaceFilename =
612           CppHelper.getLinkedFilename(ruleContext, LinkTargetType.INTERFACE_DYNAMIC_LIBRARY);
613       Artifact dynamicLibrary = env.getDerivedArtifact(
614           soImplFilename, configuration.getBinDirectory());
615       sonameLinkopts = ImmutableList.of("-Wl,-soname=" +
616           SolibSymlinkAction.getDynamicLibrarySoname(dynamicLibrary.getRootRelativePath(), false));
617     }
618 
619     // Should we also link in any libraries that this library depends on?
620     // That is required on some systems...
621     CppLinkAction action = newLinkActionBuilder(soImplFilename)
622         .setInterfaceOutputPath(soInterfaceFilename)
623         .addNonLibraryInputs(ccOutputs.getObjectFiles(usePicForSharedLibs))
624         .addNonLibraryInputs(ccOutputs.getHeaderTokenFiles())
625         .setLinkType(LinkTargetType.DYNAMIC_LIBRARY)
626         .setLinkStaticness(LinkStaticness.DYNAMIC)
627         .addLinkopts(linkopts)
628         .addLinkopts(sonameLinkopts)
629         .setRuntimeInputs(
630             CppHelper.getToolchain(ruleContext).getDynamicRuntimeLinkMiddleman(),
631             CppHelper.getToolchain(ruleContext).getDynamicRuntimeLinkInputs())
632         .build();
633     env.registerAction(action);
634 
635     LibraryToLink dynamicLibrary = action.getOutputLibrary();
636     LibraryToLink interfaceLibrary = action.getInterfaceOutputLibrary();
637     if (interfaceLibrary == null) {
638       interfaceLibrary = dynamicLibrary;
639     }
640 
641     // If shared library has neverlink=1, then leave it untouched. Otherwise,
642     // create a mangled symlink for it and from now on reference it through
643     // mangled name only.
644     if (neverLink) {
645       result.addDynamicLibrary(interfaceLibrary);
646       result.addExecutionDynamicLibrary(dynamicLibrary);
647     } else {
648       LibraryToLink libraryLink = SolibSymlinkAction.getDynamicLibrarySymlink(
649           ruleContext, interfaceLibrary.getArtifact(), false, false,
650           ruleContext.getConfiguration());
651       result.addDynamicLibrary(libraryLink);
652       LibraryToLink implLibraryLink = SolibSymlinkAction.getDynamicLibrarySymlink(
653           ruleContext, dynamicLibrary.getArtifact(), false, false,
654           ruleContext.getConfiguration());
655       result.addExecutionDynamicLibrary(implLibraryLink);
656     }
657     return result.build();
658   }

702   /**
703    * Create the actions for "--save_temps".
704    */
705   private ImmutableList<Artifact> createTempsActions(Artifact source, PathFragment outputName,
706       CppCompileActionBuilder builder, boolean usePic, PathFragment ccRelativeName) {
707     if (!cppConfiguration.getSaveTemps()) {
708       return ImmutableList.of();
709     }
710 
711     String path = source.getFilename();
712     boolean isCFile = CppFileTypes.C_SOURCE.matches(path);
713     boolean isCppFile = CppFileTypes.CPP_SOURCE.matches(path);
714 
715     if (!isCFile && !isCppFile) {
716       return ImmutableList.of();
717     }
718 
719     String iExt = isCFile ? ".i" : ".ii";
720     String picExt = usePic ? ".pic" : "";
721     CppCompileActionBuilder dBuilder = new CppCompileActionBuilder(builder);
722     setupBuildVariables(dBuilder, usePic, ccRelativeName);
723     CppCompileActionBuilder sdBuilder = new CppCompileActionBuilder(builder);
724     setupBuildVariables(sdBuilder, usePic, ccRelativeName);
725 
726     dBuilder
727         .setOutputFile(ruleContext.getRelatedArtifact(outputName, picExt + iExt))
728         .setDotdFile(outputName, picExt + iExt + ".d");
729     semantics.finalizeCompileActionBuilder(ruleContext, dBuilder);
730     CppCompileAction dAction = dBuilder.build();
731     ruleContext.registerAction(dAction);
732 
733     sdBuilder
734         .setOutputFile(ruleContext.getRelatedArtifact(outputName, picExt + ".s"))
735         .setDotdFile(outputName, picExt + ".s.d");
736     semantics.finalizeCompileActionBuilder(ruleContext, sdBuilder);
737     CppCompileAction sdAction = sdBuilder.build();
738     ruleContext.registerAction(sdAction);
739     return ImmutableList.of(
740         dAction.getOutputFile(),
741         sdAction.getOutputFile());
742   }

744   /**
745    * Returns true iff code coverage is enabled for the given target.
746    */
747   private boolean isCodeCoverageEnabled() {
748     if (configuration.isCodeCoverageEnabled()) {
749       final RegexFilter filter = configuration.getInstrumentationFilter();
750       // If rule is matched by the instrumentation filter, enable instrumentation
751       if (filter.isIncluded(ruleContext.getLabel().toString())) {
752         return true;
753       }
754       // At this point the rule itself is not matched by the instrumentation filter. However, we
755       // might still want to instrument C++ rules if one of the targets listed in "deps" is
756       // instrumented and, therefore, can supply header files that we would want to collect code
757       // coverage for. For example, think about cc_test rule that tests functionality defined in a
758       // header file that is supplied by the cc_library.
759       //
760       // Note that we only check direct prerequisites and not the transitive closure. This is done
761       // for two reasons:
762       // a) It is a good practice to declare libraries which you directly rely on. Including headers
763       //    from a library hidden deep inside the transitive closure makes build dependencies less
764       //    readable and can lead to unexpected breakage.
765       // b) Traversing the transitive closure for each C++ compile action would require more complex
766       //    implementation (with caching results of this method) to avoid O(N^2) slowdown.
767       if (ruleContext.getRule().isAttrDefined("deps", Type.LABEL_LIST)) {
768         for (TransitiveInfoCollection dep : ruleContext.getPrerequisites("deps", Mode.TARGET)) {
769           if (dep.getProvider(CppCompilationContext.class) != null
770               && filter.isIncluded(dep.getLabel().toString())) {
771             return true;
772           }
773         }
774       }
775     }
776     return false;
777   }
