30  /**
31   * The value returned by {@link PrepareDepsOfPatternFunction}. Because that function is
32   * invoked only for its side effect (i.e. ensuring the graph contains targets matching the
33   * pattern and its transitive dependencies), this value carries no information.
34   *
35   * <p>Because the returned value is always the same object, this value and the
36   * {@link PrepareDepsOfPatternFunction} which computes it are incompatible with change pruning. It
37   * should only be requested by consumers who do not require reevaluation when
38   * {@link PrepareDepsOfPatternFunction} is reevaluated. Safe consumers include, e.g., top-level
39   * consumers, and other functions which invoke {@link PrepareDepsOfPatternFunction} solely for its
40   * side-effects.
41   */
42  public class PrepareDepsOfPatternValue implements SkyValue {
43    public static final PrepareDepsOfPatternValue INSTANCE = new PrepareDepsOfPatternValue();
44  
45    private PrepareDepsOfPatternValue() {
46    }
47  
48    /**
49     * Returns an iterable of {@link PrepareDepsOfPatternSkyKeyOrException}, with {@link
50     * TargetPatternKey} arguments. Negative target patterns of type other than {@link
51     * Type#TARGETS_BELOW_DIRECTORY} are not permitted. If a provided pattern fails to parse or is
52     * negative but not a {@link Type#TARGETS_BELOW_DIRECTORY}, an element in the returned iterable
53     * will throw when its {@link PrepareDepsOfPatternSkyKeyOrException#getSkyKey} method is called
54     * and will return the failing pattern when its {@link
55     * PrepareDepsOfPatternSkyKeyOrException#getOriginalPattern} method is called.
56     *
57     * <p>There may be fewer returned elements than patterns provided as input. This function will
58     * combine negative {@link Type#TARGETS_BELOW_DIRECTORY} patterns with preceding patterns to
59     * return an iterable of SkyKeys that avoids loading excluded directories during evaluation.
60     *
61     * @param patterns The list of patterns, e.g. [//foo/..., -//foo/biz/...]. If a pattern's first
62     *     character is "-", it is treated as a negative pattern.
63     * @param offset The offset to apply to relative target patterns.
64     */
65    @ThreadSafe
66    public static Iterable<PrepareDepsOfPatternSkyKeyOrException> keys(List<String> patterns,
67        String offset) {
68      List<TargetPatternSkyKeyOrException> keysMaybe =
69          ImmutableList.copyOf(TargetPatternValue.keys(patterns, FilteringPolicies.NO_FILTER,
70              offset));
71  
72      // This code path is evaluated only for query universe preloading, and the quadratic cost of
73      // the code below (i.e. for each pattern, consider each later pattern as a candidate for
74      // subdirectory exclusion) is only acceptable because all the use cases for query universe
75      // preloading involve short (<10 items) pattern sequences.
76      ImmutableList.Builder<PrepareDepsOfPatternSkyKeyOrException> builder = ImmutableList.builder();
77      for (int i = 0; i < keysMaybe.size(); i++) {
78        TargetPatternSkyKeyOrException keyMaybe = keysMaybe.get(i);
79        SkyKey skyKey;
80        try {
81          skyKey = keyMaybe.getSkyKey();
82        } catch (TargetParsingException e) {
83          // keyMaybe.getSkyKey() may throw TargetParsingException if its corresponding pattern
84          // failed to parse. If so, wrap the exception and return it, so that our caller can
85          // deal with it.
86          skyKey = null;
87          builder.add(new PrepareDepsOfPatternSkyKeyException(e, keyMaybe.getOriginalPattern()));
88        }
89        if (skyKey != null) {
90          TargetPatternKey targetPatternKey = (TargetPatternKey) skyKey.argument();
91          if (targetPatternKey.isNegative()) {
92            if (!targetPatternKey.getParsedPattern().getType().equals(Type.TARGETS_BELOW_DIRECTORY)) {
93              builder.add(
94                  new PrepareDepsOfPatternSkyKeyException(
95                      new TargetParsingException(
96                          "Negative target patterns of types other than \"targets below directory\""
97                              + " are not permitted."), targetPatternKey.toString()));
98            }
99            // Otherwise it's a negative TBD pattern which was combined with previous patterns as an
100           // excluded directory. These can be skipped because there's no PrepareDepsOfPattern work
101           // to be done for them.
102         } else {
103           builder.add(new PrepareDepsOfPatternSkyKeyValue(setExcludedDirectories(targetPatternKey,
104               excludedDirectoriesBeneath(targetPatternKey, i, keysMaybe))));
105         }
106       }
107     }
108     return builder.build();
109   }
110 
111   private static TargetPatternKey setExcludedDirectories(TargetPatternKey original,
112       ImmutableSet<String> excludedSubdirectories) {
113     return new TargetPatternKey(original.getParsedPattern(), original.getPolicy(),
114         original.isNegative(), original.getOffset(), excludedSubdirectories);
115   }
116 
117   private static ImmutableSet<String> excludedDirectoriesBeneath(TargetPatternKey targetPatternKey,
118       int position, List<TargetPatternSkyKeyOrException> keysMaybe) {
119     ImmutableSet.Builder<String> excludedDirectoriesBuilder = ImmutableSet.builder();
120     for (int j = position + 1; j < keysMaybe.size(); j++) {
121       TargetPatternSkyKeyOrException laterPatternMaybe = keysMaybe.get(j);
122       SkyKey laterSkyKey;
123       try {
124         laterSkyKey = laterPatternMaybe.getSkyKey();
125       } catch (TargetParsingException ignored) {
126         laterSkyKey = null;
127       }
128       if (laterSkyKey != null) {
129         TargetPatternKey laterTargetPatternKey = (TargetPatternKey) laterSkyKey.argument();
130         TargetPattern laterParsedPattern = laterTargetPatternKey.getParsedPattern();
131         if (laterTargetPatternKey.isNegative()
132             && targetPatternKey.getParsedPattern().containsBelowDirectory(laterParsedPattern)) {
133           excludedDirectoriesBuilder.add(laterParsedPattern.getDirectory());
134         }
135       }
136     }
137     return excludedDirectoriesBuilder.build();
138   }

140   /**
141    * Wrapper for a prepare deps of pattern {@link SkyKey} or the {@link TargetParsingException}
142    * thrown when trying to create it.
143    */
144   public interface PrepareDepsOfPatternSkyKeyOrException {
145 
146     /**
147      * Returns the stored {@link SkyKey} or throws {@link TargetParsingException} if one was thrown
148      * when creating the key.
149      */
150     SkyKey getSkyKey() throws TargetParsingException;
151 
152     /**
153      * Returns the pattern that resulted in the stored {@link SkyKey} or {@link
154      * TargetParsingException}.
155      */
156     String getOriginalPattern();
157   }
158 
159 
160   private static class PrepareDepsOfPatternSkyKeyException implements
161       PrepareDepsOfPatternSkyKeyOrException {
162 
163     private final TargetParsingException exception;
164     private final String originalPattern;
165 
166     public PrepareDepsOfPatternSkyKeyException(TargetParsingException exception,
167         String originalPattern) {
168       this.exception = exception;
169       this.originalPattern = originalPattern;
170     }
171 
172     @Override
173     public SkyKey getSkyKey() throws TargetParsingException {
174       throw exception;
175     }
176 
177     @Override
178     public String getOriginalPattern() {
179       return originalPattern;
180     }
181   }
182 
183   private static class PrepareDepsOfPatternSkyKeyValue implements
184       PrepareDepsOfPatternSkyKeyOrException {
185 
186     private final TargetPatternKey targetPatternKey;
187 
188     public PrepareDepsOfPatternSkyKeyValue(TargetPatternKey targetPatternKey) {
189       this.targetPatternKey = targetPatternKey;
190     }
191 
192     @Override
193     public SkyKey getSkyKey() throws TargetParsingException {
194       return new SkyKey(SkyFunctions.PREPARE_DEPS_OF_PATTERN, targetPatternKey);
195     }
196 
197     @Override
198     public String getOriginalPattern() {
199       return targetPatternKey.getPattern();
200     }
201   }
