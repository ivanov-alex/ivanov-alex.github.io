159   private ImmutableSet<String> getEnabledFeatures() {
160     Set<String> globallyEnabled = new HashSet<>();
161     Set<String> globallyDisabled = new HashSet<>();
162     parseFeatures(getConfiguration().getDefaultFeatures(), globallyEnabled, globallyDisabled);
163     Set<String> packageEnabled = new HashSet<>();
164     Set<String> packageDisabled = new HashSet<>();
165     parseFeatures(getRule().getPackage().getFeatures(), packageEnabled, packageDisabled);
166     Set<String> packageFeatures =
167         Sets.difference(Sets.union(globallyEnabled, packageEnabled), packageDisabled);
168     Set<String> ruleFeatures = packageFeatures;
169     if (attributes().has("features", Type.STRING_LIST)) {
170       Set<String> ruleEnabled = new HashSet<>();
171       Set<String> ruleDisabled = new HashSet<>();
172       parseFeatures(attributes().get("features", Type.STRING_LIST), ruleEnabled, ruleDisabled);
173       ruleFeatures = Sets.difference(Sets.union(packageFeatures, ruleEnabled), ruleDisabled);
174     }
175     return ImmutableSortedSet.copyOf(Sets.difference(ruleFeatures, globallyDisabled));
176   }
177 
178   private void parseFeatures(Iterable<String> features, Set<String> enabled, Set<String> disabled) {
179     for (String feature : features) {
180       if (feature.startsWith("-")) {
181         disabled.add(feature.substring(1));
182       } else if (feature.equals("no_layering_check")) {
183         // TODO(bazel-team): Remove once we do not have BUILD files left that contain
184         // 'no_layering_check'.
185         disabled.add(feature.substring(3));
186       } else {
187         enabled.add(feature);
188       }
189     }
190   }

456   /**
457    * Returns the list of transitive info collections that feed into this target through the
458    * specified attribute. Note that you need to specify the correct mode for the attribute,
459    * otherwise an assertion will be raised.
460    */
461   public List<? extends TransitiveInfoCollection> getPrerequisites(String attributeName,
462       Mode mode) {
463     Attribute attributeDefinition = getAttribute(attributeName);
464     if ((mode == Mode.TARGET)
465         && (attributeDefinition.getConfigurationTransition() instanceof SplitTransition)) {
466       // TODO(bazel-team): If you request a split-configured attribute in the target configuration,
467       // we return only the list of configured targets for the first architecture; this is for
468       // backwards compatibility with existing code in cases where the call to getPrerequisites is
469       // deeply nested and we can't easily inject the behavior we want. However, we should fix all
470       // such call sites.
471       checkAttribute(attributeName, Mode.SPLIT);
472       Map<String, ? extends List<? extends TransitiveInfoCollection>> map =
473           getSplitPrerequisites(attributeName, /*requireSplit=*/false);
474       return map.isEmpty()
475           ? ImmutableList.<TransitiveInfoCollection>of()
476           : map.entrySet().iterator().next().getValue();
477     }
478 
479     checkAttribute(attributeName, mode);
480     return targetMap.get(attributeName);
481   }

483   /**
484    * Returns the a prerequisites keyed by the CPU of their configurations; this method throws an
485    * exception if the split transition is not active.
486    */
487   public Map<String, ? extends List<? extends TransitiveInfoCollection>>
488       getSplitPrerequisites(String attributeName) {
489     return getSplitPrerequisites(attributeName, /*requireSplit*/true);
490   }
491 
492   private Map<String, ? extends List<? extends TransitiveInfoCollection>>
493       getSplitPrerequisites(String attributeName, boolean requireSplit) {
494     checkAttribute(attributeName, Mode.SPLIT);
495 
496     Attribute attributeDefinition = getAttribute(attributeName);
497     SplitTransition<?> transition =
498         (SplitTransition<?>) attributeDefinition.getConfigurationTransition();
499     List<BuildConfiguration> configurations =
500         getConfiguration().getTransitions().getSplitConfigurations(transition);
501     if (configurations.size() == 1) {
502       // There are two cases here:
503       // 1. Splitting is enabled, but only one target cpu.
504       // 2. Splitting is disabled, and no --cpu value was provided on the command line.
505       // In the first case, the cpu value is non-null, but in the second case it is null. We only
506       // allow that to proceed if the caller specified that he is going to ignore the cpu value
507       // anyway.
508       String cpu = configurations.get(0).getCpu();
509       if (cpu == null) {
510         Preconditions.checkState(!requireSplit);
511         cpu = "DO_NOT_USE";
512       }
513       return ImmutableMap.of(cpu, targetMap.get(attributeName));
514     }
515 
516     Set<String> cpus = new HashSet<>();
517     for (BuildConfiguration config : configurations) {
518       // This method should only be called when the split config is enabled on the command line, in
519       // which case this cpu can't be null.
520       Preconditions.checkNotNull(config.getCpu());
521       cpus.add(config.getCpu());
522     }
523 
524     // Use an ImmutableListMultimap.Builder here to preserve ordering.
525     ImmutableListMultimap.Builder<String, TransitiveInfoCollection> result =
526         ImmutableListMultimap.builder();
527     for (TransitiveInfoCollection t : targetMap.get(attributeName)) {
528       if (t.getConfiguration() != null) {
529         result.put(t.getConfiguration().getCpu(), t);
530       } else {
531         // Source files don't have a configuration, so we add them to all architecture entries.
532         for (String cpu : cpus) {
533           result.put(cpu, t);
534         }
535       }
536     }
537     return Multimaps.asMap(result.build());
538   }

587   /**
588    * Returns the prerequisite referred to by the specified attribute. Also checks whether
589    * the attribute is marked as executable and that the target referred to can actually be
590    * executed.
591    *
592    * <p>The {@code mode} argument must match the configuration transition specified in the
593    * definition of the attribute.
594    *
595    * @param attributeName the name of the attribute
596    * @param mode the configuration transition of the attribute
597    *
598    * @return the {@link FilesToRunProvider} interface of the prerequisite.
599    */
600   public FilesToRunProvider getExecutablePrerequisite(String attributeName, Mode mode) {
601     Attribute ruleDefinition = getAttribute(attributeName);
602 
603     if (ruleDefinition == null) {
604       throw new IllegalStateException(getRule().getRuleClass() + " attribute " + attributeName
605           + " is not defined");
606     }
607     if (!ruleDefinition.isExecutable()) {
608       throw new IllegalStateException(getRule().getRuleClass() + " attribute " + attributeName
609           + " is not configured to be executable");
610     }
611 
612     TransitiveInfoCollection prerequisite = getPrerequisite(attributeName, mode);
613     if (prerequisite == null) {
614       return null;
615     }
616 
617     FilesToRunProvider result = prerequisite.getProvider(FilesToRunProvider.class);
618     if (result == null || result.getExecutable() == null) {
619       attributeError(
620           attributeName, prerequisite.getLabel() + " does not refer to a valid executable target");
621     }
622     return result;
623   }

636   /**
637    * Gets an attribute of type STRING_LIST expanding Make variables, $(location) tags into the
638    * dependency location (see {@link LocationExpander} for details) and tokenizes the result.
639    *
640    * @param attributeName the name of the attribute to process
641    * @param ruleContext the rule context to look for $(location) tag replacement, or null if
642    *        location should not be expanded
643    * @return a list of strings containing the expanded and tokenized values for the attribute
644    */
645   public List<String> getTokenizedStringListAttr(String attributeName,
646       @Nullable RuleContext ruleContext) {
647     if (!getRule().isAttrDefined(attributeName, Type.STRING_LIST)) {
648       // TODO(bazel-team): This should be an error.
649       return ImmutableList.of();
650     }
651     List<String> original = attributes().get(attributeName, Type.STRING_LIST);
652     if (original.isEmpty()) {
653       return ImmutableList.of();
654     }
655     List<String> tokens = new ArrayList<>();
656     LocationExpander locationExpander =

772   private void checkAttribute(String attributeName, Mode mode) {
773     Attribute attributeDefinition = getAttribute(attributeName);
774     if (attributeDefinition == null) {
775       throw new IllegalStateException(getRule().getLocation() + ": " + getRule().getRuleClass()
776         + " attribute " + attributeName + " is not defined");
777     }
778     if (!(attributeDefinition.getType() == Type.LABEL
779         || attributeDefinition.getType() == Type.LABEL_LIST)) {
780       throw new IllegalStateException(rule.getRuleClass() + " attribute " + attributeName
781         + " is not a label type attribute");
782     }
783     if (mode == Mode.HOST) {
784       if (attributeDefinition.getConfigurationTransition() != ConfigurationTransition.HOST) {
785         throw new IllegalStateException(getRule().getLocation() + ": "
786             + getRule().getRuleClass() + " attribute " + attributeName
787             + " is not configured for the host configuration");
788       }
789     } else if (mode == Mode.TARGET) {
790       if (attributeDefinition.getConfigurationTransition() != ConfigurationTransition.NONE) {
791         throw new IllegalStateException(getRule().getLocation() + ": "
792             + getRule().getRuleClass() + " attribute " + attributeName
793             + " is not configured for the target configuration");
794       }
795     } else if (mode == Mode.DATA) {
796       if (attributeDefinition.getConfigurationTransition() != ConfigurationTransition.DATA) {
797         throw new IllegalStateException(getRule().getLocation() + ": "
798             + getRule().getRuleClass() + " attribute " + attributeName
799             + " is not configured for the data configuration");
800       }
801     } else if (mode == Mode.SPLIT) {
802       if (!(attributeDefinition.getConfigurationTransition() instanceof SplitTransition)) {
803         throw new IllegalStateException(getRule().getLocation() + ": "
804             + getRule().getRuleClass() + " attribute " + attributeName
805             + " is not configured for a split transition");
806       }
807     }
808   }

814   public Mode getAttributeMode(String attributeName) {
815     Attribute attributeDefinition = getAttribute(attributeName);
816     if (attributeDefinition == null) {
817       throw new IllegalStateException(getRule().getLocation() + ": " + getRule().getRuleClass()
818         + " attribute " + attributeName + " is not defined");
819     }
820     if (!(attributeDefinition.getType() == Type.LABEL
821         || attributeDefinition.getType() == Type.LABEL_LIST)) {
822       throw new IllegalStateException(rule.getRuleClass() + " attribute " + attributeName
823         + " is not a label type attribute");
824     }
825     if (attributeDefinition.getConfigurationTransition() == ConfigurationTransition.HOST) {
826       return Mode.HOST;
827     } else if (attributeDefinition.getConfigurationTransition() == ConfigurationTransition.NONE) {
828       return Mode.TARGET;
829     } else if (attributeDefinition.getConfigurationTransition() == ConfigurationTransition.DATA) {
830       return Mode.DATA;
831     } else if (attributeDefinition.getConfigurationTransition() instanceof SplitTransition) {
832       return Mode.SPLIT;
833     }
834     throw new IllegalStateException(getRule().getLocation() + ": "
835         + getRule().getRuleClass() + " attribute " + attributeName + " is not configured");
836   }

1182     private boolean validateFilesetEntry(FilesetEntry filesetEntry, ConfiguredTarget src) {
1183       if (src.getProvider(FilesetProvider.class) != null) {
1184         return true;
1185       }
1186       if (filesetEntry.isSourceFileset()) {
1187         return true;
1188       }
1189 
1190       Target srcTarget = src.getTarget();
1191       if (!(srcTarget instanceof FileTarget)) {
1192         attributeError("entries", String.format(
1193             "Invalid 'srcdir' target '%s'. Must be another Fileset or package",
1194             srcTarget.getLabel()));
1195         return false;
1196       }
1197 
1198       if (srcTarget instanceof OutputFile) {
1199         attributeWarning("entries", String.format("'srcdir' target '%s' is not an input file. "
1200             + "This forces the Fileset to be executed unconditionally",
1201             srcTarget.getLabel()));
1202       }
1203 
1204       return true;
1205     }

1207     /**
1208      * Determines and returns a map from attribute name to list of configured fileset entries, based
1209      * on a PrerequisiteMap instance.
1210      */
1211     private ListMultimap<String, ConfiguredFilesetEntry> createFilesetEntryMap(
1212         final Rule rule, Set<ConfigMatchingProvider> configConditions) {
1213       final ImmutableSortedKeyListMultimap.Builder<String, ConfiguredFilesetEntry> mapBuilder =
1214           ImmutableSortedKeyListMultimap.builder();
1215       for (Attribute attr : rule.getAttributes()) {
1216         if (attr.getType() != Type.FILESET_ENTRY_LIST) {
1217           continue;
1218         }
1219         String attributeName = attr.getName();
1220         Map<Label, ConfiguredTarget> ctMap = new HashMap<>();
1221         for (ConfiguredTarget prerequisite : prerequisiteMap.get(attr)) {
1222           ctMap.put(prerequisite.getLabel(), prerequisite);
1223         }
1224         List<FilesetEntry> entries = ConfiguredAttributeMapper.of(rule, configConditions)
1225             .get(attributeName, Type.FILESET_ENTRY_LIST);
1226         for (FilesetEntry entry : entries) {
1227           if (entry.getFiles() == null) {
1228             Label label = entry.getSrcLabel();
1229             ConfiguredTarget src = ctMap.get(label);
1230             if (!validateFilesetEntry(entry, src)) {
1231               continue;
1232             }
1233 
1234             mapBuilder.put(attributeName, new ConfiguredFilesetEntry(entry, src));
1235           } else {
1236             ImmutableList.Builder<TransitiveInfoCollection> files = ImmutableList.builder();
1237             for (Label file : entry.getFiles()) {
1238               files.add(ctMap.get(file));
1239             }
1240             mapBuilder.put(attributeName, new ConfiguredFilesetEntry(entry, files.build()));
1241           }
1242         }
1243       }
1244       return mapBuilder.build();
1245     }

1247     /**
1248      * Determines and returns a map from attribute name to list of configured targets.
1249      */
1250     private ImmutableSortedKeyListMultimap<String, ConfiguredTarget> createTargetMap() {
1251       ImmutableSortedKeyListMultimap.Builder<String, ConfiguredTarget> mapBuilder =
1252           ImmutableSortedKeyListMultimap.builder();
1253 
1254       for (Map.Entry<Attribute, Collection<ConfiguredTarget>> entry :
1255           prerequisiteMap.asMap().entrySet()) {
1256         Attribute attribute = entry.getKey();
1257         if (attribute == null) {
1258           continue;
1259         }
1260         if (attribute.isSilentRuleClassFilter()) {
1261           Predicate<RuleClass> filter = attribute.getAllowedRuleClassesPredicate();
1262           for (ConfiguredTarget configuredTarget : entry.getValue()) {
1263             Target prerequisiteTarget = configuredTarget.getTarget();
1264             if ((prerequisiteTarget instanceof Rule)
1265                 && filter.apply(((Rule) prerequisiteTarget).getRuleClassObject())) {
1266               validateDirectPrerequisite(attribute, configuredTarget);
1267               mapBuilder.put(attribute.getName(), configuredTarget);
1268             }
1269           }
1270         } else {
1271           for (ConfiguredTarget configuredTarget : entry.getValue()) {
1272             validateDirectPrerequisite(attribute, configuredTarget);
1273             mapBuilder.put(attribute.getName(), configuredTarget);
1274           }
1275         }
1276       }
1277 
1278       // Handle abi_deps+deps error.
1279       Attribute abiDepsAttr = rule.getAttributeDefinition("abi_deps");
1280       if ((abiDepsAttr != null) && rule.isAttributeValueExplicitlySpecified("abi_deps")
1281           && rule.isAttributeValueExplicitlySpecified("deps")) {
1282         attributeError("deps", "Only one of deps and abi_deps should be provided");
1283       }
1284       return mapBuilder.build();
1285     }

1324     private void reportBadPrerequisite(Attribute attribute, String targetKind,
1325         Label prerequisiteLabel, String reason, boolean isWarning) {
1326       String msgPrefix = targetKind != null ? targetKind + " " : "";
1327       String msgReason = reason != null ? " (" + reason + ")" : "";
1328       if (isWarning) {
1329         attributeWarning(attribute.getName(), String.format(
1330             "%s'%s' is unexpected here%s; continuing anyway",
1331             msgPrefix, prerequisiteLabel, msgReason));

1334             "%s'%s' is misplaced here%s", msgPrefix, prerequisiteLabel, msgReason));

1338     private void validateDirectPrerequisiteType(ConfiguredTarget prerequisite,
1339         Attribute attribute) {
1340       Target prerequisiteTarget = prerequisite.getTarget();
1341       Label prerequisiteLabel = prerequisiteTarget.getLabel();
1342 
1343       if (prerequisiteTarget instanceof Rule) {
1344         Rule prerequisiteRule = (Rule) prerequisiteTarget;
1345 
1346         String reason = attribute.getValidityPredicate().checkValid(rule, prerequisiteRule);
1347         if (reason != null) {
1348           reportBadPrerequisite(attribute, prerequisiteTarget.getTargetKind(),
1349               prerequisiteLabel, reason, false);
1350         }
1351       }
1352 
1353       if (attribute.isStrictLabelCheckingEnabled()) {
1354         if (prerequisiteTarget instanceof Rule) {
1355           RuleClass ruleClass = ((Rule) prerequisiteTarget).getRuleClassObject();
1356           if (!attribute.getAllowedRuleClassesPredicate().apply(ruleClass)) {
1357             boolean allowedWithWarning = attribute.getAllowedRuleClassesWarningPredicate()
1358                 .apply(ruleClass);
1359             reportBadPrerequisite(attribute, prerequisiteTarget.getTargetKind(), prerequisiteLabel,
1360                 "expected " + attribute.getAllowedRuleClassesPredicate(), allowedWithWarning);
1361           }
1362         } else if (prerequisiteTarget instanceof FileTarget) {
1363           if (!attribute.getAllowedFileTypesPredicate()
1364               .apply(((FileTarget) prerequisiteTarget).getFilename())) {
1365             if (prerequisiteTarget instanceof InputFile
1366                 && !((InputFile) prerequisiteTarget).getPath().exists()) {
1367               // Misplaced labels, no corresponding target exists
1368               if (attribute.getAllowedFileTypesPredicate().isNone()
1369                   && !((InputFile) prerequisiteTarget).getFilename().contains(".")) {
1370                 // There are no allowed files in the attribute but it's not a valid rule,
1371                 // and the filename doesn't contain a dot --> probably a misspelled rule
1372                 attributeError(attribute.getName(),
1373                     "rule '" + prerequisiteLabel + "' does not exist");
1374               } else {
1375                 attributeError(attribute.getName(),
1376                     "target '" + prerequisiteLabel + "' does not exist");
1377               }
1378             } else {
1379               // The file exists but has a bad extension
1380               reportBadPrerequisite(attribute, "file", prerequisiteLabel,
1381                   "expected " + attribute.getAllowedFileTypesPredicate(), false);
1382             }
1383           }
1384         }
1385       }
1386     }

1408     private void validateDirectPrerequisiteFileTypes(ConfiguredTarget prerequisite,
1409         Attribute attribute) {
1410       if (attribute.isSkipAnalysisTimeFileTypeCheck()) {
1411         return;
1412       }
1413       FileTypeSet allowedFileTypes = attribute.getAllowedFileTypesPredicate();
1414       if (allowedFileTypes == null) {
1415         // It's not a label or label_list attribute.
1416         return;
1417       }
1418       if (allowedFileTypes == FileTypeSet.ANY_FILE && !attribute.isNonEmpty()
1419           && !attribute.isSingleArtifact()) {
1420         return;
1421       }
1422 
1423       // If we allow any file we still need to check if there are actually files generated
1424       // Note that this check only runs for ANY_FILE predicates if the attribute is NON_EMPTY
1425       // or SINGLE_ARTIFACT
1426       // If we performed this check when allowedFileTypes == NO_FILE this would
1427       // always throw an error in those cases
1428       if (allowedFileTypes != FileTypeSet.NO_FILE) {
1429         Iterable<Artifact> artifacts = prerequisite.getProvider(FileProvider.class)
1430             .getFilesToBuild();
1431         if (attribute.isSingleArtifact() && Iterables.size(artifacts) != 1) {
1432           attributeError(attribute.getName(),
1433               "'" + prerequisite.getLabel() + "' must produce a single file");
1434           return;
1435         }
1436         for (Artifact sourceArtifact : artifacts) {
1437           if (allowedFileTypes.apply(sourceArtifact.getFilename())) {
1438             return;
1439           }
1440         }
1441         attributeError(attribute.getName(), "'" + prerequisite.getLabel()
1442             + "' does not produce any " + rule.getRuleClass() + " " + attribute.getName()
1443             + " files (expected " + allowedFileTypes + ")");
1444       }
1445     }

1447     private void validateMandatoryProviders(ConfiguredTarget prerequisite, Attribute attribute) {
1448       for (String provider : attribute.getMandatoryProviders()) {
1449         if (prerequisite.get(provider) == null) {
1450           attributeError(attribute.getName(), "'" + prerequisite.getLabel()
1451               + "' does not have mandatory provider '" + provider + "'");
1452         }
1453       }
1454     }

1456     private void validateDirectPrerequisite(Attribute attribute, ConfiguredTarget prerequisite) {
1457       validateDirectPrerequisiteType(prerequisite, attribute);
1458       validateDirectPrerequisiteFileTypes(prerequisite, attribute);
1459       validateMandatoryProviders(prerequisite, attribute);
1460       prerequisiteValidator.validate(this, prerequisite, attribute);
1461     }
