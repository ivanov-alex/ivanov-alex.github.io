92    /**
93     * Command string for "sed" that tries to extract the application version number from a larger
94     * string. For example, from "foo_1.2.3_RC00" this would extract "1.2.3". This regex looks for
95     * versions of the format "x.y" or "x.y.z", which may be preceded and/or followed by other text,
96     * such as a project name or release candidate number.
97     *
98     * <p>This command also preserves double quotes around the string, if any.
99     */
100   private static final String EXTRACT_VERSION_NUMBER_SED_COMMAND =
101       "s#\\(\"\\)\\{0,1\\}\\(.*_\\)\\{0,1\\}\\([0-9][0-9]*\\(\\.[0-9][0-9]*\\)\\{1,2\\}\\)"
102       + "\\(_[^\"]*\\)\\{0,1\\}\\(\"\\)\\{0,1\\}#\\1\\3\\6#";

121     LOCAL_AND_DEPENDENCIES,

129   /**
130    * Creates a new application support within the given rule context.
131    *
132    * @param ruleContext context for the application-generating rule
133    * @param objcProvider provider containing all dependencies' information as well as some of this
134    *    rule's
135    * @param linkedBinary whether to look for a linked binary from this rule and dependencies or just
136    *    the latter
137    * @param bundleDirFormat format string representing the bundle's directory with a single
138    *    placeholder for the target name (e.g. {@code "Payload/%s.app"})
139    * @param bundleMinimumOsVersion the minimum OS version this bundle's plist should be generated
140    *    for (<b>not</b> the minimum OS version its binary is compiled with, that needs to be set
141    *    through the configuration)
142    */
143   ReleaseBundlingSupport(RuleContext ruleContext, ObjcProvider objcProvider,
144       LinkedBinary linkedBinary, String bundleDirFormat, String bundleMinimumOsVersion) {
145     this.linkedBinary = linkedBinary;
146     this.attributes = new Attributes(ruleContext);
147     this.ruleContext = ruleContext;
148     this.objcProvider = objcProvider;
149     this.intermediateArtifacts = ObjcRuleClasses.intermediateArtifacts(ruleContext);
150     bundling = bundling(ruleContext, objcProvider, bundleDirFormat, bundleMinimumOsVersion);
151     bundleSupport = new BundleSupport(ruleContext, bundling, extraActoolArgs());
152   }

154   /**
155    * Validates application-related attributes set on this rule and registers any errors with the
156    * rule context.
157    *
158    * @return this application support
159    */
160   ReleaseBundlingSupport validateAttributes() {
161     // No asset catalogs. That means you cannot specify app_icon or
162     // launch_image attributes, since they must not exist. However, we don't
163     // run actool in this case, which means it does not do validity checks,
164     // and we MUST raise our own error somehow...
165     if (!objcProvider.hasAssetCatalogs()) {
166       if (attributes.appIcon() != null) {
167         ruleContext.attributeError("app_icon",
168             String.format(NO_ASSET_CATALOG_ERROR_FORMAT, attributes.appIcon()));
169       }
170       if (attributes.launchImage() != null) {
171         ruleContext.attributeError("launch_image",
172             String.format(NO_ASSET_CATALOG_ERROR_FORMAT, attributes.launchImage()));
173       }
174     }
175 
176     if (bundleSupport.targetDeviceFamilies().isEmpty()) {
177       ruleContext.attributeError("families", INVALID_FAMILIES_ERROR);
178     }
179 
180     return this;
181   }

194   /**
195    * Registers actions required to build an application. This includes any
196    * {@link BundleSupport#registerActions(ObjcProvider) bundle} and bundle merge actions, signing
197    * this application if appropriate and combining several single-architecture binaries into one
198    * multi-architecture binary.
199    *
200    * @return this application support
201    */
202   ReleaseBundlingSupport registerActions() {
203     bundleSupport.registerActions(objcProvider);
204 
205     registerCombineArchitecturesAction();
206     registerTransformAndCopyBreakpadFilesAction();
207     registerSwiftStdlibActionsIfNecessary();
208 
209     ObjcConfiguration objcConfiguration = ObjcRuleClasses.objcConfiguration(ruleContext);
210     Artifact ipaOutput = ruleContext.getImplicitOutputArtifact(IPA);
211 
212     Artifact maybeSignedIpa;
213     if (objcConfiguration.getBundlingPlatform() == Platform.SIMULATOR) {
214       maybeSignedIpa = ipaOutput;
215     } else if (attributes.provisioningProfile() == null) {
216       throw new IllegalStateException(DEVICE_NO_PROVISIONING_PROFILE);
217     } else {
218       maybeSignedIpa = registerBundleSigningActions(ipaOutput);
219     }
220 
221     registerEmbedLabelPlistAction();
222     registerEnvironmentPlistAction();
223 
224     BundleMergeControlBytes bundleMergeControlBytes = new BundleMergeControlBytes(
225         bundling, maybeSignedIpa, objcConfiguration, bundleSupport.targetDeviceFamilies());
226     registerBundleMergeActions(
227         maybeSignedIpa, bundling.getBundleContentArtifacts(), bundleMergeControlBytes);
228 
229     return this;
230   }

232   private void registerEmbedLabelPlistAction() {
233     Artifact buildInfo = Iterables.getOnlyElement(
234         ruleContext.getBuildInfo(ObjcBuildInfoFactory.KEY));
235     String generatedVersionPlistPath = getGeneratedVersionPlist().getShellEscapedExecPathString();
236     String shellCommand = "VERSION=\"$("
237         + "grep \"^" + BuildInfo.BUILD_EMBED_LABEL + "\" "
238         + buildInfo.getShellEscapedExecPathString()
239         + " | cut -d' ' -f2- | sed -e '" + EXTRACT_VERSION_NUMBER_SED_COMMAND + "' | "
240         + "sed -e 's#\"#\\\"#g')\" && "
241         + "cat >" + generatedVersionPlistPath + " <<EOF\n"
242         + "<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n"
243         + "<!DOCTYPE plist PUBLIC \"-//Apple//DTD PLIST 1.0//EN\" "
244         + "\"http://www.apple.com/DTDs/PropertyList-1.0.dtd\">\n"
245         + "<plist version=\"1.0\">\n"
246         + "<dict>\n"
247         + "EOF\n"
248 
249             + "if [[ -n \"${VERSION}\" ]]; then\n"
250             + "  for KEY in CFBundleVersion CFBundleShortVersionString; do\n"
251             + "    echo \"  <key>${KEY}</key>\n\" >> "
252             + generatedVersionPlistPath + "\n"
253             + "    echo \"  <string>${VERSION}</string>\n\" >> "
254             + generatedVersionPlistPath + "\n"
255             + "  done\n"
256             + "fi\n"
257 
258             + "cat >>" + generatedVersionPlistPath + " <<EOF\n"
259             + "</dict>\n"
260             + "</plist>\n"
261             + "EOF\n";
262     ruleContext.registerAction(new SpawnAction.Builder()
263         .setMnemonic("ObjcVersionPlist")
264         .setShellCommand(shellCommand)
265         .addInput(buildInfo)
266         .addOutput(getGeneratedVersionPlist())
267         .build(ruleContext));
268   }

270   private void registerEnvironmentPlistAction() {
271     ObjcConfiguration configuration = ObjcRuleClasses.objcConfiguration(ruleContext);
272     // Generates a .plist that contains environment values (such as the SDK used to build, the Xcode
273     // version, etc), which are parsed from various .plist files of the OS, namely XCodes' and
274     // Platforms' plists.
275     // The resulting file is meant to be merged with the final bundle.
276     String command = Joiner.on(" && ").join(
277         "PLATFORM_PLIST=" + IosSdkCommands.platformDir(configuration) + "/Info.plist",
278         "PLIST=$(mktemp -d -t bazel_environment)/env.plist",
279         "os_build=$(/usr/bin/defaults read \"${PLATFORM_PLIST}\" BuildMachineOSBuild)",
280         "compiler=$(/usr/bin/defaults read \"${PLATFORM_PLIST}\" DTCompiler)",
281         "platform_version=$(/usr/bin/defaults read \"${PLATFORM_PLIST}\" Version)",
282         "sdk_build=$(/usr/bin/defaults read \"${PLATFORM_PLIST}\" DTSDKBuild)",
283         "platform_build=$(/usr/bin/defaults read \"${PLATFORM_PLIST}\" DTPlatformBuild)",
284         "xcode_build=$(/usr/bin/defaults read \"${PLATFORM_PLIST}\" DTXcodeBuild)",
285         "xcode_version=$(/usr/bin/defaults read \"${PLATFORM_PLIST}\" DTXcode)",
286         "/usr/bin/defaults write \"${PLIST}\" DTPlatformBuild -string ${platform_build}",
287         "/usr/bin/defaults write \"${PLIST}\" DTSDKBuild -string ${sdk_build}",
288         "/usr/bin/defaults write \"${PLIST}\" DTPlatformVersion -string ${platform_version}",
289         "/usr/bin/defaults write \"${PLIST}\" DTXcode -string ${xcode_version}",
290         "/usr/bin/defaults write \"${PLIST}\" DTXCodeBuild -string ${xcode_build}",
291         "/usr/bin/defaults write \"${PLIST}\" DTCompiler -string ${compiler}",
292         "/usr/bin/defaults write \"${PLIST}\" BuildMachineOSBuild -string ${os_build}",
293         "cat \"${PLIST}\" > " + getGeneratedEnvironmentPlist().getShellEscapedExecPathString(),
294         "rm -rf \"${PLIST}\"");
295     ruleContext.registerAction(ObjcRuleClasses.spawnBashOnDarwinActionBuilder(command)
296         .setMnemonic("EnvironmentPlist")
297         .addOutput(getGeneratedEnvironmentPlist())
298         .build(ruleContext));
299   }

301   private Artifact registerBundleSigningActions(Artifact ipaOutput) {
302     PathFragment entitlementsDirectory = ruleContext.getUniqueDirectory("entitlements");
303     Artifact teamPrefixFile =
304         ruleContext.getRelatedArtifact(entitlementsDirectory, ".team_prefix_file");
305     registerExtractTeamPrefixAction(teamPrefixFile);
306 
307     Artifact entitlementsNeedingSubstitution = attributes.entitlements();
308     if (entitlementsNeedingSubstitution == null) {
309       entitlementsNeedingSubstitution = ruleContext.getRelatedArtifact(
310           entitlementsDirectory, ".entitlements_with_variables");
311       registerExtractEntitlementsAction(entitlementsNeedingSubstitution);
312     }
313     Artifact entitlements =
314         ruleContext.getRelatedArtifact(entitlementsDirectory, ".entitlements");
315     registerEntitlementsVariableSubstitutionAction(
316         entitlementsNeedingSubstitution, entitlements, teamPrefixFile);
317     Artifact ipaUnsigned = ObjcRuleClasses.artifactByAppendingToRootRelativePath(
318         ruleContext, ipaOutput.getExecPath(), ".unsigned");
319     registerSignBundleAction(entitlements, ipaOutput, ipaUnsigned);
320     return ipaUnsigned;
321   }

337   /**
338    * Adds any files to the given nested set builder that should be built if this application is the
339    * top level target in a blaze invocation.
340    *
341    * @return this application support
342    */
343   ReleaseBundlingSupport addFilesToBuild(NestedSetBuilder<Artifact> filesToBuild) {
344     NestedSetBuilder<Artifact> debugSymbolBuilder = NestedSetBuilder.<Artifact>stableOrder()
345         .addTransitive(objcProvider.get(ObjcProvider.DEBUG_SYMBOLS));
346 
347     for (Artifact breakpadFile : getBreakpadFiles().values()) {
348       filesToBuild.add(breakpadFile);
349     }
350 
351     if (linkedBinary == LinkedBinary.LOCAL_AND_DEPENDENCIES
352         && ObjcRuleClasses.objcConfiguration(ruleContext).generateDebugSymbols()) {
353       IntermediateArtifacts intermediateArtifacts =
354           ObjcRuleClasses.intermediateArtifacts(ruleContext);
355       debugSymbolBuilder.add(intermediateArtifacts.dsymPlist())
356           .add(intermediateArtifacts.dsymSymbol())
357           .add(intermediateArtifacts.breakpadSym());
358     }
359 
360     filesToBuild.add(ruleContext.getImplicitOutputArtifact(ReleaseBundlingSupport.IPA))
361         // TODO(bazel-team): Fat binaries may require some merging of these file rather than just
362         // making them available.
363         .addTransitive(debugSymbolBuilder.build());
364     return this;
365   }

383         .addTransitiveAndPropagate(ObjcProvider.WEAK_SDK_FRAMEWORK, objcProvider)
384         .addTransitiveAndPropagate(ObjcProvider.FRAMEWORK_DIR, objcProvider)
385         .addTransitiveAndPropagate(ObjcProvider.FRAMEWORK_FILE, objcProvider)
386         .build();
387     // TODO(bazel-team): Handle the FRAMEWORK_DIR key properly. We probably want to add it to

438   private Bundling bundling(RuleContext ruleContext, ObjcProvider objcProvider,
439       String bundleDirFormat, String minimumOsVersion) {
440     ImmutableList<BundleableFile> extraBundleFiles;
441     ObjcConfiguration objcConfiguration = ObjcRuleClasses.objcConfiguration(ruleContext);
442     if (objcConfiguration.getBundlingPlatform() == Platform.DEVICE) {
443       extraBundleFiles = ImmutableList.of(new BundleableFile(
444           new Attributes(ruleContext).provisioningProfile(),
445           PROVISIONING_PROFILE_BUNDLE_FILE));
446     } else {
447       extraBundleFiles = ImmutableList.of();
448     }
449 
450     String primaryBundleId = null;
451     String fallbackBundleId = null;
452 
453     if (ruleContext.attributes().isAttributeValueExplicitlySpecified("bundle_id")) {
454       primaryBundleId = ruleContext.attributes().get("bundle_id", Type.STRING);
455     } else {
456       fallbackBundleId = ruleContext.attributes().get("bundle_id", Type.STRING);
457     }
458 
459     return new Bundling.Builder()
460         .setName(ruleContext.getLabel().getName())
461         // Architecture that determines which nested bundles are kept.
462         .setArchitecture(objcConfiguration.getDependencySingleArchitecture())
463         .setBundleDirFormat(bundleDirFormat)
464         .addExtraBundleFiles(extraBundleFiles)
465         .setObjcProvider(objcProvider)
466         .addInfoplistInputFromRule(ruleContext)
467         .addInfoplistInput(getGeneratedVersionPlist())
468         .addInfoplistInput(getGeneratedEnvironmentPlist())
469         .setIntermediateArtifacts(ObjcRuleClasses.intermediateArtifacts(ruleContext))
470         .setPrimaryBundleId(primaryBundleId)
471         .setFallbackBundleId(fallbackBundleId)
472         .setMinimumOsVersion(minimumOsVersion)
473         .build();
474   }

476   private void registerCombineArchitecturesAction() {
477     Artifact resultingLinkedBinary = intermediateArtifacts.combinedArchitectureBinary();
478     NestedSet<Artifact> linkedBinaries = linkedBinaries();
479 
480     ruleContext.registerAction(ObjcRuleClasses.spawnOnDarwinActionBuilder()
481         .setMnemonic("ObjcCombiningArchitectures")
482         .addTransitiveInputs(linkedBinaries)
483         .addOutput(resultingLinkedBinary)
484         .setExecutable(ObjcRuleClasses.LIPO)
485         .setCommandLine(CustomCommandLine.builder()
486             .addExecPaths("-create", linkedBinaries)
487             .addExecPath("-o", resultingLinkedBinary)
488             .build())
489         .build(ruleContext));
490   }

492   private NestedSet<Artifact> linkedBinaries() {
493     NestedSetBuilder<Artifact> linkedBinariesBuilder = NestedSetBuilder.<Artifact>stableOrder()
494         .addTransitive(attributes.dependentLinkedBinaries());
495     if (linkedBinary == LinkedBinary.LOCAL_AND_DEPENDENCIES) {
496       linkedBinariesBuilder.add(intermediateArtifacts.strippedSingleArchitectureBinary());
497     }
498     return linkedBinariesBuilder.build();
499   }

501   /** Returns this target's Xcode build settings. */
502   private Iterable<XcodeprojBuildSetting> buildSettings() {
503     ImmutableList.Builder<XcodeprojBuildSetting> buildSettings = new ImmutableList.Builder<>();
504     if (attributes.appIcon() != null) {
505       buildSettings.add(XcodeprojBuildSetting.newBuilder()
506           .setName("ASSETCATALOG_COMPILER_APPICON_NAME")
507           .setValue(attributes.appIcon())
508           .build());
509     }
510     if (attributes.launchImage() != null) {
511       buildSettings.add(XcodeprojBuildSetting.newBuilder()
512           .setName("ASSETCATALOG_COMPILER_LAUNCHIMAGE_NAME")
513           .setValue(attributes.launchImage())
514           .build());
515     }
516 
517     // Convert names to a sequence containing "1" and/or "2" for iPhone and iPad, respectively.
518     ImmutableSet<TargetDeviceFamily> families = bundleSupport.targetDeviceFamilies();
519     Iterable<Integer> familyIndexes =
520         families.isEmpty() ? ImmutableList.<Integer>of() : UI_DEVICE_FAMILY_VALUES.get(families);
521     buildSettings.add(XcodeprojBuildSetting.newBuilder()
522         .setName("TARGETED_DEVICE_FAMILY")
523         .setValue(Joiner.on(',').join(familyIndexes))
524         .build());
525 
526     Artifact entitlements = attributes.entitlements();
527     if (entitlements != null) {
528       buildSettings.add(XcodeprojBuildSetting.newBuilder()
529           .setName("CODE_SIGN_ENTITLEMENTS")
530           .setValue("$(WORKSPACE_ROOT)/" + entitlements.getExecPathString())
531           .build());
532     }
533 
534     return buildSettings.build();
535   }

537   private ReleaseBundlingSupport registerSignBundleAction(
538       Artifact entitlements, Artifact ipaOutput, Artifact ipaUnsigned) {
539     // TODO(bazel-team): Support variable substitution
540 
541     ImmutableList.Builder<String> dirsToSign = new ImmutableList.Builder<>();
542 
543     // Explicitly sign Swift frameworks. Unfortunately --deep option on codesign doesn't do this
544     // automatically.
545     // The order here is important. The innermost code must singed first.
546     String bundleDir = ShellUtils.shellEscape(bundling.getBundleDir());
547     if (objcProvider.is(USES_SWIFT)) {
548       dirsToSign.add(bundleDir + "/Frameworks/*");
549     }
550     dirsToSign.add(bundleDir);
551 
552     StringBuilder codesignCommandLineBuilder = new StringBuilder();
553     for (String dir : dirsToSign.build()) {
554       codesignCommandLineBuilder
555           .append(codesignCommand(attributes.provisioningProfile(), entitlements, "${t}/" + dir))
556           .append(" && ");
557     }
558 
559     // TODO(bazel-team): Support nested code signing.
560     String shellCommand = "set -e && "
561         + "t=$(mktemp -d -t signing_intermediate) && "
562         + "trap \"rm -rf ${t}\" EXIT && "
563         // Get an absolute path since we need to cd into the temp directory for zip.
564         + "signed_ipa=${PWD}/" + ipaOutput.getShellEscapedExecPathString() + " && "
565         + "/usr/bin/unzip -qq " + ipaUnsigned.getShellEscapedExecPathString() + " -d ${t} && "
566         + codesignCommandLineBuilder.toString()
567         // Using zip since we need to preserve permissions
568         + "cd ${t} && /usr/bin/zip -q -r \"${signed_ipa}\" .";
569     ruleContext.registerAction(ObjcRuleClasses.spawnBashOnDarwinActionBuilder(shellCommand)
570         .setMnemonic("IosSignBundle")
571         .setProgressMessage("Signing iOS bundle: " + ruleContext.getLabel())
572         .addInput(ipaUnsigned)
573         .addInput(attributes.provisioningProfile())
574         .addInput(entitlements)
575         .addOutput(ipaOutput)
576         .build(ruleContext));
577 
578     return this;
579   }

601   /**
602    * Registers the actions that transform and copy the breakpad files from the CPU-specific binaries
603    * that are part of this application. There are two steps involved: 1) The breakpad files have to
604    * be renamed to include their corresponding CPU architecture as a suffix. 2) The first line of
605    * the breakpad file has to be rewritten, as it has to include the name of the application instead
606    * of the name of the binary artifact.
607    *
608    * <p>Example:<br>
609    * The ios_application "PrenotCalculator" is specified to use "PrenotCalculatorBinary" as its
610    * binary. Assuming that the application is built for armv7 and arm64 CPUs, in the build process
611    * two binaries with a corresponding breakpad file each will be built:
612    *
613    * <pre>blaze-out/xyz-crosstool-ios-arm64/.../PrenotCalculatorBinary_bin
614    * blaze-out/xyz-crosstool-ios-arm64/.../PrenotCalculatorBinary.breakpad
615    * blaze-out/xyz-crosstool-ios-armv7/.../PrenotCalculatorBinary_bin
616    * blaze-out/xyz-crosstool-ios-armv7/.../PrenotCalculatorBinary.breakpad</pre>
617    *
618    * <p>The first line of the breakpad files will look like this:
619    * <pre>MODULE mac arm64 8A7A2DDD28E83E27B339E63631ADBEF30 PrenotCalculatorBinary_bin</pre>
620    *
621    * <p>For our application, we have to transform & copy these breakpad files like this:
622    * <pre>$ head -n1 blaze-bin/.../PrenotCalculator_arm64.breakpad
623    * MODULE mac arm64 8A7A2DDD28E83E27B339E63631ADBEF30 PrenotCalculator</pre>
624    */
625   private void registerTransformAndCopyBreakpadFilesAction() {
626     for (Entry<Artifact, Artifact> breakpadFiles : getBreakpadFiles().entrySet()) {
627       ruleContext.registerAction(
628           new SpawnAction.Builder().setMnemonic("CopyBreakpadFile")
629               .setShellCommand(String.format(
630                   // This sed command replaces the last word of the first line with the application
631                   // name.
632                   "sed -r \"1 s/^(MODULE \\w* \\w* \\w*).*$/\\1 %s/\" < %s > %s",
633                   ruleContext.getLabel().getName(), breakpadFiles.getKey().getExecPathString(),
634                   breakpadFiles.getValue().getExecPathString()))
635               .addInput(breakpadFiles.getKey())
636               .addOutput(breakpadFiles.getValue())
637               .build(ruleContext));
638     }
639   }
640 
641   /**
642    * Returns a map of input breakpad artifacts from the CPU-specific binaries built for this
643    * ios_application to the new output breakpad artifacts.
644    */
645   private ImmutableMap<Artifact, Artifact> getBreakpadFiles() {
646     ImmutableMap.Builder<Artifact, Artifact> results = ImmutableMap.builder();
647     for (Entry<String, Artifact> breakpadFile : attributes.cpuSpecificBreakpadFiles().entrySet()) {
648       Artifact destBreakpad = intermediateArtifacts.breakpadSym(breakpadFile.getKey());
649       results.put(breakpadFile.getValue(), destBreakpad);
650     }
651     return results.build();
652   }

667   private ReleaseBundlingSupport registerExtractEntitlementsAction(Artifact entitlements) {
668     // See Apple Glossary (http://goo.gl/EkhXOb)

677         + entitlements.getShellEscapedExecPathString();
678     ruleContext.registerAction(ObjcRuleClasses.spawnBashOnDarwinActionBuilder(shellCommand)
679         .setMnemonic("ExtractIosEntitlements")
680         .setProgressMessage("Extracting entitlements: " + ruleContext.getLabel())
681         .addInput(attributes.provisioningProfile())
682         .addOutput(entitlements)
683         .build(ruleContext));

688   private void registerEntitlementsVariableSubstitutionAction(Artifact in, Artifact out,
689       Artifact prefix) {
690     String escapedBundleId = ShellUtils.shellEscape(attributes.bundleId());
691     String shellCommand = "set -e && "
692         + "PREFIX=\"$(cat " + prefix.getShellEscapedExecPathString() + ")\" && "
693         + "sed "
694         // Replace .* from default entitlements file with bundle ID where suitable.
695         + "-e \"s#${PREFIX}\\.\\*#${PREFIX}." + escapedBundleId + "#g\" "
696 
697         // Replace some variables that people put in their own entitlements files
698         + "-e \"s#\\$(AppIdentifierPrefix)#${PREFIX}.#g\" "
699         + "-e \"s#\\$(CFBundleIdentifier)#" + escapedBundleId + "#g\" "
700 
701         + in.getShellEscapedExecPathString() + " "
702         + "> " + out.getShellEscapedExecPathString();
703     ruleContext.registerAction(new SpawnAction.Builder()
704         .setMnemonic("SubstituteIosEntitlements")
705         .setShellCommand(shellCommand)
706         .addInput(in)
707         .addInput(prefix)
708         .addOutput(out)
709         .build(ruleContext));
710   }

742   private String codesignCommand(
743       Artifact provisioningProfile, Artifact entitlements, String appDir) {
744     String fingerprintCommand =
745         "PLIST=$(mktemp -t cert.plist) && trap \"rm ${PLIST}\" EXIT && "
746             + extractPlistCommand(provisioningProfile) + " > ${PLIST} && "
747             + "/usr/libexec/PlistBuddy -c 'Print DeveloperCertificates:0' ${PLIST} | "
748             + "openssl x509 -inform DER -noout -fingerprint | "
749             + "cut -d= -f2 | sed -e 's#:##g'";
750     return String.format(
751         "/usr/bin/codesign --force --sign $(%s) --entitlements %s %s",
752         fingerprintCommand,
753         entitlements.getShellEscapedExecPathString(),
754         appDir);
755   }

869   /**
870    * Transition that results in one configured target per architecture set in {@code
871    * --ios_multi_cpus}.
872    */
873   protected static class SplitArchTransition implements SplitTransition<BuildOptions> {
874 
875     @Override
876     public final List<BuildOptions> split(BuildOptions buildOptions) {
877       List<String> iosMultiCpus = buildOptions.get(ObjcCommandLineOptions.class).iosMultiCpus;
878       if (iosMultiCpus.isEmpty()) {
879         return defaultOptions(buildOptions);
880       }
881 
882       ImmutableList.Builder<BuildOptions> splitBuildOptions = ImmutableList.builder();
883       for (String iosCpu : iosMultiCpus) {
884         BuildOptions splitOptions = buildOptions.clone();
885         setArchitectureOptions(splitOptions, iosCpu);
886         setAdditionalOptions(splitOptions, buildOptions);
887         splitOptions.get(ObjcCommandLineOptions.class).configurationDistinguisher =
888             getConfigurationDistinguisher();
889         splitBuildOptions.add(splitOptions);
890       }
891       return splitBuildOptions.build();
892     }
893 
894     /**
895      * Returns the default options to use if no split architectures are specified.
896      *
897      * @param originalOptions original options before this transition
898      */
899     protected ImmutableList<BuildOptions> defaultOptions(BuildOptions originalOptions) {
900       return ImmutableList.of();
901     }
902 
903     /**
904      * Sets or overwrites flags on the given split options.
905      *
906      * <p>Invoked once for each configuration produced by this transition.
907      *
908      * @param splitOptions options to use after this transition
909      * @param originalOptions original options before this transition
910      */
911     protected void setAdditionalOptions(BuildOptions splitOptions, BuildOptions originalOptions) {}
912 
913     private void setArchitectureOptions(BuildOptions splitOptions, String iosCpu) {
914       splitOptions.get(ObjcCommandLineOptions.class).iosSplitCpu = iosCpu;
915       splitOptions.get(ObjcCommandLineOptions.class).iosCpu = iosCpu;
916       if (splitOptions.get(ObjcCommandLineOptions.class).enableCcDeps) {
917         // Only set the (CC-compilation) CPU for dependencies if explicitly required by the user.
918         // This helps users of the iOS rules who do not depend on CC rules as these CPU values
919         // require additional flags to work (e.g. a custom crosstool) which now only need to be set
920         // if this feature is explicitly requested.
921         splitOptions.get(BuildConfiguration.Options.class).cpu = "ios_" + iosCpu;
922       }
923     }
924 
925     @Override
926     public boolean defaultsToSelf() {
927       return true;
928     }
929 
930     /**
931      * Returns the configuration distinguisher for this transition instance.
932      */
933     protected ConfigurationDistinguisher getConfigurationDistinguisher() {
934       return ConfigurationDistinguisher.APPLICATION;
935     }
936 
937     /**
938      * Value used to avoid multiple configurations from conflicting. No two instances of this
939      * transition may exist with the same value in a single Bazel invocation.
940      */
941     enum ConfigurationDistinguisher {
942       EXTENSION, APPLICATION, UNKNOWN
943     }
944   }
