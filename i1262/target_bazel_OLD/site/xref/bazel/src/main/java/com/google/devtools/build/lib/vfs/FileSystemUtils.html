60    /**
61     * Throws exceptions if {@code baseName} is not a valid base name. A valid
62     * base name:
63     * <ul>
64     * <li>Is not null
65     * <li>Is not an empty string
66     * <li>Is not "." or ".."
67     * <li>Does not contain a slash
68     * </ul>
69     */
70    @ThreadSafe
71    public static void checkBaseName(String baseName) {
72      if (baseName.length() == 0) {
73        throw new IllegalArgumentException("Child must not be empty string ('')");
74      }
75      if (baseName.equals(".") || baseName.equals("..")) {
76        throw new IllegalArgumentException("baseName must not be '" + baseName + "'");
77      }
78      if (baseName.indexOf('/') != -1) {
79        throw new IllegalArgumentException("baseName must not contain a slash: '" + baseName + "'");
80      }
81    }

83    /**
84     * Returns the common ancestor between two paths, or null if none (including
85     * if they are on different filesystems).
86     */
87    public static Path commonAncestor(Path a, Path b) {
88      while (a != null && !b.startsWith(a)) {
89        a = a.getParentDirectory();  // returns null at root
90      }
91      return a;
92    }

105   /**
106    * Returns a path fragment from a given from-dir to a given to-path. May be
107    * either a short relative path "foo/bar", an up'n'over relative path
108    * "../../foo/bar" or an absolute path.
109    */
110   public static PathFragment relativePath(Path fromDir, Path to) {
111     if (to.getFileSystem() != fromDir.getFileSystem()) {
112       throw new IllegalArgumentException("fromDir and to must be on the same FileSystem");
113     }
114 
115     return relativePath(fromDir.asFragment(), to.asFragment());
116   }

118   /**
119    * Returns a path fragment from a given from-dir to a given to-path.
120    */
121   public static PathFragment relativePath(PathFragment fromDir, PathFragment to) {
122     if (to.equals(fromDir)) {
123       return new PathFragment(".");  // same dir, just return '.'
124     }
125     if (to.startsWith(fromDir)) {
126       return to.relativeTo(fromDir);  // easy case--it's a descendant
127     }
128     PathFragment ancestor = commonAncestor(fromDir, to);
129     if (ancestor == null) {
130       return to;  // no common ancestor, use 'to'
131     }
132     int levels = fromDir.relativeTo(ancestor).segmentCount();
133     StringBuilder dotdots = new StringBuilder();
134     for (int i = 0; i < levels; i++) {
135       dotdots.append("../");
136     }
137     return new PathFragment(dotdots.toString()).getRelative(to.relativeTo(ancestor));
138   }

140   /**
141    * Returns the longest prefix from a given set of 'prefixes' that are
142    * contained in 'path'. I.e the closest ancestor directory containing path.
143    * Returns null if none found.
144    */
145   public static PathFragment longestPathPrefix(PathFragment path, Set<PathFragment> prefixes) {
146     for (int i = path.segmentCount(); i >= 1; i--) {
147       PathFragment prefix = path.subFragment(0, i);
148       if (prefixes.contains(prefix)) {
149         return prefix;
150       }
151     }
152     return null;
153   }

155   /**
156    * Removes the shortest suffix beginning with '.' from the basename of the
157    * filename string. If the basename contains no '.', the filename is returned
158    * unchanged.
159    *
160    * <p>e.g. "foo/bar.x" -> "foo/bar"
161    *
162    * <p>Note that if the filename is composed entirely of ".", this method will return the string
163    * with one fewer ".", which may have surprising effects.
164    */
165   @ThreadSafe
166   public static String removeExtension(String filename) {
167     int lastDotIndex = filename.lastIndexOf('.');
168     if (lastDotIndex == -1) { return filename; }
169     int lastSlashIndex = filename.lastIndexOf('/');
170     if (lastSlashIndex > lastDotIndex) {
171       return filename;
172     }
173     return filename.substring(0, lastDotIndex);
174   }

215   /**
216    * Returns a new {@code PathFragment} formed by replacing the extension of the
217    * last path segment of {@code path} with {@code newExtension}. Null is
218    * returned iff {@code path} has zero segments or it doesn't end with {@code oldExtension}.
219    */
220   public static PathFragment replaceExtension(PathFragment path, String newExtension,
221       String oldExtension) {
222     String base = path.getBaseName();
223     if (!base.endsWith(oldExtension)) {
224       return null;
225     }
226     String newBase = base.substring(0, base.length() - oldExtension.length()) + newExtension;
227     return path.replaceName(newBase);
228   }

248   /**
249    * Returns a new {@code PathFragment} formed by replacing the first, or all if
250    * {@code replaceAll} is true, {@code oldSegment} of {@code path} with {@code
251    * newSegment}.
252    */
253   public static PathFragment replaceSegments(PathFragment path,
254       String oldSegment, String newSegment, boolean replaceAll) {
255     int count = path.segmentCount();
256     for (int i = 0; i < count; i++) {
257       if (path.getSegment(i).equals(oldSegment)) {
258         path = new PathFragment(path.subFragment(0, i),
259                                 new PathFragment(newSegment),
260                                 path.subFragment(i+1, count));
261         if (!replaceAll) {
262           return path;
263         }
264       }
265     }
266     return path;
267   }

278   /**
279    * Given a string that represents a file with an extension separated by a '.' and a string
280    * to append, return a string in which {@code toAppend} has been appended to {@code name}
281    * before the last '.' character.  If {@code name} does not include a '.', appends {@code
282    * toAppend} at the end.
283    *
284    * <p>For example,
285    * ("libfoo.jar", "-src") ==> "libfoo-src.jar"
286    * ("libfoo", "-src") ==> "libfoo-src"
287    */
288   private static String appendWithoutExtension(String name, String toAppend) {
289     int dotIndex = name.lastIndexOf('.');
290     if (dotIndex > 0) {
291       String baseName = name.substring(0, dotIndex);
292       String extension = name.substring(dotIndex);
293       return baseName + toAppend + extension;
294     } else {
295       return name + toAppend;
296     }
297   }

323   /**
324    * "Touches" the file or directory specified by the path, following symbolic
325    * links. If it does not exist, it is created as an empty file; otherwise, the
326    * time of last access is updated to the current time.
327    *
328    * @throws IOException if there was an error while touching the file
329    */
330   @ThreadSafe
331   public static void touchFile(Path path) throws IOException {
332     if (path.exists()) {
333       // -1L means "use the current time", and is ultimately implemented by
334       // utime(path, null), thereby using the kernel's clock, not the JVM's.
335       // (A previous implementation based on the JVM clock was found to be
336       // skewy.)
337       path.setLastModifiedTime(-1L);
338     } else {
339       createEmptyFile(path);
340     }
341   }

388   /**
389    * Creates or updates a symbolic link from 'link' to 'target'. Replaces
390    * existing symbolic links with target, and skips the link creation if it is
391    * already present. Will also create any missing ancestor directories of the
392    * link. This method is non-atomic
393    *
394    * <p>Note: this method will throw an IOException if there is an unequal
395    * non-symlink at link.
396    *
397    * @throws IOException if the creation of the symbolic link was unsuccessful
398    *         for any reason.
399    */
400   @ThreadSafe  // but not atomic
401   public static void ensureSymbolicLink(Path link, PathFragment target) throws IOException {
402     // TODO(bazel-team): (2009) consider adding the logic for recovering from the case when
403     // we have already created a parent directory symlink earlier.
404     try {
405       if (link.readSymbolicLink().equals(target)) {
406         return;  // Do nothing if the link is already there.
407       }
408     } catch (IOException e) { // link missing or broken
409       /* fallthru and do the work below */
410     }
411     if (link.isSymbolicLink()) {
412       link.delete();  // Remove the symlink since it is pointing somewhere else.
413     } else {
414       createDirectoryAndParents(link.getParentDirectory());
415     }
416     try {
417       link.createSymbolicLink(target);
418     } catch (IOException e) {
419       // Only pass on exceptions caused by a true link creation failure.
420       if (!link.isSymbolicLink() ||
421           !link.resolveSymbolicLinks().equals(link.getRelative(target))) {
422         throw e;
423       }
424     }
425   }

447   /**
448    * Copies the file from location "from" to location "to", while overwriting a
449    * potentially existing "to". File's last modified time, executable and
450    * writable bits are also preserved.
451    *
452    * <p>If no error occurs, the method returns normally. If a parent directory does
453    * not exist, a FileNotFoundException is thrown. An IOException is thrown when
454    * other erroneous situations occur. (e.g. read errors)
455    */
456   @ThreadSafe  // but not atomic
457   public static void copyFile(Path from, Path to) throws IOException {
458     try {
459       to.delete();
460     } catch (IOException e) {
461       throw new IOException("error copying file: "
462           + "couldn't delete destination: " + e.getMessage());
463     }
464     asByteSource(from).copyTo(asByteSink(to));
465     to.setLastModifiedTime(from.getLastModifiedTime()); // Preserve mtime.
466     if (!from.isWritable()) {
467       to.setWritable(false); // Make file read-only if original was read-only.
468     }
469     to.setExecutable(from.isExecutable()); // Copy executable bit.
470   }

472   /**
473    * Copies a tool binary from one path to another, returning the target path.
474    * The directory of the target path must already exist. The target copy's time
475    * is set to match, as well as its read-only and executable flags. The
476    * operation is skipped if the target file has the same time and size as the
477    * source.
478    */
479   public static Path copyTool(Path source, Path target) throws IOException {
480     FileStatus sourceStat = null;
481     FileStatus targetStat = target.statNullable();
482     if (targetStat != null) {
483       // stat the source file only if we'll need the stat.
484       sourceStat = source.stat(Symlinks.FOLLOW);
485     }
486     if (targetStat == null ||
487         targetStat.getLastModifiedTime() != sourceStat.getLastModifiedTime() ||
488         targetStat.getSize() != sourceStat.getSize()) {
489       copyFile(source, target);
490       target.setWritable(source.isWritable());
491       target.setExecutable(source.isExecutable());
492       target.setLastModifiedTime(source.getLastModifiedTime());
493     }
494     return target;
495   }

501   /**
502    * Returns a new collection containing all of the paths below a given root
503    * path, for which the given predicate is true. Symbolic links are not
504    * followed, and may appear in the result.
505    *
506    * @throws IOException If the root does not denote a directory

508   @ThreadSafe
509   public static Collection<Path> traverseTree(Path root, Predicate<? super Path> predicate)
510       throws IOException {
511     List<Path> paths = new ArrayList<>();
512     traverseTree(paths, root, predicate);
513     return paths;
514   }

516   /**
517    * Populates an existing Path List, adding all of the paths below a given root
518    * path for which the given predicate is true. Symbolic links are not
519    * followed, and may appear in the result.
520    *
521    * @throws IOException If the root does not denote a directory
522    */
523   @ThreadSafe
524   public static void traverseTree(Collection<Path> paths, Path root,
525       Predicate<? super Path> predicate) throws IOException {
526     for (Path p : root.getDirectoryEntries()) {
527       if (predicate.apply(p)) {
528         paths.add(p);
529       }
530       if (p.isDirectory(Symlinks.NOFOLLOW)) {
531         traverseTree(paths, p, predicate);
532       }
533     }
534   }

548   /**
549    * Deletes all dir trees recursively beneath 'dir' if it's a directory,
550    * nothing otherwise. Does not follow any symbolic links.
551    *
552    * @throws IOException if any file could not be removed.
553    */
554   @ThreadSafe
555   public static void deleteTreesBelow(Path dir) throws IOException {
556     if (dir.isDirectory(Symlinks.NOFOLLOW)) {  // real directories (not symlinks)
557       dir.setReadable(true);
558       dir.setWritable(true);
559       dir.setExecutable(true);
560       for (Path child : dir.getDirectoryEntries()) {
561         deleteTree(child);
562       }
563     }
564   }

566   /**
567    * Delete all dir trees under a given 'dir' that don't start with one of a set
568    * of given 'prefixes'. Does not follow any symbolic links.
569    */
570   @ThreadSafe
571   public static void deleteTreesBelowNotPrefixed(Path dir, String[] prefixes) throws IOException {
572     dirloop:
573     for (Path p : dir.getDirectoryEntries()) {
574       String name = p.getBaseName();
575       for (int i = 0; i < prefixes.length; i++) {
576         if (name.startsWith(prefixes[i])) {
577           continue dirloop;
578         }
579       }
580       deleteTree(p);
581     }
582   }

584   /**
585    * Copies all dir trees under a given 'from' dir to location 'to', while overwriting
586    * all files in the potentially existing 'to'. Does not follow any symbolic links,
587    * but copies them instead.
588    *
589    * <p>The source and the destination must be non-overlapping, otherwise an
590    * IllegalArgumentException will be thrown. This method cannot be used to copy
591    * a dir tree to a sub tree of itself.
592    *
593    * <p>If no error occurs, the method returns normally. If the given 'from' does
594    * not exist, a FileNotFoundException is thrown. An IOException is thrown when
595    * other erroneous situations occur. (e.g. read errors)
596    */
597   @ThreadSafe
598   public static void copyTreesBelow(Path from , Path to) throws IOException {
599     if (to.startsWith(from)) {
600       throw new IllegalArgumentException(to + " is a subdirectory of " + from);
601     }
602 
603     Collection<Path> entries = from.getDirectoryEntries();
604     for (Path entry : entries) {
605       if (entry.isDirectory(Symlinks.NOFOLLOW)) {
606         Path subDir = to.getChild(entry.getBaseName());
607         subDir.createDirectory();
608         copyTreesBelow(entry, subDir);
609       } else if (entry.isSymbolicLink()) {
610         Path newLink = to.getChild(entry.getBaseName());
611         newLink.createSymbolicLink(entry.readSymbolicLink());
612       } else {
613         Path newEntry = to.getChild(entry.getBaseName());
614         copyFile(entry, newEntry);
615       }
616     }
617   }

619   /**
620    * Attempts to create a directory with the name of the given path, creating
621    * ancestors as necessary.
622    *
623    * <p>Postcondition: completes normally iff {@code dir} denotes an existing
624    * directory (not necessarily canonical); completes abruptly otherwise.
625    *
626    * @return true if the directory was successfully created anew, false if it
627    *   already existed (including the case where {@code dir} denotes a symlink
628    *   to an existing directory)
629    * @throws IOException if the directory could not be created
630    */
631   @ThreadSafe
632   public static boolean createDirectoryAndParents(Path dir) throws IOException {
633     // Optimised for minimal number of I/O calls.
634 
635     // Don't attempt to create the root directory.
636     if (dir.getParentDirectory() == null) { return false; }
637 
638     FileSystem filesystem = dir.getFileSystem();
639     if (filesystem instanceof UnionFileSystem) {
640       // If using UnionFS, make sure that we do not traverse filesystem boundaries when creating
641       // parent directories by rehoming the path on the most specific filesystem.
642       FileSystem delegate = ((UnionFileSystem) filesystem).getDelegate(dir);
643       dir = delegate.getPath(dir.asFragment());
644     }
645 
646     try {
647       return dir.createDirectory();
648     } catch (IOException e) {
649       if (e.getMessage().endsWith(" (No such file or directory)")) { // ENOENT
650         createDirectoryAndParents(dir.getParentDirectory());
651         return dir.createDirectory();
652       } else if (e.getMessage().endsWith(" (File exists)") && dir.isDirectory()) { // EEXIST
653         return false;
654       } else {
655         throw e; // some other error (e.g. ENOTDIR, EACCES, etc.)
656       }
657     }
658   }

660   /**
661    * Attempts to remove a relative chain of directories under a given base.
662    * Returns {@code true} if the removal was successful, and returns {@code
663    * false} if the removal fails because a directory was not empty. An
664    * {@link IOException} is thrown for any other errors.
665    */
666   @ThreadSafe
667   public static boolean removeDirectoryAndParents(Path base, PathFragment toRemove) {
668     if (toRemove.isAbsolute()) {
669       return false;
670     }
671     try {
672       for (; toRemove.segmentCount() > 0; toRemove = toRemove.getParentDirectory()) {
673         Path p = base.getRelative(toRemove);
674         if (p.exists()) {
675           p.delete();
676         }
677       }
678     } catch (IOException e) {
679       return false;
680     }
681     return true;
682   }

684   /**
685    * Takes a map of directory fragments to root paths, and creates a symlink
686    * forest under an existing linkRoot to the corresponding source dirs or
687    * files. Symlink are made at the highest dir possible, linking files directly
688    * only when needed with nested packages.
689    */
690   public static void plantLinkForest(ImmutableMap<PathFragment, Path> packageRootMap, Path linkRoot)
691       throws IOException {
692     Path emptyPackagePath = null;
693 
694     // Create a sorted map of all dirs (packages and their ancestors) to sets of their roots.
695     // Packages come from exactly one root, but their shared ancestors may come from more.
696     // The map is maintained sorted lexicographically, so parents are before their children.
697     Map<PathFragment, Set<Path>> dirRootsMap = Maps.newTreeMap();
698     for (Map.Entry<PathFragment, Path> entry : packageRootMap.entrySet()) {
699       PathFragment pkgDir = entry.getKey();
700       Path pkgRoot = entry.getValue();
701       if (pkgDir.segmentCount() == 0) {
702         emptyPackagePath = entry.getValue();
703       }
704       for (int i = 1; i <= pkgDir.segmentCount(); i++) {
705         PathFragment dir = pkgDir.subFragment(0, i);
706         Set<Path> roots = dirRootsMap.get(dir);
707         if (roots == null) {
708           roots = Sets.newHashSet();
709           dirRootsMap.put(dir, roots);
710         }
711         roots.add(pkgRoot);
712       }
713     }
714     // Now add in roots for all non-pkg dirs that are in between two packages, and missed above.
715     for (Map.Entry<PathFragment, Set<Path>> entry : dirRootsMap.entrySet()) {
716       PathFragment dir = entry.getKey();
717       if (!packageRootMap.containsKey(dir)) {
718         PathFragment pkgDir = longestPathPrefix(dir, packageRootMap.keySet());
719         if (pkgDir != null) {
720           entry.getValue().add(packageRootMap.get(pkgDir));
721         }
722       }
723     }
724     // Create output dirs for all dirs that have more than one root and need to be split.
725     for (Map.Entry<PathFragment, Set<Path>> entry : dirRootsMap.entrySet()) {
726       PathFragment dir = entry.getKey();
727       if (entry.getValue().size() > 1) {
728         if (LOG_FINER) {
729           LOG.finer("mkdir " + linkRoot.getRelative(dir));
730         }
731         createDirectoryAndParents(linkRoot.getRelative(dir));
732       }
733     }
734     // Make dir links for single rooted dirs.
735     for (Map.Entry<PathFragment, Set<Path>> entry : dirRootsMap.entrySet()) {
736       PathFragment dir = entry.getKey();
737       Set<Path> roots = entry.getValue();
738       // Simple case of one root for this dir.
739       if (roots.size() == 1) {
740         if (dir.segmentCount() > 1 && dirRootsMap.get(dir.getParentDirectory()).size() == 1) {
741           continue;  // skip--an ancestor will link this one in from above
742         }
743         // This is the top-most dir that can be linked to a single root. Make it so.
744         Path root = roots.iterator().next();  // lone root in set
745         if (LOG_FINER) {
746           LOG.finer("ln -s " + root.getRelative(dir) + " " + linkRoot.getRelative(dir));
747         }
748         linkRoot.getRelative(dir).createSymbolicLink(root.getRelative(dir));
749       }
750     }
751     // Make links for dirs within packages, skip parent-only dirs.
752     for (Map.Entry<PathFragment, Set<Path>> entry : dirRootsMap.entrySet()) {
753       PathFragment dir = entry.getKey();
754       if (entry.getValue().size() > 1) {
755         // If this dir is at or below a package dir, link in its contents.
756         PathFragment pkgDir = longestPathPrefix(dir, packageRootMap.keySet());
757         if (pkgDir != null) {
758           Path root = packageRootMap.get(pkgDir);
759           try {
760             Path absdir = root.getRelative(dir);
761             if (absdir.isDirectory()) {
762               if (LOG_FINER) {
763                 LOG.finer("ln -s " + absdir + "/* " + linkRoot.getRelative(dir) + "/");
764               }
765               for (Path target : absdir.getDirectoryEntries()) {
766                 PathFragment p = target.relativeTo(root);
767                 if (!dirRootsMap.containsKey(p)) {
768                   //LOG.finest("ln -s " + target + " " + linkRoot.getRelative(p));
769                   linkRoot.getRelative(p).createSymbolicLink(target);
770                 }
771               }
772             } else {
773               LOG.fine("Symlink planting skipping dir '" + absdir + "'");
774             }
775           } catch (IOException e) {
776             e.printStackTrace();
777           }
778           // Otherwise its just an otherwise empty common parent dir.
779         }
780       }
781     }
782 
783     if (emptyPackagePath != null) {
784       // For the top-level directory, generate symlinks to everything in the directory instead of
785       // the directory itself.
786       for (Path target : emptyPackagePath.getDirectoryEntries()) {
787         String baseName = target.getBaseName();
788         // Create any links that don't exist yet and don't start with bazel-.
789         if (!baseName.startsWith(Constants.PRODUCT_NAME + "-")
790             && !linkRoot.getRelative(baseName).exists()) {
791           linkRoot.getRelative(baseName).createSymbolicLink(target);
792         }
793       }
794     }
795   }

802   private static char[] convertFromLatin1(byte[] content) {
803     char[] latin1 = new char[content.length];
804     for (int i = 0; i < latin1.length; i++) { // yeah, latin1 is this easy! :-)
805       latin1[i] = (char) (0xff & content[i]);
806     }
807     return latin1;
808   }
