14  package com.google.devtools.build.lib.syntax;
15  
16  import com.google.common.base.Preconditions;
17  
18  import java.util.List;
19  
20  import javax.annotation.Nullable;
21  
22  /**
23   * Syntax node for a function argument.
24   *
25   * <p>Argument is a base class for arguments passed in a call (@see Argument.Passed)
26   * or defined as part of a function definition (@see Parameter).
27   * It is notably used by some {@link Parser} and printer functions.
28   */
29  public abstract class Argument extends ASTNode {
30  
31    public boolean isStar() {
32      return false;
33    }
34  
35    public boolean isStarStar() {
36      return false;
37    }
38  
39    /**
40     * Argument.Passed is the class of arguments passed in a function call
41     * (as opposed to being used in a definition -- @see Parameter for that).
42     * Argument.Passed is usually what we mean when informally say "argument".
43     *
44     * <p>An Argument.Passed can be Positional, Keyword, Star, or StarStar.
45     */
46    public abstract static class Passed extends Argument {
47      /** the value to be passed by this argument */
48      protected final Expression value;
49  
50      private Passed(Expression value) {
51        this.value = Preconditions.checkNotNull(value);
52      }
53  
54      public boolean isPositional() {
55        return false;
56      }
57      public boolean isKeyword() {
58        return false;
59      }
60      @Nullable public String getName() { // only for keyword arguments
61        return null;
62      }
63      public Expression getValue() {
64        return value;
65      }
66      @Override
67      public void accept(SyntaxTreeVisitor visitor) {
68        visitor.visit(this);
69      }
70    }
71  
72    /** positional argument: Expression */
73    public static class Positional extends Passed {
74  
75      public Positional(Expression value) {
76        super(value);
77      }
78  
79      @Override public boolean isPositional() {
80        return true;
81      }
82      @Override
83      public String toString() {
84        return String.valueOf(value);
85      }
86    }
87  
88    /** keyword argument: K = Expression */
89    public static class Keyword extends Passed {
90  
91      final String name;
92  
93      public Keyword(String name, Expression value) {
94        super(value);
95        this.name = name;
96      }
97  
98      @Override public String getName() {
99        return name;
100     }
101     @Override public boolean isKeyword() {
102       return true;
103     }
104     @Override
105     public String toString() {
106       return name + " = " + value;
107     }
108   }
109 
110   /** positional rest (starred) argument: *Expression */
111   public static class Star extends Passed {
112 
113     public Star(Expression value) {
114       super(value);
115     }
116 
117     @Override public boolean isStar() {
118       return true;
119     }
120     @Override
121     public String toString() {
122       return "*" + value;
123     }
124   }
125 
126   /** keyword rest (star_starred) parameter: **Expression */
127   public static class StarStar extends Passed {
128 
129     public StarStar(Expression value) {
130       super(value);
131     }
132 
133     @Override public boolean isStarStar() {
134       return true;
135     }
136     @Override
137     public String toString() {
138       return "**" + value;
139     }
140   }
141 
142   /** Some arguments failed to satisfy python call convention strictures */
143   protected static class ArgumentException extends Exception {
144     /** construct an ArgumentException from a message only */
145     public ArgumentException(String message) {
146       super(message);
147     }
148   }
149 
150   /**
151    * Validate that the list of Argument's, whether gathered by the Parser or from annotations,
152    * satisfies the requirements of the Python calling conventions: all Positional's first,
153    * at most one Star, at most one StarStar, at the end only.
154    */
155   public static void validateFuncallArguments(List<Passed> arguments)
156       throws ArgumentException {
157     boolean hasNamed = false;
158     boolean hasStar = false;
159     boolean hasKwArg = false;
160     for (Passed arg : arguments) {
161       if (hasKwArg) {
162         throw new ArgumentException("argument after **kwargs");
163       }
164       if (arg.isPositional()) {
165         if (hasNamed) {
166           throw new ArgumentException("non-keyword arg after keyword arg");
167         } else if (arg.isStar()) {
168           throw new ArgumentException("only named arguments may follow *expression");
169         }
170       } else if (arg.isKeyword()) {
171         hasNamed = true;
172       } else if (arg.isStar()) {
173         if (hasStar) {
174           throw new ArgumentException("more than one *stararg");
175         }
176         hasStar = true;
177       } else {
178         hasKwArg = true;
179       }
180     }
181   }
