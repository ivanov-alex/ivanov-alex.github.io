40  /** Represents the collection of native libraries (.so) to be installed in the APK. */
41  public final class NativeLibs {
42    public static final NativeLibs EMPTY =
43        new NativeLibs(ImmutableMap.<String, Iterable<Artifact>>of(), null);
44  
45    public static NativeLibs fromPrecompiledObjects(
46        RuleContext ruleContext, Multimap<String, TransitiveInfoCollection> depsByArchitecture) {
47      ImmutableMap.Builder<String, Iterable<Artifact>> builder = ImmutableMap.builder();
48      for (Map.Entry<String, Collection<TransitiveInfoCollection>> entry :
49          depsByArchitecture.asMap().entrySet()) {
50        NestedSet<LinkerInput> nativeLibraries =
51            AndroidCommon.collectTransitiveNativeLibraries(entry.getValue());
52        builder.put(entry.getKey(), checkUniqueBaseNames(ruleContext, nativeLibraries));
53      }
54      return new NativeLibs(builder.build(), null);
55    }
56  
57    public static NativeLibs fromLinkedNativeDeps(
58        RuleContext ruleContext,
59        String nativeDepsFileName,
60        Multimap<String, TransitiveInfoCollection> depsByArchitecture,
61        Map<String, CcToolchainProvider> toolchainMap,
62        Map<String, BuildConfiguration> configurationMap) {
63      Map<String, Iterable<Artifact>> result = new LinkedHashMap<>();
64      for (Map.Entry<String, Collection<TransitiveInfoCollection>> entry :
65          depsByArchitecture.asMap().entrySet()) {
66        CcLinkParams linkParams = AndroidCommon.getCcLinkParamsStore(entry.getValue())
67            .get(/* linkingStatically */ true, /* linkShared */ true);
68        Artifact nativeDepsLibrary = NativeDepsHelper.maybeCreateAndroidNativeDepsAction(
69            ruleContext, linkParams, configurationMap.get(entry.getKey()),
70            toolchainMap.get(entry.getKey()));
71        if (nativeDepsLibrary != null) {
72          result.put(entry.getKey(), ImmutableList.of(nativeDepsLibrary));
73        }
74      }
75      if (result.isEmpty()) {
76        return NativeLibs.EMPTY;
77      } else {
78        Artifact anyNativeLibrary =
79            result.entrySet().iterator().next().getValue().iterator().next();
80        // The native deps name file must be the only file in its directory because ApkBuilder does
81        // not have an option to add a particular file to the .apk, only one to add every file in a
82        // particular directory.
83        Artifact nativeDepsName = ruleContext.getAnalysisEnvironment().getDerivedArtifact(
84            ruleContext.getUniqueDirectory("nativedeps_filename").getRelative(nativeDepsFileName),
85            ruleContext.getBinOrGenfilesDirectory());
86        ruleContext.registerAction(new FileWriteAction(ruleContext.getActionOwner(), nativeDepsName,
87            anyNativeLibrary.getExecPath().getBaseName(), false));
88  
89        return new NativeLibs(ImmutableMap.copyOf(result), nativeDepsName);
90      }
91    }
92  
93    // Map from architecture (CPU folder to place the library in) to libraries for that CPU
94    private final Map<String, Iterable<Artifact>> nativeLibs;
95    private final Artifact nativeLibsName;
96  
97    private NativeLibs(Map<String, Iterable<Artifact>> nativeLibs, Artifact nativeLibsName) {
98      this.nativeLibs = nativeLibs;
99      this.nativeLibsName = nativeLibsName;
100   }
101 
102   /**
103    * Returns a map from the name of the architecture (CPU folder to place the library in) to the
104    * set of libraries for that architecture.
105    */
106   public Map<String, Iterable<Artifact>> getMap() {
107     return nativeLibs;
108   }
109 
110   public ImmutableList<Artifact> createApkBuilderSymlinks(RuleContext ruleContext) {
111       ImmutableList.Builder<Artifact> result = ImmutableList.builder();
112     for (Map.Entry<String, Iterable<Artifact>> entry : nativeLibs.entrySet()) {
113       String arch = entry.getKey();
114       for (Artifact lib : entry.getValue()) {
115         Artifact symlink = AndroidBinary.getDxArtifact(ruleContext,
116             "native_symlinks/" + arch + "/" + lib.getExecPath().getBaseName());
117         ruleContext.registerAction(new SymlinkAction(
118             ruleContext.getActionOwner(), lib, symlink,
119             "Symlinking Android native library for " + ruleContext.getLabel()));
120         result.add(symlink);
121       }
122     }
123 
124     return result.build();
125   }
126 
127   /**
128    * Returns the artifact containing the names of the native libraries or null if it does not exist.
129    *
130    * <p>This artifact will be put in the root directory of the .apk and can be used to load the
131    * libraries programmatically without knowing their names.
132    */
133   @Nullable
134   public Artifact getName() {
135     return nativeLibsName;
136   }
137 
138   private static Iterable<Artifact> checkUniqueBaseNames(
139       RuleContext ruleContext, NestedSet<LinkerInput> libraries) {
140     Map<String, Artifact> basenames = new HashMap<>();
141     Set<Artifact> artifacts = new HashSet<>();
142     for (LinkerInput linkerInput : libraries) {
143       Artifact artifact = linkerInput.getOriginalLibraryArtifact();
144       if (!artifacts.add(artifact)) {
145         // We have already reached this library, e.g., through a different solib symlink.
146         continue;
147       }
148       String basename = artifact.getExecPath().getBaseName();
149       Artifact oldArtifact = basenames.put(basename, artifact);
150       if (oldArtifact != null) {
151         // Without linking, there may be name collisions in the libraries which were provided, so
152         // check for this at this step.
153         ruleContext.ruleError("Each library in the transitive closure must have a unique "
154             + "basename, but two libraries had the basename '" + basename + "': "
155             + artifact.prettyPrint() + " and " + oldArtifact.prettyPrint());
156       }
157     }
158     return artifacts;
159   }
