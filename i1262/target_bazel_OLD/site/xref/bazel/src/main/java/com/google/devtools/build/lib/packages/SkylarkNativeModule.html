15  package com.google.devtools.build.lib.packages;

17  import com.google.devtools.build.lib.packages.Type.ConversionException;
18  import com.google.devtools.build.lib.syntax.BuiltinFunction;
19  import com.google.devtools.build.lib.syntax.Environment;
20  import com.google.devtools.build.lib.syntax.EvalException;
21  import com.google.devtools.build.lib.syntax.FuncallExpression;
22  import com.google.devtools.build.lib.syntax.GlobList;
23  import com.google.devtools.build.lib.syntax.SkylarkList;
24  import com.google.devtools.build.lib.syntax.SkylarkModule;
25  import com.google.devtools.build.lib.syntax.SkylarkSignature;
26  import com.google.devtools.build.lib.syntax.SkylarkSignature.Param;
27  import com.google.devtools.build.lib.syntax.SkylarkSignatureProcessor;
28  
29  /**
30   * A class for the Skylark native module.
31   */
32  @SkylarkModule(name = "native", namespace = true, onlyLoadingPhase = true, doc =
33        "A built-in module to support native rules and other package helper functions. "
34      + "All native rules appear as functions in this module. Note that the native module is only "
35      + "available in the loading phase (i.e. for macros, not for rule implementations).<br/>"
36      + "Extra helper functions:")
37  public class SkylarkNativeModule {
38  
39    // TODO(bazel-team): shouldn't we return a SkylarkList instead?
40    @SkylarkSignature(name = "glob", objectType = SkylarkNativeModule.class,
41        returnType = GlobList.class,
42        doc = "Glob returns a list of every file in the current package that:<ul>\n"
43            + "<li>Matches at least one pattern in <code>include</code>.</li>\n"
44            + "<li>Does not match any of the patterns in <code>exclude</code> "
45            + "(default <code>[]</code>).</li></ul>\n"
46            + "If the <code>exclude_directories</code> argument is enabled (set to <code>1</code>), "
47            + "files of type directory will be omitted from the results (default <code>1</code>).",
48        mandatoryPositionals = {
49        @Param(name = "include", type = SkylarkList.class, generic1 = String.class,
50            defaultValue = "[]", doc = "The list of glob patterns to include.")},
51        optionalPositionals = {
52        @Param(name = "exclude", type = SkylarkList.class, generic1 = String.class,
53            defaultValue = "[]", doc = "The list of glob patterns to exclude."),
54        @Param(name = "excludes", type = SkylarkList.class, generic1 = String.class,
55            defaultValue = "[]", doc = "The list of glob patterns to exclude."),
56        // TODO(bazel-team): accept booleans as well as integers? (and eventually migrate?)
57        @Param(name = "exclude_directories", type = Integer.class, defaultValue = "1",
58            doc = "A flag whether to exclude directories or not.")},
59        useAst = true, useEnvironment = true)
60    private static final BuiltinFunction glob = new BuiltinFunction("glob") {
61        public GlobList<String> invoke(
62            SkylarkList include, SkylarkList exclude, SkylarkList excludes,
63            Integer excludeDirectories, FuncallExpression ast, Environment env)
64            throws EvalException, ConversionException, InterruptedException {
65          // TODO(bazel-team): Remove 'excludes' argument in July 2015.
66          if (exclude.size() == 0) {
67            exclude = excludes;
68          }
69          return PackageFactory.callGlob(
70              null, false, include, exclude, excludeDirectories != 0, ast, env);
71      }
72    };
73  
74    @SkylarkSignature(name = "package_group", objectType = SkylarkNativeModule.class,
75        returnType = Environment.NoneType.class,
76        doc = "This function defines a set of packages and assigns a label to the group. "
77            + "The label can be referenced in <code>visibility</code> attributes.",
78        mandatoryNamedOnly = {
79        @Param(name = "name", type = String.class,
80            doc = "The unique name for this rule.")},
81        optionalNamedOnly = {
82        @Param(name = "packages", type = SkylarkList.class, generic1 = String.class,
83            defaultValue = "[]",
84            doc = "A complete enumeration of packages in this group."),
85        @Param(name = "includes", type = SkylarkList.class, generic1 = String.class,
86            defaultValue = "[]",
87            doc = "Other package groups that are included in this one.")},
88        useAst = true, useEnvironment = true)
89    private static final BuiltinFunction packageGroup = new BuiltinFunction("package_group") {
90        public Environment.NoneType invoke(String name, SkylarkList packages, SkylarkList includes,
91                  FuncallExpression ast, Environment env) throws EvalException, ConversionException {
92          return PackageFactory.callPackageFunction(name, packages, includes, ast, env);
93        }
94      };
95  
96    @SkylarkSignature(name = "exports_files", objectType = SkylarkNativeModule.class,
97        returnType = Environment.NoneType.class,
98        doc = "Specifies a list of files belonging to this package that are exported to other "
99            + "packages but not otherwise mentioned.",
100       mandatoryPositionals = {
101       @Param(name = "srcs", type = SkylarkList.class, generic1 = String.class,
102           doc = "The list of files to export.")},
103       optionalPositionals = {
104       // TODO(bazel-team): make it possible to express the precise type ListOf(LabelDesignator)
105       @Param(name = "visibility", type = SkylarkList.class,
106           noneable = true,
107           doc = "A visibility declaration can to be specified. The files will be visible to the "
108               + "targets specified. If no visibility is specified, the files will be visible to "
109               + "every package."),
110       @Param(name = "licenses", type = SkylarkList.class, generic1 = String.class, noneable = true,
111           doc = "Licenses to be specified.")},
112       useAst = true, useEnvironment = true)
113   private static final BuiltinFunction exportsFiles = new BuiltinFunction("exports_files") {
114       public Environment.NoneType invoke(SkylarkList srcs, Object visibility, Object licenses,
115           FuncallExpression ast, Environment env)
116           throws EvalException, ConversionException {
117         return PackageFactory.callExportsFiles(srcs, visibility, licenses, ast, env);
118       }
119     };
120 
121   public static final SkylarkNativeModule NATIVE_MODULE = new SkylarkNativeModule();
122 
123   static {
124     SkylarkSignatureProcessor.configureSkylarkFunctions(SkylarkNativeModule.class);
125   }
126 }
