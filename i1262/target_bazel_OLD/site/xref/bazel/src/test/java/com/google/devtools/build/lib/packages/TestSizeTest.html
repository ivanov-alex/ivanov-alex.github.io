14  package com.google.devtools.build.lib.packages;
15  
16  import static org.junit.Assert.assertEquals;
17  import static org.junit.Assert.assertNull;
18  
19  import org.junit.Test;
20  import org.junit.runner.RunWith;
21  import org.junit.runners.JUnit4;
22  
23  /**
24   * Tests the various methods of {@link TestSize}
25   */
26  @RunWith(JUnit4.class)
27  public class TestSizeTest {
28  
29    @Test
30    public void testBasicConversion() {
31      assertEquals(TestSize.valueOf("SMALL"), TestSize.SMALL);
32      assertEquals(TestSize.valueOf("MEDIUM"), TestSize.MEDIUM);
33      assertEquals(TestSize.valueOf("LARGE"), TestSize.LARGE);
34      assertEquals(TestSize.valueOf("ENORMOUS"), TestSize.ENORMOUS);
35    }
36  
37    @Test
38    public void testGetDefaultTimeout() {
39      assertEquals(TestTimeout.SHORT, TestSize.SMALL.getDefaultTimeout());
40      assertEquals(TestTimeout.MODERATE, TestSize.MEDIUM.getDefaultTimeout());
41      assertEquals(TestTimeout.LONG, TestSize.LARGE.getDefaultTimeout());
42      assertEquals(TestTimeout.ETERNAL, TestSize.ENORMOUS.getDefaultTimeout());
43    }
44  
45    @Test
46    public void testGetDefaultShards() {
47      assertEquals(2, TestSize.SMALL.getDefaultShards());
48      assertEquals(10, TestSize.MEDIUM.getDefaultShards());
49      assertEquals(20, TestSize.LARGE.getDefaultShards());
50      assertEquals(30, TestSize.ENORMOUS.getDefaultShards());
51    }
52  
53    @Test
54    public void testGetTestSizeFromString() {
55      assertNull(TestSize.getTestSize("Small"));
56      assertNull(TestSize.getTestSize("Koala"));
57      assertEquals(TestSize.SMALL, TestSize.getTestSize("small"));
58      assertEquals(TestSize.MEDIUM, TestSize.getTestSize("medium"));
59      assertEquals(TestSize.LARGE, TestSize.getTestSize("large"));
60      assertEquals(TestSize.ENORMOUS, TestSize.getTestSize("enormous"));
61    }
62  
63    @Test
64    public void testGetTestSizeFromDefaultTimeout() {
65      assertEquals(TestSize.SMALL, TestSize.getTestSize(TestTimeout.SHORT));
66      assertEquals(TestSize.MEDIUM, TestSize.getTestSize(TestTimeout.MODERATE));
67      assertEquals(TestSize.LARGE, TestSize.getTestSize(TestTimeout.LONG));
68      assertEquals(TestSize.ENORMOUS, TestSize.getTestSize(TestTimeout.ETERNAL));
69    }
70  }
