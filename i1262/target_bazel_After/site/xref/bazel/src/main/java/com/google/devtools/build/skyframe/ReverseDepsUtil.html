28  /**
29   * A utility class that allows us to keep the reverse dependencies as an array list instead of a
30   * set. This is more memory-efficient. At the same time it allows us to group the removals and
31   * uniqueness checks so that it also performs well.
32   *
33   * <p>The reason of this class it to share non-trivial code between BuildingState and NodeEntry. We
34   * could simply make those two classes extend this class instead, but we would be less
35   * memory-efficient since object memory alignment does not cross classes ( you would have two memory
36   * alignments, one for the base class and one for the extended one).
37   */
38  abstract class ReverseDepsUtil<T> {
39  
40    static final int MAYBE_CHECK_THRESHOLD = 10;
41  
42    abstract void setReverseDepsObject(T container, Object object);
43  
44    abstract void setSingleReverseDep(T container, boolean singleObject);
45  
46    abstract void setReverseDepsToRemove(T container, List<SkyKey> object);
47  
48    abstract Object getReverseDepsObject(T container);
49  
50    abstract boolean isSingleReverseDep(T container);
51  
52    abstract List<SkyKey> getReverseDepsToRemove(T container);
53  
54    /**
55     * We check that the reverse dependency is not already present. We only do that if reverseDeps is
56     * small, so that it does not impact performance.
57     */
58    void maybeCheckReverseDepNotPresent(T container, SkyKey reverseDep) {
59      if (isSingleReverseDep(container)) {
60        Preconditions.checkState(!getReverseDepsObject(container).equals(reverseDep),
61            "Reverse dep %s already present", reverseDep);
62        return;
63      }
64      @SuppressWarnings("unchecked")
65      List<SkyKey> asList = (List<SkyKey>) getReverseDepsObject(container);
66      if (asList.size() < MAYBE_CHECK_THRESHOLD) {
67        Preconditions.checkState(!asList.contains(reverseDep), "Reverse dep %s already present"
68            + " in %s", reverseDep, asList);
69      }
70    }
71  
72    /**
73     * We use a memory-efficient trick to keep reverseDeps memory usage low. Edges in Bazel are
74     * dominant over the number of nodes.
75     *
76     * <p>Most of the nodes have zero or one reverse dep. That is why we use immutable versions of the
77     * lists for those cases. In case of the size being > 1 we switch to an ArrayList. That is because
78     * we also have a decent number of nodes for which the reverseDeps are huge (for example almost
79     * everything depends on BuildInfo node).
80     *
81     * <p>We also optimize for the case where we have only one dependency. In that case we keep the
82     * object directly instead of a wrapper list.
83     */
84    @SuppressWarnings("unchecked")
85    void addReverseDeps(T container, Collection<SkyKey> newReverseDeps) {
86      if (newReverseDeps.isEmpty()) {
87        return;
88      }
89      Object reverseDeps = getReverseDepsObject(container);
90      int reverseDepsSize = isSingleReverseDep(container) ? 1 : ((List<SkyKey>) reverseDeps).size();
91      int newSize = reverseDepsSize + newReverseDeps.size();
92      if (newSize == 1) {
93        overwriteReverseDepsWithObject(container, Iterables.getOnlyElement(newReverseDeps));
94      } else if (reverseDepsSize == 0) {
95        overwriteReverseDepsList(container, Lists.newArrayList(newReverseDeps));
96      } else if (reverseDepsSize == 1) {
97        List<SkyKey> newList = Lists.newArrayListWithExpectedSize(newSize);
98        newList.add((SkyKey) reverseDeps);
99        newList.addAll(newReverseDeps);
100       overwriteReverseDepsList(container, newList);
101     } else {
102       ((List<SkyKey>) reverseDeps).addAll(newReverseDeps);
103     }
104   }
105 
106   /**
107    * See {@code addReverseDeps} method.
108    */
109   void removeReverseDep(T container, SkyKey reverseDep) {
110     if (isSingleReverseDep(container)) {
111       // This removal is cheap so let's do it and not keep it in reverseDepsToRemove.
112       // !equals should only happen in case of catastrophe.
113       if (getReverseDepsObject(container).equals(reverseDep)) {
114         overwriteReverseDepsList(container, ImmutableList.<SkyKey>of());
115       }
116       return;
117     }
118     @SuppressWarnings("unchecked")
119     List<SkyKey> reverseDepsAsList = (List<SkyKey>) getReverseDepsObject(container);
120     if (reverseDepsAsList.isEmpty()) {
121       return;
122     }
123     List<SkyKey> reverseDepsToRemove = getReverseDepsToRemove(container);
124     if (reverseDepsToRemove == null) {
125       reverseDepsToRemove = Lists.newArrayListWithExpectedSize(1);
126       setReverseDepsToRemove(container, reverseDepsToRemove);
127     }
128     reverseDepsToRemove.add(reverseDep);
129   }
130 
131   ImmutableSet<SkyKey> getReverseDeps(T container) {
132     consolidateReverseDepsRemovals(container);
133 
134     // TODO(bazel-team): Unfortunately, we need to make a copy here right now to be on the safe side
135     // wrt. thread-safety. The parents of a node get modified when any of the parents is deleted,
136     // and we can't handle that right now.
137     if (isSingleReverseDep(container)) {
138       return ImmutableSet.of((SkyKey) getReverseDepsObject(container));
139     } else {
140       @SuppressWarnings("unchecked")
141       List<SkyKey> reverseDeps = (List<SkyKey>) getReverseDepsObject(container);
142       ImmutableSet<SkyKey> set = ImmutableSet.copyOf(reverseDeps);
143       Preconditions.checkState(set.size() == reverseDeps.size(),
144           "Duplicate reverse deps present in %s: %s. %s", this, reverseDeps, container);
145       return set;
146     }
147   }
148 
149   void consolidateReverseDepsRemovals(T container) {
150     List<SkyKey> reverseDepsToRemove = getReverseDepsToRemove(container);
151     Object reverseDeps = getReverseDepsObject(container);
152     if (reverseDepsToRemove == null) {
153       return;
154     }
155     Preconditions.checkState(!isSingleReverseDep(container),
156         "We do not use reverseDepsToRemove for single lists: %s", container);
157     // Should not happen, as we only create reverseDepsToRemove in case we have at least one
158     // reverse dep to remove.
159     Preconditions.checkState((!((List<?>) reverseDeps).isEmpty()),
160         "Could not remove %s elements from %s.\nReverse deps to remove: %s. %s",
161         reverseDepsToRemove.size(), reverseDeps, reverseDepsToRemove, container);
162 
163     Set<SkyKey> toRemove = Sets.newHashSet(reverseDepsToRemove);
164     int expectedRemovals = toRemove.size();
165     Preconditions.checkState(expectedRemovals == reverseDepsToRemove.size(),
166         "A reverse dependency tried to remove itself twice: %s. %s", reverseDepsToRemove,
167         container);
168 
169     @SuppressWarnings("unchecked")
170     List<SkyKey> reverseDepsAsList = (List<SkyKey>) reverseDeps;
171     List<SkyKey> newReverseDeps = Lists
172         .newArrayListWithExpectedSize(Math.max(0, reverseDepsAsList.size() - expectedRemovals));
173 
174     for (SkyKey reverseDep : reverseDepsAsList) {
175       if (!toRemove.contains(reverseDep)) {
176         newReverseDeps.add(reverseDep);
177       }
178     }
179     Preconditions.checkState(newReverseDeps.size() == reverseDepsAsList.size() - expectedRemovals,
180         "Could not remove some elements from %s.\nReverse deps to remove: %s. %s", reverseDeps,
181         toRemove, container);
182 
183     if (newReverseDeps.isEmpty()) {
184       overwriteReverseDepsList(container, ImmutableList.<SkyKey>of());
185     } else if (newReverseDeps.size() == 1) {
186       overwriteReverseDepsWithObject(container, newReverseDeps.get(0));
187     } else {
188       overwriteReverseDepsList(container, newReverseDeps);
189     }
190     setReverseDepsToRemove(container, null);
191   }
192 
193   String toString(T container) {
194     return MoreObjects.toStringHelper("ReverseDeps")
195         .add("reverseDeps", getReverseDepsObject(container))
196         .add("singleReverseDep", isSingleReverseDep(container))
197         .add("reverseDepsToRemove", getReverseDepsToRemove(container))
198         .toString();
199   }
200 
201   private void overwriteReverseDepsWithObject(T container, SkyKey newObject) {
202     setReverseDepsObject(container, newObject);
203     setSingleReverseDep(container, true);
204   }
205 
206   private void overwriteReverseDepsList(T container, List<SkyKey> list) {
207     setReverseDepsObject(container, list);
208     setSingleReverseDep(container, false);
209   }
