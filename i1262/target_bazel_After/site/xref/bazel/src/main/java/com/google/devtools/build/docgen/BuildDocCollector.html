34  /**
35   * Class that parses the documentation fragments of rule-classes and
36   * generates the html format documentation.
37   */
38  class BuildDocCollector {
39    private ConfiguredRuleClassProvider ruleClassProvider;
40    private boolean printMessages;
41  
42    public BuildDocCollector(ConfiguredRuleClassProvider ruleClassProvider,
43        boolean printMessages) {
44      this.ruleClassProvider = ruleClassProvider;
45      this.printMessages = printMessages;
46    }
47  
48    /**
49     * Collects all the rule and attribute documentation present in inputDirs, integrates the
50     * attribute documentation in the rule documentation and returns the rule documentation.
51     */
52    public Map<String, RuleDocumentation> collect(String[] inputDirs)
53        throws BuildEncyclopediaDocException, IOException {
54      // RuleDocumentations are generated in order (based on rule type then alphabetically).
55      // The ordering is also used to determine in which rule doc the common attribute docs are
56      // generated (they are generated at the first appearance).
57      Map<String, RuleDocumentation> ruleDocEntries = new TreeMap<>();
58      // RuleDocumentationAttribute objects equal based on attributeName so they have to be
59      // collected in a List instead of a Set.
60      ListMultimap<String, RuleDocumentationAttribute> attributeDocEntries =
61          LinkedListMultimap.create();
62  
63      // Map of rule class name to file that defined it.
64      Map<String, File> ruleClassFiles = new HashMap<>();
65  
66      // Set of files already processed. The same file may be encountered multiple times because
67      // directories are processed recursively, and an input directory may be a subdirectory of
68      // another one.
69      Set<File> processedFiles = new HashSet<>();
70  
71      for (String inputDir : inputDirs) {
72        if (printMessages) {
73          System.out.println(" Processing input directory: " + inputDir);
74        }
75        int ruleNum = ruleDocEntries.size();
76        collectDocs(processedFiles, ruleClassFiles, ruleDocEntries, attributeDocEntries,
77            new File(inputDir));
78        if (printMessages) {
79          System.out.println(" " + (ruleDocEntries.size() - ruleNum)
80              + " rule documentations found.");
81        }
82      }
83  
84      processAttributeDocs(ruleDocEntries.values(), attributeDocEntries);
85      return ruleDocEntries;
86    }
87  
88    /**
89     * Go through all attributes of all documented rules and search the best attribute documentation
90     * if exists. The best documentation is the closest documentation in the ancestor graph. E.g. if
91     * java_library.deps documented in $rule and $java_rule then the one in $java_rule is going to
92     * apply since it's a closer ancestor of java_library.
93     */
94    private void processAttributeDocs(Iterable<RuleDocumentation> ruleDocEntries,
95        ListMultimap<String, RuleDocumentationAttribute> attributeDocEntries)
96            throws BuildEncyclopediaDocException {
97      for (RuleDocumentation ruleDoc : ruleDocEntries) {
98        RuleClass ruleClass = ruleClassProvider.getRuleClassMap().get(ruleDoc.getRuleName());
99        if (ruleClass != null) {
100         if (ruleClass.isDocumented()) {
101           Class<? extends RuleDefinition> ruleDefinition =
102               ruleClassProvider.getRuleClassDefinition(ruleDoc.getRuleName());
103           for (Attribute attribute : ruleClass.getAttributes()) {
104             String attrName = attribute.getName();
105             List<RuleDocumentationAttribute> attributeDocList =
106                 attributeDocEntries.get(attrName);
107             if (attributeDocList != null) {
108               // There are attribute docs for this attribute.
109               // Search the closest one in the ancestor graph.
110               // Note that there can be only one 'closest' attribute since we forbid multiple
111               // inheritance of the same attribute in RuleClass.
112               int minLevel = Integer.MAX_VALUE;
113               RuleDocumentationAttribute bestAttributeDoc = null;
114               for (RuleDocumentationAttribute attributeDoc : attributeDocList) {
115                 int level = attributeDoc.getDefinitionClassAncestryLevel(ruleDefinition);
116                 if (level >= 0 && level < minLevel) {
117                   bestAttributeDoc = attributeDoc;
118                   minLevel = level;
119                 }
120               }
121               if (bestAttributeDoc != null) {
122                 ruleDoc.addAttribute(bestAttributeDoc);
123               // If there is no matching attribute doc try to add the common.
124               } else if (ruleDoc.getRuleType().equals(RuleType.BINARY)
125                   && PredefinedAttributes.BINARY_ATTRIBUTES.containsKey(attrName)) {
126                 ruleDoc.addAttribute(PredefinedAttributes.BINARY_ATTRIBUTES.get(attrName));
127               } else if (ruleDoc.getRuleType().equals(RuleType.TEST)
128                   && PredefinedAttributes.TEST_ATTRIBUTES.containsKey(attrName)) {
129                 ruleDoc.addAttribute(PredefinedAttributes.TEST_ATTRIBUTES.get(attrName));
130               } else if (PredefinedAttributes.COMMON_ATTRIBUTES.containsKey(attrName)) {
131                 ruleDoc.addAttribute(PredefinedAttributes.COMMON_ATTRIBUTES.get(attrName));
132               }
133             }
134           }
135         }
136       } else {
137         throw ruleDoc.createException("Can't find RuleClass for " + ruleDoc.getRuleName());
138       }
139     }
140   }
141 
142   /**
143    * Goes through all the html files and subdirs under inputPath and collects the rule
144    * and attribute documentations using the ruleDocEntries and attributeDocEntries variable.
145    */
146   public void collectDocs(
147       Set<File> processedFiles,
148       Map<String, File> ruleClassFiles,
149       Map<String, RuleDocumentation> ruleDocEntries,
150       ListMultimap<String, RuleDocumentationAttribute> attributeDocEntries,
151       File inputPath) throws BuildEncyclopediaDocException, IOException {
152     if (processedFiles.contains(inputPath)) {
153       return;
154     }
155 
156     if (inputPath.isFile()) {
157       if (DocgenConsts.JAVA_SOURCE_FILE_SUFFIX.apply(inputPath.getName())) {
158         SourceFileReader sfr = new SourceFileReader(
159             ruleClassProvider, inputPath.getAbsolutePath());
160         sfr.readDocsFromComments();
161         for (RuleDocumentation d : sfr.getRuleDocEntries()) {
162           String ruleName = d.getRuleName();
163           if (ruleDocEntries.containsKey(ruleName)
164               && !ruleClassFiles.get(ruleName).equals(inputPath)) {
165             System.err.printf("WARNING: '%s' from '%s' overrides value already in map from '%s'\n",
166                 d.getRuleName(), inputPath, ruleClassFiles.get(ruleName));
167           }
168           ruleClassFiles.put(ruleName, inputPath);
169           ruleDocEntries.put(ruleName, d);
170         }
171         if (attributeDocEntries != null) {
172           // Collect all attribute documentations from this file.
173           attributeDocEntries.putAll(sfr.getAttributeDocEntries());
174         }
175       }
176     } else if (inputPath.isDirectory()) {
177       for (File childPath : inputPath.listFiles()) {
178         collectDocs(processedFiles, ruleClassFiles, ruleDocEntries, attributeDocEntries, childPath);
179       }
180     }
181 
182     processedFiles.add(inputPath);
183   }
