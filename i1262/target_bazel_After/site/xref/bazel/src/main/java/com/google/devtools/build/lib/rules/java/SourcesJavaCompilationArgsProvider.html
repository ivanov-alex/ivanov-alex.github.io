15  package com.google.devtools.build.lib.rules.java;
16  
17  import com.google.devtools.build.lib.analysis.TransitiveInfoProvider;
18  import com.google.devtools.build.lib.concurrent.ThreadSafety.Immutable;
19  
20  /**
21   * An interface that marks configured targets that can provide Java compilation arguments through
22   * the 'srcs' attribute of Java rules.
23   *
24   * <p>In a perfect world, this would not be necessary for a million reasons, but
25   * this world is far from perfect, thus, we need this.
26   *
27   * <p>Please do not implement this interface with configured target implementations.
28   */
29  @Immutable
30  public final class SourcesJavaCompilationArgsProvider implements TransitiveInfoProvider {
31    private final JavaCompilationArgs javaCompilationArgs;
32    private final JavaCompilationArgs recursiveJavaCompilationArgs;
33  
34    public SourcesJavaCompilationArgsProvider(
35        JavaCompilationArgs javaCompilationArgs,
36        JavaCompilationArgs recursiveJavaCompilationArgs) {
37      this.javaCompilationArgs = javaCompilationArgs;
38      this.recursiveJavaCompilationArgs = recursiveJavaCompilationArgs;
39    }
40  
41    /**
42     * Returns non-recursively collected Java compilation information for
43     * building this target (called when strict_java_deps = 1).
44     *
45     * <p>Note that some of the parameters are still collected from the complete
46     * transitive closure. The non-recursive collection applies mainly to
47     * compile-time jars.
48     */
49    public JavaCompilationArgs getJavaCompilationArgs() {
50      return javaCompilationArgs;
51    }
52  
53    /**
54     * Returns recursively collected Java compilation information for building
55     * this target (called when strict_java_deps = 0).
56     */
57    public JavaCompilationArgs getRecursiveJavaCompilationArgs() {
58      return recursiveJavaCompilationArgs;
59    }
60  }
