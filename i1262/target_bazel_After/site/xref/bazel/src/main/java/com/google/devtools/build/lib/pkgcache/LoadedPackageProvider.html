14  package com.google.devtools.build.lib.pkgcache;
15  
16  import com.google.devtools.build.lib.packages.NoSuchPackageException;
17  import com.google.devtools.build.lib.packages.NoSuchTargetException;
18  import com.google.devtools.build.lib.packages.Package;
19  import com.google.devtools.build.lib.packages.PackageIdentifier;
20  import com.google.devtools.build.lib.packages.Target;
21  import com.google.devtools.build.lib.syntax.Label;
22  
23  /**
24   * Read-only API for retrieving packages, i.e., calling this API should not result in packages being
25   * loaded.
26   *
27   * <p><b>Concurrency</b>: Implementations should be thread-safe.
28   */
29  // TODO(bazel-team): Skyframe doesn't really implement this - can we remove it?
30  public interface LoadedPackageProvider {
31  
32    /**
33     * Returns a package if it was recently loaded, i.e., since the most recent cache sync. This
34     * throws an exception if the package was not loaded, even if it exists on disk.
35     */
36    Package getLoadedPackage(PackageIdentifier packageIdentifier) throws NoSuchPackageException;
37  
38    /**
39     * Returns a target if it was recently loaded, i.e., since the most recent cache sync. This
40     * throws an exception if the target was not loaded or not validated, even if it exists in the
41     * surrounding package.
42     */
43    Target getLoadedTarget(Label label) throws NoSuchPackageException, NoSuchTargetException;
44  
45    /**
46     * Returns true iff the specified target is current, i.e. a request for its label using {@link
47     * #getLoadedTarget} would return the same target instance.
48     */
49    boolean isTargetCurrent(Target target);
50  }
