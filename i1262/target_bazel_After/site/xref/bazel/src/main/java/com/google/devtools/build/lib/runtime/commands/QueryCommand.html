47  /**
48   * Command line wrapper for executing a query with blaze.
49   */
50  @Command(name = "query",
51           options = { PackageCacheOptions.class,
52                       QueryOptions.class },
53           help = "resource:query.txt",
54           shortDescription = "Executes a dependency graph query.",
55           allowResidue = true,
56           binaryStdOut = true,
57           completion = "label",
58           canRunInOutputDirectory = true)
59  public final class QueryCommand implements BlazeCommand {
60  
61    @Override
62    public void editOptions(BlazeRuntime runtime, OptionsParser optionsParser) { }
63  
64    /**
65     * Exit codes:
66     *   0   on successful evaluation.
67     *   1   if query evaluation did not complete.
68     *   2   if query parsing failed.
69     *   3   if errors were reported but evaluation produced a partial result
70     *        (only when --keep_going is in effect.)
71     */
72    @Override
73    public ExitCode exec(BlazeRuntime runtime, OptionsProvider options) {
74      QueryOptions queryOptions = options.getOptions(QueryOptions.class);
75  
76      try {
77        runtime.setupPackageCache(
78            options.getOptions(PackageCacheOptions.class),
79            runtime.getDefaultsPackageContent());
80      } catch (InterruptedException e) {
81        runtime.getReporter().handle(Event.error("query interrupted"));
82        return ExitCode.INTERRUPTED;
83      } catch (AbruptExitException e) {
84        runtime.getReporter().handle(Event.error(null, "Unknown error: " + e.getMessage()));
85        return e.getExitCode();
86      }
87  
88      if (options.getResidue().isEmpty()) {
89        runtime.getReporter().handle(Event.error(String.format(
90            "missing query expression. Type '%s help query' for syntax and help",
91            Constants.PRODUCT_NAME)));
92        return ExitCode.COMMAND_LINE_ERROR;
93      }
94  
95      Iterable<OutputFormatter> formatters = runtime.getQueryOutputFormatters();
96      OutputFormatter formatter =
97          OutputFormatter.getFormatter(formatters, queryOptions.outputFormat);
98      if (formatter == null) {
99        runtime.getReporter().handle(Event.error(
100           String.format("Invalid output format '%s'. Valid values are: %s",
101               queryOptions.outputFormat, OutputFormatter.formatterNames(formatters))));
102       return ExitCode.COMMAND_LINE_ERROR;
103     }
104 
105     String query = Joiner.on(' ').join(options.getResidue());
106 
107     Set<Setting> settings = queryOptions.toSettings();
108     AbstractBlazeQueryEnvironment<Target> env = newQueryEnvironment(
109         runtime,
110         queryOptions.keepGoing,
111         QueryOutputUtils.orderResults(queryOptions, formatter),
112         queryOptions.universeScope, queryOptions.loadingPhaseThreads,
113         settings);
114 
115     // 1. Parse query:
116     QueryExpression expr;
117     try {
118       expr = QueryExpression.parse(query, env);
119     } catch (QueryException e) {
120       runtime.getReporter().handle(Event.error(
121           null, "Error while parsing '" + query + "': " + e.getMessage()));
122       return ExitCode.COMMAND_LINE_ERROR;
123     }
124 
125     // 2. Evaluate expression:
126     QueryEvalResult<Target> result;
127     try {
128       result = env.evaluateQuery(expr);
129     } catch (QueryException | InterruptedException e) {
130       // Keep consistent with reportBuildFileError()
131       runtime
132           .getReporter()
133           // TODO(bazel-team): this is a kludge to fix a bug observed in the wild. We should make
134           // sure no null error messages ever get in.
135           .handle(Event.error(e.getMessage() == null ? e.toString() : e.getMessage()));
136       return ExitCode.ANALYSIS_FAILURE;
137     }
138 
139     runtime.getReporter().switchToAnsiAllowingHandler();
140     // 3. Output results:
141     PrintStream output = new PrintStream(runtime.getReporter().getOutErr().getOutputStream());
142     try {
143       QueryOutputUtils.output(queryOptions, result, formatter, output,
144           queryOptions.aspectDeps.createResolver(
145               runtime.getPackageManager(), runtime.getReporter()));
146     } catch (ClosedByInterruptException | InterruptedException e) {
147       runtime.getReporter().handle(Event.error("query interrupted"));
148       return ExitCode.INTERRUPTED;
149     } catch (IOException e) {
150       runtime.getReporter().handle(Event.error("I/O error: " + e.getMessage()));
151       return ExitCode.LOCAL_ENVIRONMENTAL_ERROR;
152     } finally {
153       output.flush();
154     }
155     if (result.getResultSet().isEmpty()) {
156       runtime.getReporter().handle(Event.info("Empty results"));
157     }
158 
159     return result.getSuccess() ? ExitCode.SUCCESS : ExitCode.PARTIAL_ANALYSIS_FAILURE;
160   }

162   @VisibleForTesting // for com.google.devtools.deps.gquery.test.QueryResultTestUtil
163   public static AbstractBlazeQueryEnvironment<Target> newQueryEnvironment(BlazeRuntime runtime,
164       boolean keepGoing, boolean orderedResults, int loadingPhaseThreads,
165       Set<Setting> settings) {
166     return newQueryEnvironment(runtime, keepGoing, orderedResults, ImmutableList.<String>of(),
167         loadingPhaseThreads, settings);
168   }
169 
170   public static AbstractBlazeQueryEnvironment<Target> newQueryEnvironment(BlazeRuntime runtime,
171       boolean keepGoing, boolean orderedResults, List<String> universeScope,
172       int loadingPhaseThreads,
173       Set<Setting> settings) {
174     ImmutableList.Builder<QueryFunction> functions = ImmutableList.builder();
175     for (BlazeModule module : runtime.getBlazeModules()) {
176       functions.addAll(module.getQueryFunctions());
177     }
178     return AbstractBlazeQueryEnvironment.newQueryEnvironment(
179         runtime.getPackageManager().newTransitiveLoader(),
180         runtime.getSkyframeExecutor(),
181         runtime.getPackageManager(),
182         runtime.getTargetPatternEvaluator(),
183         keepGoing, orderedResults, universeScope, loadingPhaseThreads, runtime.getReporter(),
184         settings,
185         functions.build(),
186         runtime.getPackageManager().getPackagePath());
187   }
