119   @Override
120   public SkyValue compute(SkyKey key, Environment env) throws ConfiguredTargetFunctionException,
121       InterruptedException {
122     SkyframeBuildView view = buildViewProvider.getSkyframeBuildView();
123 
124     ConfiguredTargetKey configuredTargetKey = (ConfiguredTargetKey) key.argument();
125     LabelAndConfiguration lc = LabelAndConfiguration.of(
126         configuredTargetKey.getLabel(), configuredTargetKey.getConfiguration());
127 
128     BuildConfiguration configuration = lc.getConfiguration();
129 
130     PackageValue packageValue =
131         (PackageValue) env.getValue(PackageValue.key(lc.getLabel().getPackageIdentifier()));
132     if (packageValue == null) {
133       return null;
134     }
135 
136     Target target;
137     try {
138       target = packageValue.getPackage().getTarget(lc.getLabel().getName());
139     } catch (NoSuchTargetException e1) {
140       throw new ConfiguredTargetFunctionException(new NoSuchTargetException(lc.getLabel(),
141           "No such target"));
142     }
143     // TODO(bazel-team): This is problematic - we create the right key, but then end up with a value
144     // that doesn't match; we can even have the same value multiple times. However, I think it's
145     // only triggered in tests (i.e., in normal operation, the configuration passed in is already
146     // null).
147     if (target instanceof InputFile) {
148       // InputFileConfiguredTarget expects its configuration to be null since it's not used.
149       configuration = null;
150     } else if (target instanceof PackageGroup) {
151       // Same for PackageGroupConfiguredTarget.
152       configuration = null;
153     }
154     TargetAndConfiguration ctgValue =
155         new TargetAndConfiguration(target, configuration);
156 
157     SkyframeDependencyResolver resolver = view.createDependencyResolver(env);
158     if (resolver == null) {
159       return null;
160     }
161 
162     try {
163       // Get the configuration targets that trigger this rule's configurable attributes.
164       Set<ConfigMatchingProvider> configConditions =
165           getConfigConditions(ctgValue.getTarget(), env, resolver, ctgValue);
166       if (configConditions == null) {
167         // Those targets haven't yet been resolved.
168         return null;
169       }
170 
171       ListMultimap<Attribute, ConfiguredTarget> depValueMap =
172           computeDependencies(env, resolver, ctgValue, null, configConditions);
173 
174       // TODO(bazel-team): Support dynamically created host configurations.
175       BuildConfiguration hostConfiguration = configuration == null
176           ? null : configuration.getConfiguration(Attribute.ConfigurationTransition.HOST);
177 
178       return createConfiguredTarget(
179           view, env, target, configuration, hostConfiguration, depValueMap, configConditions);
180     } catch (DependencyEvaluationException e) {
181       throw new ConfiguredTargetFunctionException(e.getRootCauseSkyKey(), e.getCause());
182     }
183   }

185   /**
186    * Computes the direct dependencies of a node in the configured target graph (a configured
187    * target or an aspect).
188    *
189    * <p>Returns null if Skyframe hasn't evaluated the required dependencies yet. In this case, the
190    * caller should also return null to Skyframe.
191    *
192    * @param env the Skyframe environment
193    * @param resolver The dependency resolver
194    * @param ctgValue The label and the configuration of the node
195    * @param aspectDefinition the aspect of the node (if null, the node is a configured target,
196    *     otherwise it's an asect)
197    * @param configConditions the configuration conditions for evaluating the attributes of the node
198    * @return an attribute -&gt; direct dependency multimap
199    */
200   @Nullable
201   static ListMultimap<Attribute, ConfiguredTarget> computeDependencies(
202       Environment env, SkyframeDependencyResolver resolver, TargetAndConfiguration ctgValue,
203       AspectDefinition aspectDefinition, Set<ConfigMatchingProvider> configConditions)
204       throws DependencyEvaluationException {
205 
206     // 1. Create the map from attributes to list of (target, configuration) pairs.
207     ListMultimap<Attribute, Dependency> depValueNames;
208     try {
209       depValueNames = resolver.dependentNodeMap(ctgValue, aspectDefinition, configConditions);
210     } catch (EvalException e) {
211       env.getListener().handle(Event.error(e.getLocation(), e.getMessage()));
212       throw new DependencyEvaluationException(new ConfiguredValueCreationException(e.print()));
213     }
214 
215     // 2. Resolve configured target dependencies and handle errors.
216     Map<SkyKey, ConfiguredTarget> depValues =
217         resolveConfiguredTargetDependencies(env, depValueNames.values(), ctgValue.getTarget());
218     if (depValues == null) {
219       return null;
220     }
221 
222     // 3. Resolve required aspects.
223     ListMultimap<SkyKey, Aspect> depAspects = resolveAspectDependencies(
224         env, depValues, depValueNames.values());
225     if (depAspects == null) {
226       return null;
227     }
228 
229     // 3. Merge the dependent configured targets and aspects into a single map.
230     return mergeAspects(depValueNames, depValues, depAspects);
231   }
232 
233   /**
234    * Merges the each direct dependency configured target with the aspects associated with it.
235    *
236    * <p>Note that the combination of a configured target and its associated aspects are not
237    * represented by a Skyframe node. This is because there can possibly be many different
238    * combinations of aspects for a particular configured target, so it would result in a
239    * combinatiorial explosion of Skyframe nodes.
240    */
241   private static ListMultimap<Attribute, ConfiguredTarget> mergeAspects(
242       ListMultimap<Attribute, Dependency> depValueNames,
243       Map<SkyKey, ConfiguredTarget> depConfiguredTargetMap,
244       ListMultimap<SkyKey, Aspect> depAspectMap) {
245     ListMultimap<Attribute, ConfiguredTarget> result = ArrayListMultimap.create();
246 
247     for (Map.Entry<Attribute, Dependency> entry : depValueNames.entries()) {
248       Dependency dep = entry.getValue();
249       SkyKey depKey = TO_KEYS.apply(dep);
250       ConfiguredTarget depConfiguredTarget = depConfiguredTargetMap.get(depKey);
251       result.put(entry.getKey(),
252           RuleConfiguredTarget.mergeAspects(depConfiguredTarget, depAspectMap.get(depKey)));
253     }
254 
255     return result;
256   }

258   /**
259    * Given a list of {@link Dependency} objects, returns a multimap from the {@link SkyKey} of the
260    * dependency to the {@link Aspect} instances that should be merged into it.
261    *
262    * <p>Returns null if the required aspects are not computed yet.
263    */
264   @Nullable
265   private static ListMultimap<SkyKey, Aspect> resolveAspectDependencies(Environment env,
266       Map<SkyKey, ConfiguredTarget> configuredTargetMap, Iterable<Dependency> deps)
267       throws DependencyEvaluationException {
268     ListMultimap<SkyKey, Aspect> result = ArrayListMultimap.create();
269     Set<SkyKey> aspectKeys = new HashSet<>();
270     for (Dependency dep : deps) {
271       for (Class<? extends ConfiguredAspectFactory> depAspect : dep.getAspects()) {
272         aspectKeys.add(AspectValue.key(dep.getLabel(), dep.getConfiguration(), depAspect));
273       }
274     }
275 
276     Map<SkyKey, ValueOrException3<
277         AspectCreationException, NoSuchThingException, ConfiguredValueCreationException>>
278         depAspects = env.getValuesOrThrow(aspectKeys, AspectCreationException.class,
279             NoSuchThingException.class, ConfiguredValueCreationException.class);
280 
281     for (Dependency dep : deps) {
282       SkyKey depKey = TO_KEYS.apply(dep);
283       // If the same target was declared in different attributes of rule, we should not process it
284       // twice.
285       if (result.containsKey(depKey)) {
286         continue;
287       }
288       ConfiguredTarget depConfiguredTarget = configuredTargetMap.get(depKey);
289       for (Class<? extends ConfiguredAspectFactory> depAspect : dep.getAspects()) {
290         if (!aspectMatchesConfiguredTarget(depConfiguredTarget, depAspect)) {
291           continue;
292         }
293 
294         SkyKey aspectKey = AspectValue.key(dep.getLabel(), dep.getConfiguration(), depAspect);
295         AspectValue aspectValue = null;
296         try {
297           aspectValue = (AspectValue) depAspects.get(aspectKey).get();
298         } catch (ConfiguredValueCreationException e) {
299           // The configured target should have been created in resolveConfiguredTargetDependencies()
300           throw new IllegalStateException(e);
301         } catch (NoSuchThingException | AspectCreationException e) {
302           AspectFactory<?, ?, ?> depAspectFactory = AspectFactory.Util.create(depAspect);
303           throw new DependencyEvaluationException(new ConfiguredValueCreationException(
304               String.format("Evaluation of aspect %s on %s failed: %s",
305                   depAspectFactory.getDefinition().getName(), dep.getLabel(), e.toString())));
306         }
307 
308         if (aspectValue == null) {
309           // Dependent aspect has either not been computed yet or is in error.
310           return null;
311         }
312         result.put(depKey, aspectValue.get());
313       }
314     }
315 
316     return result;
317   }

319   private static boolean aspectMatchesConfiguredTarget(ConfiguredTarget dep,
320       Class<? extends ConfiguredAspectFactory> aspectFactory) {
321     AspectDefinition aspectDefinition = AspectFactory.Util.create(aspectFactory).getDefinition();
322     for (Class<?> provider : aspectDefinition.getRequiredProviders()) {
323       if (dep.getProvider((Class<? extends TransitiveInfoProvider>) provider) == null) {
324         return false;
325       }
326     }
327 
328     return true;
329   }

331   /**
332    * Returns the set of {@link ConfigMatchingProvider}s that key the configurable attributes
333    * used by this rule.
334    *
335    * <p>>If the configured targets supplying those providers aren't yet resolved by the
336    * dependency resolver, returns null.
337    */
338   @Nullable
339   static Set<ConfigMatchingProvider> getConfigConditions(Target target, Environment env,
340       SkyframeDependencyResolver resolver, TargetAndConfiguration ctgValue)
341       throws DependencyEvaluationException {
342     if (!(target instanceof Rule)) {
343       return ImmutableSet.of();
344     }
345 
346     ImmutableSet.Builder<ConfigMatchingProvider> configConditions = ImmutableSet.builder();
347 
348     // Collect the labels of the configured targets we need to resolve.
349     ListMultimap<Attribute, LabelAndConfiguration> configLabelMap = ArrayListMultimap.create();
350     RawAttributeMapper attributeMap = RawAttributeMapper.of(((Rule) target));
351     for (Attribute a : ((Rule) target).getAttributes()) {
352       for (Label configLabel : attributeMap.getConfigurabilityKeys(a.getName(), a.getType())) {
353         if (!Type.Selector.isReservedLabel(configLabel)) {
354           configLabelMap.put(a, LabelAndConfiguration.of(
355               configLabel, ctgValue.getConfiguration()));
356         }
357       }
358     }
359     if (configLabelMap.isEmpty()) {
360       return ImmutableSet.of();
361     }
362 
363     // Collect the corresponding Skyframe configured target values. Abort early if they haven't
364     // been computed yet.
365     Collection<Dependency> configValueNames =
366         resolver.resolveRuleLabels(ctgValue, null, configLabelMap);
367     Map<SkyKey, ConfiguredTarget> configValues =
368         resolveConfiguredTargetDependencies(env, configValueNames, target);
369     if (configValues == null) {
370       return null;
371     }
372 
373     // Get the configured targets as ConfigMatchingProvider interfaces.
374     for (Dependency entry : configValueNames) {
375       ConfiguredTarget value = configValues.get(TO_KEYS.apply(entry));
376       // The code above guarantees that value is non-null here.
377       ConfigMatchingProvider provider = value.getProvider(ConfigMatchingProvider.class);
378       if (provider != null) {
379         configConditions.add(provider);
380       } else {
381         // Not a valid provider for configuration conditions.
382         String message =
383             entry.getLabel() + " is not a valid configuration key for " + target.getLabel();
384         env.getListener().handle(Event.error(TargetUtils.getLocationMaybe(target), message));
385         throw new DependencyEvaluationException(new ConfiguredValueCreationException(message));
386       }
387     }
388 
389     return configConditions.build();
390   }

392   /***
393    * Resolves the targets referenced in depValueNames and returns their ConfiguredTarget
394    * instances.
395    *
396    * <p>Returns null if not all instances are available yet.
397    *
398    */
399   @Nullable
400   private static Map<SkyKey, ConfiguredTarget> resolveConfiguredTargetDependencies(
401       Environment env, Collection<Dependency> deps, Target target)
402       throws DependencyEvaluationException {
403     boolean ok = !env.valuesMissing();
404     String message = null;
405     Iterable<SkyKey> depKeys = Iterables.transform(deps, TO_KEYS);
406     Map<SkyKey, ValueOrException2<NoSuchTargetException,
407         NoSuchPackageException>> depValuesOrExceptions = env.getValuesOrThrow(depKeys,
408             NoSuchTargetException.class, NoSuchPackageException.class);
409     Map<SkyKey, ConfiguredTarget> depValues = new HashMap<>(depValuesOrExceptions.size());
410     SkyKey childKey = null;
411     NoSuchThingException transitiveChildException = null;
412     for (Map.Entry<SkyKey, ValueOrException2<NoSuchTargetException, NoSuchPackageException>> entry
413         : depValuesOrExceptions.entrySet()) {
414       ConfiguredTargetKey depKey = (ConfiguredTargetKey) entry.getKey().argument();
415       LabelAndConfiguration depLabelAndConfiguration = LabelAndConfiguration.of(
416           depKey.getLabel(), depKey.getConfiguration());
417       Label depLabel = depLabelAndConfiguration.getLabel();
418       ConfiguredTargetValue depValue = null;
419       NoSuchThingException directChildException = null;
420       try {
421         depValue = (ConfiguredTargetValue) entry.getValue().get();
422       } catch (NoSuchTargetException e) {
423         if (depLabel.equals(e.getLabel())) {
424           directChildException = e;
425         } else {
426           childKey = entry.getKey();
427           transitiveChildException = e;
428         }
429       } catch (NoSuchPackageException e) {
430         if (depLabel.getPackageIdentifier().equals(e.getPackageId())) {
431           directChildException = e;
432         } else {
433           childKey = entry.getKey();
434           transitiveChildException = e;
435         }
436       }
437       // If an exception wasn't caused by a direct child target value, we'll treat it the same
438       // as any other missing dep by setting ok = false below, and returning null at the end.
439       if (directChildException != null) {
440         // Only update messages for missing targets we depend on directly.
441         message = TargetUtils.formatMissingEdge(target, depLabel, directChildException);
442         env.getListener().handle(Event.error(TargetUtils.getLocationMaybe(target), message));
443       }
444 
445       if (depValue == null) {
446         ok = false;
447       } else {
448         depValues.put(entry.getKey(), depValue.getConfiguredTarget());
449       }
450     }
451     if (message != null) {
452       throw new DependencyEvaluationException(new NoSuchTargetException(message));
453     }
454     if (childKey != null) {
455       throw new DependencyEvaluationException(childKey, transitiveChildException);
456     }
457     if (!ok) {
458       return null;
459     } else {
460       return depValues;
461     }
462   }

470   @Nullable
471   private ConfiguredTargetValue createConfiguredTarget(SkyframeBuildView view,
472       Environment env, Target target, BuildConfiguration configuration,
473       BuildConfiguration hostConfiguration, ListMultimap<Attribute, ConfiguredTarget> depValueMap,
474       Set<ConfigMatchingProvider> configConditions)
475       throws ConfiguredTargetFunctionException, InterruptedException {
476     StoredEventHandler events = new StoredEventHandler();
477     BuildConfiguration ownerConfig = (configuration == null)
478         ? null : configuration.getArtifactOwnerConfiguration();
479     CachingAnalysisEnvironment analysisEnvironment = view.createAnalysisEnvironment(
480         new ConfiguredTargetKey(target.getLabel(), ownerConfig), false,
481         events, env, configuration);
482     if (env.valuesMissing()) {
483       return null;
484     }
485 
486     ConfiguredTarget configuredTarget = view.createConfiguredTarget(target, configuration,
487         hostConfiguration, analysisEnvironment, depValueMap, configConditions);
488 
489     events.replayOn(env.getListener());
490     if (events.hasErrors()) {
491       analysisEnvironment.disable(target);
492       throw new ConfiguredTargetFunctionException(new ConfiguredValueCreationException(
493               "Analysis of target '" + target.getLabel() + "' failed; build aborted"));
494     }
495     Preconditions.checkState(!analysisEnvironment.hasErrors(),
496         "Analysis environment hasError() but no errors reported");
497     if (env.valuesMissing()) {
498       return null;
499     }
500 
501     analysisEnvironment.disable(target);
502     Preconditions.checkNotNull(configuredTarget, target);
503 
504     try {
505       return new ConfiguredTargetValue(configuredTarget,
506           filterSharedActionsAndThrowIfConflict(analysisEnvironment.getRegisteredActions()));
507     } catch (ActionConflictException e) {
508       throw new ConfiguredTargetFunctionException(e);
509     }
510   }
511 
512   static Map<Artifact, Action> filterSharedActionsAndThrowIfConflict(Iterable<Action> actions)
513       throws ActionConflictException {
514     Map<Artifact, Action> generatingActions = new HashMap<>();
515     for (Action action : actions) {
516       for (Artifact artifact : action.getOutputs()) {
517         Action previousAction = generatingActions.put(artifact, action);
518         if (previousAction != null && previousAction != action
519             && !Actions.canBeShared(previousAction, action)) {
520           throw new ActionConflictException(artifact, previousAction, action);
521         }
522       }
523     }
524     return generatingActions;
525   }
