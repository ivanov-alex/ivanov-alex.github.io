14  package com.google.devtools.build.lib.runtime.commands;
15  
16  import com.google.devtools.build.lib.Constants;
17  
18  /**
19   * An enumeration of all the valid info keys, excepting the make environment
20   * variables.
21   */
22  public enum InfoKey {
23    // directories
24    WORKSPACE("workspace", "The working directory of the server."),
25    INSTALL_BASE("install_base", "The installation base directory."),
26    OUTPUT_BASE("output_base",
27        "A directory for shared " + Constants.PRODUCT_NAME + " state as well as tool and strategy "
28        + "specific subdirectories."),
29    EXECUTION_ROOT("execution_root",
30        "A directory that makes all input and output files visible to the build."),
31    OUTPUT_PATH("output_path", "Output directory"),
32    BLAZE_BIN(Constants.PRODUCT_NAME + "-bin",
33        "Configuration dependent directory for binaries."),
34    BLAZE_GENFILES(Constants.PRODUCT_NAME + "-genfiles",
35        "Configuration dependent directory for generated files."),
36    BLAZE_TESTLOGS(Constants.PRODUCT_NAME + "-testlogs",
37        "Configuration dependent directory for logs from a test run."),
38  
39    // logs
40    COMMAND_LOG("command_log", "Location of the log containg the output from the build commands."),
41    MESSAGE_LOG("message_log" ,
42        "Location of a log containing machine readable message in LogMessage protobuf format."),
43  
44    // misc
45    RELEASE("release", Constants.PRODUCT_NAME + " release identifier"),
46    SERVER_PID("server_pid", Constants.PRODUCT_NAME + " process id"),
47    PACKAGE_PATH("package_path", "The search path for resolving package labels."),
48  
49    // memory statistics
50    USED_HEAP_SIZE("used-heap-size", "The amount of used memory in bytes. Note that this is not a "
51        + "good indicator of the actual memory use, as it includes any remaining inaccessible "
52        + "memory."),
53    USED_HEAP_SIZE_AFTER_GC("used-heap-size-after-gc",
54        "The amount of used memory in bytes after a call to System.gc().", true),
55    COMMITTED_HEAP_SIZE("committed-heap-size",
56        "The amount of memory in bytes that is committed for the Java virtual machine to use"),
57    MAX_HEAP_SIZE("max-heap-size",
58        "The maximum amount of memory in bytes that can be used for memory management."),
59    GC_COUNT("gc-count", "Number of garbage collection runs."),
60    GC_TIME("gc-time", "The approximate accumulated time spend on garbage collection."),
61  
62    // These are deprecated, they still work, when explicitly requested, but are not shown by default
63  
64    // These keys print multi-line messages and thus don't play well with grep. We don't print them
65    // unless explicitly requested
66    DEFAULTS_PACKAGE("defaults-package", "Default packages used as implicit dependencies", true),
67    BUILD_LANGUAGE("build-language", "A protobuffer with the build language structure", true),
68    DEFAULT_PACKAGE_PATH("default-package-path", "The default package path", true);
69  
70    private final String name;
71    private final String description;
72    private final boolean hidden;
73  
74    private InfoKey(String name, String description) {
75      this(name, description, false);
76    }
77  
78    private InfoKey(String name, String description, boolean hidden) {
79      this.name = name;
80      this.description = description;
81      this.hidden = hidden;
82    }
83  
84    public String getName() {
85      return name;
86    }
87  
88    public String getDescription() {
89      return description;
90    }
91  
92    public boolean isHidden() {
93      return hidden;
94    }
95  }
