59  /**
60   * Resolver for dependencies between configured targets.
61   *
62   * <p>Includes logic to derive the right configurations depending on transition type.
63   */
64  public abstract class DependencyResolver {
65    /**
66     * A dependency of a configured target through a label.
67     *
68     * <p>Includes the target and the configuration of the dependency configured target and any
69     * aspects that may be required.
70     *
71     * <p>Note that the presence of an aspect here does not necessarily mean that it will be created.
72     * They will be filtered based on the {@link TransitiveInfoProvider} instances their associated
73     * configured targets have (we cannot do that here because the configured targets are not
74     * available yet). No error or warning is reported in this case, because it is expected that rules
75     * sometimes over-approximate the providers they supply in their definitions.
76     */
77    public static final class Dependency {
78  
79      /**
80       * Returns the {@link ConfiguredTargetKey} for a direct dependency.
81       *
82       * <p>Essentially the same information as {@link Dependency} minus the aspects.
83       */
84      public static final Function<Dependency, ConfiguredTargetKey>
85          TO_CONFIGURED_TARGET_KEY = new Function<Dependency, ConfiguredTargetKey>() {
86            @Override
87            public ConfiguredTargetKey apply(Dependency input) {
88              return new ConfiguredTargetKey(input.getLabel(), input.getConfiguration());
89            }
90          };

94      @Nullable private final BuildConfiguration configuration;

98      public Dependency(
99          Label label,
100         @Nullable BuildConfiguration configuration,
101         ImmutableSet<Class<? extends ConfiguredAspectFactory>> aspects) {
102       this.label = Preconditions.checkNotNull(label);
103       this.configuration = configuration;
104       this.aspects = Preconditions.checkNotNull(aspects);
105     }
106 
107     public Dependency(Label label, @Nullable BuildConfiguration configuration) {
108       this(label, configuration, ImmutableSet.<Class<? extends ConfiguredAspectFactory>>of());
109     }
110 
111     public Label getLabel() {
112       return label;
113     }
114 
115     @Nullable
116     public BuildConfiguration getConfiguration() {
117       return configuration;
118     }
119 
120     public ImmutableSet<Class<? extends ConfiguredAspectFactory>> getAspects() {
121       return aspects;
122     }
123 
124     @Override
125     public int hashCode() {
126       return Objects.hash(label, configuration, aspects);
127     }
128 
129     @Override
130     public boolean equals(Object other) {
131       if (!(other instanceof Dependency)) {
132         return false;
133       }
134       Dependency otherDep = (Dependency) other;
135       return label.equals(otherDep.label)
136           && (configuration == otherDep.configuration
137               || (configuration != null && configuration.equals(otherDep.configuration)))
138           && aspects.equals(otherDep.aspects);
139     }
140 
141     @Override
142     public String toString() {
143       return String.format(
144           "Dependency{label=%s, configuration=%s, aspects=%s}", label, configuration, aspects);
145     }

151   /**
152    * Returns ids for dependent nodes of a given node, sorted by attribute. Note that some
153    * dependencies do not have a corresponding attribute here, and we use the null attribute to
154    * represent those edges. Visibility attributes are only visited if {@code visitVisibility} is
155    * {@code true}.
156    *
157    * <p>If {@code aspect} is null, returns the dependent nodes of the configured target node
158    * representing the given target and configuration, otherwise that of the aspect node accompanying
159    * the aforementioned configured target node for the specified aspect.
160    *
161    * <p>The values are not simply labels because this also implements the first step of applying
162    * configuration transitions, namely, split transitions. This needs to be done before the labels
163    * are resolved because late bound attributes depend on the configuration. A good example for this
164    * is @{code :cc_toolchain}.
165    *
166    * <p>The long-term goal is that most configuration transitions be applied here. However, in order
167    * to do that, we first have to eliminate transitions that depend on the rule class of the
168    * dependency.
169    */
170   public final ListMultimap<Attribute, Dependency> dependentNodeMap(
171       TargetAndConfiguration node, AspectDefinition aspect,
172       Set<ConfigMatchingProvider> configConditions)
173       throws EvalException {
174     Target target = node.getTarget();
175     BuildConfiguration config = node.getConfiguration();
176     ListMultimap<Attribute, Dependency> outgoingEdges = ArrayListMultimap.create();
177     if (target instanceof OutputFile) {
178       Preconditions.checkNotNull(config);
179       visitTargetVisibility(node, outgoingEdges.get(null));
180       Rule rule = ((OutputFile) target).getGeneratingRule();
181       outgoingEdges.put(null, new Dependency(rule.getLabel(), config));
182     } else if (target instanceof InputFile) {
183       visitTargetVisibility(node, outgoingEdges.get(null));
184     } else if (target instanceof EnvironmentGroup) {
185       visitTargetVisibility(node, outgoingEdges.get(null));
186     } else if (target instanceof Rule) {
187       Preconditions.checkNotNull(config);
188       visitTargetVisibility(node, outgoingEdges.get(null));
189       Rule rule = (Rule) target;
190       ListMultimap<Attribute, LabelAndConfiguration> labelMap =
191           resolveAttributes(rule, aspect, config, configConditions);
192       visitRule(rule, aspect, labelMap, outgoingEdges);
193     } else if (target instanceof PackageGroup) {
194       visitPackageGroup(node, (PackageGroup) target, outgoingEdges.get(null));
195     } else {
196       throw new IllegalStateException(target.getLabel().toString());
197     }
198     return outgoingEdges;
199   }

201   private ListMultimap<Attribute, LabelAndConfiguration> resolveAttributes(
202       Rule rule, AspectDefinition aspect, BuildConfiguration configuration,
203       Set<ConfigMatchingProvider> configConditions)
204       throws EvalException {
205     ConfiguredAttributeMapper attributeMap = ConfiguredAttributeMapper.of(rule, configConditions);
206     attributeMap.validateAttributes();
207     List<Attribute> attributes;
208     if (aspect == null) {
209       attributes = rule.getRuleClassObject().getAttributes();
210     } else {
211       attributes = new ArrayList<>();
212       attributes.addAll(rule.getRuleClassObject().getAttributes());
213       if (aspect != null) {
214         attributes.addAll(aspect.getAttributes().values());
215       }
216     }
217 
218     ImmutableSortedKeyListMultimap.Builder<Attribute, LabelAndConfiguration> result =
219         ImmutableSortedKeyListMultimap.builder();
220 
221     resolveExplicitAttributes(rule, configuration, attributeMap, result);
222     resolveImplicitAttributes(rule, configuration, attributeMap, attributes, result);
223     resolveLateBoundAttributes(rule, configuration, attributeMap, attributes, result);
224 
225     // Add the rule's visibility labels (which may come from the rule or from package defaults).
226     addExplicitDeps(result, rule, "visibility", rule.getVisibility().getDependencyLabels(),
227         configuration);
228 
229     // Add package default constraints when the rule doesn't explicitly declare them.
230     //
231     // Note that this can have subtle implications for constraint semantics. For example: say that
232     // package defaults declare compatibility with ':foo' and rule R declares compatibility with
233     // ':bar'. Does that mean that R is compatible with [':foo', ':bar'] or just [':bar']? In other
234     // words, did R's author intend to add additional compatibility to the package defaults or to
235     // override them? More severely, what if package defaults "restrict" support to just [':baz']?
236     // Should R's declaration signify [':baz'] + ['bar'], [ORIGINAL_DEFAULTS] + ['bar'], or
237     // something else?
238     //
239     // Rather than try to answer these questions with possibly confusing logic, we take the
240     // simple approach of assigning the rule's "restriction" attribute to the rule-declared value if
241     // it exists, else the package defaults value (and likewise for "compatibility"). This may not
242     // always provide what users want, but it makes it easy for them to understand how rule
243     // declarations and package defaults intermix (and how to refactor them to get what they want).
244     //
245     // An alternative model would be to apply the "rule declaration" / "rule class defaults"
246     // relationship, i.e. the rule class' "compatibility" and "restriction" declarations are merged
247     // to generate a set of default environments, then the rule's declarations are independently
248     // processed on top of that. This protects against obscure coupling behavior between
249     // declarations from wildly different places (e.g. it offers clear answers to the examples posed
250     // above). But within the scope of a single package it seems better to keep the model simple and
251     // make the user responsible for resolving ambiguities.
252     if (!rule.isAttributeValueExplicitlySpecified(RuleClass.COMPATIBLE_ENVIRONMENT_ATTR)) {
253       addExplicitDeps(result, rule, RuleClass.COMPATIBLE_ENVIRONMENT_ATTR,
254           rule.getPackage().getDefaultCompatibleWith(), configuration);
255     }
256     if (!rule.isAttributeValueExplicitlySpecified(RuleClass.RESTRICTED_ENVIRONMENT_ATTR)) {
257       addExplicitDeps(result, rule, RuleClass.RESTRICTED_ENVIRONMENT_ATTR,
258           rule.getPackage().getDefaultRestrictedTo(), configuration);
259     }
260 
261     return result.build();
262   }

264   /**
265    * Adds new dependencies to the given rule under the given attribute name
266    *
267    * @param result the builder for the attribute --> dependency labels map
268    * @param rule the rule being evaluated
269    * @param attrName the name of the attribute to add dependency labels to
270    * @param labels the dependencies to add
271    * @param configuration the configuration to apply to those dependencies
272    */
273   private void addExplicitDeps(
274       ImmutableSortedKeyListMultimap.Builder<Attribute, LabelAndConfiguration> result, Rule rule,
275       String attrName, Iterable<Label> labels, BuildConfiguration configuration) {
276     if (!rule.isAttrDefined(attrName, Type.LABEL_LIST)
277         && !rule.isAttrDefined(attrName, Type.NODEP_LABEL_LIST)) {
278       return;
279     }
280     Attribute attribute = rule.getRuleClassObject().getAttributeByName(attrName);
281     for (Label label : labels) {
282       // The configuration must be the configuration after the first transition step (applying
283       // split configurations). The proper configuration (null) for package groups will be set
284       // later.
285       result.put(attribute, LabelAndConfiguration.of(label, configuration));
286     }
287   }

289   private void resolveExplicitAttributes(Rule rule, final BuildConfiguration configuration,
290       AttributeMap attributes,
291       final ImmutableSortedKeyListMultimap.Builder<Attribute, LabelAndConfiguration> builder) {
292     attributes.visitLabels(
293         new AttributeMap.AcceptsLabelAttribute() {
294           @Override
295           public void acceptLabelAttribute(Label label, Attribute attribute) {
296             String attributeName = attribute.getName();
297             if (attributeName.equals("abi_deps")) {
298               // abi_deps is handled specially: we visit only the branch that
299               // needs to be taken based on the configuration.
300               return;
301             }
302 
303             if (attribute.getType() == Type.NODEP_LABEL) {
304               return;
305             }
306 
307             if (attribute.isImplicit() || attribute.isLateBound()) {
308               return;
309             }
310 
311             builder.put(attribute, LabelAndConfiguration.of(label, configuration));
312           }
313         });
314 
315     // TODO(bazel-team): Remove this in favor of the new configurable attributes.
316     if (attributes.getAttributeDefinition("abi_deps") != null) {
317       Attribute depsAttribute = attributes.getAttributeDefinition("deps");
318       MakeVariableExpander.Context context = new ConfigurationMakeVariableContext(
319           rule.getPackage(), configuration);
320       String abi = null;
321       try {
322         abi = MakeVariableExpander.expand(attributes.get("abi", Type.STRING), context);
323       } catch (MakeVariableExpander.ExpansionException e) {
324         // Ignore this. It will be handled during the analysis phase.
325       }
326 
327       if (abi != null) {
328         for (Map.Entry<String, List<Label>> entry
329             : attributes.get("abi_deps", Type.LABEL_LIST_DICT).entrySet()) {
330           try {
331             if (Pattern.matches(entry.getKey(), abi)) {
332               for (Label label : entry.getValue()) {
333                 builder.put(depsAttribute, LabelAndConfiguration.of(label, configuration));
334               }
335             }
336           } catch (PatternSyntaxException e) {
337             // Ignore this. It will be handled during the analysis phase.
338           }
339         }
340       }
341     }
342   }

344   private void resolveImplicitAttributes(Rule rule, BuildConfiguration configuration,
345       AttributeMap attributeMap, Iterable<Attribute> attributes,
346       ImmutableSortedKeyListMultimap.Builder<Attribute, LabelAndConfiguration> builder) {
347     // Since the attributes that come from aspects do not appear in attributeMap, we have to get
348     // their values from somewhere else. This incidentally means that aspects attributes are not
349     // configurable. It would be nice if that wasn't the case, but we'd have to revamp how
350     // attribute mapping works, which is a large chunk of work.
351     ImmutableSet<String> mappedAttributes = ImmutableSet.copyOf(attributeMap.getAttributeNames());
352     for (Attribute attribute : attributes) {
353       if (!attribute.isImplicit() || !attribute.getCondition().apply(attributeMap)) {
354         continue;
355       }
356 
357       if (attribute.getType() == Type.LABEL) {
358         Label label = mappedAttributes.contains(attribute.getName())
359             ? attributeMap.get(attribute.getName(), Type.LABEL)
360             : Type.LABEL.cast(attribute.getDefaultValue(rule));
361 
362         if (label != null) {
363           builder.put(attribute, LabelAndConfiguration.of(label, configuration));
364         }
365       } else if (attribute.getType() == Type.LABEL_LIST) {
366         List<Label> labelList = mappedAttributes.contains(attribute.getName())
367             ? attributeMap.get(attribute.getName(), Type.LABEL_LIST)
368             : Type.LABEL_LIST.cast(attribute.getDefaultValue(rule));
369 
370         for (Label label : labelList) {
371           builder.put(attribute, LabelAndConfiguration.of(label, configuration));
372         }
373       }
374     }
375   }

377   private void resolveLateBoundAttributes(
378       Rule rule,
379       BuildConfiguration configuration,
380       AttributeMap attributeMap,
381       Iterable<Attribute> attributes,
382       ImmutableSortedKeyListMultimap.Builder<Attribute, LabelAndConfiguration> builder)
383       throws EvalException {
384     for (Attribute attribute : attributes) {
385       if (!attribute.isLateBound() || !attribute.getCondition().apply(attributeMap)) {
386         continue;
387       }
388 
389       List<BuildConfiguration> actualConfigurations = ImmutableList.of(configuration);
390       if (attribute.getConfigurationTransition() instanceof SplitTransition<?>) {
391         Preconditions.checkState(attribute.getConfigurator() == null);
392         // TODO(bazel-team): This ends up applying the split transition twice, both here and in the
393         // visitRule method below - this is not currently a problem, because the configuration graph
394         // never contains nested split transitions, so the second application is idempotent.
395         actualConfigurations = configuration.getSplitConfigurations(
396             (SplitTransition<?>) attribute.getConfigurationTransition());
397       }
398 
399       for (BuildConfiguration actualConfig : actualConfigurations) {
400         @SuppressWarnings("unchecked")
401         LateBoundDefault<BuildConfiguration> lateBoundDefault =
402             (LateBoundDefault<BuildConfiguration>) attribute.getLateBoundDefault();
403         if (lateBoundDefault.useHostConfiguration()) {
404           actualConfig =
405               actualConfig.getConfiguration(ConfigurationTransition.HOST);
406         }
407         // TODO(bazel-team): This might be too expensive - can we cache this somehow?
408         if (!lateBoundDefault.getRequiredConfigurationFragments().isEmpty()) {
409           if (!actualConfig.hasAllFragments(lateBoundDefault.getRequiredConfigurationFragments())) {
410             continue;
411           }
412         }
413 
414         // TODO(bazel-team): We should check if the implementation tries to access an undeclared
415         // fragment.
416         Object actualValue = lateBoundDefault.getDefault(rule, actualConfig);
417         if (EvalUtils.isNullOrNone(actualValue)) {
418           continue;
419         }
420         try {
421           if (attribute.getType() == Type.LABEL) {
422             Label label = Type.LABEL.cast(actualValue);
423             builder.put(attribute, LabelAndConfiguration.of(label, actualConfig));
424           } else if (attribute.getType() == Type.LABEL_LIST) {
425             for (Label label : Type.LABEL_LIST.cast(actualValue)) {
426               builder.put(attribute, LabelAndConfiguration.of(label, actualConfig));
427             }
428           } else {
429             throw new IllegalStateException(
430                 String.format(
431                     "Late bound attribute '%s' is not a label or a label list",
432                     attribute.getName()));
433           }
434         } catch (ClassCastException e) {
435           throw new EvalException(
436               rule.getLocation(),
437               String.format(
438                   "When computing the default value of %s, expected '%s', got '%s'",
439                   attribute.getName(),
440                   attribute.getType(),
441                   EvalUtils.getDataTypeName(actualValue, true)));
442         }
443       }
444     }
445   }

447   /**
448    * A variant of {@link #dependentNodeMap} that only returns the values of the resulting map, and
449    * also converts any internally thrown {@link EvalException} instances into {@link
450    * IllegalStateException}.
451    */
452   public final Collection<Dependency> dependentNodes(
453       TargetAndConfiguration node, Set<ConfigMatchingProvider> configConditions) {
454     try {
455       return dependentNodeMap(node, null, configConditions).values();
456     } catch (EvalException e) {
457       throw new IllegalStateException(e);

461   /**
462    * Converts the given multimap of attributes to labels into a multi map of attributes to
463    * {@link Dependency} objects using the proper configuration transition for each attribute.
464    *
465    * @throws IllegalArgumentException if the {@code node} does not refer to a {@link Rule} instance
466    */
467   public final Collection<Dependency> resolveRuleLabels(
468       TargetAndConfiguration node, AspectDefinition aspect, ListMultimap<Attribute,
469       LabelAndConfiguration> labelMap) {
470     Preconditions.checkArgument(node.getTarget() instanceof Rule);
471     Rule rule = (Rule) node.getTarget();
472     ListMultimap<Attribute, Dependency> outgoingEdges = ArrayListMultimap.create();
473     visitRule(rule, aspect, labelMap, outgoingEdges);
474     return outgoingEdges.values();
475   }

477   private void visitPackageGroup(TargetAndConfiguration node, PackageGroup packageGroup,
478       Collection<Dependency> outgoingEdges) {
479     for (Label label : packageGroup.getIncludes()) {
480       try {
481         Target target = getTarget(label);
482         if (target == null) {
483           return;
484         }
485         if (!(target instanceof PackageGroup)) {
486           // Note that this error could also be caught in PackageGroupConfiguredTarget, but since
487           // these have the null configuration, visiting the corresponding target would trigger an
488           // analysis of a rule with a null configuration, which doesn't work.
489           invalidPackageGroupReferenceHook(node, label);
490           continue;
491         }
492 
493         outgoingEdges.add(new Dependency(label, node.getConfiguration()));
494       } catch (NoSuchThingException e) {
495         // Don't visit targets that don't exist (--keep_going)
496       }
497     }
498   }

500   private ImmutableSet<Class<? extends ConfiguredAspectFactory>> requiredAspects(
501       AspectDefinition aspect, Attribute attribute, Target target) {
502     if (!(target instanceof Rule)) {
503       return ImmutableSet.of();
504     }
505 
506     RuleClass ruleClass = ((Rule) target).getRuleClassObject();
507 
508     // The order of this set will be deterministic. This is necessary because this order eventually
509     // influences the order in which aspects are merged into the main configured target, which in
510     // turn influences which aspect takes precedence if two emit the same provider (maybe this
511     // should be an error)
512     Set<Class<? extends AspectFactory<?, ?, ?>>> aspectCandidates = new LinkedHashSet<>();
513     aspectCandidates.addAll(attribute.getAspects());
514     if (aspect != null) {
515       aspectCandidates.addAll(aspect.getAttributeAspects().get(attribute.getName()));
516     }
517 
518     ImmutableSet.Builder<Class<? extends ConfiguredAspectFactory>> result = ImmutableSet.builder();
519     for (Class<? extends AspectFactory<?, ?, ?>> candidateClass : aspectCandidates) {
520       ConfiguredAspectFactory candidate =
521           (ConfiguredAspectFactory) AspectFactory.Util.create(candidateClass);
522       if (Sets.difference(
523           candidate.getDefinition().getRequiredProviders(),
524           ruleClass.getAdvertisedProviders()).isEmpty()) {
525         result.add(candidateClass.asSubclass(ConfiguredAspectFactory.class));
526       }
527     }
528 
529     return result.build();
530   }

532   private void visitRule(Rule rule, AspectDefinition aspect,
533       ListMultimap<Attribute, LabelAndConfiguration> labelMap,
534       ListMultimap<Attribute, Dependency> outgoingEdges) {
535     Preconditions.checkNotNull(labelMap);
536     for (Map.Entry<Attribute, Collection<LabelAndConfiguration>> entry :
537         labelMap.asMap().entrySet()) {
538       Attribute attribute = entry.getKey();
539       for (LabelAndConfiguration dep : entry.getValue()) {
540         Label label = dep.getLabel();
541         BuildConfiguration config = dep.getConfiguration();
542 
543         Target toTarget;
544         try {
545           toTarget = getTarget(label);
546         } catch (NoSuchThingException e) {
547           throw new IllegalStateException("not found: " + label + " from " + rule + " in "
548               + attribute.getName());
549         }
550         if (toTarget == null) {
551           continue;
552         }
553         Iterable<BuildConfiguration> toConfigurations = config.evaluateTransition(
554             rule, attribute, toTarget);
555         for (BuildConfiguration toConfiguration : toConfigurations) {
556           outgoingEdges.put(
557               entry.getKey(),
558               new Dependency(label, toConfiguration, requiredAspects(aspect, attribute, toTarget)));
559         }
560       }
561     }
562   }

564   private void visitTargetVisibility(TargetAndConfiguration node,
565       Collection<Dependency> outgoingEdges) {
566     for (Label label : node.getTarget().getVisibility().getDependencyLabels()) {
567       try {
568         Target visibilityTarget = getTarget(label);
569         if (visibilityTarget == null) {
570           return;
571         }
572         if (!(visibilityTarget instanceof PackageGroup)) {
573           // Note that this error could also be caught in
574           // AbstractConfiguredTarget.convertVisibility(), but we have an
575           // opportunity here to avoid dependency cycles that result from
576           // the visibility attribute of a rule referring to a rule that
577           // depends on it (instead of its package)
578           invalidVisibilityReferenceHook(node, label);
579           continue;
580         }
581 
582         // Visibility always has null configuration
583         outgoingEdges.add(new Dependency(label, null));
584       } catch (NoSuchThingException e) {
585         // Don't visit targets that don't exist (--keep_going)
586       }
587     }
588   }

609    *
610    * <p>Throws {@link NoSuchThingException} if the target is known not to exist.
611    *
