75     * A Skylark built-in function to create and register a SpawnAction using a
76     * dictionary of parameters:
77     * createSpawnAction(
78     *         inputs = [input1, input2, ...],
79     *         outputs = [output1, output2, ...],
80     *         executable = executable,
81     *         arguments = [argument1, argument2, ...],
82     *         mnemonic = 'Mnemonic',
83     *         command = 'command',
84     *     )

86    @SkylarkSignature(name = "action",
87        doc = "Creates an action that runs an executable or a shell command. You must specify either "
88          + "<code>command</code> or <code>executable</code>.",
89        objectType = SkylarkRuleContext.class,
90        returnType = Environment.NoneType.class,
91        mandatoryPositionals = {
92          @Param(name = "self", type = SkylarkRuleContext.class, doc = "This RuleContext.")},
93        mandatoryNamedOnly = {
94          @Param(name = "outputs", type = SkylarkList.class, generic1 = Artifact.class,
95              doc = "list of the output files of the action")},
96        optionalNamedOnly = {
97          @Param(name = "inputs", type = SkylarkList.class, generic1 = Artifact.class,
98              defaultValue = "[]", doc = "list of the input files of the action"),
99          @Param(name = "executable", type = Object.class, // File or PathFragment or None
100             defaultValue = "None",
101             doc = "the executable file to be called by the action"),
102         @Param(name = "arguments", type = SkylarkList.class, generic1 = String.class,
103             defaultValue = "[]", doc = "command line arguments of the action"),
104         @Param(name = "mnemonic", type = String.class, noneable = true,
105             defaultValue = "None",
106             doc = "a one-word description of the action, e.g. CppCompile or GoLink"),
107         @Param(name = "command", type = Object.class, // string or ListOf(string) or NoneType
108             defaultValue = "None", doc = "shell command to execute. It is usually preferable to "
109             + "use <code>executable</code> instead. Arguments are available with <code>$1</code>, "
110             + "<code>$2</code>, etc."),
111         @Param(name = "progress_message", type = String.class, noneable = true,
112             defaultValue = "None",
113             doc = "progress message to show to the user during the build, "
114             + "e.g. \"Compiling foo.cc to create foo.o\""),
115         @Param(name = "use_default_shell_env", type = Boolean.class, defaultValue = "False",
116             doc = "whether the action should use the built in shell environment or not"),
117         @Param(name = "env", type = Map.class, noneable = true, defaultValue = "None",
118             doc = "sets the dictionary of environment variables"),
119         @Param(name = "execution_requirements", type = Map.class, noneable = true,
120             defaultValue = "None",
121             doc = "information for scheduling the action"),
122         @Param(name = "input_manifests", type = Map.class, noneable = true,
123             defaultValue = "None",
124             doc = "sets the map of input manifests files; "
125             + "they are typicially generated by the command_helper")},
126       useLocation = true)
127   private static final BuiltinFunction createSpawnAction = new BuiltinFunction("action") {
128     public Environment.NoneType invoke(
129         SkylarkRuleContext ctx,
130         SkylarkList outputs,
131         SkylarkList inputs,
132         Object executableO,
133         SkylarkList arguments,
134         Object mnemonicO,
135         Object commandO,
136         Object progressMessage,
137         Boolean useDefaultShellEnv,
138         Object envO,
139         Object executionRequirementsO,
140         Object inputManifestsO,
141         Location loc) throws EvalException, ConversionException {
142       SpawnAction.Builder builder = new SpawnAction.Builder();
143       // TODO(bazel-team): builder still makes unnecessary copies of inputs, outputs and args.
144       builder.addInputs(castList(inputs, Artifact.class));
145       builder.addOutputs(castList(outputs, Artifact.class));
146       if (commandO != Environment.NONE && arguments.size() > 0) {
147         // When we use a shell command, add an empty argument before other arguments.
148         //   e.g.  bash -c "cmd" '' 'arg1' 'arg2'
149         // bash will use the empty argument as the value of $0 (which we don't care about).
150         // arg1 and arg2 will be $1 and $2, as a user exects.
151         builder.addArgument("");
152       }
153       builder.addArguments(castList(arguments, String.class));
154       if (executableO != Environment.NONE) {
155         if (executableO instanceof Artifact) {
156           Artifact executable = (Artifact) executableO;
157           builder.addInput(executable);
158           FilesToRunProvider provider = ctx.getExecutableRunfiles(executable);
159           if (provider == null) {
160             builder.setExecutable(executable);
161           } else {
162             builder.setExecutable(provider);
163           }
164         } else if (executableO instanceof PathFragment) {
165           builder.setExecutable((PathFragment) executableO);
166         } else {
167           throw new EvalException(loc, "expected file or PathFragment for "
168               + "executable but got " + EvalUtils.getDataTypeName(executableO) + " instead");
169         }
170       }
171       if ((commandO == Environment.NONE) == (executableO == Environment.NONE)) {
172         throw new EvalException(loc, "You must specify either 'command' or 'executable' argument");
173       }
174       if (commandO != Environment.NONE) {
175         if (commandO instanceof String) {
176           builder.setShellCommand((String) commandO);
177         } else if (commandO instanceof SkylarkList) {
178           SkylarkList commandList = (SkylarkList) commandO;
179           if (commandList.size() < 3) {
180             throw new EvalException(loc, "'command' list has to be of size at least 3");
181           }
182           builder.setShellCommand(castList(commandList, String.class, "command"));
183         } else {
184           throw new EvalException(loc, "expected string or list of strings for "
185               + "command instead of " + EvalUtils.getDataTypeName(commandO));
186         }
187       }
188       if (mnemonicO != Environment.NONE) {
189         builder.setMnemonic((String) mnemonicO);
190       }
191       if (envO != Environment.NONE) {
192         builder.setEnvironment(ImmutableMap.copyOf(
193             castMap(envO, String.class, String.class, "env")));
194       }
195       if (progressMessage != Environment.NONE) {
196         builder.setProgressMessage((String) progressMessage);
197       }
198       if (EvalUtils.toBoolean(useDefaultShellEnv)) {
199         builder.useDefaultShellEnvironment();
200       }
201       if (executionRequirementsO != Environment.NONE) {
202         builder.setExecutionInfo(ImmutableMap.copyOf(castMap(
203             executionRequirementsO, String.class, String.class, "execution_requirements")));
204       }
205       if (inputManifestsO != Environment.NONE) {
206         for (Map.Entry<PathFragment, Artifact> entry : castMap(inputManifestsO,
207             PathFragment.class, Artifact.class, "input manifest file map").entrySet()) {
208           builder.addInputManifest(entry.getValue(), entry.getKey());
209         }
210       }
211       // Always register the action
212       ctx.getRuleContext().registerAction(builder.build(ctx.getRuleContext()));
213       return Environment.NONE;
214     }

217   @SkylarkSignature(name = "expand_location",
218       doc =
219       "Expands the given string so that all labels are replaced with the location "
220       + "of their target file(s). Currently, the algorithm uses output, srcs, deps, "
221       + "tools and data attributes for looking up mappings from label to locations.",
222       objectType = SkylarkRuleContext.class, returnType = String.class,
223       mandatoryPositionals = {
224           @Param(name = "self", type = SkylarkRuleContext.class, doc = "this context"),
225           @Param(name = "input", type = String.class, doc = "string to be expanded"),
226       },
227       optionalPositionals = {
228           @Param(name = "targets", type = SkylarkList.class,
229               generic1 = AbstractConfiguredTarget.class, defaultValue = "[]",
230               doc = "list of targets for additional lookup information"),
231       },
232       useLocation = true, useEnvironment = true)
233   private static final BuiltinFunction expandLocation = new BuiltinFunction("expand_location") {
234     @SuppressWarnings("unused")
235     public String invoke(SkylarkRuleContext ctx, String input, SkylarkList targets,
236         Location loc, Environment env) throws EvalException {
237       try {
238         return new LocationExpander(ctx.getRuleContext(),
239                    makeLabelMap(castList(targets, AbstractConfiguredTarget.class)), false)
240             .expand(input);
241       } catch (IllegalStateException ise) {
242         throw new EvalException(loc, ise);
243       }
244     }
245   };

247   /**
248    * Builds a map: Label -> List of files from the given labels
249    * @param knownLabels List of known labels
250    * @return Immutable map with immutable collections as values
251    */
252   private static ImmutableMap<Label, ImmutableCollection<Artifact>> makeLabelMap(
253       Iterable<AbstractConfiguredTarget> knownLabels) {
254     ImmutableMap.Builder<Label, ImmutableCollection<Artifact>> builder = ImmutableMap.builder();
255 
256     for (AbstractConfiguredTarget current : knownLabels) {
257       builder.put(
258           current.getLabel(),
259           ImmutableList.copyOf(current.getProvider(FileProvider.class).getFilesToBuild()));
260     }
261 
262     return builder.build();
263   }

289   @SkylarkSignature(name = "empty_action",
290       doc =
291       "Creates an empty action that neither executes a command nor produces any "
292       + "output, but that is useful for inserting 'extra actions'.",
293       objectType = SkylarkRuleContext.class, returnType = NoneType.class,
294       mandatoryPositionals = {
295           @Param(name = "self", type = SkylarkRuleContext.class, doc = "this context"),
296       },
297       mandatoryNamedOnly = {
298           @Param(name = "mnemonic", type = String.class, defaultValue = "None",
299           doc = "a one-word description of the action, e.g. CppCompile or GoLink"),
300       },
301       optionalNamedOnly = {
302           @Param(name = "inputs", type = SkylarkList.class, generic1 = Artifact.class,
303               defaultValue = "[]", doc = "list of the input files of the action"),
304       })
305   private static final BuiltinFunction createEmptyAction = new BuiltinFunction("empty_action") {
306     @SuppressWarnings("unused")
307     public NoneType invoke(SkylarkRuleContext ctx, String mnemonic, SkylarkList inputs)
308         throws EvalException, ConversionException {
309       RuleContext ruleContext = ctx.getRuleContext();
310       Action action = new PseudoAction<SpawnInfo>(generateUuid(ruleContext),
311           ruleContext.getActionOwner(), convertInputs(inputs), generateDummyOutputs(ruleContext),
312           mnemonic, SpawnInfo.spawnInfo, createEmptySpawnInfo());
313       ruleContext.registerAction(action);
314 
315       return Environment.NONE;
316     }

318     private NestedSet<Artifact> convertInputs(SkylarkList inputs) {
319       return NestedSetBuilder.<Artifact>compileOrder().addAll(inputs.to(Artifact.class)).build();
320     }
321 
322     protected UUID generateUuid(RuleContext ruleContext) {
323       return UUID.nameUUIDFromBytes(
324           String.format("empty action %s", ruleContext.getLabel())
325               .getBytes(StandardCharsets.UTF_8));
326     }
327 
328     protected ImmutableList<Artifact> generateDummyOutputs(RuleContext ruleContext) {
329       return ImmutableList.of(PseudoAction.getDummyOutput(ruleContext));
330     }
331 
332     protected SpawnInfo createEmptySpawnInfo() {
333       return SpawnInfo.newBuilder().build();
334     }

337   @SkylarkSignature(name = "template_action",
338       doc = "Creates a template expansion action.",
339       objectType = SkylarkRuleContext.class,
340       returnType = TemplateExpansionAction.class,
341       mandatoryPositionals = {
342         @Param(name = "self", type = SkylarkRuleContext.class, doc = "this context")},
343       mandatoryNamedOnly = {
344         @Param(name = "template", type = Artifact.class,
345             doc = "the template file"),
346         @Param(name = "output", type = Artifact.class,
347             doc = "the output file"),
348         @Param(name = "substitutions", type = Map.class,
349             doc = "substitutions to make when expanding the template")},
350       optionalNamedOnly = {
351         @Param(name = "executable", type = Boolean.class,
352             doc = "whether the output file should be executable (default is False)")})
353   private static final BuiltinFunction createTemplateAction =
354       new BuiltinFunction("template_action", Arrays.<Object>asList(false)) {
355         public TemplateExpansionAction invoke(SkylarkRuleContext ctx,
356             Artifact template, Artifact output, Map<?, ?> substitutionsO, Boolean executable)
357             throws EvalException, ConversionException {
358           ImmutableList.Builder<Substitution> substitutions = ImmutableList.builder();
359           for (Map.Entry<String, String> substitution : castMap(
360               substitutionsO, String.class, String.class, "substitutions").entrySet()) {
361             substitutions.add(Substitution.of(substitution.getKey(), substitution.getValue()));
362           }
363           TemplateExpansionAction action = new TemplateExpansionAction(
364               ctx.getRuleContext().getActionOwner(),
365               template,
366               output,
367               substitutions.build(),
368               executable);
369           ctx.getRuleContext().registerAction(action);
370           return action;
371         }
372       };

376    * Transitive info providers of Transitive info collections.

378   @SkylarkSignature(name = "provider",
379       doc = "Returns the transitive info provider provided by the target.",
380       mandatoryPositionals = {
381         @Param(name = "target", type = TransitiveInfoCollection.class,
382             doc = "the configured target which provides the provider"),
383         @Param(name = "type", type = String.class, doc = "the class type of the provider")},
384       useLocation = true)
385   private static final BuiltinFunction provider = new BuiltinFunction("provider") {
386       public Object invoke(TransitiveInfoCollection target, String type,
387           Location loc) throws EvalException {
388       try {
389         Class<?> classType = SkylarkRuleContext.classCache.get(type);
390         Class<? extends TransitiveInfoProvider> convertedClass =
391             classType.asSubclass(TransitiveInfoProvider.class);
392         Object result = target.getProvider(convertedClass);
393         return result == null ? Environment.NONE : result;
394       } catch (ExecutionException e) {
395         throw new EvalException(loc, "Unknown class type " + type);
396       } catch (ClassCastException e) {
397         throw new EvalException(loc, "Not a TransitiveInfoProvider " + type);
398       }
399     }

412         // TODO(bazel-team): If we have a memory efficient support for lazy list containing

414         // Also, allow empty set for init
415         @Param(name = "transitive_files", type = SkylarkNestedSet.class, generic1 = Artifact.class,
416             noneable = true, defaultValue = "None",
417             doc = "The (transitive) set of files to be added to the runfiles."),
418         @Param(name = "collect_data", type = Boolean.class, defaultValue = "False",
419             doc = "Whether to collect the data "
420             + "runfiles from the dependencies in srcs, data and deps attributes."),
421         @Param(name = "collect_default", type = Boolean.class, defaultValue = "False",
422             doc = "Whether to collect the default "
423             + "runfiles from the dependencies in srcs, data and deps attributes.")},
424       useLocation = true)
425   private static final BuiltinFunction runfiles = new BuiltinFunction("runfiles") {
426     public Runfiles invoke(SkylarkRuleContext ctx, SkylarkList files, Object transitiveFiles,
427         Boolean collectData, Boolean collectDefault,
428         Location loc) throws EvalException, ConversionException {
429       Runfiles.Builder builder = new Runfiles.Builder();
430       if (EvalUtils.toBoolean(collectData)) {
431         builder.addRunfiles(ctx.getRuleContext(), RunfilesProvider.DATA_RUNFILES);
432       }
433       if (EvalUtils.toBoolean(collectDefault)) {
434         builder.addRunfiles(ctx.getRuleContext(), RunfilesProvider.DEFAULT_RUNFILES);
435       }
436       if (!files.isEmpty()) {
437         builder.addArtifacts(castList(files, Artifact.class));
438       }
439       if (transitiveFiles != Environment.NONE) {
440         builder.addTransitiveArtifacts(((SkylarkNestedSet) transitiveFiles).getSet(Artifact.class));
441       }
442       return builder.build();
443     }
