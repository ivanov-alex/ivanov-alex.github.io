78    // Convert string index in the same way Python does.
79    // If index is negative, starts from the end.
80    // If index is outside bounds, it is restricted to the valid range.
81    private static int clampIndex(int index, int length) {
82      if (index < 0) {
83        index += length;
84      }
85      return Math.max(Math.min(index, length), 0);
86    }

88    // Emulate Python substring function
89    // It converts out of range indices, and never fails
90    private static String pythonSubstring(String str, int start, Object end, String msg)
91        throws ConversionException {
92      if (start == 0 && EvalUtils.isNullOrNone(end)) {
93        return str;
94      }
95      start = clampIndex(start, str.length());
96      int stop;
97      if (EvalUtils.isNullOrNone(end)) {
98        stop = str.length();
99      } else {
100       stop = clampIndex(Type.INTEGER.convert(end, msg), str.length());
101     }
102     if (start >= stop) {
103       return "";
104     }
105     return str.substring(start, stop);
106   }

243   /**
244    * Splits the given string into a list of words, using {@code separator} as a 
245    * delimiter.
246    * 
247    * <p>At most {@code maxSplits} will be performed, going from right to left.
248    *
249    * @param input The input string.
250    * @param separator The separator string.
251    * @param maxSplits The maximum number of splits. Negative values mean unlimited splits.
252    * @return A list of words
253    * @throws IllegalArgumentException
254    */
255   private static List<String> stringRSplit(String input, String separator, int maxSplits)
256       throws IllegalArgumentException {
257     if (separator.isEmpty()) {
258       throw new IllegalArgumentException("Empty separator");
259     }
260 
261     if (maxSplits <= 0) {
262       maxSplits = Integer.MAX_VALUE;
263     }
264 
265     LinkedList<String> result = new LinkedList<>();
266     String[] parts = input.split(Pattern.quote(separator), -1);
267     int sepLen = separator.length();
268     int remainingLength = input.length();    
269     int splitsSoFar = 0;
270 
271     // Copies parts from the array into the final list, starting at the end (because 
272     // it's rsplit), as long as fewer than maxSplits splits are performed. The 
273     // last spot in the list is reserved for the remaining string, whose length 
274     // has to be tracked throughout the loop.
275     for (int pos = parts.length - 1; (pos >= 0) && (splitsSoFar < maxSplits); --pos) {
276       String current = parts[pos];
277       result.addFirst(current);
278 
279       ++splitsSoFar;
280       remainingLength -= sepLen + current.length();
281     }
282 
283     if (splitsSoFar == maxSplits && remainingLength >= 0)   {
284       result.addFirst(input.substring(0, remainingLength));      
285     }
286     
287     return result;  
288   }

351   /**
352    * Splits the input string at the {first|last} occurrence of the given separator
353    * and returns the resulting partition as a three-tuple of Strings, contained 
354    * in a {@code List}.
355    * 
356    * <p>If the input string does not contain the separator, the tuple will
357    * consist of the original input string and two empty strings.
358    * 
359    * <p>This method emulates the behavior of Python's str.partition() and 
360    * str.rpartition(), depending on the value of the {@code forward} flag.
361    * 
362    * @param input The input string
363    * @param separator The string to split on
364    * @param forward A flag that controls whether the input string is split around 
365    *    the first ({@code true}) or last ({@code false}) occurrence of the separator. 
366    * @return A three-tuple (List) of the form [part_before_separator, separator, 
367    *    part_after_separator].
368    *    
369    */
370   private static List<String> stringPartition(String input, String separator, boolean forward)
371       throws IllegalArgumentException {
372     if (separator.isEmpty()) {
373       throw new IllegalArgumentException("Empty separator");
374     }
375 
376     int partitionSize = 3;
377     ArrayList<String> result = new ArrayList<>(partitionSize);
378     int pos = forward ? input.indexOf(separator) : input.lastIndexOf(separator);
379 
380     if (pos < 0) {
381       for (int i = 0; i < partitionSize; ++i) {
382         result.add("");
383       }
384 
385       // Following Python's implementation of str.partition() and str.rpartition(),
386       // the input string is copied to either the first or the last position in the
387       // list, depending on the value of the forward flag.
388       result.set(forward ? 0 : partitionSize - 1, input);
389     } else {
390       result.add(input.substring(0, pos));
391       result.add(separator);
392 
393       // pos + sep.length() is at most equal to input.length(). This worst-case
394       // happens when the separator is at the end of the input string. However,
395       // substring() will return an empty string in this scenario, thus making
396       // any additional safety checks obsolete.
397       result.add(input.substring(pos + separator.length()));
398     }
399 
400     return result;
401   }

411     int subpos = forward ? substr.indexOf(sub) : substr.lastIndexOf(sub);

413     return subpos < 0 ? subpos : subpos + start;
414   }
