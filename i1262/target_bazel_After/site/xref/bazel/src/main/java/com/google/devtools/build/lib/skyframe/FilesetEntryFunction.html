46    private static final class FilesetEntryFunctionException extends SkyFunctionException {
47      FilesetEntryFunctionException(RecursiveFilesystemTraversalException e) {
48        super(e, Transience.PERSISTENT);
49      }

52    @Override
53    public SkyValue compute(SkyKey key, Environment env) throws FilesetEntryFunctionException {
54      FilesetTraversalParams t = (FilesetTraversalParams) key.argument();
55      Preconditions.checkState(
56          t.getNestedTraversal().isPresent() != t.getDirectTraversal().isPresent(),
57          "Exactly one of the nested and direct traversals must be specified: %s", t);
58  
59      // Create the set of excluded files. Only top-level files can be excluded, i.e. ones that are
60      // directly under the root if the root is a directory.
61      Set<String> exclusions = createExclusionSet(t.getExcludedFiles());
62  
63      // The map of output symlinks. Each key is the path of a output symlink that the Fileset must
64      // create, relative to the Fileset.out directory, and each value specifies extra information
65      // about the link (its target, associated metadata and again its name).
66      Map<PathFragment, FilesetOutputSymlink> outputSymlinks = new LinkedHashMap<>();
67  
68      if (t.getNestedTraversal().isPresent()) {
69        // The "nested" traversal parameters are present if and only if FilesetEntry.srcdir specifies
70        // another Fileset (a "nested" one).
71        FilesetEntryValue nested = (FilesetEntryValue) env.getValue(
72            FilesetEntryValue.key(t.getNestedTraversal().get()));
73        if (env.valuesMissing()) {
74          return null;
75        }
76  
77        for (FilesetOutputSymlink s : nested.getSymlinks()) {
78          maybeStoreSymlink(s, t.getDestPath(), exclusions, outputSymlinks);
79        }
80      } else {
81        // The "nested" traversal params are absent if and only if the "direct" traversal params are
82        // present, which is the case when the FilesetEntry specifies a package's BUILD file, a
83        // directory or a list of files.
84  
85        // The root of the direct traversal is defined as follows.
86        //
87        // If FilesetEntry.files is specified, then a TraversalRequest is created for each entry, the
88        // root being the respective entry itself. These are all traversed for they may be
89        // directories or symlinks to directories, and we need to establish Skyframe dependencies on
90        // their contents for incremental correctness. If an entry is indeed a directory (but not when
91        // it's a symlink to one) then we have to create symlinks to each of their childen.
92        // (NB: there seems to be no good reason for this, it's just how legacy Fileset works. We may
93        // want to consider creating a symlink just for the directory and not for its child elements.)
94        //
95        // If FilesetEntry.files is not specified, then srcdir refers to either a BUILD file or a
96        // directory. For the former, the root will be the parent of the BUILD file. For the latter,
97        // the root will be srcdir itself.
98        DirectTraversal direct = t.getDirectTraversal().get();
99  
100       RecursiveFilesystemTraversalValue rftv;
101       try {
102         // Traverse the filesystem to establish skyframe dependencies.
103         rftv = traverse(env, createErrorInfo(t), direct);
104       } catch (MissingDepException e) {
105         return null;
106       }
107 
108       // The root can only be absent for the EMPTY rftv instance.
109       if (!rftv.getResolvedRoot().isPresent()) {
110         return FilesetEntryValue.EMPTY;
111       }
112 
113       ResolvedFile resolvedRoot = rftv.getResolvedRoot().get();
114 
115       // Handle dangling symlinks gracefully be returning empty results.
116       if (!resolvedRoot.type.exists()) {
117         return FilesetEntryValue.EMPTY;
118       }
119 
120       // The prefix to remove is the entire path of the root. This is OK:
121       // - when the root is a file, this removes the entire path, but the traversal's destination
122       //   path is actually the name of the output symlink, so this works out correctly
123       // - when the root is a directory or a symlink to one then we need to strip off the
124       //   directory's path from every result (this is how the output symlinks must be created)
125       //   before making them relative to the destination path
126       PathFragment prefixToRemove = direct.getRoot().getRelativePart();
127 
128       Iterable<ResolvedFile> results = null;
129 
130       if (direct.isRecursive()
131           || (resolvedRoot.type.isDirectory() && !resolvedRoot.type.isSymlink())) {
132         // The traversal is recursive (requested for an entire FilesetEntry.srcdir) or it was
133         // requested for a FilesetEntry.files entry which turned out to be a directory. We need to
134         // create an output symlink for every file in it and all of its subdirectories. Only
135         // exception is when the subdirectory is really a symlink to a directory -- no output
136         // shall be created for the contents of those.
137         // Now we create Dir objects to model the filesystem tree. The object employs a trick to
138         // find directory symlinks: directory symlinks have corresponding ResolvedFile entries and
139         // are added as files too, while their children, also added as files, contain the path of
140         // the parent. Finding and discarding the children is easy if we traverse the tree from
141         // root to leaf.
142         DirectoryTree root = new DirectoryTree();
143         for (ResolvedFile f : rftv.getTransitiveFiles().toCollection()) {
144           PathFragment path = f.getNameInSymlinkTree().relativeTo(prefixToRemove);
145           if (path.segmentCount() > 0) {
146             path = t.getDestPath().getRelative(path);
147             DirectoryTree dir = root;
148             for (int i = 0; i < path.segmentCount() - 1; ++i) {
149               dir = dir.addOrGetSubdir(path.getSegment(i));
150             }
151             dir.maybeAddFile(f);
152           }
153         }
154         // Here's where the magic happens. The returned iterable will yield all files in the
155         // directory that are not under symlinked directories, as well as all directory symlinks.
156         results = root.iterateFiles();
157       } else {
158         // If we're on this branch then the traversal was done for just one entry in
159         // FilesetEntry.files (which was not a directory, so it was either a file, a symlink to one
160         // or a symlink to a directory), meaning we'll have only one output symlink.
161         results = ImmutableList.of(resolvedRoot);
162       }
163 
164       // Create one output symlink for each entry in the results.
165       for (ResolvedFile f : results) {
166         PathFragment targetName;
167         try {
168           targetName = f.getTargetInSymlinkTree(direct.isFollowingSymlinks());
169         } catch (DanglingSymlinkException e) {
170           throw new FilesetEntryFunctionException(e);
171         }
172 
173         // Metadata field must be present. It can only be absent when stripped by tests.
174         String metadata = Integer.toHexString(f.metadata.get().hashCode());
175 
176         // The linkName has to be under the traversal's root, which is also the prefix to remove.
177         PathFragment linkName = f.getNameInSymlinkTree().relativeTo(prefixToRemove);
178         maybeStoreSymlink(linkName, targetName, metadata, t.getDestPath(), exclusions,
179             outputSymlinks);
180       }
181     }
182 
183     return FilesetEntryValue.of(ImmutableSet.copyOf(outputSymlinks.values()));
184   }

186   /** Stores an output symlink unless it's excluded or would overwrite an existing one. */
187   private static void maybeStoreSymlink(FilesetOutputSymlink nestedLink, PathFragment destPath,
188       Set<String> exclusions, Map<PathFragment, FilesetOutputSymlink> result) {
189     maybeStoreSymlink(nestedLink.name, nestedLink.target, nestedLink.metadata, destPath,
190         exclusions, result);
191   }
192 
193   /** Stores an output symlink unless it's excluded or would overwrite an existing one. */
194   private static void maybeStoreSymlink(PathFragment linkName, PathFragment linkTarget,
195       String metadata, PathFragment destPath, Set<String> exclusions,
196       Map<PathFragment, FilesetOutputSymlink> result) {
197     if (!exclusions.contains(linkName.getPathString())) {
198       linkName = destPath.getRelative(linkName);
199       if (!result.containsKey(linkName)) {
200         result.put(linkName, new FilesetOutputSymlink(linkName, linkTarget, metadata));
201       }
202     }
203   }

205   private static Set<String> createExclusionSet(Set<String> input) {
206     return Sets.filter(input, new Predicate<String>() {
207       @Override
208       public boolean apply(String e) {
209         // Keep the top-level exclusions only. Do not look for "/" but count the path segments
210         // instead, in anticipation of future Windows support.
211         return new PathFragment(e).segmentCount() == 1;
212       }
213     });
214   }

221   private RecursiveFilesystemTraversalValue traverse(Environment env, String errorInfo,
222       DirectTraversal traversal) throws MissingDepException {
223     SkyKey depKey = RecursiveFilesystemTraversalValue.key(
224         new RecursiveFilesystemTraversalValue.TraversalRequest(traversal.getRoot().asRootedPath(),
225             traversal.isGenerated(), traversal.getPackageBoundaryMode(), traversal.isPackage(),
226             errorInfo, /*pattern=*/null));
227     RecursiveFilesystemTraversalValue v = (RecursiveFilesystemTraversalValue) env.getValue(depKey);
228     if (env.valuesMissing()) {
229       throw new MissingDepException();
230     }
231     return v;
232   }

234   private static String createErrorInfo(FilesetTraversalParams t) {
235     if (t.getDirectTraversal().isPresent()) {
236       DirectTraversal direct = t.getDirectTraversal().get();
237       return String.format("Fileset '%s' traversing %s '%s'", t.getOwnerLabel(),
238           direct.isPackage() ? "package" : "file (or directory)",
239           direct.getRoot().getRelativePart().getPathString());
240     } else {
241       return String.format("Fileset '%s' traversing another Fileset", t.getOwnerLabel());
242     }
243   }

245   /**
246    * Models a FilesetEntryFunction's portion of the symlink output tree created by a Fileset rule.
247    *
248    * <p>A Fileset rule's output is computed by zero or more {@link FilesetEntryFunction}s, resulting
249    * in one {@link FilesetEntryValue} for each. Each of those represents a portion of the grand
250    * output tree of the Fileset. These portions are later merged and written to the fileset manifest
251    * file, which is then consumed by a tool that ultimately creates the symlinks in the filesystem.
252    *
253    * <p>Because the Fileset doesn't process the lists in the FilesetEntryValues any further than
254    * merging them, they have to adhere to the conventions of the manifest file. One of these is that
255    * files are alphabetically ordered (enables the consumer of the manifest to work faster than
256    * otherwise) and another is that the contents of regular directories are listed, but contents
257    * of directory symlinks are not, only the symlinks are. (Other details of the manifest file are
258    * not relevant here.)
259    *
260    * <p>See {@link DirectoryTree#iterateFiles()} for more details.
261    */
262   private static final class DirectoryTree {
263     // Use TreeMaps for the benefit of alphabetically ordered iteration.
264     public final Map<String, ResolvedFile> files = new TreeMap<>();
265     public final Map<String, DirectoryTree> dirs = new TreeMap<>();
266 
267     DirectoryTree addOrGetSubdir(String name) {
268       DirectoryTree result = dirs.get(name);
269       if (result == null) {
270         result = new DirectoryTree();
271         dirs.put(name, result);
272       }
273       return result;
274     }
275 
276     void maybeAddFile(ResolvedFile r) {
277       String name = r.getNameInSymlinkTree().getBaseName();
278       if (!files.containsKey(name)) {
279         files.put(name, r);
280       }
281     }
282 
283     /**
284      * Lazily yields all files in this directory and all of its subdirectories.
285      *
286      * <p>The function first yields all the files directly under this directory, in alphabetical
287      * order. Then come the contents of subdirectories, processed recursively in the same fashion
288      * as this directory, and also in alphabetical order.
289      *
290      * <p>If a directory symlink is encountered its contents are not listed, only the symlink is.
291      */
292     Iterable<ResolvedFile> iterateFiles() {
293       // 1. Filter directory symlinks. If the symlink target contains files, those were added
294       // as normal files so their parent directory (the symlink) would show up in the dirs map
295       // (as a directory) as well as in the files map (as a symlink to a directory).
296       final Set<String> fileNames = files.keySet();
297       Iterable<Map.Entry<String, DirectoryTree>> noDirSymlinkes = Iterables.filter(dirs.entrySet(),
298           new Predicate<Map.Entry<String, DirectoryTree>>() {
299             @Override
300             public boolean apply(Map.Entry<String, DirectoryTree> input) {
301               return !fileNames.contains(input.getKey());
302             }
303           });
304 
305       // 2. Extract the iterables of the true subdirectories.
306       Iterable<Iterable<ResolvedFile>> subdirIters = Iterables.transform(noDirSymlinkes,
307           new Function<Map.Entry<String, DirectoryTree>, Iterable<ResolvedFile>>() {
308             @Override
309             public Iterable<ResolvedFile> apply(Entry<String, DirectoryTree> input) {
310               return input.getValue().iterateFiles();
311             }
312           });
313 
314       // 3. Just concat all subdirectory iterations for one, seamless iteration.
315       Iterable<ResolvedFile> dirsIter = Iterables.concat(subdirIters);
316 
317       return Iterables.concat(files.values(), dirsIter);
318     }
319   }
