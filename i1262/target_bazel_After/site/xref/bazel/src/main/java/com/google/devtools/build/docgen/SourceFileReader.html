36  /**
37   * A helper class to read and process documentations for rule classes and attributes
38   * from exactly one java source file.
39   */
40  public class SourceFileReader {
41  
42    private Collection<RuleDocumentation> ruleDocEntries;
43    private ListMultimap<String, RuleDocumentationAttribute> attributeDocEntries;
44    private final ConfiguredRuleClassProvider ruleClassProvider;
45    private final String javaSourceFilePath;
46  
47    public SourceFileReader(
48        ConfiguredRuleClassProvider ruleClassProvider, String javaSourceFilePath) {
49      this.ruleClassProvider = ruleClassProvider;
50      this.javaSourceFilePath = javaSourceFilePath;
51    }

53    /**
54     * The handler class of the line read from the text file.
55     */
56    public abstract static class ReadAction {
57  
58      // Text file line indexing starts from 1
59      private int lineCnt = 1;
60  
61      protected abstract void readLineImpl(String line)
62          throws BuildEncyclopediaDocException, IOException;
63  
64      protected int getLineCnt() {
65        return lineCnt;
66      }
67  
68      public void readLine(String line)
69          throws BuildEncyclopediaDocException, IOException {
70        readLineImpl(line);
71        lineCnt++;
72      }
73    }
74  
75    private static final String LS = DocgenConsts.LS;
76  
77    /**
78     * Reads the attribute and rule documentation present in the file represented by
79     * SourceFileReader.javaSourceFilePath. The rule doc variables are added to the rule
80     * documentation (which therefore must be defined in the same file). The attribute docs are
81     * stored in a different class member, so they need to be handled outside this method.
82     */
83    public void readDocsFromComments() throws BuildEncyclopediaDocException, IOException {
84      final Map<String, RuleDocumentation> docMap = new HashMap<>();
85      final List<RuleDocumentationVariable> docVariables = new LinkedList<>();
86      final ListMultimap<String, RuleDocumentationAttribute> docAttributes =
87          LinkedListMultimap.create();
88      readTextFile(javaSourceFilePath, new ReadAction() {
89  
90        private boolean inBlazeRuleDocs = false;
91        private boolean inBlazeRuleVarDocs = false;
92        private boolean inBlazeAttributeDocs = false;
93        private StringBuilder sb = new StringBuilder();
94        private String ruleName;
95        private String ruleType;
96        private String ruleFamily;
97        private String variableName;
98        private String attributeName;
99        private ImmutableSet<String> flags;
100       private int startLineCnt;
101 
102       @Override
103       public void readLineImpl(String line) throws BuildEncyclopediaDocException {
104         // TODO(bazel-team): check if copy paste code can be reduced using inner classes
105         if (inBlazeRuleDocs) {
106           if (DocgenConsts.BLAZE_RULE_END.matcher(line).matches()) {
107             endBlazeRuleDoc(docMap);
108           } else {
109             appendLine(line);
110           }
111         } else if (inBlazeRuleVarDocs) {
112           if (DocgenConsts.BLAZE_RULE_VAR_END.matcher(line).matches()) {
113             endBlazeRuleVarDoc(docVariables);
114           } else {
115             appendLine(line);
116           }
117         } else if (inBlazeAttributeDocs) {
118           if (DocgenConsts.BLAZE_RULE_ATTR_END.matcher(line).matches()) {
119             endBlazeAttributeDoc(docAttributes);
120           } else {
121             appendLine(line);
122           }
123         }
124         Matcher ruleStartMatcher = DocgenConsts.BLAZE_RULE_START.matcher(line);
125         Matcher ruleVarStartMatcher = DocgenConsts.BLAZE_RULE_VAR_START.matcher(line);
126         Matcher ruleAttrStartMatcher = DocgenConsts.BLAZE_RULE_ATTR_START.matcher(line);
127         if (ruleStartMatcher.find()) {
128           startBlazeRuleDoc(line, ruleStartMatcher);
129         } else if (ruleVarStartMatcher.find()) {
130           startBlazeRuleVarDoc(ruleVarStartMatcher);
131         } else if (ruleAttrStartMatcher.find()) {
132           startBlazeAttributeDoc(line, ruleAttrStartMatcher);
133         }
134       }
135 
136       private void appendLine(String line) {
137         // Add another line of html code to the building rule documentation
138         // Removing whitespace and java comment asterisk from the beginning of the line
139         sb.append(line.replaceAll("^[\\s]*\\*", "") + LS);
140       }
141 
142       private void startBlazeRuleDoc(String line, Matcher matcher)
143           throws BuildEncyclopediaDocException {
144         checkDocValidity();
145         // Start of a new rule.
146         // e.g.: matcher.group(1) = "NAME = cc_binary, TYPE = BINARY, FAMILY = C / C++"
147         for (String group : Splitter.on(",").split(matcher.group(1))) {
148           List<String> parts = Splitter.on("=").limit(2).splitToList(group);
149           boolean good = false;
150           if (parts.size() == 2) {
151             String key = parts.get(0).trim();
152             String value = parts.get(1).trim();
153             good = true;
154             if (DocgenConsts.META_KEY_NAME.equals(key)) {
155               ruleName = value;
156             } else if (DocgenConsts.META_KEY_TYPE.equals(key)) {
157               ruleType = value;
158             } else if (DocgenConsts.META_KEY_FAMILY.equals(key)) {
159               ruleFamily = value;
160             } else {
161               good = false;
162             }
163           }
164           if (!good) {
165             System.err.printf("WARNING: bad rule definition in line %d: '%s'", getLineCnt(), line);
166           }
167         }
168 
169         startLineCnt = getLineCnt();
170         addFlags(line);
171         inBlazeRuleDocs = true;
172       }
173 
174       private void endBlazeRuleDoc(final Map<String, RuleDocumentation> documentations)
175           throws BuildEncyclopediaDocException {
176         // End of a rule, create RuleDocumentation object
177         documentations.put(ruleName, new RuleDocumentation(ruleName, ruleType,
178             ruleFamily, sb.toString(), getLineCnt(), javaSourceFilePath, flags,
179             ruleClassProvider));
180         sb = new StringBuilder();
181         inBlazeRuleDocs = false;
182       }
183 
184       private void startBlazeRuleVarDoc(Matcher matcher) throws BuildEncyclopediaDocException {
185         checkDocValidity();
186         // Start of a new rule variable
187         ruleName = matcher.group(1).replaceAll("[\\s]", "");
188         variableName = matcher.group(2).replaceAll("[\\s]", "");
189         startLineCnt = getLineCnt();
190         inBlazeRuleVarDocs = true;
191       }
192 
193       private void endBlazeRuleVarDoc(final List<RuleDocumentationVariable> docVariables) {
194         // End of a rule, create RuleDocumentationVariable object
195         docVariables.add(
196             new RuleDocumentationVariable(ruleName, variableName, sb.toString(), startLineCnt));
197         sb = new StringBuilder();
198         inBlazeRuleVarDocs = false;
199       }
200 
201       private void startBlazeAttributeDoc(String line, Matcher matcher)
202           throws BuildEncyclopediaDocException {
203         checkDocValidity();
204         // Start of a new attribute
205         ruleName = matcher.group(1).replaceAll("[\\s]", "");
206         attributeName = matcher.group(2).replaceAll("[\\s]", "");
207         startLineCnt = getLineCnt();
208         addFlags(line);
209         inBlazeAttributeDocs = true;
210       }
211 
212       private void endBlazeAttributeDoc(
213           final ListMultimap<String, RuleDocumentationAttribute> docAttributes) {
214         // End of a attribute, create RuleDocumentationAttribute object
215         docAttributes.put(attributeName, RuleDocumentationAttribute.create(
216             ruleClassProvider.getRuleClassDefinition(ruleName),
217             attributeName, sb.toString(), startLineCnt, flags));
218         sb = new StringBuilder();
219         inBlazeAttributeDocs = false;
220       }
221 
222       private void addFlags(String line) {
223         // Add flags if there's any
224         Matcher matcher = DocgenConsts.BLAZE_RULE_FLAGS.matcher(line);
225         if (matcher.find()) {
226           flags = ImmutableSet.<String>copyOf(matcher.group(1).split(","));
227         } else {
228           flags = ImmutableSet.<String>of();
229         }
230       }
231 
232       private void checkDocValidity() throws BuildEncyclopediaDocException {
233         if (inBlazeRuleDocs || inBlazeRuleVarDocs || inBlazeAttributeDocs) {
234           throw new BuildEncyclopediaDocException(javaSourceFilePath, getLineCnt(),
235               "Malformed documentation, #BLAZE_RULE started after another #BLAZE_RULE.");
236         }
237       }
238     });
239 
240     // Adding rule doc variables to the corresponding rules
241     for (RuleDocumentationVariable docVariable : docVariables) {
242       if (docMap.containsKey(docVariable.getRuleName())) {
243         docMap.get(docVariable.getRuleName()).addDocVariable(
244           docVariable.getVariableName(), docVariable.getValue());
245       } else {
246         throw new BuildEncyclopediaDocException(javaSourceFilePath, docVariable.getStartLineCnt(),
247             String.format("Malformed rule variable #BLAZE_RULE(%s).%s, rule %s not found in file.",
248                 docVariable.getRuleName(), docVariable.getVariableName(),
249                 docVariable.getRuleName()));
250       }
251     }
252     ruleDocEntries = docMap.values();
253     attributeDocEntries = docAttributes;
254   }
255 
256   public Collection<RuleDocumentation> getRuleDocEntries() {
257     return ruleDocEntries;
258   }
259 
260   public ListMultimap<String, RuleDocumentationAttribute> getAttributeDocEntries() {
261     return attributeDocEntries;
262   }

264   /**
265    * Reads the template file without variable substitution.
266    */
267   public static String readTemplateContents(String templateFilePath)
268       throws BuildEncyclopediaDocException, IOException {
269     return readTemplateContents(templateFilePath, null);
270   }
271 
272   /**
273    * Reads a template file and substitutes variables of the format ${FOO}.
274    *
275    * @param variables keys are the possible variable names, e.g. "FOO", values are the substitutions
276    *     (can be null)
277    */
278   public static String readTemplateContents(String templateFilePath,
279       final Map<String, String> variables) throws BuildEncyclopediaDocException, IOException {
280     final StringBuilder sb = new StringBuilder();
281     readTextFile(templateFilePath, new ReadAction() {
282       @Override
283       public void readLineImpl(String line) {
284         sb.append(expandVariables(line, variables)).append(LS);
285       }
286     });
287     return sb.toString();
288   }
289 
290   private static String expandVariables(String line, Map<String, String> variables) {
291     if (variables == null || line.indexOf("${") == -1) {
292       return line;
293     }
294 
295     for (Entry<String, String> variable : variables.entrySet()) {
296       line = line.replace("${" + variable.getKey() + "}", variable.getValue());
297     }
298     return line;
299   }
300 
301   public static void readTextFile(String filePath, ReadAction action)
302       throws BuildEncyclopediaDocException, IOException {
303     BufferedReader br = null;
304     try {
305       File file = new File(filePath);
306       if (file.exists()) {
307         br = new BufferedReader(new FileReader(file));
308       } else {
309         InputStream is = SourceFileReader.class.getResourceAsStream(filePath);
310         if (is != null) {
311           br = new BufferedReader(new InputStreamReader(is));
312         }
313       }
314       if (br != null) {
315         String line = null;
316         while ((line = br.readLine()) != null) {
317           action.readLine(line);
318         }
319       } else {
320         System.out.println("Couldn't find file or resource: " + filePath);
321       }
322     } finally {
323       if (br != null) {
324         br.close();
325       }
326     }
327   }
