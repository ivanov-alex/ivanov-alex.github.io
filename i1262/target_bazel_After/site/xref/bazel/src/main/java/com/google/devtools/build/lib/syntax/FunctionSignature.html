33  /**
34   * Function Signatures for BUILD language (same as Python)
35   *
36   * <p>Skylark's function signatures are just like Python3's.
37   * A function may have 6 kinds of arguments:
38   * positional mandatory, positional optional, positional rest (aka *star argument),
39   * key-only mandatory, key-only optional, key rest (aka **star_star argument).
40   * A caller may specify all arguments but the *star and **star_star arguments by name,
41   * and thus all mandatory and optional arguments are named arguments.
42   *
43   * <p>To enable various optimizations in the argument processing routine,
44   * we sort arguments according the following constraints, enabling corresponding optimizations:
45   * <ol>
46   * <li>The positional mandatories come just before the positional optionals,
47   *   so they can be filled in one go.
48   * <li>Positionals come first, so it's easy to prepend extra positional arguments such as "self"
49   *   to an argument list, and we optimize for the common case of no key-only mandatory parameters.
50   *   key-only parameters are thus grouped together.
51   *   positional mandatory and key-only mandatory parameters are separate,
52   *   but there no loop over a contiguous chunk of them, anyway.
53   * <li>The named are all grouped together, with star and star_star rest arguments coming last.
54   * <li>Mandatory arguments in each category (positional and named-only) come before the optional
55   *   arguments, for the sake of slightly better clarity to human implementers. This eschews an
56   *   optimization whereby grouping optionals together allows to iterate over them in one go instead
57   *   of two; however, this relatively minor optimization only matters when keyword arguments are
58   *   passed, at which point it is dwarfed by the slowness of keyword processing.
59   * </ol>
60   *
61   * <p>Parameters are thus sorted in the following obvious order:
62   * positional mandatory arguments (if any), positional optional arguments (if any),
63   * key-only mandatory arguments (if any), key-only optional arguments (if any),
64   * then star argument (if any), then star_star argument (if any).
65   */
66  @AutoValue
67  public abstract class FunctionSignature implements Serializable {
68  
69    /**
70     * The shape of a FunctionSignature, without names
71     */
72    @AutoValue
73    public abstract static class Shape implements Serializable {
74      private static final Interner<Shape> interner = Interners.newWeakInterner();
75  
76      /** Create a function signature */
77      public static Shape create(
78          int mandatoryPositionals,
79          int optionalPositionals,
80          int mandatoryNamedOnly,
81          int optionalNamedOnly,
82          boolean starArg,
83          boolean kwArg) {
84        Preconditions.checkArgument(
85            0 <= mandatoryPositionals && 0 <= optionalPositionals
86            && 0 <= mandatoryNamedOnly && 0 <= optionalNamedOnly);
87        return interner.intern(new AutoValue_FunctionSignature_Shape(
88            mandatoryPositionals, optionalPositionals,
89            mandatoryNamedOnly, optionalNamedOnly, starArg, kwArg));
90      }
91  
92      // These abstract getters specify the actual argument count fields to be defined by AutoValue.
93      /** number of mandatory positional arguments */
94      public abstract int getMandatoryPositionals();
95  
96      /** number of optional positional arguments */
97      public abstract int getOptionalPositionals();
98  
99      /** number of mandatory named-only arguments. */
100     public abstract int getMandatoryNamedOnly();
101 
102     /** number of optional named-only arguments */
103     public abstract int getOptionalNamedOnly();
104 
105     /** indicator for presence of a star argument for extra positional arguments */
106     public abstract boolean hasStarArg();
107 
108     /** indicator for presence of a star-star argument for extra keyword arguments */
109     public abstract boolean hasKwArg();
110 
111 
112     // The are computed argument counts
113     /** number of optional positional arguments. */
114     public int getPositionals() {
115       return getMandatoryPositionals() + getOptionalPositionals();
116     }
117 
118     /** number of optional named-only arguments. */
119     public int getNamedOnly() {
120       return getMandatoryNamedOnly() + getOptionalNamedOnly();
121     }
122 
123     /** number of optional arguments. */
124     public int getOptionals() {
125       return getOptionalPositionals() + getOptionalNamedOnly();
126     }
127 
128     /** total number of arguments */
129     public int getArguments() {
130       return getPositionals() + getNamedOnly() + (hasStarArg() ? 1 : 0) + (hasKwArg() ? 1 : 0);
131     }
132   }

153   /**
154    * Signatures proper.
155    *
156    * <p>A signature is a Shape and an ImmutableList of argument variable names
157    * NB: we assume these lists are short, so we may do linear scans.
158    */
159   public static FunctionSignature create(Shape shape, ImmutableList<String> names) {
160     Preconditions.checkArgument(names.size() == shape.getArguments());
161     return signatureInterner.intern(new AutoValue_FunctionSignature(shape, names(names)));

186   /**
187    * FunctionSignature.WithValues: also specifies a List of default values and types.
188    *
189    * <p>The lists can be null, which is an optimized path for specifying all null values.
190    *
191    * <p>Note that if some values can be null (for BuiltinFunction, not for UserDefinedFunction),
192    * you should use an ArrayList; otherwise, we recommend an ImmutableList.
193    *
194    * <p>V is the class of defaultValues and T is the class of types.
195    * When parsing a function definition at compile-time, they are &lt;Expression, Expression&gt;;
196    * when processing a @SkylarkSignature annotation at build-time, &lt;Object, SkylarkType&gt;.
197    */
198   @AutoValue
199   public abstract static class WithValues<V, T> implements Serializable {
200 
201     // The fields
202     /** The underlying signature with parameter shape and names */
203     public abstract FunctionSignature getSignature();
204 
205     /** The default values (if any) as a List of one per optional parameter.
206      * We might have preferred ImmutableList, but we care about
207      * supporting null's for some BuiltinFunction's, and we don't spit on speed.
208      */
209     @Nullable public abstract List<V> getDefaultValues();
210 
211     /** The parameter types (if specified) as a List of one per parameter, including * and **.
212      * We might have preferred ImmutableList, but we care about supporting null's
213      * so we can take shortcut for untyped values.
214      */
215     @Nullable public abstract List<T> getTypes();
216 
217 
218     /**
219      * Create a signature with (default and type) values.
220      * If you supply mutable List's, we trust that you won't modify them afterwards.
221      */
222     public static <V, T> WithValues<V, T> create(FunctionSignature signature,
223         @Nullable List<V> defaultValues, @Nullable List<T> types) {
224       Shape shape = signature.getShape();
225       Preconditions.checkArgument(defaultValues == null
226           || defaultValues.size() == shape.getOptionals());
227       Preconditions.checkArgument(types == null
228           || types.size() == shape.getArguments());
229       return new AutoValue_FunctionSignature_WithValues<>(signature, defaultValues, types);
230     }
231 
232     public static <V, T> WithValues<V, T> create(FunctionSignature signature,
233         @Nullable List<V> defaultValues) {
234       return create(signature, defaultValues, null);
235     }
236 
237     public static <V, T> WithValues<V, T> create(FunctionSignature signature,
238         @Nullable V[] defaultValues) {
239       return create(signature, Arrays.asList(defaultValues), null);
240     }
241 
242     public static <V, T> WithValues<V, T> create(FunctionSignature signature) {
243       return create(signature, null, null);
244     }

246     /**
247      * Parse a list of Parameter into a FunctionSignature.
248      *
249      * <p>To be used both by the Parser and by the SkylarkSignature annotation processor.
250      */
251     public static <V, T> WithValues<V, T> of(Iterable<Parameter<V, T>> parameters)
252         throws SignatureException {
253       int mandatoryPositionals = 0;
254       int optionalPositionals = 0;
255       int mandatoryNamedOnly = 0;
256       int optionalNamedOnly = 0;
257       boolean hasStarStar = false;
258       boolean hasStar = false;
259       @Nullable String star = null;
260       @Nullable String starStar = null;
261       @Nullable T starType = null;
262       @Nullable T starStarType = null;
263       ArrayList<String> params = new ArrayList<>();
264       ArrayList<V> defaults = new ArrayList<>();
265       ArrayList<T> types = new ArrayList<>();
266       // optional named-only parameters are kept aside to be spliced after the mandatory ones.
267       ArrayList<String> optionalNamedOnlyParams = new ArrayList<>();
268       ArrayList<T> optionalNamedOnlyTypes = new ArrayList<>();
269       ArrayList<V> optionalNamedOnlyDefaultValues = new ArrayList<>();
270       boolean defaultRequired = false; // true after mandatory positionals and before star.
271       Set<String> paramNameSet = new HashSet<>(); // set of names, to avoid duplicates
272 
273       for (Parameter<V, T> param : parameters) {
274         if (hasStarStar) {
275           throw new SignatureException("illegal parameter after star-star parameter", param);
276         }
277         @Nullable String name = param.getName();
278         @Nullable T type = param.getType();
279         if (param.hasName()) {
280           if (paramNameSet.contains(name)) {
281             throw new SignatureException("duplicate parameter name in function definition", param);
282           }
283           paramNameSet.add(name);
284         }
285         if (param.isStarStar()) {
286           hasStarStar = true;
287           starStar = name;
288           starStarType = type;
289         } else if (param.isStar()) {
290           if (hasStar) {
291             throw new SignatureException(
292                 "duplicate star parameter in function definition", param);
293           }
294           hasStar = true;
295           defaultRequired = false;
296           if (param.hasName()) {
297             star = name;
298             starType = type;
299           }
300         } else if (hasStar && param.isOptional()) {
301           optionalNamedOnly++;
302           optionalNamedOnlyParams.add(name);
303           optionalNamedOnlyTypes.add(type);
304           optionalNamedOnlyDefaultValues.add(param.getDefaultValue());
305         } else {
306           params.add(name);
307           types.add(type);
308           if (param.isOptional()) {
309             optionalPositionals++;
310             defaults.add(param.getDefaultValue());
311             defaultRequired = true;
312           } else if (hasStar) {
313             mandatoryNamedOnly++;
314           } else if (defaultRequired) {
315               throw new SignatureException(
316                   "a mandatory positional parameter must not follow an optional parameter",
317                   param);
318           } else {
319             mandatoryPositionals++;
320           }
321         }
322       }
323       params.addAll(optionalNamedOnlyParams);
324       types.addAll(optionalNamedOnlyTypes);
325       defaults.addAll(optionalNamedOnlyDefaultValues);
326 
327       if (star != null) {
328         params.add(star);
329         types.add(starType);
330       }
331       if (starStar != null) {
332         params.add(starStar);
333         types.add(starStarType);
334       }
335       return WithValues.<V, T>create(
336           FunctionSignature.create(
337               Shape.create(
338                   mandatoryPositionals, optionalPositionals,
339                   mandatoryNamedOnly, optionalNamedOnly,
340                   star != null, starStar != null),
341               ImmutableList.<String>copyOf(params)),
342           FunctionSignature.<V>valueListOrNull(defaults),
343           FunctionSignature.<T>valueListOrNull(types));
344     }

346     /**
347      * Append a representation of this signature to a string buffer.
348      */
349     public StringBuilder toStringBuilder(final StringBuilder sb) {
350       FunctionSignature signature = getSignature();
351       Shape shape = signature.getShape();
352       final ImmutableList<String> names = signature.getNames();
353       @Nullable final List<V> defaultValues = getDefaultValues();
354       @Nullable final List<T> types = getTypes();
355 
356       int mandatoryPositionals = shape.getMandatoryPositionals();
357       int optionalPositionals = shape.getOptionalPositionals();
358       int mandatoryNamedOnly = shape.getMandatoryNamedOnly();
359       int optionalNamedOnly = shape.getOptionalNamedOnly();
360       boolean starArg = shape.hasStarArg();
361       boolean kwArg = shape.hasKwArg();
362       int positionals = mandatoryPositionals + optionalPositionals;
363       int namedOnly = mandatoryNamedOnly + optionalNamedOnly;
364       int named = positionals + namedOnly;
365       int args = named + (starArg ? 1 : 0) + (kwArg ? 1 : 0);
366       int endMandatoryNamedOnly = positionals + mandatoryNamedOnly;
367       boolean hasStar = starArg || (namedOnly > 0);
368       int iStarArg = named;
369       int iKwArg = args - 1;
370 
371       class Show {
372         private boolean isMore = false;
373         private int j = 0;
374 
375         public void comma() {
376           if (isMore) { sb.append(", "); }
377           isMore = true;
378         }
379         public void type(int i) {
380           if (types != null && types.get(i) != null) {
381             sb.append(": ").append(types.get(i).toString());
382           }
383         }
384         public void mandatory(int i) {
385           comma();
386           sb.append(names.get(i));
387           type(i);
388         }
389         public void optional(int i) {
390           mandatory(i);
391           sb.append(" = ");
392           if (defaultValues == null) {
393             sb.append("?");
394           } else {
395             Printer.write(sb, defaultValues.get(j++));
396           }
397         }
398       };
399       Show show = new Show();
400 
401       int i = 0;
402       for (; i < mandatoryPositionals; i++) {
403         show.mandatory(i);
404       }
405       for (; i < positionals; i++) {
406         show.optional(i);
407       }
408       if (hasStar) {
409         show.comma();
410         sb.append("*");
411         if (starArg) {
412           sb.append(names.get(iStarArg));
413         }
414       }
415       for (; i < endMandatoryNamedOnly; i++) {
416         show.mandatory(i);
417       }
418       for (; i < named; i++) {
419         show.optional(i);
420       }
421       if (kwArg) {
422         show.comma();
423         sb.append("**");
424         sb.append(names.get(iKwArg));
425       }
426 
427       return sb;
428     }
429 
430     @Override
431     public String toString() {
432       StringBuilder sb = new StringBuilder();
433       toStringBuilder(sb);
434       return sb.toString();
435     }
436   }

438   /** The given List, or null if all the list elements are null. */
439   @Nullable public static <E> List<E> valueListOrNull(List<E> list) {
440     if (list == null) {
441       return null;
442     }
443     for (E value : list) {
444       if (value != null) {
445         return list;
446       }
447     }
448     return null;
449   }

451   /**
452    * Constructs a function signature (with names) from signature description and names.
453    * This method covers the general case.
454    * The number of optional named-only parameters is deduced from the other arguments.
455    *
456    * @param numMandatoryPositionals an int for the number of mandatory positional parameters
457    * @param numOptionalPositionals an int for the number of optional positional parameters
458    * @param numMandatoryNamedOnly an int for the number of mandatory named-only parameters
459    * @param starArg a boolean for whether there is a starred parameter
460    * @param kwArg a boolean for whether there is a star-starred parameter
461    * @param names an Array of String for the parameter names
462    * @return a FunctionSignature
463    */
464   public static FunctionSignature of(int numMandatoryPositionals, int numOptionalPositionals,
465       int numMandatoryNamedOnly, boolean starArg, boolean kwArg, String... names) {
466     return create(Shape.create(
467         numMandatoryPositionals,
468         numOptionalPositionals,
469         numMandatoryNamedOnly,
470         names.length - (kwArg ? 1 : 0) - (starArg ? 1 : 0)
471             - numMandatoryPositionals - numOptionalPositionals - numMandatoryNamedOnly,
472         starArg, kwArg),
473         ImmutableList.<String>copyOf(names));
474   }
