29  /**
30   * Rule definition for the java_binary rule.
31   */
32  public final class BazelJavaBinaryRule implements RuleDefinition {
33    @Override
34    public RuleClass build(Builder builder, RuleDefinitionEnvironment env) {
35      /* <!-- #BLAZE_RULE(java_binary).NAME -->
36      <br/>It is good practice to use the name of the source file that is the main entry point of the
37      application (minus the extension). For example, if your entry point is called
38      <code>Main.java</code>, then your name could be <code>Main</code>.
39      <!-- #END_BLAZE_RULE.NAME --> */
40      return builder
41          /* <!-- #BLAZE_RULE(java_binary).IMPLICIT_OUTPUTS -->
42          <ul>
43            <li><code><var>name</var>.jar</code>: A Java archive, containing the class files and other
44              resources corresponding to the binary's direct dependencies.</li>
45            <li><code><var>name</var>-src.jar</code>: An archive containing the sources ("source
46              jar").</li>
47            <li><code><var>name</var>_deploy.jar</code>: A Java archive suitable for deployment (only
48              built if explicitly requested).
49              <p>
50                Building the <code>&lt;<var>name</var>&gt;_deploy.jar</code> target for your rule
51                creates a self-contained jar file with a manifest that allows it to be run with the
52                <code>java -jar</code> command or with the wrapper script's <code>--singlejar</code>
53                option. Using the wrapper script is preferred to <code>java -jar</code> because it
54                also passes the <a href="#java_binary.jvm_flags">JVM flags</a> and the options to load
55                native libraries.
56              </p>
57              <p>
58                The deploy jar contains all the classes that would be found by a classloader that
59                searched the classpath from the binary's wrapper script from beginning to end. It also
60                contains the native libraries needed for dependencies. These are automatically loaded
61                into the JVM at runtime.
62              </p>
63            </li>
64            <li><code><var>name</var>_deploy-src.jar</code>: An archive containing the sources
65              collected from the transitive closure of the target. These will match the classes in the
66              <code>deploy.jar</code> except where jars have no matching source jar.</li>
67            <li><code><var>name</var>-gen.jar</code>: An archive containing the class files for
68              source files generated by annotation processors ("gen jar").</li>
69          </ul>
70          <!-- #END_BLAZE_RULE.IMPLICIT_OUTPUTS --> */
71          .setImplicitOutputsFunction(BazelJavaRuleClasses.JAVA_BINARY_IMPLICIT_OUTPUTS)
72          .override(attr("$is_executable", BOOLEAN).nonconfigurable("automatic").value(
73              new Attribute.ComputedDefault() {
74                @Override
75                public Object getDefault(AttributeMap rule) {
76                  return rule.get("create_executable", BOOLEAN);
77                }
78              }))
79          .build();
80    }
81  
82    @Override
83    public Metadata getMetadata() {
84      return RuleDefinition.Metadata.builder()
85          .name("java_binary")
86          .ancestors(BaseJavaBinaryRule.class, BazelBaseRuleClasses.BinaryBaseRule.class)
87          .factoryClass(BazelJavaBinary.class)
88          .build();
89    }
90  }

92  /*<!-- #BLAZE_RULE (NAME = java_binary, TYPE = BINARY, FAMILY = Java) -->
93  
94  ${ATTRIBUTE_SIGNATURE}
95  
96  <p>
97    Builds a Java archive ("jar file"), plus a wrapper shell script with the same name as the rule.
98    The wrapper shell script uses a classpath that includes, among other things, a jar file for each
99    library on which the binary depends.
100 </p>
101 <p>
102   The wrapper script accepts several unique flags. Refer to
103   <code>//src/main/java/com/google/devtools/build/lib/bazel/rules/java/java_stub_template.txt</code>
104   for a list of configurable flags and environment variables accepted by the wrapper.
105 </p>
106 
107 ${IMPLICIT_OUTPUTS}
108 
109 <p>
110   A <code>deps</code> attribute is not allowed in a <code>java_binary</code> rule without
111   <a href="#java_binary.srcs"><code>srcs</code></a>; such a rule requires a
112   <a href="#java_binary.main_class"><code>main_class</code></a> provided by
113   <a href="#java_binary.runtime_deps"><code>runtime_deps</code></a>.
114 </p>
115 
116 <p>The following code snippet illustrates a common mistake:</p>
117 
118 <pre class="code">
119 java_binary(
120     name = "DontDoThis",
121     srcs = [
122         <var>...</var>,
123         <code class="deprecated">"GeneratedJavaFile.java"</code>,  # a generated .java file
124     ],
125     deps = [<code class="deprecated">":generating_rule",</code>],  # rule that generates that file
126 )
127 </pre>
128 
129 <p>Do this instead:</p>
130 
131 <pre class="code">
132 java_binary(
133     name = "DoThisInstead",
134     srcs = [
135         <var>...</var>,
136         ":generating_rule",
137     ],
138 )
139 </pre>
140 
141 ${ATTRIBUTE_DEFINITION}
142 
143 <!-- #END_BLAZE_RULE -->*/
