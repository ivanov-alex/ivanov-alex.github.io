25  /**
26   * Options that will be evaluated by the blaze client startup code and passed
27   * to the blaze server upon startup.
28   *
29   * <h4>IMPORTANT</h4> These options and their defaults must be kept in sync with those in the
30   * source of the launcher.  The latter define the actual default values; this class exists only to
31   * provide the help message, which displays the default values.
32   *
33   * The same relationship holds between {@link HostJvmStartupOptions} and the launcher.
34   */
35  public class BlazeServerStartupOptions extends OptionsBase {
36    /**
37     * Converter for the <code>option_sources</code> option. Takes a string in the form of
38     * "option_name1:source1:option_name2:source2:.." and converts it into an option name to
39     * source map.
40     */
41    public static class OptionSourcesConverter implements Converter<Map<String, String>> {
42      private String unescape(String input) {
43        return input.replace("_C", ":").replace("_U", "_");
44      }
45  
46      @Override
47      public Map<String, String> convert(String input) {
48        ImmutableMap.Builder<String, String> builder = ImmutableMap.builder();
49        if (input.isEmpty()) {
50          return builder.build();
51        }
52  
53        String[] elements = input.split(":");
54        for (int i = 0; i < (elements.length + 1) / 2; i++) {
55          String name = elements[i * 2];
56          String value = "";
57          if (elements.length > i * 2 + 1) {
58            value = elements[i * 2 + 1];
59          }
60          builder.put(unescape(name), unescape(value));
61        }
62        return builder.build();
63      }
64  
65      @Override
66      public String getTypeDescription() {
67        return "a list of option-source pairs";
68      }
69    }

71    /* Passed from the client to the server, specifies the installation
72     * location. The location should be of the form:
73     * $OUTPUT_BASE/_blaze_${USER}/install/${MD5_OF_INSTALL_MANIFEST}.
74     * The server code will only accept a non-empty path; it's the
75     * responsibility of the client to compute a proper default if
76     * necessary.
77     */
78    @Option(name = "install_base",
79        defaultValue = "", // NOTE: purely decorative!  See class docstring.
80        category = "hidden",
81        converter = OptionsUtils.PathFragmentConverter.class,
82        help = "This launcher option is intended for use only by tests.")
83    public PathFragment installBase;
84  
85    /* Note: The help string in this option applies to the client code; not
86     * the server code. The server code will only accept a non-empty path; it's
87     * the responsibility of the client to compute a proper default if
88     * necessary.
89     */
90    @Option(name = "output_base",
91        defaultValue = "null", // NOTE: purely decorative!  See class docstring.
92        category = "server startup",
93        converter = OptionsUtils.PathFragmentConverter.class,
94        help = "If set, specifies the output location to which all build output will be written. "
95            + "Otherwise, the location will be "
96            + "${OUTPUT_ROOT}/_blaze_${USER}/${MD5_OF_WORKSPACE_ROOT}. Note: If you specify a "
97            + "different option from one to the next Blaze invocation for this value, you'll likely "
98            + "start up a new, additional Blaze server. Blaze starts exactly one server per "
99            + "specified output base. Typically there is one output base per workspace--however, "
100           + "with this option you may have multiple output bases per workspace and thereby run "
101           + "multiple builds for the same client on the same machine concurrently. See "
102           + "'blaze help shutdown' on how to shutdown a Blaze server.")
103   public PathFragment outputBase;
104 
105   /* Note: This option is only used by the C++ client, never by the Java server.
106    * It is included here to make sure that the option is documented in the help
107    * output, which is auto-generated by Java code.
108    */
109   @Option(name = "output_user_root",
110       defaultValue = "null", // NOTE: purely decorative!  See class docstring.
111       category = "server startup",
112       converter = OptionsUtils.PathFragmentConverter.class,
113       help = "The user-specific directory beneath which all build outputs are written; "
114           + "by default, this is a function of $USER, but by specifying a constant, build outputs "
115           + "can be shared between collaborating users.")
116   public PathFragment outputUserRoot;
117 
118   @Option(name = "workspace_directory",
119       defaultValue = "",
120       category = "hidden",
121       converter = OptionsUtils.PathFragmentConverter.class,
122       help = "The root of the workspace, that is, the directory that Blaze uses as the root of the "
123           + "build. This flag is only to be set by the blaze client.")

126   @Option(name = "max_idle_secs",
127       defaultValue = "" + (3 * 3600), // NOTE: purely decorative!  See class docstring.
128       category = "server startup",
129       help = "The number of seconds the build server will wait idling " +
130              "before shutting down. Note: Blaze will ignore this option " +
131              "unless you are starting a new instance. See also 'blaze help " +
132              "shutdown'.")
133   public int maxIdleSeconds;

135   @Option(name = "batch",
136       defaultValue = "false", // NOTE: purely decorative!  See class docstring.
137       category = "server startup",
138       help = "If set, Blaze will be run in batch mode, instead of " +
139              "the standard client/server. Doing so may provide " +
140              "more predictable semantics with respect to signal handling and job control, " +
141              "Batch mode retains proper queueing semantics within the same output_base. " +
142              "That is, simultaneous invocations will be processed in order, without overlap. " +
143              "If a batch mode Blaze is run on a client with a running server, it first kills "  +
144              "the server before processing the command." +
145              "Blaze will run slower in batch mode, compared to client/server mode. " +
146              "Among other things, the build file cache is memory-resident, so it is not " +
147              "preserved between sequential batch invocations. Therefore, using batch mode " +
148              "often makes more sense in cases where performance is less critical, " +
149              "such as continuous builds.")
150   public boolean batch;
