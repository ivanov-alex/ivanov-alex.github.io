49  /**
50   * An object that combines multiple ZIP files into a single file. It only
51   * supports a subset of the ZIP format, specifically:
52   * <ul>
53   *   <li>It only supports STORE and DEFLATE storage methods.</li>
54   *   <li>It only supports 32-bit ZIP files.</li>
55   * </ul>
56   *
57   * <p>These restrictions are also present in the JDK implementations
58   * {@link java.util.jar.JarInputStream}, {@link java.util.zip.ZipInputStream},
59   * though they are not documented there.
60   *
61   * <p>IMPORTANT NOTE: Callers must call {@link #finish()} or {@link #close()}
62   * at the end of processing to ensure that the output buffers are flushed and
63   * the ZIP file is complete.
64   *
65   * <p>This class performs only rudimentary data checking. If the input files
66   * are damaged, the output will likely also be damaged.
67   *
68   * <p>Also see:
69   * <a href="http://www.pkware.com/documents/casestudies/APPNOTE.TXT">ZIP format</a>
70   */
71  public class ZipCombiner implements AutoCloseable {
72    public static final Date DOS_EPOCH = new Date(ZipUtil.DOS_EPOCH);
73    /**
74     * Whether to compress or decompress entries.
75     */
76    public enum OutputMode {
77  
78      /**
79       * Output entries using any method.
80       */
81      DONT_CARE,
82  
83      /**
84       * Output all entries using DEFLATE method, except directory entries. It is always more
85       * efficient to store directory entries uncompressed.
86       */
87      FORCE_DEFLATE,
88  
89      /**
90       * Output all entries using STORED method.
91       */
92      FORCE_STORED,
93    }

95    /**
96     * The type of action to take for a ZIP file entry.
97     */
98    private enum ActionType {
99  
100     /**
101      * Skip the entry.
102      */
103     SKIP,
104 
105     /**
106      * Copy the entry.
107      */
108     COPY,
109 
110     /**
111      * Rename the entry.
112      */
113     RENAME,
114 
115     /**
116      * Merge the entry.
117      */
118     MERGE;
119   }

121   /**
122    * Encapsulates the action to take for a ZIP file entry along with optional details specific to
123    * the action type. The minimum requirements per type are:
124    * <ul>
125    *    <li>SKIP: none.</li>
126    *    <li>COPY: none.</li>
127    *    <li>RENAME: newName.</li>
128    *    <li>MERGE: strategy, mergeBuffer.</li>
129    * </ul>
130    *
131    * <p>An action can be easily changed from one type to another by using
132    * {@link EntryAction#EntryAction(ActionType, EntryAction)}.
133    */
134   private static final class EntryAction {
135     private final ActionType type;
136     @Nullable private final Date date;
137     @Nullable private final String newName;
138     @Nullable private final CustomMergeStrategy strategy;
139     @Nullable private final ByteArrayOutputStream mergeBuffer;
140 
141     /**
142      * Create an action of the specified type with no extra details.
143      */
144     public EntryAction(ActionType type) {
145       this(type, null, null, null, null);
146     }
147 
148     /**
149      * Create a duplicate action with a different {@link ActionType}.
150      */
151     public EntryAction(ActionType type, EntryAction action) {
152       this(type, action.getDate(), action.getNewName(), action.getStrategy(),
153           action.getMergeBuffer());
154     }
155 
156     /**
157      * Create an action of the specified type and details.
158      *
159      * @param type the type of action
160      * @param date the custom date to set on the entry
161      * @param newName the custom name to create the entry as
162      * @param strategy the {@link CustomMergeStrategy} to use for merging this entry
163      * @param mergeBuffer the output stream to use for merge results
164      */
165     public EntryAction(ActionType type, Date date, String newName, CustomMergeStrategy strategy,
166         ByteArrayOutputStream mergeBuffer) {
167       checkArgument(type != ActionType.RENAME || newName != null,
168           "NewName must not be null if the ActionType is RENAME.");
169       checkArgument(type != ActionType.MERGE || strategy != null,
170           "Strategy must not be null if the ActionType is MERGE.");
171       checkArgument(type != ActionType.MERGE || mergeBuffer != null,
172           "MergeBuffer must not be null if the ActionType is MERGE.");
173       this.type = type;
174       this.date = date;
175       this.newName = newName;
176       this.strategy = strategy;
177       this.mergeBuffer = mergeBuffer;
178     }
179 
180     /** Returns the type. */
181     public ActionType getType() {
182       return type;
183     }
184 
185     /** Returns the date. */
186     public Date getDate() {
187       return date;
188     }
189 
190     /** Returns the new name. */
191     public String getNewName() {
192       return newName;
193     }
194 
195     /** Returns the strategy. */
196     public CustomMergeStrategy getStrategy() {
197       return strategy;
198     }
199 
200     /** Returns the mergeBuffer. */
201     public ByteArrayOutputStream getMergeBuffer() {
202       return mergeBuffer;
203     }
204   }

206   private final class FilterCallback implements StrategyCallback {
207     private String filename;
208     private final AtomicBoolean called = new AtomicBoolean();
209 
210     public void resetForFile(String filename) {
211       this.filename = filename;
212       this.called.set(false);
213     }
214 
215     @Override public void skip() throws IOException {
216       checkCall();
217       actions.put(filename, new EntryAction(ActionType.SKIP));
218     }
219 
220     @Override public void copy(Date date) throws IOException {
221       checkCall();
222       actions.put(filename, new EntryAction(ActionType.COPY, date, null, null, null));
223     }
224 
225     @Override public void rename(String newName, Date date) throws IOException {
226       checkCall();
227       actions.put(filename, new EntryAction(ActionType.RENAME, date, newName, null, null));
228     }
229 
230     @Override public void customMerge(Date date, CustomMergeStrategy strategy) throws IOException {
231       checkCall();
232       actions.put(filename, new EntryAction(ActionType.MERGE, date, null, strategy,
233           new ByteArrayOutputStream()));
234     }
235 
236     private void checkCall() {
237       checkState(called.compareAndSet(false, true), "The callback was already called once.");
238     }
239   }

251   /** Copies all data from the input stream to the output stream. */
252   private static long copyStream(InputStream from, OutputStream to) throws IOException {
253     byte[] buf = new byte[0x1000];
254     long total = 0;
255     int r;
256     while ((r = from.read(buf)) != -1) {
257       to.write(buf, 0, r);
258       total += r;
259     }
260     return total;
261   }

346   /**
347    * Adds a directory entry to the combined ZIP file using the specified filename, date, and extra
348    * data.
349    *
350    * @param filename the name of the directory to create
351    * @param date the modified time to assign to the directory
352    * @param extra the extra field data to add to the directory entry
353    * @throws IOException if there is an error writing the directory entry
354    */
355   public void addDirectory(String filename, Date date, ExtraData[] extra) throws IOException {
356     checkArgument(filename.endsWith("/"), "Directory names must end with a /");
357     checkState(!entries.containsKey(filename),
358         "Zip already contains a directory named %s", filename);
359 
360     ZipFileEntry entry = new ZipFileEntry(filename);
361     entry.setMethod(Compression.STORED);
362     entry.setCrc(0);
363     entry.setSize(0);
364     entry.setCompressedSize(0);
365     entry.setTime(date != null ? date.getTime() : new Date().getTime());
366     entry.setExtra(new ExtraDataList(extra));
367     out.putNextEntry(entry);
368     out.closeEntry();
369     entries.put(filename, entry);
370   }

401   /**
402    * Adds a file with attributes specified by the {@link ZipFileEntry} to the combined ZIP file.
403    *
404    * @param entry the {@link ZipFileEntry} containing the entry meta-data
405    * @param in the {@link InputStream} containing the file data
406    * @throws IOException if there is an error writing the file entry
407    * @throws IllegalArgumentException if the combined ZIP file already contains a file of the same
408    *     name.
409    */
410   public void addFile(ZipFileEntry entry, InputStream in) throws IOException {
411     checkNotNull(entry, "Zip entry must not be null.");
412     checkNotNull(in, "Input stream must not be null.");
413     checkArgument(!entries.containsKey(entry.getName()), "Zip already contains a file named '%s'.",
414         entry.getName());
415 
416     ByteArrayOutputStream uncompressed = new ByteArrayOutputStream();
417     copyStream(in, uncompressed);
418 
419     writeEntryFromBuffer(new ZipFileEntry(entry), uncompressed.toByteArray());
420   }

422   /**
423    * Adds the contents of a ZIP file to the combined ZIP file using the specified
424    * {@link ZipEntryFilter} to determine the appropriate action for each file. 
425    *
426    * @param zipFile the ZIP file to add to the combined ZIP file
427    * @throws IOException if there is an error reading the ZIP file or writing entries to the
428    *     combined ZIP file
429    */
430   public void addZip(File zipFile) throws IOException {
431     try (ZipReader zip = new ZipReader(zipFile)) {
432       for (ZipFileEntry entry : zip.entries()) {
433         String filename = entry.getName();
434         EntryAction action = getAction(filename);
435         switch (action.getType()) {
436           case SKIP:
437             break;
438           case COPY:
439           case RENAME:
440             writeEntry(zip, entry, action);
441             break;
442           case MERGE:
443             entries.put(filename, null);
444             InputStream in = zip.getRawInputStream(entry);
445             if (entry.getMethod() == Compression.DEFLATED) {
446               in = new InflaterInputStream(in, getInflater());
447             }
448             action.getStrategy().merge(in, action.getMergeBuffer());
449             break;
450         }
451       }
452     }
453   }

455   /** Returns the action to take for a file of the given filename. */
456   private EntryAction getAction(String filename) throws IOException {
457     // If this filename has not been encountered before (no entry for filename) or this filename
458     // has been renamed (RENAME entry for filename), the desired action should be recomputed.
459     if (!actions.containsKey(filename) || actions.get(filename).getType() == ActionType.RENAME) {
460       callback.resetForFile(filename);
461       entryFilter.accept(filename, callback);
462     }
463     checkState(actions.containsKey(filename),
464         "Action for file '%s' should have been set by ZipEntryFilter.", filename);
465 
466     EntryAction action = actions.get(filename);
467     // Only copy if this is the first instance of filename.
468     if (action.getType() == ActionType.COPY && entries.containsKey(filename)) {
469       action = new EntryAction(ActionType.SKIP, action);
470       actions.put(filename, action);
471     }
472     // Only rename if there is not already an entry with filename or filename's action is SKIP.
473     if (action.getType() == ActionType.RENAME) {
474       if (actions.containsKey(action.getNewName())
475           && actions.get(action.getNewName()).getType() == ActionType.SKIP) {
476         action = new EntryAction(ActionType.SKIP, action);
477       }
478       if (entries.containsKey(action.getNewName())) {
479         action = new EntryAction(ActionType.SKIP, action);
480       }
481     }
482     return action;
483   }

485   /** Writes an entry with the given name, date and external file attributes from the buffer. */
486   private void writeEntryFromBuffer(ZipFileEntry entry, byte[] uncompressed) throws IOException {
487     CRC32 crc = new CRC32();
488     crc.update(uncompressed);
489 
490     entry.setCrc(crc.getValue());
491     entry.setSize(uncompressed.length);
492     if (mode == OutputMode.FORCE_STORED) {
493       entry.setMethod(Compression.STORED);
494       entry.setCompressedSize(uncompressed.length);
495       writeEntry(entry, new ByteArrayInputStream(uncompressed));
496     } else {
497       ByteArrayOutputStream compressed = new ByteArrayOutputStream();
498       copyStream(new DeflaterInputStream(new ByteArrayInputStream(uncompressed), getDeflater()),
499           compressed);
500       entry.setMethod(Compression.DEFLATED);
501       entry.setCompressedSize(compressed.size());
502       writeEntry(entry, new ByteArrayInputStream(compressed.toByteArray()));
503     }
504   }

506   /**
507    * Writes an entry from the specified source {@link ZipReader} and {@link ZipFileEntry} using the
508    * specified {@link EntryAction}.
509    * 
510    *  <p>Writes the output entry from the input entry performing inflation or deflation as needed
511    *  and applies any values from the {@link EntryAction} as needed.
512    */
513   private void writeEntry(ZipReader zip, ZipFileEntry entry, EntryAction action)
514       throws IOException {
515     checkArgument(action.getType() != ActionType.SKIP,
516         "Cannot write a zip entry whose action is of type SKIP.");
517 
518     ZipFileEntry outEntry = new ZipFileEntry(entry);
519     if (action.getType() == ActionType.RENAME) {
520       checkNotNull(action.getNewName(),
521           "ZipEntryFilter actions of type RENAME must not have a null filename.");
522       outEntry.setName(action.getNewName());
523     }
524 
525     if (action.getDate() != null) {
526       outEntry.setTime(action.getDate().getTime());
527     }
528 
529     InputStream data;
530     if (mode == OutputMode.FORCE_DEFLATE && entry.getMethod() != Compression.DEFLATED) {
531       // The output mode is deflate, but the entry compression is not. Create a deflater stream
532       // from the raw file data and deflate to a temporary byte array to determine the deflated
533       // size. Then use this byte array as the input stream for writing the entry.
534       ByteArrayOutputStream tmp = new ByteArrayOutputStream();
535       copyStream(new DeflaterInputStream(zip.getRawInputStream(entry), getDeflater()), tmp);
536       data = new ByteArrayInputStream(tmp.toByteArray());
537       outEntry.setMethod(Compression.DEFLATED);
538       outEntry.setCompressedSize(tmp.size());
539     } else if (mode == OutputMode.FORCE_STORED && entry.getMethod() != Compression.STORED) {
540       // The output mode is stored, but the entry compression is not; create an inflater stream
541       // from the raw file data. 
542       data = new InflaterInputStream(zip.getRawInputStream(entry), getInflater());
543       outEntry.setMethod(Compression.STORED);
544       outEntry.setCompressedSize(entry.getSize());
545     } else {
546       // Entry compression agrees with output mode; use the raw file data as is.
547       data = zip.getRawInputStream(entry);
548     }
549     writeEntry(outEntry, data);
550   }

552   /**
553    * Writes the specified {@link ZipFileEntry} using the data from the given {@link InputStream}.
554    */
555   private void writeEntry(ZipFileEntry entry, InputStream data) throws IOException {
556     out.putNextEntry(entry);
557     copyStream(data, out);
558     out.closeEntry();
559     entries.put(entry.getName(), entry);
560   }

574   /**
575    * Writes any remaining output data to the output stream and also creates the merged entries by
576    * calling the {@link CustomMergeStrategy} implementations given back from the
577    * {@link ZipEntryFilter}.
578    *
579    * @throws IOException if the output stream or the filter throws an IOException
580    * @throws IllegalStateException if this method was already called earlier
581    */
582   public void finish() throws IOException {
583     for (Entry<String, EntryAction> entry : actions.entrySet()) {
584       String filename = entry.getKey();
585       EntryAction action = entry.getValue();
586       if (action.getType() == ActionType.MERGE) {
587         ByteArrayOutputStream uncompressed = action.getMergeBuffer();
588         action.getStrategy().finish(uncompressed);
589 
590         ZipFileEntry e = new ZipFileEntry(filename);
591         e.setTime(action.getDate() != null ? action.getDate().getTime() : new Date().getTime());
592         writeEntryFromBuffer(e, uncompressed.toByteArray());
593       }
594     }
595     out.finish();
596   }

598   /**
599    * Writes any remaining output data to the output stream and closes it.
600    *
601    * @throws IOException if the output stream or the filter throws an IOException
602    */
603   @Override public void close() throws IOException {
604     finish();
605     out.close();
606   }
