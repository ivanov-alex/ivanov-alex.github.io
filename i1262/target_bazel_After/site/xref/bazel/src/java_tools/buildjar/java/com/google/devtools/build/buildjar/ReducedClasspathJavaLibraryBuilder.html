15  package com.google.devtools.build.buildjar;
16  
17  import com.google.devtools.build.buildjar.javac.JavacRunner;
18  
19  import com.sun.tools.javac.main.Main.Result;
20  
21  import java.io.IOException;
22  import java.io.PrintWriter;
23  import java.io.StringWriter;
24  
25  /**
26   * A variant of SimpleJavaLibraryBuilder that attempts to reduce the compile-time classpath right
27   * before invoking the compiler, based on extra information from provided .jdeps files. This mode is
28   * enabled via the --reduce_classpath flag, only when Blaze runs with --experimental_java_classpath.
29   *
30   * <p>A fall-back mechanism detects whether javac fails because the classpath is incorrectly
31   * discarding required entries, and re-attempts to compile with the full classpath.
32   */
33  public class ReducedClasspathJavaLibraryBuilder extends SimpleJavaLibraryBuilder {
34  
35    /**
36     * Attempts to minimize the compile-time classpath before invoking javac, falling back to a
37     * regular compile.
38     *
39     * @param build A JavaLibraryBuildRequest request object describing what to compile
40     * @return result code of the javac compilation
41     * @throws IOException clean-up up the output directory fails
42     */
43    @Override
44    Result compileSources(JavaLibraryBuildRequest build, JavacRunner javacRunner, PrintWriter err)
45        throws IOException {
46      // Minimize classpath, but only if we're actually compiling some sources (some invocations of
47      // JavaBuilder are only building resource jars).
48      String compressedClasspath = build.getClassPath();
49      if (!build.getSourceFiles().isEmpty()) {
50        compressedClasspath = build.getDependencyModule()
51            .computeStrictClasspath(build.getClassPath(), build.getClassDir());
52      }
53      String[] javacArguments = makeJavacArguments(build, compressedClasspath);
54  
55      // Compile!
56      StringWriter javacOutput = new StringWriter();
57      PrintWriter javacOutputWriter = new PrintWriter(javacOutput);
58      Result result = javacRunner.invokeJavac(javacArguments, javacOutputWriter);
59      javacOutputWriter.close();
60  
61      // If javac errored out because of missing entries on the classpath, give it another try.
62      // TODO(bazel-team): check performance impact of additional retries.
63      if (!result.isOK() && hasRecognizedError(javacOutput.toString())) {
64        if (debug) {
65          err.println("warning: [transitive] Target uses transitive classpath to compile.");
66        }
67  
68        // Reset output directories
69        prepareSourceCompilation(build);
70  
71        // Fall back to the regular compile, but add extra checks to catch transitive uses
72        javacArguments = makeJavacArguments(build);
73        result = javacRunner.invokeJavac(javacArguments, err);
74      } else {
75        err.print(javacOutput.getBuffer());
76      }
77      return result;
78    }
79    
80    private boolean hasRecognizedError(String javacOutput) {
81      return javacOutput.contains("error: cannot access")
82          || javacOutput.contains("error: cannot find symbol")
83          || javacOutput.contains("com.sun.tools.javac.code.Symbol$CompletionFailure");
84    }
85  }
