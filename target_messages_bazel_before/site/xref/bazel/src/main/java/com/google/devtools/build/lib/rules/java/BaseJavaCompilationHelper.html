37   * A helper class for compiling Java targets. This helper does not rely on the
38   * presence of rule-specific attributes.

41    /**
42     * Also see DeployArchiveBuilder.SINGLEJAR_MAX_MEMORY. We don't expect that anyone has more
43     * than ~500,000 files in a source jar, so 256 MB of memory should be plenty.
44     */
45    private static final String SINGLEJAR_MAX_MEMORY = "-Xmx256m";

49    public BaseJavaCompilationHelper(RuleContext ruleContext) {

53    /**
54     * Returns the artifacts required to invoke {@code javahome} relative binary
55     * in the action.
56     */
57    public static NestedSet<Artifact> getHostJavabaseInputs(RuleContext ruleContext) {
58      // This must have a different name than above, because the middleman creation uses the rule's
59      // configuration, although it should use the host configuration.
60      return AnalysisUtils.getMiddlemanFor(ruleContext, ":host_jdk");
61    }

63    private static final ImmutableList<String> SOURCE_JAR_COMMAND_LINE_ARGS = ImmutableList.of(
64        "--compression",
65        "--normalize",
66        "--exclude_build_data",
67        "--warn_duplicate_resources");
68  
69    private CommandLine sourceJarCommandLine(JavaSemantics semantics, Artifact outputJar,
70        Iterable<Artifact> resources, Iterable<Artifact> resourceJars) {
71      CustomCommandLine.Builder args = CustomCommandLine.builder();
72      args.addExecPath("--output", outputJar);
73      args.add(SOURCE_JAR_COMMAND_LINE_ARGS);
74      args.addExecPaths("--sources", resourceJars);
75      args.add("--resources");
76      for (Artifact resource : resources) {
77        args.addPaths("%s:%s", resource.getExecPath(),
78            semantics.getJavaResourcePath(resource.getRootRelativePath()));
79      }
80      return args.build();
81    }

83    /**
84     * Creates an Action that packages files into a Jar file.
85     *
86     * @param semantics delegate semantics for java.
87     * @param resources the resources to put into the Jar.
88     * @param resourceJars the resource jars to merge into the jar
89     * @param outputJar the Jar to create
90     */
91    public void createSourceJarAction(JavaSemantics semantics, Collection<Artifact> resources,
92        Collection<Artifact> resourceJars, Artifact outputJar) {
93      ruleContext.registerAction(new SpawnAction.Builder()
94          .addOutput(outputJar)
95          .addInputs(resources)
96          .addInputs(resourceJars)
97          .addTransitiveInputs(JavaCompilationHelper.getHostJavabaseInputs(ruleContext))
98          .setJarExecutable(
99              ruleContext.getHostConfiguration().getFragment(Jvm.class).getJavaExecutable(),
100             ruleContext.getPrerequisiteArtifact("$singlejar", Mode.HOST),
101             ImmutableList.of("-client", SINGLEJAR_MAX_MEMORY))
102         .setCommandLine(sourceJarCommandLine(semantics, outputJar, resources, resourceJars))
103         .useParameterFile(ParameterFileType.SHELL_QUOTED)
104         .setProgressMessage("Building source jar " + outputJar.prettyPrint())
105         .setMnemonic("JavaSourceJar")
106         .build(ruleContext));
107   }

144   private Artifact getIjarArtifact(Artifact jar, boolean addPrefix) {
145     if (addPrefix) {
146       PathFragment ruleBase = ruleContext.getUniqueDirectory("_ijar");
147       PathFragment artifactDirFragment = jar.getRootRelativePath().getParentDirectory();
148       String ijarBasename = FileSystemUtils.removeExtension(jar.getFilename()) + "-ijar.jar";
149       return ruleContext.getDerivedArtifact(
150           ruleBase.getRelative(artifactDirFragment).getRelative(ijarBasename),
151           getConfiguration().getGenfilesDirectory());
152     } else {
153       return derivedArtifact(jar, "", "-ijar.jar");
154     }
155   }
156 
157   /**
158    * Creates the Action that creates ijars from Jar files.
159    *
160    * @param inputJar the Jar to create the ijar for
161    * @param addPrefix whether to prefix the path of the generated ijar with the package and
162    *     name of the current rule
163    * @return the Artifact to create with the Action
164    */
165   protected Artifact createIjarAction(final Artifact inputJar, boolean addPrefix) {
166     Artifact interfaceJar = getIjarArtifact(inputJar, addPrefix);
167     final FilesToRunProvider ijarTarget =
168         ruleContext.getExecutablePrerequisite("$ijar", Mode.HOST);
169     if (!ruleContext.hasErrors()) {
170       ruleContext.registerAction(new SpawnAction.Builder()
171           .addInput(inputJar)
172           .addOutput(interfaceJar)
173           .setExecutable(ijarTarget)
174           .addArgument(inputJar.getExecPathString())
175           .addArgument(interfaceJar.getExecPathString())
176           .setProgressMessage("Extracting interface " + ruleContext.getLabel())
177           .setMnemonic("JavaIjar")
178           .build(ruleContext));
179     }
180     return interfaceJar;
181   }

183   protected final JavaCompileAction.Builder createJavaCompileActionBuilder(
184       JavaSemantics semantics) {
185     JavaCompileAction.Builder builder = new JavaCompileAction.Builder(ruleContext, semantics);
186     builder.setJavaExecutable(
187         ruleContext.getHostConfiguration().getFragment(Jvm.class).getJavaExecutable());
188     builder.setJavaBaseInputs(BaseJavaCompilationHelper.getHostJavabaseInputs(ruleContext));
189     return builder;
190   }

220   /**
221    * For an output jar and a suffix, produces a derived directory under
222    * {@code bin} directory with a given suffix.
223    */
224   private PathFragment workDir(Artifact outputJar, String suffix) {
225     PathFragment path = outputJar.getRootRelativePath();
226     String basename = FileSystemUtils.removeExtension(path.getBaseName()) + suffix;
227     path = path.replaceName(basename);
228     return getConfiguration().getBinDirectory().getExecPath().getRelative(path);
229   }

231   /**
232    * Creates a derived artifact from the given artifact by adding the given
233    * prefix and removing the extension and replacing it by the given suffix.
234    * The new artifact will have the same root as the given one.
235    */
236   protected Artifact derivedArtifact(Artifact artifact, String prefix, String suffix) {
237     return derivedArtifact(artifact, prefix, suffix, artifact.getRoot());
238   }
239 
240   protected Artifact derivedArtifact(Artifact artifact, String prefix, String suffix, Root root) {
241     PathFragment path = artifact.getRootRelativePath();
242     String basename = FileSystemUtils.removeExtension(path.getBaseName()) + suffix;
243     path = path.replaceName(prefix + basename);
244     return ruleContext.getDerivedArtifact(path, root);
245   }
