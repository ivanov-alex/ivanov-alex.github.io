103   static CompilationArtifacts compilationArtifacts(RuleContext ruleContext) {
104     return new CompilationArtifacts.Builder()
105         .addSrcs(ruleContext.getPrerequisiteArtifacts("srcs", Mode.TARGET)
106             .errorsForNonMatching(SRCS_TYPE)
107             .list())
108         .addNonArcSrcs(ruleContext.getPrerequisiteArtifacts("non_arc_srcs", Mode.TARGET)
109             .errorsForNonMatching(NON_ARC_SRCS_TYPE)
110             .list())
111         .setIntermediateArtifacts(ObjcRuleClasses.intermediateArtifacts(ruleContext))
112         .setPchFile(Optional.fromNullable(ruleContext.getPrerequisiteArtifact("pch", Mode.TARGET)))
113         .build();
114   }

128    * Registers all actions necessary to compile this rule's sources and archive them.
129    *
130    * @param common common information about this rule and its dependencies
131    * @return this compilation support
132    */
133   CompilationSupport registerCompileAndArchiveActions(ObjcCommon common) {
134     if (common.getCompilationArtifacts().isPresent()) {
135       registerCompileAndArchiveActions(
136           common.getCompilationArtifacts().get(),
137           ObjcRuleClasses.intermediateArtifacts(ruleContext),
138           common.getObjcProvider(),
139           ruleContext.getConfiguration().isCodeCoverageEnabled());
140     }
141     return this;
142   }

144   /**
145    * Creates actions to compile each source file individually, and link all the compiled object
146    * files into a single archive library.
147    */
148   private void registerCompileAndArchiveActions(CompilationArtifacts compilationArtifacts,
149       IntermediateArtifacts intermediateArtifacts, ObjcProvider objcProvider,
150       boolean isCodeCoverageEnabled) {
151     ImmutableList.Builder<Artifact> objFiles = new ImmutableList.Builder<>();
152     for (Artifact sourceFile : compilationArtifacts.getSrcs()) {
153       Artifact objFile = intermediateArtifacts.objFile(sourceFile);
154       objFiles.add(objFile);
155       if (ObjcRuleClasses.SWIFT_SOURCES.matches(sourceFile.getFilename())) {
156         registerSwiftCompileAction(sourceFile, objFile, intermediateArtifacts);
157       } else {
158         registerCompileAction(sourceFile, objFile, objcProvider, intermediateArtifacts,
159             compilationArtifacts, ImmutableList.of("-fobjc-arc"), isCodeCoverageEnabled);
160       }
161     }
162     for (Artifact nonArcSourceFile : compilationArtifacts.getNonArcSrcs()) {
163       Artifact objFile = intermediateArtifacts.objFile(nonArcSourceFile);
164       objFiles.add(objFile);
165       registerCompileAction(nonArcSourceFile, objFile, objcProvider, intermediateArtifacts,
166           compilationArtifacts, ImmutableList.of("-fno-objc-arc"), isCodeCoverageEnabled);
167     }
168 
169     if (compilationArtifacts.hasSwiftSources()) {
170       registerSwiftModuleMergeAction(intermediateArtifacts, compilationArtifacts);
171     }
172 
173     for (Artifact archive : compilationArtifacts.getArchive().asSet()) {
174       registerArchiveActions(intermediateArtifacts, objFiles, archive);
175     }
176   }

178   private void registerCompileAction(
179       Artifact sourceFile,
180       Artifact objFile,
181       ObjcProvider objcProvider,
182       IntermediateArtifacts intermediateArtifacts,
183       CompilationArtifacts compilationArtifacts,
184       Iterable<String> otherFlags,
185       boolean isCodeCoverageEnabled) {
186     ObjcConfiguration objcConfiguration = ObjcRuleClasses.objcConfiguration(ruleContext);
187     ImmutableList.Builder<String> coverageFlags = new ImmutableList.Builder<>();
188     ImmutableList.Builder<Artifact> gcnoFiles = new ImmutableList.Builder<>();
189     ImmutableList.Builder<Artifact> additionalInputs = new ImmutableList.Builder<>();
190     if (isCodeCoverageEnabled) {
191       coverageFlags.addAll(CLANG_COVERAGE_FLAGS);
192       gcnoFiles.add(intermediateArtifacts.gcnoFile(sourceFile));
193     }
194     CustomCommandLine.Builder commandLine = new CustomCommandLine.Builder();
195     if (ObjcRuleClasses.CPP_SOURCES.matches(sourceFile.getExecPath())) {
196       commandLine.add("-stdlib=libc++");
197     }
198     if (ObjcRuleClasses.PREPROCESSED_ASSEMBLY_SOURCES.matches(sourceFile.getExecPath())) {
199       commandLine.add("-x").add("assembler-with-cpp");
200     }
201 
202     if (compilationArtifacts.hasSwiftSources()) {
203       // Add the directory that contains merged TargetName-Swift.h header to search path, in case
204       // any of ObjC files use it.
205       commandLine.add("-I");
206       commandLine.addPath(intermediateArtifacts.swiftHeader().getExecPath().getParentDirectory());
207       additionalInputs.add(intermediateArtifacts.swiftHeader());
208     }
209 
210     // The linker needs full debug symbol information to perform binary dead-code stripping.
211     if (objcConfiguration.shouldStripBinary()) {
212       commandLine.add("-g");
213     }
214 
215     commandLine
216         .add(IosSdkCommands.compileFlagsForClang(objcConfiguration))
217         .add(IosSdkCommands.commonLinkAndCompileFlagsForClang(
218             objcProvider, objcConfiguration))
219         .add(objcConfiguration.getCoptsForCompilationMode())
220         .addBeforeEachPath(
221             "-iquote", ObjcCommon.userHeaderSearchPaths(ruleContext.getConfiguration()))
222         .addBeforeEachExecPath("-include", compilationArtifacts.getPchFile().asSet())
223         .addBeforeEachPath("-I", objcProvider.get(INCLUDE))
224         .add(otherFlags)
225         .addFormatEach("-D%s", objcProvider.get(DEFINE))
226         .add(coverageFlags.build())
227         .add(objcConfiguration.getCopts())
228         .add(attributes.copts())
229         .add(attributes.optionsCopts())
230         .addExecPath("-c", sourceFile)
231         .addExecPath("-o", objFile);
232 
233     ruleContext.registerAction(ObjcRuleClasses.spawnOnDarwinActionBuilder()
234         .setMnemonic("ObjcCompile")
235         .setExecutable(CLANG)
236         .setCommandLine(commandLine.build())
237         .addInput(sourceFile)
238         .addInputs(additionalInputs.build())
239         .addOutput(objFile)
240         .addOutputs(gcnoFiles.build())
241         .addTransitiveInputs(objcProvider.get(HEADER))
242         .addTransitiveInputs(objcProvider.get(FRAMEWORK_FILE))
243         .addInputs(compilationArtifacts.getPchFile().asSet())
244         .build(ruleContext));
245   }

247   /**
248    * Compiles a single swift file.
249    *
250    * @param sourceFile the artifact to compile
251    * @param objFile the resulting object artifact
252    */
253   private void registerSwiftCompileAction(
254       Artifact sourceFile,
255       Artifact objFile,
256       IntermediateArtifacts intermediateArtifacts) {
257     ObjcConfiguration objcConfiguration = ObjcRuleClasses.objcConfiguration(ruleContext);
258 
259     // Compiling a single swift file requires knowledge of all of the other
260     // swift files in the same module. The primary file ({@code sourceFile}) is
261     // compiled to an object file, while the remaining files are used to resolve
262     // symbols (they behave like c/c++ headers in this context).
263     ImmutableSet.Builder<Artifact> otherSwiftSourcesBuilder = ImmutableSet.builder();
264     for (Artifact otherSourceFile : compilationArtifacts(ruleContext).getSrcs()) {
265       if (ObjcRuleClasses.SWIFT_SOURCES.matches(otherSourceFile.getFilename())
266           && otherSourceFile != sourceFile) {
267         otherSwiftSourcesBuilder.add(otherSourceFile);
268       }
269     }
270     ImmutableSet<Artifact> otherSwiftSources = otherSwiftSourcesBuilder.build();
271 
272     CustomCommandLine.Builder commandLine = new CustomCommandLine.Builder()
273         .add("-frontend")
274         .add("-emit-object")
275         .add("-target").add(IosSdkCommands.swiftTarget(objcConfiguration))
276         .add("-sdk").add(IosSdkCommands.sdkDir(objcConfiguration))
277         .add("-enable-objc-interop");
278 
279     if (objcConfiguration.generateDebugSymbols()) {
280       commandLine.add("-g");
281     }
282 
283     commandLine
284       .add("-Onone")
285       .add("-module-name").add(getModuleName())
286       .add("-parse-as-library")
287       .addExecPath("-primary-file", sourceFile)
288       .addExecPaths(otherSwiftSources)
289       .addExecPath("-o", objFile)
290       .addExecPath("-emit-module-path", intermediateArtifacts.swiftModuleFile(sourceFile));
291 
292     // Add all ObjC headers to the compiler, in case Swift code is calling into Objc
293     // TODO(bazel-team): This can be augmented by an explicit bridging header field in the rule.
294     commandLine.addBeforeEachExecPath("-import-objc-header", attributes.hdrs());
295 
296     ruleContext.registerAction(ObjcRuleClasses.spawnOnDarwinActionBuilder()
297         .setMnemonic("SwiftCompile")
298         .setExecutable(SWIFT)
299         .setCommandLine(commandLine.build())
300         .addInput(sourceFile)
301         .addInputs(otherSwiftSources)
302         .addInputs(attributes.hdrs())
303         .addOutput(objFile)
304         .addOutput(intermediateArtifacts.swiftModuleFile(sourceFile))
305         .build(ruleContext));
306   }

308   /**
309    * Merges multiple .partial_swiftmodule files together. Also produces a swift header that can be
310    * used by Objective-C code.
311    */
312   private void registerSwiftModuleMergeAction(
313       IntermediateArtifacts intermediateArtifacts,
314       CompilationArtifacts compilationArtifacts) {
315     ObjcConfiguration objcConfiguration = ObjcRuleClasses.objcConfiguration(ruleContext);
316 
317     ImmutableList.Builder<Artifact> moduleFiles = new ImmutableList.Builder<>();
318     for (Artifact src : compilationArtifacts.getSrcs()) {
319       if (ObjcRuleClasses.SWIFT_SOURCES.matches(src.getFilename())) {
320         moduleFiles.add(intermediateArtifacts.swiftModuleFile(src));
321       }
322     }
323 
324     CustomCommandLine.Builder commandLine = new CustomCommandLine.Builder();
325     commandLine.add("-frontend");
326     commandLine.add("-emit-module");
327     commandLine.add("-sdk").add(IosSdkCommands.sdkDir(objcConfiguration));
328     commandLine.add("-target").add(IosSdkCommands.swiftTarget(objcConfiguration));
329     if (objcConfiguration.generateDebugSymbols()) {
330       commandLine.add("-g");
331     }
332 
333     commandLine.add("-module-name").add(getModuleName());
334     commandLine.add("-parse-as-library");
335     commandLine.addExecPaths(moduleFiles.build());
336     commandLine.addExecPath("-o", intermediateArtifacts.swiftModule());
337     commandLine.addExecPath("-emit-objc-header-path", intermediateArtifacts.swiftHeader());
338 
339     ruleContext.registerAction(ObjcRuleClasses.spawnOnDarwinActionBuilder()
340         .setMnemonic("SwiftModuleMerge")
341         .setExecutable(SWIFT)
342         .setCommandLine(commandLine.build())
343         .addInputs(moduleFiles.build())
344         .addOutput(intermediateArtifacts.swiftModule())
345         .addOutput(intermediateArtifacts.swiftHeader())
346         .build(ruleContext));
347   }

349   private void registerArchiveActions(IntermediateArtifacts intermediateArtifacts,
350       ImmutableList.Builder<Artifact> objFiles, Artifact archive) {
351     for (Action action : archiveActions(ruleContext, objFiles.build(), archive,
352         ObjcRuleClasses.objcConfiguration(ruleContext),
353         intermediateArtifacts.objList())) {
354       ruleContext.registerAction(action);
355     }
356   }

358   private static Iterable<Action> archiveActions(
359       ActionConstructionContext context,
360       Iterable<Artifact> objFiles,
361       Artifact archive,
362       ObjcConfiguration objcConfiguration,
363       Artifact objList) {
364 
365     ImmutableList.Builder<Action> actions = new ImmutableList.Builder<>();
366 
367     actions.add(new FileWriteAction(
368         context.getActionOwner(),
369         objList,
370         Artifact.joinExecPaths("\n", objFiles),
371         /*makeExecutable=*/ false));
372 
373     actions.add(ObjcRuleClasses.spawnOnDarwinActionBuilder()
374         .setMnemonic("ObjcLink")
375         .setExecutable(ObjcRuleClasses.LIBTOOL)
376         .setCommandLine(new CustomCommandLine.Builder()
377             .add("-static")
378             .add("-filelist").add(objList.getExecPathString())
379             .add("-arch_only").add(objcConfiguration.getIosCpu())
380             .add("-syslibroot").add(IosSdkCommands.sdkDir(objcConfiguration))
381             .add("-o").add(archive.getExecPathString())
382             .build())
383         .addInputs(objFiles)
384         .addInput(objList)
385         .addOutput(archive)
386         .build(context));
387 
388     return actions.build();
389   }

391   /**
392    * Registers any actions necessary to link this rule and its dependencies.
393    *
394    * <p>Dsym bundle and breakpad files are generated if
395    * {@link ObjcConfiguration#generateDebugSymbols()} is set.
396    *
397    * <p>When Bazel flags {@code --compilation_mode=opt} and {@code --objc_enable_binary_stripping}
398    * are specified, additional optimizations will be performed on the linked binary: all-symbol
399    * stripping (using {@code /usr/bin/strip}) and dead-code stripping (using linker flags:
400    * {@code -dead_strip} and {@code -no_dead_strip_inits_and_terms}).
401    *
402    * @param objcProvider common information about this rule's attributes and its dependencies
403    * @param extraLinkArgs any additional arguments to pass to the linker
404    * @param extraLinkInputs any additional input artifacts to pass to the link action
405    *
406    * @return this compilation support
407    */
408   CompilationSupport registerLinkActions(ObjcProvider objcProvider, ExtraLinkArgs extraLinkArgs,
409       Iterable<Artifact> extraLinkInputs) {
410     IntermediateArtifacts intermediateArtifacts =
411         ObjcRuleClasses.intermediateArtifacts(ruleContext);
412     Optional<Artifact> dsymBundle;
413     if (ObjcRuleClasses.objcConfiguration(ruleContext).generateDebugSymbols()) {
414       registerDsymActions();
415       dsymBundle = Optional.of(intermediateArtifacts.dsymBundle());
416     } else {
417       dsymBundle = Optional.absent();
418     }
419 
420     registerLinkAction(objcProvider, extraLinkArgs, extraLinkInputs, dsymBundle);
421     return this;
422   }
423 
424   private void registerLinkAction(ObjcProvider objcProvider, ExtraLinkArgs extraLinkArgs,
425       Iterable<Artifact> extraLinkInputs, Optional<Artifact> dsymBundle) {
426     ObjcConfiguration objcConfiguration = ObjcRuleClasses.objcConfiguration(ruleContext);
427     IntermediateArtifacts intermediateArtifacts =
428         ObjcRuleClasses.intermediateArtifacts(ruleContext);
429 
430     // When compilation_mode=opt and objc_enable_binary_stripping are specified, the unstripped
431     // binary containing debug symbols is generated by the linker, which also needs the debug
432     // symbols for dead-code removal. The binary is also used to generate dSYM bundle if
433     // --objc_generate_debug_symbol is specified. A symbol strip action is later registered to strip
434     // the symbol table from the unstripped binary.
435     Artifact binaryToLink =
436         objcConfiguration.shouldStripBinary()
437             ? intermediateArtifacts.unstrippedSingleArchitectureBinary()
438             : intermediateArtifacts.strippedSingleArchitectureBinary();
439 
440     ImmutableList<Artifact> ccLibraries = ccLibraries(objcProvider);
441     ruleContext.registerAction(
442         ObjcRuleClasses.spawnOnDarwinActionBuilder()
443             .setMnemonic("ObjcLink")
444             .setShellCommand(ImmutableList.of("/bin/bash", "-c"))
445             .setCommandLine(
446                 linkCommandLine(extraLinkArgs, objcProvider, binaryToLink, dsymBundle, ccLibraries))
447             .addOutput(binaryToLink)
448             .addOutputs(dsymBundle.asSet())
449             .addTransitiveInputs(objcProvider.get(LIBRARY))
450             .addTransitiveInputs(objcProvider.get(IMPORTED_LIBRARY))
451             .addTransitiveInputs(objcProvider.get(FRAMEWORK_FILE))
452             .addInputs(ccLibraries)
453             .addInputs(extraLinkInputs)
454             .build(ruleContext));
455 
456     if (objcConfiguration.shouldStripBinary()) {
457       // For test targets, only debug symbols are stripped off, since /usr/bin/strip is not able
458       // to strip off all symbols in XCTest bundle.
459       boolean isTestTarget = TargetUtils.isTestRule(ruleContext.getRule());
460       Iterable<String> stripArgs =
461           isTestTarget ? ImmutableList.of("-S") : ImmutableList.<String>of();
462       Artifact strippedBinary = intermediateArtifacts.strippedSingleArchitectureBinary();
463 
464       ruleContext.registerAction(
465           ObjcRuleClasses.spawnOnDarwinActionBuilder()
466               .setMnemonic("ObjcBinarySymbolStrip")
467               .setExecutable(STRIP)
468               .setCommandLine(symbolStripCommandLine(stripArgs, binaryToLink, strippedBinary))
469               .addOutput(strippedBinary)
470               .addInput(binaryToLink)
471               .build(ruleContext));
472     }
473   }

475   private ImmutableList<Artifact> ccLibraries(ObjcProvider objcProvider) {
476     ImmutableList.Builder<Artifact> ccLibraryBuilder = ImmutableList.builder();
477     for (LinkerInputs.LibraryToLink libraryToLink : objcProvider.get(ObjcProvider.CC_LIBRARY)) {
478       ccLibraryBuilder.add(libraryToLink.getArtifact());
479     }
480     return ccLibraryBuilder.build();
481   }

492   private CommandLine linkCommandLine(ExtraLinkArgs extraLinkArgs,
493       ObjcProvider objcProvider, Artifact linkedBinary, Optional<Artifact> dsymBundle,
494       ImmutableList<Artifact> ccLibraries) {
495     ObjcConfiguration objcConfiguration = ObjcRuleClasses.objcConfiguration(ruleContext);
496 
497     CustomCommandLine.Builder commandLine = CustomCommandLine.builder();
498 
499     if (objcProvider.is(USES_CPP)) {
500       commandLine
501           .addPath(CLANG_PLUSPLUS)
502           .add("-stdlib=libc++");
503     } else {
504       commandLine.addPath(CLANG);
505     }
506 
507     // Do not perform code stripping on tests because XCTest binary is linked not as an executable
508     // but as a bundle without any entry point.
509     boolean isTestTarget = TargetUtils.isTestRule(ruleContext.getRule());
510     if (objcConfiguration.shouldStripBinary() && !isTestTarget) {
511       commandLine.add("-dead_strip").add("-no_dead_strip_inits_and_terms");
512     }
513 
514     commandLine
515         .add(IosSdkCommands.commonLinkAndCompileFlagsForClang(objcProvider, objcConfiguration))
516         .add("-Xlinker")
517         .add("-objc_abi_version")
518         .add("-Xlinker")
519         .add("2")
520         .add("-fobjc-link-runtime")
521         .add(IosSdkCommands.DEFAULT_LINKER_FLAGS)
522         .addBeforeEach("-framework", frameworkNames(objcProvider))
523         .addBeforeEach("-weak_framework", SdkFramework.names(objcProvider.get(WEAK_SDK_FRAMEWORK)))
524         .addExecPath("-o", linkedBinary)
525         .addExecPaths(objcProvider.get(LIBRARY))
526         .addExecPaths(objcProvider.get(IMPORTED_LIBRARY))
527         .addExecPaths(ccLibraries)
528         .add(dylibPaths(objcProvider))
529         .addBeforeEach("-force_load", Artifact.toExecPaths(objcProvider.get(FORCE_LOAD_LIBRARY)))
530         .add(extraLinkArgs)
531         .build();
532 
533     if (ruleContext.getConfiguration().isCodeCoverageEnabled()) {
534       commandLine.add(LINKER_COVERAGE_FLAGS);
535     }
536 
537     if (objcProvider.is(USES_SWIFT)) {
538       commandLine
539           .add("-L").add(IosSdkCommands.swiftLibDir(objcConfiguration))
540           .add("-Xlinker").add("-rpath")
541           .add("-Xlinker").add("@executable_path/Frameworks");
542     }
543 
544     // Call to dsymutil for debug symbol generation must happen in the link action.
545     // All debug symbol information is encoded in object files inside archive files. To generate
546     // the debug symbol bundle, dsymutil will look inside the linked binary for the encoded
547     // absolute paths to archive files, which are only valid in the link action.
548     if (dsymBundle.isPresent()) {
549       commandLine
550           .add("&&")
551           .addPath(DSYMUTIL)
552           .add(linkedBinary.getExecPathString())
553           .addExecPath("-o", dsymBundle.get());
554     }
555 
556     return new SingleArgCommandLine(commandLine.build());
557   }

559   /**
560    * Command line that converts its input's arg array to a single input.
561    *
562    * <p>Required as a hack to the link command line because that may contain two commands, which are
563    * then passed to {@code /bin/bash -c}, and accordingly need to be a single argument.
564    */
565   private static class SingleArgCommandLine extends CommandLine {
566 
567     private final CommandLine original;
568 
569     private SingleArgCommandLine(CommandLine original) {
570       this.original = original;
571     }
572 
573     @Override
574     public Iterable<String> arguments() {
575       return ImmutableList.of(Joiner.on(' ').join(original.arguments()));
576     }
577   }

589   /**
590    * All framework names to pass to the linker using {@code -framework} flags. For a framework in
591    * the directory foo/bar.framework, the name is "bar". Each framework is found without using the
592    * full path by means of the framework search paths. The search paths are added by
593    * {@link IosSdkCommands#commonLinkAndCompileFlagsForClang(ObjcProvider, ObjcConfiguration)}).
594    *
595    * <p>It's awful that we can't pass the full path to the framework and avoid framework search
596    * paths, but this is imposed on us by clang. clang does not support passing the full path to the
597    * framework, so Bazel cannot do it either.
598    */
599   private Iterable<String> frameworkNames(ObjcProvider provider) {
600     List<String> names = new ArrayList<>();
601     Iterables.addAll(names, SdkFramework.names(provider.get(SDK_FRAMEWORK)));
602     for (PathFragment frameworkDir : provider.get(FRAMEWORK_DIR)) {
603       String segment = frameworkDir.getBaseName();
604       Preconditions.checkState(segment.endsWith(FRAMEWORK_SUFFIX),
605           "expect %s to end with %s, but it does not", segment, FRAMEWORK_SUFFIX);
606       names.add(segment.substring(0, segment.length() - FRAMEWORK_SUFFIX.length()));
607     }
608     return names;
609   }

611   /**
612    * Registers actions that compile and archive j2Objc dependencies of this rule.
613    *
614    * @param objcProvider common information about this rule's attributes and its dependencies
615    *
616    * @return this compilation support
617    */
618   CompilationSupport registerJ2ObjcCompileAndArchiveActions(ObjcProvider objcProvider) {
619     J2ObjcSrcsProvider provider = J2ObjcSrcsProvider.buildFrom(ruleContext);
620     Iterable<J2ObjcSource> j2ObjcSources = provider.getSrcs();
621     J2ObjcConfiguration j2objcConfiguration = ruleContext.getFragment(J2ObjcConfiguration.class);
622 
623     // Only perform J2ObjC dead code stripping if flag --j2objc_dead_code_removal is specified and
624     // users have specified entry classes.
625     boolean stripJ2ObjcDeadCode = j2objcConfiguration.removeDeadCode()
626         && !provider.getEntryClasses().isEmpty();
627 
628     if (stripJ2ObjcDeadCode) {
629       registerJ2ObjcDeadCodeRemovalActions(j2ObjcSources, provider.getEntryClasses());
630     }
631 
632     for (J2ObjcSource j2ObjcSource : j2ObjcSources) {
633       J2ObjcSource sourceToCompile =
634           j2ObjcSource.getSourceType() == SourceType.JAVA && stripJ2ObjcDeadCode
635               ? j2ObjcSource.toPrunedSource(ruleContext)
636               : j2ObjcSource;
637       IntermediateArtifacts intermediateArtifacts =
638           ObjcRuleClasses.j2objcIntermediateArtifacts(ruleContext, sourceToCompile);
639       CompilationArtifacts compilationArtifact = new CompilationArtifacts.Builder()
640           .addNonArcSrcs(sourceToCompile.getObjcSrcs())
641           .setIntermediateArtifacts(intermediateArtifacts)
642           .setPchFile(Optional.<Artifact>absent())
643           .build();
644       registerCompileAndArchiveActions(compilationArtifact, intermediateArtifacts, objcProvider,
645           ruleContext.getConfiguration().isCodeCoverageEnabled());
646     }
647 
648     return this;
649   }
650 
651   private void registerJ2ObjcDeadCodeRemovalActions(Iterable<J2ObjcSource> j2ObjcSources,
652       Iterable<String> entryClasses) {
653     Artifact pruner = ruleContext.getPrerequisiteArtifact("$j2objc_dead_code_pruner", Mode.HOST);
654     J2ObjcMappingFileProvider provider = ObjcRuleClasses.j2ObjcMappingFileProvider(ruleContext);
655     NestedSet<Artifact> j2ObjcDependencyMappingFiles = provider.getDependencyMappingFiles();
656     NestedSet<Artifact> j2ObjcHeaderMappingFiles = provider.getHeaderMappingFiles();
657 
658     for (J2ObjcSource j2ObjcSource : j2ObjcSources) {
659       if (j2ObjcSource.getSourceType() == SourceType.JAVA) {
660         Iterable<Artifact> sourceArtifacts = j2ObjcSource.getObjcSrcs();
661         Iterable<Artifact> prunedSourceArtifacts =
662             j2ObjcSource.toPrunedSource(ruleContext).getObjcSrcs();
663         PathFragment objcFilePath = j2ObjcSource.getObjcFilePath();
664         ruleContext.registerAction(new SpawnAction.Builder()
665             .setMnemonic("DummyPruner")
666             .setExecutable(pruner)
667             .addInput(pruner)
668             .addInputs(sourceArtifacts)
669             .addTransitiveInputs(j2ObjcDependencyMappingFiles)
670             .addTransitiveInputs(j2ObjcHeaderMappingFiles)
671             .setCommandLine(CustomCommandLine.builder()
672                 .addJoinExecPaths("--input_files", ",", sourceArtifacts)
673                 .addJoinExecPaths("--output_files", ",", prunedSourceArtifacts)
674                 .addJoinExecPaths("--dependency_mapping_files", ",", j2ObjcDependencyMappingFiles)
675                 .addJoinExecPaths("--header_mapping_files", ",", j2ObjcHeaderMappingFiles)
676                 .add("--entry_classes").add(Joiner.on(",").join(entryClasses))
677                 .add("--objc_file_path").add(objcFilePath.getPathString())
678                 .build())
679             .addOutputs(prunedSourceArtifacts)
680             .build(ruleContext));
681       }
682     }
683   }

723   private CompilationSupport registerDsymActions() {
724     IntermediateArtifacts intermediateArtifacts =
725         ObjcRuleClasses.intermediateArtifacts(ruleContext);
726     ObjcConfiguration objcConfiguration = ObjcRuleClasses.objcConfiguration(ruleContext);
727 
728     Artifact dsymBundle = intermediateArtifacts.dsymBundle();
729     Artifact linkedBinary =
730         objcConfiguration.shouldStripBinary()
731             ? intermediateArtifacts.unstrippedSingleArchitectureBinary()
732             : intermediateArtifacts.strippedSingleArchitectureBinary();
733     Artifact debugSymbolFile = intermediateArtifacts.dsymSymbol();
734     Artifact dsymPlist = intermediateArtifacts.dsymPlist();
735 
736     PathFragment dsymOutputDir =
737         replaceSuffix(
738             dsymBundle.getExecPath(), IntermediateArtifacts.TMP_DSYM_BUNDLE_SUFFIX, ".app.dSYM");
739     PathFragment dsymPlistZipEntry = dsymPlist.getExecPath().relativeTo(dsymOutputDir);
740     PathFragment debugSymbolFileZipEntry =
741         debugSymbolFile
742             .getExecPath()
743             .replaceName(linkedBinary.getFilename())
744             .relativeTo(dsymOutputDir);
745 
746     StringBuilder unzipDsymCommand = new StringBuilder();
747     unzipDsymCommand
748         .append(
749             String.format(
750                 "unzip -p %s %s > %s",
751                 dsymBundle.getExecPathString(),
752                 dsymPlistZipEntry,
753                 dsymPlist.getExecPathString()))
754         .append(
755             String.format(
756                 " && unzip -p %s %s > %s",
757                 dsymBundle.getExecPathString(),
758                 debugSymbolFileZipEntry,
759                 debugSymbolFile.getExecPathString()));
760 
761     ruleContext.registerAction(
762         new SpawnAction.Builder()
763             .setMnemonic("UnzipDsym")
764             .setShellCommand(unzipDsymCommand.toString())
765             .addInput(dsymBundle)
766             .addOutput(dsymPlist)
767             .addOutput(debugSymbolFile)
768             .build(ruleContext));
769 
770     Artifact dumpsyms = ruleContext.getPrerequisiteArtifact(":dumpsyms", Mode.HOST);
771     Artifact breakpadFile = intermediateArtifacts.breakpadSym();
772     ruleContext.registerAction(ObjcRuleClasses.spawnOnDarwinActionBuilder()
773         .setMnemonic("GenBreakpad")
774         .setProgressMessage("Generating breakpad file: " + ruleContext.getLabel())
775         .setShellCommand(ImmutableList.of("/bin/bash", "-c"))
776         .addInput(dumpsyms)
777         .addInput(debugSymbolFile)
778         .addArgument(String.format("%s %s > %s",
779             ShellUtils.shellEscape(dumpsyms.getExecPathString()),
780             ShellUtils.shellEscape(debugSymbolFile.getExecPathString()),
781             ShellUtils.shellEscape(breakpadFile.getExecPathString())))
782         .addOutput(breakpadFile)
783         .build(ruleContext));
784     return this;
785   }

787   private PathFragment replaceSuffix(PathFragment path, String suffix, String newSuffix) {
788     // TODO(bazel-team): Throw instead of returning null?
789     String name = path.getBaseName();
790     if (name.endsWith(suffix)) {
791       return path.replaceName(name.substring(0, name.length() - suffix.length()) + newSuffix);
792     } else {
793       return null;
794     }
795   }
