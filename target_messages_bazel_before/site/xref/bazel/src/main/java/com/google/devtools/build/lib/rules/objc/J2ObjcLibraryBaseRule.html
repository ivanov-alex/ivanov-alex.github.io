28  /**
29   * Abstract rule definition for j2objc_library.
30   */
31  public class J2ObjcLibraryBaseRule implements RuleDefinition {
32    @Override
33    public RuleClass build(Builder builder, RuleDefinitionEnvironment env) {
34      // TODO(rduan): Add support for package prefixes.
35      return builder
36            /* <!-- #BLAZE_RULE(j2objc_library).ATTRIBUTE(entry_classes) -->
37            The list of Java classes whose translated ObjC counterparts will be referenced directly
38            by user ObjC code. This attibute is required if flag <code>--j2objc_dead_code_removal
39            </code> is on. The Java classes should be specified in their canonical names as defined by
40            <a href="http://docs.oracle.com/javase/specs/jls/se8/html/jls-6.html#jls-6.7">the Java
41            Language Specification.</a>
42            ${SYNOPSIS}
43            When flag <code>--j2objc_dead_code_removal</code> is specified, the list of entry classes
44            will be collected transitively and used as entry points to perform dead code analysis.
45            Unused classes will then be removed from the final ObjC app bundle.
46            <!-- #END_BLAZE_RULE.ATTRIBUTE -->*/
47          .add(attr("entry_classes", STRING_LIST))
48          .add(attr("$jre_emul_jar", LABEL)
49              .value(env.getLabel("//third_party/java_src/j2objc/jre_emul:libjre_emul.jar")))
50          .add(attr("$jre_emul_lib", LABEL)
51              .value(env.getLabel("//third_party/java_src/j2objc/jre_emul:jre_emul_lib")))
52          .build();
53    }
54  
55    @Override
56    public Metadata getMetadata() {
57      return RuleDefinition.Metadata.builder()
58          .name("$j2objc_library_base")
59          .type(RuleClassType.ABSTRACT)
60          .ancestors(BaseRuleClasses.BaseRule.class, ObjcRuleClasses.CoptsRule.class)
61          .build();
62    }
63  }
