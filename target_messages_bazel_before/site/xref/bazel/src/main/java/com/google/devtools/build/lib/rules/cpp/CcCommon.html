109   /**
110    * Returns our own linkopts from the rule attribute. This determines linker
111    * options to use when building this target and anything that depends on it.
112    */
113   public ImmutableList<String> getLinkopts() {
114     Preconditions.checkState(hasAttribute("linkopts", Type.STRING_LIST));
115     List<String> ourLinkopts = ruleContext.attributes().get("linkopts", Type.STRING_LIST);
116     List<String> result = new ArrayList<>();
117     if (ourLinkopts != null) {
118       boolean allowDashStatic = !cppConfiguration.forceIgnoreDashStatic()
119           && (cppConfiguration.getDynamicMode() != DynamicMode.FULLY);
120       for (String linkopt : ourLinkopts) {
121         if (linkopt.equals("-static") && !allowDashStatic) {
122           continue;
123         }
124         CppHelper.expandAttribute(ruleContext, result, "linkopts", linkopt, true);
125       }
126     }
127     return ImmutableList.copyOf(result);
128   }

130   public ImmutableList<String> getCopts() {
131     Preconditions.checkState(hasAttribute("copts", Type.STRING_LIST));
132     // TODO(bazel-team): getAttributeCopts should not tokenize the strings. Make a warning for now.
133     List<String> tokens = new ArrayList<>();
134     for (String str : ruleContext.attributes().get("copts", Type.STRING_LIST)) {
135       tokens.clear();
136       try {
137         ShellUtils.tokenize(tokens, str);
138         if (tokens.size() > 1) {
139           ruleContext.attributeWarning("copts",
140               "each item in the list should contain only one option");
141         }
142       } catch (ShellUtils.TokenizationException e) {
143         // ignore, the error is reported in the getAttributeCopts call
144       }
145     }
146 
147     Pattern nocopts = getNoCopts(ruleContext);
148     if (nocopts != null && nocopts.matcher("-Wno-future-warnings").matches()) {
149       ruleContext.attributeWarning("nocopts",
150           "Regular expression '" + nocopts.pattern() + "' is too general; for example, it matches "
151           + "'-Wno-future-warnings'.  Thus it might *re-enable* compiler warnings we wish to "
152           + "disable globally.  To disable all compiler warnings, add '-w' to copts instead");
153     }
154 
155     return ImmutableList.<String>builder()
156         .addAll(getPackageCopts(ruleContext))
157         .addAll(CppHelper.getAttributeCopts(ruleContext, "copts"))
158         .build();
159   }

168   public static DwoArtifactsCollector collectTransitiveDwoArtifacts(
169       RuleContext ruleContext,
170       CcCompilationOutputs compilationOutputs) {
171     ImmutableList.Builder<TransitiveInfoCollection> deps =
172         ImmutableList.<TransitiveInfoCollection>builder();
173 
174     deps.addAll(ruleContext.getPrerequisites("deps", Mode.TARGET));
175 
176     if (ruleContext.attributes().has("malloc", Type.LABEL)) {
177       deps.add(CppHelper.mallocForTarget(ruleContext));
178     }
179 
180     return compilationOutputs == null  // Possible in LIPO collection mode (see initializationHook).
181         ? DwoArtifactsCollector.emptyCollector()
182         : DwoArtifactsCollector.transitiveCollector(compilationOutputs, deps.build());
183   }

201   /**
202    * Returns a list of ({@link Artifact}, {@link Label}) pairs. Each pair represents an input
203    * source file and the label of the rule that generates it (or the label of the source file
204    * itself if it is an input file)
205    */
206   ImmutableList<Pair<Artifact, Label>> getCAndCppSources() {
207     Map<Artifact, Label> map = Maps.newLinkedHashMap();
208     Iterable<FileProvider> providers =
209         ruleContext.getPrerequisites("srcs", Mode.TARGET, FileProvider.class);
210     // TODO(bazel-team): Move header processing logic down in the stack (to CcLibraryHelper or
211     // such).
212     boolean processHeaders = shouldProcessHeaders();
213     if (processHeaders && hasAttribute("hdrs", Type.LABEL_LIST)) {
214       providers = Iterables.concat(providers,
215           ruleContext.getPrerequisites("hdrs", Mode.TARGET, FileProvider.class));
216     }
217     for (FileProvider provider : providers) {
218       for (Artifact artifact : FileType.filter(provider.getFilesToBuild(), SOURCE_TYPES)) {
219         boolean isHeader = CppFileTypes.CPP_HEADER.matches(artifact.getExecPath());
220         if ((isHeader && !processHeaders)
221             || CppFileTypes.CPP_TEXTUAL_INCLUDE.matches(artifact.getExecPath())) {
222           continue;
223         }
224         Label oldLabel = map.put(artifact, provider.getLabel());
225         // TODO(bazel-team): We currently do not warn for duplicate headers with
226         // different labels, as that would require cleaning up the code base
227         // without significant benefit; we should eventually make this
228         // consistent one way or the other.
229         if (!isHeader && oldLabel != null && !oldLabel.equals(provider.getLabel())) {
230           ruleContext.attributeError("srcs", String.format(
231               "Artifact '%s' is duplicated (through '%s' and '%s')",
232               artifact.getExecPathString(), oldLabel, provider.getLabel()));
233         }
234       }
235     }
236 
237     ImmutableList.Builder<Pair<Artifact, Label>> result = ImmutableList.builder();
238     for (Map.Entry<Artifact, Label> entry : map.entrySet()) {
239       result.add(Pair.of(entry.getKey(), entry.getValue()));
240     }
241 
242     return result.build();
243   }

245   /**
246    * Returns the files from headers and does some sanity checks. Note that this method reports
247    * warnings to the {@link RuleContext} as a side effect, and so should only be called once for any
248    * given rule.
249    */
250   public static List<Artifact> getHeaders(RuleContext ruleContext) {
251     List<Artifact> hdrs = new ArrayList<>();
252     for (TransitiveInfoCollection target :
253         ruleContext.getPrerequisitesIf("hdrs", Mode.TARGET, FileProvider.class)) {
254       FileProvider provider = target.getProvider(FileProvider.class);
255       for (Artifact artifact : provider.getFilesToBuild()) {
256         if (!CppRuleClasses.DISALLOWED_HDRS_FILES.matches(artifact.getFilename())) {
257           hdrs.add(artifact);
258         } else {
259           ruleContext.attributeWarning("hdrs", "file '" + artifact.getFilename()
260               + "' from target '" + target.getLabel() + "' is not allowed in hdrs");
261         }
262       }
263     }
264     return hdrs;
265   }

267   HeadersCheckingMode determineHeadersCheckingMode() {
268     HeadersCheckingMode headersCheckingMode = cppConfiguration.getHeadersCheckingMode();
269 
270     // Package default overrides command line option.
271     if (ruleContext.getRule().getPackage().isDefaultHdrsCheckSet()) {
272       String value =
273           ruleContext.getRule().getPackage().getDefaultHdrsCheck().toUpperCase(Locale.ENGLISH);
274       headersCheckingMode = HeadersCheckingMode.valueOf(value);
275     }
276 
277     // 'hdrs_check' attribute overrides package default.
278     if (hasAttribute("hdrs_check", Type.STRING)
279         && ruleContext.getRule().isAttributeValueExplicitlySpecified("hdrs_check")) {
280       try {
281         String value = ruleContext.attributes().get("hdrs_check", Type.STRING)
282             .toUpperCase(Locale.ENGLISH);
283         headersCheckingMode = HeadersCheckingMode.valueOf(value);
284       } catch (IllegalArgumentException e) {
285         ruleContext.attributeError("hdrs_check", "must be one of: 'loose', 'warn' or 'strict'");
286       }
287     }
288 
289     return headersCheckingMode;
290   }

301   /**
302    * Returns nocopts pattern built from the make variable expanded nocopts
303    * attribute.
304    */
305   private static Pattern getNoCopts(RuleContext ruleContext) {
306     Pattern nocopts = null;
307     if (ruleContext.getRule().isAttrDefined(NO_COPTS_ATTRIBUTE, Type.STRING)) {
308       String nocoptsAttr = ruleContext.expandMakeVariables(NO_COPTS_ATTRIBUTE,
309           ruleContext.attributes().get(NO_COPTS_ATTRIBUTE, Type.STRING));
310       try {
311         nocopts = Pattern.compile(nocoptsAttr);
312       } catch (PatternSyntaxException e) {
313         ruleContext.attributeError(NO_COPTS_ATTRIBUTE,
314             "invalid regular expression '" + nocoptsAttr + "': " + e.getMessage());
315       }
316     }
317     return nocopts;
318   }

320   // TODO(bazel-team): calculating nocopts every time is not very efficient,
321   // fix this after the rule migration. The problem is that in some cases we call this after
322   // the RCT is created (so RuleContext is not accessible), in some cases during the creation.
323   // It would probably make more sense to use TransitiveInfoProviders.
324   /**
325    * Returns true if the rule context has a nocopts regex that matches the given value, false
326    * otherwise.
327    */
328   static boolean noCoptsMatches(String option, RuleContext ruleContext) {
329     Pattern nocopts = getNoCopts(ruleContext);
330     return nocopts == null ? false : nocopts.matcher(option).matches();
331   }

335   /**
336    * Returns a list of define tokens from "defines" attribute.
337    *
338    * <p>We tokenize the "defines" attribute, to ensure that the handling of
339    * quotes and backslash escapes is consistent Bazel's treatment of the "copts" attribute.
340    *
341    * <p>But we require that the "defines" attribute consists of a single token.
342    */
343   public List<String> getDefines() {
344     List<String> defines = new ArrayList<>();
345     for (String define :
346       ruleContext.attributes().get(DEFINES_ATTRIBUTE, Type.STRING_LIST)) {
347       List<String> tokens = new ArrayList<>();
348       try {
349         ShellUtils.tokenize(tokens, ruleContext.expandMakeVariables(DEFINES_ATTRIBUTE, define));
350         if (tokens.size() == 1) {
351           defines.add(tokens.get(0));
352         } else if (tokens.isEmpty()) {
353           ruleContext.attributeError(DEFINES_ATTRIBUTE, "empty definition not allowed");
354         } else {
355           ruleContext.attributeError(DEFINES_ATTRIBUTE,
356               "definition contains too many tokens (found " + tokens.size()
357               + ", expecting exactly one)");
358         }
359       } catch (ShellUtils.TokenizationException e) {
360         ruleContext.attributeError(DEFINES_ATTRIBUTE, e.getMessage());
361       }
362     }
363     return defines;
364   }

366   /**
367    * Determines a list of loose include directories that are only allowed to be referenced when
368    * headers checking is {@link HeadersCheckingMode#LOOSE} or {@link HeadersCheckingMode#WARN}.
369    */
370   List<PathFragment> getLooseIncludeDirs() {
371     List<PathFragment> result = new ArrayList<>();
372     // The package directory of the rule contributes includes. Note that this also covers all
373     // non-subpackage sub-directories.
374     PathFragment rulePackage = ruleContext.getLabel().getPackageIdentifier().getPathFragment();
375     result.add(rulePackage);
376 
377     // Gather up all the dirs from the rule's srcs as well as any of the srcs outputs.
378     if (hasAttribute("srcs", Type.LABEL_LIST)) {
379       for (FileProvider src :
380           ruleContext.getPrerequisites("srcs", Mode.TARGET, FileProvider.class)) {
381         PathFragment packageDir = src.getLabel().getPackageIdentifier().getPathFragment();
382         for (Artifact a : src.getFilesToBuild()) {
383           result.add(packageDir);
384           // Attempt to gather subdirectories that might contain include files.
385           result.add(a.getRootRelativePath().getParentDirectory());
386         }
387       }
388     }
389 
390     // Add in any 'includes' attribute values as relative path fragments
391     if (ruleContext.getRule().isAttributeValueExplicitlySpecified("includes")) {
392       PathFragment packageFragment = ruleContext.getLabel().getPackageIdentifier()
393           .getPathFragment();
394       // For now, anything with an 'includes' needs a blanket declaration
395       result.add(packageFragment.getRelative("**"));
396     }
397     return result;
398   }

400   List<PathFragment> getSystemIncludeDirs() {
401     // Add in any 'includes' attribute values as relative path fragments
402     if (!ruleContext.getRule().isAttributeValueExplicitlySpecified("includes")
403         || !cppConfiguration.useIsystemForIncludes()) {
404       return ImmutableList.of();
405     }
406     return getIncludeDirsFromIncludesAttribute();
407   }

409   List<PathFragment> getIncludeDirs() {
410     if (!ruleContext.getRule().isAttributeValueExplicitlySpecified("includes")
411         || cppConfiguration.useIsystemForIncludes()) {
412       return ImmutableList.of();
413     }
414     return getIncludeDirsFromIncludesAttribute();
415   }

417   private List<PathFragment> getIncludeDirsFromIncludesAttribute() {
418     List<PathFragment> result = new ArrayList<>();
419     PathFragment packageFragment = ruleContext.getLabel().getPackageIdentifier().getPathFragment();
420     for (String includesAttr : ruleContext.attributes().get("includes", Type.STRING_LIST)) {
421       includesAttr = ruleContext.expandMakeVariables("includes", includesAttr);
422       if (includesAttr.startsWith("/")) {
423         ruleContext.attributeWarning("includes",
424             "ignoring invalid absolute path '" + includesAttr + "'");
425         continue;
426       }
427       PathFragment includesPath = packageFragment.getRelative(includesAttr).normalize();
428       if (!includesPath.isNormalized()) {
429         ruleContext.attributeError("includes",
430             "Path references a path above the execution root.");
431       }
432       result.add(includesPath);
433       result.add(ruleContext.getConfiguration().getGenfilesFragment().getRelative(includesPath));
434     }
435     return result;
436   }

439    * Collects compilation prerequisite artifacts.

441   static NestedSet<Artifact> collectCompilationPrerequisites(
442       RuleContext ruleContext, CppCompilationContext context) {
443     // TODO(bazel-team): Use context.getCompilationPrerequisites() instead.
444     NestedSetBuilder<Artifact> prerequisites = NestedSetBuilder.stableOrder();
445     if (ruleContext.attributes().has("srcs", Type.LABEL_LIST)) {

448         prerequisites.addAll(FileType.filter(provider.getFilesToBuild(), SOURCE_TYPES));

450     }
451     prerequisites.addTransitive(context.getDeclaredIncludeSrcs());
452     return prerequisites.build();
453   }

497   /**
498    * Creates the feature configuration for a given rule.
499    * 
500    * @param ruleContext the context of the rule we want the feature configuration for. 
501    * @param ruleSpecificRequestedFeatures features that will be requested, and thus be always
502    * enabled if the toolchain supports them.
503    * @param ruleSpecificUnsupportedFeatures features that are not supported in the current context.
504    * @return the feature configuration for the given {@code ruleContext}.
505    */
506   public static FeatureConfiguration configureFeatures(RuleContext ruleContext,
507       Set<String> ruleSpecificRequestedFeatures,
508       Set<String> ruleSpecificUnsupportedFeatures) {
509     CcToolchainProvider toolchain = CppHelper.getToolchain(ruleContext);    
510     ImmutableSet.Builder<String> unsupportedFeaturesBuilder = ImmutableSet.builder();
511     unsupportedFeaturesBuilder.addAll(ruleSpecificUnsupportedFeatures);
512     if (!toolchain.supportsHeaderParsing()) {
513       // TODO(bazel-team): Remove once supports_header_parsing has been removed from the
514       // cc_toolchain rule.
515       unsupportedFeaturesBuilder.add(CppRuleClasses.PARSE_HEADERS);
516       unsupportedFeaturesBuilder.add(CppRuleClasses.PREPROCESS_HEADERS);
517     }
518     if (toolchain.getCppCompilationContext().getCppModuleMap() == null) {
519       unsupportedFeaturesBuilder.add(CppRuleClasses.MODULE_MAPS);
520     }
521     Set<String> unsupportedFeatures = unsupportedFeaturesBuilder.build();
522     ImmutableSet.Builder<String> requestedFeatures = ImmutableSet.builder();
523     for (String feature : Iterables.concat(
524         ImmutableSet.of(toolchain.getCompilationMode().toString()), DEFAULT_FEATURES,
525         ruleContext.getFeatures())) {
526       if (!unsupportedFeatures.contains(feature)) {
527         requestedFeatures.add(feature);
528       }
529     }
530     requestedFeatures.addAll(ruleSpecificRequestedFeatures);
531 
532     // Enable FDO related features requested by options.
533     CppConfiguration cppConfiguration =
534         ruleContext.getConfiguration().getFragment(CppConfiguration.class);
535     FdoSupport fdoSupport = cppConfiguration.getFdoSupport();
536     if (fdoSupport.getFdoInstrument() != null) {
537       requestedFeatures.add(CppRuleClasses.FDO_INSTRUMENT);
538     }
539     if (fdoSupport.getFdoOptimizeProfile() != null
540         && !fdoSupport.isAutoFdoEnabled()) {
541       requestedFeatures.add(CppRuleClasses.FDO_OPTIMIZE);
542     }
543     if (fdoSupport.isAutoFdoEnabled()) {
544       requestedFeatures.add(CppRuleClasses.AUTOFDO);
545     }
546     if (cppConfiguration.isLipoOptimizationOrInstrumentation()) {
547       requestedFeatures.add(CppRuleClasses.LIPO);
548     }
549 
550     FeatureConfiguration configuration =
551         toolchain.getFeatures().getFeatureConfiguration(requestedFeatures.build());
552     for (String feature : unsupportedFeatures) {
553       if (configuration.isEnabled(feature)) {
554         ruleContext.ruleError("The C++ toolchain '"
555             + ruleContext.getPrerequisite(":cc_toolchain", Mode.TARGET).getLabel()
556             + "' unconditionally implies feature '" + feature
557             + "', which is unsupported by this rule. "
558             + "This is most likely a misconfiguration in the C++ toolchain.");
559       }
560     }
561     return configuration; 
562   }

564   /**
565    * Creates a feature configuration for a given rule.
566    * 
567    * @param ruleContext the context of the rule we want the feature configuration for. 
568    * @return the feature configuration for the given {@code ruleContext}.
569    */
570   public static FeatureConfiguration configureFeatures(RuleContext ruleContext) {
571     return configureFeatures(ruleContext, ImmutableSet.<String>of(), ImmutableSet.<String>of());
