15  package com.google.devtools.build.lib.analysis;
16  
17  import com.google.common.collect.ImmutableMap;
18  import com.google.common.collect.ImmutableSet;
19  import com.google.devtools.build.lib.actions.Artifact;
20  import com.google.devtools.build.lib.collect.nestedset.NestedSet;
21  import com.google.devtools.build.lib.collect.nestedset.NestedSetBuilder;
22  import com.google.devtools.build.lib.collect.nestedset.Order;
23  import com.google.devtools.build.lib.concurrent.ThreadSafety.Immutable;
24  
25  /**
26   * {@code ConfiguredTarget}s implementing this interface can provide artifacts that <b>can</b> be
27   * built when the target is mentioned on the command line (as opposed to being always built, like
28   * {@link com.google.devtools.build.lib.analysis.FileProvider})
29   *
30   * <p>The artifacts are grouped into "output groups". Which output groups are built is controlled
31   * by the {@code --output_groups} undocumented command line option, which in turn is added to the
32   * command line at the discretion of the build command being run.
33   *
34   * <p>Output groups starting with an underscore are "not important". This means that artifacts built
35   * because such an output group is mentioned in a {@code --output_groups} command line option are
36   * not mentioned on the output.
37   */
38  @Immutable
39  public final class OutputGroupProvider implements TransitiveInfoProvider {
40  
41    /**
42     * Prefix for output groups that are not reported to the user on the terminal output of Blaze when
43     * they are built.
44     */
45    public static final String HIDDEN_OUTPUT_GROUP_PREFIX = "_";
46  
47    /**
48     * Baseline coverage artifacts.
49     */
50    public static final String BASELINE_COVERAGE = HIDDEN_OUTPUT_GROUP_PREFIX + "baseline_coverage";
51  
52    /**
53     * Building these artifacts only results in the compilation (and not e.g. linking) of the
54     * associated target. Mostly useful for C++, less so for e.g. Java.
55     */
56    public static final String FILES_TO_COMPILE = "files_to_compile";
57  
58    /**
59     * These artifacts are the direct requirements for compilation, but building these does not
60     * actually compile the target. Mostly useful when IDEs want Blaze to emit generated code so that
61     * they can do the compilation in their own way.
62     */
63    public static final String COMPILATION_PREREQUISITES = "compilation_prerequisites";
64  
65    /**
66     * These files are built when a target is mentioned on the command line, but are not reported to
67     * the user. This is mostly runfiles, which is necessary because we don't want a target to
68     * successfully build if a file in its runfiles is broken.
69     */
70    public static final String HIDDEN_TOP_LEVEL = HIDDEN_OUTPUT_GROUP_PREFIX + "hidden_top_level";
71  
72    /**
73     * Temporary files created during building a rule, for example, .i, .d and .s files for C++
74     * compilation.
75     *
76     * <p>This output group is somewhat special: it is always built, but it only contains files when
77     * the {@code --save_temps} command line option present. I'm not sure if this is to save RAM by
78     * not creating the associated actions and artifacts if we don't need them or just historical
79     * baggage.
80     */
81    public static final String TEMP_FILES = "temp_files";
82  
83    /**
84     * The default group of files built by a target when it is mentioned on the command line.
85     */
86    public static final String DEFAULT = "default";
87  
88    /**
89     * The default set of OutputGroups we typically want to build.
90     */
91    public static final ImmutableSet<String> DEFAULT_GROUPS =
92        ImmutableSet.of(DEFAULT, TEMP_FILES, HIDDEN_TOP_LEVEL);
93  
94    private final ImmutableMap<String, NestedSet<Artifact>> outputGroups;
95  
96    OutputGroupProvider(ImmutableMap<String, NestedSet<Artifact>> outputGroups) {
97      this.outputGroups = outputGroups;
98    }
99  
100   /** Return the artifacts in a particular output group.
101    *
102    * @return the artifacts in the output group with the given name. The return value is never null.
103    *     If the specified output group is not present, the empty set is returned.
104    */
105   public NestedSet<Artifact> getOutputGroup(String outputGroupName) {
106     return outputGroups.containsKey(outputGroupName)
107         ? outputGroups.get(outputGroupName)
108         : NestedSetBuilder.<Artifact>emptySet(Order.STABLE_ORDER);
109   }
110 }
