41  /**
42   * Immutable store of information needed for C++ compilation that is aggregated
43   * across dependencies.
44   */
45  @Immutable

47    /** An empty compilation context. */
48    public static final CppCompilationContext EMPTY = new Builder(null).build();

50    private final CommandLineContext commandLineContext;
51    private final ImmutableList<DepsContext> depsContexts;
52    private final CppModuleMap cppModuleMap;
53    private final Artifact headerModule;
54    private final Artifact picHeaderModule;

56    // True if this context is for a compilation that uses header modules.
57    private final boolean usingHeaderModules;

59    // Derived from depsContexts; no need to consider it for equals/hashCode.
60    private final ImmutableSet<Artifact> compilationPrerequisites;
61  
62    private CppCompilationContext(CommandLineContext commandLineContext,
63        List<DepsContext> depsContexts, CppModuleMap cppModuleMap, Artifact headerModule,
64        Artifact picHeaderModule, boolean usingHeaderModules) {
65      Preconditions.checkNotNull(commandLineContext);
66      Preconditions.checkArgument(!depsContexts.isEmpty());
67      this.commandLineContext = commandLineContext;
68      this.depsContexts = ImmutableList.copyOf(depsContexts);
69      this.cppModuleMap = cppModuleMap;
70      this.headerModule = headerModule;
71      this.picHeaderModule = picHeaderModule;
72      this.usingHeaderModules = usingHeaderModules;
73  
74      if (depsContexts.size() == 1) {
75        // Only LIPO targets have more than one DepsContexts. This codepath avoids creating
76        // an ImmutableSet.Builder for the vast majority of the cases.
77        compilationPrerequisites = (depsContexts.get(0).compilationPrerequisiteStampFile != null)
78            ? ImmutableSet.<Artifact>of(depsContexts.get(0).compilationPrerequisiteStampFile)
79            : ImmutableSet.<Artifact>of();
80      } else {
81        ImmutableSet.Builder<Artifact> prerequisites = ImmutableSet.builder();
82        for (DepsContext depsContext : depsContexts) {
83          if (depsContext.compilationPrerequisiteStampFile != null) {
84            prerequisites.add(depsContext.compilationPrerequisiteStampFile);
85          }
86        }
87        compilationPrerequisites = prerequisites.build();
88      }
89    }

105    * contains a single middleman, but if LIPO is used there can be two.

144   /**
145    * Returns the immutable set of declared include directories, relative to a
146    * "-I" or "-iquote" directory" (possibly empty but never null). The returned
147    * collection may contain duplicate elements.
148    *
149    * <p>Note: The iteration order of this list is preserved as ide_build_info
150    * writes these directories and sources out and the ordering will help when
151    * used by consumers.
152    */
153   public NestedSet<PathFragment> getDeclaredIncludeDirs() {
154     if (depsContexts.isEmpty()) {
155       return NestedSetBuilder.emptySet(Order.STABLE_ORDER);
156     }
157 
158     if (depsContexts.size() == 1) {
159       return depsContexts.get(0).declaredIncludeDirs;
160     }
161 
162     NestedSetBuilder<PathFragment> builder = NestedSetBuilder.stableOrder();
163     for (DepsContext depsContext : depsContexts) {
164       builder.addTransitive(depsContext.declaredIncludeDirs);
165     }
166 
167     return builder.build();
168   }

244   /**
245    * Returns the immutable set of additional transitive inputs needed for
246    * compilation, like C++ module map artifacts.
247    */
248   public NestedSet<Artifact> getAdditionalInputs() {
249     if (depsContexts.isEmpty()) {
250       return NestedSetBuilder.emptySet(Order.STABLE_ORDER);
251     }
252 
253     NestedSetBuilder<Artifact> builder = NestedSetBuilder.stableOrder();
254     for (DepsContext depsContext : depsContexts) {
255       builder.addTransitive(depsContext.topLevelHeaderModules);
256       builder.addTransitive(depsContext.impliedHeaderModules);
257       builder.addTransitive(depsContext.directModuleMaps);
258       if (usingHeaderModules) {
259         builder.addTransitive(depsContext.transitiveModuleMaps);
260       }
261     }
262     if (cppModuleMap != null) {
263       builder.add(cppModuleMap.getArtifact());
264     }
265 
266     return builder.build();
267   }

370   /**
371    * Returns a context that is based on a given context but returns empty sets
372    * for {@link #getDeclaredIncludeDirs()} and {@link #getDeclaredIncludeWarnDirs()}.
373    */
374   public static CppCompilationContext disallowUndeclaredHeaders(CppCompilationContext context) {
375     ImmutableList.Builder<DepsContext> builder = ImmutableList.builder();
376     for (DepsContext depsContext : context.depsContexts) {
377       builder.add(new DepsContext(
378           depsContext.compilationPrerequisiteStampFile,
379           NestedSetBuilder.<PathFragment>emptySet(Order.STABLE_ORDER),
380           NestedSetBuilder.<PathFragment>emptySet(Order.STABLE_ORDER),
381           depsContext.declaredIncludeSrcs,
382           depsContext.pregreppedHdrs,
383           depsContext.headerModuleSrcs,
384           depsContext.topLevelHeaderModules,
385           depsContext.impliedHeaderModules,
386           depsContext.transitiveHeaderModuleSrcs,
387           depsContext.transitiveModuleMaps,
388           depsContext.directModuleMaps));
389     }
390     return new CppCompilationContext(context.commandLineContext, builder.build(),
391         context.cppModuleMap, context.headerModule, context.picHeaderModule,
392         context.usingHeaderModules);
393   }

395   /**
396    * Returns the context for a LIPO compile action. This uses the include dirs
397    * and defines of the library, but the declared inclusion dirs/srcs from both
398    * the library and the owner binary.
399    *
400    * <p>TODO(bazel-team): this might make every LIPO target have an unnecessary large set of
401    * inclusion dirs/srcs. The correct behavior would be to merge only the contexts
402    * of actual referred targets (as listed in .imports file).
403    *
404    * <p>Undeclared inclusion checking ({@link #getDeclaredIncludeDirs()},
405    * {@link #getDeclaredIncludeWarnDirs()}, and
406    * {@link #getDeclaredIncludeSrcs()}) needs to use the union of the contexts
407    * of the involved source files.
408    *
409    * <p>For include and define command line flags ({@link #getIncludeDirs()}
410    * {@link #getQuoteIncludeDirs()}, {@link #getSystemIncludeDirs()}, and
411    * {@link #getDefines()}) LIPO compilations use the same values as non-LIPO
412    * compilation.
413    *
414    * <p>Include scanning is not handled by this method. See
415    * {@code IncludeScannable#getAuxiliaryScannables()} instead.
416    *
417    * @param ownerContext the compilation context of the owner binary
418    * @param libContext the compilation context of the library
419    */
420   public static CppCompilationContext mergeForLipo(CppCompilationContext ownerContext,
421       CppCompilationContext libContext) {
422     return new CppCompilationContext(libContext.commandLineContext,
423         ImmutableList.copyOf(Iterables.concat(ownerContext.depsContexts, libContext.depsContexts)),
424         libContext.cppModuleMap, libContext.headerModule, libContext.picHeaderModule,
425         /*usingHeaderModules=*/false);
426   }

498     private final NestedSet<PathFragment> declaredIncludeDirs;
499     private final NestedSet<PathFragment> declaredIncludeWarnDirs;
500     private final NestedSet<Artifact> declaredIncludeSrcs;
501     /**
502      * Module maps from direct dependencies.
503      */
504     private final NestedSet<Artifact> directModuleMaps;
505 
506     /**
507      * All declared headers of the current module, if compiled as a header module.
508      */
509     private final NestedSet<Artifact> headerModuleSrcs;
510 
511     /**
512      * All header modules in the transitive closure of {@code topLevelHeaderModules}.
513      */
514     private final NestedSet<Artifact> impliedHeaderModules;
515 
516     private final NestedSet<Pair<Artifact, Artifact>> pregreppedHdrs;

518     /**
519      * All header modules in our transitive closure that are not in the transitive closure of

521      */
522     private final NestedSet<Artifact> topLevelHeaderModules;
523 
524     /**
525      * Headers whose transitive closure of includes needs to be available when compiling the current

527      * header module, contains all headers that are part of its header module.
528      */
529     private final NestedSet<Artifact> transitiveHeaderModuleSrcs;
530 
531     /**
532      * The module maps from all targets the current target depends on transitively.
533      */
534     private final NestedSet<Artifact> transitiveModuleMaps;
535 
536     DepsContext(Artifact compilationPrerequisiteStampFile,
537         NestedSet<PathFragment> declaredIncludeDirs,
538         NestedSet<PathFragment> declaredIncludeWarnDirs,
539         NestedSet<Artifact> declaredIncludeSrcs,
540         NestedSet<Pair<Artifact, Artifact>> pregreppedHdrs,
541         NestedSet<Artifact> headerModuleSrcs,
542         NestedSet<Artifact> topLevelHeaderModules,
543         NestedSet<Artifact> impliedHeaderModules,
544         NestedSet<Artifact> transitiveHeaderModuleSrcs,
545         NestedSet<Artifact> transitiveModuleMaps,
546         NestedSet<Artifact> directModuleMaps) {
547       this.compilationPrerequisiteStampFile = compilationPrerequisiteStampFile;
548       this.declaredIncludeDirs = declaredIncludeDirs;
549       this.declaredIncludeWarnDirs = declaredIncludeWarnDirs;
550       this.declaredIncludeSrcs = declaredIncludeSrcs;
551       this.directModuleMaps = directModuleMaps;
552       this.headerModuleSrcs = headerModuleSrcs;
553       this.impliedHeaderModules = impliedHeaderModules;
554       this.pregreppedHdrs = pregreppedHdrs;
555       this.topLevelHeaderModules = topLevelHeaderModules;
556       this.transitiveModuleMaps = transitiveModuleMaps;
557       this.transitiveHeaderModuleSrcs = transitiveHeaderModuleSrcs;
558     }

560     @Override

563         return true;
564       }

566         return false;
567       }

570               compilationPrerequisiteStampFile, other.compilationPrerequisiteStampFile)
571           && Objects.equals(declaredIncludeDirs, other.declaredIncludeDirs)
572           && Objects.equals(declaredIncludeWarnDirs, other.declaredIncludeWarnDirs)
573           && Objects.equals(declaredIncludeSrcs, other.declaredIncludeSrcs)
574           && Objects.equals(directModuleMaps, other.directModuleMaps)
575           && Objects.equals(headerModuleSrcs, other.headerModuleSrcs)
576           && Objects.equals(impliedHeaderModules, other.impliedHeaderModules)
577           // TODO(bazel-team): add pregreppedHdrs?
578           && Objects.equals(topLevelHeaderModules, other.topLevelHeaderModules)
579           && Objects.equals(transitiveHeaderModuleSrcs, other.transitiveHeaderModuleSrcs)
580           && Objects.equals(transitiveModuleMaps, other.transitiveModuleMaps);

665     /**
666      * Merges the context of a dependency into this one by adding the contents
667      * of all of its attributes.
668      */
669     public Builder mergeDependentContext(CppCompilationContext otherContext) {
670       Preconditions.checkNotNull(otherContext);
671       compilationPrerequisites.addAll(otherContext.getCompilationPrerequisites());
672       includeDirs.addAll(otherContext.getIncludeDirs());
673       quoteIncludeDirs.addAll(otherContext.getQuoteIncludeDirs());
674       systemIncludeDirs.addAll(otherContext.getSystemIncludeDirs());
675       declaredIncludeDirs.addTransitive(otherContext.getDeclaredIncludeDirs());
676       declaredIncludeWarnDirs.addTransitive(otherContext.getDeclaredIncludeWarnDirs());
677       declaredIncludeSrcs.addTransitive(otherContext.getDeclaredIncludeSrcs());
678       pregreppedHdrs.addTransitive(otherContext.getPregreppedHeaders());
679 
680       NestedSet<Artifact> othersTransitiveModuleMaps = otherContext.getTransitiveModuleMaps();
681       NestedSet<Artifact> othersDirectModuleMaps = otherContext.getDirectModuleMaps();
682       NestedSet<Artifact> othersTopLevelHeaderModules = otherContext.getTopLevelHeaderModules();
683 
684       // Forward transitive information.
685       // The other target's transitive module maps do not include its direct module maps, so we
686       // add both.
687       transitiveModuleMaps.addTransitive(othersTransitiveModuleMaps);
688       transitiveModuleMaps.addTransitive(othersDirectModuleMaps);
689       transitiveHeaderModuleSrcs.addTransitive(otherContext.getTransitiveHeaderModuleSrcs());
690       impliedHeaderModules.addTransitive(otherContext.getImpliedHeaderModules());
691       topLevelHeaderModules.addTransitive(othersTopLevelHeaderModules);
692 
693       // All module maps of direct dependencies are inputs to the current compile independently of
694       // the build type.
695       if (otherContext.getCppModuleMap() != null) {
696         directModuleMaps.add(otherContext.getCppModuleMap().getArtifact());
697       }
698       if (otherContext.getHeaderModule() != null || otherContext.getPicHeaderModule() != null) {
699         // If the other context is for a target that compiles a header module, that context's
700         // header module becomes our top-level header module, and its top-level header modules
701         // become our implied header modules.
702         impliedHeaderModules.addTransitive(othersTopLevelHeaderModules);
703         if (otherContext.getHeaderModule() != null) {
704           topLevelHeaderModules.add(otherContext.getHeaderModule());
705         }
706         if (otherContext.getPicHeaderModule() != null) {
707           topLevelHeaderModules.add(otherContext.getPicHeaderModule());
708         }
709         // All targets transitively depending on us will need to have the full transitive #include
710         // closure of the headers in that module available.
711         transitiveHeaderModuleSrcs.addTransitive(otherContext.getHeaderModuleSrcs());
712       }
713 
714       defines.addAll(otherContext.getDefines());
715       return this;
716     }

719      * Merges the context of some targets into this one by adding the contents
720      * of all of their attributes. Targets that do not implement

723     public Builder mergeDependentContexts(Iterable<CppCompilationContext> targets) {

893     @VisibleForTesting  // productionVisibility = Visibility.PRIVATE
894     public CppCompilationContext build(ActionOwner owner, MiddlemanFactory middlemanFactory) {
895       // During merging we might have put header modules into topLevelHeaderModules that were
896       // also in the transitive closure of a different header module; we need to filter those out.
897       NestedSet<Artifact> impliedHeaderModules = this.impliedHeaderModules.build();
898       Set<Artifact> impliedHeaderModulesSet = impliedHeaderModules.toSet();
899       NestedSetBuilder<Artifact> topLevelHeaderModules = NestedSetBuilder.stableOrder();
900       for (Artifact artifact : this.topLevelHeaderModules.build()) {
901         if (!impliedHeaderModulesSet.contains(artifact)) {
902           topLevelHeaderModules.add(artifact);
903         }
904       }
905 
906       // We don't create middlemen in LIPO collector subtree, because some target CT
907       // will do that instead.
908       Artifact prerequisiteStampFile = (ruleContext != null
909           && ruleContext.getFragment(CppConfiguration.class).isLipoContextCollector())
910           ? getMiddlemanArtifact(middlemanFactory)
911           : createMiddleman(owner, middlemanFactory);
912 
913       return new CppCompilationContext(
914           new CommandLineContext(ImmutableList.copyOf(includeDirs),
915               ImmutableList.copyOf(quoteIncludeDirs), ImmutableList.copyOf(systemIncludeDirs),
916               ImmutableList.copyOf(defines)),
917           ImmutableList.of(new DepsContext(prerequisiteStampFile,
918               declaredIncludeDirs.build(),
919               declaredIncludeWarnDirs.build(),
920               declaredIncludeSrcs.build(),
921               pregreppedHdrs.build(),
922               headerModuleSrcs.build(),
923               topLevelHeaderModules.build(),
924               impliedHeaderModules,
925               transitiveHeaderModuleSrcs.build(),
926               transitiveModuleMaps.build(),
927               directModuleMaps.build())),
928           cppModuleMap,
929           headerModule,
930           picHeaderModule,
931           usingHeaderModules);
932     }

934     /**
935      * Creates a middleman for the compilation prerequisites.
936      *
937      * @return the middleman or null if there are no prerequisites
938      */
939     private Artifact createMiddleman(ActionOwner owner,
940         MiddlemanFactory middlemanFactory) {
941       if (compilationPrerequisites.isEmpty()) {
942         return null;
943       }
944 
945       // Compilation prerequisites gathered in the compilationPrerequisites
946       // must be generated prior to executing C++ compilation step that depends
947       // on them (since these prerequisites include all potential header files, etc
948       // that could be referenced during compilation). So there is a definite need
949       // to ensure scheduling edge dependency. However, those prerequisites should
950       // have no effect on the decision whether C++ compilation should happen in
951       // the first place - only CppCompileAction outputs (*.o and *.d files) and
952       // all files referenced by the *.d file should be used to make that decision.
953       // If this action was never executed, then *.d file would be missing, forcing
954       // compilation to occur. If *.d file is present and has not changed then the
955       // only reason that would force us to re-compile would be change in one of
956       // the files referenced by the *.d file, since no other files participated
957       // in the compilation. We also need to propagate errors through this
958       // dependency link. So we use an error propagating middleman.
959       // Such middleman will be ignored by the dependency checker yet will still
960       // represent an edge in the action dependency graph - forcing proper execution
961       // order and error propagation.
962       return middlemanFactory.createErrorPropagatingMiddleman(
963           owner, ruleContext.getLabel().toString(), purpose,
964           ImmutableList.copyOf(compilationPrerequisites),
965           ruleContext.getConfiguration().getMiddlemanDirectory());
966     }
967 
968     /**
969      * Returns the same set of artifacts as createMiddleman() would, but without
970      * actually creating middlemen.
971      */
972     private Artifact getMiddlemanArtifact(MiddlemanFactory middlemanFactory) {
973       if (compilationPrerequisites.isEmpty()) {
974         return null;
975       }
976 
977       return middlemanFactory.getErrorPropagatingMiddlemanArtifact(ruleContext.getLabel()
978           .toString(), purpose, ruleContext.getConfiguration().getMiddlemanDirectory());
979     }
