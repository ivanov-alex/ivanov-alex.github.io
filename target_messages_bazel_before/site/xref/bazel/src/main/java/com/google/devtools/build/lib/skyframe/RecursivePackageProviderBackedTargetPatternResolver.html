37  /**
38   * A {@link TargetPatternResolver} backed by a {@link RecursivePackageProvider}.
39   */
40  public class RecursivePackageProviderBackedTargetPatternResolver
41      implements TargetPatternResolver<Target> {
42  
43    private final RecursivePackageProvider recursivePackageProvider;
44    private final EventHandler eventHandler;
45    private final FilteringPolicy policy;
46    private final PathPackageLocator pkgPath;
47  
48    public RecursivePackageProviderBackedTargetPatternResolver(
49        RecursivePackageProvider recursivePackageProvider,
50        EventHandler eventHandler,
51        FilteringPolicy policy,
52        PathPackageLocator pkgPath) {
53      this.recursivePackageProvider = recursivePackageProvider;
54      this.eventHandler = eventHandler;
55      this.policy = policy;
56      this.pkgPath = pkgPath;
57    }
58  
59    @Override
60    public void warn(String msg) {
61      eventHandler.handle(Event.warn(msg));
62    }
63  
64    /**
65     * Gets a {@link Package} from the {@link RecursivePackageProvider}. May return a {@link Package}
66     * that has errors.
67     */
68    private Package getPackage(PackageIdentifier pkgIdentifier)
69        throws NoSuchPackageException, InterruptedException {
70      Package pkg;
71      try {
72        pkg = recursivePackageProvider.getPackage(eventHandler, pkgIdentifier);
73      } catch (NoSuchPackageException e) {
74        pkg = e.getPackage();
75        if (pkg == null) {
76          throw e;
77        }
78      }
79      return pkg;
80    }
81  
82    @Override
83    public Target getTargetOrNull(String targetName) throws InterruptedException {
84      try {
85        Label label = Label.parseAbsolute(targetName);
86        if (!isPackage(label.getPackageName())) {
87          return null;
88        }
89        return recursivePackageProvider.getTarget(eventHandler, label);
90      } catch (Label.SyntaxException | NoSuchThingException e) {
91        return null;
92      }
93    }
94  
95    @Override
96    public ResolvedTargets<Target> getExplicitTarget(String targetName)
97        throws TargetParsingException, InterruptedException {
98      Label label = TargetPatternResolverUtil.label(targetName);
99      try {
100       Target target = recursivePackageProvider.getTarget(eventHandler, label);
101       return policy.shouldRetain(target, true)
102           ? ResolvedTargets.of(target)
103           : ResolvedTargets.<Target>empty();
104     } catch (NoSuchThingException e) {
105       throw new TargetParsingException(e.getMessage(), e);
106     }
107   }
108 
109   @Override
110   public ResolvedTargets<Target> getTargetsInPackage(String originalPattern, String packageName,
111                                                      boolean rulesOnly)
112       throws TargetParsingException, InterruptedException {
113     FilteringPolicy actualPolicy = rulesOnly
114         ? FilteringPolicies.and(FilteringPolicies.RULES_ONLY, policy)
115         : policy;
116     return getTargetsInPackage(originalPattern, new PathFragment(packageName), actualPolicy);
117   }
118 
119   private ResolvedTargets<Target> getTargetsInPackage(String originalPattern,
120       PathFragment packageNameFragment, FilteringPolicy policy)
121       throws TargetParsingException, InterruptedException {
122     TargetPatternResolverUtil.validatePatternPackage(originalPattern, packageNameFragment, this);
123     try {
124       Package pkg = getPackage(PackageIdentifier.createInDefaultRepo(packageNameFragment));
125       return TargetPatternResolverUtil.resolvePackageTargets(pkg, policy);
126     } catch (NoSuchThingException e) {
127       String message = TargetPatternResolverUtil.getParsingErrorMessage(
128           "package contains errors", originalPattern);
129       throw new TargetParsingException(message, e);
130     }
131   }
132 
133   @Override
134   public boolean isPackage(String packageName) {
135     // TODO(bazel-team): this should get the whole PackageIdentifier. Using only the package name
136     // makes it impossible to use the //... wildcard to refer to targets in remote repositories.
137     return recursivePackageProvider.isPackage(
138         eventHandler, PackageIdentifier.createInDefaultRepo(packageName));
139   }
140 
141   @Override
142   public String getTargetKind(Target target) {
143     return target.getTargetKind();
144   }
145 
146   @Override
147   public ResolvedTargets<Target> findTargetsBeneathDirectory(String originalPattern,
148       String directory, boolean rulesOnly, ImmutableSet<String> excludedSubdirectories)
149       throws TargetParsingException, InterruptedException {
150     FilteringPolicy actualPolicy = rulesOnly
151         ? FilteringPolicies.and(FilteringPolicies.RULES_ONLY, policy)
152         : policy;
153     ImmutableSet<PathFragment> excludedPathFragments =
154         TargetPatternResolverUtil.getPathFragments(excludedSubdirectories);
155     PathFragment pathFragment = TargetPatternResolverUtil.getPathFragment(directory);
156     ResolvedTargets.Builder<Target> targetBuilder = ResolvedTargets.builder();
157     for (Path root : pkgPath.getPathEntries()) {
158       RootedPath rootedPath = RootedPath.toRootedPath(root, pathFragment);
159       Iterable<PathFragment> packagesUnderDirectory =
160           recursivePackageProvider.getPackagesUnderDirectory(rootedPath, excludedPathFragments);
161       for (PathFragment pkg : packagesUnderDirectory) {
162         targetBuilder.merge(getTargetsInPackage(originalPattern, pkg, FilteringPolicies.NO_FILTER));
163       }
164     }
165 
166     // Perform the no-targets-found check before applying the filtering policy so we only return the
167     // error if the input directory's subtree really contains no targets.
168     if (targetBuilder.isEmpty()) {
169       throw new TargetParsingException("no targets found beneath '" + pathFragment + "'");
170     }
171     ResolvedTargets<Target> prefilteredTargets = targetBuilder.build();
172      
173     ResolvedTargets.Builder<Target> filteredBuilder = ResolvedTargets.builder();
174     if (prefilteredTargets.hasError()) {
175       filteredBuilder.setError();
176     }
177     for (Target target : prefilteredTargets.getTargets()) {
178       if (actualPolicy.shouldRetain(target, false)) {
179         filteredBuilder.add(target);
180       }
181     }
182     return filteredBuilder.build();
183   }
184 }
