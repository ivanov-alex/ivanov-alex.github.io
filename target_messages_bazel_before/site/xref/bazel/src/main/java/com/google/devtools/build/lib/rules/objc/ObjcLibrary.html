32  /**
33   * Implementation for {@code objc_library}.
34   */
35  public class ObjcLibrary implements RuleConfiguredTargetFactory {
36  
37    /**
38     * An {@link IterableWrapper} containing extra library {@link Artifact}s to be linked into the
39     * final ObjC application bundle.
40     */
41    static final class ExtraImportLibraries extends IterableWrapper<Artifact> {
42  
43      ExtraImportLibraries(Artifact... extraImportLibraries) {
44        super(extraImportLibraries);
45      }
46  
47      ExtraImportLibraries(Iterable<Artifact> extraImportLibraries) {
48        super(extraImportLibraries);
49      }
50    }
51  
52    /**
53     * Constructs an {@link ObjcCommon} instance based on the attributes of the given rule. The rule
54     * should inherit from {@link ObjcLibraryRule}..
55     */
56    static ObjcCommon common(RuleContext ruleContext, Iterable<SdkFramework> extraSdkFrameworks,
57        boolean alwayslink, ExtraImportLibraries extraImportLibraries,
58        Iterable<ObjcProvider> extraDepObjcProviders) {
59      CompilationArtifacts compilationArtifacts =
60          CompilationSupport.compilationArtifacts(ruleContext);
61  
62      return new ObjcCommon.Builder(ruleContext)
63          .setCompilationAttributes(new CompilationAttributes(ruleContext))
64          .setResourceAttributes(new ResourceAttributes(ruleContext))
65          .addExtraSdkFrameworks(extraSdkFrameworks)
66          .addDefines(ruleContext.getTokenizedStringListAttr("defines"))
67          .setCompilationArtifacts(compilationArtifacts)
68          .addDepObjcProviders(ruleContext.getPrerequisites("deps", Mode.TARGET, ObjcProvider.class))
69          .addDepObjcProviders(
70              ruleContext.getPrerequisites("bundles", Mode.TARGET, ObjcProvider.class))
71          .addNonPropagatedDepObjcProviders(ruleContext.getPrerequisites("non_propagated_deps",
72              Mode.TARGET, ObjcProvider.class))
73          .addDepCcHeaderProviders(
74              ruleContext.getPrerequisites("deps", Mode.TARGET, CppCompilationContext.class))
75          .addDepCcLinkProviders(
76              ruleContext.getPrerequisites("deps", Mode.TARGET, CcLinkParamsProvider.class))
77          .setIntermediateArtifacts(ObjcRuleClasses.intermediateArtifacts(ruleContext))
78          .setAlwayslink(alwayslink)
79          .addExtraImportLibraries(extraImportLibraries)
80          .addDepObjcProviders(extraDepObjcProviders)
81          .build();
82    }
83  
84    @Override
85    public ConfiguredTarget create(RuleContext ruleContext) throws InterruptedException {
86      ObjcCommon common = common(
87          ruleContext, ImmutableList.<SdkFramework>of(),
88          ruleContext.attributes().get("alwayslink", Type.BOOLEAN), new ExtraImportLibraries(),
89          ImmutableList.<ObjcProvider>of());
90  
91      XcodeProvider.Builder xcodeProviderBuilder = new XcodeProvider.Builder();
92      NestedSetBuilder<Artifact> filesToBuild = NestedSetBuilder.<Artifact>stableOrder()
93          .addAll(common.getCompiledArchive().asSet());
94  
95      new CompilationSupport(ruleContext)
96          .registerCompileAndArchiveActions(common)
97          .addXcodeSettings(xcodeProviderBuilder, common)
98          .validateAttributes();
99  
100     new ResourceSupport(ruleContext)
101         .validateAttributes()
102         .addXcodeSettings(xcodeProviderBuilder);
103 
104     new XcodeSupport(ruleContext)
105         .addFilesToBuild(filesToBuild)
106         .addXcodeSettings(xcodeProviderBuilder, common.getObjcProvider(), LIBRARY_STATIC)
107         .addDependencies(xcodeProviderBuilder, new Attribute("bundles", Mode.TARGET))
108         .addDependencies(xcodeProviderBuilder, new Attribute("deps", Mode.TARGET))
109         .addNonPropagatedDependencies(
110             xcodeProviderBuilder, new Attribute("non_propagated_deps", Mode.TARGET))
111         .registerActions(xcodeProviderBuilder.build());
112 
113     return ObjcRuleClasses.ruleConfiguredTarget(ruleContext, filesToBuild.build())
114         .addProvider(XcodeProvider.class, xcodeProviderBuilder.build())
115         .addProvider(ObjcProvider.class, common.getObjcProvider())
116         .addProvider(J2ObjcSrcsProvider.class, J2ObjcSrcsProvider.buildFrom(ruleContext))
117         .addProvider(
118             J2ObjcMappingFileProvider.class, ObjcRuleClasses.j2ObjcMappingFileProvider(ruleContext))
119         .build();
120   }
