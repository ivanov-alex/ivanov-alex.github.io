36  /**
37   * Rule class definitions for C++ rules.
38   */
39  public class CppRuleClasses {
40    // Artifacts of these types are discarded from the 'hdrs' attribute in cc rules
41    static final FileTypeSet DISALLOWED_HDRS_FILES = FileTypeSet.of(
42        ARCHIVE,
43        PIC_ARCHIVE,
44        ALWAYS_LINK_LIBRARY,
45        ALWAYS_LINK_PIC_LIBRARY,
46        SHARED_LIBRARY,
47        VERSIONED_SHARED_LIBRARY,
48        OBJECT_FILE,
49        PIC_OBJECT_FILE);
50  
51    /**
52     * The set of instrumented source file types; keep this in sync with the list above. Note that
53     * extension-less header files cannot currently be declared, so we cannot collect coverage for
54     * those.
55     */
56    static final InstrumentationSpec INSTRUMENTATION_SPEC = new InstrumentationSpec(
57        FileTypeSet.of(CPP_SOURCE, C_SOURCE, CPP_HEADER, ASSEMBLER_WITH_C_PREPROCESSOR),
58        "srcs", "deps", "data", "hdrs");
59  
60    public static final LibraryLanguage LANGUAGE = new LibraryLanguage("C++");
61  
62    /**
63     * Implicit outputs for cc_binary rules.
64     */
65    public static final SafeImplicitOutputsFunction CC_BINARY_STRIPPED =
66        fromTemplates("%{name}.stripped");
67  
68  
69    // Used for requesting dwp "debug packages".
70    public static final SafeImplicitOutputsFunction CC_BINARY_DEBUG_PACKAGE =
71        fromTemplates("%{name}.dwp");
72  
73  
74    /**
75     * Path of the build_interface_so script in the Blaze binary.
76     */
77    public static final String BUILD_INTERFACE_SO = "build_interface_so";
78  
79    /**
80     * A string constant for the parse_headers feature.
81     */
82    public static final String PARSE_HEADERS = "parse_headers";
83  
84    /**
85     * A string constant for the preprocess_headers feature.
86     */
87    public static final String PREPROCESS_HEADERS = "preprocess_headers";
88  
89    /**
90     * A string constant for the module_maps feature; this is a precondition to the layering_check and
91     * header_modules features.
92     */
93    public static final String MODULE_MAPS = "module_maps";
94  
95    /**
96     * A string constant for the module_map_home_cwd feature.
97     */
98    public static final String MODULE_MAP_HOME_CWD = "module_map_home_cwd";
99  
100   /**
101    * A string constant for the layering_check feature.
102    */
103   public static final String LAYERING_CHECK = "layering_check";
104 
105   /**
106    * A string constant for the header_modules feature.
107    */
108   public static final String HEADER_MODULES = "header_modules";
109 
110   /**
111    * A string constant for the use_header_modules feature.
112    *
113    * <p>This feature is only used during rollout; we expect to default enable this once we
114    * have verified that module-enabled compilation is stable enough.
115    */
116   public static final String USE_HEADER_MODULES = "use_header_modules";
117 
118   /**
119    * A string constant for the generate_submodules feature.
120    *
121    * <p>This feature is only used temporarily to make the switch to using submodules easier. With
122    * submodules, each header of a cc_library is placed into a submodule of the module generated for
123    * the appropriate target. As this influences the layering_check semantics and needs to be synced
124    * with a clang release, we want to be able to switch back and forth easily.
125    */
126   public static final String GENERATE_SUBMODULES = "generate_submodules";
127 
128   /**
129    * A string constant for switching on that a header module file includes information about
130    * all its dependencies, so we do not need to pass all transitive dependent header modules on
131    * the command line.
132    */
133   public static final String HEADER_MODULE_INCLUDES_DEPENDENCIES =
134       "header_module_includes_dependencies";
135 
136   /**
137    * A string constant for the no_legacy_features feature.
138    *
139    * <p>If this feature is enabled, Bazel will not extend the crosstool configuration with the
140    * default legacy feature set.
141    */
142   public static final String NO_LEGACY_FEATURES = "no_legacy_features";
143 
144   /**
145    * A string constant for the include_paths feature.
146    */
147   public static final String INCLUDE_PATHS = "include_paths";
148 
149   /**
150    * A string constant for the fdo_instrument feature.
151    */
152   public static final String FDO_INSTRUMENT = "fdo_instrument";
153 
154   /**
155    * A string constant for the fdo_optimize feature.
156    */
157   public static final String FDO_OPTIMIZE = "fdo_optimize";
158 
159   /**
160    * A string constant for the autofdo feature.
161    */
162   public static final String AUTOFDO = "autofdo";
163 
164   /**
165    * A string constant for the lipo feature.
166    */
167   public static final String LIPO = "lipo";
168 }
