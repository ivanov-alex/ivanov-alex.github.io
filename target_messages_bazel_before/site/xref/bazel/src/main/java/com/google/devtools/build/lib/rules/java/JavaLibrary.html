43  /**
44   * Implementation for the java_library rule.
45   */
46  public class JavaLibrary implements RuleConfiguredTargetFactory {
47    private final JavaSemantics semantics;
48  
49    protected JavaLibrary(JavaSemantics semantics) {
50      this.semantics = semantics;
51    }
52  
53    @Override
54    public ConfiguredTarget create(RuleContext ruleContext) {
55      JavaCommon common = new JavaCommon(ruleContext, semantics);
56      RuleConfiguredTargetBuilder builder = init(ruleContext, common);
57      return builder != null ? builder.build() : null;
58    }
59  
60    public RuleConfiguredTargetBuilder init(RuleContext ruleContext, final JavaCommon common) {
61      common.initializeJavacOpts();
62      JavaTargetAttributes.Builder attributesBuilder = common.initCommon();
63  
64      // Collect the transitive dependencies.
65      JavaCompilationHelper helper = new JavaCompilationHelper(
66          ruleContext, semantics, common.getJavacOpts(), attributesBuilder);
67      helper.addLibrariesToAttributes(common.targetsTreatedAsDeps(ClasspathType.COMPILE_ONLY));
68      helper.addProvidersToAttributes(common.compilationArgsFromSources(),
69          JavaCommon.isNeverLink(ruleContext));
70  
71      if (ruleContext.hasErrors()) {
72        return null;
73      }
74  
75      semantics.checkRule(ruleContext, common);
76  
77      JavaCompilationArtifacts.Builder javaArtifactsBuilder = new JavaCompilationArtifacts.Builder();
78  
79      if (ruleContext.hasErrors()) {
80        common.setJavaCompilationArtifacts(JavaCompilationArtifacts.EMPTY);
81        return null;
82      }
83  
84      JavaConfiguration javaConfig = ruleContext.getFragment(JavaConfiguration.class);
85      NestedSetBuilder<Artifact> filesBuilder = NestedSetBuilder.stableOrder();
86  
87      JavaTargetAttributes attributes = helper.getAttributes();
88      if (attributes.hasJarFiles()) {
89        // This rule is repackaging some source jars as a java library.
90        Set<Artifact> jarFiles = attributes.getJarFiles();
91        javaArtifactsBuilder.addRuntimeJars(jarFiles);
92        javaArtifactsBuilder.addCompileTimeJars(attributes.getCompileTimeJarFiles());
93  
94        filesBuilder.addAll(jarFiles);
95      }
96      if (attributes.hasMessages()) {
97        helper.addTranslations(semantics.translate(ruleContext, javaConfig,
98            attributes.getMessages()));
99      }
100 
101     ruleContext.checkSrcsSamePackage(true);
102 
103     Artifact jar = null;
104 
105     Artifact srcJar = ruleContext.getImplicitOutputArtifact(
106         JavaSemantics.JAVA_LIBRARY_SOURCE_JAR);
107 
108     Artifact classJar = ruleContext.getImplicitOutputArtifact(
109         JavaSemantics.JAVA_LIBRARY_CLASS_JAR);
110 
111     if (attributes.hasSourceFiles() || attributes.hasSourceJars() || attributes.hasResources()
112         || attributes.hasMessages()) {
113       // We only want to add a jar to the classpath of a dependent rule if it has content.
114       javaArtifactsBuilder.addRuntimeJar(classJar);
115       jar = classJar;
116     }
117 
118     filesBuilder.add(classJar);
119 
120     // The gensrc jar is created only if the target uses annotation processing.
121     // Otherwise, it is null, and the source jar action will not depend on the compile action.
122     Artifact gensrcJar = helper.createGensrcJar(classJar);
123     Artifact manifestProtoOutput = helper.createManifestProtoOutput(classJar);
124 
125     Artifact genClassJar = ruleContext.getImplicitOutputArtifact(
126         JavaSemantics.JAVA_LIBRARY_GEN_JAR);
127     helper.createGenJarAction(classJar, manifestProtoOutput, genClassJar);
128 
129     Artifact outputDepsProto = helper.createOutputDepsProtoArtifact(classJar, javaArtifactsBuilder);
130 
131     helper.createCompileActionWithInstrumentation(classJar, manifestProtoOutput, gensrcJar,
132         outputDepsProto, javaArtifactsBuilder);
133     helper.createSourceJarAction(srcJar, gensrcJar);
134 
135     if ((attributes.hasSourceFiles() || attributes.hasSourceJars()) && jar != null) {
136       helper.createCompileTimeJarAction(jar, outputDepsProto,
137           javaArtifactsBuilder);
138     }
139 
140     boolean neverLink = JavaCommon.isNeverLink(ruleContext);
141     common.setJavaCompilationArtifacts(javaArtifactsBuilder.build());
142     common.setClassPathFragment(new ClasspathConfiguredFragment(
143         common.getJavaCompilationArtifacts(), attributes, neverLink));
144     CppCompilationContext transitiveCppDeps = common.collectTransitiveCppDeps();
145 
146     NestedSet<Artifact> transitiveSourceJars = common.collectTransitiveSourceJars(srcJar);
147 
148     // If sources are empty, treat this library as a forwarding node for dependencies.
149     JavaCompilationArgs javaCompilationArgs = common.collectJavaCompilationArgs(
150         false, neverLink, common.compilationArgsFromSources());
151     JavaCompilationArgs recursiveJavaCompilationArgs = common.collectJavaCompilationArgs(
152         true, neverLink, common.compilationArgsFromSources());
153     NestedSet<Artifact> compileTimeJavaDepArtifacts = common.collectCompileTimeDependencyArtifacts(
154         common.getJavaCompilationArtifacts().getCompileTimeDependencyArtifact());
155     NestedSet<Artifact> runTimeJavaDepArtifacts = NestedSetBuilder.emptySet(Order.STABLE_ORDER);
156     NestedSet<LinkerInput> transitiveJavaNativeLibraries =
157         common.collectTransitiveJavaNativeLibraries();
158 
159     ImmutableList<String> exportedProcessorClasses = ImmutableList.of();
160     NestedSet<Artifact> exportedProcessorClasspath =
161         NestedSetBuilder.emptySet(Order.NAIVE_LINK_ORDER);
162     ImmutableList.Builder<String> processorClasses = ImmutableList.builder();
163     NestedSetBuilder<Artifact> processorClasspath = NestedSetBuilder.naiveLinkOrder();
164     for (JavaPluginInfoProvider provider : Iterables.concat(
165         common.getPluginInfoProvidersForAttribute("exported_plugins", Mode.HOST),
166         common.getPluginInfoProvidersForAttribute("exports", Mode.TARGET))) {
167       processorClasses.addAll(provider.getProcessorClasses());
168       processorClasspath.addTransitive(provider.getProcessorClasspath());
169     }
170     exportedProcessorClasses = processorClasses.build();
171     exportedProcessorClasspath = processorClasspath.build();
172 
173     CcLinkParamsStore ccLinkParamsStore = new CcLinkParamsStore() {
174       @Override
175       protected void collect(CcLinkParams.Builder builder, boolean linkingStatically,
176                              boolean linkShared) {
177         builder.addTransitiveTargets(common.targetsTreatedAsDeps(ClasspathType.BOTH),
178             JavaCcLinkParamsProvider.TO_LINK_PARAMS, CcLinkParamsProvider.TO_LINK_PARAMS);
179       }
180     };
181 
182     // The "neverlink" attribute is transitive, so we don't add any
183     // runfiles from this target or its dependencies.
184     Runfiles runfiles = Runfiles.EMPTY;
185     if (!neverLink) {
186       Runfiles.Builder runfilesBuilder = new Runfiles.Builder().addArtifacts(
187           common.getJavaCompilationArtifacts().getRuntimeJars());
188 
189 
190       runfilesBuilder.addRunfiles(ruleContext, RunfilesProvider.DEFAULT_RUNFILES);
191       runfilesBuilder.add(ruleContext, JavaRunfilesProvider.TO_RUNFILES);
192 
193       List<TransitiveInfoCollection> depsForRunfiles = new ArrayList<>();
194       if (ruleContext.getRule().isAttrDefined("runtime_deps", Type.LABEL_LIST)) {
195         depsForRunfiles.addAll(ruleContext.getPrerequisites("runtime_deps", Mode.TARGET));
196       }
197       if (ruleContext.getRule().isAttrDefined("exports", Type.LABEL_LIST)) {
198         depsForRunfiles.addAll(ruleContext.getPrerequisites("exports", Mode.TARGET));
199       }
200 
201       runfilesBuilder.addTargets(depsForRunfiles, RunfilesProvider.DEFAULT_RUNFILES);
202       runfilesBuilder.addTargets(depsForRunfiles, JavaRunfilesProvider.TO_RUNFILES);
203 
204       TransitiveInfoCollection launcher = JavaHelper.launcherForTarget(semantics, ruleContext);
205       if (launcher != null) {
206         runfilesBuilder.addTarget(launcher, RunfilesProvider.DATA_RUNFILES);
207       }
208 
209       semantics.addRunfilesForLibrary(ruleContext, runfilesBuilder);
210       runfiles = runfilesBuilder.build();
211     }
212 
213     RuleConfiguredTargetBuilder builder =
214         new RuleConfiguredTargetBuilder(ruleContext);
215 
216     semantics.addProviders(
217         ruleContext, common, ImmutableList.<String>of(), classJar, srcJar, genClassJar, gensrcJar,
218         ImmutableMap.<Artifact, Artifact>of(), helper, filesBuilder, builder);
219 
220     NestedSet<Artifact> filesToBuild = filesBuilder.build();
221     common.addTransitiveInfoProviders(builder, filesToBuild, classJar);
222 
223     builder
224         .add(JavaRuntimeJarProvider.class,
225             new JavaRuntimeJarProvider(common.getJavaCompilationArtifacts().getRuntimeJars()))
226         .add(RunfilesProvider.class, RunfilesProvider.simple(runfiles))
227         .setFilesToBuild(filesToBuild)
228         .addSkylarkTransitiveInfo(JavaSkylarkApiProvider.NAME, new JavaSkylarkApiProvider())
229         .add(JavaNeverlinkInfoProvider.class, new JavaNeverlinkInfoProvider(neverLink))
230         .add(CppCompilationContext.class, transitiveCppDeps)
231         .add(JavaCompilationArgsProvider.class, new JavaCompilationArgsProvider(
232             javaCompilationArgs, recursiveJavaCompilationArgs,
233             compileTimeJavaDepArtifacts, runTimeJavaDepArtifacts))
234         .add(CcLinkParamsProvider.class, new CcLinkParamsProvider(ccLinkParamsStore))
235         .add(JavaNativeLibraryProvider.class, new JavaNativeLibraryProvider(
236             transitiveJavaNativeLibraries))
237         .add(JavaSourceInfoProvider.class,
238             JavaSourceInfoProvider.fromJavaTargetAttributes(attributes, semantics))
239         .add(JavaSourceJarsProvider.class, new JavaSourceJarsProvider(
240             transitiveSourceJars, ImmutableList.of(srcJar)))
241         // TODO(bazel-team): this should only happen for java_plugin
242         .add(JavaPluginInfoProvider.class, new JavaPluginInfoProvider(
243             exportedProcessorClasses, exportedProcessorClasspath))
244         .addOutputGroup(JavaSemantics.SOURCE_JARS_OUTPUT_GROUP, transitiveSourceJars)
245         .addOutputGroup(JavaSemantics.GENERATED_JARS_OUTPUT_GROUP, genClassJar);
246 
247     if (ruleContext.hasErrors()) {
248       return null;
249     }
250 
251     return builder;
252   }
