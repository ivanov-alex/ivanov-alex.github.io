36  /**
37   * Test summary entry. Stores summary information for a single test rule.
38   * Also used to sort summary output by status.
39   *
40   * <p>Invariant:
41   * All TestSummary mutations should be performed through the Builder.
42   * No direct TestSummary methods (except the constructor) may mutate the object.
43   */
44  @VisibleForTesting // Ideally package-scoped.
45  public class TestSummary implements Comparable<TestSummary> {

53      private Builder() {
54        summary = new TestSummary();
55        built = false;
56      }

58      private void mergeFrom(TestSummary existingSummary) {
59        // Yuck, manually fill in fields.
60        summary.shardRunStatuses = ArrayListMultimap.create(existingSummary.shardRunStatuses);
61        setTarget(existingSummary.target);
62        setStatus(existingSummary.status);
63        addCoverageFiles(existingSummary.coverageFiles);
64        addPassedLogs(existingSummary.passedLogs);
65        addFailedLogs(existingSummary.failedLogs);
66  
67        if (existingSummary.failedTestCasesStatus != null) {
68          addFailedTestCases(existingSummary.getFailedTestCases(),
69              existingSummary.getFailedTestCasesStatus());
70        }
71  
72        addTestTimes(existingSummary.testTimes);
73        addWarnings(existingSummary.warnings);
74        setActionRan(existingSummary.actionRan);
75        setNumCached(existingSummary.numCached);
76        setRanRemotely(existingSummary.ranRemotely);
77        setWasUnreportedWrongSize(existingSummary.wasUnreportedWrongSize);
78      }
79  
80      // Implements copy on write logic, allowing reuse of the same builder.
81      private void checkMutation() {
82        // If mutating the builder after an object was built, create another copy.
83        if (built) {
84          built = false;
85          TestSummary lastSummary = summary;
86          summary = new TestSummary();
87          mergeFrom(lastSummary);
88        }
89      }
90  
91      // This used to return a reference to the value on success.
92      // However, since it can alter the summary member, inlining it in an
93      // assignment to a property of summary was unsafe.
94      private void checkMutation(Object value) {
95        Preconditions.checkNotNull(value);
96        checkMutation();
97      }

129     public Builder collectFailedTests(TestCase testCase) {
130       if (testCase == null) {
131         summary.failedTestCasesStatus = FailedTestCasesStatus.NOT_AVAILABLE;
132         return this;
133       }
134       summary.failedTestCasesStatus = FailedTestCasesStatus.FULL;
135       return collectFailedTestCases(testCase);
136     }
137 
138     private Builder collectFailedTestCases(TestCase testCase) {
139       if (testCase.getChildCount() > 0) {
140         // This is a non-leaf result. Traverse its children, but do not add its
141         // name to the output list. It should not contain any 'failure' or
142         // 'error' tags, but we want to be lax here, because the syntax of the
143         // test.xml file is also lax.
144         for (TestCase child : testCase.getChildList()) {
145           collectFailedTestCases(child);
146         }
147       } else {
148         // This is a leaf result. If it passed, don't add it.
149         if (testCase.getStatus() == TestCase.Status.PASSED) {
150           return this;
151         }
152 
153         String name = testCase.getName();
154         String className = testCase.getClassName();
155         if (name == null || className == null) {
156           // A test case detail is not really interesting if we cannot tell which
157           // one it is.
158           this.summary.failedTestCasesStatus = FailedTestCasesStatus.PARTIAL;
159           return this;
160         }
161 
162         this.summary.failedTestCases.add(testCase);
163       }
164       return this;
165     }
166 
167     public Builder addFailedTestCases(List<TestCase> testCases, FailedTestCasesStatus status) {
168       checkMutation(status);
169       checkMutation(testCases);
170 
171       if (summary.failedTestCasesStatus == null) {
172         summary.failedTestCasesStatus = status;
173       } else if (summary.failedTestCasesStatus != status) {
174         summary.failedTestCasesStatus = FailedTestCasesStatus.PARTIAL;
175       }
176 
177       if (testCases.isEmpty()) {
178         return this;
179       }
180 
181       // union of summary.failedTestCases, testCases
182       Map<String, TestCase> allCases = new TreeMap<>();
183       if (summary.failedTestCases != null) {
184         for (TestCase detail : summary.failedTestCases) {
185           allCases.put(detail.getClassName() + "." + detail.getName(), detail);
186         }
187       }
188       for (TestCase detail : testCases) {
189         allCases.put(detail.getClassName() + "." + detail.getName(), detail);
190       }
191 
192       summary.failedTestCases = new ArrayList<>(allCases.values());
193       return this;
194     }

244     /**
245      * Records a new result for the given shard of the test.
246      *
247      * @return an immutable view of the statuses associated with the shard, with the new element.
248      */
249     public List<BlazeTestStatus> addShardStatus(int shardNumber, BlazeTestStatus status) {
250       Preconditions.checkState(summary.shardRunStatuses.put(shardNumber, status),
251           "shardRunStatuses must allow duplicate statuses");
252       return ImmutableList.copyOf(summary.shardRunStatuses.get(shardNumber));
253     }
254 
255     /**
256      * Returns the created TestSummary object.
257      * Any actions following a build() will create another copy of the same values.
258      * Since no mutators are provided directly by TestSummary, a copy will not
259      * be produced if two builds are invoked in a row without calling a setter.
260      */
261     public TestSummary build() {
262       peek();
263       if (!built) {
264         makeSummaryImmutable();
265         // else: it is already immutable.
266       }
267       Preconditions.checkState(built, "Built flag was not set");
268       return summary;
269     }
270 
271     /**
272      * Within-package, it is possible to read directly from an
273      * incompletely-built TestSummary. Used to pass Builders around directly.
274      */
275     TestSummary peek() {
276       Preconditions.checkNotNull(summary.target, "Target cannot be null");
277       Preconditions.checkNotNull(summary.status, "Status cannot be null");
278       return summary;
279     }

281     private void makeSummaryImmutable() {
282       // Once finalized, the list types are immutable.
283       summary.passedLogs = Collections.unmodifiableList(summary.passedLogs);
284       summary.failedLogs = Collections.unmodifiableList(summary.failedLogs);
285       summary.warnings = Collections.unmodifiableList(summary.warnings);
286       summary.coverageFiles = Collections.unmodifiableList(summary.coverageFiles);
287       summary.testTimes = Collections.unmodifiableList(summary.testTimes);
288 
289       built = true;
290     }

412   private static int getSortKey(BlazeTestStatus status) {
413     return status == BlazeTestStatus.PASSED ? -1 : status.ordinal();
414   }
415 
416   @Override
417   public int compareTo(TestSummary that) {
418     if (this.isCached() != that.isCached()) {
419       return this.isCached() ? -1 : 1;
420     } else if ((this.isCached() && that.isCached()) && (this.numUncached() != that.numUncached())) {
421       return this.numUncached() - that.numUncached();
422     } else if (this.status != that.status) {
423       return getSortKey(this.status) - getSortKey(that.status);
424     } else {
425       Artifact thisExecutable = this.target.getProvider(FilesToRunProvider.class).getExecutable();
426       Artifact thatExecutable = that.target.getProvider(FilesToRunProvider.class).getExecutable();
427       return thisExecutable.getPath().compareTo(thatExecutable.getPath());
428     }
429   }

444   static Mode getStatusMode(BlazeTestStatus status) {
