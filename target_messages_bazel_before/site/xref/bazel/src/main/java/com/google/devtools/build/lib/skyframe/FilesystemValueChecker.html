57  /**
58   * A helper class to find dirty values by accessing the filesystem directly (contrast with
59   * {@link DiffAwareness}).
60   */
61  class FilesystemValueChecker {
62  
63    private static final int DIRTINESS_CHECK_THREADS = 50;
64    private static final Logger LOG = Logger.getLogger(FilesystemValueChecker.class.getName());
65  
66    private static final Predicate<SkyKey> ACTION_FILTER =
67        SkyFunctionName.functionIs(SkyFunctions.ACTION_EXECUTION);
68  
69    private final TimestampGranularityMonitor tsgm;
70    private final Range<Long> lastExecutionTimeRange;
71    private final Supplier<Map<SkyKey, SkyValue>> valuesSupplier;
72    private AtomicInteger modifiedOutputFilesCounter = new AtomicInteger(0);
73    private AtomicInteger modifiedOutputFilesIntraBuildCounter = new AtomicInteger(0);
74  
75    FilesystemValueChecker(final MemoizingEvaluator evaluator, TimestampGranularityMonitor tsgm,
76        Range<Long> lastExecutionTimeRange) {
77      this.tsgm = tsgm;
78      this.lastExecutionTimeRange = lastExecutionTimeRange;
79  
80      // Construct the full map view of the entire graph at most once ("memoized"), lazily. If
81      // getDirtyFilesystemValues(Iterable<SkyKey>) is called on an empty Iterable, we avoid having
82      // to create the Map of value keys to values. This is useful in the case where the graph
83      // getValues() method could be slow.
84      this.valuesSupplier = Suppliers.memoize(new Supplier<Map<SkyKey, SkyValue>>() {
85        @Override
86        public Map<SkyKey, SkyValue> get() {
87          return evaluator.getValues();
88        }
89      });
90    }
91  
92    /**
93     * Returns a {@link Differencer.Diff} containing keys that are dirty according to the passed-in
94     * {@param dirtinessChecker}.
95     */
96    Differencer.Diff getDirtyKeys(SkyValueDirtinessChecker dirtinessChecker)
97        throws InterruptedException {
98      return getDirtyValues(valuesSupplier.get().keySet(), dirtinessChecker);
99    }

101   /**
102    * Return a collection of action values which have output files that are not in-sync with
103    * the on-disk file value (were modified externally).
104    */
105   Collection<SkyKey> getDirtyActionValues(@Nullable final BatchStat batchStatter)
106       throws InterruptedException {
107     // CPU-bound (usually) stat() calls, plus a fudge factor.
108     LOG.info("Accumulating dirty actions");
109     final int numOutputJobs = Runtime.getRuntime().availableProcessors() * 4;
110     final Set<SkyKey> actionSkyKeys =
111         Sets.filter(valuesSupplier.get().keySet(), ACTION_FILTER);
112     final Sharder<Pair<SkyKey, ActionExecutionValue>> outputShards =
113         new Sharder<>(numOutputJobs, actionSkyKeys.size());
114 
115     for (SkyKey key : actionSkyKeys) {
116       outputShards.add(Pair.of(key, (ActionExecutionValue) valuesSupplier.get().get(key)));
117     }
118     LOG.info("Sharded action values for batching");
119 
120     ExecutorService executor = Executors.newFixedThreadPool(
121         numOutputJobs,
122         new ThreadFactoryBuilder().setNameFormat("FileSystem Output File Invalidator %d").build());
123 
124     Collection<SkyKey> dirtyKeys = Sets.newConcurrentHashSet();
125     ThrowableRecordingRunnableWrapper wrapper =
126         new ThrowableRecordingRunnableWrapper("FileSystemValueChecker#getDirtyActionValues");
127 
128     modifiedOutputFilesCounter.set(0);
129     modifiedOutputFilesIntraBuildCounter.set(0);
130     for (List<Pair<SkyKey, ActionExecutionValue>> shard : outputShards) {
131       Runnable job = (batchStatter == null)
132           ? outputStatJob(dirtyKeys, shard)
133           : batchStatJob(dirtyKeys, shard, batchStatter);
134       executor.submit(wrapper.wrap(job));
135     }
136 
137     boolean interrupted = ExecutorUtil.interruptibleShutdown(executor);
138     Throwables.propagateIfPossible(wrapper.getFirstThrownError());
139     LOG.info("Completed output file stat checks");
140     if (interrupted) {
141       throw new InterruptedException();
142     }
143     return dirtyKeys;
144   }
145 
146   private Runnable batchStatJob(final Collection<SkyKey> dirtyKeys,
147                                        final List<Pair<SkyKey, ActionExecutionValue>> shard,
148                                        final BatchStat batchStatter) {
149     return new Runnable() {
150       @Override
151       public void run() {
152         Map<Artifact, Pair<SkyKey, ActionExecutionValue>> artifactToKeyAndValue = new HashMap<>();
153         for (Pair<SkyKey, ActionExecutionValue> keyAndValue : shard) {
154           ActionExecutionValue actionValue = keyAndValue.getSecond();
155           if (actionValue == null) {
156             dirtyKeys.add(keyAndValue.getFirst());
157           } else {
158             for (Artifact artifact : actionValue.getAllOutputArtifactData().keySet()) {
159               artifactToKeyAndValue.put(artifact, keyAndValue);
160             }
161           }
162         }
163 
164         List<Artifact> artifacts = ImmutableList.copyOf(artifactToKeyAndValue.keySet());
165         List<FileStatusWithDigest> stats;
166         try {
167           stats = batchStatter.batchStat(/*includeDigest=*/true, /*includeLinks=*/true,
168                                          Artifact.asPathFragments(artifacts));
169         } catch (IOException e) {
170           // Batch stat did not work. Log an exception and fall back on system calls.
171           LoggingUtil.logToRemote(Level.WARNING, "Unable to process batch stat", e);
172           outputStatJob(dirtyKeys, shard).run();
173           return;
174         } catch (InterruptedException e) {
175           // We handle interrupt in the main thread.
176           return;
177         }
178 
179         Preconditions.checkState(artifacts.size() == stats.size(),
180             "artifacts.size() == %s stats.size() == %s", artifacts.size(), stats.size());
181         for (int i = 0; i < artifacts.size(); i++) {
182           Artifact artifact = artifacts.get(i);
183           FileStatusWithDigest stat = stats.get(i);
184           Pair<SkyKey, ActionExecutionValue> keyAndValue = artifactToKeyAndValue.get(artifact);
185           ActionExecutionValue actionValue = keyAndValue.getSecond();
186           SkyKey key = keyAndValue.getFirst();
187           FileValue lastKnownData = actionValue.getAllOutputArtifactData().get(artifact);
188           try {
189             FileValue newData = ActionMetadataHandler.fileValueFromArtifact(artifact, stat, tsgm);
190             if (!newData.equals(lastKnownData)) {
191               updateIntraBuildModifiedCounter(stat != null ? stat.getLastChangeTime() : -1,
192                   lastKnownData.isSymlink(), newData.isSymlink());
193               modifiedOutputFilesCounter.getAndIncrement();
194               dirtyKeys.add(key);
195             }
196           } catch (IOException e) {
197             // This is an unexpected failure getting a digest or symlink target.
198             modifiedOutputFilesCounter.getAndIncrement();
199             dirtyKeys.add(key);
200           }
201         }
202       }
203     };
204   }
205 
206   private void updateIntraBuildModifiedCounter(long time, boolean oldWasSymlink,
207       boolean newIsSymlink) {
208     if (lastExecutionTimeRange != null
209         && lastExecutionTimeRange.contains(time)
210         && !(oldWasSymlink && newIsSymlink)) {
211       modifiedOutputFilesIntraBuildCounter.incrementAndGet();
212     }
213   }

215   private Runnable outputStatJob(final Collection<SkyKey> dirtyKeys,
216                                  final List<Pair<SkyKey, ActionExecutionValue>> shard) {
217     return new Runnable() {
218       @Override
219       public void run() {
220         for (Pair<SkyKey, ActionExecutionValue> keyAndValue : shard) {
221           ActionExecutionValue value = keyAndValue.getSecond();
222           if (value == null || actionValueIsDirtyWithDirectSystemCalls(value)) {
223             dirtyKeys.add(keyAndValue.getFirst());
224           }
225         }
226       }
227     };
228   }

230   /**
231    * Returns the number of modified output files inside of dirty actions.
232    */
233   int getNumberOfModifiedOutputFiles() {
234     return modifiedOutputFilesCounter.get();
235   }
236 
237   /** Returns the number of modified output files that occur during the previous build. */
238   int getNumberOfModifiedOutputFilesDuringPreviousBuild() {
239     return modifiedOutputFilesIntraBuildCounter.get();
240   }

242   private boolean actionValueIsDirtyWithDirectSystemCalls(ActionExecutionValue actionValue) {
243     boolean isDirty = false;
244     for (Map.Entry<Artifact, FileValue> entry :
245         actionValue.getAllOutputArtifactData().entrySet()) {
246       Artifact artifact = entry.getKey();
247       FileValue lastKnownData = entry.getValue();
248       try {
249         FileValue fileValue = ActionMetadataHandler.fileValueFromArtifact(artifact, null, tsgm);
250         if (!fileValue.equals(lastKnownData)) {
251           updateIntraBuildModifiedCounter(fileValue.exists()
252               ? fileValue.realRootedPath().asPath().getLastModifiedTime()
253               : -1, lastKnownData.isSymlink(), fileValue.isSymlink());
254           modifiedOutputFilesCounter.getAndIncrement();
255           isDirty = true;
256         }
257       } catch (IOException e) {
258         // This is an unexpected failure getting a digest or symlink target.
259         modifiedOutputFilesCounter.getAndIncrement();
260         isDirty = true;
261       }
262     }
263     return isDirty;
264   }

266   private BatchDirtyResult getDirtyValues(
267       Iterable<SkyKey> values, final SkyValueDirtinessChecker checker) throws InterruptedException {
268     ExecutorService executor =
269         Executors.newFixedThreadPool(
270             DIRTINESS_CHECK_THREADS,
271             new ThreadFactoryBuilder().setNameFormat("FileSystem Value Invalidator %d").build());
272 
273     final BatchDirtyResult batchResult = new BatchDirtyResult();
274     ThrowableRecordingRunnableWrapper wrapper =
275         new ThrowableRecordingRunnableWrapper("FilesystemValueChecker#getDirtyValues");
276     for (final SkyKey key : values) {
277       final SkyValue value = valuesSupplier.get().get(key);
278       executor.execute(
279           wrapper.wrap(
280               new Runnable() {
281                 @Override
282                 public void run() {
283                   if (value != null) {
284                     DirtyResult result = checker.maybeCheck(key, value, tsgm);
285                     if (result != null && result.isDirty()) {
286                       batchResult.add(key, result.getNewValue());
287                     }
288                   }
289                 }
290               }));
291     }
292 
293     boolean interrupted = ExecutorUtil.interruptibleShutdown(executor);
294     Throwables.propagateIfPossible(wrapper.getFirstThrownError());
295     if (interrupted) {
296       throw new InterruptedException();
297     }
298     return batchResult;
299   }

301   /**
302    * Result of a batch call to {@link SkyValueDirtinessChecker#maybeCheck}. Partitions the dirty
303    * values based on whether we have a new value available for them or not.
304    */
305   private static class BatchDirtyResult implements Differencer.Diff {
306 
307     private final Set<SkyKey> concurrentDirtyKeysWithoutNewValues =
308         Collections.newSetFromMap(new ConcurrentHashMap<SkyKey, Boolean>());
309     private final ConcurrentHashMap<SkyKey, SkyValue> concurrentDirtyKeysWithNewValues =
310         new ConcurrentHashMap<>();
311 
312     private void add(SkyKey key, @Nullable SkyValue newValue) {
313       if (newValue == null) {
314         concurrentDirtyKeysWithoutNewValues.add(key);
315       } else {
316         concurrentDirtyKeysWithNewValues.put(key, newValue);
317       }
318     }
319 
320     @Override
321     public Collection<SkyKey> changedKeysWithoutNewValues() {
322       return concurrentDirtyKeysWithoutNewValues;
323     }
324 
325     @Override
326     public Map<SkyKey, ? extends SkyValue> changedKeysWithNewValues() {
327       return concurrentDirtyKeysWithNewValues;
328     }
329   }
