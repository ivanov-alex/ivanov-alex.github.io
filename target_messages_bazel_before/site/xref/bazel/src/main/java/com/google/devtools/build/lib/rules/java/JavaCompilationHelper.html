44  /**
45   * A helper class for compiling Java targets. It contains method to create the
46   * various intermediate Artifacts for using ijars and source ijars.
47   * <p>
48   * Also supports the creation of resource and source only Jars.
49   */
50  public class JavaCompilationHelper extends BaseJavaCompilationHelper {

66    public JavaCompilationHelper(RuleContext ruleContext, JavaSemantics semantics,
67        ImmutableList<String> javacOpts, JavaTargetAttributes.Builder attributes) {
68      super(ruleContext);
69      this.attributes = attributes;
70      this.customJavacOpts = javacOpts;
71      this.customJavacJvmOpts =
72          ImmutableList.copyOf(JavaToolchainProvider.getDefaultJavacJvmOptions(ruleContext));
73      this.semantics = semantics;
74    }

76    public JavaCompilationHelper(RuleContext ruleContext, JavaSemantics semantics,
77        JavaTargetAttributes.Builder attributes) {
78      this(ruleContext, semantics, getDefaultJavacOptsFromRule(ruleContext), attributes);

88    /**
89     * Creates the Action that compiles Java source files.
90     *
91     * @param outputJar the class jar Artifact to create with the Action
92     * @param manifestProtoOutput the output artifact for the manifest proto emitted from JavaBuilder
93     * @param gensrcOutputJar the generated sources jar Artifact to create with the Action
94     *        (null if no sources will be generated).
95     * @param outputDepsProto the compiler-generated jdeps file to create with the Action
96     *        (null if not requested)
97     * @param outputMetadata metadata file (null if no instrumentation is needed).
98     */
99    public void createCompileAction(
100       Artifact outputJar,
101       Artifact manifestProtoOutput,
102       @Nullable Artifact gensrcOutputJar,
103       @Nullable Artifact outputDepsProto,
104       @Nullable Artifact outputMetadata) {
105 
106     JavaTargetAttributes attributes = getAttributes();
107     JavaCompileAction.Builder builder = createJavaCompileActionBuilder(semantics);
108     builder.setClasspathEntries(attributes.getCompileTimeClassPath());
109     builder.addResources(attributes.getResources());
110     builder.addClasspathResources(attributes.getClassPathResources());
111     // Only add default bootclasspath entries if not explicitly set in attributes.
112     if (!attributes.getBootClassPath().isEmpty()) {
113       builder.setBootclasspathEntries(attributes.getBootClassPath());
114     } else {
115       builder.setBootclasspathEntries(getBootClasspath());
116     }
117     builder.setExtdirInputs(getExtdirInputs());
118     builder.setLangtoolsJar(getLangtoolsJar());
119     builder.setJavaBuilderJar(getJavaBuilderJar());
120     builder.addTranslations(getTranslations());
121     builder.setOutputJar(outputJar);
122     builder.setManifestProtoOutput(manifestProtoOutput);
123     builder.setGensrcOutputJar(gensrcOutputJar);
124     builder.setOutputDepsProto(outputDepsProto);
125     builder.setMetadata(outputMetadata);
126     builder.setInstrumentationJars(getInstrumentationJars(semantics));
127     builder.addSourceFiles(attributes.getSourceFiles());
128     builder.addSourceJars(attributes.getSourceJars());
129     builder.setJavacOpts(customJavacOpts);
130     builder.setJavacJvmOpts(customJavacJvmOpts);
131     builder.setCompressJar(true);
132     builder.setSourceGenDirectory(sourceGenDir(outputJar));
133     builder.setTempDirectory(tempDir(outputJar));
134     builder.addProcessorPaths(attributes.getProcessorPath());
135     builder.addProcessorNames(attributes.getProcessorNames());
136     builder.setStrictJavaDeps(attributes.getStrictJavaDeps());
137     builder.addDirectJars(attributes.getDirectJars());
138     builder.addCompileTimeDependencyArtifacts(attributes.getCompileTimeDependencyArtifacts());
139     builder.setRuleKind(attributes.getRuleKind());
140     builder.setTargetLabel(attributes.getTargetLabel());
141     getAnalysisEnvironment().registerAction(builder.build());
142   }

144   /**
145    * Creates the Action that compiles Java source files and optionally instruments them for
146    * coverage.
147    *
148    * @param outputJar the class jar Artifact to create with the Action
149    * @param manifestProtoOutput the output artifact for the manifest proto emitted from JavaBuilder
150    * @param gensrcJar the generated sources jar Artifact to create with the Action
151    * @param outputDepsProto the compiler-generated jdeps file to create with the Action
152    * @param javaArtifactsBuilder the build to store the instrumentation metadata in
153    */
154   public void createCompileActionWithInstrumentation(
155       Artifact outputJar,
156       Artifact manifestProtoOutput,
157       @Nullable Artifact gensrcJar,
158       @Nullable Artifact outputDepsProto,
159       JavaCompilationArtifacts.Builder javaArtifactsBuilder) {
160     createCompileAction(
161         outputJar,
162         manifestProtoOutput,
163         gensrcJar,
164         outputDepsProto,
165         createInstrumentationMetadata(outputJar, javaArtifactsBuilder));
166   }

168   /**
169    * Creates the instrumentation metadata artifact if needed.
170    *
171    * @return the instrumentation metadata artifact or null if instrumentation is
172    *         disabled
173    */
174   public Artifact createInstrumentationMetadata(Artifact outputJar,
175       JavaCompilationArtifacts.Builder javaArtifactsBuilder) {
176     // If we need to instrument the jar, add additional output (the coverage metadata file) to the
177     // JavaCompileAction.
178     Artifact instrumentationMetadata = null;
179     if (shouldInstrumentJar()) {
180       instrumentationMetadata = semantics.createInstrumentationMetadataArtifact(
181           getAnalysisEnvironment(), outputJar);
182 
183       if (instrumentationMetadata != null) {
184         javaArtifactsBuilder.addInstrumentationMetadata(instrumentationMetadata);
185       }
186     }
187     return instrumentationMetadata;
188   }

190   private boolean shouldInstrumentJar() {
191     // TODO(bazel-team): What about source jars?
192     return getConfiguration().isCodeCoverageEnabled() && attributes.hasSourceFiles() &&
193         getConfiguration().getInstrumentationFilter().isIncluded(
194             getRuleContext().getLabel().toString());
195   }

197   /**
198    * Returns the artifact for a jar file containing source files that were generated by
199    * annotation processors or null if no annotation processors are used.
200    */
201   public Artifact createGensrcJar(@Nullable Artifact outputJar) {
202     if (usesAnnotationProcessing()) {
203       return getRuleContext().getDerivedArtifact(
204           FileSystemUtils.appendWithoutExtension(outputJar.getRootRelativePath(), "-gensrc"),
205           outputJar.getRoot());
206     } else {
207       return null;
208     }
209   }

211   /**
212    * Returns whether this target uses annotation processing.
213    */
214   private boolean usesAnnotationProcessing() {
215     JavaTargetAttributes attributes = getAttributes();
216     return getJavacOpts().contains("-processor") || !attributes.getProcessorNames().isEmpty();
217   }

232   /**
233    * Creates the action for creating the gen jar.
234    *
235    * @param classJar The artifact for the class jar emitted from JavaBuilder
236    * @param manifestProto The artifact for the manifest proto emitted from JavaBuilder
237    * @param genClassJar The artifact for the gen jar to output
238    */
239   public void createGenJarAction(Artifact classJar, Artifact manifestProto,
240       Artifact genClassJar) {
241     getRuleContext().registerAction(new SpawnAction.Builder()
242       .addInput(manifestProto)
243       .addInput(classJar)
244       .addOutput(genClassJar)
245       .addTransitiveInputs(JavaCompilationHelper.getHostJavabaseInputs(getRuleContext()))
246       .setJarExecutable(
247           getRuleContext().getHostConfiguration().getFragment(Jvm.class).getJavaExecutable(),
248           getRuleContext().getPrerequisiteArtifact("$genclass", Mode.HOST),
249           ImmutableList.of("-client", GENCLASS_MAX_MEMORY))
250       .setCommandLine(CustomCommandLine.builder()
251           .addExecPath("--manifest_proto", manifestProto)
252           .addExecPath("--class_jar", classJar)
253           .addExecPath("--output_jar", genClassJar)
254           .add("--temp_dir").addPath(tempDir(genClassJar))
255           .build())
256       .useParameterFile(ParameterFileType.SHELL_QUOTED)
257       .setProgressMessage("Building genclass jar " + genClassJar.prettyPrint())
258       .setMnemonic("JavaSourceJar")
259       .build(getRuleContext()));
260   }

266   /**
267    * Creates the jdeps file artifact if needed. Returns null if the target can't emit dependency
268    * information (i.e there is no compilation step, the target acts as an alias).
269    *
270    * @param outputJar output jar artifact used to derive the name
271    * @return the jdeps file artifact or null if the target can't generate such a file
272    */
273   public Artifact createOutputDepsProtoArtifact(Artifact outputJar,
274       JavaCompilationArtifacts.Builder builder) {
275     if (!generatesOutputDeps()) {
276       return null;
277     }
278 
279     outputDepsProtoArtifact = getRuleContext().getDerivedArtifact(
280           FileSystemUtils.replaceExtension(outputJar.getRootRelativePath(), ".jdeps"),
281           outputJar.getRoot());
282 
283     builder.setRunTimeDependencies(outputDepsProtoArtifact);
284     return outputDepsProtoArtifact;
285   }

287   /**
288    * Returns whether this target emits dependency information. Compilation must occur, so certain
289    * targets acting as aliases have to be filtered out.
290    */
291   private boolean generatesOutputDeps() {
292     return getJavaConfiguration().getGenerateJavaDeps() &&
293         (attributes.hasSourceFiles() || attributes.hasSourceJars());
294   }

296   /**
297    * Creates an Action that packages all of the resources into a Jar. This
298    * includes the declared resources, the classpath resources and the translated
299    * messages.
300    *
301    * <p>The resource jar artifact is derived from the given original jar, by
302    * prepending the given prefix and appending the given suffix. The new jar
303    * uses the same root as the original jar.
304    */
305   // TODO(bazel-team): Extract this method to make it easier to create simple
306   // zip/jar archives without having to first create a JavaCompilationhelper and
307   // JavaTargetAttributes.
308   public Artifact createResourceJarAction(Artifact resourceJar) {
309     JavaTargetAttributes attributes = getAttributes();
310     JavaCompileAction.Builder builder = createJavaCompileActionBuilder(semantics);
311     builder.setOutputJar(resourceJar);
312     builder.addResources(attributes.getResources());
313     builder.addClasspathResources(attributes.getClassPathResources());
314     builder.setExtdirInputs(getExtdirInputs());
315     builder.setLangtoolsJar(getLangtoolsJar());
316     builder.addTranslations(getTranslations());
317     builder.setCompressJar(true);
318     builder.setTempDirectory(tempDir(resourceJar));
319     builder.setJavaBuilderJar(getJavaBuilderJar());
320     builder.setJavacOpts(getDefaultJavacOptsFromRule(getRuleContext()));
321     builder.setJavacJvmOpts(
322         ImmutableList.copyOf(JavaToolchainProvider.getDefaultJavacOptions(getRuleContext())));
323     builder.setJavacJvmOpts(
324         ImmutableList.copyOf(JavaToolchainProvider.getDefaultJavacJvmOptions(getRuleContext())));
325     getAnalysisEnvironment().registerAction(builder.build());
326     return resourceJar;
327   }

329   /**
330    * Creates an Action that packages the Java source files into a Jar.  If {@code gensrcJar} is
331    * non-null, includes the contents of the {@code gensrcJar} with the output source jar.
332    *
333    * @param outputJar the Artifact to create with the Action
334    * @param gensrcJar the generated sources jar Artifact that should be included with the
335    *        sources in the output Artifact.  May be null.
336    */
337   public void createSourceJarAction(Artifact outputJar, @Nullable Artifact gensrcJar) {
338     JavaTargetAttributes attributes = getAttributes();
339     Collection<Artifact> resourceJars = new ArrayList<>(attributes.getSourceJars());
340     if (gensrcJar != null) {
341       resourceJars.add(gensrcJar);
342     }
343     createSourceJarAction(semantics, attributes.getSourceFiles(), resourceJars, outputJar);
344   }

346   /**
347    * Creates the actions that produce the interface jar. Adds the jar artifacts to the given
348    * JavaCompilationArtifacts builder.
349    */
350   public void createCompileTimeJarAction(Artifact runtimeJar,
351       @Nullable Artifact runtimeDeps, JavaCompilationArtifacts.Builder builder) {
352     Artifact jar = getJavaConfiguration().getUseIjars()
353         ? createIjarAction(runtimeJar, false)
354         : runtimeJar;
355     Artifact deps = runtimeDeps;
356 
357     builder.addCompileTimeJar(jar);
358     builder.setCompileTimeDependencies(deps);
359   }

361   /**
362    * Creates actions that create ijars from generated jars that are an input to
363    * the Java target.
364    *
365    * @return the generated ijars or original jars that are not generated by a
366    *         genrule
367    */
368   public Iterable<Artifact> filterGeneratedJarsThroughIjar(Iterable<Artifact> jars) {
369     if (!getJavaConfiguration().getUseIjars()) {
370       return jars;
371     }
372     // We need to copy this list in order to avoid generating a new action each time the iterator
373     // is enumerated
374     return ImmutableList.copyOf(Iterables.transform(jars, new Function<Artifact, Artifact>() {
375       @Override
376       public Artifact apply(Artifact jar) {
377         return !jar.isSourceArtifact() ? createIjarAction(jar, true) : jar;
378       }
379     }));
380   }

382   private void addArgsAndJarsToAttributes(JavaCompilationArgs args, Iterable<Artifact> directJars) {
383     // Can only be non-null when isStrict() returns true.
384     if (directJars != null) {
385       attributes.addDirectCompileTimeClassPathEntries(directJars);
386       attributes.addDirectJars(directJars);
387     }
388 
389     attributes.merge(args);
390   }

392   private void addLibrariesToAttributesInternal(Iterable<? extends TransitiveInfoCollection> deps) {
393     JavaCompilationArgs args = JavaCompilationArgs.builder()
394         .addTransitiveTargets(deps).build();
395 
396     NestedSet<Artifact> directJars = isStrict()
397         ? getNonRecursiveCompileTimeJarsFromCollection(deps)
398         : null;
399     addArgsAndJarsToAttributes(args, directJars);
400   }
401 
402   private void addProvidersToAttributesInternal(
403       Iterable<? extends SourcesJavaCompilationArgsProvider> deps, boolean isNeverLink) {
404     JavaCompilationArgs args = JavaCompilationArgs.builder()
405         .addSourcesTransitiveCompilationArgs(deps, true,
406             isNeverLink ? ClasspathType.COMPILE_ONLY : ClasspathType.BOTH)
407         .build();
408 
409     NestedSet<Artifact> directJars = isStrict()
410         ? getNonRecursiveCompileTimeJarsFromProvider(deps, isNeverLink)
411         : null;
412     addArgsAndJarsToAttributes(args, directJars);
413   }
414 
415   private boolean isStrict() {
416     return getStrictJavaDeps() != OFF;
417   }

419   private NestedSet<Artifact> getNonRecursiveCompileTimeJarsFromCollection(
420       Iterable<? extends TransitiveInfoCollection> deps) {
421     JavaCompilationArgs.Builder builder = JavaCompilationArgs.builder();
422     builder.addTransitiveTargets(deps, /*recursive=*/false);
423     return builder.build().getCompileTimeJars();
424   }
425 
426   private NestedSet<Artifact> getNonRecursiveCompileTimeJarsFromProvider(
427       Iterable<? extends SourcesJavaCompilationArgsProvider> deps, boolean isNeverLink) {
428     return JavaCompilationArgs.builder()
429         .addSourcesTransitiveCompilationArgs(deps, false,
430             isNeverLink ? ClasspathType.COMPILE_ONLY : ClasspathType.BOTH)
431         .build().getCompileTimeJars();
432   }

434   private void addDependencyArtifactsToAttributes(
435       Iterable<? extends TransitiveInfoCollection> deps) {
436     NestedSetBuilder<Artifact> compileTimeBuilder = NestedSetBuilder.stableOrder();
437     NestedSetBuilder<Artifact> runTimeBuilder = NestedSetBuilder.stableOrder();
438     for (JavaCompilationArgsProvider provider : AnalysisUtils.getProviders(
439         deps, JavaCompilationArgsProvider.class)) {
440       compileTimeBuilder.addTransitive(provider.getCompileTimeJavaDependencyArtifacts());
441       runTimeBuilder.addTransitive(provider.getRunTimeJavaDependencyArtifacts());
442     }
443     attributes.addCompileTimeDependencyArtifacts(compileTimeBuilder.build());
444     attributes.addRuntimeDependencyArtifacts(runTimeBuilder.build());
445   }

447   /**
448    * Adds the compile time and runtime Java libraries in the transitive closure
449    * of the deps to the attributes.
450    *
451    * @param deps the dependencies to be included as roots of the transitive
452    *        closure
453    */
454   public void addLibrariesToAttributes(Iterable<? extends TransitiveInfoCollection> deps) {
455     // Enforcing strict Java dependencies: when the --strict_java_deps flag is
456     // WARN or ERROR, or is DEFAULT and strict_java_deps attribute is unset,
457     // we use a stricter javac compiler to perform direct deps checks.
458     attributes.setStrictJavaDeps(getStrictJavaDeps());
459     addLibrariesToAttributesInternal(deps);
460 
461     JavaClasspathMode classpathMode = getJavaConfiguration().getReduceJavaClasspath();
462     if (isStrict() && classpathMode != JavaClasspathMode.OFF) {
463       addDependencyArtifactsToAttributes(deps);
464     }
465   }
466 
467   public void addProvidersToAttributes(Iterable<? extends SourcesJavaCompilationArgsProvider> deps,
468       boolean isNeverLink) {
469     // see addLibrariesToAttributes() for explanation
470     attributes.setStrictJavaDeps(getStrictJavaDeps());
471     addProvidersToAttributesInternal(deps, isNeverLink);
472   }

507   public void addTranslations(Collection<Artifact> translations) {
