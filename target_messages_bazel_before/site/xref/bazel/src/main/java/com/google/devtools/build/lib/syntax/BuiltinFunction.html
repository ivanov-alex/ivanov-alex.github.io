120   @Override
121   @Nullable
122   public Object call(Object[] args,
123       @Nullable FuncallExpression ast, @Nullable Environment env)
124       throws EvalException, InterruptedException {
125     final Location loc = (ast == null) ? location : ast.getLocation();
126 
127     // Add extra arguments, if needed
128     if (extraArgs != null) {
129       int i = args.length - extraArgs.length;
130       for (BuiltinFunction.ExtraArgKind extraArg : extraArgs) {
131         switch(extraArg) {
132           case LOCATION:
133             args[i] = loc;
134             break;
135 
136           case SYNTAX_TREE:
137             args[i] = ast;
138             break;
139 
140           case ENVIRONMENT:
141             args[i] = env;
142             break;
143         }
144         i++;
145       }
146     }
147 
148     // Last but not least, actually make an inner call to the function with the resolved arguments.
149     try {
150       return invokeMethod.invoke(this, args);
151     } catch (InvocationTargetException x) {
152       Throwable e = x.getCause();
153       if (e instanceof EvalException) {
154         throw ((EvalException) e).ensureLocation(loc);
155       } else if (e instanceof InterruptedException) {
156         throw (InterruptedException) e;
157       } else if (e instanceof ClassCastException
158           || e instanceof ExecutionException
159           || e instanceof IllegalStateException) {
160         throw new EvalException(loc, "in call to " + getName(), e);
161       } else if (e instanceof IllegalArgumentException) {
162         throw new EvalException(loc, "Illegal argument in call to " + getName(), e);
163       } else {
164         throw badCallException(loc, e, args);
165       }
166     } catch (IllegalArgumentException e) {
167         // Either this was thrown by Java itself, or it's a bug
168         // To cover the first case, let's manually check the arguments.
169         final int len = args.length - ((extraArgs == null) ? 0 : extraArgs.length);
170         final Class<?>[] types = invokeMethod.getParameterTypes();
171         for (int i = 0; i < args.length; i++) {
172           if (args[i] != null && !types[i].isAssignableFrom(args[i].getClass())) {
173             String paramName = i < len
174                 ? signature.getSignature().getNames().get(i) : extraArgs[i - len].name();
175             throw new EvalException(loc, String.format(
176                 "expected %s for '%s' while calling %s but got %s instead",
177                 EvalUtils.getDataTypeNameFromClass(types[i]), paramName, getName(),
178                 EvalUtils.getDataTypeName(args[i])));
179           }
180         }
181         throw badCallException(loc, e, args);
182     } catch (IllegalAccessException e) {
183       throw badCallException(loc, e, args);
184     }
185   }
186 
187   private IllegalStateException badCallException(Location loc, Throwable e, Object... args) {
188     // If this happens, it's a bug in our code.
189     return new IllegalStateException(String.format("%s%s (%s)\n"
190             + "while calling %s with args %s\nJava parameter types: %s\nSkylark type checks: %s",
191             (loc == null) ? "" : loc + ": ",
192             e.getClass().getName(), e.getMessage(), this,
193             Arrays.asList(args),
194             Arrays.asList(invokeMethod.getParameterTypes()),
195             signature.getTypes()), e);
196   }

199   /** Configure the reflection mechanism */
200   @Override
201   public void configure(SkylarkSignature annotation) {
202     Preconditions.checkState(!isConfigured()); // must not be configured yet
203     enforcedArgumentTypes = new ArrayList<>();
204     this.extraArgs = SkylarkSignatureProcessor.getExtraArgs(annotation);
205     this.returnType = annotation.returnType();
206     super.configure(annotation);
207   }
208 
209   // finds the method and makes it accessible (which is needed to find it, and later to use it)
210   protected Method findMethod(final String name) {
211     Method found = null;
212     for (Method method : this.getClass().getDeclaredMethods()) {
213       method.setAccessible(true);
214       if (name.equals(method.getName())) {
215         if (found != null) {
216           throw new IllegalArgumentException(String.format(
217               "function %s has more than one method named %s", getName(), name));
218         }
219         found = method;
220       }
221     }
222     if (found == null) {
223       throw new NoSuchElementException(String.format(
224           "function %s doesn't have a method named %s", getName(), name));
225     }
226     return found;
227   }

229   /** Configure the reflection mechanism */
230   @Override
231   protected void configure() {
232     invokeMethod = findMethod("invoke");
233 
234     int arguments = signature.getSignature().getShape().getArguments();
235     innerArgumentCount = arguments + (extraArgs == null ? 0 : extraArgs.length);
236     Class<?>[] parameterTypes = invokeMethod.getParameterTypes();
237     Preconditions.checkArgument(innerArgumentCount == parameterTypes.length,
238         "bad argument count for %s: method has %s arguments, type list has %s",
239         getName(), innerArgumentCount, parameterTypes.length);
240 
241     if (enforcedArgumentTypes != null) {
242       for (int i = 0; i < arguments; i++) {
243         SkylarkType enforcedType = enforcedArgumentTypes.get(i);
244         if (enforcedType != null) {
245           Class<?> parameterType = parameterTypes[i];
246           String msg = String.format(
247               "fun %s(%s), param %s, enforcedType: %s (%s); parameterType: %s",
248               getName(), signature, signature.getSignature().getNames().get(i),
249               enforcedType, enforcedType.getType(), parameterType);
250           if (enforcedType instanceof SkylarkType.Simple
251               || enforcedType instanceof SkylarkFunctionType) {
252             Preconditions.checkArgument(
253                 enforcedType.getType() == parameterType, msg);
254             // No need to enforce Simple types on the Skylark side, the JVM will do it for us.
255             enforcedArgumentTypes.set(i, null);
256           } else if (enforcedType instanceof SkylarkType.Combination) {
257             Preconditions.checkArgument(enforcedType.getType() == parameterType, msg);
258           } else {
259             Preconditions.checkArgument(
260                 parameterType == Object.class || parameterType == null, msg);
261           }
262         }
263       }
264     }
265     // No need for the enforcedArgumentTypes List if all the types were Simple
266     enforcedArgumentTypes = FunctionSignature.<SkylarkType>valueListOrNull(enforcedArgumentTypes);
267 
268     if (returnType != null) {
269       Class<?> type = returnType;
270       if (type == HackHackEitherList.class) {
271         type = Object.class;
272       }
273       Class<?> methodReturnType = invokeMethod.getReturnType();
274       Preconditions.checkArgument(type == methodReturnType,
275           "signature for function %s says it returns %s but its invoke method returns %s",
276           getName(), returnType, methodReturnType);
277     }
278   }

280   /** Configure by copying another function's configuration */
281   // Alternatively, we could have an extension BuiltinFunctionSignature of FunctionSignature,
282   // and use *that* instead of a Factory.
283   public void configure(BuiltinFunction.Factory factory) {
284     // this function must not be configured yet, but the factory must be
285     Preconditions.checkState(!isConfigured());
286     Preconditions.checkState(factory.isConfigured(),
287         "function factory is not configured for %s", getName());
288 
289     this.paramDoc = factory.getParamDoc();
290     this.signature = factory.getSignature();
291     this.extraArgs = factory.getExtraArgs();
292     this.objectType = factory.getObjectType();
293     this.onlyLoadingPhase = factory.isOnlyLoadingPhase();
294     configure();
295   }

297   /**
298    * A Factory allows for a @SkylarkSignature annotation to be provided and processed in advance
299    * for a function that will be defined later as a closure (see e.g. in PackageFactory).
300    *
301    * <p>Each instance of this class must define a method create that closes over some (final)
302    * variables and returns a BuiltinFunction.
303    */
304   public abstract static class Factory extends BuiltinFunction {
305     @Nullable private Method createMethod;
306 
307     /** Create unconfigured function Factory from its name */
308     public Factory(String name) {
309       super(name);
310     }
311 
312     /** Creates an unconfigured function Factory with the given name and defaultValues */
313     public Factory(String name, Iterable<Object> defaultValues) {
314       super(name, defaultValues);
315     }
316 
317     @Override
318     public void configure() {
319       if (createMethod != null) {
320         return;
321       }
322       createMethod = findMethod("create");
323     }
324 
325     @Override
326     public Object call(Object[] args, @Nullable FuncallExpression ast, @Nullable Environment env)
327       throws EvalException {
328       throw new EvalException(null, "Tried to invoke a Factory for function " + this);
329     }
330 
331     /** Instantiate the Factory
332      * @param args arguments to pass to the create method
333      * @return a new BuiltinFunction that closes over the arguments
334      */
335     public BuiltinFunction apply(Object... args) {
336       try {
337         return (BuiltinFunction) createMethod.invoke(this, args);
338       } catch (InvocationTargetException | IllegalArgumentException | IllegalAccessException e) {
339         throw new RuntimeException(String.format(
340             "Exception while applying BuiltinFunction.Factory %s: %s",
341             this, e.getMessage()), e);
342       }
343     }
344   }
