121   private void setDeserializedArtifactOwners() throws ViewCreationFailedException {
122     Map<PathFragment, Artifact> deserializedArtifactMap =
123         artifactFactory.getDeserializedArtifacts();
124     Set<Artifact> deserializedArtifacts = new HashSet<>();
125     for (Artifact artifact : deserializedArtifactMap.values()) {
126       if (!artifact.getExecPath().getBaseName().endsWith(".gcda")) {
127         // gcda files are classified as generated artifacts, but are not actually generated. All
128         // others need owners.
129         deserializedArtifacts.add(artifact);
130       }
131     }
132     if (deserializedArtifacts.isEmpty()) {
133       // If there are no deserialized artifacts to process, don't pay the price of iterating over
134       // the graph.
135       return;
136     }
137     for (Map.Entry<SkyKey, ActionLookupValue> entry :
138       skyframeExecutor.getActionLookupValueMap().entrySet()) {
139       for (Action action : entry.getValue().getActionsForFindingArtifactOwners()) {
140         for (Artifact output : action.getOutputs()) {
141           Artifact deserializedArtifact = deserializedArtifactMap.get(output.getExecPath());
142           if (deserializedArtifact != null) {
143             deserializedArtifact.setArtifactOwner((ActionLookupKey) entry.getKey().argument());
144             deserializedArtifacts.remove(deserializedArtifact);
145           }
146         }
147       }
148     }
149     if (!deserializedArtifacts.isEmpty()) {
150       throw new ViewCreationFailedException("These artifacts were read in from the FDO profile but"
151       + " have no generating action that could be found. If you are confident that your profile was"
152       + " collected from the same source state at which you're building, please report this:\n"
153       + Artifact.asExecPaths(deserializedArtifacts));
154     }
155     artifactFactory.clearDeserializedArtifacts();
156   }

158   /**
159    * Analyzes the specified targets using Skyframe as the driving framework.
160    *
161    * @return the configured targets that should be built along with a WalkableGraph of the analysis.
162    */
163   public Pair<Collection<ConfiguredTarget>, WalkableGraph> configureTargets(
164       List<ConfiguredTargetKey> values, EventBus eventBus, boolean keepGoing)
165           throws InterruptedException, ViewCreationFailedException {
166     enableAnalysis(true);
167     EvaluationResult<ConfiguredTargetValue> result;
168     try {
169       result = skyframeExecutor.configureTargets(values, keepGoing);
170     } finally {
171       enableAnalysis(false);
172     }
173     ImmutableMap<Action, ConflictException> badActions = skyframeExecutor.findArtifactConflicts();
174 
175     // Filter out all CTs that have a bad action and convert to a list of configured targets. This
176     // code ensures that the resulting list of configured targets has the same order as the incoming
177     // list of values, i.e., that the order is deterministic.
178     Collection<ConfiguredTarget> goodCts = Lists.newArrayListWithCapacity(values.size());
179     for (ConfiguredTargetKey value : values) {
180       ConfiguredTargetValue ctValue = result.get(ConfiguredTargetValue.key(value));
181       if (ctValue == null) {
182         continue;
183       }
184       goodCts.add(ctValue.getConfiguredTarget());
185     }
186 
187     if (!result.hasError() && badActions.isEmpty()) {
188       setDeserializedArtifactOwners();
189       return Pair.of(goodCts, result.getWalkableGraph());
190     }
191 
192     // --nokeep_going so we fail with an exception for the first error.
193     // TODO(bazel-team): We might want to report the other errors through the event bus but
194     // for keeping this code in parity with legacy we just report the first error for now.
195     if (!keepGoing) {
196       for (Map.Entry<Action, ConflictException> bad : badActions.entrySet()) {
197         ConflictException ex = bad.getValue();
198         try {
199           ex.rethrowTyped();
200         } catch (MutableActionGraph.ActionConflictException ace) {
201           ace.reportTo(skyframeExecutor.getReporter());
202           String errorMsg = "Analysis of target '" + bad.getKey().getOwner().getLabel()
203               + "' failed; build aborted";
204           throw new ViewCreationFailedException(errorMsg);
205         } catch (ArtifactPrefixConflictException apce) {
206           skyframeExecutor.getReporter().handle(Event.error(apce.getMessage()));
207         }
208         throw new ViewCreationFailedException(ex.getMessage());
209       }
210 
211       Map.Entry<SkyKey, ErrorInfo> error = result.errorMap().entrySet().iterator().next();
212       SkyKey topLevel = error.getKey();
213       ErrorInfo errorInfo = error.getValue();
214       assertSaneAnalysisError(errorInfo, topLevel);
215       skyframeExecutor.getCyclesReporter().reportCycles(errorInfo.getCycleInfo(), topLevel,
216           skyframeExecutor.getReporter());
217       Throwable cause = errorInfo.getException();
218       Preconditions.checkState(cause != null || !Iterables.isEmpty(errorInfo.getCycleInfo()),
219           errorInfo);
220       String errorMsg = "Analysis of target '" + ConfiguredTargetValue.extractLabel(topLevel)
221           + "' failed; build aborted";
222       if (cause instanceof ActionConflictException) {
223         ((ActionConflictException) cause).reportTo(skyframeExecutor.getReporter());
224       }
225       throw new ViewCreationFailedException(errorMsg);
226     }
227 
228     // --keep_going : We notify the error and return a ConfiguredTargetValue
229     for (Map.Entry<SkyKey, ErrorInfo> errorEntry : result.errorMap().entrySet()) {
230       if (values.contains(errorEntry.getKey().argument())) {
231         SkyKey errorKey = errorEntry.getKey();
232         ConfiguredTargetKey label = (ConfiguredTargetKey) errorKey.argument();
233         ErrorInfo errorInfo = errorEntry.getValue();
234         assertSaneAnalysisError(errorInfo, errorKey);
235 
236         skyframeExecutor.getCyclesReporter().reportCycles(errorInfo.getCycleInfo(), errorKey,
237             skyframeExecutor.getReporter());
238         // We try to get the root cause key first from ErrorInfo rootCauses. If we don't have one
239         // we try to use the cycle culprit if the error is a cycle. Otherwise we use the top-level
240         // error key.
241         Label root;
242         if (!Iterables.isEmpty(errorEntry.getValue().getRootCauses())) {
243           SkyKey culprit = Preconditions.checkNotNull(Iterables.getFirst(
244               errorEntry.getValue().getRootCauses(), null));
245           root = ((ConfiguredTargetKey) culprit.argument()).getLabel();
246         } else {
247           root = maybeGetConfiguredTargetCycleCulprit(errorInfo.getCycleInfo());
248         }
249         Exception cause = errorInfo.getException();
250         if (cause instanceof ActionConflictException) {
251           ((ActionConflictException) cause).reportTo(skyframeExecutor.getReporter());
252         }
253         skyframeExecutor.getReporter().handle(
254             Event.warn("errors encountered while analyzing target '"
255                 + label + "': it will not be built"));
256         eventBus.post(new AnalysisFailureEvent(
257             LabelAndConfiguration.of(label.getLabel(), label.getConfiguration()), root));
258       }
259     }
260 
261     Collection<Exception> reportedExceptions = Sets.newHashSet();
262     for (Map.Entry<Action, ConflictException> bad : badActions.entrySet()) {
263       ConflictException ex = bad.getValue();
264       try {
265         ex.rethrowTyped();
266       } catch (MutableActionGraph.ActionConflictException ace) {
267         ace.reportTo(skyframeExecutor.getReporter());
268         skyframeExecutor.getReporter()
269             .handle(Event.warn("errors encountered while analyzing target '"
270                 + bad.getKey().getOwner().getLabel() + "': it will not be built"));
271       } catch (ArtifactPrefixConflictException apce) {
272         if (reportedExceptions.add(apce)) {
273           skyframeExecutor.getReporter().handle(Event.error(apce.getMessage()));
274         }
275       }
276     }
277 
278     if (!badActions.isEmpty()) {
279       // In order to determine the set of configured targets transitively error free from action
280       // conflict issues, we run a post-processing update() that uses the bad action map.
281       EvaluationResult<PostConfiguredTargetValue> actionConflictResult =
282           skyframeExecutor.postConfigureTargets(values, keepGoing, badActions);
283 
284       goodCts = Lists.newArrayListWithCapacity(values.size());
285       for (ConfiguredTargetKey value : values) {
286         PostConfiguredTargetValue postCt =
287             actionConflictResult.get(PostConfiguredTargetValue.key(value));
288         if (postCt != null) {
289           goodCts.add(postCt.getCt());
290         }
291       }
292     }
293     setDeserializedArtifactOwners();
294     return Pair.of(goodCts, result.getWalkableGraph());
295   }

297   @Nullable
298   Label maybeGetConfiguredTargetCycleCulprit(Iterable<CycleInfo> cycleInfos) {
299     for (CycleInfo cycleInfo : cycleInfos) {
300       SkyKey culprit = Iterables.getFirst(cycleInfo.getCycle(), null);
301       if (culprit == null) {
302         continue;
303       }
304       if (culprit.functionName().equals(SkyFunctions.CONFIGURED_TARGET)) {
305         return ((LabelAndConfiguration) culprit.argument()).getLabel();
306       }
307     }
308     return null;
309   }

311   private static void assertSaneAnalysisError(ErrorInfo errorInfo, SkyKey key) {
312     Throwable cause = errorInfo.getException();
313     if (cause != null) {
314       // We should only be trying to configure targets when the loading phase succeeds, meaning
315       // that the only errors should be analysis errors.
316       Preconditions.checkState(cause instanceof ConfiguredValueCreationException
317           || cause instanceof ActionConflictException, "%s -> %s", key, errorInfo);
318     }
319   }

326    * Because we don't know what build-info artifacts this configured target may request, we
327    * conservatively register a dep on all of them.
328    */
329   // TODO(bazel-team): Allow analysis to return null so the value builder can exit and wait for a
330   // restart deps are not present.
331   private boolean getWorkspaceStatusValues(Environment env, BuildConfiguration config) {
332     env.getValue(WorkspaceStatusValue.SKY_KEY);
333     Map<BuildInfoKey, BuildInfoFactory> buildInfoFactories =
334         PrecomputedValue.BUILD_INFO_FACTORIES.get(env);
335     if (buildInfoFactories == null) {
336       return false;
337     }
338     // These factories may each create their own build info artifacts, all depending on the basic
339     // build-info.txt and build-changelist.txt.
340     List<SkyKey> depKeys = Lists.newArrayList();
341     for (BuildInfoKey key : buildInfoFactories.keySet()) {
342       if (buildInfoFactories.get(key).isEnabled(config)) {
343         depKeys.add(BuildInfoCollectionValue.key(new BuildInfoKeyAndConfig(key, config)));
344       }
345     }
346     env.getValues(depKeys);
347     return !env.valuesMissing();
348   }

350   /** Returns null if any build-info values are not ready. */
351   @Nullable
352   CachingAnalysisEnvironment createAnalysisEnvironment(ArtifactOwner owner,
353       boolean isSystemEnv, EventHandler eventHandler,
354       Environment env, BuildConfiguration config) {
355     if (config != null && !getWorkspaceStatusValues(env, config)) {
356       return null;
357     }
358     boolean extendedSanityChecks = config != null && config.extendedSanityChecks();
359     boolean allowRegisteringActions = config == null || config.isActionsEnabled();
360     return new CachingAnalysisEnvironment(
361         artifactFactory, owner, isSystemEnv, extendedSanityChecks, eventHandler, env,
362         allowRegisteringActions, binTools);
363   }

455   private class ConfiguredTargetValueInvalidationReceiver implements EvaluationProgressReceiver {
456     @Override
457     public void invalidated(SkyValue value, InvalidationState state) {
458       if (value instanceof ConfiguredTargetValue) {
459         ConfiguredTargetValue ctValue = (ConfiguredTargetValue) value;
460         // If the value was just dirtied and not deleted, then it may not be truly invalid, since
461         // it may later get re-validated.
462         if (state == InvalidationState.DELETED) {
463           anyConfiguredTargetDeleted = true;
464         } else {
465           dirtyConfiguredTargets.add(ctValue);
466         }
467       }
468     }
469 
470     @Override
471     public void enqueueing(SkyKey skyKey) {}
472 
473     @Override
474     public void evaluated(SkyKey skyKey, SkyValue value, EvaluationState state) {
475       if (skyKey.functionName() == SkyFunctions.CONFIGURED_TARGET && value != null) {
476         if (state == EvaluationState.BUILT) {
477           evaluatedConfiguredTargets.add(skyKey);
478           // During multithreaded operation, this is only set to true, so no concurrency issues.
479           someConfiguredTargetEvaluated = true;
480         }
481         Preconditions.checkNotNull(value, "%s %s", skyKey, state);
482         ConfiguredTargetValue ctValue = (ConfiguredTargetValue) value;
483         dirtyConfiguredTargets.remove(ctValue);
484       }
485     }
