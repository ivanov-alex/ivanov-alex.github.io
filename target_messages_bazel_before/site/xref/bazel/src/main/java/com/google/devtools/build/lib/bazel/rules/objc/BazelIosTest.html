31  /**
32   * Implementation for ios_test rule in Bazel.
33   */
34  public final class BazelIosTest extends IosTest {
35    static final String IOS_TEST_ON_BAZEL_ATTR = "$ios_test_on_bazel";
36  
37    @Override
38    public ConfiguredTarget create(RuleContext ruleContext, ObjcCommon common,
39        XcodeProvider xcodeProvider, NestedSet<Artifact> filesToBuild) throws InterruptedException {
40      Artifact testRunner = ruleContext.getPrerequisiteArtifact(IOS_TEST_ON_BAZEL_ATTR, Mode.TARGET);
41      Runfiles runfiles = new Runfiles.Builder()
42          .addArtifact(testRunner)
43          .build();
44      RunfilesSupport runfilesSupport =
45          RunfilesSupport.withExecutable(ruleContext, runfiles, testRunner);
46  
47      return new RuleConfiguredTargetBuilder(ruleContext)
48          .setFilesToBuild(NestedSetBuilder.<Artifact>stableOrder()
49              .addTransitive(filesToBuild)
50              .add(testRunner)
51              .build())
52          .add(XcodeProvider.class, xcodeProvider)
53          .add(RunfilesProvider.class, RunfilesProvider.simple(runfiles))
54          .setRunfilesSupport(runfilesSupport, testRunner)
55          .build();
56    }
57  }
