14  package com.google.devtools.build.lib.syntax;
15  
16  import com.google.common.base.Preconditions;
17  import com.google.common.collect.ImmutableList;
18  
19  import java.util.List;
20  
21  // TODO(bazel-team): maybe we should get rid of the ConditionalStatements and
22  // create a chain of if-else statements for elif-s.
23  /**
24   * Syntax node for an if/else statement.
25   */
26  public final class IfStatement extends Statement {
27  
28    /**
29     * Syntax node for an [el]if statement.
30     */
31    static final class ConditionalStatements extends Statement {
32  
33      private final Expression condition;
34      private final ImmutableList<Statement> stmts;
35  
36      public ConditionalStatements(Expression condition, List<Statement> stmts) {
37        this.condition = Preconditions.checkNotNull(condition);
38        this.stmts = ImmutableList.copyOf(stmts);
39      }
40  
41      @Override
42      void exec(Environment env) throws EvalException, InterruptedException {
43        for (Statement stmt : stmts) {
44          stmt.exec(env);
45        }
46      }
47  
48      @Override
49      public String toString() {
50        // TODO(bazel-team): see TODO in the outer class
51        return "[el]if " + condition + ": " + stmts + "\n";
52      }
53  
54      @Override
55      public void accept(SyntaxTreeVisitor visitor) {
56        visitor.visit(this);
57      }
58  
59      Expression getCondition() {
60        return condition;
61      }
62  
63      ImmutableList<Statement> getStmts() {
64        return stmts;
65      }
66  
67      @Override
68      void validate(ValidationEnvironment env) throws EvalException {
69        // EvalUtils.toBoolean() evaluates everything so we don't need type check here.
70        condition.validate(env);
71        validateStmts(env, stmts);
72      }
73    }
74  
75    private final ImmutableList<ConditionalStatements> thenBlocks;
76    private final ImmutableList<Statement> elseBlock;
77  
78    /**
79     * Constructs a if-elif-else statement. The else part is mandatory, but the list may be empty.
80     * ThenBlocks has to have at least one element.
81     */
82    IfStatement(List<ConditionalStatements> thenBlocks, List<Statement> elseBlock) {
83      Preconditions.checkArgument(!thenBlocks.isEmpty());
84      this.thenBlocks = ImmutableList.copyOf(thenBlocks);
85      this.elseBlock = ImmutableList.copyOf(elseBlock);
86    }
87  
88    public ImmutableList<ConditionalStatements> getThenBlocks() {
89      return thenBlocks;
90    }
91  
92    public ImmutableList<Statement> getElseBlock() {
93      return elseBlock;
94    }
95  
96    @Override
97    public String toString() {
98      // TODO(bazel-team): if we want to print the complete statement, the function
99      // needs an extra argument to specify indentation level.
100     return "if : ...\n";
101   }
102 
103   @Override
104   void exec(Environment env) throws EvalException, InterruptedException {
105     for (ConditionalStatements stmt : thenBlocks) {
106       if (EvalUtils.toBoolean(stmt.getCondition().eval(env))) {
107         stmt.exec(env);
108         return;
109       }
110     }
111     for (Statement stmt : elseBlock) {
112       stmt.exec(env);
113     }
114   }
115 
116   @Override
117   public void accept(SyntaxTreeVisitor visitor) {
118     visitor.visit(this);
119   }
120 
121   @Override
122   void validate(ValidationEnvironment env) throws EvalException {
123     env.startTemporarilyDisableReadonlyCheckSession();
124     for (ConditionalStatements stmts : thenBlocks) {
125       stmts.validate(env);
126     }
127     validateStmts(env, elseBlock);
128     env.finishTemporarilyDisableReadonlyCheckSession();
129   }
130 
131   private static void validateStmts(ValidationEnvironment env, List<Statement> stmts)
132       throws EvalException {
133     for (Statement stmt : stmts) {
134       stmt.validate(env);
135     }
136     env.finishTemporarilyDisableReadonlyCheckBranch();
137   }
138 }
