29  /**
30   * A customizable, serializable class for building memory efficient command lines.
31   */
32  @Immutable
33  public final class CustomCommandLine extends CommandLine {
34  
35    private abstract static class ArgvFragment {
36      abstract void eval(ImmutableList.Builder<String> builder);
37    }
38  
39    // It's better to avoid anonymous classes if we want to serialize command lines
40  
41    private static final class ObjectArg extends ArgvFragment {
42      private final Object arg;
43  
44      private ObjectArg(Object arg) {
45        this.arg = arg;
46      }
47  
48      @Override
49      void eval(ImmutableList.Builder<String> builder) {
50        builder.add(arg.toString());
51      }
52    }
53  
54    private static final class JoinExecPathsArg extends ArgvFragment {
55  
56      private final String delimiter;
57      private final Iterable<Artifact> artifacts;
58  
59      private JoinExecPathsArg(String delimiter, Iterable<Artifact> artifacts) {
60        this.delimiter = delimiter;
61        this.artifacts = CollectionUtils.makeImmutable(artifacts);
62      }
63  
64      @Override
65      void eval(ImmutableList.Builder<String> builder) {
66        builder.add(Artifact.joinExecPaths(delimiter, artifacts));
67      }
68    }
69  
70    private static final class PathWithTemplateArg extends ArgvFragment {
71  
72      private final String template;
73      private final PathFragment[] paths;
74  
75      private PathWithTemplateArg(String template, PathFragment... paths) {
76        this.template = template;
77        this.paths = paths;
78      }
79  
80      @Override
81      void eval(ImmutableList.Builder<String> builder) {
82        // PathFragment.toString() uses getPathString()
83        builder.add(String.format(template, (Object[]) paths));
84      }
85    }

103   /**
104    * Custom Java code producing a List of String arguments. Usage of this class is discouraged.
105    */
106   public abstract static class CustomMultiArgv extends ArgvFragment {
107 
108     @Override
109     void eval(ImmutableList.Builder<String> builder) {
110       builder.addAll(argv());
111     }
112 
113     public abstract Iterable<String> argv();
114   }

116   private static final class JoinPathsArg extends ArgvFragment {
117 
118     private final String delimiter;
119     private final Iterable<PathFragment> paths;
120 
121     private JoinPathsArg(String delimiter, Iterable<PathFragment> paths) {
122       this.delimiter = delimiter;
123       this.paths = CollectionUtils.makeImmutable(paths);
124     }
125 
126     @Override
127     void eval(ImmutableList.Builder<String> builder) {
128       builder.add(Joiner.on(delimiter).join(paths));
129     }
130   }

132   /**
133    * Arguments that intersperse strings between the items in a sequence. There are two forms of
134    * interspersing, and either may be used by this implementation:
135    * <ul>
136    *   <li>before each - a string is added before each item in a sequence. e.g.
137    *       {@code -f foo -f bar -f baz}
138    *   <li>format each - a format string is used to format each item in a sequence. e.g.
139    *       {@code -I/foo -I/bar -I/baz} for the format {@code "-I%s"}
140    * </ul>
141    *
142    * <p>This class could be used both with both the "before" and "format" features at the same
143    * time, but this is probably more confusion than it is worth. If you need this functionality,
144    * consider using "before" only but storing the strings pre-formated in a {@link NestedSet}.
145    */
146   private static final class InterspersingArgs extends ArgvFragment {
147     private final Iterable<?> sequence;
148     private final String beforeEach;
149     private final String formatEach;
150 
151     /**
152      * Do not call from outside this class because this does not guarantee that {@code sequence} is
153      * immutable.
154      */
155     private InterspersingArgs(Iterable<?> sequence, String beforeEach, String formatEach) {
156       this.sequence = sequence;
157       this.beforeEach = beforeEach;
158       this.formatEach = formatEach;
159     }
160 
161     static InterspersingArgs fromStrings(
162         Iterable<?> sequence, String beforeEach, String formatEach) {
163       return new InterspersingArgs(
164           CollectionUtils.makeImmutable(sequence), beforeEach, formatEach);
165     }
166 
167     static InterspersingArgs fromExecPaths(
168         Iterable<Artifact> sequence, String beforeEach, String formatEach) {
169       return new InterspersingArgs(
170           Artifact.toExecPaths(CollectionUtils.makeImmutable(sequence)), beforeEach, formatEach);
171     }
172 
173     @Override
174     void eval(ImmutableList.Builder<String> builder) {
175       for (Object item : sequence) {
176         if (item == null) {
177           continue;
178         }
179 
180         if (beforeEach != null) {
181           builder.add(beforeEach);
182         }
183         String arg = item.toString();
184         if (formatEach != null) {
185           arg = String.format(formatEach, arg);
186         }
187         builder.add(arg);
188       }
189     }
190   }

192   /**
193    * A Builder class for CustomCommandLine with the appropriate methods.
194    *
195    * <p>{@link Iterable} instances passed to {@code add*} methods will be stored internally as
196    * collections that are known to be immutable copies. This means that any {@link Iterable} that is
197    * not a {@link NestedSet} or {@link ImmutableList} may be copied.
198    *
199    * <p>{@code addFormatEach*} methods take an {@link Iterable} but use these as arguments to
200    * {@link String#format(String, Object...)} with a certain constant format string. For instance,
201    * if {@code format} is {@code "-I%s"}, then the final arguments may be
202    * {@code -Ifoo -Ibar -Ibaz}
203    *
204    * <p>{@code addBeforeEach*} methods take an {@link Iterable} but insert a certain {@link String}
205    * once before each element in the string, meaning the total number of elements added is twice the
206    * length of the {@link Iterable}. For instance: {@code -f foo -f bar -f baz}
207    */
208   public static final class Builder {
209 
210     private final List<ArgvFragment> arguments = new ArrayList<>();
211 
212     public Builder add(CharSequence arg) {
213       if (arg != null) {
214         arguments.add(new ObjectArg(arg));
215       }
216       return this;
217     }
218 
219     public Builder add(Label arg) {
220       if (arg != null) {
221         arguments.add(new ObjectArg(arg));
222       }
223       return this;
224     }
225 
226     public Builder add(String arg, Iterable<String> args) {
227       if (arg != null && args != null) {
228         arguments.add(new ObjectArg(arg));
229         arguments.add(InterspersingArgs.fromStrings(args, /*beforeEach=*/null, "%s"));
230       }
231       return this;
232     }
233 
234     public Builder add(Iterable<String> args) {
235       if (args != null) {
236         arguments.add(InterspersingArgs.fromStrings(args, /*beforeEach=*/null, "%s"));
237       }
238       return this;
239     }
240 
241     public Builder addExecPath(String arg, Artifact artifact) {
242       if (arg != null && artifact != null) {
243         arguments.add(new ObjectArg(arg));
244         arguments.add(new ObjectArg(artifact.getExecPath()));
245       }
246       return this;
247     }
248 
249     public Builder addExecPaths(String arg, Iterable<Artifact> artifacts) {
250       if (arg != null && artifacts != null) {
251         arguments.add(new ObjectArg(arg));
252         arguments.add(InterspersingArgs.fromExecPaths(artifacts, /*beforeEach=*/null, "%s"));
253       }
254       return this;
255     }
256 
257     public Builder addExecPaths(Iterable<Artifact> artifacts) {
258       if (artifacts != null) {
259         arguments.add(InterspersingArgs.fromExecPaths(artifacts, /*beforeEach=*/null, "%s"));
260       }
261       return this;
262     }
263 
264     public Builder addJoinExecPaths(String arg, String delimiter, Iterable<Artifact> artifacts) {
265       if (arg != null && artifacts != null) {
266         arguments.add(new ObjectArg(arg));
267         arguments.add(new JoinExecPathsArg(delimiter, artifacts));
268       }
269       return this;
270     }
271 
272     public Builder addPath(PathFragment path) {
273       if (path != null) {
274         arguments.add(new ObjectArg(path));
275       }
276       return this;
277     }
278 
279     public Builder addPaths(String template, PathFragment... path) {
280       if (template != null && path != null) {
281         arguments.add(new PathWithTemplateArg(template, path));
282       }
283       return this;
284     }
285 
286     public Builder addJoinPaths(String delimiter, Iterable<PathFragment> paths) {
287       if (delimiter != null && paths != null) {
288         arguments.add(new JoinPathsArg(delimiter, paths));
289       }
290       return this;
291     }
292 
293     public Builder addBeforeEachPath(String repeated, Iterable<PathFragment> paths) {
294       if (repeated != null && paths != null) {
295         arguments.add(InterspersingArgs.fromStrings(paths, repeated, "%s"));
296       }
297       return this;
298     }
299 
300     public Builder addBeforeEach(String repeated, Iterable<String> strings) {
301       if (repeated != null && strings != null) {
302         arguments.add(InterspersingArgs.fromStrings(strings, repeated, "%s"));
303       }
304       return this;
305     }
306 
307     public Builder addBeforeEachExecPath(String repeated, Iterable<Artifact> artifacts) {
308       if (repeated != null && artifacts != null) {
309         arguments.add(InterspersingArgs.fromExecPaths(artifacts, repeated, "%s"));
310       }
311       return this;
312     }
313 
314     public Builder addFormatEach(String format, Iterable<String> strings) {
315       if (format != null && strings != null) {
316         arguments.add(InterspersingArgs.fromStrings(strings, /*beforeEach=*/null, format));
317       }
318       return this;
319     }
320 
321     public Builder add(CustomArgv arg) {
322       if (arg != null) {
323         arguments.add(arg);
324       }
325       return this;
326     }
327 
328     public Builder add(CustomMultiArgv arg) {
329       if (arg != null) {
330         arguments.add(arg);
331       }
332       return this;
333     }
334 
335     public CustomCommandLine build() {
336       return new CustomCommandLine(arguments);
337     }
338   }
339 
340   public static Builder builder() {
341     return new Builder();
342   }
343 
344   private final ImmutableList<ArgvFragment> arguments;
345 
346   private CustomCommandLine(List<ArgvFragment> arguments) {
347     this.arguments = ImmutableList.copyOf(arguments);
348   }
349 
350   @Override
351   public Iterable<String> arguments() {
352     ImmutableList.Builder<String> builder = ImmutableList.builder();
353     for (ArgvFragment arg : arguments) {
354       arg.eval(builder);
355     }
356     return builder.build();
357   }
