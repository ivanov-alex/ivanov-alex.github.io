74    /**
75     * Returns the visibility of the given target. Errors during package group resolution are reported
76     * to the {@code AnalysisEnvironment}.
77     */
78    private NestedSet<PackageSpecification> convertVisibility(
79        ListMultimap<Attribute, ConfiguredTarget> prerequisiteMap, EventHandler reporter,
80        Target target, BuildConfiguration packageGroupConfiguration) {
81      RuleVisibility ruleVisibility = target.getVisibility();
82      if (ruleVisibility instanceof ConstantRuleVisibility) {
83        return ((ConstantRuleVisibility) ruleVisibility).isPubliclyVisible()
84            ? NestedSetBuilder.<PackageSpecification>create(
85                Order.STABLE_ORDER, PackageSpecification.EVERYTHING)
86            : NestedSetBuilder.<PackageSpecification>emptySet(Order.STABLE_ORDER);
87      } else if (ruleVisibility instanceof PackageGroupsRuleVisibility) {
88        PackageGroupsRuleVisibility packageGroupsVisibility =
89            (PackageGroupsRuleVisibility) ruleVisibility;
90  
91        NestedSetBuilder<PackageSpecification> packageSpecifications =
92            NestedSetBuilder.stableOrder();
93        for (Label groupLabel : packageGroupsVisibility.getPackageGroups()) {
94          // PackageGroupsConfiguredTargets are always in the package-group configuration.
95          ConfiguredTarget group =
96              findPrerequisite(prerequisiteMap, groupLabel, packageGroupConfiguration);
97          PackageSpecificationProvider provider = null;
98          // group == null can only happen if the package group list comes
99          // from a default_visibility attribute, because in every other case,
100         // this missing link is caught during transitive closure visitation or
101         // if the RuleConfiguredTargetGraph threw out a visibility edge
102         // because if would have caused a cycle. The filtering should be done
103         // in a single place, ConfiguredTargetGraph, but for now, this is the
104         // minimally invasive way of providing a sane error message in case a
105         // cycle is created by a visibility attribute.
106         if (group != null) {
107           provider = group.getProvider(PackageSpecificationProvider.class);
108         }
109         if (provider != null) {
110           packageSpecifications.addTransitive(provider.getPackageSpecifications());
111         } else {
112           reporter.handle(Event.error(target.getLocation(),
113               String.format("Label '%s' does not refer to a package group", groupLabel)));
114         }
115       }
116 
117       packageSpecifications.addAll(packageGroupsVisibility.getDirectPackages());
118       return packageSpecifications.build();
119     } else {
120       throw new IllegalStateException("unknown visibility");
121     }
122   }

124   private ConfiguredTarget findPrerequisite(
125       ListMultimap<Attribute, ConfiguredTarget> prerequisiteMap, Label label,
126       BuildConfiguration config) {
127     for (ConfiguredTarget prerequisite : prerequisiteMap.get(null)) {
128       if (prerequisite.getLabel().equals(label) && (prerequisite.getConfiguration() == config)) {
129         return prerequisite;
130       }
131     }
132     return null;
133   }

135   private Artifact getOutputArtifact(OutputFile outputFile, BuildConfiguration configuration,
136       boolean isFileset, ArtifactFactory artifactFactory) {
137     Rule rule = outputFile.getAssociatedRule();
138     Root root = rule.hasBinaryOutput()
139         ? configuration.getBinDirectory()
140         : configuration.getGenfilesDirectory();
141     ArtifactOwner owner =
142         new ConfiguredTargetKey(rule.getLabel(), configuration.getArtifactOwnerConfiguration());
143     PathFragment rootRelativePath =
144         outputFile.getLabel().getPackageIdentifier().getPathFragment().getRelative(
145             outputFile.getLabel().getName());
146     Artifact result = isFileset
147         ? artifactFactory.getFilesetArtifact(rootRelativePath, root, owner)
148         : artifactFactory.getDerivedArtifact(rootRelativePath, root, owner);
149     // The associated rule should have created the artifact.
150     Preconditions.checkNotNull(result, "no artifact for %s", rootRelativePath);
151     return result;
152   }

154   /**
155    * Invokes the appropriate constructor to create a {@link ConfiguredTarget} instance.
156    * <p>For use in {@code ConfiguredTargetFunction}.
157    *
158    * <p>Returns null if Skyframe deps are missing or upon certain errors.
159    */
160   @Nullable
161   public final ConfiguredTarget createConfiguredTarget(AnalysisEnvironment analysisEnvironment,
162       ArtifactFactory artifactFactory, Target target, BuildConfiguration config,
163       BuildConfiguration hostConfig, ListMultimap<Attribute, ConfiguredTarget> prerequisiteMap,
164       Set<ConfigMatchingProvider> configConditions)
165       throws InterruptedException {
166     if (target instanceof Rule) {
167       return createRule(analysisEnvironment, (Rule) target, config, hostConfig,
168           prerequisiteMap, configConditions);
169     }
170 
171     // Visibility, like all package groups, doesn't have a configuration
172     NestedSet<PackageSpecification> visibility = convertVisibility(
173         prerequisiteMap, analysisEnvironment.getEventHandler(), target, null);
174     TargetContext targetContext = new TargetContext(analysisEnvironment, target, config,
175         prerequisiteMap.get(null), visibility);
176     if (target instanceof OutputFile) {
177       OutputFile outputFile = (OutputFile) target;
178       boolean isFileset = outputFile.getGeneratingRule().getRuleClass().equals("Fileset");
179       Artifact artifact = getOutputArtifact(outputFile, config, isFileset, artifactFactory);
180       TransitiveInfoCollection rule = targetContext.findDirectPrerequisite(
181           outputFile.getGeneratingRule().getLabel(), config);
182       if (isFileset) {
183         return new FilesetOutputConfiguredTarget(
184             targetContext,
185             outputFile,
186             rule,
187             artifact,
188             rule.getProvider(FilesetProvider.class).getTraversals());
189       } else {
190         return new OutputFileConfiguredTarget(targetContext, outputFile, rule, artifact);
191       }
192     } else if (target instanceof InputFile) {
193       InputFile inputFile = (InputFile) target;
194       Artifact artifact = artifactFactory.getSourceArtifact(
195           inputFile.getExecPath(),
196           Root.asSourceRoot(inputFile.getPackage().getSourceRoot()),
197           new ConfiguredTargetKey(target.getLabel(), config));
198 
199       return new InputFileConfiguredTarget(targetContext, inputFile, artifact);
200     } else if (target instanceof PackageGroup) {
201       PackageGroup packageGroup = (PackageGroup) target;
202       return new PackageGroupConfiguredTarget(targetContext, packageGroup);
203     } else {
204       throw new AssertionError("Unexpected target class: " + target.getClass().getName());
205     }
206   }

208   /**
209    * Factory method: constructs a RuleConfiguredTarget of the appropriate class, based on the rule
210    * class. May return null if an error occurred.
211    */
212   @Nullable
213   private ConfiguredTarget createRule(
214       AnalysisEnvironment env, Rule rule, BuildConfiguration configuration,
215       BuildConfiguration hostConfiguration,
216       ListMultimap<Attribute, ConfiguredTarget> prerequisiteMap,
217       Set<ConfigMatchingProvider> configConditions) throws InterruptedException {
218     // Visibility computation and checking is done for every rule.
219     RuleContext ruleContext = new RuleContext.Builder(env, rule, configuration, hostConfiguration,
220         ruleClassProvider.getPrerequisiteValidator())
221         .setVisibility(convertVisibility(prerequisiteMap, env.getEventHandler(), rule, null))
222         .setPrerequisites(prerequisiteMap)
223         .setConfigConditions(configConditions)
224         .build();
225     if (ruleContext.hasErrors()) {
226       return null;
227     }
228     if (!rule.getRuleClassObject().getRequiredConfigurationFragments().isEmpty()) {
229       MissingFragmentPolicy missingFragmentPolicy =
230           rule.getRuleClassObject().missingFragmentPolicy();
231       if (missingFragmentPolicy != MissingFragmentPolicy.IGNORE
232           && !configuration.hasAllFragments(
233               rule.getRuleClassObject().getRequiredConfigurationFragments())) {
234         if (missingFragmentPolicy == MissingFragmentPolicy.FAIL_ANALYSIS) {
235           ruleContext.ruleError(missingFragmentError(ruleContext));
236           return null;
237         }
238         return createFailConfiguredTarget(ruleContext);
239       }
240     }
241     if (rule.getRuleClassObject().isSkylarkExecutable()) {
242       // TODO(bazel-team): maybe merge with RuleConfiguredTargetBuilder?
243       return SkylarkRuleConfiguredTargetBuilder.buildRule(
244           ruleContext, rule.getRuleClassObject().getConfiguredTargetFunction());
245     } else {
246       RuleClass.ConfiguredTargetFactory<ConfiguredTarget, RuleContext> factory =
247           rule.getRuleClassObject().<ConfiguredTarget, RuleContext>getConfiguredTargetFactory();
248       Preconditions.checkNotNull(factory, rule.getRuleClassObject());
249       return factory.create(ruleContext);
250     }
251   }

253   private String missingFragmentError(RuleContext ruleContext) {
254     RuleClass ruleClass = ruleContext.getRule().getRuleClassObject();
255     Set<Class<?>> missingFragments = new LinkedHashSet<>();
256     for (Class<?> fragment : ruleClass.getRequiredConfigurationFragments()) {
257       if (!ruleContext.getConfiguration().hasFragment(fragment.asSubclass(Fragment.class))) {
258         missingFragments.add(fragment);
259       }
260     }
261     Preconditions.checkState(!missingFragments.isEmpty());
262     StringBuilder result = new StringBuilder();
263     result.append("all rules of type " + ruleClass.getName() + " require the presence of ");
264     List<String> names = new ArrayList<>();
265     for (Class<?> fragment : missingFragments) {
266       // TODO(bazel-team): Using getSimpleName here is sub-optimal, but we don't have anything
267       // better right now.
268       names.add(fragment.getSimpleName());
269     }
270     result.append("all of [");
271     Joiner.on(",").appendTo(result, names);
272     result.append("], but these were all disabled");
273     return result.toString();
274   }

276   /**
277    * Constructs an {@link Aspect}. Returns null if an error occurs; in that case,
278    * {@code aspectFactory} should call one of the error reporting methods of {@link RuleContext}.
279    */
280   public Aspect createAspect(
281       AnalysisEnvironment env, RuleConfiguredTarget associatedTarget,
282       ConfiguredAspectFactory aspectFactory,
283       ListMultimap<Attribute, ConfiguredTarget> prerequisiteMap,
284       Set<ConfigMatchingProvider> configConditions) {
285     RuleContext.Builder builder = new RuleContext.Builder(env,
286         associatedTarget.getTarget(),
287         associatedTarget.getConfiguration(),
288         getHostConfiguration(associatedTarget.getConfiguration()),
289         ruleClassProvider.getPrerequisiteValidator());
290     RuleContext ruleContext = builder
291         .setVisibility(convertVisibility(
292             prerequisiteMap, env.getEventHandler(), associatedTarget.getTarget(), null))
293         .setPrerequisites(prerequisiteMap)
294         .setConfigConditions(configConditions)
295         .build();
296     if (ruleContext.hasErrors()) {
297       return null;
298     }
299 
300     return aspectFactory.create(associatedTarget, ruleContext);
301   }
302 
303   private static BuildConfiguration getHostConfiguration(BuildConfiguration config) {
304     // TODO(bazel-team): support dynamic transitions.
305     return config == null ? null : config.getConfiguration(Attribute.ConfigurationTransition.HOST);
306   }

308   /**
309    * A pseudo-implementation for configured targets that creates fail actions for all declared
310    * outputs, both implicit and explicit.
311    */
312   private static ConfiguredTarget createFailConfiguredTarget(RuleContext ruleContext) {
313     RuleConfiguredTargetBuilder builder = new RuleConfiguredTargetBuilder(ruleContext);
314     if (!ruleContext.getOutputArtifacts().isEmpty()) {
315       ruleContext.registerAction(new FailAction(ruleContext.getActionOwner(),
316           ruleContext.getOutputArtifacts(), "Can't build this"));
317     }
318     builder.add(RunfilesProvider.class, RunfilesProvider.simple(Runfiles.EMPTY));
319     return builder.build();
320   }
