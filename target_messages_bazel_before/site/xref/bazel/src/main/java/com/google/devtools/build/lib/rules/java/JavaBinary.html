48  public class JavaBinary implements RuleConfiguredTargetFactory {
49    private static final PathFragment CPP_RUNTIMES = new PathFragment("_cpp_runtimes");

51    private final JavaSemantics semantics;
52  
53    protected JavaBinary(JavaSemantics semantics) {
54      this.semantics = semantics;
55    }

57    @Override
58    public ConfiguredTarget create(RuleContext ruleContext) {
59      final JavaCommon common = new JavaCommon(ruleContext, semantics);
60      DeployArchiveBuilder deployArchiveBuilder =  new DeployArchiveBuilder(semantics, ruleContext);
61      Runfiles.Builder runfilesBuilder = new Runfiles.Builder();
62      List<String> jvmFlags = new ArrayList<>();
63  
64      common.initializeJavacOpts();
65      JavaTargetAttributes.Builder attributesBuilder = common.initCommon();
66      attributesBuilder.addClassPathResources(
67          ruleContext.getPrerequisiteArtifacts("classpath_resources", Mode.TARGET).list());
68  
69      List<String> userJvmFlags = common.getJvmFlags();
70  
71      ruleContext.checkSrcsSamePackage(true);
72      boolean createExecutable = ruleContext.attributes().get("create_executable", Type.BOOLEAN);
73      List<TransitiveInfoCollection> deps =
74          Lists.newArrayList(common.targetsTreatedAsDeps(ClasspathType.COMPILE_ONLY));
75      semantics.checkRule(ruleContext, common);
76      String mainClass = semantics.getMainClass(ruleContext, common);
77      String originalMainClass = mainClass;
78      if (ruleContext.hasErrors()) {
79        return null;
80      }
81  
82      // Collect the transitive dependencies.
83      JavaCompilationHelper helper = new JavaCompilationHelper(
84          ruleContext, semantics, common.getJavacOpts(), attributesBuilder);
85      helper.addLibrariesToAttributes(deps);
86      helper.addProvidersToAttributes(common.compilationArgsFromSources(), /* isNeverLink */ false);
87      attributesBuilder.addNativeLibraries(
88          collectNativeLibraries(common.targetsTreatedAsDeps(ClasspathType.BOTH)));
89  
90      // deploy_env is valid for java_binary, but not for java_test.
91      if (ruleContext.getRule().isAttrDefined("deploy_env", Type.LABEL_LIST)) {
92        for (JavaRuntimeClasspathProvider envTarget : ruleContext.getPrerequisites(
93                 "deploy_env", Mode.TARGET, JavaRuntimeClasspathProvider.class)) {
94          attributesBuilder.addExcludedArtifacts(envTarget.getRuntimeClasspath());
95        }
96      }
97  
98      Artifact srcJar =
99          ruleContext.getImplicitOutputArtifact(JavaSemantics.JAVA_BINARY_SOURCE_JAR);
100 
101     Artifact classJar =
102         ruleContext.getImplicitOutputArtifact(JavaSemantics.JAVA_BINARY_CLASS_JAR);
103 
104     ImmutableList<Artifact> srcJars = ImmutableList.of(srcJar);
105 
106     Artifact launcher = semantics.getLauncher(ruleContext, common, deployArchiveBuilder,
107         runfilesBuilder, jvmFlags, attributesBuilder);
108     JavaCompilationArtifacts.Builder javaArtifactsBuilder = new JavaCompilationArtifacts.Builder();
109     Artifact instrumentationMetadata =
110         helper.createInstrumentationMetadata(classJar, javaArtifactsBuilder);
111 
112     NestedSetBuilder<Artifact> filesBuilder = NestedSetBuilder.stableOrder();
113     Artifact executable = null;
114     if (createExecutable) {
115       executable = ruleContext.createOutputArtifact(); // the artifact for the rule itself
116       filesBuilder.add(classJar).add(executable);
117 
118       if (ruleContext.getConfiguration().isCodeCoverageEnabled()) {
119         mainClass = semantics.addCoverageSupport(helper, attributesBuilder,
120             executable, instrumentationMetadata, javaArtifactsBuilder, mainClass);
121       }
122     } else {
123       filesBuilder.add(classJar);
124     }
125 
126     JavaTargetAttributes attributes = helper.getAttributes();
127     List<Artifact> nativeLibraries = attributes.getNativeLibraries();
128     if (!nativeLibraries.isEmpty()) {
129       jvmFlags.add("-Djava.library.path="
130           + JavaCommon.javaLibraryPath(nativeLibraries, ruleContext.getRule().getWorkspaceName()));
131     }
132 
133     JavaConfiguration javaConfig = ruleContext.getFragment(JavaConfiguration.class);
134     if (attributes.hasMessages()) {
135       helper.addTranslations(semantics.translate(ruleContext, javaConfig,
136           attributes.getMessages()));
137     }
138 
139     if (attributes.hasSourceFiles() || attributes.hasSourceJars()
140         || attributes.hasResources() || attributes.hasClassPathResources()) {
141       // We only want to add a jar to the classpath of a dependent rule if it has content.
142       javaArtifactsBuilder.addRuntimeJar(classJar);
143     }
144 
145     // Any JAR files should be added to the collection of runtime jars.
146     javaArtifactsBuilder.addRuntimeJars(attributes.getJarFiles());
147 
148     Artifact outputDepsProto = helper.createOutputDepsProtoArtifact(classJar, javaArtifactsBuilder);
149 
150     common.setJavaCompilationArtifacts(javaArtifactsBuilder.build());
151 
152     // The gensrc jar is created only if the target uses annotation processing. Otherwise,
153     // it is null, and the source jar action will not depend on the compile action.
154     Artifact gensrcJar = helper.createGensrcJar(classJar);
155     Artifact manifestProtoOutput = helper.createManifestProtoOutput(classJar);
156 
157     helper.createCompileAction(
158         classJar, manifestProtoOutput, gensrcJar, outputDepsProto, instrumentationMetadata);
159     helper.createSourceJarAction(srcJar, gensrcJar);
160 
161     Artifact genClassJar = ruleContext.getImplicitOutputArtifact(JavaSemantics.JAVA_BINARY_GEN_JAR);
162     helper.createGenJarAction(classJar, manifestProtoOutput, genClassJar);
163 
164     common.setClassPathFragment(new ClasspathConfiguredFragment(
165         common.getJavaCompilationArtifacts(), attributes, false));
166 
167     // Collect the action inputs for the runfiles collector here because we need to access the
168     // analysis environment, and that may no longer be safe when the runfiles collector runs.
169     Iterable<Artifact> dynamicRuntimeActionInputs =
170         CppHelper.getToolchain(ruleContext).getDynamicRuntimeLinkInputs();
171 
172 
173     Iterables.addAll(jvmFlags, semantics.getJvmFlags(ruleContext, common, launcher, userJvmFlags));
174     if (ruleContext.hasErrors()) {
175       return null;
176     }
177 
178     if (createExecutable) {
179       // Create a shell stub for a Java application
180       semantics.createStubAction(ruleContext, common, jvmFlags, executable, mainClass,
181           common.getJavaBinSubstitution(launcher));
182     }
183 
184     NestedSet<Artifact> transitiveSourceJars = collectTransitiveSourceJars(common, srcJar);
185 
186     // TODO(bazel-team): if (getOptions().sourceJars) then make this a dummy prerequisite for the
187     // DeployArchiveAction ? Needs a few changes there as we can't pass inputs
188     helper.createSourceJarAction(semantics, ImmutableList.<Artifact>of(),
189         transitiveSourceJars.toCollection(),
190         ruleContext.getImplicitOutputArtifact(JavaSemantics.JAVA_BINARY_DEPLOY_SOURCE_JAR));
191 
192     RuleConfiguredTargetBuilder builder =
193         new RuleConfiguredTargetBuilder(ruleContext);
194 
195     semantics.addProviders(ruleContext, common, jvmFlags, classJar, srcJar, genClassJar, gensrcJar,
196         ImmutableMap.<Artifact, Artifact>of(), helper, filesBuilder, builder);
197 
198     NestedSet<Artifact> filesToBuild = filesBuilder.build();
199 
200     collectDefaultRunfiles(runfilesBuilder, ruleContext, common, filesToBuild, launcher,
201         dynamicRuntimeActionInputs);
202     Runfiles defaultRunfiles = runfilesBuilder.build();
203 
204     RunfilesSupport runfilesSupport = createExecutable
205         ? runfilesSupport = RunfilesSupport.withExecutable(
206             ruleContext, defaultRunfiles, executable,
207             semantics.getExtraArguments(ruleContext, common))
208         : null;
209 
210     RunfilesProvider runfilesProvider = RunfilesProvider.withData(
211         defaultRunfiles,
212         new Runfiles.Builder().merge(runfilesSupport).build());
213 
214     ImmutableList<String> deployManifestLines =
215         getDeployManifestLines(ruleContext, originalMainClass);
216 
217     Artifact deployJar =
218         ruleContext.getImplicitOutputArtifact(JavaSemantics.JAVA_BINARY_DEPLOY_JAR);
219 
220     deployArchiveBuilder
221         .setOutputJar(deployJar)
222         .setJavaStartClass(mainClass)
223         .setDeployManifestLines(deployManifestLines)
224         .setAttributes(attributes)
225         .addRuntimeJars(common.getJavaCompilationArtifacts().getRuntimeJars())
226         .setIncludeBuildData(true)
227         .setRunfilesMiddleman(
228             runfilesSupport == null ? null : runfilesSupport.getRunfilesMiddleman())
229         .setCompression(COMPRESSED)
230         .setLauncher(launcher);
231 
232     deployArchiveBuilder.build();
233 
234     common.addTransitiveInfoProviders(builder, filesToBuild, classJar);
235 
236     return builder
237         .setFilesToBuild(filesToBuild)
238         .add(RunfilesProvider.class, runfilesProvider)
239         .setRunfilesSupport(runfilesSupport, executable)
240         .add(JavaRuntimeClasspathProvider.class,
241             new JavaRuntimeClasspathProvider(common.getRuntimeClasspath()))
242         .add(JavaSourceJarsProvider.class,
243             new JavaSourceJarsProvider(transitiveSourceJars, srcJars))
244         .addOutputGroup(JavaSemantics.SOURCE_JARS_OUTPUT_GROUP, transitiveSourceJars)
245         .addOutputGroup(JavaSemantics.GENERATED_JARS_OUTPUT_GROUP, genClassJar)
246         .build();
247   }

249   // Create the deploy jar and make it dependent on the runfiles middleman if an executable is
250   // created. Do not add the deploy jar to files to build, so we will only build it when it gets
251   // requested.
252   private ImmutableList<String> getDeployManifestLines(RuleContext ruleContext,
253       String originalMainClass) {
254     ImmutableList.Builder<String> builder = ImmutableList.<String>builder()
255           .addAll(ruleContext.attributes().get("deploy_manifest_lines", Type.STRING_LIST));
256     if (ruleContext.getConfiguration().isCodeCoverageEnabled()) {
257       builder.add("Coverage-Main-Class: " + originalMainClass);
258     }
259     return builder.build();
260   }

262   private void collectDefaultRunfiles(Runfiles.Builder builder, RuleContext ruleContext,
263       JavaCommon common, NestedSet<Artifact> filesToBuild, Artifact launcher,
264       Iterable<Artifact> dynamicRuntimeActionInputs) {
265     // Convert to iterable: filesToBuild has a different order.
266     builder.addArtifacts((Iterable<Artifact>) filesToBuild);
267     builder.addArtifacts(common.getJavaCompilationArtifacts().getRuntimeJars());
268     if (launcher != null) {
269       final TransitiveInfoCollection defaultLauncher =
270           JavaHelper.launcherForTarget(semantics, ruleContext);
271       final Artifact defaultLauncherArtifact =
272           JavaHelper.launcherArtifactForTarget(semantics, ruleContext);
273       if (!defaultLauncherArtifact.equals(launcher)) {
274         builder.addArtifact(launcher);
275 
276         // N.B. The "default launcher" referred to here is the launcher target specified through
277         // an attribute or flag. We wish to retain the runfiles of the default launcher, *except*
278         // for the original cc_binary artifact, because we've swapped it out with our custom
279         // launcher. Hence, instead of calling builder.addTarget(), or adding an odd method
280         // to Runfiles.Builder, we "unravel" the call and manually add things to the builder.
281         // Because the NestedSet representing each target's launcher runfiles is re-built here,
282         // we may see increased memory consumption for representing the target's runfiles.
283         Runfiles runfiles =
284             defaultLauncher.getProvider(RunfilesProvider.class)
285               .getDefaultRunfiles();
286         NestedSetBuilder<Artifact> unconditionalArtifacts = NestedSetBuilder.compileOrder();
287         for (Artifact a : runfiles.getUnconditionalArtifacts()) {
288           if (!a.equals(defaultLauncherArtifact)) {
289             unconditionalArtifacts.add(a);
290           }
291         }
292         builder.addTransitiveArtifacts(unconditionalArtifacts.build());
293         builder.addSymlinks(runfiles.getSymlinks());
294         builder.addRootSymlinks(runfiles.getRootSymlinks());
295         builder.addPruningManifests(runfiles.getPruningManifests());
296       } else {
297         builder.addTarget(defaultLauncher, RunfilesProvider.DEFAULT_RUNFILES);
298       }
299     }
300 
301     semantics.addRunfilesForBinary(ruleContext, launcher, builder);
302     builder.addRunfiles(ruleContext, RunfilesProvider.DEFAULT_RUNFILES);
303     builder.add(ruleContext, JavaRunfilesProvider.TO_RUNFILES);
304 
305     List<? extends TransitiveInfoCollection> runtimeDeps =
306         ruleContext.getPrerequisites("runtime_deps", Mode.TARGET);
307     builder.addTargets(runtimeDeps, JavaRunfilesProvider.TO_RUNFILES);
308     builder.addTargets(runtimeDeps, RunfilesProvider.DEFAULT_RUNFILES);
309     semantics.addDependenciesForRunfiles(ruleContext, builder);
310 
311     if (ruleContext.getConfiguration().isCodeCoverageEnabled()) {
312       Artifact instrumentedJar = common.getJavaCompilationArtifacts().getInstrumentedJar();
313       if (instrumentedJar != null) {
314         builder.addArtifact(instrumentedJar);
315       }
316     }
317 
318     builder.addArtifacts((Iterable<Artifact>) common.getRuntimeClasspath());
319 
320     // Add the JDK files if it comes from the source repository (see java_stub_template.txt).
321     TransitiveInfoCollection javabaseTarget = ruleContext.getPrerequisite(":jvm", Mode.HOST);
322     if (javabaseTarget != null) {
323       builder.addArtifacts(
324           (Iterable<Artifact>) javabaseTarget.getProvider(FileProvider.class).getFilesToBuild());
325 
326       // Add symlinks to the C++ runtime libraries under a path that can be built
327       // into the Java binary without having to embed the crosstool, gcc, and grte
328       // version information contained within the libraries' package paths.
329       for (Artifact lib : dynamicRuntimeActionInputs) {
330         PathFragment path = CPP_RUNTIMES.getRelative(lib.getExecPath().getBaseName());
331         builder.addSymlink(path, lib);
332       }
333     }
334   }

336   private NestedSet<Artifact> collectTransitiveSourceJars(JavaCommon common, Artifact srcJar) {
337     NestedSetBuilder<Artifact> builder = NestedSetBuilder.stableOrder();
338 
339     builder.add(srcJar);
340     for (JavaSourceJarsProvider dep : common.getDependencies(JavaSourceJarsProvider.class)) {
341       builder.addTransitive(dep.getTransitiveSourceJars());
342     }
343     return builder.build();
344   }

346   /**
347    * Collects the native libraries in the transitive closure of the deps.
348    *
349    * @param deps the dependencies to be included as roots of the transitive closure.
350    * @return the native libraries found in the transitive closure of the deps.
351    */
352   public static Collection<Artifact> collectNativeLibraries(
353       Iterable<? extends TransitiveInfoCollection> deps) {
354     NestedSet<LinkerInput> linkerInputs = new NativeLibraryNestedSetBuilder()
355         .addJavaTargets(deps)
356         .build();
357     ImmutableList.Builder<Artifact> result = ImmutableList.builder();
358     for (LinkerInput linkerInput : linkerInputs) {
359       result.add(linkerInput.getArtifact());
360     }
361 
362     return result.build();
363   }
