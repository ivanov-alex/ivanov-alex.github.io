60    public TestActionBuilder(RuleContext ruleContext) {
61      this.ruleContext = ruleContext;
62      this.extraEnv = ImmutableMap.of();
63    }
64  
65    /**
66     * Creates the test actions and artifacts using the previously set parameters.
67     *
68     * @return ordered list of test status artifacts
69     */
70    public TestParams build() {
71      Preconditions.checkState(runfilesSupport != null);
72      boolean local = TargetUtils.isTestRuleAndRunsLocally(ruleContext.getRule());
73      TestShardingStrategy strategy = ruleContext.getConfiguration().testShardingStrategy();
74      int shards = strategy.getNumberOfShards(
75          local, explicitShardCount, isTestShardingCompliant(),
76          TestSize.getTestSize(ruleContext.getRule()));
77      Preconditions.checkState(shards >= 0);
78      return createTestAction(Util.getWorkspaceRelativePath(ruleContext.getLabel()), shards);
79    }
80  
81    private boolean isTestShardingCompliant() {
82      // See if it has a data dependency on the special target
83      // //tools:test_sharding_compliant. Test runners add this dependency
84      // to show they speak the sharding protocol.
85      // There are certain cases where this heuristic may fail, giving
86      // a "false positive" (where we shard the test even though the
87      // it isn't supported). We may want to refine this logic, but
88      // heuristically sharding is currently experimental. Also, we do detect
89      // false-positive cases and return an error.
90      return runfilesSupport.getRunfilesSymlinkNames().contains(
91          new PathFragment("tools/test_sharding_compliant"));
92    }

140   /**
141    * A strategy for running the same tests in many processes.
142    */
143   public static enum TestShardingStrategy {
144     EXPLICIT {
145       @Override public int getNumberOfShards(boolean isLocal, int shardCountFromAttr,
146           boolean testShardingCompliant, TestSize testSize) {
147         return Math.max(shardCountFromAttr, 0);
148       }
149     },
150 
151     EXPERIMENTAL_HEURISTIC {
152       @Override public int getNumberOfShards(boolean isLocal, int shardCountFromAttr,
153           boolean testShardingCompliant, TestSize testSize) {
154         if (shardCountFromAttr >= 0) {
155           return shardCountFromAttr;
156         }
157         if (isLocal || !testShardingCompliant) {
158           return 0;
159         }
160         return testSize.getDefaultShards();
161       }
162     },
163 
164     DISABLED {
165       @Override public int getNumberOfShards(boolean isLocal, int shardCountFromAttr,
166           boolean testShardingCompliant, TestSize testSize) {
167         return 0;
168       }
169     };
170 
171     public abstract int getNumberOfShards(boolean isLocal, int shardCountFromAttr,
172         boolean testShardingCompliant, TestSize testSize);
173   }

175   /**
176    * Creates a test action and artifacts for the given rule. The test action will
177    * use the specified executable and runfiles.
178    *
179    * @param targetName the relative path of the target to run
180    * @return ordered list of test artifacts, one per action. These are used to drive
181    *    execution in Skyframe, and by AggregatingTestListener and
182    *    TestResultAnalyzer to keep track of completed and pending test runs.
183    */
184   private TestParams createTestAction(PathFragment targetName, int shards) {
185     BuildConfiguration config = ruleContext.getConfiguration();
186     AnalysisEnvironment env = ruleContext.getAnalysisEnvironment();
187     Root root = config.getTestLogsDirectory();
188 
189     NestedSetBuilder<Artifact> inputsBuilder = NestedSetBuilder.stableOrder();
190     inputsBuilder.addTransitive(
191         NestedSetBuilder.create(Order.STABLE_ORDER, runfilesSupport.getRunfilesMiddleman()));
192     for (TransitiveInfoCollection dep : ruleContext.getPrerequisites("$test_runtime", Mode.HOST)) {
193       inputsBuilder.addTransitive(dep.getProvider(FileProvider.class).getFilesToBuild());
194     }
195     TestTargetProperties testProperties = new TestTargetProperties(
196         ruleContext, executionRequirements);
197 
198     // If the test rule does not provide InstrumentedFilesProvider, there's not much that we can do.
199     final boolean collectCodeCoverage = config.isCodeCoverageEnabled()
200         && instrumentedFiles != null;
201 
202     TestTargetExecutionSettings executionSettings;
203     if (collectCodeCoverage) {
204       // Add instrumented file manifest artifact to the list of inputs. This file will contain
205       // exec paths of all source files that should be included into the code coverage output.
206       Collection<Artifact> metadataFiles =
207           ImmutableList.copyOf(instrumentedFiles.getInstrumentationMetadataFiles());
208       inputsBuilder.addTransitive(NestedSetBuilder.wrap(Order.STABLE_ORDER, metadataFiles));
209       for (TransitiveInfoCollection dep :
210           ruleContext.getPrerequisites(":coverage_support", Mode.HOST)) {
211         inputsBuilder.addTransitive(dep.getProvider(FileProvider.class).getFilesToBuild());
212       }
213       for (TransitiveInfoCollection dep :
214           ruleContext.getPrerequisites(":gcov", Mode.HOST)) {
215         inputsBuilder.addTransitive(dep.getProvider(FileProvider.class).getFilesToBuild());
216       }
217       Artifact instrumentedFileManifest =
218           InstrumentedFileManifestAction.getInstrumentedFileManifest(ruleContext,
219               ImmutableList.copyOf(instrumentedFiles.getInstrumentedFiles()),
220               metadataFiles);
221       executionSettings = new TestTargetExecutionSettings(ruleContext, runfilesSupport,
222           executable, instrumentedFileManifest, shards);
223       inputsBuilder.add(instrumentedFileManifest);
224     } else {
225       executionSettings = new TestTargetExecutionSettings(ruleContext, runfilesSupport,
226           executable, null, shards);
227     }
228 
229     if (config.getRunUnder() != null) {
230       Artifact runUnderExecutable = executionSettings.getRunUnderExecutable();
231       if (runUnderExecutable != null) {
232         inputsBuilder.add(runUnderExecutable);
233       }
234     }
235 
236     int runsPerTest = config.getRunsPerTestForLabel(ruleContext.getLabel());
237 
238     Iterable<Artifact> inputs = inputsBuilder.build();
239     int shardRuns = (shards > 0 ? shards : 1);
240     List<Artifact> results = Lists.newArrayListWithCapacity(runsPerTest * shardRuns);
241     ImmutableList.Builder<Artifact> coverageArtifacts = ImmutableList.builder();
242 
243     for (int run = 0; run < runsPerTest; run++) {
244       // Use a 1-based index for user friendliness.
245       String runSuffix =
246           runsPerTest > 1 ? String.format("_run_%d_of_%d", run + 1, runsPerTest) : "";
247       for (int shard = 0; shard < shardRuns; shard++) {
248         String suffix = (shardRuns > 1 ? String.format("_shard_%d_of_%d", shard + 1, shards) : "")
249             + runSuffix;
250         Artifact testLog = env.getDerivedArtifact(
251             targetName.getChild("test" + suffix + ".log"), root);
252         Artifact cacheStatus = env.getDerivedArtifact(
253             targetName.getChild("test" + suffix + ".cache_status"), root);
254 
255         Artifact coverageArtifact = null;
256         if (collectCodeCoverage) {
257           coverageArtifact =
258               env.getDerivedArtifact(targetName.getChild("coverage" + suffix + ".dat"), root);
259           coverageArtifacts.add(coverageArtifact);
260         }
261 
262         Artifact microCoverageArtifact = null;
263         if (collectCodeCoverage && config.isMicroCoverageEnabled()) {
264           microCoverageArtifact =
265               env.getDerivedArtifact(targetName.getChild("coverage" + suffix + ".micro.dat"), root);
266         }
267 
268         env.registerAction(new TestRunnerAction(
269             ruleContext.getActionOwner(), inputs,
270             testLog, cacheStatus,
271             coverageArtifact, microCoverageArtifact,
272             testProperties, extraEnv, executionSettings,
273             shard, run, config, ruleContext.getWorkspaceName()));
274         results.add(cacheStatus);
275       }
276     }
277     // TODO(bazel-team): Passing the reportGenerator to every TestParams is a bit strange.
278     Artifact reportGenerator = collectCodeCoverage
279         ? ruleContext.getPrerequisiteArtifact(":coverage_report_generator", Mode.HOST) : null;
280     return new TestParams(runsPerTest, shards, TestTimeout.getTestTimeout(ruleContext.getRule()),
281         ruleContext.getRule().getRuleClass(), ImmutableList.copyOf(results),
282         coverageArtifacts.build(), reportGenerator);
283   }
