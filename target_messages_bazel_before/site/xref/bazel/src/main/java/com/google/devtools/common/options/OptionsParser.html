33  /**
34   * A parser for options. Typical use case in a main method:
35   *
36   * <pre>
37   * OptionsParser parser = OptionsParser.newOptionsParser(FooOptions.class, BarOptions.class);
38   * parser.parseAndExitUponError(args);
39   * FooOptions foo = parser.getOptions(FooOptions.class);
40   * BarOptions bar = parser.getOptions(BarOptions.class);
41   * List&lt;String&gt; otherArguments = parser.getResidue();
42   * </pre>
43   *
44   * <p>FooOptions and BarOptions would be options specification classes, derived
45   * from OptionsBase, that contain fields annotated with @Option(...).
46   *
47   * <p>Alternatively, rather than calling
48   * {@link #parseAndExitUponError(OptionPriority, String, String[])},
49   * client code may call {@link #parse(OptionPriority,String,List)}, and handle
50   * parser exceptions usage messages themselves.
51   *
52   * <p>This options parsing implementation has (at least) one design flaw. It
53   * allows both '--foo=baz' and '--foo baz' for all options except void, boolean
54   * and tristate options. For these, the 'baz' in '--foo baz' is not treated as
55   * a parameter to the option, making it is impossible to switch options between
56   * void/boolean/tristate and everything else without breaking backwards
57   * compatibility.
58   *
59   * @see Options a simpler class which you can use if you only have one options
60   * specification class
61   */
62  public class OptionsParser implements OptionsProvider {
63  
64    /**
65     * A cache for the parsed options data. Both keys and values are immutable, so
66     * this is always safe. Only access this field through the {@link
67     * #getOptionsData} method for thread-safety! The cache is very unlikely to
68     * grow to a significant amount of memory, because there's only a fixed set of
69     * options classes on the classpath.
70     */
71    private static final Map<ImmutableList<Class<? extends OptionsBase>>, OptionsData> optionsData =
72        Maps.newHashMap();
73  
74    private static synchronized OptionsData getOptionsData(
75        ImmutableList<Class<? extends OptionsBase>> optionsClasses) {
76      OptionsData result = optionsData.get(optionsClasses);
77      if (result == null) {
78        result = OptionsData.of(optionsClasses);
79        optionsData.put(optionsClasses, result);
80      }
81      return result;
82    }

84    /**
85     * Returns all the annotated fields for the given class, including inherited
86     * ones.
87     */
88    static Collection<Field> getAllAnnotatedFields(Class<? extends OptionsBase> optionsClass) {
89      OptionsData data = getOptionsData(ImmutableList.<Class<? extends OptionsBase>>of(optionsClass));
90      return data.getFieldsForClass(optionsClass);
91    }

116   /**
117    * Canonicalizes a list of options using the given option classes. The
118    * contract is that if the returned set of options is passed to an options
119    * parser with the same options classes, then that will have the same effect
120    * as using the original args (which are passed in here), except for cosmetic
121    * differences.
122    */
123   public static List<String> canonicalize(
124       Collection<Class<? extends OptionsBase>> optionsClasses, List<String> args)
125       throws OptionsParsingException {
126     OptionsParser parser = new OptionsParser(optionsClasses);
127     parser.setAllowResidue(false);
128     parser.parse(args);
129     return parser.impl.asCanonicalizedList();
130   }

166   /**
167    * A convenience function for use in main methods. Parses the command line
168    * parameters, and exits upon error. Also, prints out the usage message
169    * if "--help" appears anywhere within {@code args}.
170    */
171   public void parseAndExitUponError(OptionPriority priority, String source, String[] args) {
172     try {
173       parse(priority, source, Arrays.asList(args));
174     } catch (OptionsParsingException e) {
175       System.err.println("Error parsing command line: " + e.getMessage());
176       System.err.println("Try --help.");
177       System.exit(2);
178     }
179     for (String arg : args) {
180       if (arg.equals("--help")) {
181         System.out.println(describeOptions(Collections.<String, String>emptyMap(),
182                                            HelpVerbosity.LONG));
183         System.exit(0);
184       }
185     }
186   }

360   /**
361    * The level of documentation. Only documented options are output as part of
362    * the help.
363    *
364    * <p>We use 'hidden' so that options that form the protocol between the
365    * client and the server are not logged.
366    */
367   enum DocumentationLevel {
368     DOCUMENTED, UNDOCUMENTED, HIDDEN
369   }

371   /**
372    * Returns a description of all the options this parser can digest.
373    * In addition to {@link Option} annotations, this method also
374    * interprets {@link OptionsUsage} annotations which give an intuitive short
375    * description for the options.
376    *
377    * @param categoryDescriptions a mapping from category names to category
378    *   descriptions.  Options of the same category (see {@link
379    *   Option#category}) will be grouped together, preceded by the description
380    *   of the category.
381    * @param helpVerbosity if {@code long}, the options will be described
382    *   verbosely, including their types, defaults and descriptions.  If {@code
383    *   medium}, the descriptions are omitted, and if {@code short}, the options
384    *   are just enumerated.
385    */
386   public String describeOptions(Map<String, String> categoryDescriptions,
387                                 HelpVerbosity helpVerbosity) {
388     StringBuilder desc = new StringBuilder();
389     if (!impl.getOptionsClasses().isEmpty()) {
390 
391       List<Field> allFields = Lists.newArrayList();
392       for (Class<? extends OptionsBase> optionsClass : impl.getOptionsClasses()) {
393         allFields.addAll(impl.getAnnotatedFieldsFor(optionsClass));
394       }
395       Collections.sort(allFields, OptionsUsage.BY_CATEGORY);
396       String prevCategory = null;
397 
398       for (Field optionField : allFields) {
399         String category = optionField.getAnnotation(Option.class).category();
400         if (!category.equals(prevCategory)) {
401           prevCategory = category;
402           String description = categoryDescriptions.get(category);
403           if (description == null) {
404             description = "Options category '" + category + "'";
405           }
406           if (documentationLevel(category) == DocumentationLevel.DOCUMENTED) {
407             desc.append("\n").append(description).append(":\n");
408           }
409         }
410 
411         if (documentationLevel(prevCategory) == DocumentationLevel.DOCUMENTED) {
412           OptionsUsage.getUsage(optionField, desc, helpVerbosity);
413         }
414       }
415     }
416     return desc.toString().trim();
417   }

419   /**
420    * Returns a string listing the possible flag completion for this command along with the command
421    * completion if any. See {@link OptionsUsage#getCompletion(Field, StringBuilder)} for more
422    * details on the format for the flag completion.
423    */
424   public String getOptionsCompletion() {
425     StringBuilder desc = new StringBuilder();
426 
427     // List all options
428     List<Field> allFields = Lists.newArrayList();
429     for (Class<? extends OptionsBase> optionsClass : impl.getOptionsClasses()) {
430       allFields.addAll(impl.getAnnotatedFieldsFor(optionsClass));
431     }
432     for (Field optionField : allFields) {
433       String category = optionField.getAnnotation(Option.class).category();
434       if (documentationLevel(category) == DocumentationLevel.DOCUMENTED) {
435         OptionsUsage.getCompletion(optionField, desc);
436       }
437     }
438 
439     return desc.toString();
440   }

452   static DocumentationLevel documentationLevel(String category) {
453     if ("undocumented".equals(category)) {
454       return DocumentationLevel.UNDOCUMENTED;
455     } else if ("hidden".equals(category)) {
456       return DocumentationLevel.HIDDEN;
457     } else {
458       return DocumentationLevel.DOCUMENTED;
459     }
460   }

478   /**
479    * Parses {@code args}, using the classes registered with this parser.
480    * {@link #getOptions(Class)} and {@link #getResidue()} return the results.
481    * May be called multiple times; later options override existing ones if they
482    * have equal or higher priority. The source of options is a free-form string
483    * that can be used for debugging. Strings that cannot be parsed as options
484    * accumulates as residue, if this parser allows it.
485    *
486    * @see OptionPriority
487    */
488   public void parse(OptionPriority priority, String source,
489       List<String> args) throws OptionsParsingException {
490     parseWithSourceFunction(priority, Functions.constant(source), args);
491   }
492 
493   /**
494    * Parses {@code args}, using the classes registered with this parser.
495    * {@link #getOptions(Class)} and {@link #getResidue()} return the results. May be called
496    * multiple times; later options override existing ones if they have equal or higher priority.
497    * The source of options is given as a function that maps option names to the source of the
498    * option. Strings that cannot be parsed as options accumulates as* residue, if this parser
499    * allows it.
500    */
501   public void parseWithSourceFunction(OptionPriority priority,
502       Function<? super String, String> sourceFunction, List<String> args)
503       throws OptionsParsingException {
504     Preconditions.checkNotNull(priority);
505     Preconditions.checkArgument(priority != OptionPriority.DEFAULT);
506     residue.addAll(impl.parse(priority, sourceFunction, args));
507     if (!allowResidue && !residue.isEmpty()) {
508       String errorMsg = "Unrecognized arguments: " + Joiner.on(' ').join(residue);
509       throw new OptionsParsingException(errorMsg);
510     }
511   }
