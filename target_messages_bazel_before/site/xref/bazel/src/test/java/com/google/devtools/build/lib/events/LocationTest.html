14  package com.google.devtools.build.lib.events;
15  
16  import static org.junit.Assert.assertEquals;
17  import static org.junit.Assert.assertNull;
18  
19  import com.google.devtools.build.lib.vfs.PathFragment;
20  
21  import org.junit.Test;
22  import org.junit.runner.RunWith;
23  import org.junit.runners.JUnit4;
24  
25  @RunWith(JUnit4.class)
26  public class LocationTest extends EventTestTemplate {
27  
28    @Test
29    public void fromFile() throws Exception {
30      Location location = Location.fromPathFragment(path);
31      assertEquals(path, location.getPath());
32      assertEquals(0, location.getStartOffset());
33      assertEquals(0, location.getEndOffset());
34      assertNull(location.getStartLineAndColumn());
35      assertNull(location.getEndLineAndColumn());
36      assertEquals(path + ":1", location.print());
37    }
38    
39    @Test
40    public void testPrintRelative() throws Exception {
41      Location location = Location.fromPathFragment(path);
42      assertEquals("/path/to/workspace/my/sample/path.txt:1",
43          location.print(new PathFragment("/some/other/path"), new PathFragment("baz")));
44      assertEquals("new/sample/path.txt:1",
45          location.print(new PathFragment("/path/to/workspace/my"), new PathFragment("new")));
46      assertEquals("new/path.txt:1",
47          location.print(new PathFragment("/path/to/workspace/my/sample"), new PathFragment("new")));
48      assertEquals("new:1", location.print(new PathFragment("/path/to/workspace/my/sample/path.txt"),
49          new PathFragment("new")));
50    }
51  }
