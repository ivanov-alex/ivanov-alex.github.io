28  /**
29   * This class tests the functionality of the PathFragment.
30   */
31  @RunWith(JUnit4.class)
32  public class PathFragmentWindowsTest {
33    
34    @Test
35    public void testWindowsSeparator() {
36      assertEquals("bar/baz", new PathFragment("bar\\baz").toString());
37      assertEquals("C:/bar/baz", new PathFragment("c:\\bar\\baz").toString());
38    }
39  
40    @Test
41    public void testIsAbsoluteWindows() {
42      assertTrue(new PathFragment("C:/").isAbsolute());
43      assertTrue(new PathFragment("C:/").isAbsolute());
44      assertTrue(new PathFragment("C:/foo").isAbsolute());
45      assertTrue(new PathFragment("d:/foo/bar").isAbsolute());
46  
47      assertFalse(new PathFragment("*:/").isAbsolute());
48  
49      // C: is not an absolute path, it points to the current active directory on drive C:.
50      assertFalse(new PathFragment("C:").isAbsolute());
51      assertFalse(new PathFragment("C:foo").isAbsolute());
52    }
53  
54    @Test
55    public void testIsAbsoluteWindowsBackslash() {
56      assertTrue(new PathFragment(new File("C:\\blah")).isAbsolute());
57      assertTrue(new PathFragment(new File("C:\\")).isAbsolute());
58      assertTrue(new PathFragment(new File("\\blah")).isAbsolute());
59      assertTrue(new PathFragment(new File("\\")).isAbsolute());
60    }
61  
62    @Test
63    public void testIsNormalizedWindows() {
64      assertTrue(new PathFragment("C:/").isNormalized());
65      assertTrue(new PathFragment("C:/absolute/path").isNormalized());
66      assertFalse(new PathFragment("C:/absolute/./path").isNormalized());
67      assertFalse(new PathFragment("C:/absolute/../path").isNormalized());
68    }
69  
70    @Test
71    public void testRootNodeReturnsRootStringWindows() {
72      PathFragment rootFragment = new PathFragment("C:/");
73      assertEquals("C:/", rootFragment.getPathString());
74    }
75  
76    @Test
77    public void testGetRelativeWindows() {
78      assertEquals("C:/a/b", new PathFragment("C:/a").getRelative("b").getPathString());
79      assertEquals("C:/a/b/c/d", new PathFragment("C:/a/b").getRelative("c/d").getPathString());
80      assertEquals("C:/b", new PathFragment("C:/a").getRelative("C:/b").getPathString());
81      assertEquals("C:/c/d", new PathFragment("C:/a/b").getRelative("C:/c/d").getPathString());
82      assertEquals("C:/b", new PathFragment("a").getRelative("C:/b").getPathString());
83      assertEquals("C:/c/d", new PathFragment("a/b").getRelative("C:/c/d").getPathString());
84    }
85  
86    @Test
87    public void testGetRelativeMixed() {
88      assertEquals("/b", new PathFragment("C:/a").getRelative("/b").getPathString());
89      assertEquals("C:/b", new PathFragment("/a").getRelative("C:/b").getPathString());
90    }
91  
92    @Test
93    public void testGetChildWorks() {
94      PathFragment pf = new PathFragment("../some/path");
95      assertEquals(new PathFragment("../some/path/hi"), pf.getChild("hi"));
96    }

98    // Tests after here test the canonicalization
99    private void assertRegular(String expected, String actual) {
100     assertEquals(expected, new PathFragment(actual).getPathString()); // compare string forms
101     assertEquals(new PathFragment(expected), new PathFragment(actual)); // compare fragment forms
102   }
103 
104   @Test
105   public void testEmptyPathToEmptyPathWindows() {
106     assertRegular("C:/", "C:/");
107   }

109   @Test

111     assertRegular("C:", "C:");
112   }

120   public void testRedundantSlashesWindows() {
121     assertRegular("C:/", "C:///");
122     assertRegular("C:/foo/bar", "C:/foo///bar");
123     assertRegular("C:/foo/bar", "C:////foo//bar");
124   }

126   @Test
127   public void testSimpleNameToSimpleNameWindows() {
128     assertRegular("C:/foo", "C:/foo");
129   }

131   @Test
132   public void testStripsTrailingSlashWindows() {
133     assertRegular("C:/foo/bar", "C:/foo/bar/");

136   @Test
137   public void testGetParentDirectoryWindows() {
138     PathFragment fooBarWizAbs = new PathFragment("C:/foo/bar/wiz");
139     PathFragment fooBarAbs = new PathFragment("C:/foo/bar");
140     PathFragment fooAbs = new PathFragment("C:/foo");
141     PathFragment rootAbs = new PathFragment("C:/");
142     assertEquals(fooBarAbs, fooBarWizAbs.getParentDirectory());
143     assertEquals(fooAbs, fooBarAbs.getParentDirectory());
144     assertEquals(rootAbs, fooAbs.getParentDirectory());
145     assertNull(rootAbs.getParentDirectory());
146 
147     // Note, this is suprising but correct behaviour:
148     assertEquals(fooBarAbs,
149                  new PathFragment("C:/foo/bar/..").getParentDirectory());
150   }
151 
152   @Test
153   public void testSegmentsCountWindows() {
154     assertEquals(1, new PathFragment("C:/foo").segmentCount());
155     assertEquals(0, new PathFragment("C:/").segmentCount());
156   }
157 
158   @Test
159   public void testGetSegmentWindows() {
160     assertEquals("foo", new PathFragment("C:/foo/bar").getSegment(0));
161     assertEquals("bar", new PathFragment("C:/foo/bar").getSegment(1));
162     assertEquals("foo", new PathFragment("C:/foo/").getSegment(0));
163     assertEquals("foo", new PathFragment("C:/foo").getSegment(0));
164   }
165 
166   @Test
167   public void testBasenameWindows() throws Exception {
168     assertEquals("bar", new PathFragment("C:/foo/bar").getBaseName());
169     assertEquals("foo", new PathFragment("C:/foo").getBaseName());
170     // Never return the drive name as a basename.
171     assertThat(new PathFragment("C:/").getBaseName()).isEmpty();
172   }

174   private static void assertPath(String expected, PathFragment actual) {
175     assertEquals(expected, actual.getPathString());
176   }
177 
178   @Test
179   public void testReplaceNameWindows() throws Exception {
180     assertPath("C:/foo/baz", new PathFragment("C:/foo/bar").replaceName("baz"));
181     assertNull(new PathFragment("C:/").replaceName("baz"));
182   }
183 
184   @Test
185   public void testStartsWithWindows() {
186     assertTrue(new PathFragment("C:/foo/bar").startsWith(new PathFragment("C:/foo")));
187     assertTrue(new PathFragment("C:/foo/bar").startsWith(new PathFragment("C:/")));
188     assertTrue(new PathFragment("C:foo/bar").startsWith(new PathFragment("C:")));
189     assertTrue(new PathFragment("C:/").startsWith(new PathFragment("C:/")));
190     assertTrue(new PathFragment("C:").startsWith(new PathFragment("C:")));
191 
192     // The first path is absolute, the second is not.
193     assertFalse(new PathFragment("C:/foo/bar").startsWith(new PathFragment("C:")));
194     assertFalse(new PathFragment("C:/").startsWith(new PathFragment("C:")));
195   }
196 
197   @Test
198   public void testEndsWithWindows() {
199     assertTrue(new PathFragment("C:/foo/bar").endsWith(new PathFragment("bar")));
200     assertTrue(new PathFragment("C:/foo/bar").endsWith(new PathFragment("foo/bar")));
201     assertTrue(new PathFragment("C:/foo/bar").endsWith(new PathFragment("C:/foo/bar")));
202     assertTrue(new PathFragment("C:/").endsWith(new PathFragment("C:/")));
203   }

216     assertEquals(new PathFragment("C:/b"), new PathFragment("C:/a/../b").normalize());
