38  /**
39   * Tests {@link UnixGlob} recursive globs.
40   */
41  @RunWith(JUnit4.class)
42  public class RecursiveGlobTest {
43  
44    private Path tmpPath;
45    private FileSystem fileSystem;
46    
47    @Before
48    public void setUp() throws Exception {
49      fileSystem = new InMemoryFileSystem(BlazeClock.instance());
50      tmpPath = fileSystem.getPath("/rglobtmp");
51      for (String dir : ImmutableList.of("foo/bar/wiz",
52                           "foo/baz/wiz",
53                           "foo/baz/quip/wiz",
54                           "food/baz/wiz",
55                           "fool/baz/wiz")) {
56        FileSystemUtils.createDirectoryAndParents(tmpPath.getRelative(dir));
57      }
58      FileSystemUtils.createEmptyFile(tmpPath.getRelative("foo/bar/wiz/file"));
59    }
60  
61    @Test
62    public void testDoubleStar() throws Exception {
63      assertGlobMatches("**", ".", "foo", "foo/bar", "foo/bar/wiz", "foo/baz", "foo/baz/quip",
64                        "foo/baz/quip/wiz", "foo/baz/wiz", "foo/bar/wiz/file", "food", "food/baz",
65                        "food/baz/wiz", "fool", "fool/baz", "fool/baz/wiz");
66    }
67  
68    @Test
69    public void testDoubleDoubleStar() throws Exception {
70      assertGlobMatches("**/**", ".", "foo", "foo/bar", "foo/bar/wiz", "foo/baz", "foo/baz/quip",
71                        "foo/baz/quip/wiz", "foo/baz/wiz", "foo/bar/wiz/file", "food", "food/baz",
72                        "food/baz/wiz", "fool", "fool/baz", "fool/baz/wiz");
73    }
74  
75    @Test
76    public void testDirectoryWithDoubleStar() throws Exception {
77      assertGlobMatches("foo/**", "foo", "foo/bar", "foo/bar/wiz", "foo/baz", "foo/baz/quip",
78                        "foo/baz/quip/wiz", "foo/baz/wiz", "foo/bar/wiz/file");
79    }
80  
81    @Test
82    public void testIllegalPatterns() throws Exception {
83      for (String prefix : Lists.newArrayList("", "*/", "**/", "ba/")) {
84        String suffix = ("/" + prefix).substring(0, prefix.length());
85        for (String pattern : Lists.newArrayList("**fo", "fo**", "**fo**", "fo**fo", "fo**fo**fo")) {
86          assertIllegalWildcard(prefix + pattern);
87          assertIllegalWildcard(pattern + suffix);
88          assertIllegalWildcard("foo", pattern + suffix);
89        }
90      }
91    }

93    @Test
94    public void testDoubleStarPatternWithNamedChild() throws Exception {
95      assertGlobMatches("**/bar", "foo/bar");
96    }

98    @Test
99    public void testDoubleStarPatternWithChildGlob() throws Exception {
100     assertGlobMatches("**/ba*",
101         "foo/bar", "foo/baz", "food/baz", "fool/baz");

107   }

115   public void testDoubleStarGlobWithNonExistentBase() throws Exception {

118         .globInterruptible();

127   @Test
128   public void testSingleFileExclude() throws Exception {
129     assertGlobWithExcludeMatches("**", "food", ".", "foo", "foo/bar", "foo/bar/wiz", "foo/baz",
130                                  "foo/baz/quip", "foo/baz/quip/wiz", "foo/baz/wiz",
131                                  "foo/bar/wiz/file", "food/baz", "food/baz/wiz", "fool", "fool/baz",
132                                  "fool/baz/wiz");
133   }

135   @Test
136   public void testSingleFileExcludeForDirectoryWithChildGlob()
137       throws Exception {
138     assertGlobWithExcludeMatches("foo/**", "foo", "foo/bar", "foo/bar/wiz", "foo/baz",
139                                  "foo/baz/quip", "foo/baz/quip/wiz", "foo/baz/wiz",
140                                  "foo/bar/wiz/file");
141   }

143   @Test
144   public void testGlobExcludeForDirectoryWithChildGlob()
145       throws Exception {
146     assertGlobWithExcludeMatches("foo/**", "foo/*", "foo", "foo/bar/wiz", "foo/baz/quip",
147                                  "foo/baz/quip/wiz", "foo/baz/wiz", "foo/bar/wiz/file");
148   }

150   @Test
151   public void testExcludeAll() throws Exception {
152     assertGlobWithExcludesMatches(Lists.newArrayList("**"),
153                                   Lists.newArrayList("*", "*/*", "*/*/*", "*/*/*/*"), ".");
154   }
155 
156   @Test
157   public void testManualGlobExcludeForDirectoryWithChildGlob()
158       throws Exception {
159     assertGlobWithExcludesMatches(Lists.newArrayList("foo/**"),
160                                   Lists.newArrayList("foo", "foo/*", "foo/*/*", "foo/*/*/*"));
161   }

163   private void assertGlobMatches(String pattern, String... expecteds)
164       throws Exception {
165     assertGlobWithExcludesMatches(
166         Collections.singleton(pattern), Collections.<String>emptyList(),
167         expecteds);
168   }
169 
170   private void assertGlobWithExcludeMatches(String pattern, String exclude,
171                                             String... expecteds)
172       throws Exception {
173     assertGlobWithExcludesMatches(
174         Collections.singleton(pattern), Collections.singleton(exclude),
175         expecteds);
176   }
177 
178   private void assertGlobWithExcludesMatches(Collection<String> pattern,
179                                              Collection<String> excludes,
180                                              String... expecteds) throws Exception {
181     assertSameContents(resolvePaths(expecteds),
182         new UnixGlob.Builder(tmpPath)
183             .addPatterns(pattern)
184             .addExcludes(excludes)
185             .globInterruptible());
186   }
187 
188   private Set<Path> resolvePaths(String... relativePaths) {
189     Set<Path> expectedFiles = new HashSet<>();
190     for (String expected : relativePaths) {
191       Path file = expected.equals(".")
192           ? tmpPath
193           : tmpPath.getRelative(expected);
194       expectedFiles.add(file);
195     }
196     return expectedFiles;
197   }
198 
199   /**
200    * Tests that a recursive glob returns files in sorted order.
201    */
202   @Test
203   public void testGlobEntriesAreSorted() throws Exception {
204     List<Path> globResult = new UnixGlob.Builder(tmpPath)
205         .addPattern("**")
206         .setExcludeDirectories(false)
207         .globInterruptible();
208 
209     assertThat(Ordering.natural().sortedCopy(globResult)).containsExactlyElementsIn(globResult)
210         .inOrder();
211   }
212 
213   private void assertIllegalWildcard(String pattern, String... excludePatterns)
214       throws Exception {
215     try {
216       new UnixGlob.Builder(tmpPath)
217           .addPattern(pattern)
218           .addExcludes(excludePatterns)
219           .globInterruptible();
220       fail();
221     } catch (IllegalArgumentException e) {
222       MoreAsserts.assertContainsRegex("recursive wildcard must be its own segment", e.getMessage());
223     }
224   }
