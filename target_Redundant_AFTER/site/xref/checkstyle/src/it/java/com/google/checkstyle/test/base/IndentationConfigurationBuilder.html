3   import java.io.BufferedReader;
4   import java.io.File;
5   import java.io.FileReader;
6   import java.io.IOException;
7   import java.util.ArrayList;
8   import java.util.Arrays;
9   import java.util.List;
10  import java.util.regex.Matcher;
11  import java.util.regex.Pattern;
12  
13  import com.puppycrawl.tools.checkstyle.api.CheckstyleException;
14  import com.puppycrawl.tools.checkstyle.Utils;
15  
16  public class IndentationConfigurationBuilder extends ConfigurationBuilder
17  {
18      private static final int TAB_WIDTH = 4;
19  
20      public IndentationConfigurationBuilder(File aROOT)
21          throws CheckstyleException, IOException
22      {
23          super(aROOT);
24      }
25  
26      @Override
27      public Integer[] getLinesWithWarn(String aFileName) throws IOException {
28          return getLinesWithWarnAndCheckComments(aFileName, TAB_WIDTH);
29      }
30  
31      private static final Pattern NONEMPTY_LINE_REGEX =
32                      Pattern.compile(".*?\\S+.*?");
33  
34      private static final Pattern LINE_WITH_COMMENT_REGEX =
35                      Pattern.compile(".*?\\S+.*?(//indent:(\\d+) exp:((>=\\d+)|(\\d+(,\\d+)*?))( warn)?)");
36  
37      private static final Pattern GET_INDENT_FROM_COMMENT_REGEX =
38                      Pattern.compile("//indent:(\\d+).*?");
39  
40      private static final Pattern MULTILEVEL_COMMENT_REGEX =
41                      Pattern.compile("//indent:\\d+ exp:(\\d+(,\\d+)+?)( warn)?");
42  
43      private static final Pattern SINGLELEVEL_COMMENT_REGEX =
44                      Pattern.compile("//indent:\\d+ exp:(\\d+)( warn)?");
45  
46      private static final Pattern NONSTRICT_LEVEL_COMMENT_REGEX =
47                      Pattern.compile("//indent:\\d+ exp:>=(\\d+)( warn)?");
48  
49      protected static Integer[] getLinesWithWarnAndCheckComments(String aFileName,
50                      final int tabWidth)
51                      throws IOException
52      {
53          int lineNumber = 1;
54          List<Integer> result = new ArrayList<>();
55          try (BufferedReader br = new BufferedReader(new FileReader(aFileName))) {
56              for (String line = br.readLine(); line != null; line = br.readLine()) {
57                  Matcher match = LINE_WITH_COMMENT_REGEX.matcher(line);
58                  if (match.matches()) {
59                      final String comment = match.group(1);
60                      final int indentInComment = getIndentFromComment(comment);
61                      final int actualIndent = getLineStart(line, tabWidth);
62  
63                      if (actualIndent != indentInComment) {
64                          throw new RuntimeException(String.format(
65                                          "File \"%1$s\" has incorrect indentation in comment."
66                                                          + "Line %2$d: comment:%3$d, actual:%4$d.",
67                                          aFileName,
68                                          lineNumber,
69                                          indentInComment,
70                                          actualIndent));
71                      }
72  
73                      if (isWarnComment(comment)) {
74                          result.add(lineNumber);
75                      }
76  
77                      if (!isCommentConsistent(comment)) {
78                          throw new RuntimeException(String.format(
79                                          "File \"%1$s\" has inconsistent comment on line %2$d",
80                                          aFileName,
81                                          lineNumber));
82                      }
83                  }
84                  else if (NONEMPTY_LINE_REGEX.matcher(line).matches()) {
85                      throw new RuntimeException(String.format(
86                                      "File \"%1$s\" has no indentation comment or its format "
87                                                      + "malformed. Error on line: %2$d(%3$s)",
88                                      aFileName,
89                                      lineNumber,
90                                      line));
91                  }
92                  lineNumber++;
93              }
94          }
95          return result.toArray(new Integer[result.size()]);
96      }
97  
98      private static int getIndentFromComment(String comment)
99      {
100         final Matcher match = GET_INDENT_FROM_COMMENT_REGEX.matcher(comment);
101         match.matches();
102         return Integer.parseInt(match.group(1));
103     }
104 
105     private static boolean isWarnComment(String comment)
106     {
107         return comment.endsWith(" warn");
108     }
109 
110     private static boolean isCommentConsistent(String comment)
111     {
112         final int indentInComment = getIndentFromComment(comment);
113         final boolean isWarnComment = isWarnComment(comment);
114 
115         Matcher match;
116 
117         match = MULTILEVEL_COMMENT_REGEX.matcher(comment);
118         if (match.matches()) {
119             final String[] levels = match.group(1).split(",");
120             final String indentInCommentStr = String.valueOf(indentInComment);
121             final boolean containsActualLevel =
122                             Arrays.asList(levels).contains(indentInCommentStr);
123 
124             return containsActualLevel && !isWarnComment
125                     || !containsActualLevel && isWarnComment;
126         }
127 
128         match = SINGLELEVEL_COMMENT_REGEX.matcher(comment);
129         if (match.matches()) {
130             final int expectedLevel = Integer.parseInt(match.group(1));
131 
132             return (expectedLevel == indentInComment) && !isWarnComment
133                     || (expectedLevel != indentInComment) && isWarnComment;
134         }
135 
136         match = NONSTRICT_LEVEL_COMMENT_REGEX.matcher(comment);
137         if (match.matches()) {
138             final int expectedMinimalIndent = Integer.parseInt(match.group(1));
139 
140             return (indentInComment >= expectedMinimalIndent) && !isWarnComment
141                     || (indentInComment < expectedMinimalIndent) && isWarnComment;
142         }
143 
144         throw new IllegalArgumentException();
145     }
146 
147     private static int getLineStart(String line, final int tabWidth)
148     {
149         for (int index = 0; index < line.length(); ++index) {
150             if (!Character.isWhitespace(line.charAt(index))) {
151                 return Utils.lengthExpandedTabs(line, index, tabWidth);
152             }
153         }
154         return 0;
155     }
