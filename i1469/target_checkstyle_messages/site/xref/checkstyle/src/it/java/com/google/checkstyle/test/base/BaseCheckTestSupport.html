55      protected Checker createChecker(Configuration aCheckConfig)
56          throws Exception
57      {
58          final DefaultConfiguration dc = createCheckerConfig(aCheckConfig);
59          final Checker checker = new Checker();
60          // make sure the tests always run with english error messages
61          // so the tests don't fail in supported locales like german
62          final Locale locale = Locale.ENGLISH;
63          checker.setLocaleCountry(locale.getCountry());
64          checker.setLocaleLanguage(locale.getLanguage());
65          checker.setModuleClassLoader(Thread.currentThread().getContextClassLoader());
66          checker.configure(dc);
67          checker.addListener(new BriefLogger(stream));
68          return checker;
69      }

71      protected DefaultConfiguration createCheckerConfig(Configuration aConfig)
72      {
73          final DefaultConfiguration dc =
74              new DefaultConfiguration("configuration");
75          final DefaultConfiguration twConf = createCheckConfig(TreeWalker.class);
76          // make sure that the tests always run with this charset
77          dc.addAttribute("charset", "iso-8859-1");
78          dc.addChild(twConf);
79          twConf.addChild(aConfig);
80          return dc;
81      }

118     protected void verify(Checker aC,
119                           File[] aProcessedFiles,
120                           String aMessageFileName,
121                           String[] aExpected,
122                           Integer... aWarnsExpected)
123         throws Exception
124     {
125         stream.flush();
126         final List<File> theFiles = Lists.newArrayList();
127         Collections.addAll(theFiles, aProcessedFiles);
128         final int errs = aC.process(theFiles);
129 
130         // process each of the lines
131         final ByteArrayInputStream bais =
132             new ByteArrayInputStream(baos.toByteArray());
133         final LineNumberReader lnr =
134             new LineNumberReader(new InputStreamReader(bais, StandardCharsets.UTF_8));
135 
136         for (int i = 0; i < aExpected.length; i++) {
137             final String expected = aMessageFileName + ":" + aExpected[i];
138             String actual = lnr.readLine();
139             assertEquals("error message " + i, expected, actual);
140             String parseInt = removeDeviceFromPathOnWindows(actual);
141             parseInt = parseInt.substring(parseInt.indexOf(':') + 1);
142             parseInt = parseInt.substring(0, parseInt.indexOf(':'));
143             int lineNumber = Integer.parseInt(parseInt);
144 			Integer integer = Arrays.asList(aWarnsExpected).contains(lineNumber) ? lineNumber : 0;
145             assertEquals("error message " + i, (long) integer, lineNumber);
146         }
147 
148         assertEquals("unexpected output: " + lnr.readLine(),
149                      aExpected.length, errs);
150         aC.destroy();
151     }

153     /**
154      * Gets the check message 'as is' from appropriate 'messages.properties'
155      * file.
156      *
157      * @param messageKey
158      *            the key of message in 'messages.properties' file.
159      */
160     public String getCheckMessage(Class<? extends AbstractViolationReporter> aClass,
161             String messageKey)
162     {
163         Properties pr = new Properties();
164         try {
165             pr.load(aClass.getResourceAsStream("messages.properties"));
166         }
167         catch (IOException e) {
168             return null;
169         }
170         return pr.getProperty(messageKey);
171     }

173     /**
174      * Gets the check message 'as is' from appropriate 'messages.properties' file.
175      * @param messageKey the key of message in 'messages.properties' file.
176      * @param arguments the arguments of message in 'messages.properties' file.
177      */
178     public String getCheckMessage(Class<? extends AbstractViolationReporter> aClass,
179             String messageKey, Object ... arguments) {
180         return format(getCheckMessage(aClass, messageKey), arguments);
181     }
182 
183     /**
184      * Gets the check message 'as is' from appropriate 'messages.properties' file.
185      * @param messageKey the key of message in 'messages.properties' file.
186      * @param arguments the arguments of message in 'messages.properties' file.
187      */
188    public String getCheckMessage(Map<String, String> messages, String messageKey, Object ... arguments)
189    {
190        for (Map.Entry<String, String> entry : messages.entrySet()) {
191            if (messageKey.equals(entry.getKey())) {
192                return format(entry.getValue(), arguments);
193            }
194        }
195        return null;
196    }

198    private static String removeDeviceFromPathOnWindows(String path) {
199        String os = System.getProperty("os.name", "Unix");
200        if (os.startsWith("Windows")) {
201            return path.substring(path.indexOf(':') + 1);
202        }
203        return path;
204    }
