51  /**
52   * Displays information about a parse tree.
53   * The user can change the file that is parsed and displayed
54   * through a JFileChooser.
55   *
56   * @author Lars KÃ¼hne
57   */
58  public class ParseTreeInfoPanel extends JPanel {
59      /** For Serialisation that will never happen. */
60      private static final long serialVersionUID = -4243405131202059043L;
61  
62      private final transient ParseTreeModel parseTreeModel;
63      private final JTextArea textArea;
64      private File lastDirectory;
65      private File currentFile;
66      private final ReloadAction reloadAction;
67      private final List<Integer>   linesToPosition  = new ArrayList<>();
68  
69      /**
70       * Create a new ParseTreeInfoPanel instance.
71       */
72      public ParseTreeInfoPanel() {
73          setLayout(new BorderLayout());
74  
75          parseTreeModel = new ParseTreeModel(null);
76          final JTreeTable treeTable = new JTreeTable(parseTreeModel);
77          final JScrollPane sp = new JScrollPane(treeTable);
78          add(sp, BorderLayout.PAGE_START);
79  
80          final JButton fileSelectionButton =
81              new JButton(new FileSelectionAction());
82  
83          reloadAction = new ReloadAction();
84          reloadAction.setEnabled(false);
85          final JButton reloadButton = new JButton(reloadAction);
86  
87          textArea = new JTextArea(20, 15);
88          textArea.setEditable(false);
89          treeTable.setEditor(textArea);
90          treeTable.setLinePositionMap(linesToPosition);
91  
92          final JScrollPane sp2 = new JScrollPane(textArea);
93          add(sp2, BorderLayout.CENTER);
94  
95          final JPanel p = new JPanel(new GridLayout(1, 2));
96          add(p, BorderLayout.PAGE_END);
97          p.add(fileSelectionButton);
98          p.add(reloadButton);
99  
100         try {
101             new FileDrop(sp, new FileDropListener(sp));
102         }
103         catch (final TooManyListenersException ignored) {
104             showErrorDialog(null, "Cannot initialize Drag and Drop support");
105         }
106 
107     }

109     public void openAst(DetailAST parseTree) {
110         parseTreeModel.setParseTree(parseTree);
111         reloadAction.setEnabled(true);
112 
113         // clear for each new file
114         clearLinesToPosition();
115         // starts line counting at 1
116         addLineToPosition(0);
117         // insert the contents of the file to the text area
118 
119         // clean the text area before inserting the lines of the new file
120         if (!textArea.getText().isEmpty()) {
121             textArea.replaceRange("", 0, textArea.getText().length());
122         }
123 
124         // move back to the top of the file
125         textArea.moveCaretPosition(0);
126     }

128     public void openFile(File file, final Component parent) {
129         if (file != null) {
130             try {
131                 Main.getFrame().setTitle("Checkstyle : " + file.getName());
132                 final FileText text = new FileText(file.getAbsoluteFile(),
133                                                    getEncoding());
134                 final DetailAST parseTree = parseFile(text);
135                 parseTreeModel.setParseTree(parseTree);
136                 currentFile = file;
137                 lastDirectory = file.getParentFile();
138                 reloadAction.setEnabled(true);
139 
140                 final String[] sourceLines = text.toLinesArray();
141 
142                 // clear for each new file
143                 clearLinesToPosition();
144                 // starts line counting at 1
145                 addLineToPosition(0);
146                 // insert the contents of the file to the text area
147                 for (String element : sourceLines) {
148                     addLineToPosition(textArea.getText().length());
149                     textArea.append(element + System.lineSeparator());
150                 }
151 
152                 //clean the text area before inserting the lines of the new file
153                 if (!textArea.getText().isEmpty()) {
154                     textArea.replaceRange("", 0, textArea.getText()
155                             .length());
156                 }
157 
158                 // insert the contents of the file to the text area
159                 for (final String element : sourceLines) {
160                     textArea.append(element + System.lineSeparator());
161                 }
162 
163                 // move back to the top of the file
164                 textArea.moveCaretPosition(0);
165             }
166             catch (final IOException | ANTLRException ex) {
167                 showErrorDialog(
168                         parent,
169                         "Could not parse" + file + ": " + ex.getMessage());
170             }
171         }
172     }

174     /**
175      * Parses a file and returns the parse tree.
176      * @param text the file to parse
177      * @return the root node of the parse tree
178      * @throws ANTLRException if the file is not a Java source
179      */
180     public static DetailAST parseFile(FileText text)
181         throws ANTLRException {
182         final FileContents contents = new FileContents(text);
183         return TreeWalker.parse(contents);
184     }

186     /**
187      * Returns the configured file encoding.
188      * This can be set using the {@code file.encoding} system property.
189      * It defaults to UTF-8.
190      * @return the configured file encoding

192     private static String getEncoding() {
193         return System.getProperty("file.encoding", "UTF-8");
194     }

196     private static void showErrorDialog(final Component parent, final String msg) {
197         final Runnable showError = new FrameShower(parent, msg);
198         SwingUtilities.invokeLater(showError);
199     }

201     void addLineToPosition(int value) {
202         linesToPosition.add(value);

204 
205     void clearLinesToPosition() {
206         linesToPosition.clear();
207     }

212     private static class FrameShower implements Runnable {

240     private static class JavaFileFilter extends FileFilter {
241         @Override
242         public boolean accept(File file) {
243             if (file == null) {
244                 return false;
245             }
246             return file.isDirectory() || file.getName().endsWith(".java");
247         }
248 
249         @Override
250         public String getDescription() {
251             return "Java Source Code";
252         }
253     }

255     private class FileSelectionAction extends AbstractAction {
256         /**
257          *
258          */
259         private static final long serialVersionUID = -1926935338069418119L;
260 
261         FileSelectionAction() {
262             super("Select Java File");
263             putValue(Action.MNEMONIC_KEY, KeyEvent.VK_S);
264         }
265 
266         @Override
267         public void actionPerformed(ActionEvent e) {
268             final JFileChooser fc = new JFileChooser(lastDirectory);
269             final FileFilter filter = new JavaFileFilter();
270             fc.setFileFilter(filter);
271             final Component parent =
272                 SwingUtilities.getRoot(ParseTreeInfoPanel.this);
273             fc.showDialog(parent, "Open");
274             final File file = fc.getSelectedFile();
275             openFile(file, parent);
276 
277         }
278     }

285 
286         ReloadAction() {
287             super("Reload Java File");
288             putValue(Action.MNEMONIC_KEY, KeyEvent.VK_R);
289         }
290 
291         @Override
292         public void actionPerformed(ActionEvent e) {
293             final Component parent =
294                 SwingUtilities.getRoot(ParseTreeInfoPanel.this);
295             openFile(currentFile, parent);
296         }
297     }

299     private class FileDropListener implements FileDrop.Listener {
300         private final JScrollPane scrollPane;
301 
302         FileDropListener(JScrollPane scrollPane) {
303             this.scrollPane = scrollPane;
304         }
305 
306         @Override
307         public void filesDropped(File... files) {
308             if (files != null && files.length > 0) {
309                 final File file = files[0];
310                 openFile(file, scrollPane);
311             }
312         }
