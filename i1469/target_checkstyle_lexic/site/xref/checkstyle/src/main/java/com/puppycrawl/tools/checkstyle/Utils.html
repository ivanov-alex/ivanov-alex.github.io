48      /** Maps from a token name to value */
49      private static final ImmutableMap<String, Integer> TOKEN_NAME_TO_VALUE;
50      /** Maps from a token value to name */
51      private static final String[] TOKEN_VALUE_TO_NAME;
52  
53      /** Array of all token IDs */
54      private static final int[] TOKEN_IDS;
55  
56      /** Prefix for exception when getting token by given id. */
57      private static final String TOKEN_ID_EXCEPTION_PREFIX = "given id ";
58  
59      /** Prefix for exception when getting token by given name. */
60      private static final String TOKEN_NAME_EXCEPTION_PREFIX = "given name ";
61  
62      // initialise the constants
63      static {
64          final ImmutableMap.Builder<String, Integer> builder =
65                  ImmutableMap.builder();
66          final Field[] fields = TokenTypes.class.getDeclaredFields();
67          String[] tempTokenValueToName = ArrayUtils.EMPTY_STRING_ARRAY;
68          for (final Field field : fields) {
69              // Only process the int declarations.
70              if (field.getType() != Integer.TYPE) {
71                  continue;
72              }
73  
74              final String name = field.getName();
75              final int tokenValue = getIntFromField(field, name);
76              builder.put(name, tokenValue);
77              if (tokenValue > tempTokenValueToName.length - 1) {
78                  final String[] temp = new String[tokenValue + 1];
79                  System.arraycopy(tempTokenValueToName, 0,
80                          temp, 0, tempTokenValueToName.length);
81                  tempTokenValueToName = temp;
82              }
83              tempTokenValueToName[tokenValue] = name;
84          }
85  
86          TOKEN_NAME_TO_VALUE = builder.build();
87          TOKEN_VALUE_TO_NAME = tempTokenValueToName;
88          final ImmutableCollection<Integer> values = TOKEN_NAME_TO_VALUE.values();
89          final Integer[] ids = values.toArray(new Integer[values.size()]);
90          TOKEN_IDS = ArrayUtils.toPrimitive(ids);
91      }
92  
93      /** Stop instances being created **/
94      private Utils() {
95      }

107     /**
108      * Returns whether the file extension matches what we are meant to
109      * process.
110      * @param file the file to be checked.
111      * @param fileExtensions files extensions, empty property in config makes it matches to all.
112      * @return whether there is a match.
113      */
114     public static boolean fileExtensionMatches(File file, String... fileExtensions) {
115         boolean result = false;
116         if (fileExtensions == null || fileExtensions.length == 0) {
117             result = true;
118         }
119         else {
120             // normalize extensions so all of them have a leading dot
121             final String[] withDotExtensions = new String[fileExtensions.length];
122             for (int i = 0; i < fileExtensions.length; i++) {
123                 final String extension = fileExtensions[i];
124                 if (startsWithChar(extension, '.')) {
125                     withDotExtensions[i] = extension;
126                 }
127                 else {
128                     withDotExtensions[i] = "." + extension;
129                 }
130             }
131 
132             final String fileName = file.getName();
133             for (final String fileExtension : withDotExtensions) {
134                 if (fileName.endsWith(fileExtension)) {
135                     result = true;
136                 }
137             }
138         }
139 
140         return result;
141     }

143     /**
144      * Returns whether the specified string contains only whitespace up to the
145      * specified index.
146      *
147      * @param index index to check up to
148      * @param line the line to check
149      * @return whether there is only whitespace
150      */
151     public static boolean whitespaceBefore(int index, String line) {
152         for (int i = 0; i < index; i++) {
153             if (!Character.isWhitespace(line.charAt(i))) {
154                 return false;
155             }
156         }
157         return true;
158     }
159 
160     /**
161      * Returns the length of a string ignoring all trailing whitespace. It is a
162      * pity that there is not a trim() like method that only removed the
163      * trailing whitespace.
164      * @param line the string to process
165      * @return the length of the string ignoring all trailing whitespace
166      **/
167     public static int lengthMinusTrailingWhitespace(String line) {
168         int len = line.length();
169         for (int i = len - 1; i >= 0; i--) {
170             if (!Character.isWhitespace(line.charAt(i))) {
171                 break;
172             }
173             len--;
174         }
175         return len;
176     }

178     /**
179      * Returns the length of a String prefix with tabs expanded.
180      * Each tab is counted as the number of characters is takes to
181      * jump to the next tab stop.
182      * @param inputString the input String
183      * @param toIdx index in string (exclusive) where the calculation stops
184      * @param tabWidth the distance between tab stop position.
185      * @return the length of string.substring(0, toIdx) with tabs expanded.
186      */
187     public static int lengthExpandedTabs(String inputString,
188                                          int toIdx,
189                                          int tabWidth) {
190         int len = 0;
191         for (int idx = 0; idx < toIdx; idx++) {
192             if (inputString.charAt(idx) == '\t') {
193                 len = (len / tabWidth + 1) * tabWidth;
194             }
195             else {
196                 len++;
197             }
198         }
199         return len;
200     }

202     /**
203      * Validates whether passed string is a valid pattern or not.
204      * @param pattern
205      *        string to validate
206      * @return true if the pattern is valid false otherwise
207      */
208     public static boolean isPatternValid(String pattern) {
209         try {
210             Pattern.compile(pattern);
211         }
212         catch (final PatternSyntaxException ignored) {
213             return false;
214         }
215         return true;
216     }

218     /**
219      * Helper method to create a regular expression.
220      * @param pattern the pattern to match
221      * @return a created regexp object
222      * @throws ConversionException if unable to create Pattern object.
223      **/
224     public static Pattern createPattern(String pattern) {
225         try {
226             return Pattern.compile(pattern);
227         }
228         catch (final PatternSyntaxException e) {
229             throw new ConversionException(
230                 "Failed to initialise regular expression " + pattern, e);
231         }
232     }

238     public static String baseClassname(String type) {
239         final int i = type.lastIndexOf('.');

249     /**
250      * Constructs a normalized relative path between base directory and a given path.
251      * @param baseDirectory the base path to which given path is relativized
252      * @param path the path to relativize against base directory
253      * @return the relative normalized path between base directory and path or path if base
254      * directory is null
255      */
256     public static String relativizeAndNormalizePath(final String baseDirectory, final String path) {
257         if (baseDirectory == null) {
258             return path;
259         }
260         final Path pathAbsolute = Paths.get(path).normalize();
261         final Path pathBase = Paths.get(baseDirectory).normalize();
262         return pathBase.relativize(pathAbsolute).toString();
263     }

265     /**
266      * Tests if this string starts with the specified prefix.
267      * <p>
268      * It is faster version of {@link String#startsWith(String)} optimized for one-character
269      * prefixes at the expense of some readability. Suggested by SimplifyStartsWith PMD rule:
270      * http://pmd.sourceforge.net/pmd-5.3.1/pmd-java/rules/java/optimizations.html#SimplifyStartsWith
271      * </p>
272      *
273      * @param value the {@code String} to check
274      * @param prefix the prefix to find
275      * @return {@code true} if the {@code char} is a prefix of the given
276      * {@code String}; {@code false} otherwise.
277      */
278     public static boolean startsWithChar(String value, char prefix) {
279         return !value.isEmpty() && value.charAt(0) == prefix;
280     }
281 
282     /**
283      * Tests if this string ends with the specified suffix.
284      * <p>
285      * It is faster version of {@link String#endsWith(String)} optimized for one-character
286      * suffixes at the expense of some readability. Suggested by SimplifyStartsWith PMD rule:
287      * http://pmd.sourceforge.net/pmd-5.3.1/pmd-java/rules/java/optimizations.html#SimplifyStartsWith
288      * </p>
289      *
290      * @param value the {@code String} to check
291      * @param suffix the suffix to find
292      * @return {@code true} if the {@code char} is a suffix of the given
293      * {@code String}; {@code false} otherwise.
294      */
295     public static boolean endsWithChar(String value, char suffix) {
296         return !value.isEmpty() && value.charAt(value.length() - 1) == suffix;
297     }

299     /**
300      * Returns the name of a token for a given ID.
301      * @param iD the ID of the token name to get
302      * @return a token name
303      */
304     public static String getTokenName(int iD) {
305         if (iD > TOKEN_VALUE_TO_NAME.length - 1) {
306             throw new IllegalArgumentException(TOKEN_ID_EXCEPTION_PREFIX + iD);
307         }
308         final String name = TOKEN_VALUE_TO_NAME[iD];
309         if (name == null) {
310             throw new IllegalArgumentException(TOKEN_ID_EXCEPTION_PREFIX + iD);
311         }
312         return name;
313     }

315     /**
316      * Returns the ID of a token for a given name.
317      * @param name the name of the token ID to get
318      * @return a token ID
319      */
320     public static int getTokenId(String name) {
321         final Integer id = TOKEN_NAME_TO_VALUE.get(name);
322         if (id == null) {
323             throw new IllegalArgumentException(TOKEN_NAME_EXCEPTION_PREFIX + name);
324         }
325         return id;
326     }

328     /**
329      * Returns the short description of a token for a given name.
330      * @param name the name of the token ID to get
331      * @return a short description
332      */
333     public static String getShortDescription(String name) {
334         if (!TOKEN_NAME_TO_VALUE.containsKey(name)) {
335             throw new IllegalArgumentException(TOKEN_NAME_EXCEPTION_PREFIX + name);
336         }
337 
338         final String tokentypes =
339             "com.puppycrawl.tools.checkstyle.api.tokentypes";
340         final ResourceBundle bundle = ResourceBundle.getBundle(tokentypes);
341         return bundle.getString(name);
342     }

345      * Is argument comment-related type (SINGLE_LINE_COMMENT,
346      * BLOCK_COMMENT_BEGIN, BLOCK_COMMENT_END, COMMENT_CONTENT).
347      * @param type
348      *        token type.
349      * @return true if type is comment-related type.
350      */
351     public static boolean isCommentType(int type) {
352         return type == TokenTypes.SINGLE_LINE_COMMENT
353                 || type == TokenTypes.BLOCK_COMMENT_BEGIN
354                 || type == TokenTypes.BLOCK_COMMENT_END
355                 || type == TokenTypes.COMMENT_CONTENT;
356     }

369     /**
370      * @param targetClass from which constructor is returned
371      * @param parameterTypes of constructor
372      * @return constructor of targetClass or {@link IllegalStateException} if any exception occurs
373      * @see Class#getConstructor(Class[])
374      */
375     public static Constructor<?> getConstructor(Class<?> targetClass, Class<?>... parameterTypes) {
376         try {
377             return targetClass.getConstructor(parameterTypes);
378         }
379         catch (NoSuchMethodException ex) {
380             throw new IllegalStateException(ex);
381         }
382     }

384     /**
385      * @param constructor to invoke
386      * @param parameters to pass to constructor
387      * @param <T> type of constructor
388      * @return new instance of class or {@link IllegalStateException} if any exception occurs
389      * @see Constructor#newInstance(Object...)
390      */
391     public static <T> T invokeConstructor(Constructor<T> constructor, Object... parameters) {
392         try {
393             return constructor.newInstance(parameters);
394         }
395         catch (InstantiationException | IllegalAccessException | InvocationTargetException ex) {
396             throw new IllegalStateException(ex);
397         }
398     }

400     /**
401      * Gets the value of a static or instance field of type int or of another primitive type
402      * convertible to type int via a widening conversion. Does not throw any checked exceptions.
403      * @param field from which the int should be extracted
404      * @param object to extract the int value from
405      * @return the value of the field converted to type int
406      * @throws IllegalStateException if this Field object is enforcing Java language access control
407      *         and the underlying field is inaccessible
408      * @see Field#getInt(Object)
409      */
410     public static int getIntFromField(Field field, Object object) {
411         try {
412             return field.getInt(object);
413         }
414         catch (final IllegalAccessException exception) {
415             throw new IllegalStateException(exception);
416         }
417     }

419     /**
420      * Closes a stream re-throwing IOException as IllegalStateException.
421      *
422      * @param closeable Closeable object
423      */
424     public static void close(Closeable closeable) {
425         if (closeable == null) {
426             return;
427         }
428         try {
429             closeable.close();
430         }
431         catch (IOException e) {
432             throw new IllegalStateException("Cannot close the stream", e);
433         }
