403     @Test
404     public void testLoadProperties_IOException() throws Exception {
405         Class<?>[] param = new Class<?>[1];
406         param[0] = File.class;
407         Method method = Main.class.getDeclaredMethod("loadProperties", param);
408         method.setAccessible(true);
409         try {
410             if (System.getProperty("os.name").toLowerCase(Locale.ENGLISH).startsWith("windows")) {
411                 // https://support.microsoft.com/en-us/kb/177506 but this only for NTFS
412                 // WindowsServer 2012 use Resilient File System (ReFS), so any name is ok
413                 File file = new File(File.separator + ":invalid");
414                 if (file.exists()) {
415                     file.delete();
416                 }
417                 method.invoke(null, new File(file.getAbsolutePath()));
418             }
419             else {
420                 method.invoke(null, new File(File.separator + ":invalid"));
421             }
422             fail("Exception was expected");
423         }
424         catch (InvocationTargetException e) {
425             assertTrue(e.getCause() instanceof CheckstyleException);
426             // We do separate validation for message as in Windows
427             // disk drive letter appear in message,
428             // so we skip that drive letter for compatibility issues
429             assertTrue(e.getCause().getMessage().startsWith("Unable to load properties from file '"));
430             assertTrue(e.getCause().getMessage().endsWith(":invalid'."));
431         }
432         catch (Exception e) {
433             fail();
434         }
435     }

473     @Test
474     public void testExistingDirectoryWithViolations() throws Exception {
475 
476         // we just reference there all violations
477         final String[][] outputValues = {
478             {"JavaNCSSCheckTestInput", "1", "84"},
479         };
480 
481         exit.checkAssertionAfterwards(new Assertion() {
482             @Override public void checkAssertion() throws IOException {
483                 String currentPath = new File(".").getCanonicalPath();
484                 String expectedPath = currentPath
485                         + "/src/test/resources/com/puppycrawl/tools/checkstyle/metrics/"
486                         .replace("/", File.separator);
487                 StringBuilder sb = new StringBuilder();
488                 sb.append("Starting audit...").append(System.getProperty("line.separator"));
489                 String format = "%s.java:%s: warning: File length is %s lines (max allowed is 80).";
490                 for (String[] outputValue : outputValues) {
491                     String line = String.format(format,
492                             expectedPath + outputValue[0], outputValue[1],
493                             outputValue[2]);
494                     sb.append(line).append(System.getProperty("line.separator"));
495                 }
496                 sb.append("Audit done.").append(System.getProperty("line.separator"));
497                 assertEquals(sb.toString(), systemOut.getLog());
498                 assertEquals("", systemErr.getLog());
499             }
500         });
501 
502         Main.main("-c", "src/test/resources/com/puppycrawl/tools/checkstyle/config-filelength.xml",
503                 "src/test/resources/com/puppycrawl/tools/checkstyle/metrics/");
504     }
