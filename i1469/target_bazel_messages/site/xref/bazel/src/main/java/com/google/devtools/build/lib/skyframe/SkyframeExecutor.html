404   /**
405    * This method exists only to allow a module to make a top-level Skyframe call during the
406    * transition to making it fully Skyframe-compatible. Do not add additional callers!
407    */
408   public <E extends Exception> SkyValue evaluateSkyKeyForCodeMigration(final SkyKey key,
409       final Class<E> clazz) throws E {
410     try {
411       return callUninterruptibly(new Callable<SkyValue>() {
412         @Override
413         public SkyValue call() throws E, InterruptedException {
414           synchronized (valueLookupLock) {
415             // We evaluate in keepGoing mode because in the case that the graph does not store its
416             // edges, nokeepGoing builds are not allowed, whereas keepGoing builds are always
417             // permitted.
418             EvaluationResult<SkyValue> result = buildDriver.evaluate(
419                 ImmutableList.of(key), true, ResourceUsage.getAvailableProcessors(),
420                 errorEventListener);
421             if (!result.hasError()) {
422               return Preconditions.checkNotNull(result.get(key), "%s %s", result, key);
423             }
424             ErrorInfo errorInfo = Preconditions.checkNotNull(result.getError(key),
425                 "%s %s", key, result);
426             Throwables.propagateIfPossible(errorInfo.getException(), clazz);
427             if (errorInfo.getException() != null) {
428               throw new IllegalStateException(errorInfo.getException());
429             }
430             throw new IllegalStateException(errorInfo.toString());
431           }
432         }
433       });
434     } catch (Exception e) {
435       Throwables.propagateIfPossible(e, clazz);
436       throw new IllegalStateException(e);
437     }
438   }

725   protected Differencer.Diff getDiff(Iterable<PathFragment> modifiedSourceFiles,
726       final Path pathEntry) throws InterruptedException {
727     // TODO(bazel-team): change ModifiedFileSet to work with RootedPaths instead of PathFragments.
728     Iterable<SkyKey> dirtyFileStateSkyKeys = Iterables.transform(modifiedSourceFiles,
729         new Function<PathFragment, SkyKey>() {
730           @Override
731           public SkyKey apply(PathFragment pathFragment) {
732             Preconditions.checkState(!pathFragment.isAbsolute(),
733                 "found absolute PathFragment: %s", pathFragment);
734             return FileStateValue.key(RootedPath.toRootedPath(pathEntry, pathFragment));
735           }
736         });
737     // We only need to invalidate directory values when a file has been created or deleted or
738     // changes type, not when it has merely been modified. Unfortunately we do not have that
739     // information here, so we compute it ourselves.
740     // TODO(bazel-team): Fancy filesystems could provide it with a hypothetically modified
741     // DiffAwareness interface.
742     Supplier<Map<SkyKey, SkyValue>> valuesSupplier = Suppliers.memoize(
743         new Supplier<Map<SkyKey, SkyValue>>() {
744       @Override
745       public Map<SkyKey, SkyValue> get() {
746         return memoizingEvaluator.getValues();
747       }
748     });
749     FilesystemValueChecker fsvc = new FilesystemValueChecker(valuesSupplier, tsgm, null);
750     Differencer.DiffWithDelta diff =
751         fsvc.getNewAndOldValues(dirtyFileStateSkyKeys, new FileDirtinessChecker());
752 
753     Set<SkyKey> valuesToInvalidate = new HashSet<>();
754     Map<SkyKey, SkyValue> valuesToInject = new HashMap<>();
755     for (Map.Entry<SkyKey, Delta> entry : diff.changedKeysWithNewAndOldValues().entrySet()) {
756       SkyKey key = entry.getKey();
757       Preconditions.checkState(key.functionName().equals(SkyFunctions.FILE_STATE), key);
758       RootedPath rootedPath = (RootedPath) key.argument();
759       Delta delta = entry.getValue();
760       FileStateValue oldValue = (FileStateValue) delta.getOldValue();
761       FileStateValue newValue = (FileStateValue) delta.getNewValue();
762       if (newValue != null) {
763         valuesToInject.put(key, newValue);
764       } else {
765         valuesToInvalidate.add(key);
766       }
767       SkyKey dirListingStateKey = parentDirectoryListingStateKey(rootedPath);
768       // Invalidate the directory listing for the path's parent directory if the change was
769       // relevant (e.g. path turned from a symlink into a directory) OR if we don't have enough
770       // information to determine it was irrelevant.
771       boolean changedType = false;
772       if (newValue == null) {
773         changedType = true;
774       } else if (oldValue != null) {
775         changedType = !oldValue.getType().equals(newValue.getType());
776       } else {
777         DirectoryListingStateValue oldDirListingStateValue =
778             (DirectoryListingStateValue) valuesSupplier.get().get(dirListingStateKey);
779         if (oldDirListingStateValue != null) {
780           String baseName = rootedPath.getRelativePath().getBaseName();
781           Dirent oldDirent = oldDirListingStateValue.getDirents().maybeGetDirent(baseName);
782           changedType = (oldDirent == null)
783               || !compatibleFileTypes(oldDirent.getType(), newValue.getType());
784         } else {
785           changedType = true;
786         }
787       }
788       if (changedType) {
789         valuesToInvalidate.add(dirListingStateKey);
790       }
791     }
792     for (SkyKey key : diff.changedKeysWithoutNewValues()) {
793       Preconditions.checkState(key.functionName().equals(SkyFunctions.FILE_STATE), key);
794       RootedPath rootedPath = (RootedPath) key.argument();
795       valuesToInvalidate.add(parentDirectoryListingStateKey(rootedPath));
796     }
797     return new ImmutableDiff(valuesToInvalidate, valuesToInject);
798   }

1072   @ThreadSafety.ThreadSafe
1073   public ImmutableMap<Dependency, ConfiguredTarget> getConfiguredTargetMap(
1074       BuildConfiguration originalConfig, Iterable<Dependency> keys, boolean useOriginalConfig) {
1075     checkActive();
1076     if (skyframeBuildView == null) {
1077       // If build view has not yet been initialized, no configured targets can have been created.
1078       // This is most likely to happen after a failed loading phase.
1079       return ImmutableMap.of();
1080     }
1081 
1082     Map<Dependency, BuildConfiguration> configs;
1083     if (originalConfig != null) {
1084       if (useOriginalConfig) {
1085         // This flag is used because of some unfortunate complexity in the configuration machinery:
1086         // Most callers of this method pass a <Label, Configuration> pair to directly create a
1087         // ConfiguredTarget from, but happen to use the Dependency data structure to pass that
1088         // info (even though the data has nothing to do with dependencies). If this configuration
1089         // includes a split transition, a dynamic configuration created from it will *not*
1090         // include that transition (because dynamic configurations don't embed transitions to
1091         // other configurations. In that case, we need to preserve the original configuration.
1092         // TODO(bazel-team); make this unnecessary once split transition logic is properly ported
1093         // out of configurations.
1094         configs = new HashMap<>();
1095         configs.put(Iterables.getOnlyElement(keys), originalConfig);
1096       } else {
1097         configs = getConfigurations(originalConfig.getOptions(), keys);
1098       }
1099     } else {
1100       configs = new HashMap<>();
1101       for (Dependency key : keys) {
1102         configs.put(key, null);
1103       }
1104     }
1105 
1106     final List<SkyKey> skyKeys = new ArrayList<>();
1107     for (Dependency key : keys) {
1108       skyKeys.add(ConfiguredTargetValue.key(key.getLabel(), configs.get(key)));
1109       for (Class<? extends ConfiguredAspectFactory> aspect : key.getAspects()) {
1110         skyKeys.add(AspectValue.key(key.getLabel(), configs.get(key), aspect));
1111       }
1112     }
1113 
1114     EvaluationResult<SkyValue> result = evaluateSkyKeys(skyKeys);
1115     ImmutableMap.Builder<Dependency, ConfiguredTarget> cts = ImmutableMap.builder();
1116 
1117   DependentNodeLoop:
1118     for (Dependency key : keys) {
1119       SkyKey configuredTargetKey = ConfiguredTargetValue.key(
1120           key.getLabel(), configs.get(key));
1121       if (result.get(configuredTargetKey) == null) {
1122         continue;
1123       }
1124 
1125       ConfiguredTarget configuredTarget =
1126           ((ConfiguredTargetValue) result.get(configuredTargetKey)).getConfiguredTarget();
1127       List<Aspect> aspects = new ArrayList<>();
1128

1144   /**
1145    * Retrieves the configurations needed for the given deps, trimming down their fragments
1146    * to those only needed by their transitive closures.
1147    */
1148   private Map<Dependency, BuildConfiguration> getConfigurations(BuildOptions fromOptions,
1149       Iterable<Dependency> keys) {
1150     Map<Dependency, BuildConfiguration> builder = new HashMap<>();
1151     Set<Dependency> depsToEvaluate = new HashSet<>();
1152 
1153     // Check: if !Configuration.useDynamicConfigs then just return the original configs.
1154 
1155     // Get the fragments needed for dynamic configuration nodes.
1156     final List<SkyKey> transitiveFragmentSkyKeys = new ArrayList<>();
1157     Map<Label, Set<Class<? extends BuildConfiguration.Fragment>>> fragmentsMap = new HashMap<>();
1158     Set<Label> labelsWithErrors = new HashSet<>();
1159     for (Dependency key : keys) {
1160       if (key.hasStaticConfiguration()) {
1161         builder.put(key, key.getConfiguration());
1162       } else if (key.getTransition() == Attribute.ConfigurationTransition.NULL) {
1163         builder.put(key, null);
1164       } else {
1165         depsToEvaluate.add(key);
1166         transitiveFragmentSkyKeys.add(TransitiveTargetValue.key(key.getLabel()));
1167       }
1168     }
1169     EvaluationResult<SkyValue> fragmentsResult = evaluateSkyKeys(transitiveFragmentSkyKeys);
1170     for (Dependency key : keys) {
1171       if (!depsToEvaluate.contains(key)) {
1172         // No fragments to compute here.
1173       } else if (fragmentsResult.getError(TransitiveTargetValue.key(key.getLabel())) != null) {
1174         labelsWithErrors.add(key.getLabel());
1175       } else {
1176         TransitiveTargetValue ttv =
1177             (TransitiveTargetValue) fragmentsResult.get(TransitiveTargetValue.key(key.getLabel()));
1178         fragmentsMap.put(key.getLabel(), ttv.getTransitiveConfigFragments().toSet());
1179       }
1180     }
1181 
1182     // Now get the configurations.
1183     final List<SkyKey> configSkyKeys = new ArrayList<>();
1184     for (Dependency key : keys) {
1185       if (!depsToEvaluate.contains(key) || labelsWithErrors.contains(key.getLabel())) {
1186         continue;
1187       }
1188       configSkyKeys.add(BuildConfigurationValue.key(fragmentsMap.get(key.getLabel()),
1189           getDynamicConfigOptions(key, fromOptions)));
1190     }
1191     EvaluationResult<SkyValue> configsResult = evaluateSkyKeys(configSkyKeys);
1192     for (Dependency key : keys) {
1193       if (!depsToEvaluate.contains(key) || labelsWithErrors.contains(key.getLabel())) {
1194         continue;
1195       }
1196       SkyKey configKey = BuildConfigurationValue.key(fragmentsMap.get(key.getLabel()),
1197           getDynamicConfigOptions(key, fromOptions));
1198       builder.put(key, ((BuildConfigurationValue) configsResult.get(configKey)).getConfiguration());
1199     }
1200 
1201     return builder;
1202   }

1469     /**
1470      * Looks up a particular package (mostly used after the loading phase, so packages should
1471      * already be present, but occasionally used pre-loading phase). Use should be discouraged,
1472      * since this cannot be used inside a Skyframe evaluation, and concurrent calls are
1473      * synchronized.
1474      *
1475      * <p>Note that this method needs to be synchronized since InMemoryMemoizingEvaluator.evaluate()
1476      * method does not support concurrent calls.
1477      */
1478     Package getPackage(EventHandler eventHandler, PackageIdentifier pkgName)
1479         throws InterruptedException, NoSuchPackageException {
1480       synchronized (valueLookupLock) {
1481         SkyKey key = PackageValue.key(pkgName);
1482         // Any call to this method post-loading phase should either be error-free or be in a
1483         // keep_going build, since otherwise the build would have failed during loading. Thus
1484         // we set keepGoing=true unconditionally.
1485         EvaluationResult<PackageValue> result =
1486             buildDriver.evaluate(ImmutableList.of(key), /*keepGoing=*/true,

1489           ErrorInfo error = result.getError();

1499           // errors as NoSuchPackageExceptions.

1501           throw new IllegalStateException("Unexpected Exception type from PackageValue for '"
1502               + pkgName + "'' with root causes: " + Iterables.toString(error.getRootCauses()), e);
1503         }
1504         return result.get(key).getPackage();
1505       }
