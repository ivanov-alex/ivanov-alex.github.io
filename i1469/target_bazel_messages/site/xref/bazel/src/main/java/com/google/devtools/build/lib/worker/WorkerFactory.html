14  package com.google.devtools.build.lib.worker;
15  
16  import org.apache.commons.pool2.BaseKeyedPooledObjectFactory;
17  import org.apache.commons.pool2.PooledObject;
18  import org.apache.commons.pool2.impl.DefaultPooledObject;
19  
20  /**
21   * Factory used by the pool to create / destroy / validate worker processes.
22   */
23  final class WorkerFactory extends BaseKeyedPooledObjectFactory<WorkerKey, Worker> {
24    @Override
25    public Worker create(WorkerKey key) throws Exception {
26      return Worker.create(key);
27    }
28  
29    /**
30     * Use the DefaultPooledObject implementation.
31     */
32    @Override
33    public PooledObject<Worker> wrap(Worker worker) {
34      return new DefaultPooledObject<Worker>(worker);
35    }
36  
37    /**
38     * When a worker process is discarded, destroy its process, too.
39     */
40    @Override
41    public void destroyObject(WorkerKey key, PooledObject<Worker> p) throws Exception {
42      p.getObject().destroy();
43    }
44  
45    /**
46     * The worker is considered to be valid when its process is still alive.
47     */
48    @Override
49    public boolean validateObject(WorkerKey key, PooledObject<Worker> p) {
50      return p.getObject().isAlive();
51    }
52  }
