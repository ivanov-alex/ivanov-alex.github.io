83  /**
84   * {@link AbstractBlazeQueryEnvironment} that introspects the Skyframe graph to find forward and
85   * reverse edges. Results obtained by calling {@link #evaluateQuery} are not guaranteed to be in
86   * any particular order. As well, this class eagerly loads the full transitive closure of targets,
87   * even if the full closure isn't needed.
88   */

112   public SkyQueryEnvironment(boolean keepGoing, boolean strictScope, int loadingPhaseThreads,
113       Predicate<Label> labelFilter,
114       EventHandler eventHandler,
115       Set<Setting> settings,
116       Iterable<QueryFunction> extraFunctions, String parserPrefix,
117       WalkableGraphFactory graphFactory,
118       List<String> universeScope, PathPackageLocator pkgPath) {
119     super(keepGoing, strictScope, labelFilter,
120         eventHandler,
121         settings,
122         extraFunctions);
123     this.loadingPhaseThreads = loadingPhaseThreads;
124     this.graphFactory = graphFactory;
125     this.pkgPath = pkgPath;
126     this.universeScope = Preconditions.checkNotNull(universeScope);
127     this.parserPrefix = parserPrefix;
128     Preconditions.checkState(!universeScope.isEmpty(),
129         "No queries can be performed with an empty universe");
130   }

132   private void init() throws InterruptedException {
133     long startTime = Profiler.nanoTimeMaybe();
134     EvaluationResult<SkyValue> result =
135         graphFactory.prepareAndGet(universeScope, loadingPhaseThreads, eventHandler);
136     graph = result.getWalkableGraph();
137     Collection<SkyValue> values = result.values();
138     long duration = Profiler.nanoTimeMaybe() - startTime;
139     if (duration > 0) {
140       LOG.info("Spent " + (duration / 1000 / 1000) + " ms on evaluation and walkable graph");
141     }
142 
143     // The universe query may fail if there are errors during its evaluation, e.g. because of
144     // cycles in the target graph.
145     boolean singleValueEvaluated = values.size() == 1;
146     boolean foundError = !result.errorMap().isEmpty();
147     boolean evaluationFoundCycle =
148         foundError && !Iterables.isEmpty(result.getError().getCycleInfo());
149     Preconditions.checkState(singleValueEvaluated || evaluationFoundCycle,
150         "Universe query \"%s\" unexpectedly did not result in a single value as expected (%s"
151             + " values in result) and it did not fail because of a cycle.%s",
152         universeScope, values.size(), foundError ? " Error: " + result.getError().toString() : "");
153     if (singleValueEvaluated) {
154       PrepareDepsOfPatternsValue prepareDepsOfPatternsValue =
155           (PrepareDepsOfPatternsValue) Iterables.getOnlyElement(values);
156       universeTargetPatternKeys = prepareDepsOfPatternsValue.getTargetPatternKeys();
157     } else {
158       // The error is because of a cycle, so keep going with the graph we managed to load.
159       universeTargetPatternKeys = ImmutableList.of();
160     }
161   }

164   public QueryEvalResult<Target> evaluateQuery(QueryExpression expr)
165       throws QueryException, InterruptedException {
166     // Some errors are reported as QueryExceptions and others as ERROR events (if --keep_going). The
167     // result is set to have an error iff there were errors emitted during the query, so we reset

169     eventHandler.resetErrors();
170     init();
171     return super.evaluateQuery(expr);
172   }

174   private Map<Target, Collection<Target>> makeTargetsMap(Map<SkyKey, Iterable<SkyKey>> input) {
175     ImmutableMap.Builder<Target, Collection<Target>> result = ImmutableMap.builder();
176 
177     for (Map.Entry<SkyKey, Target> entry : makeTargetsWithAssociations(input.keySet()).entrySet()) {
178       result.put(entry.getValue(), makeTargets(input.get(entry.getKey())));
179     }
180     return result.build();
181   }

191   private Set<Label> getAllowedDeps(Rule rule) {
192     Set<Label> allowedLabels = new HashSet<>(rule.getTransitions(dependencyFilter).values());
193     allowedLabels.addAll(rule.getVisibility().getDependencyLabels());
194     // We should add deps from aspects, otherwise they are going to be filtered out.
195     allowedLabels.addAll(rule.getAspectLabelsSuperset(dependencyFilter));
196     return allowedLabels;
197   }

199   private Collection<Target> filterFwdDeps(Target target, Collection<Target> rawFwdDeps) {
200     if (!(target instanceof Rule)) {
201       return rawFwdDeps;
202     }
203     final Set<Label> allowedLabels = getAllowedDeps((Rule) target);
204     return Collections2.filter(rawFwdDeps,
205         new Predicate<Target>() {
206           @Override
207           public boolean apply(Target target) {
208             return allowedLabels.contains(target.getLabel());
209           }
210         });
211   }

213   @Override
214   public Collection<Target> getFwdDeps(Iterable<Target> targets) {
215     Set<Target> result = new HashSet<>();
216     for (Map.Entry<Target, Collection<Target>> entry : getRawFwdDeps(targets).entrySet()) {
217       result.addAll(filterFwdDeps(entry.getKey(), entry.getValue()));
218     }
219     return result;
220   }

222   @Override
223   public Collection<Target> getReverseDeps(Iterable<Target> targets) {
224     Set<Target> result = CompactHashSet.create();
225     Map<Target, Collection<Target>> rawReverseDeps = getRawReverseDeps(targets);
226 
227     CompactHashSet<Target> visited = CompactHashSet.create();
228 
229     Set<Label> keys = CompactHashSet.create(Collections2.transform(rawReverseDeps.keySet(),
230         TARGET_LABEL_FUNCTION));
231     for (Collection<Target> parentCollection : rawReverseDeps.values()) {
232       for (Target parent : parentCollection) {
233         if (visited.add(parent)) {
234           if (parent instanceof Rule) {
235             for (Label label : getAllowedDeps((Rule) parent)) {
236               if (keys.contains(label)) {
237                 result.add(parent);
238               }
239             }
240           } else {
241             result.add(parent);
242           }
243         }
244       }
245     }
246     return result;
247   }

249   @Override
250   public Set<Target> getTransitiveClosure(Set<Target> targets) {
251     Set<Target> visited = new HashSet<>();
252     Collection<Target> current = targets;
253     while (!current.isEmpty()) {
254       Collection<Target> toVisit = Collections2.filter(current,
255           Predicates.not(Predicates.in(visited)));
256       current = getFwdDeps(toVisit);
257       visited.addAll(toVisit);
258     }
259     return ImmutableSet.copyOf(visited);
260   }

262   // Implemented with a breadth-first search.
263   @Override
264   public Set<Target> getNodesOnPath(Target from, Target to) {
265     // Tree of nodes visited so far.
266     Map<Target, Target> nodeToParent = new HashMap<>();
267     // Contains all nodes left to visit in a (LIFO) stack.
268     Deque<Target> toVisit = new ArrayDeque<>();
269     toVisit.add(from);
270     nodeToParent.put(from, null);
271     while (!toVisit.isEmpty()) {
272       Target current = toVisit.removeFirst();
273       if (to.equals(current)) {
274         return ImmutableSet.copyOf(Digraph.getPathToTreeNode(nodeToParent, to));
275       }
276       for (Target dep : getFwdDeps(ImmutableList.of(current))) {
277         if (!nodeToParent.containsKey(dep)) {
278           nodeToParent.put(dep, current);
279           toVisit.addFirst(dep);
280         }
281       }
282     }
283     // Note that the only current caller of this method checks first to see if there is a path
284     // before calling this method. It is not clear what the return value should be here.
285     return null;
286   }

288   @Override
289   public Set<Target> getTargetsMatchingPattern(QueryExpression owner, String pattern)
290       throws QueryException {
291     Set<Target> targets = new LinkedHashSet<>(resolvedTargetPatterns.get(pattern));
292 
293     // Sets.filter would be more convenient here, but can't deal with exceptions.
294     Iterator<Target> targetIterator = targets.iterator();
295     while (targetIterator.hasNext()) {
296       Target target = targetIterator.next();
297       if (!validateScope(target.getLabel(), strictScope)) {
298         targetIterator.remove();
299       }
300     }
301     return targets;
302   }

304   @Override
305   public Set<Target> getBuildFiles(QueryExpression caller, Set<Target> nodes)
306       throws QueryException {
307     Set<Target> dependentFiles = new LinkedHashSet<>();
308     Set<Package> seenPackages = new HashSet<>();
309     // Keep track of seen labels, to avoid adding a fake subinclude label that also exists as a
310     // real target.
311     Set<Label> seenLabels = new HashSet<>();
312 
313     // Adds all the package definition files (BUILD files and build
314     // extensions) for package "pkg", to "buildfiles".
315     for (Target x : nodes) {
316       Package pkg = x.getPackage();
317       if (seenPackages.add(pkg)) {
318         addIfUniqueLabel(pkg.getBuildFile(), seenLabels, dependentFiles);
319         for (Label subinclude
320             : Iterables.concat(pkg.getSubincludeLabels(), pkg.getSkylarkFileDependencies())) {
321           addIfUniqueLabel(getSubincludeTarget(subinclude, pkg), seenLabels, dependentFiles);
322 
323           // Also add the BUILD file of the subinclude.
324           try {
325             addIfUniqueLabel(getSubincludeTarget(
326                 subinclude.getLocalTargetLabel("BUILD"), pkg), seenLabels, dependentFiles);
327           } catch (Label.SyntaxException e) {
328             throw new AssertionError("BUILD should always parse as a target name", e);
329           }
330         }
331       }
332     }
333     return dependentFiles;
334   }

338       nodes.add(node);

351   private SkyKey getPackageKeyAndValidateLabel(Label label) throws QueryException {
352     // Can't use strictScope here because we are expecting a target back.
353     validateScope(label, true);
354     return PackageValue.key(label.getPackageIdentifier());

357   @Override
358   public Target getTarget(Label label) throws TargetNotFoundException, QueryException {
359     SkyKey packageKey = getPackageKeyAndValidateLabel(label);
360     if (!graph.exists(packageKey)) {
361       throw new QueryException(packageKey + " does not exist in graph");
362     }
363     try {
364       PackageValue packageValue = (PackageValue) graph.getValue(packageKey);
365       if (packageValue != null) {
366         return packageValue.getPackage().getTarget(label.getName());
367       } else {
368         throw (NoSuchThingException) Preconditions.checkNotNull(
369             graph.getException(packageKey), label);
370       }
371     } catch (NoSuchThingException e) {
372       throw new TargetNotFoundException(e);
373     }
374   }

376   @Override
377   public void buildTransitiveClosure(QueryExpression caller, Set<Target> targets, int maxDepth)
378       throws QueryException {
379     // Everything has already been loaded, so here we just check for errors so that we can
380     // pre-emptively throw/report if needed.
381     for (Map.Entry<SkyKey, Exception> entry :
382       graph.getMissingAndExceptions(makeKeys(targets)).entrySet()) {
383       if (entry.getValue() == null) {
384         throw new QueryException(entry.getKey().argument() + " does not exist in graph");
385       }
386       reportBuildFileError(caller, entry.getValue().getMessage());
387     }
388   }

390   private Set<Target> filterTargetsNotInGraph(Set<Target> targets) {
391     Map<Target, SkyKey> map = Maps.toMap(targets, TARGET_TO_SKY_KEY);
392     Set<SkyKey> present = graph.getDoneValues(map.values()).keySet();
393     if (present.size() == targets.size()) {
394       // Optimize for case of all targets being in graph.
395       return targets;
396     }
397     return Maps.filterValues(map, Predicates.in(present)).keySet();
398   }

400   @Override
401   protected Map<String, Set<Target>> preloadOrThrow(
402       QueryExpression caller, Collection<String> patterns)
403       throws QueryException, TargetParsingException {
404     GraphBackedRecursivePackageProvider provider =
405         new GraphBackedRecursivePackageProvider(graph, universeTargetPatternKeys);
406     Map<String, Set<Target>> result = Maps.newHashMapWithExpectedSize(patterns.size());
407     for (String pattern : patterns) {
408       SkyKey patternKey = TargetPatternValue.key(pattern,
409           TargetPatternEvaluator.DEFAULT_FILTERING_POLICY, parserPrefix);
410 
411       TargetPatternValue.TargetPatternKey targetPatternKey =
412           ((TargetPatternValue.TargetPatternKey) patternKey.argument());
413 
414       TargetParsingException targetParsingException = null;
415       if (graph.exists(patternKey)) {
416         // The graph already contains a value or exception for this target pattern, so we use it.
417         TargetPatternValue value = (TargetPatternValue) graph.getValue(patternKey);
418         if (value != null) {
419           result.put(
420               pattern, ImmutableSet.copyOf(makeTargetsFromLabels(value.getTargets().getTargets())));
421         } else {
422           // Because the graph was always initialized via a keep_going build, we know that the
423           // exception stored here must be a TargetParsingException. Thus the comment in
424           // SkyframeTargetPatternEvaluator#parseTargetPatternKeys describing the situation in which
425           // the exception acceptance must be looser does not apply here.
426           targetParsingException =
427               (TargetParsingException)
428                   Preconditions.checkNotNull(graph.getException(patternKey), pattern);
429         }
430       } else {
431         // If the graph doesn't contain a value for this target pattern, try to directly evaluate
432         // it, by making use of packages already present in the graph.
433         RecursivePackageProviderBackedTargetPatternResolver resolver =
434             new RecursivePackageProviderBackedTargetPatternResolver(provider, eventHandler,
435                 targetPatternKey.getPolicy(), pkgPath);
436         TargetPattern parsedPattern = targetPatternKey.getParsedPattern();
437         try {
438           result.put(pattern, filterTargetsNotInGraph(parsedPattern.eval(resolver).getTargets()));
439         } catch (TargetParsingException e) {
440           targetParsingException = e;
441         } catch (InterruptedException e) {
442           throw new QueryException(e.getMessage());
443         }
444       }
445 
446       if (targetParsingException != null) {
447         if (!keepGoing) {
448           throw targetParsingException;
449         } else {
450           eventHandler.handle(Event.error("Evaluation of query \"" + caller + "\" failed: "
451               + targetParsingException.getMessage()));
452           result.put(pattern, ImmutableSet.<Target>of());
453         }
454       }
455     }
456     return result;
457   }

463   private static final Function<SkyKey, Label> SKYKEY_TO_LABEL = new Function<SkyKey, Label>() {
464     @Nullable
465     @Override
466     public Label apply(SkyKey skyKey) {
467       SkyFunctionName functionName = skyKey.functionName();
468       if (!functionName.equals(SkyFunctions.TRANSITIVE_TRAVERSAL)) {
469         // Skip non-targets.

471       }

473     }

484   private <E> Map<E, Target> makeTargetsWithAssociations(Iterable<E> keys,
485       Function<E, Label> toLabel) {
486     Multimap<SkyKey, E> packageKeyToTargetKeyMap = ArrayListMultimap.create();
487     for (E key : keys) {
488       Label label = toLabel.apply(key);
489       if (label == null) {
490         continue;
491       }
492       try {
493         packageKeyToTargetKeyMap.put(getPackageKeyAndValidateLabel(label), key);
494       } catch (QueryException e) {
495         // Skip disallowed labels.
496       }
497     }
498     ImmutableMap.Builder<E, Target> result = ImmutableMap.builder();
499     Map<SkyKey, SkyValue> packageMap = graph.getDoneValues(packageKeyToTargetKeyMap.keySet());
500     for (Map.Entry<SkyKey, SkyValue> entry : packageMap.entrySet()) {
501       for (E targetKey : packageKeyToTargetKeyMap.get(entry.getKey())) {
502         try {
503           result.put(targetKey, ((PackageValue) entry.getValue()).getPackage()
504               .getTarget((toLabel.apply(targetKey)).getName()));
505         } catch (NoSuchTargetException e) {
506           // Skip missing target.
507         }
508       }
509     }
510     return result.build();
511   }

530   /**
531    * Get SkyKeys for the FileValues for the given {@param pathFragments}. To do this, we look for a
532    * package lookup node for each path fragment, since package lookup nodes contain the "root" of a
533    * package. The returned SkyKeys correspond to FileValues that may not exist in the graph.
534    */
535   private Collection<SkyKey> getSkyKeysForFileFragments(Iterable<PathFragment> pathFragments) {
536     Set<SkyKey> result = new HashSet<>();
537     Multimap<PathFragment, PathFragment> currentToOriginal = ArrayListMultimap.create();
538     for (PathFragment pathFragment : pathFragments) {
539       currentToOriginal.put(pathFragment, pathFragment);
540     }
541     while (!currentToOriginal.isEmpty()) {
542       Map<SkyKey, PathFragment> keys = new HashMap<>();
543       for (PathFragment pathFragment : currentToOriginal.keySet()) {
544         keys.put(
545             PackageLookupValue.key(PackageIdentifier.createInDefaultRepo(pathFragment)),
546             pathFragment);
547       }
548       Map<SkyKey, SkyValue> lookupValues = graph.getDoneValues(keys.keySet());
549       for (Map.Entry<SkyKey, SkyValue> entry : lookupValues.entrySet()) {
550         PackageLookupValue packageLookupValue = (PackageLookupValue) entry.getValue();
551         PathFragment dir = keys.get(entry.getKey());
552         if (packageLookupValue.packageExists()) {
553           Collection<PathFragment> originalFiles = currentToOriginal.get(dir);
554           Preconditions.checkState(!originalFiles.isEmpty(), entry);
555           for (PathFragment fileName : originalFiles) {
556             result.add(
557                 FileValue.key(RootedPath.toRootedPath(packageLookupValue.getRoot(), fileName)));
558           }
559         }
560         currentToOriginal.removeAll(dir);
561       }
562       Multimap<PathFragment, PathFragment> newCurrentToOriginal = ArrayListMultimap.create();
563       for (PathFragment pathFragment : currentToOriginal.keySet()) {
564         PathFragment parent = pathFragment.getParentDirectory();
565         if (parent != null) {
566           newCurrentToOriginal.putAll(parent, currentToOriginal.get(pathFragment));
567         }
568       }
569       currentToOriginal = newCurrentToOriginal;
570     }
571     return result;
572   }

574   /**
575    * Calculates the set of {@link Package} objects, represented as source file targets, that depend
576    * on the given list of BUILD files and subincludes (other files are filtered out).
577    */
578   @Nullable
579   Set<Target> getRBuildFiles(Collection<PathFragment> fileIdentifiers) {
580     Collection<SkyKey> files = getSkyKeysForFileFragments(fileIdentifiers);
581     Collection<SkyKey> current = graph.getDoneValues(files).keySet();
582     Set<SkyKey> resultKeys = CompactHashSet.create();
583     while (!current.isEmpty()) {
584       Collection<Iterable<SkyKey>> reverseDeps = graph.getReverseDeps(current).values();
585       current = new HashSet<>();
586       for (SkyKey rdep : Iterables.concat(reverseDeps)) {
587         if (rdep.functionName().equals(SkyFunctions.PACKAGE)) {
588           resultKeys.add(rdep);
589         } else if (!rdep.functionName().equals(SkyFunctions.PACKAGE_LOOKUP)) {
590           // Packages may depend on subpackages for existence, but we don't report them as rdeps.
591           current.add(rdep);
592         }
593       }
594     }
595     Map<SkyKey, SkyValue> packageValues = graph.getDoneValues(resultKeys);
596     if (packageValues.size() != resultKeys.size()) {
597       throw new IllegalStateException(
598           "Missing values: " + Sets.difference(resultKeys, packageValues.keySet()));
599     }
600     ImmutableSet.Builder<Target> result = ImmutableSet.builder();
601     for (SkyValue value : packageValues.values()) {
602       result.add(((PackageValue) value).getPackage().getBuildFile());
603     }
604     return result.build();
605   }

607   @Override
608   public Iterable<QueryFunction> getFunctions() {
609     return ImmutableList.<QueryFunction>builder()
610         .addAll(super.getFunctions())
611         .add(new AllRdepsFunction())
612         .add(new RBuildFilesFunction())
613         .build();
614   }
