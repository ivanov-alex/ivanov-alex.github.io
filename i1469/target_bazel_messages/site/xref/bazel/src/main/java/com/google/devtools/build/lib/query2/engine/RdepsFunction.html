14  package com.google.devtools.build.lib.query2.engine;
15  
16  import com.google.common.base.Predicates;
17  import com.google.common.collect.ImmutableList;
18  import com.google.devtools.build.lib.query2.engine.QueryEnvironment.Argument;
19  import com.google.devtools.build.lib.query2.engine.QueryEnvironment.ArgumentType;
20  
21  import java.util.List;
22  import java.util.Set;
23  
24  /**
25   * An "rdeps" query expression, which computes the reverse dependencies of the argument within the
26   * transitive closure of the universe. An optional integer-literal third argument may be
27   * specified; its value bounds the search from the arguments.
28   *
29   * <pre>expr ::= RDEPS '(' expr ',' expr ')'</pre>
30   * <pre>       | RDEPS '(' expr ',' expr ',' WORD ')'</pre>
31   */
32  final class RdepsFunction extends AllRdepsFunction {
33    RdepsFunction() {}
34  
35    @Override
36    public String getName() {
37      return "rdeps";
38    }
39  
40    @Override
41    public int getMandatoryArguments() {
42      return super.getMandatoryArguments() + 1;  // +1 for the universe.
43    }
44  
45    @Override
46    public List<ArgumentType> getArgumentTypes() {
47      return ImmutableList.<ArgumentType>builder()
48          .add(ArgumentType.EXPRESSION).addAll(super.getArgumentTypes()).build();
49    }
50  
51    /**
52     * Compute the transitive closure of the universe, then breadth-first search from the argument
53     * towards the universe while staying within the transitive closure.
54     */
55    @Override
56    public <T> Set<T> eval(QueryEnvironment<T> env, QueryExpression expression, List<Argument> args)
57        throws QueryException, InterruptedException {
58      Set<T> universeValue = args.get(0).getExpression().eval(env);
59      env.buildTransitiveClosure(expression, universeValue, Integer.MAX_VALUE);
60  
61      return eval(env, args.subList(1, args.size()),
62          Predicates.in(env.getTransitiveClosure(universeValue)));
63    }
64  }
