47  /**
48   * Module implementing the rule set of Bazel.
49   */
50  public class BazelRulesModule extends BlazeModule {
51    /**
52     * Execution options affecting how we execute the build actions (but not their semantics).
53     */
54    public static class BazelExecutionOptions extends OptionsBase {
55      @Option(
56        name = "spawn_strategy",
57        defaultValue = "",
58        category = "strategy",
59        help =
60            "Specify how spawn actions are executed by default."
61                + "'standalone' means run all of them locally."
62                + "'sandboxed' means run them in namespaces based sandbox (available only on Linux)"
63      )
64      public String spawnStrategy;
65  
66      @Option(
67        name = "genrule_strategy",
68        defaultValue = "",
69        category = "strategy",
70        help =
71            "Specify how to execute genrules."
72                + "'standalone' means run all of them locally."
73                + "'sandboxed' means run them in namespaces based sandbox (available only on Linux)"
74      )
75      public String genruleStrategy;
76  
77      @Option(name = "strategy",
78          allowMultiple = true,
79          converter = AssignmentConverter.class,
80          defaultValue = "",
81          category = "strategy",
82          help = "Specify how to distribute compilation of other spawn actions. "
83              + "Example: 'Javac=local' means to spawn Java compilation locally. "
84              + "'JavaIjar=sandboxed' means to spawn Java Ijar actions in a sandbox. ")
85      public List<Map.Entry<String, String>> strategy;
86    }
87  
88    private static class BazelActionContextConsumer implements ActionContextConsumer {
89      BazelExecutionOptions options;
90  
91      private BazelActionContextConsumer(BazelExecutionOptions options) {
92        this.options = options;
93  
94      }
95      @Override
96      public Map<String, String> getSpawnActionContexts() {
97        Map<String, String> contexts = new TreeMap<>(String.CASE_INSENSITIVE_ORDER);
98  
99        contexts.put("Genrule", options.genruleStrategy);
100 
101       for (Map.Entry<String, String> strategy : options.strategy) {
102         String strategyName = strategy.getValue();
103         // TODO(philwo) - remove this when the standalone / local mess is cleaned up.
104         // Some flag expansions use "local" as the strategy name, but the strategy is now called
105         // "standalone", so we'll translate it here.
106         if (strategyName.equals("local")) {
107           strategyName = "standalone";
108         }
109         contexts.put(strategy.getKey(), strategyName);
110       }
111 
112       // TODO(bazel-team): put this in getActionContexts (key=SpawnActionContext.class) instead
113       contexts.put("", options.spawnStrategy);
114 
115       return ImmutableMap.copyOf(contexts);
116     }
117 
118     @Override
119     public Map<Class<? extends ActionContext>, String> getActionContexts() {
120       return ImmutableMap.<Class<? extends ActionContext>, String>builder()
121           .put(CppCompileActionContext.class, "")
122           .put(CppLinkActionContext.class, "")
123           .put(IncludeScanningContext.class, "")
124           .put(FileWriteActionContext.class, "")
125           .put(WriteAdbArgsActionContext.class, "")
126           .build();
127     }
128   }
129 
130   private BlazeRuntime runtime;
131   private OptionsProvider optionsProvider;
132 
133   @Override
134   public void beforeCommand(BlazeRuntime blazeRuntime, Command command) {
135     this.runtime = blazeRuntime;
136     runtime.getEventBus().register(this);
137   }
138 
139   @Override
140   public Iterable<Class<? extends OptionsBase>> getCommandOptions(Command command) {
141     return command.builds()
142         ? ImmutableList.<Class<? extends OptionsBase>>of(BazelExecutionOptions.class)
143         : ImmutableList.<Class<? extends OptionsBase>>of();
144   }
145 
146   @Override
147   public Iterable<ActionContextProvider> getActionContextProviders() {
148     return ImmutableList.<ActionContextProvider>of(new SimpleActionContextProvider(
149         new WriteAdbArgsActionContext(runtime.getClientEnv().get("HOME"))));
150   }
151 
152   @Override
153   public Iterable<ActionContextConsumer> getActionContextConsumers() {
154     return ImmutableList.<ActionContextConsumer>of(new BazelActionContextConsumer(
155         optionsProvider.getOptions(BazelExecutionOptions.class)));
156   }
157 
158   @Subscribe
159   public void gotOptions(GotOptionsEvent event) {
160     optionsProvider = event.getOptions();
161   }
162 
163   @Override
164   public void initializeRuleClasses(ConfiguredRuleClassProvider.Builder builder) {
165     BazelRuleClassProvider.setup(builder);
166   }
167 
168   @Override
169   public Iterable<PrecomputedValue.Injected> getPrecomputedSkyframeValues() {
170     return ImmutableList.of(PrecomputedValue.injected(
171         GenQuery.QUERY_OUTPUT_FORMATTERS,
172         new Supplier<ImmutableList<OutputFormatter>>() {
173           @Override
174           public ImmutableList<OutputFormatter> get() {
175             return runtime.getQueryOutputFormatters();
176           }
177         }));
178   }
