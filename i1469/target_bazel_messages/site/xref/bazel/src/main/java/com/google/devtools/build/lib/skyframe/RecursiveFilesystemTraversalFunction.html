76     * Thrown when a dangling symlink is attempted to be dereferenced.
77     *
78     * <p>Note: this class is not identical to the one in com.google.devtools.build.lib.view.fileset
79     * and it's not easy to merge the two because of the dependency structure. The other one will

82    public static final class DanglingSymlinkException extends RecursiveFilesystemTraversalException {
83      public final String path;
84      public final String unresolvedLink;
85  
86      public DanglingSymlinkException(String path, String unresolvedLink) {
87        super("Found dangling symlink: " + path + ", unresolved path: ");
88        Preconditions.checkArgument(path != null && !path.isEmpty());
89        Preconditions.checkArgument(unresolvedLink != null && !unresolvedLink.isEmpty());
90        this.path = path;
91        this.unresolvedLink = unresolvedLink;
92      }
93  
94      public String getPath() {
95        return path;
96      }

99    /** Exception type thrown by {@link RecursiveFilesystemTraversalFunction#compute}. */
100   private static final class RecursiveFilesystemTraversalFunctionException extends
101       SkyFunctionException {
102     RecursiveFilesystemTraversalFunctionException(RecursiveFilesystemTraversalException e) {
103       super(e, Transience.PERSISTENT);
104     }
105   }

107   @Override
108   public SkyValue compute(SkyKey skyKey, Environment env)
109       throws RecursiveFilesystemTraversalFunctionException {
110     TraversalRequest traversal = (TraversalRequest) skyKey.argument();
111     try {
112       // Stat the traversal root.
113       FileInfo rootInfo = lookUpFileInfo(env, traversal);
114 
115       if (!rootInfo.type.exists()) {
116         // May be a dangling symlink or a non-existent file. Handle gracefully.
117         if (rootInfo.type.isSymlink()) {
118           return resultForDanglingSymlink(traversal.path, rootInfo);
119         } else {
120           return RecursiveFilesystemTraversalValue.EMPTY;
121         }
122       }
123 
124       if (rootInfo.type.isFile()) {
125         if (traversal.pattern == null
126             || traversal.pattern.matcher(
127                    rootInfo.realPath.getRelativePath().getPathString()).matches()) {
128           // The root is a file or a symlink to one.
129           return resultForFileRoot(traversal.path, rootInfo);
130         } else {
131           return RecursiveFilesystemTraversalValue.EMPTY;
132         }
133       }
134 
135       // Otherwise the root is a directory or a symlink to one.
136       PkgLookupResult pkgLookupResult = checkIfPackage(env, traversal, rootInfo);
137       traversal = pkgLookupResult.traversal;
138 
139       if (pkgLookupResult.isConflicting()) {
140         // The traversal was requested for an output directory whose root-relative path conflicts
141         // with a source package. We can't handle that, bail out.
142         throw new RecursiveFilesystemTraversalFunctionException(
143             new GeneratedPathConflictException(traversal));
144       } else if (pkgLookupResult.isPackage() && !traversal.skipTestingForSubpackage) {
145         // The traversal was requested for a directory that defines a package.
146         String msg = traversal.errorInfo + " crosses package boundary into package rooted at "
147             + traversal.path.getRelativePath().getPathString();
148         switch (traversal.crossPkgBoundaries) {
149           case CROSS:
150             // We are free to traverse the subpackage but we need to display a warning.
151             env.getListener().handle(new Event(EventKind.WARNING, null, msg));
152             break;
153           case DONT_CROSS:
154             // We cannot traverse the subpackage and should skip it silently. Return empty results.
155             return RecursiveFilesystemTraversalValue.EMPTY;
156           case REPORT_ERROR:
157             // We cannot traverse the subpackage and should complain loudly (display an error).
158             throw new RecursiveFilesystemTraversalFunctionException(
159                 new CannotCrossPackageBoundaryException(msg));
160           default:
161             throw new IllegalStateException(traversal.toString());
162         }
163       }
164 
165       // We are free to traverse this directory.
166       Collection<SkyKey> dependentKeys = createRecursiveTraversalKeys(env, traversal);
167       return resultForDirectory(traversal, rootInfo, traverseChildren(env, dependentKeys));
168     } catch (MissingDepException e) {
169       return null;
170     }
171   }

178   private static final class FileInfo {
179     final FileType type;
180     final FileStateValue metadata;
181     @Nullable final RootedPath realPath;
182     @Nullable final PathFragment unresolvedSymlinkTarget;
183 
184     FileInfo(FileType type, FileStateValue metadata, @Nullable RootedPath realPath,
185         @Nullable PathFragment unresolvedSymlinkTarget) {
186       this.type = Preconditions.checkNotNull(type);
187       this.metadata = Preconditions.checkNotNull(metadata);
188       this.realPath = realPath;
189       this.unresolvedSymlinkTarget = unresolvedSymlinkTarget;
190     }
191 
192     @Override
193     public String toString() {
194       if (type.isSymlink()) {
195         return String.format("(%s: link_value=%s, real_path=%s)", type,
196             unresolvedSymlinkTarget.getPathString(), realPath);
197       } else {
198         return String.format("(%s: real_path=%s)", type, realPath);
199       }
200     }
201   }

203   private static FileInfo lookUpFileInfo(Environment env, TraversalRequest traversal)
204       throws MissingDepException {
205     // Stat the file.
206     FileValue fileValue = (FileValue) getDependentSkyValue(env, FileValue.key(traversal.path));
207     if (fileValue.exists()) {
208       // If it exists, it may either be a symlink or a file/directory.
209       PathFragment unresolvedLinkTarget = null;
210       FileType type = null;
211       if (fileValue.isSymlink()) {
212         unresolvedLinkTarget = fileValue.getUnresolvedLinkTarget();
213         type = fileValue.isDirectory() ? FileType.SYMLINK_TO_DIRECTORY : FileType.SYMLINK_TO_FILE;
214       } else {
215         type = fileValue.isDirectory() ? FileType.DIRECTORY : FileType.FILE;
216       }
217       return new FileInfo(type, fileValue.realFileStateValue(),
218           fileValue.realRootedPath(), unresolvedLinkTarget);
219     } else {
220       // If it doesn't exist, or it's a dangling symlink, we still want to handle that gracefully.
221       return new FileInfo(
222           fileValue.isSymlink() ? FileType.DANGLING_SYMLINK : FileType.NONEXISTENT,
223           fileValue.realFileStateValue(), null,
224           fileValue.isSymlink() ? fileValue.getUnresolvedLinkTarget() : null);
225     }
226   }

228   private static final class PkgLookupResult {
229     private enum Type {
230       CONFLICT, DIRECTORY, PKG
231     }
232 
233     private final Type type;
234     final TraversalRequest traversal;
235     final FileInfo rootInfo;
236 
237     /** Result for a generated directory that conflicts with a source package. */
238     static PkgLookupResult conflict(TraversalRequest traversal, FileInfo rootInfo) {
239       return new PkgLookupResult(Type.CONFLICT, traversal, rootInfo);
240     }
241 
242     /** Result for a source or generated directory (not a package). */
243     static PkgLookupResult directory(TraversalRequest traversal, FileInfo rootInfo) {
244       return new PkgLookupResult(Type.DIRECTORY, traversal, rootInfo);
245     }
246 
247     /** Result for a package, i.e. a directory  with a BUILD file. */
248     static PkgLookupResult pkg(TraversalRequest traversal, FileInfo rootInfo) {
249       return new PkgLookupResult(Type.PKG, traversal, rootInfo);
250     }
251 
252     private PkgLookupResult(Type type, TraversalRequest traversal, FileInfo rootInfo) {
253       this.type = Preconditions.checkNotNull(type);
254       this.traversal = Preconditions.checkNotNull(traversal);
255       this.rootInfo = Preconditions.checkNotNull(rootInfo);
256     }
257 
258     boolean isPackage() {
259       return type == Type.PKG;
260     }
261 
262     boolean isConflicting() {
263       return type == Type.CONFLICT;
264     }
265 
266     @Override
267     public String toString() {
268       return String.format("(%s: info=%s, traversal=%s)", type, rootInfo, traversal);
269     }
270   }

272   /**
273    * Checks whether the {@code traversal}'s path refers to a package directory.
274    *
275    * @return the result of the lookup; it contains potentially new {@link TraversalRequest} and
276    *     {@link FileInfo} so the caller should use these instead of the old ones (this happens when
277    *     a package is found, but under a different root than expected)
278    */
279   private static PkgLookupResult checkIfPackage(Environment env, TraversalRequest traversal,
280       FileInfo rootInfo) throws MissingDepException {
281     Preconditions.checkArgument(rootInfo.type.exists() && !rootInfo.type.isFile(),
282         "{%s} {%s}", traversal, rootInfo);
283     PackageLookupValue pkgLookup = (PackageLookupValue) getDependentSkyValue(env,
284         PackageLookupValue.key(traversal.path.getRelativePath()));
285 
286     if (pkgLookup.packageExists()) {
287       if (traversal.isGenerated) {
288         // The traversal's root was a generated directory, but its root-relative path conflicts with
289         // an existing package.
290         return PkgLookupResult.conflict(traversal, rootInfo);
291       } else {
292         // The traversal's root was a source directory and it defines a package.
293         Path pkgRoot = pkgLookup.getRoot();
294         if (!pkgRoot.equals(traversal.path.getRoot())) {
295           // However the root of this package is different from what we expected. stat() the real
296           // BUILD file of that package.
297           traversal = traversal.forChangedRootPath(pkgRoot);
298           rootInfo = lookUpFileInfo(env, traversal);
299           Verify.verify(rootInfo.type.exists(), "{%s} {%s}", traversal, rootInfo);
300         }
301         return PkgLookupResult.pkg(traversal, rootInfo);
302       }
303     } else {
304       // The traversal's root was a directory (source or generated one), no package exists under the
305       // same root-relative path.
306       return PkgLookupResult.directory(traversal, rootInfo);
307     }
308   }

310   /**
311    * List the directory and create {@code SkyKey}s to request contents of its children recursively.
312    *
313    * <p>The returned keys are of type {@link SkyFunctions#RECURSIVE_FILESYSTEM_TRAVERSAL}.
314    */
315   private static Collection<SkyKey> createRecursiveTraversalKeys(Environment env,
316       TraversalRequest traversal) throws MissingDepException {
317     // Use the traversal's path, even if it's a symlink. The contents of the directory, as listed
318     // in the result, must be relative to it.
319     DirectoryListingValue dirListing = (DirectoryListingValue) getDependentSkyValue(env,
320         DirectoryListingValue.key(traversal.path));
321 
322     List<SkyKey> result = new ArrayList<>();
323     for (Dirent dirent : dirListing.getDirents()) {
324       RootedPath childPath = RootedPath.toRootedPath(traversal.path.getRoot(),
325           traversal.path.getRelativePath().getRelative(dirent.getName()));
326       TraversalRequest childTraversal = traversal.forChildEntry(childPath);
327       result.add(RecursiveFilesystemTraversalValue.key(childTraversal));
328     }
329     return result;
330   }

332   /**
333    * Creates result for a dangling symlink.
334    *
335    * @param linkName path to the symbolic link
336    * @param info the {@link FileInfo} associated with the link file
337    */
338   private static RecursiveFilesystemTraversalValue resultForDanglingSymlink(RootedPath linkName,
339       FileInfo info) {
340     Preconditions.checkState(info.type.isSymlink() && !info.type.exists(), "{%s} {%s}", linkName,
341         info.type);
342     return RecursiveFilesystemTraversalValue.of(
343         ResolvedFile.danglingSymlink(linkName, info.unresolvedSymlinkTarget, info.metadata));
344   }

346   /**
347    * Creates results for a file or for a symlink that points to one.
348    *
349    * <p>A symlink may be direct (points to a file) or transitive (points at a direct or transitive
350    * symlink).
351    */
352   private static RecursiveFilesystemTraversalValue resultForFileRoot(RootedPath path,
353       FileInfo info) {
354     Preconditions.checkState(info.type.isFile() && info.type.exists(), "{%s} {%s}", path,
355         info.type);
356     if (info.type.isSymlink()) {
357       return RecursiveFilesystemTraversalValue.of(ResolvedFile.symlinkToFile(info.realPath, path,
358           info.unresolvedSymlinkTarget, info.metadata));
359     } else {
360       return RecursiveFilesystemTraversalValue.of(ResolvedFile.regularFile(path, info.metadata));
361     }
362   }
363 
364   private static RecursiveFilesystemTraversalValue resultForDirectory(TraversalRequest traversal,
365       FileInfo rootInfo, Collection<RecursiveFilesystemTraversalValue> subdirTraversals) {
366     // Collect transitive closure of files in subdirectories.
367     NestedSetBuilder<ResolvedFile> paths = NestedSetBuilder.stableOrder();
368     for (RecursiveFilesystemTraversalValue child : subdirTraversals) {
369       paths.addTransitive(child.getTransitiveFiles());
370     }
371     ResolvedFile root;
372     if (rootInfo.type.isSymlink()) {
373       root = ResolvedFile.symlinkToDirectory(rootInfo.realPath, traversal.path,
374           rootInfo.unresolvedSymlinkTarget, rootInfo.metadata);
375       paths.add(root);
376     } else {
377       root = ResolvedFile.directory(rootInfo.realPath);
378     }
379     return RecursiveFilesystemTraversalValue.of(root, paths.build());
380   }

382   private static SkyValue getDependentSkyValue(Environment env, SkyKey key)
383       throws MissingDepException {
384     SkyValue value = env.getValue(key);
385     if (env.valuesMissing()) {
386       throw new MissingDepException();
387     }
388     return value;
389   }

391   /**
392    * Requests Skyframe to compute the dependent values and returns them.
393    *
394    * <p>The keys must all be {@link SkyFunctions#RECURSIVE_FILESYSTEM_TRAVERSAL} keys.
395    */
396   private static Collection<RecursiveFilesystemTraversalValue> traverseChildren(
397       Environment env, Iterable<SkyKey> keys)
398       throws MissingDepException {
399     Map<SkyKey, SkyValue> values = env.getValues(keys);
400     if (env.valuesMissing()) {
401       throw new MissingDepException();
402     }
403     return Collections2.transform(values.values(),
404         new Function<SkyValue, RecursiveFilesystemTraversalValue>() {
405           @Override
406           public RecursiveFilesystemTraversalValue apply(SkyValue input) {
407             return (RecursiveFilesystemTraversalValue) input;
408           }
409         });
410   }

412   /** Type information about the filesystem entry residing at a path. */
413   enum FileType {
414     /** A regular file. */
415     FILE {
416       @Override boolean isFile() { return true; }
417       @Override boolean exists() { return true; }
418       @Override public String toString() { return "<f>"; }
419     },
420     /**
421      * A symlink to a regular file.
422      *
423      * <p>The symlink may be direct (points to a non-symlink (here a file)) or it may be transitive
424      * (points to a direct or transitive symlink).
425      */
426     SYMLINK_TO_FILE {
427       @Override boolean isFile() { return true; }
428       @Override boolean isSymlink() { return true; }
429       @Override boolean exists() { return true; }
430       @Override public String toString() { return "<lf>"; }
431     },
432     /** A directory. */
433     DIRECTORY {
434       @Override boolean isDirectory() { return true; }
435       @Override boolean exists() { return true; }
436       @Override public String toString() { return "<d>"; }
437     },
438     /**
439      * A symlink to a directory.
440      *
441      * <p>The symlink may be direct (points to a non-symlink (here a directory)) or it may be
442      * transitive (points to a direct or transitive symlink).
443      */
444     SYMLINK_TO_DIRECTORY {
445       @Override boolean isDirectory() { return true; }
446       @Override boolean isSymlink() { return true; }
447       @Override boolean exists() { return true; }
448       @Override public String toString() { return "<ld>"; }
449     },
450     /** A dangling symlink, i.e. one whose target is known not to exist. */
451     DANGLING_SYMLINK {
452       @Override boolean isFile() { throw new UnsupportedOperationException(); }
453       @Override boolean isDirectory() { throw new UnsupportedOperationException(); }
454       @Override boolean isSymlink() { return true; }
455       @Override public String toString() { return "<l?>"; }
456     },
457     /** A path that does not exist or should be ignored. */
458     NONEXISTENT {
459       @Override public String toString() { return "<?>"; }
460     };
461 
462     boolean isFile() { return false; }
463     boolean isDirectory() { return false; }
464     boolean isSymlink() { return false; }
465     boolean exists() { return false; }
466     @Override public abstract String toString();
467   }
