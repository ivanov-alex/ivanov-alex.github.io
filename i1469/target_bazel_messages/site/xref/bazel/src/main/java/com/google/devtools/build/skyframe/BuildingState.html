32  /**
33   * Data the NodeEntry uses to maintain its state before it is done building. It allows the
34   * {@link NodeEntry} to keep the current state of the entry across invalidation and successive
35   * evaluations. A done node does not contain any of this data. However, if a node is marked dirty,
36   * its entry acquires a new {@code BuildingState} object, which persists until it is done again.
37   *
38   * <p>This class should be considered a private inner class of {@link NodeEntry} -- no other
39   * classes should instantiate a {@code BuildingState} object or call any of its methods directly.
40   * It is in a separate file solely to keep the {@link NodeEntry} class readable. In particular, the
41   * caller must synchronize access to this class.
42   *
43   * <p>This class is public only for the benefit of alternative graph implementations outside of the
44   * package.
45   */
46  @ThreadCompatible
47  public class BuildingState {
48    /**
49     * During its life, a node can go through states as follows:
50     * <ol>
51     * <li>Non-existent
52     * <li>Just created ({@code evaluating} is false)
53     * <li>Evaluating ({@code evaluating} is true)
54     * <li>Done (meaning this buildingState object is null)
55     * <li>Just created (when it is dirtied during evaluation)
56     * <li>Reset (just before it is re-evaluated)
57     * <li>Evaluating
58     * <li>Done
59     * </ol>
60     *
61     * <p>The "just created" state is there to allow the {@link EvaluableGraph#createIfAbsent} and
62     * {@link NodeEntry#addReverseDepAndCheckIfDone} methods to be separate. All callers have to
63     * call both methods in that order if they want to create a node. The second method calls
64     * {@link #startEvaluating}, which transitions the current node to the "evaluating" state and
65     * returns true only the first time it was called. A caller that gets "true" back from that call
66     * must start the evaluation of this node, while any subsequent callers must not.
67     *
68     * <p>An entry is set to "evaluating" as soon as it is scheduled for evaluation. Thus, even a
69     * node that is never actually built (for instance, a dirty node that is verified as clean) is
70     * in the "evaluating" state until it is done.
71     */
72    private boolean evaluating = false;
73  
74    /**
75     * The state of a dirty node. A node is marked dirty in the BuildingState constructor, and goes
76     * into either the state {@link DirtyState#CHECK_DEPENDENCIES} or {@link DirtyState#REBUILDING},
77     * depending on whether the caller specified that the node was itself changed or not. A non-null
78     * {@code dirtyState} indicates that the node {@link #isDirty} in some way.
79     */
80    private DirtyState dirtyState = null;
81  
82    /**
83     * The number of dependencies that are known to be done in a {@link NodeEntry}. There is a
84     * potential check-then-act race here, so we need to make sure that when this is increased, we
85     * always check if the new value is equal to the number of required dependencies, and if so, we
86     * must re-schedule the node for evaluation.
87     *
88     * <p>There are two potential pitfalls here: 1) If multiple dependencies signal this node in
89     * close succession, this node should be scheduled exactly once. 2) If a thread is still working
90     * on this node, it should not be scheduled.
91     *
92     * <p>The first problem is solved by the {@link #signalDep} method, which also returns if the
93     * node needs to be re-scheduled, and ensures that only one thread gets a true return value.
94     *
95     * <p>The second problem is solved by first adding the newly discovered deps to a node's
96     * {@link #directDeps}, and then looping through the direct deps and registering this node as a
97     * reverse dependency. This ensures that the signaledDeps counter can only reach
98     * {@link #directDeps}.size() on the very last iteration of the loop, i.e., the thread is not
99     * working on the node anymore. Note that this requires that there is no code after the loop in
100    * {@code ParallelEvaluator.Evaluate#run}.
101    */
102   private int signaledDeps = 0;
103 
104   /**
105    * Direct dependencies discovered during the build. They will be written to the immutable field
106    * {@code ValueEntry#directDeps} and the dependency group data to {@code ValueEntry#groupData}
107    * once the node is finished building. {@link SkyFunction}s can request deps in groups, and these
108    * groupings are preserved in this field.
109    */
110   private final GroupedList<SkyKey> directDeps = new GroupedList<>();
111 
112   /**
113    * The set of reverse dependencies that are registered before the node has finished building.
114    * Upon building, these reverse deps will be signaled and then stored in the permanent
115    * {@code ValueEntry#reverseDeps}.
116    */
117   // TODO(bazel-team): Remove this field. With eager invalidation, all direct deps on this dirty
118   // node will be removed by the time evaluation starts, so reverse deps to signal can just be
119   // reverse deps in the main ValueEntry object.
120   private Object reverseDepsToSignal = ImmutableList.of();
121   private List<SkyKey> reverseDepsToRemove = null;
122   private boolean reverseDepIsSingleObject = false;

159   /**
160    * The dependencies requested (with group markers) last time the node was built (and below, the
161    * value last time the node was built). They will be compared to dependencies requested on this
162    * build to check whether this node has changed in {@link NodeEntry#setValue}. If they are null,
163    * it means that this node is being built for the first time. See {@link #directDeps} for more on
164    * dependency group storage.
165    */
166   private final GroupedList<SkyKey> lastBuildDirectDeps;
167 
168   /**
169    * Which child should be re-evaluated next in the process of determining if this entry needs to
170    * be re-evaluated. Used by {@link #getNextDirtyDirectDeps} and {@link #signalDep(boolean)}.
171    */
172   private Iterator<Iterable<SkyKey>> dirtyDirectDepIterator = null;
173 
174   BuildingState() {
175     lastBuildDirectDeps = null;
176   }
177 
178   protected BuildingState(boolean isChanged, GroupedList<SkyKey> lastBuildDirectDeps) {
179     this.lastBuildDirectDeps = lastBuildDirectDeps;
180     Preconditions.checkState(isChanged || !this.lastBuildDirectDeps.isEmpty(),
181         "%s is being marked dirty, not changed, but has no children that could have dirtied it",
182         this);
183     dirtyState = isChanged ? DirtyState.REBUILDING
184         : DirtyState.CHECK_DEPENDENCIES;
185     if (dirtyState == DirtyState.CHECK_DEPENDENCIES) {
186       // We need to iterate through the deps to see if they have changed. Initialize the iterator.
187       dirtyDirectDepIterator = lastBuildDirectDeps.iterator();
188     }
189   }
190 
191   static BuildingState newDirtyState(boolean isChanged,
192       GroupedList<SkyKey> lastBuildDirectDeps, SkyValue lastBuildValue) {
193     return new DirtyBuildingState(isChanged, lastBuildDirectDeps, lastBuildValue);
194   }
195 
196   void markChanged() {
197     Preconditions.checkState(isDirty(), this);
198     Preconditions.checkState(!isChanged(), this);
199     Preconditions.checkState(!evaluating, this);
200     dirtyState = DirtyState.REBUILDING;
201   }
202 
203   void forceChanged() {
204     Preconditions.checkState(isDirty(), this);
205     Preconditions.checkState(!isChanged(), this);
206     Preconditions.checkState(evaluating, this);
207     Preconditions.checkState(isReady(), this);
208     dirtyState = DirtyState.REBUILDING;
209   }
210 
211   /**
212    * Returns whether all known children of this node have signaled that they are done.
213    */
214   boolean isReady() {
215     int directDepsSize = directDeps.size();
216     Preconditions.checkState(signaledDeps <= directDepsSize, "%s %s", directDepsSize, this);
217     return signaledDeps == directDepsSize;
218   }

220   /**
221    * Returns true if the entry is marked dirty, meaning that at least one of its transitive
222    * dependencies is marked changed.
223    *
224    * @see NodeEntry#isDirty()
225    */
226   boolean isDirty() {
227     return dirtyState != null;
228   }
229 
230   /**
231    * Returns true if the entry is known to require re-evaluation.
232    *
233    * @see NodeEntry#isChanged()
234    */
235   boolean isChanged() {
236     return dirtyState == DirtyState.REBUILDING;
237   }
238 
239   private boolean rebuilding() {
240     return dirtyState == DirtyState.REBUILDING;
241   }
242 
243   /**
244    * Helper method to assert that node has finished building, as far as we can tell. We would
245    * actually like to check that the node has been evaluated, but that is not available in
246    * this context.
247    */
248   private void checkNotProcessing() {
249     Preconditions.checkState(evaluating, "not started building %s", this);
250     Preconditions.checkState(!isDirty() || dirtyState == DirtyState.VERIFIED_CLEAN
251         || rebuilding(), "not done building %s", this);
252     Preconditions.checkState(isReady(), "not done building %s", this);
253   }
254 
255   /**
256    * Puts the node in the "evaluating" state if it is not already in it. Returns true if the
257    * node wasn't already evaluating and false otherwise. Should only be called by
258    * {@link NodeEntry#addReverseDepAndCheckIfDone}.
259    */
260   boolean startEvaluating() {
261     boolean result = !evaluating;
262     evaluating = true;
263     return result;
264   }
265 
266   /**
267    * Increments the number of children known to be finished. Returns true if the number of children
268    * finished is equal to the number of known children.
269    *
270    * <p>If the node is dirty and checking its deps for changes, this also updates {@link
271    * #dirtyState} as needed -- {@link DirtyState#REBUILDING} if the child has changed,
272    * and {@link DirtyState#VERIFIED_CLEAN} if the child has not changed and this was the
273    * last child to be checked (as determined by {@link #dirtyDirectDepIterator} == null, isReady(),
274    * and a flag set in {@link #getNextDirtyDirectDeps}).
275    *
276    * @see NodeEntry#signalDep(Version)
277    */
278   boolean signalDep(boolean childChanged) {
279     signaledDeps++;
280     if (isDirty() && !rebuilding()) {
281       // Synchronization isn't needed here because the only caller is ValueEntry, which does it
282       // through the synchronized method signalDep(long).
283       if (childChanged) {
284         dirtyState = DirtyState.REBUILDING;
285       } else if (dirtyState == DirtyState.CHECK_DEPENDENCIES && isReady()
286           && dirtyDirectDepIterator == null) {
287         // No other dep already marked this as REBUILDING, no deps outstanding, and this was
288         // the last block of deps to be checked.
289         dirtyState = DirtyState.VERIFIED_CLEAN;
290       }
291     }
292     return isReady();
293   }
294 
295   /**
296    * Returns true if {@code newValue}.equals the value from the last time this node was built, and
297    * the deps requested during this evaluation are exactly those requested the last time this node
298    * was built, in the same order. Should only be used by {@link NodeEntry#setValue}.
299    */
300   boolean unchangedFromLastBuild(SkyValue newValue) {
301     checkNotProcessing();
302     return getLastBuildValue().equals(newValue) && lastBuildDirectDeps.equals(directDeps);
303   }
304 
305   boolean noDepsLastBuild() {
306     return lastBuildDirectDeps.isEmpty();
307   }

309   protected SkyValue getLastBuildValue() {
310     // Default BuildingState isn't dirty.
311     throw new UnsupportedOperationException(this.toString());
312   }

314   /**
315    * Gets the current state of checking this dirty entry to see if it must be re-evaluated. Must be
316    * called each time evaluation of a dirty entry starts to find the proper action to perform next,
317    * as enumerated by {@link DirtyState}.
318    *
319    * @see NodeEntry#getDirtyState()
320    */
321   DirtyState getDirtyState() {
322     // Entry may not be ready if being built just for its errors.
323     Preconditions.checkState(isDirty(), "must be dirty to get dirty state %s", this);
324     Preconditions.checkState(evaluating, "must be evaluating to get dirty state %s", this);
325     return dirtyState;
326   }
327 
328   /**
329    * Gets the next children to be re-evaluated to see if this dirty node needs to be re-evaluated.
330    *
331    * <p>If this is the last group of children to be checked, then sets {@link
332    * #dirtyDirectDepIterator} to null so that the final call to {@link #signalDep(boolean)} will
333    * know to mark this entry as {@link DirtyState#VERIFIED_CLEAN} if no deps have changed.
334    *
335    * See {@link NodeEntry#getNextDirtyDirectDeps}.
336    */
337   Collection<SkyKey> getNextDirtyDirectDeps() {
338     Preconditions.checkState(isDirty(), this);
339     Preconditions.checkState(dirtyState == DirtyState.CHECK_DEPENDENCIES, this);
340     Preconditions.checkState(evaluating, this);
341     List<SkyKey> nextDeps = ImmutableList.copyOf(dirtyDirectDepIterator.next());
342     if (!dirtyDirectDepIterator.hasNext()) {
343       // Done checking deps. If this last group is clean, the state will become VERIFIED_CLEAN.
344       dirtyDirectDepIterator = null;
345     }
346     return nextDeps;
347   }

399   /**
400    * Removes a set of deps from the set of known direct deps. This is complicated by the need
401    * to maintain the group data. If we remove a dep that ended a group, then its predecessor's
402    * group data must be changed to indicate that it now ends the group.
403    *
404    * @see NodeEntry#removeUnfinishedDeps
405    */
406   void removeDirectDeps(Set<SkyKey> unfinishedDeps) {
407     directDeps.remove(unfinishedDeps);
408   }

410   protected ToStringHelper getStringHelper() {
411     return MoreObjects.toStringHelper(this)
412         .add("evaluating", evaluating)
413         .add("dirtyState", dirtyState)
414         .add("signaledDeps", signaledDeps)
415         .add("directDeps", directDeps)
416         .add("reverseDepsToSignal", REVERSE_DEPS_UTIL.toString(this))
417         .add("lastBuildDirectDeps", lastBuildDirectDeps)
418         .add("dirtyDirectDepIterator", dirtyDirectDepIterator);
419   }
420   @Override
421   public String toString() {
422     return getStringHelper().toString();
423   }

425   private static class DirtyBuildingState extends BuildingState {
426     private final SkyValue lastBuildValue;
427     private DirtyBuildingState(boolean isChanged, GroupedList<SkyKey> lastBuildDirectDeps,
428         SkyValue lastBuildValue) {
429       super(isChanged, lastBuildDirectDeps);
430       this.lastBuildValue = Preconditions.checkNotNull(lastBuildValue);
431     }
432 
433     @Override
434     protected SkyValue getLastBuildValue() {
435       return lastBuildValue;
436     }
437 
438     @Override
439     public String toString() {
440       return getStringHelper().add("lastBuildValue", lastBuildValue).toString();
441     }
