33  /**
34   * Rule definition for objc_proto_library.
35   *
36   * This is a temporary rule until it is better known how to support proto_library rules.
37   */
38  public class ObjcProtoLibraryRule implements RuleDefinition {
39    static final String COMPILE_PROTOS_ATTR = "$googlemac_proto_compiler";
40    static final String PROTO_SUPPORT_ATTR = "$googlemac_proto_compiler_support";
41    static final String OPTIONS_FILE_ATTR = "options_file";
42    static final String OUTPUT_CPP_ATTR = "output_cpp";
43    static final String USE_OBJC_HEADER_NAMES_ATTR = "use_objc_header_names";
44    static final String PER_PROTO_INCLUDES = "per_proto_includes";
45    static final String LIBPROTOBUF_ATTR = "$lib_protobuf";
46  
47    @Override
48    public RuleClass build(Builder builder, final RuleDefinitionEnvironment env) {
49      return builder
50          .requiresConfigurationFragments(ObjcConfiguration.class)
51          /* <!-- #BLAZE_RULE(objc_proto_library).ATTRIBUTE(deps) -->
52          The directly depended upon proto_library rules.
53          ${SYNOPSIS}
54          <!-- #END_BLAZE_RULE.ATTRIBUTE -->*/
55          .override(attr("deps", LABEL_LIST)
56              .allowedRuleClasses("proto_library", "filegroup")
57              .legacyAllowAnyFileType())
58          /* <!-- #BLAZE_RULE(objc_proto_library).ATTRIBUTE(options_file) -->
59          Optional options file to apply to protos which affects compilation (e.g. class
60          whitelist/blacklist settings).
61          ${SYNOPSIS}
62          <!-- #END_BLAZE_RULE.ATTRIBUTE -->*/
63          .add(attr(OPTIONS_FILE_ATTR, LABEL).legacyAllowAnyFileType().singleArtifact().cfg(HOST))
64          /* <!-- #BLAZE_RULE(objc_proto_library).ATTRIBUTE(output_cpp) -->
65          If true, output C++ rather than ObjC.
66          ${SYNOPSIS}
67          <!-- #END_BLAZE_RULE.ATTRIBUTE -->*/
68          .add(attr(OUTPUT_CPP_ATTR, BOOLEAN).value(false))
69          /* <!-- #BLAZE_RULE(objc_proto_library).ATTRIBUTE(use_objc_header_names) -->
70          If true, output headers with .pbobjc.h, rather than .pb.h.
71          ${SYNOPSIS}
72          <!-- #END_BLAZE_RULE.ATTRIBUTE -->*/
73          .add(attr(PER_PROTO_INCLUDES, BOOLEAN).value(false))
74          /* <!-- #BLAZE_RULE(objc_proto_library).ATTRIBUTE(per_proto_includes) -->
75          If true, always add all directories to objc_library includes,
76          overriding --noobjc_per_proto_includes.
77          ${SYNOPSIS}
78          <!-- #END_BLAZE_RULE.ATTRIBUTE -->*/
79          .add(attr(COMPILE_PROTOS_ATTR, LABEL)
80              .allowedFileTypes(FileType.of(".py"))
81              .cfg(HOST)
82              .singleArtifact()
83              .value(env.getLabel("//tools/objc:compile_protos")))
84          .add(attr(PROTO_SUPPORT_ATTR, LABEL)
85              .legacyAllowAnyFileType()
86              .cfg(HOST)
87              .value(env.getLabel("//tools/objc:proto_support")))
88          .add(attr(USE_OBJC_HEADER_NAMES_ATTR, BOOLEAN).value(false))
89          .add(attr(LIBPROTOBUF_ATTR, LABEL).allowedRuleClasses("objc_library")
90              .value(new ComputedDefault(OUTPUT_CPP_ATTR) {
91                @Override
92                public Object getDefault(AttributeMap rule) {
93                  return rule.get(OUTPUT_CPP_ATTR, Type.BOOLEAN)
94                      ? env.getLabel("//external:objc_proto_cpp_lib")
95                      : env.getLabel("//external:objc_proto_lib");
96                }
97              }))
98          .add(attr("$xcodegen", LABEL).cfg(HOST).exec()
99              .value(env.getLabel("//tools/objc:xcodegen")))
100         .build();
101   }
102 
103   @Override
104   public Metadata getMetadata() {
105     return RuleDefinition.Metadata.builder()
106         .name("objc_proto_library")
107         .factoryClass(ObjcProtoLibrary.class)
108         .ancestors(BaseRuleClasses.RuleBase.class)
109         .build();
110   }
111 }
