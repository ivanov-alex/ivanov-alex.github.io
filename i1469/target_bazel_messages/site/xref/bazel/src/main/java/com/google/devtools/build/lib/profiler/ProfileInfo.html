59    public static final class AggregateAttr {
60      public final int count;
61      public final long totalTime;

63      AggregateAttr(int count, long totalTime) {
64        this.count = count;
65        this.totalTime = totalTime;
66      }

69    /**
70     * Immutable compact representation of the Map<ProfilerTask, AggregateAttr>.
71     */
72    static final class CompactStatistics {
73      final byte[] content;
74  
75      CompactStatistics(byte[] content) {
76        this.content = content;
77      }
78  
79      /**
80       * Create compact task statistic instance using provided array.
81       * Array length must exactly match ProfilerTask value space.
82       * Each statistic is stored in the array according to the ProfilerTask
83       * value ordinal() number. Absent statistics are represented by null.
84       */
85      CompactStatistics(AggregateAttr[] stats) {
86        Preconditions.checkArgument(stats.length == TASK_COUNT);
87        ByteBuffer sink = ByteBuffer.allocate(TASK_COUNT * (1 + 5 + 10));
88        for (int i = 0; i < TASK_COUNT; i++) {
89          if (stats[i] != null && stats[i].count > 0) {
90            sink.put((byte) i);
91            VarInt.putVarInt(stats[i].count, sink);
92            VarInt.putVarLong(stats[i].totalTime, sink);
93          }
94        }
95        content = sink.position() > 0 ? Arrays.copyOf(sink.array(), sink.position()) : null;
96      }
97  
98      boolean isEmpty() { return content == null; }
99  
100     /**
101      * Converts instance back into AggregateAttr[TASK_COUNT]. See
102      * constructor documentation for more information.
103      */
104     AggregateAttr[] toArray() {
105       AggregateAttr[] stats = new AggregateAttr[TASK_COUNT];
106       if (!isEmpty()) {
107         ByteBuffer source = ByteBuffer.wrap(content);
108         while (source.hasRemaining()) {
109           byte id = source.get();
110           int count = VarInt.getVarInt(source);
111           long time = VarInt.getVarLong(source);
112           stats[id] = new AggregateAttr(count, time);
113         }
114       }
115       return stats;
116     }
117 
118     /**
119      * Returns AggregateAttr instance for the given ProfilerTask value.
120      */
121     AggregateAttr getAttr(ProfilerTask task) {
122       if (isEmpty()) { return ZERO; }
123       ByteBuffer source = ByteBuffer.wrap(content);
124       byte id = (byte) task.ordinal();
125       while (source.hasRemaining()) {
126         if (id == source.get()) {
127           int count = VarInt.getVarInt(source);
128           long time = VarInt.getVarLong(source);
129           return new AggregateAttr(count, time);
130         } else {
131           VarInt.getVarInt(source);
132           VarInt.getVarLong(source);
133         }
134       }
135       return ZERO;
136     }
137 
138     /**
139      * Returns cumulative time stored in this instance across whole
140      * ProfilerTask dimension.
141      */
142     long getTotalTime() {
143       if (isEmpty()) { return 0; }
144       ByteBuffer source = ByteBuffer.wrap(content);
145       long totalTime = 0;
146       while (source.hasRemaining()) {
147         source.get();
148         VarInt.getVarInt(source);
149         totalTime += VarInt.getVarLong(source);
150       }
151       return totalTime;
152     }
153   }

156    * Container for the profile record information.

161   public final class Task implements Comparable<Task> {
162     public final long threadId;
163     public final int id;
164     public final int parentId;
165     public final long startTime;
166     public final long duration;
167     public final ProfilerTask type;

169     // Contains statistic for a task and all subtasks. Populated only for root tasks.
170     CompactStatistics aggregatedStats = null;
171     // Subtasks are stored as an array for performance and memory utilization
172     // reasons (we can easily deal with millions of those objects).
173     public Task[] subtasks = NO_TASKS;
174     final int descIndex;
175     // Reference to the related task (e.g. ACTION_GRAPH->ACTION task relation).
176     private Task relatedTask;
177 
178     Task(long threadId, int id, int parentId, long startTime, long duration,
179          ProfilerTask type, int descIndex, CompactStatistics stats) {
180       this.threadId = threadId;
181       this.id = id;
182       this.parentId = parentId;
183       this.startTime = startTime;
184       this.duration = duration;
185       this.type = type;
186       this.descIndex = descIndex;
187       this.stats = stats;
188       relatedTask = null;
189     }
190 
191     public String getDescription() {
192       return descriptionList.get(descIndex);
193     }
194 
195     public boolean hasStats() {
196       return !stats.isEmpty();
197     }
198 
199     public long getInheritedDuration() {
200       return stats.getTotalTime();
201     }

203     public AggregateAttr[] getStatAttrArray() {
204       Preconditions.checkNotNull(stats);
205       return stats.toArray();
206     }

208     private void combineStats(int[] counts, long[] duration) {
209       int ownIndex = type.ordinal();
210       if (parentId != 0) {
211         // Parent task already accounted for this task total duration. We need to adjust
212         // for the inherited duration.
213         duration[ownIndex] -= getInheritedDuration();
214       }
215       AggregateAttr[] ownStats = stats.toArray();
216       for (int i = 0; i < TASK_COUNT; i++) {
217         AggregateAttr attr = ownStats[i];
218         if (attr != null) {
219           counts[i] += attr.count;
220           duration[i] += attr.totalTime;
221         }
222       }
223       for (Task task : subtasks) {
224         task.combineStats(counts, duration);
225       }
226     }
227 
228     /**
229      * Calculates aggregated statistics covering all subtasks (including
230      * nested ones). Must be called only for parent tasks.
231      */
232     void calculateRootStats() {
233       Preconditions.checkState(parentId == 0);
234       int[] counts = new int[TASK_COUNT];
235       long[] duration = new long[TASK_COUNT];
236       combineStats(counts, duration);
237       AggregateAttr[] statArray = ProfileInfo.createEmptyStatArray();
238       for (int i = 0; i < TASK_COUNT; i++) {
239         statArray[i] = new AggregateAttr(counts[i], duration[i]);
240       }
241       this.aggregatedStats = new CompactStatistics(statArray);
242     }
243 
244     @Override
245     public boolean equals(Object o) {
246       return (o instanceof ProfileInfo.Task) && ((Task) o).id == this.id;
247     }
248 
249     @Override
250     public int hashCode() {
251       return this.id;
252     }
253 
254     @Override
255     public String toString() {
256       return type + "(" + id + "," + getDescription() + ")";
257     }

259     /**
260      * Tasks records by default sorted by their id. Since id was obtained using
261      * AtomicInteger, this comparison will correctly sort tasks in time-ascending
262      * order regardless of their origin thread.
263      */
264     @Override
265     public int compareTo(Task task) {
266       return this.id - task.id;
267     }

298   /**
299    * Helper class to create space-efficient task multimap, used to associate
300    * array of tasks with specific key.
301    */
302   private abstract static class TaskMapCreator<K> implements Comparator<Task> {
303     @Override
304     public abstract int compare(Task a, Task b);
305     public abstract K getKey(Task task);
306 
307     public Map<K, Task[]> createTaskMap(List<Task> taskList) {
308       // Created map usually will end up with thousands of entries, so we
309       // preinitialize it to the 10000.
310       Map<K, Task[]> taskMap = Maps.newHashMapWithExpectedSize(10000);
311       if (taskList.isEmpty()) {
312         return taskMap;
313       }
314       Task[] taskArray = taskList.toArray(new Task[taskList.size()]);
315       Arrays.sort(taskArray, this);
316       K key = getKey(taskArray[0]);
317       int start = 0;
318       for (int i = 0; i < taskArray.length; i++) {
319         K currentKey = getKey(taskArray[i]);
320         if (!key.equals(currentKey)) {
321           taskMap.put(key, Arrays.copyOfRange(taskArray, start, i));
322           key = currentKey;
323           start = i;
324         }
325       }
326       if (start < taskArray.length) {
327         taskMap.put(key, Arrays.copyOfRange(taskArray, start, taskArray.length));
328       }
329       return taskMap;
330     }
331   }

393   /**
394    * Initializes minimum internal data structures necessary to obtain individual
395    * task statistic. This method is sufficient to initialize data for dumping.
396    */
397   public void calculateStats() {
398     if (allTasksById.isEmpty()) {
399       return;
400     }
401 
402     Collections.sort(allTasksById);
403 
404     Map<Integer, Task[]> subtaskMap = new TaskMapCreator<Integer>() {
405       @Override
406       public int compare(Task a, Task b) {
407         return a.parentId != b.parentId ? a.parentId - b.parentId : a.compareTo(b);
408       }
409       @Override
410       public Integer getKey(Task task) { return task.parentId; }
411     }.createTaskMap(allTasksById);
412     for (Task task : allTasksById) {
413       Task[] subtasks = subtaskMap.get(task.id);
414       if (subtasks != null) {
415         task.subtasks = subtasks;
416       }
417     }
418     rootTasksById = Arrays.asList(subtaskMap.get(0));
419 
420     for (Task task : rootTasksById) {
421       task.calculateRootStats();
422       if (task.type == ProfilerTask.PHASE) {
423         if (!phaseTasks.isEmpty()) {
424           phaseTasks.get(phaseTasks.size() - 1).relatedTask = task;
425         }
426         phaseTasks.add(task);
427       }
428     }
429   }

432    * Analyzes task relationships and dependencies. Used for the detailed profile
433    * analysis.
434    */
435   public void analyzeRelationships() {
436     tasksByThread.putAll(new TaskMapCreator<Long>() {
437       @Override
438       public int compare(Task a, Task b) {
439         return a.threadId != b.threadId ? (a.threadId < b.threadId ? -1 : 1) : a.compareTo(b);
440       }
441       @Override
442       public Long getKey(Task task) { return task.threadId; }
443     }.createTaskMap(rootTasksById));
444 
445     buildDependencyMap();
446   }

448   /**
449    * Calculates cumulative time attributed to the specific task type.
450    * Expects to be called only for root (parentId = 0) tasks.
451    * calculateStats() must have been called first.
452    */
453   public AggregateAttr getStatsForType(ProfilerTask type, Collection<Task> tasks) {
454     long totalTime = 0;
455     int count = 0;
456     for (Task task : tasks) {
457       if (task.parentId > 0) {
458         throw new IllegalArgumentException("task " + task.id + " is not a root task");
459       }
460       AggregateAttr attr = task.aggregatedStats.getAttr(type);
461       count += attr.count;
462       totalTime += attr.totalTime;
463       if (task.type == type) {
464         count++;
465         totalTime += (task.duration - task.getInheritedDuration());
466       }
467     }
468     return new AggregateAttr(count, totalTime);
469   }

471   /**
472    * Returns list of all root tasks related to (in other words, started during)
473    * the specified phase task.
474    */
475   public List<Task> getTasksForPhase(Task phaseTask) {
476     Preconditions.checkArgument(phaseTask.type == ProfilerTask.PHASE,
477       "Unsupported task type %s", phaseTask.type);
478 
479     // Algorithm below takes into account fact that rootTasksById list is sorted
480     // by the task id and task id values are monotonically increasing with time
481     // (this property is guaranteed by the profiler). Thus list is effectively
482     // sorted by the startTime. We are trying to select a sublist that includes
483     // all tasks that were started later than the given task but earlier than
484     // its completion time.
485     int startIndex = Collections.binarySearch(rootTasksById, phaseTask);
486     Preconditions.checkState(startIndex >= 0,
487         "Phase task %s is not a root task", phaseTask.id);
488     int endIndex = (phaseTask.relatedTask != null)
489         ? Collections.binarySearch(rootTasksById, phaseTask.relatedTask)
490         : rootTasksById.size();
491     Preconditions.checkState(endIndex >= startIndex,
492         "Failed to find end of the phase marked by the task %s", phaseTask.id);
493     return rootTasksById.subList(startIndex, endIndex);
494   }

509   /**
510    * Returns duration of the given phase in ns.
511    */
512   public long getPhaseDuration(Task phaseTask) {
513     Preconditions.checkArgument(phaseTask.type == ProfilerTask.PHASE,
514         "Unsupported task type %s", phaseTask.type);
515 
516     long duration;
517     if (phaseTask.relatedTask != null) {
518       duration = phaseTask.relatedTask.startTime - phaseTask.startTime;
519     } else {
520       Task lastTask = rootTasksById.get(rootTasksById.size() - 1);
521       duration = lastTask.startTime + lastTask.duration - phaseTask.startTime;
522     }
523     Preconditions.checkState(duration >= 0);
524     return duration;
525   }

528   /**
529    * Builds map of dependencies between ACTION tasks based on dependencies
530    * between ACTION_GRAPH tasks
531    */
532   private Task buildActionTaskTree(Task actionGraphTask, List<Task> actionTasksByDescription) {
533     Task actionTask = actionGraphTask.relatedTask;
534     if (actionTask == null) {
535       actionTask = actionTasksByDescription.get(actionGraphTask.descIndex);
536       if (actionTask == null) {
537         // If we cannot find ACTION task that corresponds to the ACTION_GRAPH task,
538         // most likely scenario is that we dealing with either aborted or failed
539         // build. In this case we will find or create fake zero-duration action
540         // task and still reconstruct dependency graph.
541         actionTask = new Task(-1, --fakeActionId, 0, 0, 0,
542             ProfilerTask.ACTION, actionGraphTask.descIndex, new CompactStatistics((byte[]) null));
543         actionTask.calculateRootStats();
544         actionTasksByDescription.set(actionGraphTask.descIndex, actionTask);
545       }
546       actionGraphTask.relatedTask = actionTask;
547     }
548     if (actionGraphTask.subtasks.length != 0) {
549       List<Task> list = Lists.newArrayListWithCapacity(actionGraphTask.subtasks.length);
550       for (Task task : actionGraphTask.subtasks) {
551         if (task.type == ProfilerTask.ACTION_GRAPH) {
552           list.add(buildActionTaskTree(task, actionTasksByDescription));
553         }
554       }
555       if (!list.isEmpty()) {
556         Task[] actionPrerequisites = list.toArray(new Task[list.size()]);
557         Arrays.sort(actionPrerequisites);
558         actionDependencyMap.put(actionTask, actionPrerequisites);
559       }
560     }
561     return actionTask;
562   }

564   /**
565    * Builds map of dependencies between ACTION tasks based on dependencies
566    * between ACTION_GRAPH tasks. Root of that dependency tree would be
567    * getBuildPhaseTask().
568    *
569    * <p> Also marks related ACTION and ACTION_SUBMIT tasks.
570    */
571   private void buildDependencyMap() {
572     Task analysisPhaseTask = getPhaseTask(ProfilePhase.ANALYZE);
573     Task executionPhaseTask = getPhaseTask(ProfilePhase.EXECUTE);
574     if ((executionPhaseTask == null) || (analysisPhaseTask == null)) {
575       return;
576     }
577     // Association between ACTION_GRAPH tasks and ACTION tasks can be established through
578     // description id. So we create appropriate xref list.
579     List<Task> actionTasksByDescription = Lists.newArrayList(new Task[descriptionList.size()]);
580     for (Task task : getTasksForPhase(executionPhaseTask)) {
581       if (task.type == ProfilerTask.ACTION) {
582         actionTasksByDescription.set(task.descIndex, task);
583       }
584     }
585     List<Task> list = new ArrayList<>();
586     for (Task task : getTasksForPhase(analysisPhaseTask)) {
587       if (task.type == ProfilerTask.ACTION_GRAPH) {
588         list.add(buildActionTaskTree(task, actionTasksByDescription));
589       }
590     }
591     Task[] actionPrerequisites = list.toArray(new Task[list.size()]);
592     Arrays.sort(actionPrerequisites);
593     actionDependencyMap.put(executionPhaseTask, actionPrerequisites);
594 
595     // Scan through all execution phase tasks to identify ACTION_SUBMIT tasks and associate
596     // them with ACTION task counterparts. ACTION_SUBMIT tasks are not necessarily root
597     // tasks so we need to scan ALL tasks.
598     for (Task task : allTasksById.subList(executionPhaseTask.id, allTasksById.size())) {
599       if (task.type == ProfilerTask.ACTION_SUBMIT) {
600         Task actionTask = actionTasksByDescription.get(task.descIndex);
601         if (actionTask != null) {
602           task.relatedTask = actionTask;
603           actionTask.relatedTask = task;
604         }
605       } else if (task.type == ProfilerTask.ACTION_BUILDER) {
606         Task actionTask = actionTasksByDescription.get(task.descIndex);
607         if (actionTask != null) {
608           parallelBuilderCompletionQueueTasks.put(actionTask, task);
609         }
610       }
611     }
612   }

614   /**
615    * Calculates critical path for the specific action
616    * excluding specified nested task types (e.g. VFS-related time) and not
617    * accounting for overhead related to the Blaze scheduler.
618    */
619   private CriticalPathEntry computeCriticalPathForAction(
620       Set<ProfilerTask> ignoredTypes, Set<Task> ignoredTasks,
621       Task actionTask, Map<Task, CriticalPathEntry> cache, Deque<Task> stack) {
622 
623     // Loop check is expensive for the Deque (and we don't want to use hash sets because adding
624     // and removing elements was shown to be very expensive). To avoid quadratic costs we're
625     // checking for infinite loop only when deque's size equal to the power of 2 and >= 32.
626     if ((stack.size() & 0x1F) == 0 && Integer.bitCount(stack.size()) == 1) {
627       if (stack.contains(actionTask)) {
628         // This situation will appear if build has ended with the
629         // IllegalStateException thrown by the
630         // ParallelBuilder.getNextCompletedAction(), warning user about
631         // possible cycle in the dependency graph. But the exception text
632         // is more friendly and will actually identify the loop.
633         // Do not use Preconditions class below due to the very expensive
634         // toString() calls used in the message.
635         throw new IllegalStateException ("Dependency graph contains loop:\n"
636             + actionTask + " in the\n" + Joiner.on('\n').join(stack));
637       }
638     }
639     stack.addLast(actionTask);
640     CriticalPathEntry entry;
641     try {
642       entry = cache.get(actionTask);
643       long entryDuration = 0;
644       if (entry == null) {
645         Task[] actionPrerequisites = actionDependencyMap.get(actionTask);
646         if (actionPrerequisites != null) {
647           for (Task task : actionPrerequisites) {
648             CriticalPathEntry candidate =
649               computeCriticalPathForAction(ignoredTypes, ignoredTasks, task, cache, stack);
650             if (entry == null || entryDuration < candidate.cumulativeDuration) {
651               entry = candidate;
652               entryDuration = candidate.cumulativeDuration;
653             }
654           }
655         }
656         if (actionTask.type == ProfilerTask.ACTION) {
657           long duration = actionTask.duration;
658           if (ignoredTasks.contains(actionTask)) {
659             duration = 0L;
660           } else {
661             for (ProfilerTask type : ignoredTypes) {
662               duration -= actionTask.aggregatedStats.getAttr(type).totalTime;
663             }
664           }
665 
666           entry = new CriticalPathEntry(actionTask, duration, entry);
667           cache.put(actionTask, entry);
668         }
669       }
670     } finally {
671       stack.removeLast();
672     }
673     return entry;
674   }

676   /**
677    * Returns the critical path information from the {@code CriticalPathComputer} recorded stats.
678    * This code does not have the "Critical" column (Time difference if we removed this node from
679    * the critical path).
680    */
681   public CriticalPathEntry getCriticalPathNewVersion() {
682     for (Task task : rootTasksById) {
683       if (task.type == CRITICAL_PATH) {
684         CriticalPathEntry entry = null;
685         for (Task shared : task.subtasks) {
686           entry = new CriticalPathEntry(shared, shared.duration, entry);
687         }
688         return entry;
689       }
690     }
691     return null;
692   }

694   /**
695    * Calculates critical path for the given action graph excluding
696    * specified tasks (usually ones that belong to the "real" critical path).
697    */
698   public CriticalPathEntry getCriticalPath(Set<ProfilerTask> ignoredTypes) {
699     Task actionTask = getPhaseTask(ProfilePhase.EXECUTE);
700     if (actionTask == null) {
701       return null;
702     }
703     Map <Task, CriticalPathEntry> cache = Maps.newHashMapWithExpectedSize(1000);
704     CriticalPathEntry result = computeCriticalPathForAction(ignoredTypes,
705         new HashSet<Task>(), actionTask, cache,
706         new ArrayDeque<Task>());
707     if (result != null) {
708       return result;
709     }
710     return getCriticalPathNewVersion();
711   }

713   /**
714    * Calculates critical path time that will be saved by eliminating specific
715    * entry from the critical path
716    */
717   public void analyzeCriticalPath(Set<ProfilerTask> ignoredTypes, CriticalPathEntry path) {
718     // With light critical path we do not need to analyze since it is already preprocessed
719     // by blaze build.
720     if (path != null && path.task.type == CRITICAL_PATH_COMPONENT) {
721       return;
722     }
723     for (CriticalPathEntry entry = path; entry != null; entry = entry.next) {
724       Map <Task, CriticalPathEntry> cache = Maps.newHashMapWithExpectedSize(1000);
725       entry.setCriticalTime(path.cumulativeDuration -
726           computeCriticalPathForAction(ignoredTypes, Sets.newHashSet(entry.task),
727           getPhaseTask(ProfilePhase.EXECUTE), cache,  new ArrayDeque<Task>())
728           .cumulativeDuration);
729     }
730   }

732   /**
733    * Return the next critical path entry for the task or null if there is none.
734    */
735   public CriticalPathEntry getNextCriticalPathEntryForTask(CriticalPathEntry path, Task task) {
736     for (CriticalPathEntry entry = path; entry != null; entry = entry.next) {
737       if (entry.task.id == task.id) {
738         return entry;
739       }
740     }
741     return null;

744   /**
745    * Returns time action waited in the execution queue (difference between
746    * ACTION task start time and ACTION_SUBMIT task start time).
747    */
748   public long getActionWaitTime(Task actionTask) {
749     // Light critical path does not record wait time.
750     if (actionTask.type == ProfilerTask.CRITICAL_PATH_COMPONENT) {
751       return 0;
752     }
753     Preconditions.checkArgument(actionTask.type == ProfilerTask.ACTION);
754     if (actionTask.relatedTask != null) {
755       Preconditions.checkState(actionTask.relatedTask.type == ProfilerTask.ACTION_SUBMIT);
756       long time = actionTask.startTime - actionTask.relatedTask.startTime;
757       Preconditions.checkState(time >= 0);
758       return time;
759     } else {
760       return 0L; // submission time is not available.
761     }
762   }
763 
764   /**
765    * Returns time action waited in the parallel builder completion queue
766    * (difference between ACTION task end time and ACTION_BUILDER start time).
767    */
768   public long getActionQueueTime(Task actionTask) {
769     // Light critical path does not record queue time.
770     if (actionTask.type == ProfilerTask.CRITICAL_PATH_COMPONENT) {
771       return 0;
772     }
773     Preconditions.checkArgument(actionTask.type == ProfilerTask.ACTION);
774     Task related = parallelBuilderCompletionQueueTasks.get(actionTask);
775     if (related != null) {
776       Preconditions.checkState(related.type == ProfilerTask.ACTION_BUILDER);
777       long time = related.startTime - (actionTask.startTime + actionTask.duration);
778       Preconditions.checkState(time >= 0);
779       return time;
780     } else {
781       return 0L; // queue task is not available.
782     }
783   }

796   /**
797    * Loads and parses Blaze profile file.
798    *
799    * @param profileFile profile file path
800    *
801    * @return ProfileInfo object with some fields populated (call calculateStats()
802    *         and analyzeRelationships() to populate the remaining fields)
803    * @throws UnsupportedEncodingException if the file format is invalid
804    * @throws IOException if the file can't be read
805    */
806   public static ProfileInfo loadProfile(Path profileFile)
807       throws IOException {
808     // It is extremely important to wrap InflaterInputStream using
809     // BufferedInputStream because majority of reads would be done using
810     // readInt()/readLong() methods and InflaterInputStream is very inefficient
811     // in handling small read requests (performance difference with 1MB buffer
812     // used below is almost 10x).
813     DataInputStream in = new DataInputStream(
814         new BufferedInputStream(new InflaterInputStream(
815         profileFile.getInputStream(), new Inflater(false), 65536), 1024 * 1024));
816 
817     if (in.readInt() != Profiler.MAGIC) {
818       in.close();
819       throw new UnsupportedEncodingException("Invalid profile datafile format");
820     }
821     if (in.readInt() != Profiler.VERSION) {
822       in.close();
823       throw new UnsupportedEncodingException("Incompatible profile datafile version");
824     }
825     String fileComment = in.readUTF();
826 
827     // Read list of used record types
828     int typeCount = in.readInt();
829     boolean hasUnknownTypes = false;
830     Set<String> supportedTasks = new HashSet<>();
831     for (ProfilerTask task : ProfilerTask.values()) {
832       supportedTasks.add(task.toString());
833     }
834     List<ProfilerTask> typeList = new ArrayList<>();
835     for (int i = 0; i < typeCount; i++) {
836       String name = in.readUTF();
837       if (supportedTasks.contains(name)) {
838         typeList.add(ProfilerTask.valueOf(name));
839       } else {
840         hasUnknownTypes = true;
841         typeList.add(ProfilerTask.UNKNOWN);
842       }
843     }
844 
845     ProfileInfo info = new ProfileInfo(fileComment);
846 
847     // Read record until we encounter end marker (-1).
848     // TODO(bazel-team): Maybe this still should handle corrupted(truncated) files.
849     try {
850       int size;
851       while ((size = in.readInt()) != Profiler.EOF_MARKER) {
852         byte[] backingArray = new byte[size];
853         in.readFully(backingArray);
854         ByteBuffer buffer = ByteBuffer.wrap(backingArray);
855         long threadId = VarInt.getVarLong(buffer);
856         int id = VarInt.getVarInt(buffer);
857         int parentId = VarInt.getVarInt(buffer);
858         long startTime = VarInt.getVarLong(buffer);
859         long duration = VarInt.getVarLong(buffer);
860         int descIndex = VarInt.getVarInt(buffer) - 1;
861         if (descIndex == -1) {
862           String desc = in.readUTF();
863           descIndex = info.descriptionList.size();
864           info.descriptionList.add(desc);
865         }
866         ProfilerTask type = typeList.get(buffer.get());
867         byte[] stats = null;
868         if (buffer.hasRemaining()) {
869           // Copy aggregated stats.
870           int offset = buffer.position();
871           stats = Arrays.copyOfRange(backingArray, offset, size);
872           if (hasUnknownTypes) {
873             while (buffer.hasRemaining()) {
874               byte attrType = buffer.get();
875               if (typeList.get(attrType) == ProfilerTask.UNKNOWN) {
876                 // We're dealing with unknown aggregated type - update stats array to
877                 // use ProfilerTask.UNKNOWN.ordinal() value.
878                 stats[buffer.position() - 1 - offset] = (byte) ProfilerTask.UNKNOWN.ordinal();
879               }
880               VarInt.getVarInt(buffer);
881               VarInt.getVarLong(buffer);
882             }
883           }
884         }
885         ProfileInfo.Task task =  info.new Task(threadId, id, parentId, startTime, duration, type,
886             descIndex, new CompactStatistics(stats));
887         info.addTask(task);
888       }
889     } catch (IOException e) {
890       info.corruptedOrIncomplete = true;
891     } finally {
892       in.close();
893     }
894 
895     return info;
896   }

898   /**
899    * Loads and parses Blaze profile file, and reports what it is doing.
900    *
901    * @param profileFile profile file path
902    * @param reporter for progress messages and warnings
903    *
904    * @return ProfileInfo object with most fields populated
905    *         (call analyzeRelationships() to populate the remaining fields)
906    * @throws UnsupportedEncodingException if the file format is invalid
907    * @throws IOException if the file can't be read
908    */
909   public static ProfileInfo loadProfileVerbosely(Path profileFile, InfoListener reporter)
910       throws IOException {
911     reporter.info("Loading " + profileFile.getPathString());
912     ProfileInfo profileInfo = ProfileInfo.loadProfile(profileFile);
913     if (profileInfo.isCorruptedOrIncomplete()) {
914       reporter.warn("Profile file is incomplete or corrupted - not all records were parsed");
915     }
916     reporter.info(profileInfo.comment + ", " + profileInfo.allTasksById.size() + " record(s)");
917     return profileInfo;
918   }
