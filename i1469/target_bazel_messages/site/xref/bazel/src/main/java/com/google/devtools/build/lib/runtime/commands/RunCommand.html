71   * Builds and run a target with the given command line arguments.
72   */
73  @Command(name = "run",
74           builds = true,
75           options = { RunCommand.RunOptions.class },
76           inherits = { BuildCommand.class },
77           shortDescription = "Runs the specified target.",
78           help = "resource:run.txt",

87          category = "run",

91              + "target. If this option is set, the target is not run from %{product}. "
92              + "Use '%{product} run --script_path=foo //foo && foo' to invoke target '//foo' "
93              + "This differs from '%{product} run //foo' in that the %{product} lock is released "
94              + "and the executable is connected to the terminal's stdin.")

111   @VisibleForTesting  // productionVisibility = Visibility.PRIVATE
112   protected BuildResult processRequest(final BlazeRuntime runtime, BuildRequest request) {
113     return runtime.getBuildTool().processRequest(request, new TargetValidator() {
114       @Override
115       public void validateTargets(Collection<Target> targets, boolean keepGoing)
116           throws LoadingFailedException {
117         RunCommand.this.validateTargets(runtime.getReporter(), targets, keepGoing);
118       }
119     });
120   }

125   @Override
126   public ExitCode exec(BlazeRuntime runtime, OptionsProvider options) {
127     RunOptions runOptions = options.getOptions(RunOptions.class);
128     // This list should look like: ["//executable:target", "arg1", "arg2"]
129     List<String> targetAndArgs = options.getResidue();
130 
131     // The user must at the least specify an executable target.
132     if (targetAndArgs.isEmpty()) {
133       runtime.getReporter().handle(Event.error("Must specify a target to run"));
134       return ExitCode.COMMAND_LINE_ERROR;
135     }
136     String targetString = targetAndArgs.get(0);
137     List<String> runTargetArgs = targetAndArgs.subList(1, targetAndArgs.size());
138     RunUnder runUnder = options.getOptions(BuildConfiguration.Options.class).runUnder;
139 
140     OutErr outErr = runtime.getReporter().getOutErr();
141     List<String> targets = (runUnder != null) && (runUnder.getLabel() != null)
142         ? ImmutableList.of(targetString, runUnder.getLabel().toString())
143         : ImmutableList.of(targetString);
144     BuildRequest request = BuildRequest.create(
145         this.getClass().getAnnotation(Command.class).name(), options,
146         runtime.getStartupOptionsProvider(), targets, outErr,
147         runtime.getCommandId(), runtime.getCommandStartTime());
148 
149     currentRunUnder = runUnder;
150     BuildResult result;
151     try {
152       result = processRequest(runtime, request);
153     } finally {
154       currentRunUnder = null;
155     }
156 
157     if (!result.getSuccess()) {
158       runtime.getReporter().handle(Event.error("Build failed. Not running target"));
159       return result.getExitCondition();
160     }
161 
162     // Make sure that we have exactly 1 built target (excluding --run_under),
163     // and that it is executable.
164     // These checks should only fail if keepGoing is true, because we already did
165     // validation before the build began.  See {@link #validateTargets()}.
166     Collection<ConfiguredTarget> targetsBuilt = result.getSuccessfulTargets();
167     ConfiguredTarget targetToRun = null;
168     ConfiguredTarget runUnderTarget = null;
169 
170     if (targetsBuilt != null) {
171       int maxTargets = runUnder != null && runUnder.getLabel() != null ? 2 : 1;
172       if (targetsBuilt.size() > maxTargets) {
173         runtime.getReporter().handle(Event.error(SINGLE_TARGET_MESSAGE));
174         return ExitCode.COMMAND_LINE_ERROR;
175       }
176       for (ConfiguredTarget target : targetsBuilt) {
177         ExitCode targetValidation = fullyValidateTarget(runtime, target);
178         if (targetValidation != ExitCode.SUCCESS) {
179           return targetValidation;
180         }
181         if (runUnder != null && target.getLabel().equals(runUnder.getLabel())) {
182           if (runUnderTarget != null) {
183             runtime.getReporter().handle(Event.error(
184                 null, "Can't identify the run_under target from multiple options?"));
185             return ExitCode.COMMAND_LINE_ERROR;
186           }
187           runUnderTarget = target;
188         } else if (targetToRun == null) {
189           targetToRun = target;
190         } else {
191           runtime.getReporter().handle(Event.error(SINGLE_TARGET_MESSAGE));
192           return ExitCode.COMMAND_LINE_ERROR;
193         }
194       }
195     }
196     // Handle target & run_under referring to the same target.
197     if ((targetToRun == null) && (runUnderTarget != null)) {
198       targetToRun = runUnderTarget;
199     }
200     if (targetToRun == null) {
201       runtime.getReporter().handle(Event.error(NO_TARGET_MESSAGE));
202       return ExitCode.COMMAND_LINE_ERROR;
203     }
204 
205     Path executablePath = Preconditions.checkNotNull(
206         targetToRun.getProvider(FilesToRunProvider.class).getExecutable().getPath());
207     BuildConfiguration configuration = targetToRun.getConfiguration();
208     if (configuration == null) {
209       // The target may be an input file, which doesn't have a configuration. In that case, we
210       // choose any target configuration.
211       configuration = runtime.getBuildTool().getView().getConfigurationCollection()
212           .getTargetConfigurations().get(0);
213     }
214     Path workingDir;
215     try {
216       workingDir = ensureRunfilesBuilt(runtime, targetToRun);
217     } catch (CommandException e) {
218       runtime.getReporter().handle(Event.error("Error creating runfiles: " + e.getMessage()));
219       return ExitCode.LOCAL_ENVIRONMENTAL_ERROR;
220     }
221 
222     List<String> args = runTargetArgs;
223 
224     FilesToRunProvider provider = targetToRun.getProvider(FilesToRunProvider.class);
225     RunfilesSupport runfilesSupport = provider == null ? null : provider.getRunfilesSupport();
226     if (runfilesSupport != null && runfilesSupport.getArgs() != null) {
227       List<String> targetArgs = runfilesSupport.getArgs();
228       if (!targetArgs.isEmpty()) {
229         args = Lists.newArrayListWithCapacity(targetArgs.size() + runTargetArgs.size());
230         args.addAll(targetArgs);
231         args.addAll(runTargetArgs);
232       }
233     }
234 
235     //
236     // We now have a unique executable ready to be run.
237     //
238     // We build up two different versions of the command to run: one with an absolute path, which
239     // we'll actually run, and a prettier one with the long absolute path to the executable
240     // replaced with a shorter relative path that uses the symlinks in the workspace.
241     PathFragment prettyExecutablePath =
242         OutputDirectoryLinksUtils.getPrettyPath(executablePath,
243             runtime.getWorkspaceName(), runtime.getWorkspace(),
244             options.getOptions(BuildRequestOptions.class).symlinkPrefix);
245     List<String> cmdLine = new ArrayList<>();
246     if (runOptions.scriptPath == null) {
247       cmdLine.add(runtime.getDirectories().getExecRoot()
248           .getRelative(runtime.getBinTools().getExecPath(PROCESS_WRAPPER)).getPathString());
249       cmdLine.add("-1");
250       cmdLine.add("15");
251       cmdLine.add("-");
252       cmdLine.add("-");
253     }
254     List<String> prettyCmdLine = new ArrayList<>();
255     // Insert the command prefix specified by the "--run_under=<command-prefix>" option
256     // at the start of the command line.
257     if (runUnder != null) {
258       String runUnderValue = runUnder.getValue();
259       if (runUnderTarget != null) {
260         // --run_under specifies a target. Get the corresponding executable.
261         // This must be an absolute path, because the run_under target is only
262         // in the runfiles of test targets.
263         runUnderValue = runUnderTarget
264             .getProvider(FilesToRunProvider.class).getExecutable().getPath().getPathString();
265         // If the run_under command contains any options, make sure to add them
266         // to the command line as well.
267         List<String> opts = runUnder.getOptions();
268         if (!opts.isEmpty()) {
269           runUnderValue += " " + ShellEscaper.escapeJoinAll(opts);
270         }
271       }
272       cmdLine.add(configuration.getShExecutable().getPathString());
273       cmdLine.add("-c");
274       cmdLine.add(runUnderValue + " " + executablePath.getPathString() + " " +
275           ShellEscaper.escapeJoinAll(args));
276       prettyCmdLine.add(configuration.getShExecutable().getPathString());
277       prettyCmdLine.add("-c");
278       prettyCmdLine.add(runUnderValue + " " + prettyExecutablePath.getPathString() + " " +
279           ShellEscaper.escapeJoinAll(args));
280     } else {
281       cmdLine.add(executablePath.getPathString());
282       cmdLine.addAll(args);
283       prettyCmdLine.add(prettyExecutablePath.getPathString());
284       prettyCmdLine.addAll(args);
285     }
286 
287     // Add a newline between the blaze output and the binary's output.
288     outErr.printErrLn("");
289 
290     if (runOptions.scriptPath != null) {
291       String unisolatedCommand = CommandFailureUtils.describeCommand(
292           CommandDescriptionForm.COMPLETE_UNISOLATED,
293           cmdLine, null, workingDir.getPathString());
294       if (writeScript(runtime, runOptions.scriptPath, unisolatedCommand)) {
295         return ExitCode.SUCCESS;
296       } else {
297         return ExitCode.RUN_FAILURE;
298       }
299     }
300 
301     runtime.getReporter().handle(Event.info(
302         null, "Running command line: " + ShellEscaper.escapeJoinAll(prettyCmdLine)));
303 
304     com.google.devtools.build.lib.shell.Command command = new CommandBuilder()
305         .addArgs(cmdLine).setEnv(runtime.getClientEnv()).setWorkingDir(workingDir).build();
306 
307     try {
308       // Restore a raw EventHandler if it is registered. This allows for blaze run to produce the
309       // actual output of the command being run even if --color=no is specified.
310       runtime.getReporter().switchToAnsiAllowingHandler();
311 
312       // The command API is a little strange in that the following statement
313       // will return normally only if the program exits with exit code 0.
314       // If it ends with any other code, we have to catch BadExitStatusException.
315       command.execute(com.google.devtools.build.lib.shell.Command.NO_INPUT,
316           com.google.devtools.build.lib.shell.Command.NO_OBSERVER,
317           outErr.getOutputStream(),
318           outErr.getErrorStream(),
319           true /* interruptible */).getTerminationStatus().getExitCode();
320       return ExitCode.SUCCESS;
321     } catch (BadExitStatusException e) {
322       String message = "Non-zero return code '"
323                        + e.getResult().getTerminationStatus().getExitCode()
324                        + "' from command: " + e.getMessage();
325       runtime.getReporter().handle(Event.error(message));
326       return ExitCode.RUN_FAILURE;
327     } catch (AbnormalTerminationException e) {
328       // The process was likely terminated by a signal in this case.
329       return ExitCode.INTERRUPTED;
330     } catch (CommandException e) {
331       runtime.getReporter().handle(Event.error("Error running program: " + e.getMessage()));
332       return ExitCode.RUN_FAILURE;
333     }
334   }

336   /**
337    * Ensures that runfiles are built for the specified target. If they already
338    * are, does nothing, otherwise builds them.
339    *
340    * @param target the target to build runfiles for.
341    * @return the path of the runfiles directory.
342    * @throws CommandException
343    */
344   private Path ensureRunfilesBuilt(BlazeRuntime runtime, ConfiguredTarget target)
345       throws CommandException {
346     FilesToRunProvider provider = target.getProvider(FilesToRunProvider.class);
347     RunfilesSupport runfilesSupport = provider == null ? null : provider.getRunfilesSupport();
348     if (runfilesSupport == null) {
349       return runtime.getWorkingDirectory();
350     }
351 
352     Artifact manifest = runfilesSupport.getRunfilesManifest();
353     PathFragment runfilesDir = runfilesSupport.getRunfilesDirectoryExecPath();
354     Path workingDir = runtime.getExecRoot()
355         .getRelative(runfilesDir)
356         .getRelative(runfilesSupport.getRunfiles().getSuffix());
357 
358     // When runfiles are not generated, getManifest() returns the
359     // .runfiles_manifest file, otherwise it returns the MANIFEST file. This is
360     // a handy way to check whether runfiles were built or not.
361     if (!RUNFILES_MANIFEST.matches(manifest.getFilename())) {
362       // Runfiles already built, nothing to do.
363       return workingDir;
364     }
365 
366     SymlinkTreeHelper helper = new SymlinkTreeHelper(
367         manifest.getExecPath(),
368         runfilesDir,
369         false);
370     helper.createSymlinksUsingCommand(runtime.getExecRoot(), target.getConfiguration(),
371         runtime.getBinTools());
372     return workingDir;
373   }

375   private boolean writeScript(BlazeRuntime runtime, PathFragment scriptPathFrag, String cmd) {
376     final String SH_SHEBANG = "#!/bin/sh";
377     Path scriptPath = runtime.getWorkingDirectory().getRelative(scriptPathFrag);
378     try {
379       FileSystemUtils.writeContent(scriptPath, StandardCharsets.ISO_8859_1,
380           SH_SHEBANG + "\n" + cmd + " \"$@\"");
381       scriptPath.setExecutable(true);
382     } catch (IOException e) {
383       runtime.getReporter().handle(Event.error("Error writing run script:" + e.getMessage()));
384       return false;
385     }
386     return true;
387   }

389   // Make sure we are building exactly 1 binary target.
390   // If keepGoing, we'll build all the targets even if they are non-binary.
391   private void validateTargets(Reporter reporter, Collection<Target> targets, boolean keepGoing)
392       throws LoadingFailedException {
393     Target targetToRun = null;
394     Target runUnderTarget = null;
395 
396     boolean singleTargetWarningWasOutput = false;
397     int maxTargets = currentRunUnder != null && currentRunUnder.getLabel() != null ? 2 : 1;
398     if (targets.size() > maxTargets) {
399       warningOrException(reporter, SINGLE_TARGET_MESSAGE, keepGoing);
400       singleTargetWarningWasOutput = true;
401     }
402     for (Target target : targets) {
403       String targetError = validateTarget(target);
404       if (targetError != null) {
405         warningOrException(reporter, targetError, keepGoing);
406       }
407 
408       if (currentRunUnder != null && target.getLabel().equals(currentRunUnder.getLabel())) {
409         // It's impossible to have two targets with the same label.
410         Preconditions.checkState(runUnderTarget == null);
411         runUnderTarget = target;
412       } else if (targetToRun == null) {
413         targetToRun = target;
414       } else {
415         if (!singleTargetWarningWasOutput) {
416           warningOrException(reporter, SINGLE_TARGET_MESSAGE, keepGoing);
417         }
418         return;
419       }
420     }
421     // Handle target & run_under referring to the same target.
422     if ((targetToRun == null) && (runUnderTarget != null)) {
423       targetToRun = runUnderTarget;
424     }
425     if (targetToRun == null) {
426       warningOrException(reporter, NO_TARGET_MESSAGE, keepGoing);
427     }
428   }

430   // If keepGoing, print a warning and return the given collection.
431   // Otherwise, throw InvalidTargetException.
432   private void warningOrException(Reporter reporter, String message,
433       boolean keepGoing) throws LoadingFailedException {
434     if (keepGoing) {
435       reporter.handle(Event.warn(message + ". Will continue anyway"));
436     } else {
437       throw new LoadingFailedException(message);
438     }
439   }

441   private static String notExecutableError(Target target) {
442     return "Cannot run target " + target.getLabel() + ": Not executable";
443   }

445   /** Returns null if the target is a runnable rule, or an appropriate error message otherwise. */
446   private static String validateTarget(Target target) {
447     return isExecutable(target)
448         ? null
449         : notExecutableError(target);
450   }

452   /**
453    * Performs all available validation checks on an individual target.
454    *
455    * @param target ConfiguredTarget to validate
456    * @return ExitCode.SUCCESS if all checks succeeded, otherwise a different error code.
457    */
458   private ExitCode fullyValidateTarget(BlazeRuntime runtime, ConfiguredTarget target) {
459     String targetError = validateTarget(target.getTarget());
460 
461     if (targetError != null) {
462       runtime.getReporter().handle(Event.error(targetError));
463       return ExitCode.COMMAND_LINE_ERROR;
464     }
465 
466     Artifact executable = target.getProvider(FilesToRunProvider.class).getExecutable();
467     if (executable == null) {
468       runtime.getReporter().handle(Event.error(notExecutableError(target.getTarget())));
469       return ExitCode.COMMAND_LINE_ERROR;
470     }
471 
472     // Shouldn't happen: We just validated the target.
473     Preconditions.checkState(executable != null,
474         "Could not find executable for target %s", target);
475     Path executablePath = executable.getPath();
476     try {
477       if (!executablePath.exists() || !executablePath.isExecutable()) {
478         runtime.getReporter().handle(Event.error(
479             null, "Non-existent or non-executable " + executablePath));
480         return ExitCode.BLAZE_INTERNAL_ERROR;
481       }
482     } catch (IOException e) {
483       runtime.getReporter().handle(Event.error(
484           "Error checking " + executablePath.getPathString() + ": " + e.getMessage()));
485       return ExitCode.LOCAL_ENVIRONMENTAL_ERROR;
486     }
487 
488     return ExitCode.SUCCESS;
489   }

491   /**
492    * Return true iff {@code target} is a rule that has an executable file. This includes
493    * *_test rules, *_binary rules, and generated outputs.
494    */
495   private static boolean isExecutable(Target target) {
496     return isOutputFile(target) || isExecutableNonTestRule(target)
497         || TargetUtils.isTestRule(target);
498   }
499 
500   /**
501    * Return true iff {@code target} is a rule that generates an executable file and is user-executed
502    * code.
503    */
504   private static boolean isExecutableNonTestRule(Target target) {
505     if (!(target instanceof Rule)) {
506       return false;
507     }
508     Rule rule = ((Rule) target);
509     if (rule.getRuleClassObject().hasAttr("$is_executable", Type.BOOLEAN)) {
510       return NonconfigurableAttributeMapper.of(rule).get("$is_executable", Type.BOOLEAN);
511     }
512     return false;
513   }
514 
515   private static boolean isOutputFile(Target target) {
516     return (target instanceof OutputFile);
517   }
