48   * An implementation of java_binary.

50  public class JavaBinary implements RuleConfiguredTargetFactory {
51    private static final PathFragment CPP_RUNTIMES = new PathFragment("_cpp_runtimes");
52  
53    private final JavaSemantics semantics;
54  
55    protected JavaBinary(JavaSemantics semantics) {
56      this.semantics = semantics;
57    }
58  
59    @Override
60    public ConfiguredTarget create(RuleContext ruleContext) {
61      final JavaCommon common = new JavaCommon(ruleContext, semantics);
62      DeployArchiveBuilder deployArchiveBuilder =  new DeployArchiveBuilder(semantics, ruleContext);
63      Runfiles.Builder runfilesBuilder = new Runfiles.Builder(ruleContext.getWorkspaceName());
64      List<String> jvmFlags = new ArrayList<>();
65  
66      common.initializeJavacOpts();
67      JavaTargetAttributes.Builder attributesBuilder = common.initCommon();
68      attributesBuilder.addClassPathResources(
69          ruleContext.getPrerequisiteArtifacts("classpath_resources", Mode.TARGET).list());
70  
71      List<String> userJvmFlags = common.getJvmFlags();
72  
73      ruleContext.checkSrcsSamePackage(true);
74      boolean createExecutable = ruleContext.attributes().get("create_executable", Type.BOOLEAN);
75      List<TransitiveInfoCollection> deps =
76          Lists.newArrayList(common.targetsTreatedAsDeps(ClasspathType.COMPILE_ONLY));
77      semantics.checkRule(ruleContext, common);
78      String mainClass = semantics.getMainClass(ruleContext, common);
79      String originalMainClass = mainClass;
80      if (ruleContext.hasErrors()) {
81        return null;
82      }
83  
84      // Collect the transitive dependencies.
85      JavaCompilationHelper helper = new JavaCompilationHelper(
86          ruleContext, semantics, common.getJavacOpts(), attributesBuilder);
87      helper.addLibrariesToAttributes(deps);
88      helper.addProvidersToAttributes(common.compilationArgsFromSources(), /* isNeverLink */ false);
89      attributesBuilder.addNativeLibraries(
90          collectNativeLibraries(common.targetsTreatedAsDeps(ClasspathType.BOTH)));
91  
92      // deploy_env is valid for java_binary, but not for java_test.
93      if (ruleContext.getRule().isAttrDefined("deploy_env", Type.LABEL_LIST)) {
94        for (JavaRuntimeClasspathProvider envTarget : ruleContext.getPrerequisites(
95                 "deploy_env", Mode.TARGET, JavaRuntimeClasspathProvider.class)) {
96          attributesBuilder.addExcludedArtifacts(envTarget.getRuntimeClasspath());
97        }
98      }
99  
100     Artifact srcJar =
101         ruleContext.getImplicitOutputArtifact(JavaSemantics.JAVA_BINARY_SOURCE_JAR);
102 
103     Artifact classJar =
104         ruleContext.getImplicitOutputArtifact(JavaSemantics.JAVA_BINARY_CLASS_JAR);
105 
106     ImmutableList<Artifact> srcJars = ImmutableList.of(srcJar);
107 
108     CppConfiguration cppConfiguration = ruleContext.getConfiguration().getFragment(
109         CppConfiguration.class);
110     boolean stripAsDefault = cppConfiguration.useFission();
111     Artifact launcher = semantics.getLauncher(ruleContext, common, deployArchiveBuilder,
112         runfilesBuilder, jvmFlags, attributesBuilder, stripAsDefault);
113 
114     DeployArchiveBuilder unstrippedDeployArchiveBuilder = null;
115     Artifact unstrippedLauncher = null;
116     if (stripAsDefault) {
117       unstrippedDeployArchiveBuilder = new DeployArchiveBuilder(semantics, ruleContext);
118       unstrippedLauncher = semantics.getLauncher(ruleContext, common,
119           unstrippedDeployArchiveBuilder, runfilesBuilder, jvmFlags, attributesBuilder,
120           false  /* shouldStrip */);
121     }
122 
123     JavaCompilationArtifacts.Builder javaArtifactsBuilder = new JavaCompilationArtifacts.Builder();
124     Artifact instrumentationMetadata =
125         helper.createInstrumentationMetadata(classJar, javaArtifactsBuilder);
126 
127     NestedSetBuilder<Artifact> filesBuilder = NestedSetBuilder.stableOrder();
128     Artifact executable = null;
129     if (createExecutable) {
130       executable = ruleContext.createOutputArtifact(); // the artifact for the rule itself
131       filesBuilder.add(classJar).add(executable);
132 
133       if (ruleContext.getConfiguration().isCodeCoverageEnabled()) {
134         mainClass = semantics.addCoverageSupport(helper, attributesBuilder,
135             executable, instrumentationMetadata, javaArtifactsBuilder, mainClass);
136       }
137     } else {
138       filesBuilder.add(classJar);
139     }
140 
141     JavaTargetAttributes attributes = helper.getAttributes();
142     List<Artifact> nativeLibraries = attributes.getNativeLibraries();
143     if (!nativeLibraries.isEmpty()) {
144       jvmFlags.add("-Djava.library.path="
145           + JavaCommon.javaLibraryPath(nativeLibraries, ruleContext.getRule().getWorkspaceName()));
146     }
147 
148     JavaConfiguration javaConfig = ruleContext.getFragment(JavaConfiguration.class);
149     if (attributes.hasMessages()) {
150       helper.addTranslations(semantics.translate(ruleContext, javaConfig,
151           attributes.getMessages()));
152     }
153 
154     if (attributes.hasSourceFiles() || attributes.hasSourceJars()
155         || attributes.hasResources() || attributes.hasClassPathResources()) {
156       // We only want to add a jar to the classpath of a dependent rule if it has content.
157       javaArtifactsBuilder.addRuntimeJar(classJar);
158     }
159 
160     // Any JAR files should be added to the collection of runtime jars.
161     javaArtifactsBuilder.addRuntimeJars(attributes.getJarFiles());
162 
163     Artifact outputDepsProto = helper.createOutputDepsProtoArtifact(classJar, javaArtifactsBuilder);
164 
165     common.setJavaCompilationArtifacts(javaArtifactsBuilder.build());
166 
167     // The gensrc jar is created only if the target uses annotation processing. Otherwise,
168     // it is null, and the source jar action will not depend on the compile action.
169     Artifact gensrcJar = helper.createGensrcJar(classJar);
170     Artifact manifestProtoOutput = helper.createManifestProtoOutput(classJar);
171 
172     helper.createCompileAction(
173         classJar, manifestProtoOutput, gensrcJar, outputDepsProto, instrumentationMetadata);
174     helper.createSourceJarAction(srcJar, gensrcJar);
175 
176     Artifact genClassJar = ruleContext.getImplicitOutputArtifact(JavaSemantics.JAVA_BINARY_GEN_JAR);
177     helper.createGenJarAction(classJar, manifestProtoOutput, genClassJar);
178 
179     common.setClassPathFragment(new ClasspathConfiguredFragment(
180         common.getJavaCompilationArtifacts(), attributes, false));
181 
182     // Collect the action inputs for the runfiles collector here because we need to access the
183     // analysis environment, and that may no longer be safe when the runfiles collector runs.
184     Iterable<Artifact> dynamicRuntimeActionInputs =
185         CppHelper.getToolchain(ruleContext).getDynamicRuntimeLinkInputs();
186 
187 
188     Iterables.addAll(jvmFlags, semantics.getJvmFlags(ruleContext, common, launcher, userJvmFlags));
189     if (ruleContext.hasErrors()) {
190       return null;
191     }
192 
193     if (createExecutable) {
194       // Create a shell stub for a Java application
195       semantics.createStubAction(ruleContext, common, jvmFlags, executable, mainClass,
196           common.getJavaBinSubstitution(launcher));
197     }
198 
199     NestedSet<Artifact> transitiveSourceJars = collectTransitiveSourceJars(common, srcJar);
200 
201     // TODO(bazel-team): if (getOptions().sourceJars) then make this a dummy prerequisite for the
202     // DeployArchiveAction ? Needs a few changes there as we can't pass inputs
203     helper.createSourceJarAction(semantics, ImmutableList.<Artifact>of(),
204         transitiveSourceJars.toCollection(),
205         ruleContext.getImplicitOutputArtifact(JavaSemantics.JAVA_BINARY_DEPLOY_SOURCE_JAR));
206 
207     RuleConfiguredTargetBuilder builder =
208         new RuleConfiguredTargetBuilder(ruleContext);
209 
210     semantics.addProviders(ruleContext, common, jvmFlags, classJar, srcJar, genClassJar, gensrcJar,
211         ImmutableMap.<Artifact, Artifact>of(), helper, filesBuilder, builder);
212 
213     NestedSet<Artifact> filesToBuild = filesBuilder.build();
214 
215     collectDefaultRunfiles(runfilesBuilder, ruleContext, common, filesToBuild, launcher,
216         dynamicRuntimeActionInputs);
217     Runfiles defaultRunfiles = runfilesBuilder.build();
218 
219     RunfilesSupport runfilesSupport = createExecutable
220         ? runfilesSupport = RunfilesSupport.withExecutable(
221             ruleContext, defaultRunfiles, executable,
222             semantics.getExtraArguments(ruleContext, common))
223         : null;
224 
225     RunfilesProvider runfilesProvider = RunfilesProvider.withData(
226         defaultRunfiles,
227         new Runfiles.Builder(ruleContext.getWorkspaceName()).merge(runfilesSupport).build());
228 
229     ImmutableList<String> deployManifestLines =
230         getDeployManifestLines(ruleContext, originalMainClass);
231 
232     Artifact deployJar =
233         ruleContext.getImplicitOutputArtifact(JavaSemantics.JAVA_BINARY_DEPLOY_JAR);
234 
235     Artifact unstrippedDeployJar =
236         ruleContext.getImplicitOutputArtifact(JavaSemantics.JAVA_UNSTRIPPED_BINARY_DEPLOY_JAR);
237 
238     deployArchiveBuilder
239         .setOutputJar(deployJar)
240         .setJavaStartClass(mainClass)
241         .setDeployManifestLines(deployManifestLines)
242         .setAttributes(attributes)
243         .addRuntimeJars(common.getJavaCompilationArtifacts().getRuntimeJars())
244         .setIncludeBuildData(true)
245         .setRunfilesMiddleman(
246             runfilesSupport == null ? null : runfilesSupport.getRunfilesMiddleman())
247         .setCompression(COMPRESSED)
248         .setLauncher(launcher);
249 
250     deployArchiveBuilder.build();
251 
252     if (stripAsDefault) {
253       unstrippedDeployArchiveBuilder
254           .setOutputJar(unstrippedDeployJar)
255           .setJavaStartClass(mainClass)
256           .setDeployManifestLines(deployManifestLines)
257           .setAttributes(attributes)
258           .addRuntimeJars(common.getJavaCompilationArtifacts().getRuntimeJars())
259           .setIncludeBuildData(true)
260           .setRunfilesMiddleman(
261               runfilesSupport == null ? null : runfilesSupport.getRunfilesMiddleman())
262           .setCompression(COMPRESSED)
263           .setLauncher(unstrippedLauncher);
264 
265       unstrippedDeployArchiveBuilder.build();
266     } else {
267       // Write an empty file as the name_deploy.jar.unstripped when the default output jar is not
268       // stripped.
269       ruleContext.registerAction(
270           new FileWriteAction(ruleContext.getActionOwner(), unstrippedDeployJar, "", false));
271     }
272 
273     common.addTransitiveInfoProviders(builder, filesToBuild, classJar);
274 
275     return builder
276         .setFilesToBuild(filesToBuild)
277         .add(RunfilesProvider.class, runfilesProvider)
278         .setRunfilesSupport(runfilesSupport, executable)
279         .add(JavaRuntimeClasspathProvider.class,
280             new JavaRuntimeClasspathProvider(common.getRuntimeClasspath()))
281         .add(JavaSourceJarsProvider.class,
282             new JavaSourceJarsProvider(transitiveSourceJars, srcJars))
283         .addOutputGroup(JavaSemantics.SOURCE_JARS_OUTPUT_GROUP, transitiveSourceJars)
284         .addOutputGroup(JavaSemantics.GENERATED_JARS_OUTPUT_GROUP, genClassJar)
285         .build();
286   }

288   // Create the deploy jar and make it dependent on the runfiles middleman if an executable is
289   // created. Do not add the deploy jar to files to build, so we will only build it when it gets
290   // requested.
291   private ImmutableList<String> getDeployManifestLines(RuleContext ruleContext,
292       String originalMainClass) {
293     ImmutableList.Builder<String> builder = ImmutableList.<String>builder()
294           .addAll(ruleContext.attributes().get("deploy_manifest_lines", Type.STRING_LIST));
295     if (ruleContext.getConfiguration().isCodeCoverageEnabled()) {
296       builder.add("Coverage-Main-Class: " + originalMainClass);
297     }
298     return builder.build();
299   }

301   private void collectDefaultRunfiles(Runfiles.Builder builder, RuleContext ruleContext,
302       JavaCommon common, NestedSet<Artifact> filesToBuild, Artifact launcher,
303       Iterable<Artifact> dynamicRuntimeActionInputs) {
304     // Convert to iterable: filesToBuild has a different order.
305     builder.addArtifacts((Iterable<Artifact>) filesToBuild);
306     builder.addArtifacts(common.getJavaCompilationArtifacts().getRuntimeJars());
307     if (launcher != null) {
308       final TransitiveInfoCollection defaultLauncher =
309           JavaHelper.launcherForTarget(semantics, ruleContext);
310       final Artifact defaultLauncherArtifact =
311           JavaHelper.launcherArtifactForTarget(semantics, ruleContext);
312       if (!defaultLauncherArtifact.equals(launcher)) {
313         builder.addArtifact(launcher);
314 
315         // N.B. The "default launcher" referred to here is the launcher target specified through
316         // an attribute or flag. We wish to retain the runfiles of the default launcher, *except*
317         // for the original cc_binary artifact, because we've swapped it out with our custom
318         // launcher. Hence, instead of calling builder.addTarget(), or adding an odd method
319         // to Runfiles.Builder, we "unravel" the call and manually add things to the builder.
320         // Because the NestedSet representing each target's launcher runfiles is re-built here,
321         // we may see increased memory consumption for representing the target's runfiles.
322         Runfiles runfiles =
323             defaultLauncher.getProvider(RunfilesProvider.class)
324               .getDefaultRunfiles();
325         NestedSetBuilder<Artifact> unconditionalArtifacts = NestedSetBuilder.compileOrder();
326         for (Artifact a : runfiles.getUnconditionalArtifacts()) {
327           if (!a.equals(defaultLauncherArtifact)) {
328             unconditionalArtifacts.add(a);
329           }
330         }
331         builder.addTransitiveArtifacts(unconditionalArtifacts.build());
332         builder.addSymlinks(runfiles.getSymlinks());
333         builder.addRootSymlinks(runfiles.getRootSymlinks());
334         builder.addPruningManifests(runfiles.getPruningManifests());
335       } else {
336         builder.addTarget(defaultLauncher, RunfilesProvider.DEFAULT_RUNFILES);
337       }
338     }
339 
340     semantics.addRunfilesForBinary(ruleContext, launcher, builder);
341     builder.addRunfiles(ruleContext, RunfilesProvider.DEFAULT_RUNFILES);
342     builder.add(ruleContext, JavaRunfilesProvider.TO_RUNFILES);
343 
344     List<? extends TransitiveInfoCollection> runtimeDeps =
345         ruleContext.getPrerequisites("runtime_deps", Mode.TARGET);
346     builder.addTargets(runtimeDeps, JavaRunfilesProvider.TO_RUNFILES);
347     builder.addTargets(runtimeDeps, RunfilesProvider.DEFAULT_RUNFILES);
348     semantics.addDependenciesForRunfiles(ruleContext, builder);
349 
350     if (ruleContext.getConfiguration().isCodeCoverageEnabled()) {
351       Artifact instrumentedJar = common.getJavaCompilationArtifacts().getInstrumentedJar();
352       if (instrumentedJar != null) {
353         builder.addArtifact(instrumentedJar);
354       }
355     }
356 
357     builder.addArtifacts((Iterable<Artifact>) common.getRuntimeClasspath());
358 
359     // Add the JDK files if it comes from the source repository (see java_stub_template.txt).
360     TransitiveInfoCollection javabaseTarget = ruleContext.getPrerequisite(":jvm", Mode.HOST);
361     if (javabaseTarget != null) {
362       builder.addArtifacts(
363           (Iterable<Artifact>) javabaseTarget.getProvider(FileProvider.class).getFilesToBuild());
364 
365       // Add symlinks to the C++ runtime libraries under a path that can be built
366       // into the Java binary without having to embed the crosstool, gcc, and grte
367       // version information contained within the libraries' package paths.
368       for (Artifact lib : dynamicRuntimeActionInputs) {
369         PathFragment path = CPP_RUNTIMES.getRelative(lib.getExecPath().getBaseName());
370         builder.addSymlink(path, lib);
371       }
372     }
373   }

375   private NestedSet<Artifact> collectTransitiveSourceJars(JavaCommon common, Artifact srcJar) {
376     NestedSetBuilder<Artifact> builder = NestedSetBuilder.stableOrder();
377 
378     builder.add(srcJar);
379     for (JavaSourceJarsProvider dep : common.getDependencies(JavaSourceJarsProvider.class)) {
380       builder.addTransitive(dep.getTransitiveSourceJars());
381     }
382     return builder.build();
383   }

385   /**
386    * Collects the native libraries in the transitive closure of the deps.
387    *
388    * @param deps the dependencies to be included as roots of the transitive closure.
389    * @return the native libraries found in the transitive closure of the deps.
390    */
391   public static Collection<Artifact> collectNativeLibraries(
392       Iterable<? extends TransitiveInfoCollection> deps) {
393     NestedSet<LinkerInput> linkerInputs = new NativeLibraryNestedSetBuilder()
394         .addJavaTargets(deps)
395         .build();
396     ImmutableList.Builder<Artifact> result = ImmutableList.builder();
397     for (LinkerInput linkerInput : linkerInputs) {
398       result.add(linkerInput.getArtifact());
399     }
400 
401     return result.build();
402   }
