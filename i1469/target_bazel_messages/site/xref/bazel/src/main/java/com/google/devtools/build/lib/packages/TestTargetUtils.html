42     * Returns a predicate to be used for test size filtering, i.e., that only accepts tests of the
43     * given size.
44     */
45    public static Predicate<Target> testSizeFilter(final Set<TestSize> allowedSizes) {
46      return new Predicate<Target>() {
47        @Override
48        public boolean apply(Target target) {
49          if (!(target instanceof Rule)) {
50            return false;
51          }
52          return allowedSizes.contains(TestSize.getTestSize((Rule) target));
53        }
54      };
55    }

73    /**
74     * Returns a predicate to be used for test language filtering, i.e., that only accepts tests of
75     * the specified languages. The reporter and the list of rule names are only used to warn about
76     * unknown languages.
77     */
78    public static Predicate<Target> testLangFilter(List<String> langFilterList,
79        EventHandler reporter, Set<String> allRuleNames) {
80      final Set<String> requiredLangs = new HashSet<>();
81      final Set<String> excludedLangs = new HashSet<>();
82  
83      for (String lang : langFilterList) {
84        if (lang.startsWith("-")) {
85          lang = lang.substring(1);
86          excludedLangs.add(lang);
87        } else {
88          requiredLangs.add(lang);
89        }
90        if (!allRuleNames.contains(lang + "_test")) {
91          reporter.handle(
92              Event.warn("Unknown language '" + lang + "' in --test_lang_filters option"));
93        }
94      }
95  
96      return new Predicate<Target>() {
97        @Override
98        public boolean apply(Target rule) {
99          String ruleLang = TargetUtils.getRuleLanguage(rule);
100         return (requiredLangs.isEmpty() || requiredLangs.contains(ruleLang))
101             && !excludedLangs.contains(ruleLang);
102       }
103     };
104   }

106   /**
107    * Returns whether a test with the specified tags matches a filter (as specified by the set
108    * of its positive and its negative filters).
109    */
110   public static boolean testMatchesFilters(Collection<String> testTags,
111       Collection<String> requiredTags, Collection<String> excludedTags,
112       boolean mustMatchAllPositive) {
113 
114     for (String tag : excludedTags) {
115       if (testTags.contains(tag)) {
116         return false;
117       }
118     }
119 
120     // Check required tags, if there are any.
121     if (!requiredTags.isEmpty()) {
122       if (mustMatchAllPositive) {
123         // Require all tags to be present.
124         for (String tag : requiredTags) {
125           if (!testTags.contains(tag)) {
126             return false;
127           }
128         }
129         return true;
130       } else {
131         // Require at least one positive tag.
132         for (String tag : requiredTags) {
133           if (testTags.contains(tag)) {
134             return true;
135           }
136         }
137       }
138 
139       return false; // No positive tag found.
140     }
141 
142     return true; // No tags are required.
143   }

145   /**
146    * Returns a predicate to be used for test tag filtering, i.e., that only accepts tests that match
147    * all of the required tags and none of the excluded tags.
148    */
149   // TODO(bazel-team): This also applies to non-test rules, so should probably be moved to
150   // TargetUtils.
151   public static Predicate<Target> tagFilter(List<String> tagFilterList) {
152     Pair<Collection<String>, Collection<String>> tagLists = sortTagsBySense(tagFilterList);
153     final Collection<String> requiredTags = tagLists.first;
154     final Collection<String> excludedTags = tagLists.second;
155     return new Predicate<Target>() {
156       @Override
157       public boolean apply(Target input) {
158         if (!(input instanceof Rule)) {
159           return false;
160         }
161         // Note that test_tags are those originating from the XX_test rule,
162         // whereas the requiredTags and excludedTags originate from the command
163         // line or test_suite rule.
164         return testMatchesFilters(((Rule) input).getRuleTags(),
165             requiredTags, excludedTags, false);
166       }
167     };
168   }

170   /**
171    * Separates a list of text "tags" into a Pair of Collections, where
172    * the first element are the required or positive tags and the second element
173    * are the excluded or negative tags.
174    * This should work on tag list provided from the command line
175    * --test_tags_filters flag or on tag filters explicitly declared in the
176    * suite.
177    *
178    * @param tagList A collection of text targets to separate.
179    */
180   public static Pair<Collection<String>, Collection<String>> sortTagsBySense(
181       Iterable<String> tagList) {
182     Collection<String> requiredTags = new HashSet<>();
183     Collection<String> excludedTags = new HashSet<>();
184 
185     for (String tag : tagList) {
186       if (tag.startsWith("-")) {
187         excludedTags.add(tag.substring(1));
188       } else if (tag.startsWith("+")) {
189         requiredTags.add(tag.substring(1));
190       } else if (tag.equals("manual")) {
191         // Ignore manual attribute because it is an exception: it is not a filter
192         // but a property of test_suite
193         continue;
194       } else {
195         requiredTags.add(tag);
196       }
197     }
198     return Pair.of(requiredTags, excludedTags);
199   }

201   /**
202    * Returns the (new, mutable) set of test rules, expanding all 'test_suite' rules into the
203    * individual tests they group together and preserving other test target instances.
204    *
205    * Method assumes that passed collection contains only *_test and test_suite rules. While, at this
206    * point it will successfully preserve non-test rules as well, there is no guarantee that this
207    * behavior will be kept in the future.
208    *
209    * @param targetProvider a target provider
210    * @param eventHandler a failure eventHandler to report loading failures to
211    * @param targets Collection of the *_test and test_suite configured targets
212    * @return a duplicate-free iterable of the tests under the specified targets
213    */
214   public static ResolvedTargets<Target> expandTestSuites(TargetProvider targetProvider,
215       EventHandler eventHandler, Iterable<? extends Target> targets, boolean strict,
216       boolean keepGoing)
217           throws TargetParsingException {
218     Closure closure = new Closure(targetProvider, eventHandler, strict, keepGoing);
219     ResolvedTargets.Builder<Target> result = ResolvedTargets.builder();
220     for (Target target : targets) {
221       if (TargetUtils.isTestRule(target)) {
222         result.add(target);
223       } else if (TargetUtils.isTestSuiteRule(target)) {
224         result.addAll(closure.getTestsInSuite((Rule) target));
225       } else {
226         result.add(target);
227       }
228     }
229     if (closure.hasError) {
230       result.setError();
231     }
232     return result.build();
233   }

235   // TODO(bazel-team): This is a copy of TestsExpression.Closure with some minor changes; this
236   // should be unified.
237   private static final class Closure {
238     private final TargetProvider targetProvider;
239 
240     private final EventHandler eventHandler;
241 
242     private final boolean keepGoing;
243 
244     private final boolean strict;
245 
246     private final Map<Target, Set<Target>> testsInSuite = new HashMap<>();
247 
248     private boolean hasError;
249 
250     public Closure(TargetProvider targetProvider, EventHandler eventHandler, boolean strict,
251         boolean keepGoing) {
252       this.targetProvider = targetProvider;
253       this.eventHandler = eventHandler;
254       this.strict = strict;
255       this.keepGoing = keepGoing;
256     }
257 
258     /**
259      * Computes and returns the set of test rules in a particular suite.  Uses
260      * dynamic programming---a memoized version of {@link #computeTestsInSuite}.
261      */
262     private Set<Target> getTestsInSuite(Rule testSuite) throws TargetParsingException {
263       Set<Target> tests = testsInSuite.get(testSuite);
264       if (tests == null) {
265         tests = Sets.newHashSet();
266         testsInSuite.put(testSuite, tests); // break cycles by inserting empty set early.
267         computeTestsInSuite(testSuite, tests);
268       }
269       return tests;
270     }
271 
272     /**
273      * Populates 'result' with all the tests associated with the specified
274      * 'testSuite'.  Throws an exception if any target is missing.
275      *
276      * CAUTION!  Keep this logic consistent with {@code TestsSuiteConfiguredTarget}!
277      */
278     private void computeTestsInSuite(Rule testSuite, Set<Target> result)
279         throws TargetParsingException {
280       List<Target> testsAndSuites = new ArrayList<>();
281       // Note that testsAndSuites can contain input file targets; the test_suite rule does not
282       // restrict the set of targets that can appear in tests or suites.
283       testsAndSuites.addAll(getPrerequisites(testSuite, "tests"));
284       if (testSuite.getRuleClassObject().hasAttr("suites", Type.LABEL_LIST)) {
285         testsAndSuites.addAll(getPrerequisites(testSuite, "suites"));
286       }
287 
288       // 1. Add all tests
289       for (Target test : testsAndSuites) {
290         if (TargetUtils.isTestRule(test)) {
291           result.add(test);
292         } else if (strict && !TargetUtils.isTestSuiteRule(test)) {
293           // If strict mode is enabled, then give an error for any non-test, non-test-suite targets.
294           eventHandler.handle(Event.error(testSuite.getLocation(),
295               "in test_suite rule '" + testSuite.getLabel()
296               + "': expecting a test or a test_suite rule but '" + test.getLabel()
297               + "' is not one."));
298           hasError = true;
299           if (!keepGoing) {
300             throw new TargetParsingException("Test suite expansion failed.");
301           }
302         }
303       }
304 
305       // 2. Add implicit dependencies on tests in same package, if any.
306       for (Target target : getPrerequisites(testSuite, "$implicit_tests")) {
307         // The Package construction of $implicit_tests ensures that this check never fails, but we
308         // add it here anyway for compatibility with future code.
309         if (TargetUtils.isTestRule(target)) {
310           result.add(target);
311         }
312       }
313 
314       // 3. Filter based on tags, size, env.
315       filterTests(testSuite, result);
316 
317       // 4. Expand all suites recursively.
318       for (Target suite : testsAndSuites) {
319         if (TargetUtils.isTestSuiteRule(suite)) {
320           result.addAll(getTestsInSuite((Rule) suite));
321         }
322       }
323     }

325     /**
326      * Returns the set of rules named by the attribute 'attrName' of test_suite rule 'testSuite'.
327      * The attribute must be a list of labels. If a target cannot be resolved, then an error is
328      * reported to the environment (which may throw an exception if {@code keep_going} is disabled).
329      */
330     private Collection<Target> getPrerequisites(Rule testSuite, String attrName)
331         throws TargetParsingException {
332       try {
333         List<Target> targets = new ArrayList<>();
334         // TODO(bazel-team): This serializes package loading in some cases. We might want to make
335         // this multi-threaded.
336         for (Label label :
337             NonconfigurableAttributeMapper.of(testSuite).get(attrName, Type.LABEL_LIST)) {
338           targets.add(targetProvider.getTarget(eventHandler, label));
339         }
340         return targets;
341       } catch (NoSuchThingException e) {
342         if (keepGoing) {
343           hasError = true;
344           eventHandler.handle(Event.error(e.getMessage()));
345           return ImmutableList.of();
346         }
347         throw new TargetParsingException(e.getMessage(), e);
348       } catch (InterruptedException e) {
349         Thread.currentThread().interrupt();
350         throw new TargetParsingException("interrupted", e);
351       }
352     }

354     /**
355      * Filters 'tests' (by mutation) according to the 'tags' attribute, specifically those that
356      * match ALL of the tags in tagsAttribute.
357      *
358      * @precondition {@code env.getAccessor().isTestSuite(testSuite)}
359      * @precondition {@code env.getAccessor().isTestRule(test)} for all test in tests
360      */
361     private void filterTests(Rule testSuite, Set<Target> tests) {
362       List<String> tagsAttribute =
363           NonconfigurableAttributeMapper.of(testSuite).get("tags", Type.STRING_LIST);
364       // Split the tags list into positive and negative tags
365       Pair<Collection<String>, Collection<String>> tagLists = sortTagsBySense(tagsAttribute);
366       Collection<String> positiveTags = tagLists.first;
367       Collection<String> negativeTags = tagLists.second;
368 
369       Iterator<Target> it = tests.iterator();
370       while (it.hasNext()) {
371         Rule test = (Rule) it.next();
372         AttributeMap nonConfigurableAttributes = NonconfigurableAttributeMapper.of(test);
373         List<String> testTags =
374             new ArrayList<>(nonConfigurableAttributes.get("tags", Type.STRING_LIST));
375         testTags.add(nonConfigurableAttributes.get("size", Type.STRING));
376         if (!includeTest(testTags, positiveTags, negativeTags)) {
377           it.remove();
378         }
379       }
380     }

382     /**
383      * Decides whether to include a test in a test_suite or not.
384      * @param testTags Collection of all tags exhibited by a given test.
385      * @param positiveTags Tags declared by the suite. A Test must match ALL of these.
386      * @param negativeTags Tags declared by the suite. A Test must match NONE of these.
387      * @return false is the test is to be removed.
388      */
389     private static boolean includeTest(Collection<String> testTags,
390         Collection<String> positiveTags, Collection<String> negativeTags) {
391       // Add this test if it matches ALL of the positive tags and NONE of the
392       // negative tags in the tags attribute.
393       for (String tag : negativeTags) {
394         if (testTags.contains(tag)) {
395           return false;
396         }
397       }
398       for (String tag : positiveTags) {
399         if (!testTags.contains(tag)) {
400           return false;
401         }
402       }
403       return true;
404     }
405   }
