34  /**
35   * Implementation for ios_test rule in Bazel.
36   */
37  public final class BazelIosTest extends IosTest {
38    static final String IOS_TEST_ON_BAZEL_ATTR = "$ios_test_on_bazel";
39  
40    @Override
41    public ConfiguredTarget create(RuleContext ruleContext, ObjcCommon common,
42        XcodeProvider xcodeProvider, NestedSet<Artifact> filesToBuild) throws InterruptedException {
43      Artifact testRunner = ruleContext.getPrerequisiteArtifact(IOS_TEST_ON_BAZEL_ATTR, Mode.TARGET);
44  
45      Artifact testScript =
46          ObjcRuleClasses.artifactByAppendingToBaseName(ruleContext, "_test_script");
47  
48      ruleContext.registerAction(
49          new TemplateExpansionAction(
50              ruleContext.getActionOwner(),
51              testRunner,
52              testScript,
53              ImmutableList.<TemplateExpansionAction.Substitution>of(),
54              /*executable=*/ true));
55  
56      Runfiles runfiles = new Runfiles.Builder(ruleContext.getWorkspaceName())
57          .addArtifact(testScript).build();
58      RunfilesSupport runfilesSupport =
59          RunfilesSupport.withExecutable(ruleContext, runfiles, testScript);
60  
61      return new RuleConfiguredTargetBuilder(ruleContext)
62          .setFilesToBuild(NestedSetBuilder.<Artifact>stableOrder()
63              .addTransitive(filesToBuild)
64              .add(testRunner)
65              .build())
66          .add(XcodeProvider.class, xcodeProvider)
67          .add(RunfilesProvider.class, RunfilesProvider.simple(runfiles))
68          .setRunfilesSupport(runfilesSupport, testRunner)
69          .build();
70    }
71  }
