53  /**
54   * A loader that reads Crosstool configuration files and creates CToolchain
55   * instances from them.
56   *
57   * <p>Note that this class contains a cache for the text format -> proto objects mapping of
58   * Crosstool protos that is completely independent from Skyframe or anything else. This should be
59   * done in a saner way.
60   */
61  public class CrosstoolConfigurationLoader {
62    private static final String CROSSTOOL_CONFIGURATION_FILENAME = "CROSSTOOL";
63  
64    /**
65     * Cache for storing result of toReleaseConfiguration function based on path and md5 sum of
66     * input file. We can use md5 because result of this function depends only on the file content.
67     */
68    private static final Cache<String, CrosstoolRelease> crosstoolReleaseCache =
69        CacheBuilder.newBuilder().concurrencyLevel(4).maximumSize(100).build();
70    /**
71     * A class that holds the results of reading a CROSSTOOL file.
72     */
73    public static class CrosstoolFile {
74      private final String location;
75      private final CrosstoolConfig.CrosstoolRelease crosstool;
76      private final String md5;
77  
78      CrosstoolFile(String location, CrosstoolConfig.CrosstoolRelease crosstool, String md5) {
79        this.location = location;
80        this.crosstool = crosstool;
81        this.md5 = md5;
82      }
83  
84      /**
85       * Returns a user-friendly location of the CROSSTOOL proto for error messages.
86       */
87      public String getLocation() {
88        return location;
89      }
90  
91      /**
92       * Returns the parsed contents of the CROSSTOOL file.
93       */
94      public CrosstoolConfig.CrosstoolRelease getProto() {
95        return crosstool;
96      }
97  
98      /**
99       * Returns an MD5 hash of the CROSSTOOL file contents.
100      */
101     public String getMd5() {
102       return md5;
103     }
104   }

109   /**
110    * Reads the given <code>data</code> String, which must be in ascii format,
111    * into a protocol buffer. It uses the <code>name</code> parameter for error
112    * messages.
113    *
114    * @throws IOException if the parsing failed
115    */
116   @VisibleForTesting
117   static CrosstoolConfig.CrosstoolRelease toReleaseConfiguration(String name, String data)
118       throws IOException {
119     CrosstoolConfig.CrosstoolRelease.Builder builder =
120         CrosstoolConfig.CrosstoolRelease.newBuilder();
121     try {
122       TextFormat.merge(data, builder);
123       return builder.build();
124     } catch (ParseException e) {
125       throw new IOException("Could not read the crosstool configuration file '" + name + "', "
126           + "because of a parser error (" + e.getMessage() + ")");
127     } catch (UninitializedMessageException e) {
128       throw new IOException("Could not read the crosstool configuration file '" + name + "', "
129           + "because of an incomplete protocol buffer (" + e.getMessage() + ")");
130     }
131   }

133   /**
134    * This class is the in-memory representation of a text-formatted Crosstool proto file.
135    *
136    * <p>This layer of abstraction is here so that we can load these protos either from BUILD files
137    * or from CROSSTOOL files.
138    *
139    * <p>An implementation of this class should override {@link #getContents()} and call
140    * the constructor with the MD5 checksum of what that method will return and a human-readable name
141    * used in error messages.
142    */
143   private abstract static class CrosstoolProto {
144     private final byte[] md5;
145     private final String name;
146 
147     private CrosstoolProto(byte[] md5, String name) {
148       this.md5 = md5;
149       this.name = name;
150     }
151 
152     /**
153      * The binary MD5 checksum of the proto.
154      */
155     public byte[] getMd5() {
156       return md5;
157     }
158 
159     /**
160      * A user-friendly string describing the location of the proto.
161      */
162     public String getName() {
163       return name;
164     }
165 
166     /**
167      * The proto itself.
168      */
169     public abstract String getContents() throws IOException;
170   }

172   private static CrosstoolProto getCrosstoolProtofromBuildFile(
173       ConfigurationEnvironment env, Label crosstoolTop) {
174     Target target;
175     try {
176       target = env.getTarget(crosstoolTop);
177     } catch (NoSuchThingException e) {
178       throw new IllegalStateException(e);  // Should have beeen evaluated by RedirectChaser
179     }
180 
181     if (!(target instanceof Rule)) {
182       return null;
183     }
184 
185     Rule rule = (Rule) target;
186     if (!(rule.getRuleClass().equals("cc_toolchain_suite"))
187         || !rule.isAttributeValueExplicitlySpecified("proto")) {
188       return null;
189     }
190 
191     final String contents = NonconfigurableAttributeMapper.of(rule).get("proto", Type.STRING);
192     byte[] md5 = new Fingerprint().addBytes(contents.getBytes(UTF_8)).digestAndReset();
193     return new CrosstoolProto(md5, "cc_toolchain_suite rule " + crosstoolTop.toString()) {
194 
195       @Override
196       public String getContents() throws IOException {
197         return contents;
198       }
199     };
200   }

202   private static CrosstoolProto getCrosstoolProtoFromCrosstoolFile(
203       ConfigurationEnvironment env, Label crosstoolTop)
204       throws IOException, InvalidConfigurationException {
205     final Path path;
206     try {
207       Package containingPackage = env.getTarget(crosstoolTop.getLocalTargetLabel("BUILD"))
208           .getPackage();
209       if (containingPackage == null) {
210         return null;
211       }
212       path = env.getPath(containingPackage, CROSSTOOL_CONFIGURATION_FILENAME);
213     } catch (SyntaxException e) {
214       throw new InvalidConfigurationException(e);
215     } catch (NoSuchThingException e) {
216       // Handled later
217       return null;
218     }
219 
220     if (path == null || !path.exists()) {
221       return null;
222     }
223 
224     return new CrosstoolProto(path.getMD5Digest(), "CROSSTOOL file " + path.getPathString()) {
225       @Override
226       public String getContents() throws IOException {
227         return new String(FileSystemUtils.readContentAsLatin1(path.getInputStream()));
228       }
229     };
230   }

232   private static CrosstoolFile findCrosstoolConfiguration(
233       ConfigurationEnvironment env, Label crosstoolTop)
234       throws IOException, InvalidConfigurationException {
235 
236     CrosstoolProto crosstoolProto = getCrosstoolProtofromBuildFile(env, crosstoolTop);
237     if (crosstoolProto == null) {
238       crosstoolProto = getCrosstoolProtoFromCrosstoolFile(env, crosstoolTop);
239     }
240 
241     if (crosstoolProto == null) {
242       throw new InvalidConfigurationException("The crosstool_top you specified was resolved to '" +
243           crosstoolTop + "', which does not contain a CROSSTOOL file. " +
244           "You can use a crosstool from the depot by specifying its label.");
245     } else {
246       // Do this before we read the data, so if it changes, we get a different MD5 the next time.
247       // Alternatively, we could calculate the MD5 of the contents, which we also read, but this
248       // is faster if the file comes from a file system with md5 support.
249       final CrosstoolProto finalProto = crosstoolProto;
250       String md5 = BaseEncoding.base16().lowerCase().encode(finalProto.getMd5());
251       CrosstoolConfig.CrosstoolRelease release;
252       try {
253         release = crosstoolReleaseCache.get(md5, new Callable<CrosstoolRelease>() {
254           @Override
255           public CrosstoolRelease call() throws Exception {
256             return toReleaseConfiguration(finalProto.getName(), finalProto.getContents());
257           }
258         });
259       } catch (ExecutionException e) {
260         throw new InvalidConfigurationException(e);
261       }
262 
263       return new CrosstoolFile(finalProto.getName(), release, md5);
264     }
265   }

267   /**
268    * Reads a crosstool file.
269    */
270   @Nullable
271   public static CrosstoolConfigurationLoader.CrosstoolFile readCrosstool(
272       ConfigurationEnvironment env, Label crosstoolTop) throws InvalidConfigurationException {
273     crosstoolTop = RedirectChaser.followRedirects(env, crosstoolTop, "crosstool_top");
274     if (crosstoolTop == null) {
275       return null;
276     }
277     try {
278       return findCrosstoolConfiguration(env, crosstoolTop);
279     } catch (IOException e) {
280       throw new InvalidConfigurationException(e);
281     }
282   }

284   /**
285    * Selects a crosstool toolchain corresponding to the given crosstool
286    * configuration options. If all of these options are null, it returns the default
287    * toolchain specified in the crosstool release. If only cpu is non-null, it
288    * returns the default toolchain for that cpu, as specified in the crosstool
289    * release. Otherwise, all values must be non-null, and this method
290    * returns the toolchain which matches all of the values.
291    *
292    * @throws NullPointerException if {@code release} is null
293    * @throws InvalidConfigurationException if no matching toolchain can be found, or
294    *     if the input parameters do not obey the constraints described above
295    */
296   public static CrosstoolConfig.CToolchain selectToolchain(
297       CrosstoolConfig.CrosstoolRelease release, BuildOptions options,
298       Function<String, String> cpuTransformer)
299           throws InvalidConfigurationException {
300     CrosstoolConfigurationIdentifier config =
301         CrosstoolConfigurationIdentifier.fromReleaseAndCrosstoolConfiguration(release, options);
302     if ((config.getCompiler() != null) || (config.getLibc() != null)) {
303       ArrayList<CrosstoolConfig.CToolchain> candidateToolchains = new ArrayList<>();
304       for (CrosstoolConfig.CToolchain toolchain : release.getToolchainList()) {
305         if (config.isCandidateToolchain(toolchain)) {
306           candidateToolchains.add(toolchain);
307         }
308       }
309       switch (candidateToolchains.size()) {
310         case 0: {
311           StringBuilder message = new StringBuilder();
312           message.append("No toolchain found for");
313           message.append(config.describeFlags());
314           message.append(". Valid toolchains are: ");
315           describeToolchainList(message, release.getToolchainList());
316           throw new InvalidConfigurationException(message.toString());
317         }
318         case 1:
319           return candidateToolchains.get(0);
320         default: {
321           StringBuilder message = new StringBuilder();
322           message.append("Multiple toolchains found for");
323           message.append(config.describeFlags());
324           message.append(": ");
325           describeToolchainList(message, candidateToolchains);
326           throw new InvalidConfigurationException(message.toString());
327         }
328       }
329     }
330     String selectedIdentifier = null;
331     // We use fake CPU values to allow cross-platform builds for other languages that use the
332     // C++ toolchain. Translate to the actual target architecture.
333     String desiredCpu = cpuTransformer.apply(config.getCpu());
334     for (CrosstoolConfig.DefaultCpuToolchain selector : release.getDefaultToolchainList()) {
335       if (selector.getCpu().equals(desiredCpu)) {
336         selectedIdentifier = selector.getToolchainIdentifier();
337         break;
338       }
339     }
340     checkToolChain(selectedIdentifier, desiredCpu);
341     for (CrosstoolConfig.CToolchain toolchain : release.getToolchainList()) {
342       if (toolchain.getToolchainIdentifier().equals(selectedIdentifier)) {
343         return toolchain;
344       }
345     }
346     throw new InvalidConfigurationException("Inconsistent crosstool configuration; no toolchain "
347         + "corresponding to '" + selectedIdentifier + "' found for cpu '" + config.getCpu() + "'");
348   }

350   private static String describeToolchainFlags(CrosstoolConfig.CToolchain toolchain) {
351     return CrosstoolConfigurationIdentifier.fromToolchain(toolchain).describeFlags();
352   }

354   /**
355    * Appends a series of toolchain descriptions (as the blaze command line flags
356    * that would specify that toolchain) to 'message'.
357    */
358   private static void describeToolchainList(StringBuilder message,
359       Collection<CrosstoolConfig.CToolchain> toolchains) {
360     message.append("[");
361     for (CrosstoolConfig.CToolchain toolchain : toolchains) {
362       message.append(describeToolchainFlags(toolchain));
363       message.append(",");
364     }
365     message.append("]");
366   }

368   /**
369    * Makes sure that {@code selectedIdentifier} is a valid identifier for a toolchain,
370    * i.e. it starts with a letter or an underscore and continues with only dots, dashes,
371    * spaces, letters, digits or underscores (i.e. matches the following regular expression:
372    * "[a-zA-Z_][\.\- \w]*").
373    *
374    * @throws InvalidConfigurationException if selectedIdentifier is null or does not match the
375    *         aforementioned regular expression.
376    */
377   private static void checkToolChain(String selectedIdentifier, String cpu)
378       throws InvalidConfigurationException {
379     if (selectedIdentifier == null) {
380       throw new InvalidConfigurationException("No toolchain found for cpu '" + cpu + "'");
381     }
382     // If you update this regex, please do so in the javadoc comment too, and also in the
383     // crosstool_config.proto file.
384     String rx = "[a-zA-Z_][\\.\\- \\w]*";
385     if (!selectedIdentifier.matches(rx)) {
386       throw new InvalidConfigurationException(String.format(
387           "Toolchain identifier '%s' for cpu '%s' is illegal (does not match '%s')",
388           selectedIdentifier, cpu, rx));
389     }
390   }

392   public static CrosstoolConfig.CrosstoolRelease getCrosstoolReleaseProto(
393       ConfigurationEnvironment env, BuildOptions options,
394       Label crosstoolTop, Function<String, String> cpuTransformer)
395       throws InvalidConfigurationException {
396     CrosstoolConfigurationLoader.CrosstoolFile file =
397         readCrosstool(env, crosstoolTop);
398     // Make sure that we have the requested toolchain in the result. Throw an exception if not.
399     selectToolchain(file.getProto(), options, cpuTransformer);
400     return file.getProto();
401   }
