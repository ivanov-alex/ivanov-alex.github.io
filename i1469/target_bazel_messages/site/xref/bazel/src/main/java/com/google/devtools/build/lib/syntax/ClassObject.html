14  package com.google.devtools.build.lib.syntax;
15  
16  import com.google.common.base.Joiner;
17  import com.google.common.base.Preconditions;
18  import com.google.common.collect.ImmutableCollection;
19  import com.google.common.collect.ImmutableMap;
20  import com.google.common.collect.Ordering;
21  import com.google.common.collect.Sets;
22  import com.google.common.collect.Sets.SetView;
23  import com.google.devtools.build.lib.concurrent.ThreadSafety.Immutable;
24  import com.google.devtools.build.lib.events.Location;
25  
26  import java.util.Map;
27  
28  import javax.annotation.Nullable;
29  
30  /**
31   * An interface for objects behaving like Skylark structs.
32   */
33  // TODO(bazel-team): type checks
34  public interface ClassObject {
35  
36    /**
37     * Returns the value associated with the name field in this struct,
38     * or null if the field does not exist.
39     */
40    @Nullable
41    Object getValue(String name);
42  
43    /**
44     * Returns the fields of this struct.
45     */
46    ImmutableCollection<String> getKeys();
47  
48    /**
49     * Returns a customized error message to print if the name is not a valid struct field
50     * of this struct, or returns null to use the default error message.
51     */
52    @Nullable String errorMessage(String name);
53  
54    /**
55     * An implementation class of ClassObject for structs created in Skylark code.
56     */
57    // TODO(bazel-team): maybe move the SkylarkModule annotation to the ClassObject interface?
58    @Immutable
59    @SkylarkModule(name = "struct",
60        doc = "A special language element to support structs (i.e. simple value objects). "
61            + "See the global <a href=\"globals.html#struct\">struct</a> function "
62            + "for more details.")
63    public class SkylarkClassObject implements ClassObject {
64  
65      private final ImmutableMap<String, Object> values;
66      private final Location creationLoc;
67      private final String errorMessage;
68  
69      /**
70       * Creates a built-in struct (i.e. without creation loc). The errorMessage has to have
71       * exactly one '%s' parameter to substitute the struct field name.
72       */
73      public SkylarkClassObject(Map<String, Object> values, String errorMessage) {
74        this.values = ImmutableMap.copyOf(values);
75        this.creationLoc = null;
76        this.errorMessage = errorMessage;
77      }
78  
79      public SkylarkClassObject(Map<String, Object> values, Location creationLoc) {
80        this.values = ImmutableMap.copyOf(values);
81        this.creationLoc = Preconditions.checkNotNull(creationLoc);
82        this.errorMessage = null;
83      }
84  
85      @Override
86      public Object getValue(String name) {
87        return values.get(name);
88      }
89  
90      @Override
91      public ImmutableCollection<String> getKeys() {
92        return values.keySet();
93      }
94  
95      public Location getCreationLoc() {
96        return Preconditions.checkNotNull(creationLoc,
97            "This struct was not created in a Skylark code");
98      }
99  
100     static SkylarkClassObject concat(
101         SkylarkClassObject lval, SkylarkClassObject rval, Location loc) throws EvalException {
102       SetView<String> commonFields = Sets.intersection(lval.values.keySet(), rval.values.keySet());
103       if (!commonFields.isEmpty()) {
104         throw new EvalException(loc, "Cannot concat structs with common field(s): "
105             + Joiner.on(",").join(commonFields));
106       }
107       return new SkylarkClassObject(ImmutableMap.<String, Object>builder()
108           .putAll(lval.values).putAll(rval.values).build(), loc);
109     }
110 
111     @Override
112     public String errorMessage(String name) {
113       return errorMessage != null ? String.format(errorMessage, name) : null;
114     }
115 
116     /**
117      * Convert the object to string using Skylark syntax. The output tries to be
118      * reversible (but there is no guarantee, it depends on the actual values).
119      */
120     @Override
121     public String toString() {
122       StringBuilder builder = new StringBuilder();
123       boolean first = true;
124       builder.append("struct(");
125       // Sort by key to ensure deterministic output.
126       for (String key : Ordering.natural().sortedCopy(values.keySet())) {
127         if (!first) {
128           builder.append(", ");
129         }
130         first = false;
131         builder.append(key);
132         builder.append(" = ");
133         Printer.write(builder, values.get(key));
134       }
135       builder.append(")");
136       return builder.toString();
137     }
138   }
139 }
