26  /**
27   * Command-line options for the Blaze query language, revision 2.

29  public class QueryOptions extends OptionsBase {
30    /** An enum converter for {@code  AspectResolver.Mode} . Should be used internally only. */

32      public AspectResolutionModeConverter() {
33        super(AspectResolver.Mode.class, "Aspect resolution mode");
34      }
35    }

37    /** An enum converter for {@code OrderOutput} . Should be used internally only. */
38    public static class OrderOutputConverter extends EnumConverter<OrderOutput> {
39      public OrderOutputConverter() {
40        super(OrderOutput.class, "Order output setting");
41      }
42    }
43  
44    @Option(name = "output",
45        defaultValue = "label",
46        category = "query",
47        help = "The format in which the query results should be printed."
48            + " Allowed values are: label, label_kind, minrank, maxrank, package, location, graph,"
49            + " xml, proto, record.")
50    public String outputFormat;
51

53      name = "order_results",
54      defaultValue = "null",
55      category = "query",
56      deprecationWarning = "Please use --order_output=auto or --order_output=no instead of this flag",
57      expansion = {"--order_output=auto"},
58      help =
59          "Output the results in dependency-ordered (default) or unordered fashion. The "
60              + "unordered output is faster but only supported when --output is not minrank, "
61              + "maxrank, or graph."
62    )
63    public Void orderResults;
64  
65    @Option(
66      name = "noorder_results",
67      defaultValue = "null",
68      category = "query",
69      deprecationWarning = "Please use --order_output=no or --order_output=auto instead of this flag",
70      expansion = {"--order_output=no"},
71      help =
72          "Output the results in dependency-ordered (default) or unordered fashion. The "
73              + "unordered output is faster but only supported when --output is not minrank, "
74              + "maxrank, or graph."
75    )
76    public Void noOrderResults;
77  
78    /** Whether and how output should be ordered. */
79    public enum OrderOutput {
80      NO, /** Make no effort to order output besides that required by output formatter. */
81      DEPS, /** Output in dependency order when compatible with output formatter. */
82      AUTO, /** Same as full unless formatter is proto, minrank, maxrank, or graph, then deps. */
83      FULL /** Output in dependency order, breaking ties with alphabetical order when needed. */
84    }
85  
86    @Option(
87      name = "order_output",
88      converter = OrderOutputConverter.class,
89      defaultValue = "auto",
90      category = "query",
91      help =
92          "Output the results unordered (no), dependency-ordered (deps), or fully ordered (full). "
93              + "The default is 'auto', meaning that results are output either dependency-ordered or "
94              + "fully ordered, depending on the output formatter (dependency-ordered for proto, "
95              + "minrank, maxrank, and graph, fully ordered for all others). When output is fully "
96              + "ordered, nodes that would otherwise be unordered by the output formatter are "
97              + "alphabetized before output."
98    )
99    public OrderOutput orderOutput;

117   @Option(name = "host_deps",
118       defaultValue = "true",
119       category = "query",
120           help = "If enabled, dependencies on 'host configuration' targets will be included in "
121           + "the dependency graph over which the query operates.  A 'host configuration' "

125           + "discover the set of things needed during a build will typically enable this option; "
126           + "queries aimed at revealing the structure of a single program will typically  disable "

168           + "--output=xml.")

171   @Option(name = "xml:default_values",
172       defaultValue = "false",

177

179       defaultValue = "false",

181       help = "If true, the tests() expression gives an error if it encounters a test_suite "
182           + "containing non-test targets.")
183   public boolean strictTestSuite;
184 
185   @Option(name = "universe_scope",
186       converter = Converters.CommaSeparatedOptionListConverter.class,
187       defaultValue = "",

189       help = "A comma-separated set of target patterns (additive and subtractive). The query may "

194   @Option(name = "relative_locations",
195       defaultValue = "false",
196       category = "query",
197       help = "If true, the location of BUILD files in xml and proto outputs will be relative. "
198         + "By default, the location output is an absolute path and will not be consistent "
199         + "across machines. You can set this option to true to have a consistent result "
200         + "across machines.")
201   public boolean relativeLocations;
202 
203   @Option(name = "aspect_deps",
204       converter = AspectResolutionModeConverter.class,
205       defaultValue = "conservative",
206       category = "query",
207       help = "How to resolve aspect dependencies when the output format is one of "
208           + "{xml,proto,record}. 'off' means no aspect dependencies are resolved, 'conservative' "
209           + "(the default) means all declared aspect dependencies are added regardless of whether "
210           + "they are viable given the rule class of direct dependencies, 'precise' means that "
211           + "only those aspects are added that are possibly active given the rule class of the "
212           + "direct dependencies. Note that precise mode requires loading other packages to "
213           + "evaluate a single target thus making it slower than the other modes. Also note that "
214           + "even precise mode is not completely precise: the decision whether to compute an "
215           + "aspect is decided in the analysis phase, which is not run during 'blaze query'.")
216   public AspectResolver.Mode aspectDeps;
217 
218   /**
219    * Return the current options as a set of QueryEnvironment settings.
220    */
221   public Set<Setting> toSettings() {
222     Set<Setting> settings = EnumSet.noneOf(Setting.class);
223     if (strictTestSuite) {
224       settings.add(Setting.TESTS_EXPRESSION_STRICT);
225     }
226     if (!includeHostDeps) {
227       settings.add(Setting.NO_HOST_DEPS);
228     }
229     if (!includeImplicitDeps) {
230       settings.add(Setting.NO_IMPLICIT_DEPS);
231     }
232     return settings;
233   }
