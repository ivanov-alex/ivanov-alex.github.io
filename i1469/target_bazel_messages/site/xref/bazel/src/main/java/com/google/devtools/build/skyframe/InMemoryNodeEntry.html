30  /**
31   * In-memory implementation of {@link NodeEntry}. All operations on this class are thread-safe.
32   *
33   * <p>Care was taken to provide certain compound operations to avoid certain check-then-act races.
34   * That means this class is somewhat closely tied to the exact Evaluator implementation.
35   *
36   * <p>Consider the example with two threads working on two nodes, where one depends on the other,
37   * say b depends on a. If a completes first, it's done. If it completes second, it needs to signal
38   * b, and potentially re-schedule it. If b completes first, it must exit, because it will be
39   * signaled (and re-scheduled) by a. If it completes second, it must signal (and re-schedule)
40   * itself. However, if the Evaluator supported re-entrancy for a node, then this wouldn't have to
41   * be so strict, because duplicate scheduling would be less problematic.
42   *
43   * <p>The transient state of an {@code InMemoryNodeEntry} is kept in a {@link BuildingState} object.
44   * Many of the methods of {@code InMemoryNodeEntry} are just wrappers around the corresponding
45   * {@link BuildingState} methods.
46   *
47   * <p>This class is public only for the benefit of alternative graph implementations outside of the
48   * package.
49   */
50  public class InMemoryNodeEntry implements NodeEntry {
51  
52    /** Actual data stored in this entry when it is done. */
53    private SkyValue value = null;
54  
55    /**
56     * The last version of the graph at which this node entry was changed. In {@link #setValue} it
57     * may be determined that the data being written to the graph at a given version is the same as
58     * the already-stored data. In that case, the version will remain the same. The version can be
59     * thought of as the latest timestamp at which this entry was changed.
60     */
61    protected Version version = MinimalVersion.INSTANCE;
62  
63    /**
64     * This object represents a {@link GroupedList}<SkyKey> in a memory-efficient way. It stores the
65     * direct dependencies of this node, in groups if the {@code SkyFunction} requested them that way.
66     */
67    private Object directDeps = null;
68  
69    /**
70     * This list stores the reverse dependencies of this node that have been declared so far.
71     *
72     * <p>In case of a single object we store the object unwrapped, without the list, for
73     * memory-efficiency.
74     */
75    @VisibleForTesting
76    protected Object reverseDeps = ImmutableList.of();
77  
78    /**
79     * We take advantage of memory alignment to avoid doing a nasty {@code instanceof} for knowing
80     * if {@code reverseDeps} is a single object or a list.
81     */
82    protected boolean reverseDepIsSingleObject = false;
83  
84    /**
85     * During the invalidation we keep the reverse deps to be removed in this list instead of directly
86     * removing them from {@code reverseDeps}. That is because removals from reverseDeps are O(N).
87     * Originally reverseDeps was a HashSet, but because of memory consumption we switched to a list.
88     *
89     * <p>This requires that any usage of reverseDeps (contains, add, the list of reverse deps) call
90     * {@code consolidateReverseDepsRemovals} first. While this operation is not free, it can be done
91     * more effectively than trying to remove each dirty reverse dependency individually (O(N) each
92     * time).
93     */
94    private List<SkyKey> reverseDepsToRemove = null;
95  
96    protected static final ReverseDepsUtil<InMemoryNodeEntry> REVERSE_DEPS_UTIL =
97        new ReverseDepsUtil<InMemoryNodeEntry>() {
98      @Override
99      void setReverseDepsObject(InMemoryNodeEntry container, Object object) {
100       container.reverseDeps = object;
101     }
102 
103     @Override
104     void setSingleReverseDep(InMemoryNodeEntry container, boolean singleObject) {
105       container.reverseDepIsSingleObject = singleObject;
106     }
107 
108     @Override

110       container.reverseDepsToRemove = object;
111     }
112 
113     @Override
114     Object getReverseDepsObject(InMemoryNodeEntry container) {
115       return container.reverseDeps;
116     }
117 
118     @Override
119     boolean isSingleReverseDep(InMemoryNodeEntry container) {
120       return container.reverseDepIsSingleObject;
121     }
122 
123     @Override
124     List<SkyKey> getReverseDepsToRemove(InMemoryNodeEntry container) {
125       return container.reverseDepsToRemove;
126     }
127   };

164   @Override
165   public synchronized SkyValue toValue() {
166     if (isDone()) {
167       return getErrorInfo() == null ? getValue() : null;
168     } else if (isChanged() || isDirty()) {
169       return (buildingState.getLastBuildValue() == null)
170               ? null
171           : ValueWithMetadata.justValue(buildingState.getLastBuildValue());
172     }
173     throw new AssertionError("Value in bad state: " + this);
174   }

200   private synchronized Set<SkyKey> setStateFinishedAndReturnReverseDeps() {
201     // Get reverse deps that need to be signaled.
202     ImmutableSet<SkyKey> reverseDepsToSignal = buildingState.getReverseDepsToSignal();
203     REVERSE_DEPS_UTIL.consolidateReverseDepsRemovals(this);
204     REVERSE_DEPS_UTIL.addReverseDeps(this, reverseDepsToSignal);
205     this.directDeps = buildingState.getFinishedDirectDeps().compress();
206 
207     // Set state of entry to done.
208     buildingState = null;
209 
210     if (!keepEdges()) {
211       this.directDeps = null;
212       this.reverseDeps = null;
213     }
214     return reverseDepsToSignal;
215   }

217   @Override
218   public synchronized Set<SkyKey> getInProgressReverseDeps() {
219     Preconditions.checkState(!isDone(), this);
220     return buildingState.getReverseDepsToSignal();
221   }

223   @Override
224   public synchronized Set<SkyKey> setValue(SkyValue value, Version version) {
225     Preconditions.checkState(isReady(), "%s %s", this, value);
226     // This check may need to be removed when we move to a non-linear versioning sequence.
227     Preconditions.checkState(this.version.atMost(version),
228         "%s %s %s", this, version, value);
229 
230     if (isDirty() && buildingState.unchangedFromLastBuild(value)) {
231       // If the value is the same as before, just use the old value. Note that we don't use the new
232       // value, because preserving == equality is even better than .equals() equality.
233       this.value = buildingState.getLastBuildValue();
234     } else {
235       // If this is a new value, or it has changed since the last build, set the version to the
236       // current graph version.
237       this.version = version;
238       this.value = value;
239     }
240 
241     return setStateFinishedAndReturnReverseDeps();
242   }

244   @Override
245   public synchronized DependencyState addReverseDepAndCheckIfDone(SkyKey reverseDep) {
246     if (reverseDep != null) {
247       if (keepEdges()) {
248         REVERSE_DEPS_UTIL.consolidateReverseDepsRemovals(this);
249         REVERSE_DEPS_UTIL.maybeCheckReverseDepNotPresent(this, reverseDep);
250       }
251       if (isDone()) {
252         if (keepEdges()) {
253           REVERSE_DEPS_UTIL.addReverseDeps(this, ImmutableList.of(reverseDep));
254         }
255       } else {
256         // Parent should never register itself twice in the same build.
257         buildingState.addReverseDepToSignal(reverseDep);
258       }
259     }
260     if (isDone()) {
261       return DependencyState.DONE;
262     }
263     return buildingState.startEvaluating() ? DependencyState.NEEDS_SCHEDULING
264                                            : DependencyState.ADDED_DEP;
265   }
266 
267   @Override
268   public synchronized void removeReverseDep(SkyKey reverseDep) {
269     if (!keepEdges()) {
270       return;
271     }
272     REVERSE_DEPS_UTIL.removeReverseDep(this, reverseDep);
273     if (!isDone()) {
274       // This is currently unnecessary -- the only time we remove a reverse dep that was added this
275       // build is during the clean following a build failure. In that case, this node that is not
276       // done will be deleted soon, so clearing the reverse dep is not required.
277       buildingState.removeReverseDepToSignal(reverseDep);
278     }
279   }

281   @Override
282   public synchronized Collection<SkyKey> getReverseDeps() {
283     assertKeepEdges();
284     Preconditions.checkState(isDone() || buildingState.getReverseDepsToSignal().isEmpty(),
285         "Reverse deps should only be queried before the build has begun "
286             + "or after the node is done %s", this);
287     return REVERSE_DEPS_UTIL.getReverseDeps(this);
288   }

311   /** Checks that a caller is not trying to access not-stored graph edges. */
312   private void assertKeepEdges() {
313     Preconditions.checkState(keepEdges(), "Graph edges not stored. %s", this);
314   }

316   @Override
317   @Nullable
318   public synchronized Iterable<SkyKey> markDirty(boolean isChanged) {
319     assertKeepEdges();
320     if (isDone()) {
321       GroupedList<SkyKey> lastDirectDeps = GroupedList.create(directDeps);
322       buildingState = BuildingState.newDirtyState(isChanged, lastDirectDeps, value);
323       value = null;
324       directDeps = null;
325       return lastDirectDeps.toSet();
326     }
327     // The caller may be simultaneously trying to mark this node dirty and changed, and the dirty
328     // thread may have lost the race, but it is the caller's responsibility not to try to mark
329     // this node changed twice. The end result of racing markers must be a changed node, since one
330     // of the markers is trying to mark the node changed.
331     Preconditions.checkState(isChanged != isChanged(),
332         "Cannot mark node dirty twice or changed twice: %s", this);
333     Preconditions.checkState(value == null, "Value should have been reset already %s", this);
334     Preconditions.checkState(directDeps == null, "direct deps not already reset %s", this);
335     if (isChanged) {
336       // If the changed marker lost the race, we just need to mark changed in this method -- all
337       // other work was done by the dirty marker.
338       buildingState.markChanged();
339     }
340     return null;
341   }

343   @Override
344   public synchronized Set<SkyKey> markClean() {
345     this.value = buildingState.getLastBuildValue();
346     // This checks both the value and the direct deps, but since we're passing in the same value,
347     // the value check should be trivial.
348     Preconditions.checkState(buildingState.unchangedFromLastBuild(this.value),
349         "Direct deps must be the same as those found last build for node to be marked clean: %s",
350         this);
351     Preconditions.checkState(isDirty(), this);
352     Preconditions.checkState(!buildingState.isChanged(), "shouldn't be changed: %s", this);
353     return setStateFinishedAndReturnReverseDeps();
354   }

406   @Override
407   public String toString() {
408     return MoreObjects.toStringHelper(this)
409         .add("value", value)
410         .add("version", version)
411         .add("directDeps", directDeps == null ? null : GroupedList.create(directDeps))
412         .add("reverseDeps", REVERSE_DEPS_UTIL.toString(this))
413         .add("buildingState", buildingState).toString();
414   }

416   /**
417    * Do not use except in custom evaluator implementations! Added only temporarily.
418    *
419    * <p>Clones a InMemoryMutableNodeEntry iff it is a done node. Otherwise it fails.
420    */
421   public synchronized InMemoryNodeEntry cloneNodeEntry() {
422     // As this is temporary, for now let's limit to done nodes.
423     Preconditions.checkState(isDone(), "Only done nodes can be copied: %s", this);
424     InMemoryNodeEntry nodeEntry = new InMemoryNodeEntry();
425     nodeEntry.value = value;
426     nodeEntry.version = this.version;
427     REVERSE_DEPS_UTIL.addReverseDeps(nodeEntry, REVERSE_DEPS_UTIL.getReverseDeps(this));
428     nodeEntry.directDeps = directDeps;
429     nodeEntry.buildingState = null;
430     return nodeEntry;
431   }
