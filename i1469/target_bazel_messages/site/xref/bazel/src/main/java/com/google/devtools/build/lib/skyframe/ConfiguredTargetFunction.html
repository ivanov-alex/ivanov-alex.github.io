123   private final BuildViewProvider buildViewProvider;
124   private final RuleClassProvider ruleClassProvider;

126   ConfiguredTargetFunction(BuildViewProvider buildViewProvider,
127       RuleClassProvider ruleClassProvider) {
128     this.buildViewProvider = buildViewProvider;

132   @Override
133   public SkyValue compute(SkyKey key, Environment env) throws ConfiguredTargetFunctionException,
134       InterruptedException {
135     SkyframeBuildView view = buildViewProvider.getSkyframeBuildView();
136 
137     ConfiguredTargetKey configuredTargetKey = (ConfiguredTargetKey) key.argument();
138     LabelAndConfiguration lc = LabelAndConfiguration.of(
139         configuredTargetKey.getLabel(), configuredTargetKey.getConfiguration());
140 
141     BuildConfiguration configuration = lc.getConfiguration();
142 
143     PackageValue packageValue =
144         (PackageValue) env.getValue(PackageValue.key(lc.getLabel().getPackageIdentifier()));
145     if (packageValue == null) {
146       return null;
147     }
148 
149     Target target;
150     try {
151       target = packageValue.getPackage().getTarget(lc.getLabel().getName());
152     } catch (NoSuchTargetException e1) {
153       throw new ConfiguredTargetFunctionException(new NoSuchTargetException(lc.getLabel(),
154           "No such target"));
155     }
156     // TODO(bazel-team): This is problematic - we create the right key, but then end up with a value
157     // that doesn't match; we can even have the same value multiple times. However, I think it's
158     // only triggered in tests (i.e., in normal operation, the configuration passed in is already
159     // null).
160     if (target instanceof InputFile) {
161       // InputFileConfiguredTarget expects its configuration to be null since it's not used.
162       configuration = null;
163     } else if (target instanceof PackageGroup) {
164       // Same for PackageGroupConfiguredTarget.
165       configuration = null;
166     }
167     TargetAndConfiguration ctgValue =
168         new TargetAndConfiguration(target, configuration);
169 
170     SkyframeDependencyResolver resolver = view.createDependencyResolver(env);
171     if (resolver == null) {
172       return null;
173     }
174 
175     try {
176       // Get the configuration targets that trigger this rule's configurable attributes.
177       Set<ConfigMatchingProvider> configConditions =
178           getConfigConditions(ctgValue.getTarget(), env, resolver, ctgValue);
179       if (env.valuesMissing()) {
180         return null;
181       }
182 
183       ListMultimap<Attribute, ConfiguredTarget> depValueMap = computeDependencies(env, resolver,
184           ctgValue, null, configConditions, ruleClassProvider,
185           view.getHostConfiguration(configuration));
186       ConfiguredTargetValue ans = createConfiguredTarget(
187           view, env, target, configuration, depValueMap, configConditions);
188       return ans;
189     } catch (DependencyEvaluationException e) {
190       throw new ConfiguredTargetFunctionException(e.getRootCauseSkyKey(), e.getCause());
191     }
192   }

194   /**
195    * Computes the direct dependencies of a node in the configured target graph (a configured
196    * target or an aspect).
197    *
198    * <p>Returns null if Skyframe hasn't evaluated the required dependencies yet. In this case, the
199    * caller should also return null to Skyframe.
200    *
201    * @param env the Skyframe environment
202    * @param resolver The dependency resolver
203    * @param ctgValue The label and the configuration of the node
204    * @param aspectDefinition the aspect of the node (if null, the node is a configured target,
205    *     otherwise it's an asect)
206    * @param configConditions the configuration conditions for evaluating the attributes of the node
207    * @param ruleClassProvider rule class provider for determining the right configuration fragments
208    *   to apply to deps
209    * @param hostConfiguration the host configuration. There's a noticeable performance hit from
210    *     instantiating this on demand for every dependency that wants it, so it's best to compute
211    *     the host configuration as early as possible and pass this reference to all consumers
212    *     without involving Skyframe.
213    * @return an attribute -&gt; direct dependency multimap
214    */
215   @Nullable
216   static ListMultimap<Attribute, ConfiguredTarget> computeDependencies(
217       Environment env, SkyframeDependencyResolver resolver, TargetAndConfiguration ctgValue,
218       AspectDefinition aspectDefinition, Set<ConfigMatchingProvider> configConditions,
219       RuleClassProvider ruleClassProvider, BuildConfiguration hostConfiguration)
220       throws DependencyEvaluationException {
221 
222     // Create the map from attributes to list of (target, configuration) pairs.
223     ListMultimap<Attribute, Dependency> depValueNames;
224     try {
225       depValueNames = resolver.dependentNodeMap(ctgValue, hostConfiguration, aspectDefinition,
226           configConditions);
227     } catch (EvalException e) {
228       env.getListener().handle(Event.error(e.getLocation(), e.getMessage()));
229       throw new DependencyEvaluationException(new ConfiguredValueCreationException(e.print()));
230     }
231 
232     // Trim each dep's configuration so it only includes the fragments needed by its transitive
233     // closure (only dynamic configurations support this).
234     if (ctgValue.getConfiguration() != null
235         && ctgValue.getConfiguration().useDynamicConfigurations()) {
236       depValueNames = trimConfigurations(env, ctgValue, depValueNames, hostConfiguration,
237           ruleClassProvider);
238       if (depValueNames == null) {
239         return null;
240       }
241     }
242 
243     // Resolve configured target dependencies and handle errors.
244     Map<SkyKey, ConfiguredTarget> depValues =
245         resolveConfiguredTargetDependencies(env, depValueNames.values(), ctgValue.getTarget());
246     if (depValues == null) {
247       return null;
248     }
249 
250     // Resolve required aspects.
251     ListMultimap<SkyKey, Aspect> depAspects = resolveAspectDependencies(
252         env, depValues, depValueNames.values());
253     if (depAspects == null) {
254       return null;
255     }
256 
257     // Merge the dependent configured targets and aspects into a single map.
258     return mergeAspects(depValueNames, depValues, depAspects);
259   }

261   /**
262    * Helper class for {@link #trimConfigurations} - encapsulates a set of config fragments and
263    * a dynamic transition. This can be used to determine the exact build options needed to
264    * set a dynamic configuration.
265    */
266   private static final class FragmentsAndTransition {
267     // Treat this as immutable. The only reason this isn't an ImmutableSet is because it
268     // gets bound to a NestedSet.toSet() reference, which returns a Set interface.
269     final Set<Class<? extends BuildConfiguration.Fragment>> fragments;
270     final Attribute.Transition transition;
271     private final int hashCode;
272 
273     FragmentsAndTransition(Set<Class<? extends BuildConfiguration.Fragment>> fragments,
274         Attribute.Transition transition) {
275       this.fragments = fragments;
276       this.transition = transition;
277       hashCode = Objects.hash(this.fragments, this.transition);
278     }
279 
280     @Override
281     public boolean equals(Object o) {
282       if (o == this) {
283         return true;
284       } else if (o == null) {
285         return false;
286       } else {
287         FragmentsAndTransition other = (FragmentsAndTransition) o;
288         return other.transition.equals(transition) && other.fragments.equals(fragments);
289       }
290     }
291 
292     @Override
293     public int hashCode() {
294       return hashCode;
295     }
296   }

298   /**
299    * Creates a dynamic configuration for each dep that's custom-fitted specifically for that dep.
300    *
301    * <p>More specifically: given a set of {@link Dependency} instances holding dynamic config
302    * transition requests (e.g. {@link Dependency#hasStaticConfiguration()} == false}), returns
303    * equivalent dependencies containing dynamically created configurations that a) apply those
304    * transitions and b) only contain the fragments needed by the dep and everything in its
305    * transitive closure.
306    *
307    * <p>This method is heavily performance-optimized. Because it, in aggregate, reads over every
308    * edge in the configured target graph, small inefficiencies can have observable impact on
309    * build analysis time. Keep this in mind when making modifications and performance-test any
310    * changes you make.
311    */
312   @Nullable
313   static ListMultimap<Attribute, Dependency> trimConfigurations(Environment env,
314       TargetAndConfiguration ctgValue, ListMultimap<Attribute, Dependency> originalDeps,
315       BuildConfiguration hostConfiguration, RuleClassProvider ruleClassProvider)
316       throws DependencyEvaluationException {
317 
318     // Maps each Skyframe-evaluated BuildConfiguration to the dependencies that need that
319     // configuration. For cases where Skyframe isn't needed to get the configuration (e.g. when
320     // we just re-used the original rule's configuration), we should skip this outright.
321     Multimap<SkyKey, Map.Entry<Attribute, Dependency>> keysToEntries = ArrayListMultimap.create();
322 
323     // Stores the result of applying a dynamic transition to the current configuration using a
324     // particular subset of fragments. By caching this, we save from redundantly computing the
325     // same transition for every dependency edge that requests that transition. This can have
326     // real effect on analysis time for commonly triggered transitions.
327     Map<FragmentsAndTransition, BuildOptions> transitionsMap = new HashMap<>();
328 
329     // The fragments used by the current target's configuration.
330     Set<Class<? extends BuildConfiguration.Fragment>> ctgFragments =
331         ctgValue.getConfiguration().fragmentClasses();
332     BuildOptions ctgOptions = ctgValue.getConfiguration().getOptions();
333 
334     // Our output map.
335     ListMultimap<Attribute, Dependency> trimmedDeps = ArrayListMultimap.create();
336 
337     for (Map.Entry<Attribute, Dependency> depsEntry : originalDeps.entries()) {
338       Dependency dep = depsEntry.getValue();
339 
340       if (dep.hasStaticConfiguration()) {
341         // Certain targets (like output files) trivially pass their configurations to their deps.
342         // So no need to transform them in any way.
343         trimmedDeps.put(depsEntry.getKey(),
344             new Dependency(dep.getLabel(), dep.getConfiguration(), dep.getAspects()));
345         continue;
346       } else if (dep.getTransition() == Attribute.ConfigurationTransition.NULL) {
347         trimmedDeps.put(depsEntry.getKey(),
348             new Dependency(dep.getLabel(), (BuildConfiguration) null, dep.getAspects()));
349         continue;
350       }
351 
352       // Figure out the required fragments for this dep and its transitive closure.
353       SkyKey fragmentsKey = TransitiveTargetValue.key(dep.getLabel());
354       TransitiveTargetValue transitiveDepInfo = (TransitiveTargetValue) env.getValue(fragmentsKey);
355       if (transitiveDepInfo == null) {
356         // This should only be possible for tests. In actual runs, this was already called
357         // as a routine part of the loading phase.
358         // TODO(bazel-team): check this only occurs in a test context.
359         return null;
360       }
361       Set<Class<? extends BuildConfiguration.Fragment>> depFragments =
362           transitiveDepInfo.getTransitiveConfigFragments().toSet();
363 
364       boolean sameFragments = depFragments.equals(ctgFragments);
365       Attribute.Transition transition = dep.getTransition();
366 
367       if (sameFragments) {
368         if (transition == Attribute.ConfigurationTransition.NONE) {
369           // The dep uses the same exact configuration.
370           trimmedDeps.put(depsEntry.getKey(),
371               new Dependency(dep.getLabel(), ctgValue.getConfiguration(), dep.getAspects()));
372           continue;
373         } else if (transition == HostTransition.INSTANCE) {
374           // The current rule's host configuration can also be used for the dep. We short-circuit
375           // the standard transition logic for host transitions because these transitions are
376           // uniquely frequent. It's possible, e.g., for every node in the configured target graph
377           // to incur multiple host transitions. So we aggressively optimize to avoid hurting
378           // analysis time.
379           trimmedDeps.put(depsEntry.getKey(),
380               new Dependency(dep.getLabel(), hostConfiguration, dep.getAspects()));
381           continue;
382         }
383       }
384 
385       // Apply the transition or use the cached result if it was already applied.
386       FragmentsAndTransition transitionKey = new FragmentsAndTransition(depFragments, transition);
387       BuildOptions toOptions = transitionsMap.get(transitionKey);
388       if (toOptions == null) {
389         Verify.verify(transition == Attribute.ConfigurationTransition.NONE
390             || transition instanceof PatchTransition);
391         BuildOptions fromOptions = ctgOptions;
392         if (!sameFragments) {
393           // TODO(bazel-team): pre-compute getOptionsClasses in the constructor.
394           fromOptions = fromOptions.trim(BuildConfiguration.getOptionsClasses(
395               transitiveDepInfo.getTransitiveConfigFragments(), ruleClassProvider));
396         }
397         // TODO(bazel-team): safety-check that the below call never mutates fromOptions.
398         toOptions = transition == Attribute.ConfigurationTransition.NONE
399             ? fromOptions
400             : ((PatchTransition) transition).apply(fromOptions);
401         transitionsMap.put(transitionKey, toOptions);
402       }
403 
404       // If the transition doesn't change the configuration, trivially re-use the original
405       // configuration.
406       if (sameFragments && toOptions.equals(ctgOptions)) {
407         trimmedDeps.put(depsEntry.getKey(),
408             new Dependency(dep.getLabel(), ctgValue.getConfiguration(), dep.getAspects()));
409         continue;
410       }
411 
412       // If we get here, we have to get the configuration from Skyframe.
413       keysToEntries.put(BuildConfigurationValue.key(depFragments, toOptions), depsEntry);
414     }
415 
416     // Get all BuildConfigurations we need to get from Skyframe.
417     Map<SkyKey, ValueOrException<InvalidConfigurationException>> depConfigValues =
418         env.getValuesOrThrow(keysToEntries.keySet(), InvalidConfigurationException.class);
419     if (env.valuesMissing()) {
420       return null;
421     }
422 
423     // Now fill in the remaining unresolved deps with the now-resolved configurations.
424     try {
425       for (Map.Entry<SkyKey, ValueOrException<InvalidConfigurationException>> entry :
426           depConfigValues.entrySet()) {
427         SkyKey key = entry.getKey();
428         BuildConfigurationValue trimmedConfig = (BuildConfigurationValue) entry.getValue().get();
429         for (Map.Entry<Attribute, Dependency> info : keysToEntries.get(key)) {
430           Attribute attribute = info.getKey();
431           Dependency originalDep = info.getValue();
432           trimmedDeps.put(attribute,
433               new Dependency(originalDep.getLabel(), trimmedConfig.getConfiguration(),
434                   originalDep.getAspects()));
435         }
436       }
437     } catch (InvalidConfigurationException e) {
438       throw new DependencyEvaluationException(e);
439     }
440 
441     return trimmedDeps;
442   }

444   /**
445    * Merges the each direct dependency configured target with the aspects associated with it.
446    *
447    * <p>Note that the combination of a configured target and its associated aspects are not
448    * represented by a Skyframe node. This is because there can possibly be many different
449    * combinations of aspects for a particular configured target, so it would result in a
450    * combinatiorial explosion of Skyframe nodes.
451    */
452   private static ListMultimap<Attribute, ConfiguredTarget> mergeAspects(
453       ListMultimap<Attribute, Dependency> depValueNames,
454       Map<SkyKey, ConfiguredTarget> depConfiguredTargetMap,
455       ListMultimap<SkyKey, Aspect> depAspectMap) {
456     ListMultimap<Attribute, ConfiguredTarget> result = ArrayListMultimap.create();
457 
458     for (Map.Entry<Attribute, Dependency> entry : depValueNames.entries()) {
459       Dependency dep = entry.getValue();
460       SkyKey depKey = TO_KEYS.apply(dep);
461       ConfiguredTarget depConfiguredTarget = depConfiguredTargetMap.get(depKey);
462       result.put(entry.getKey(),
463           RuleConfiguredTarget.mergeAspects(depConfiguredTarget, depAspectMap.get(depKey)));
464     }
465 
466     return result;
467   }

469   /**
470    * Given a list of {@link Dependency} objects, returns a multimap from the {@link SkyKey} of the
471    * dependency to the {@link Aspect} instances that should be merged into it.
472    *
473    * <p>Returns null if the required aspects are not computed yet.
474    */
475   @Nullable
476   private static ListMultimap<SkyKey, Aspect> resolveAspectDependencies(Environment env,
477       Map<SkyKey, ConfiguredTarget> configuredTargetMap, Iterable<Dependency> deps)
478       throws DependencyEvaluationException {
479     ListMultimap<SkyKey, Aspect> result = ArrayListMultimap.create();
480     Set<SkyKey> aspectKeys = new HashSet<>();
481     for (Dependency dep : deps) {
482       for (Class<? extends ConfiguredAspectFactory> depAspect : dep.getAspects()) {
483         aspectKeys.add(AspectValue.key(dep.getLabel(), dep.getConfiguration(), depAspect));
484       }
485     }
486 
487     Map<SkyKey, ValueOrException3<
488         AspectCreationException, NoSuchThingException, ConfiguredValueCreationException>>
489         depAspects = env.getValuesOrThrow(aspectKeys, AspectCreationException.class,
490             NoSuchThingException.class, ConfiguredValueCreationException.class);
491 
492     for (Dependency dep : deps) {
493       SkyKey depKey = TO_KEYS.apply(dep);
494       // If the same target was declared in different attributes of rule, we should not process it
495       // twice.
496       if (result.containsKey(depKey)) {
497         continue;
498       }
499       ConfiguredTarget depConfiguredTarget = configuredTargetMap.get(depKey);
500       for (Class<? extends ConfiguredAspectFactory> depAspect : dep.getAspects()) {
501         if (!aspectMatchesConfiguredTarget(depConfiguredTarget, depAspect)) {
502           continue;
503         }
504 
505         SkyKey aspectKey = AspectValue.key(dep.getLabel(), dep.getConfiguration(), depAspect);
506         AspectValue aspectValue = null;
507         try {
508           aspectValue = (AspectValue) depAspects.get(aspectKey).get();
509         } catch (ConfiguredValueCreationException e) {
510           // The configured target should have been created in resolveConfiguredTargetDependencies()
511           throw new IllegalStateException(e);
512         } catch (NoSuchThingException | AspectCreationException e) {
513           AspectFactory<?, ?, ?> depAspectFactory = AspectFactory.Util.create(depAspect);
514           throw new DependencyEvaluationException(new ConfiguredValueCreationException(
515               String.format("Evaluation of aspect %s on %s failed: %s",
516                   depAspectFactory.getDefinition().getName(), dep.getLabel(), e.toString())));
517         }
518 
519         if (aspectValue == null) {
520           // Dependent aspect has either not been computed yet or is in error.
521           return null;
522         }
523         result.put(depKey, aspectValue.getAspect());
524       }
525     }
526 
527     return result;
528   }

530   private static boolean aspectMatchesConfiguredTarget(ConfiguredTarget dep,
531       Class<? extends ConfiguredAspectFactory> aspectFactory) {
532     AspectDefinition aspectDefinition = AspectFactory.Util.create(aspectFactory).getDefinition();
533     for (Class<?> provider : aspectDefinition.getRequiredProviders()) {
534       if (dep.getProvider(provider.asSubclass(TransitiveInfoProvider.class)) == null) {
535         return false;
536       }
537     }
538 
539     return true;
540   }

542   /**
543    * Returns the set of {@link ConfigMatchingProvider}s that key the configurable attributes
544    * used by this rule.
545    *
546    * <p>>If the configured targets supplying those providers aren't yet resolved by the
547    * dependency resolver, returns null.
548    */
549   @Nullable
550   static Set<ConfigMatchingProvider> getConfigConditions(Target target, Environment env,
551       SkyframeDependencyResolver resolver, TargetAndConfiguration ctgValue)
552       throws DependencyEvaluationException {
553     if (!(target instanceof Rule)) {
554       return ImmutableSet.of();
555     }
556 
557     ImmutableSet.Builder<ConfigMatchingProvider> configConditions = ImmutableSet.builder();
558 
559     // Collect the labels of the configured targets we need to resolve.
560     ListMultimap<Attribute, LabelAndConfiguration> configLabelMap = ArrayListMultimap.create();
561     RawAttributeMapper attributeMap = RawAttributeMapper.of(((Rule) target));
562     for (Attribute a : ((Rule) target).getAttributes()) {
563       for (Label configLabel : attributeMap.getConfigurabilityKeys(a.getName(), a.getType())) {
564         if (!Type.Selector.isReservedLabel(configLabel)) {
565           configLabelMap.put(a, LabelAndConfiguration.of(
566               configLabel, ctgValue.getConfiguration()));
567         }
568       }
569     }
570     if (configLabelMap.isEmpty()) {
571       return ImmutableSet.of();
572     }
573 
574     // Collect the corresponding Skyframe configured target values. Abort early if they haven't
575     // been computed yet.
576     Collection<Dependency> configValueNames =
577         resolver.resolveRuleLabels(ctgValue, null, configLabelMap);
578 
579     // No need to get new configs from Skyframe - config_setting rules always use the current
580     // target's config.
581     // TODO(bazel-team): remove the need for this special transformation. We can probably do this by
582     // simply passing this through trimConfigurations.
583     BuildConfiguration targetConfig = ctgValue.getConfiguration();
584     if (targetConfig != null && targetConfig.useDynamicConfigurations()) {
585       ImmutableList.Builder<Dependency> staticConfigs = ImmutableList.builder();
586       for (Dependency dep : configValueNames) {
587         staticConfigs.add(new Dependency(dep.getLabel(), targetConfig, dep.getAspects()));
588       }
589       configValueNames = staticConfigs.build();
590     }
591 
592     Map<SkyKey, ConfiguredTarget> configValues =
593         resolveConfiguredTargetDependencies(env, configValueNames, target);
594     if (configValues == null) {
595       return null;
596     }
597 
598     // Get the configured targets as ConfigMatchingProvider interfaces.
599     for (Dependency entry : configValueNames) {
600       ConfiguredTarget value = configValues.get(TO_KEYS.apply(entry));
601       // The code above guarantees that value is non-null here.
602       ConfigMatchingProvider provider = value.getProvider(ConfigMatchingProvider.class);
603       if (provider != null) {
604         configConditions.add(provider);
605       } else {
606         // Not a valid provider for configuration conditions.
607         String message =
608             entry.getLabel() + " is not a valid configuration key for " + target.getLabel();
609         env.getListener().handle(Event.error(TargetUtils.getLocationMaybe(target), message));
610         throw new DependencyEvaluationException(new ConfiguredValueCreationException(message));
611       }
612     }
613 
614     return configConditions.build();
615   }

617   /***
618    * Resolves the targets referenced in depValueNames and returns their ConfiguredTarget
619    * instances.
620    *
621    * <p>Returns null if not all instances are available yet.
622    *
623    */
624   @Nullable
625   private static Map<SkyKey, ConfiguredTarget> resolveConfiguredTargetDependencies(
626       Environment env, Collection<Dependency> deps, Target target)
627       throws DependencyEvaluationException {
628     boolean ok = !env.valuesMissing();
629     String message = null;
630     Iterable<SkyKey> depKeys = Iterables.transform(deps, TO_KEYS);
631     Map<SkyKey, ValueOrException2<NoSuchTargetException,
632         NoSuchPackageException>> depValuesOrExceptions = env.getValuesOrThrow(depKeys,
633             NoSuchTargetException.class, NoSuchPackageException.class);
634     Map<SkyKey, ConfiguredTarget> depValues = new HashMap<>(depValuesOrExceptions.size());
635     SkyKey childKey = null;
636     NoSuchThingException transitiveChildException = null;
637     for (Map.Entry<SkyKey, ValueOrException2<NoSuchTargetException, NoSuchPackageException>> entry
638         : depValuesOrExceptions.entrySet()) {
639       ConfiguredTargetKey depKey = (ConfiguredTargetKey) entry.getKey().argument();
640       LabelAndConfiguration depLabelAndConfiguration = LabelAndConfiguration.of(
641           depKey.getLabel(), depKey.getConfiguration());
642       Label depLabel = depLabelAndConfiguration.getLabel();
643       ConfiguredTargetValue depValue = null;
644       NoSuchThingException directChildException = null;
645       try {
646         depValue = (ConfiguredTargetValue) entry.getValue().get();
647       } catch (NoSuchTargetException e) {
648         if (depLabel.equals(e.getLabel())) {
649           directChildException = e;
650         } else {
651           childKey = entry.getKey();
652           transitiveChildException = e;
653         }
654       } catch (NoSuchPackageException e) {
655         if (depLabel.getPackageIdentifier().equals(e.getPackageId())) {
656           directChildException = e;
657         } else {
658           childKey = entry.getKey();
659           transitiveChildException = e;
660         }
661       }
662       // If an exception wasn't caused by a direct child target value, we'll treat it the same
663       // as any other missing dep by setting ok = false below, and returning null at the end.
664       if (directChildException != null) {
665         // Only update messages for missing targets we depend on directly.
666         message = TargetUtils.formatMissingEdge(target, depLabel, directChildException);
667         env.getListener().handle(Event.error(TargetUtils.getLocationMaybe(target), message));
668       }
669 
670       if (depValue == null) {
671         ok = false;
672       } else {
673         depValues.put(entry.getKey(), depValue.getConfiguredTarget());
674       }
675     }
676     if (message != null) {
677       throw new DependencyEvaluationException(new NoSuchTargetException(message));
678     }
679     if (childKey != null) {
680       throw new DependencyEvaluationException(childKey, transitiveChildException);
681     }
682     if (!ok) {
683       return null;
684     } else {
685       return depValues;
686     }
687   }

695   @Nullable
696   private ConfiguredTargetValue createConfiguredTarget(SkyframeBuildView view,
697       Environment env, Target target, BuildConfiguration configuration,
698       ListMultimap<Attribute, ConfiguredTarget> depValueMap,
699       Set<ConfigMatchingProvider> configConditions)
700       throws ConfiguredTargetFunctionException, InterruptedException {
701     StoredEventHandler events = new StoredEventHandler();
702     BuildConfiguration ownerConfig = (configuration == null)
703         ? null : configuration.getArtifactOwnerConfiguration();
704     CachingAnalysisEnvironment analysisEnvironment = view.createAnalysisEnvironment(
705         new ConfiguredTargetKey(target.getLabel(), ownerConfig), false,
706         events, env, configuration);
707     if (env.valuesMissing()) {
708       return null;
709     }
710 
711     ConfiguredTarget configuredTarget = view.createConfiguredTarget(target, configuration,
712         analysisEnvironment, depValueMap, configConditions);
713 
714     events.replayOn(env.getListener());
715     if (events.hasErrors()) {
716       analysisEnvironment.disable(target);
717       throw new ConfiguredTargetFunctionException(new ConfiguredValueCreationException(
718               "Analysis of target '" + target.getLabel() + "' failed; build aborted"));
719     }
720     Preconditions.checkState(!analysisEnvironment.hasErrors(),
721         "Analysis environment hasError() but no errors reported");
722     if (env.valuesMissing()) {
723       return null;
724     }
725 
726     analysisEnvironment.disable(target);
727     Preconditions.checkNotNull(configuredTarget, target);
728 
729     try {
730       return new ConfiguredTargetValue(configuredTarget,
731           filterSharedActionsAndThrowIfConflict(analysisEnvironment.getRegisteredActions()));
732     } catch (ActionConflictException e) {
733       throw new ConfiguredTargetFunctionException(e);
734     }
735   }
736 
737   static Map<Artifact, Action> filterSharedActionsAndThrowIfConflict(Iterable<Action> actions)
738       throws ActionConflictException {
739     Map<Artifact, Action> generatingActions = new HashMap<>();
740     for (Action action : actions) {
741       for (Artifact artifact : action.getOutputs()) {
742         Action previousAction = generatingActions.put(artifact, action);
743         if (previousAction != null && previousAction != action
744             && !Actions.canBeShared(previousAction, action)) {
745           throw new ActionConflictException(artifact, previousAction, action);
746         }
747       }
748     }
749     return generatingActions;
750   }
