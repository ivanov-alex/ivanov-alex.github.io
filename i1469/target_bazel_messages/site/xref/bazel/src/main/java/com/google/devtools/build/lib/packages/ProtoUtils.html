50  /**
51   * Shared code used in proto buffer output for rules and rule classes.
52   */
53  public class ProtoUtils {
54    /**
55     * This map contains all attribute types that are recognized by the protocol
56     * output formatter.
57     *
58     * <p>If you modify this map, please ensure that {@link #getTypeFromDiscriminator} can still
59     * resolve a {@link Discriminator} value to exactly one {@link Type} (using an optional nodep
60     * hint, as described below).
61     */
62    static final ImmutableMap<Type<?>, Discriminator> TYPE_MAP =
63        new ImmutableMap.Builder<Type<?>, Discriminator>()
64            .put(INTEGER, Discriminator.INTEGER)
65            .put(DISTRIBUTIONS, Discriminator.DISTRIBUTION_SET)
66            .put(LABEL, Discriminator.LABEL)
67            // NODEP_LABEL attributes are not really strings. This is implemented
68            // this way for the sake of backward compatibility.
69            .put(NODEP_LABEL, Discriminator.STRING)
70            .put(LABEL_LIST, Discriminator.LABEL_LIST)
71            .put(NODEP_LABEL_LIST, Discriminator.STRING_LIST)
72            .put(STRING, Discriminator.STRING)
73            .put(STRING_LIST, Discriminator.STRING_LIST)
74            .put(OUTPUT, Discriminator.OUTPUT)
75            .put(OUTPUT_LIST, Discriminator.OUTPUT_LIST)
76            .put(LICENSE, Discriminator.LICENSE)
77            .put(STRING_DICT, Discriminator.STRING_DICT)
78            .put(FILESET_ENTRY_LIST, Discriminator.FILESET_ENTRY_LIST)
79            .put(LABEL_DICT_UNARY, Discriminator.LABEL_DICT_UNARY)
80            .put(LABEL_LIST_DICT, Discriminator.LABEL_LIST_DICT)
81            .put(STRING_LIST_DICT, Discriminator.STRING_LIST_DICT)
82            .put(BOOLEAN, Discriminator.BOOLEAN)
83            .put(TRISTATE, Discriminator.TRISTATE)
84            .put(INTEGER_LIST, Discriminator.INTEGER_LIST)
85            .put(STRING_DICT_UNARY, Discriminator.STRING_DICT_UNARY)
86            .build();
87  
88    static final ImmutableSet<Type<?>> NODEP_TYPES =
89        ImmutableSet.of(NODEP_LABEL, NODEP_LABEL_LIST);
90  
91    static final ImmutableSetMultimap<Discriminator, Type<?>> INVERSE_TYPE_MAP =
92        TYPE_MAP.asMultimap().inverse();
93  
94    /**
95     * Returns the appropriate Attribute.Discriminator value from an internal attribute type.
96     */
97    public static Discriminator getDiscriminatorFromType(Type<?> type) {
98      Preconditions.checkArgument(TYPE_MAP.containsKey(type), type);
99      return TYPE_MAP.get(type);
100   }
101 
102   /**
103    * Returns the appropriate set of internal attribute types from an Attribute.Discriminator value.
104    */
105   public static ImmutableSet<Type<?>> getTypesFromDiscriminator(Discriminator discriminator) {
106     Preconditions.checkArgument(INVERSE_TYPE_MAP.containsKey(discriminator), discriminator);
107     return INVERSE_TYPE_MAP.get(discriminator);
108   }
109 
110   /**
111    * Returns the appropriate internal attribute type from an Attribute.Discriminator value, given
112    * an optional nodeps hint.
113    */
114   public static Type<?> getTypeFromDiscriminator(Discriminator discriminator,
115       Optional<Boolean> nodeps, String ruleClassName, String attrName) {
116     Preconditions.checkArgument(INVERSE_TYPE_MAP.containsKey(discriminator));
117     ImmutableSet<Type<?>> possibleTypes = ProtoUtils.getTypesFromDiscriminator(discriminator);
118     Type<?> preciseType;
119     if (possibleTypes.size() == 1) {
120       preciseType = Iterables.getOnlyElement(possibleTypes);
121     } else {
122       // If there is more than one possible type associated with the discriminator, then the
123       // discriminator must be either Discriminator.STRING or Discriminator.STRING_LIST.
124       //
125       // If it is Discriminator.STRING, then its possible Type<?>s are {NODEP_LABEL, STRING}. The
126       // nodeps hint must be present in order to distinguish between them. If nodeps is true,
127       // then the Type<?> must be NODEP_LABEL, and if false, it must be STRING.
128       //
129       // A similar relation holds for the Discriminator value STRING_LIST, and its possible
130       // Type<?>s {NODEP_LABEL_LIST, STRING_LIST}.
131 
132       Preconditions.checkArgument(nodeps.isPresent(),
133           "Nodeps hint is required when discriminator is associated with more than one type."
134               + " Discriminator: \"%s\", Rule class: \"%s\", Attr: \"%s\"", discriminator,
135           ruleClassName, attrName);
136       if (nodeps.get()) {
137         Set<Type<?>> nodepType = Sets.filter(possibleTypes, Predicates.in(NODEP_TYPES));
138         Preconditions.checkState(nodepType.size() == 1,
139             "There should be exactly one NODEP type associated with discriminator \"%s\""
140                 + ", but found these: %s. Rule class: \"%s\", Attr: \"%s\"", discriminator,
141             nodepType, ruleClassName, attrName);
142         preciseType = Iterables.getOnlyElement(nodepType);
143       } else {
144         Set<Type<?>> notNodepType =
145             Sets.filter(possibleTypes, Predicates.not(Predicates.in(NODEP_TYPES)));
146         Preconditions.checkState(notNodepType.size() == 1,
147             "There should be exactly one non-NODEP type associated with discriminator \"%s\""
148                 + ", but found these: %s. Rule class: \"%s\", Attr: \"%s\"", discriminator,
149             notNodepType, ruleClassName, attrName);
150         preciseType = Iterables.getOnlyElement(notNodepType);
151       }
152     }
153     return preciseType;
154   }
