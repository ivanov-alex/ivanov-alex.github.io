44  /**
45   * Tests for {@link DependencyResolver}.
46   *
47   * <p>These use custom rules so that all usual and unusual cases related to aspect processing can
48   * be tested.
49   *
50   * <p>It would be nicer is we didn't have a Skyframe executor, if we didn't have that, we'd need a
51   * way to create a configuration, a package manager and a whole lot of other things, so it's just
52   * easier this way.
53   */
54  @RunWith(JUnit4.class)
55  public class DependencyResolverTest extends AnalysisTestCase {
56    private DependencyResolver dependencyResolver;

58    @Override
59    @Before
60    public void setUp() throws Exception {
61      super.setUp();
62  
63      dependencyResolver = new DependencyResolver() {
64        @Override
65        protected void invalidVisibilityReferenceHook(TargetAndConfiguration node, Label label) {
66          throw new IllegalStateException();
67        }
68  
69        @Override
70        protected void invalidPackageGroupReferenceHook(TargetAndConfiguration node, Label label) {
71          throw new IllegalStateException();
72        }
73  
74        @Nullable
75        @Override
76        protected Target getTarget(Label label) throws NoSuchThingException {
77          try {
78            return packageManager.getTarget(reporter, label);
79          } catch (InterruptedException e) {
80            throw new IllegalStateException(e);

82        }
83      };
84    }

92    private void pkg(String name, String... contents) throws Exception {
93      scratch.file("" + name + "/BUILD", contents);

96    @SafeVarargs
97    private final void setRules(RuleDefinition... rules) throws Exception {
98      ConfiguredRuleClassProvider.Builder builder =
99          new ConfiguredRuleClassProvider.Builder();
100     TestRuleClassProvider.addStandardRules(builder);
101     for (RuleDefinition rule : rules) {
102       builder.addRuleDefinition(rule);
103     }
104 
105     useRuleClassProvider(builder.build());
106     update();
107   }

109   private ListMultimap<Attribute, Dependency> dependentNodeMap(
110       String targetName, Class<? extends ConfiguredAspectFactory> aspect) throws Exception {
111     AspectDefinition aspectDefinition = aspect == null
112         ? null
113         : AspectFactory.Util.create(aspect).getDefinition();
114     Target target = packageManager.getTarget(reporter, Label.parseAbsolute(targetName));
115     return dependencyResolver.dependentNodeMap(
116         new TargetAndConfiguration(target, getTargetConfiguration()),
117         getHostConfiguration(),
118         aspectDefinition,
119         ImmutableSet.<ConfigMatchingProvider>of());
120   }
121 
122   @SafeVarargs
123   private final void assertDep(
124       ListMultimap<Attribute, Dependency> dependentNodeMap,
125       String attrName,
126       String dep,
127       Class<? extends AspectFactory<?, ?, ?>>... aspects) {
128     Attribute attr = null;
129     for (Attribute candidate : dependentNodeMap.keySet()) {
130       if (candidate.getName().equals(attrName)) {
131         attr = candidate;
132         break;
133       }
134     }
135 
136     assertNotNull("Attribute '" + attrName + "' not found", attr);
137     Dependency dependency = null;
138     for (Dependency candidate : dependentNodeMap.get(attr)) {
139       if (candidate.getLabel().toString().equals(dep)) {
140         dependency = candidate;
141         break;
142       }
143     }
144 
145     assertNotNull("Dependency '" + dep + "' on attribute '" + attrName + "' not found", dependency);
146     assertThat(dependency.getAspects()).containsExactly((Object[]) aspects);
147   }

149   @Test
150   public void hasAspectsRequiredByRule() throws Exception {
151     setRules(new AspectRequiringRule(), new TestAspects.BaseRule());
152     pkg("a",
153         "aspect(name='a', foo=[':b'])",
154         "aspect(name='b', foo=[])");
155     ListMultimap<Attribute, Dependency> map = dependentNodeMap("//a:a", null);
156     assertDep(map, "foo", "//a:b", TestAspects.SimpleAspect.class);
157   }

191   @Test
192   public void equalsOnDependencyPassesEqualsTester() throws Exception {
193     update();
194 
195     Label a = Label.parseAbsolute("//a");
196     Label aExplicit = Label.parseAbsolute("//a:a");
197     Label b = Label.parseAbsolute("//b");
198 
199     BuildConfiguration host = getHostConfiguration();
200     BuildConfiguration target = getTargetConfiguration();
201 
202     ImmutableSet<Class<? extends ConfiguredAspectFactory>> twoAspects =
203         ImmutableSet.<Class<? extends ConfiguredAspectFactory>>of(
204             TestAspects.SimpleAspect.class, TestAspects.AttributeAspect.class);
205     ImmutableSet<Class<? extends ConfiguredAspectFactory>> inverseAspects =
206         ImmutableSet.<Class<? extends ConfiguredAspectFactory>>of(
207             TestAspects.AttributeAspect.class, TestAspects.SimpleAspect.class);
208     ImmutableSet<Class<? extends ConfiguredAspectFactory>> differentAspects =
209         ImmutableSet.<Class<? extends ConfiguredAspectFactory>>of(
210             TestAspects.AttributeAspect.class, TestAspects.ErrorAspect.class);
211 
212     new EqualsTester()
213         .addEqualityGroup(
214             // base set: //a, host configuration, normal aspect set
215             new Dependency(a, host, twoAspects),
216             new Dependency(aExplicit, host, twoAspects),
217             new Dependency(a, host, inverseAspects),
218             new Dependency(aExplicit, host, inverseAspects))
219         .addEqualityGroup(
220             // base set but with label //b
221             new Dependency(b, host, twoAspects),
222             new Dependency(b, host, inverseAspects))
223         .addEqualityGroup(
224             // base set but with target configuration
225             new Dependency(a, target, twoAspects),
226             new Dependency(aExplicit, target, twoAspects),
227             new Dependency(a, target, inverseAspects),
228             new Dependency(aExplicit, target, inverseAspects))
229         .addEqualityGroup(
230             // base set but with null configuration
231             new Dependency(a, (BuildConfiguration) null, twoAspects),
232             new Dependency(aExplicit, (BuildConfiguration) null, twoAspects),
233             new Dependency(a, (BuildConfiguration) null, inverseAspects),
234             new Dependency(aExplicit, (BuildConfiguration) null, inverseAspects))
235         .addEqualityGroup(
236             // base set but with different aspects
237             new Dependency(a, host, differentAspects),
238             new Dependency(aExplicit, host, differentAspects))
239         .addEqualityGroup(
240             // base set but with label //b and target configuration
241             new Dependency(b, target, twoAspects),
242             new Dependency(b, target, inverseAspects))
243         .addEqualityGroup(
244             // base set but with label //b and null configuration
245             new Dependency(b, (BuildConfiguration) null, twoAspects),
246             new Dependency(b, (BuildConfiguration) null, inverseAspects))
247         .addEqualityGroup(
248             // base set but with label //b and different aspects
249             new Dependency(b, host, differentAspects))
250         .addEqualityGroup(
251             // base set but with target configuration and different aspects
252             new Dependency(a, target, differentAspects),
253             new Dependency(aExplicit, target, differentAspects))
254         .addEqualityGroup(
255             // base set but with null configuration and different aspects
256             new Dependency(a, (BuildConfiguration) null, differentAspects),
257             new Dependency(aExplicit, (BuildConfiguration) null, differentAspects))
258         .addEqualityGroup(
259             // inverse of base set: //b, target configuration, different aspects
260             new Dependency(b, target, differentAspects))
261         .addEqualityGroup(
262             // inverse of base set: //b, null configuration, different aspects
263             new Dependency(b, (BuildConfiguration) null, differentAspects))
264         .testEquals();
265   }
