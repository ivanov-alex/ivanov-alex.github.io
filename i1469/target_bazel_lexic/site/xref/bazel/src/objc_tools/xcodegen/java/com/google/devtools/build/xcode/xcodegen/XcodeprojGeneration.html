99    /**
100    * Determines the relative path to the workspace root from the path of the project.pbxproj output
101    * file. An absolute path is preferred if available.
102    */
103   static Path relativeWorkspaceRoot(Path pbxproj) {
104     int levelsToExecRoot = pbxproj.getParent().getParent().getNameCount();
105     return pbxproj.getFileSystem().getPath(Joiner
106         .on('/')
107         .join(Collections.nCopies(levelsToExecRoot, "..")));
108   }

110   /**
111    * Writes a project to an {@code OutputStream} in the correct encoding.
112    */
113   public static void write(OutputStream out, PBXProject project) throws IOException {
114     XcodeprojSerializer ser = new XcodeprojSerializer(
115         new GidGenerator(ImmutableSet.<String>of()), project);
116     Writer outWriter = new OutputStreamWriter(out, StandardCharsets.UTF_8);
117     // toXMLPropertyList includes an XML encoding specification (UTF-8), which we specify above.
118     // Standard Xcodeproj files use the toASCIIPropertyList format, but Xcode will rewrite
119     // XML-encoded project files automatically when first opening them. We use XML to prevent
120     // encoding issues, since toASCIIPropertyList does not include the UTF-8 encoding comment, and
121     // Xcode by default apparently uses MacRoman.
122     // This encoding concern is probably why Buck also generates XML project files as well.
123     outWriter.write(ser.toPlist().toXMLPropertyList());
124     outWriter.flush();
125   }

142   /**
143    * Detects the product type of the given target based on multiple fields in {@code targetControl}.
144    * {@code productType} is set as a field on {@code PBXNativeTarget} objects in Xcode project
145    * files, and we support three values: {@link ProductType#APPLICATION},
146    * {@link ProductType#STATIC_LIBRARY}, and {@link ProductType#BUNDLE}. The product type is not
147    * only what xcodegen sets the {@code productType} field to - it also dictates what can be built
148    * with this target (e.g. a library cannot be built with resources), what build phase it should be
149    * added to of its dependers, and the name and shape of its build output.
150    */
151   public static ProductType productType(TargetControl targetControl) {
152     if (targetControl.hasProductType()) {
153       for (ProductType supportedType : SUPPORTED_PRODUCT_TYPES) {
154         if (targetControl.getProductType().equals(supportedType.identifier)) {
155           return supportedType;
156         }
157       }
158       throw new IllegalArgumentException(
159           "Unsupported product type: " + targetControl.getProductType());
160     }
161 
162     return targetControl.hasInfoplist() ? ProductType.APPLICATION : ProductType.STATIC_LIBRARY;
163   }
164 
165   private static String productName(TargetControl targetControl) {
166     if (Equaling.of(ProductType.STATIC_LIBRARY, productType(targetControl))) {
167       // The product names for static libraries must be unique since the final
168       // binary is linked with "clang -l${LIBRARY_PRODUCT_NAME}" for each static library.
169       // Unlike other product types, a full application may have dozens of static libraries,
170       // so rather than just use the target name, we use the full label to generate the product
171       // name.
172       return targetControl.getLabel();
173     } else {
174       return targetControl.getName();
175     }
176   }

178   /**
179    * Returns the file reference corresponding to the {@code productReference} of the given target.
180    * The {@code productReference} is the build output of a target, and its name and file type
181    * (stored in the {@link FileReference}) change based on the product type.
182    */
183   private static FileReference productReference(TargetControl targetControl) {
184     ProductType type = productType(targetControl);
185     String productName = productName(targetControl);
186 
187     switch (type) {
188       case APPLICATION:
189         return FileReference.of(String.format("%s.app", productName), SourceTree.BUILT_PRODUCTS_DIR)
190             .withExplicitFileType(FILE_TYPE_WRAPPER_APPLICATION);
191       case STATIC_LIBRARY:
192         return FileReference.of(
193             String.format("lib%s.a", productName), SourceTree.BUILT_PRODUCTS_DIR)
194                 .withExplicitFileType(FILE_TYPE_ARCHIVE_LIBRARY);
195       case BUNDLE:
196         return FileReference.of(
197             String.format("%s.bundle", productName), SourceTree.BUILT_PRODUCTS_DIR)
198                 .withExplicitFileType(FILE_TYPE_WRAPPER_BUNDLE);
199       case UNIT_TEST:
200         return FileReference.of(
201             String.format("%s.xctest", productName), SourceTree.BUILT_PRODUCTS_DIR)
202                 .withExplicitFileType(FILE_TYPE_WRAPPER_BUNDLE);
203       case APP_EXTENSION:
204         return FileReference.of(
205             String.format("%s.appex", productName), SourceTree.BUILT_PRODUCTS_DIR)
206                 .withExplicitFileType(FILE_TYPE_APP_EXTENSION);
207       case FRAMEWORK:
208         return FileReference.of(
209             String.format("%s.framwork", productName), SourceTree.BUILT_PRODUCTS_DIR)
210                 .withExplicitFileType(FILE_TYPE_FRAMEWORK);
211 
212       default:
213         throw new IllegalArgumentException("unknown: " + type);
214     }
215   }

218     final TargetControl control;

226     TargetInfo(TargetControl control,
227         PBXNativeTarget nativeTarget,
228         PBXFrameworksBuildPhase frameworksPhase,
229         PBXResourcesBuildPhase resourcesPhase,
230         PBXBuildFile productBuildFile,
231         PBXTargetDependency targetDependency,

233       this.control = control;
234       this.nativeTarget = nativeTarget;
235       this.frameworksPhase = frameworksPhase;
236       this.resourcesPhase = resourcesPhase;
237       this.productBuildFile = productBuildFile;
238       this.targetDependency = targetDependency;
239       this.buildConfig = buildConfig;

242     /**
243      * Returns the path to the built, statically-linked binary for this target. The path contains
244      * build-setting variables and may be used in a build setting such as {@code TEST_HOST}.
245      *
246      * <p>One example return value is {@code $(BUILT_PRODUCTS_DIR)/Foo.app/Foo}.
247      */
248     String staticallyLinkedBinary() {
249       ProductType type = productType(control);
250       Preconditions.checkArgument(
251           Containing.item(PRODUCT_TYPES_THAT_HAVE_A_BINARY, type),
252           "This product type (%s) is not known to have a binary.", type);
253       FileReference productReference = productReference(control);
254       return String.format("$(%s)/%s/%s",
255           productReference.sourceTree().name(),
256           productReference.path().or(productReference.name()),
257           control.getName());
258     }

260     /**
261      * Adds the given dependency to the list of dependencies, the
262      * appropriate build phase if applicable, and the appropriate build setting values if
263      * applicable, of this target.
264      */
265     void addDependencyInfo(
266         DependencyControl dependencyControl, Map<String, TargetInfo> targetInfoByLabel) {
267       TargetInfo dependencyInfo =
268           Mapping.of(targetInfoByLabel, dependencyControl.getTargetLabel()).get();
269       if (dependencyControl.getTestHost()) {
270         buildConfig.put("TEST_HOST", dependencyInfo.staticallyLinkedBinary());
271         buildConfig.put("BUNDLE_LOADER", dependencyInfo.staticallyLinkedBinary());
272       } else if (productType(dependencyInfo.control) == ProductType.BUNDLE) {
273         resourcesPhase.getFiles().add(dependencyInfo.productBuildFile);
274       } else if (productType(dependencyInfo.control) == ProductType.APP_EXTENSION) {
275         PBXCopyFilesBuildPhase copyFilesPhase = new PBXCopyFilesBuildPhase(
276             PBXCopyFilesBuildPhase.Destination.PLUGINS, /*path=*/"");
277         copyFilesPhase.getFiles().add(dependencyInfo.productBuildFile);
278         nativeTarget.getBuildPhases().add(copyFilesPhase);
279       } else {
280         frameworksPhase.getFiles().add(dependencyInfo.productBuildFile);
281       }
282       nativeTarget.getDependencies().add(dependencyInfo.targetDependency);
283     }
284   }

300   /**
301    * Returns the final header search paths to be placed in a build configuration.
302    */
303   private static NSArray headerSearchPaths(Iterable<String> paths) {
304     ImmutableList.Builder<String> result = new ImmutableList.Builder<>();
305     for (String path : paths) {
306       // TODO(bazel-team): Remove this hack once the released version of Bazel is prepending
307       // "$(WORKSPACE_ROOT)/" to every "source rooted" path.
308       if (!path.startsWith("$")) {
309         path = "$(WORKSPACE_ROOT)/" + path;
310       }
311       result.add(path);
312     }
313     return (NSArray) NSObject.wrap(result.build());
314   }

316   /**
317    * Returns the {@code FRAMEWORK_SEARCH_PATHS} array for a target's build config given the list of
318    * {@code .framework} directory paths.
319    */
320   private static NSArray frameworkSearchPaths(Iterable<String> frameworks) {
321     ImmutableSet.Builder<NSString> result = new ImmutableSet.Builder<>();
322     for (String framework : frameworks) {
323       result.add(new NSString("$(WORKSPACE_ROOT)/" + Paths.get(framework).getParent()));
324     }
325     // This is needed by XcTest targets (and others, just in case) for SenTestingKit.framework.
326     result.add(new NSString("$(SDKROOT)/Developer/Library/Frameworks"));
327     // This is needed by non-XcTest targets that use XcTest.framework, for instance for test
328     // utility libraries packaged as an objc_library.
329     result.add(new NSString("$(PLATFORM_DIR)/Developer/Library/Frameworks"));
330 
331     return (NSArray) NSObject.wrap(result.build().asList());
332   }

334   /**
335    * Returns the {@code ARCHS} array for a target's build config given the list of architecture
336    * strings. If none is given, an array with default architectures "armv7" and "arm64" will be
337    * returned.
338    */
339   private static NSArray cpuArchitectures(Iterable<String> architectures) {
340     if (Iterables.isEmpty(architectures)) {
341       return new NSArray(new NSString("armv7"), new NSString("arm64"));
342     } else {
343       ImmutableSet.Builder<NSString> result = new ImmutableSet.Builder<>();
344       for (String architecture : architectures) {
345         result.add(new NSString(architecture));
346       }
347       return (NSArray) NSObject.wrap(result.build().asList());
348     }
349   }

351   private static PBXFrameworksBuildPhase buildLibraryInfo(
352       LibraryObjects libraryObjects, TargetControl target) {
353     BuildPhaseBuilder builder = libraryObjects.newBuildPhase();
354     if (Containing.item(PRODUCT_TYPES_THAT_HAVE_A_BINARY, productType(target))) {
355       for (String dylib : target.getSdkDylibList()) {
356         builder.addDylib(dylib);
357       }
358     }
359     for (String sdkFramework : target.getSdkFrameworkList()) {
360       builder.addSdkFramework(sdkFramework);
361     }
362     for (String framework : target.getFrameworkList()) {
363       builder.addFramework(framework);
364     }
365     return builder.build();
366   }

368   private static ImmutableList<String> otherLdflags(TargetControl targetControl) {
369     Iterable<String> givenFlags = targetControl.getLinkoptList();
370     ImmutableList.Builder<String> flags = new ImmutableList.Builder<>();
371     flags.addAll(givenFlags);
372     if (!Equaling.of(ProductType.STATIC_LIBRARY, productType(targetControl))) {
373       for (String importedLibrary : targetControl.getImportedLibraryList()) {
374         flags.add("$(WORKSPACE_ROOT)/" + importedLibrary);
375       }
376     }
377     return flags.build();
378   }

380   /** Generates a project file. */
381   public static PBXProject xcodeproj(Path workspaceRoot, Control control,
382       Iterable<PbxReferencesProcessor> postProcessors) {
383     checkArgument(control.hasPbxproj(), "Must set pbxproj field on control proto.");
384     FileSystem fileSystem = workspaceRoot.getFileSystem();
385 
386     XcodeprojPath<Path> outputPath = XcodeprojPath.converter().fromPath(
387         RelativePaths.fromString(fileSystem, control.getPbxproj()));
388 
389     NSDictionary projBuildConfigMap = new NSDictionary();
390     projBuildConfigMap.put("ARCHS", cpuArchitectures(control.getCpuArchitectureList()));
391     projBuildConfigMap.put("CLANG_ENABLE_OBJC_ARC", "YES");
392     projBuildConfigMap.put("SDKROOT", "iphoneos");
393     projBuildConfigMap.put("IPHONEOS_DEPLOYMENT_TARGET", "7.0");
394     projBuildConfigMap.put("GCC_VERSION", "com.apple.compilers.llvm.clang.1_0");
395     projBuildConfigMap.put("CODE_SIGN_IDENTITY[sdk=iphoneos*]", "iPhone Developer");
396 
397     for (XcodeprojBuildSetting projectSetting : control.getBuildSettingList()) {
398       projBuildConfigMap.put(projectSetting.getName(), projectSetting.getValue());
399     }
400 
401     PBXProject project = new PBXProject(outputPath.getProjectName());
402     project.getMainGroup().setPath(workspaceRoot.toString());
403     if (workspaceRoot.isAbsolute()) {
404       project.getMainGroup().setSourceTree(SourceTree.ABSOLUTE);
405     }
406     try {
407       project
408           .getBuildConfigurationList()
409           .getBuildConfigurationsByName()
410           .get(DEFAULT_OPTIONS_NAME)
411           .setBuildSettings(projBuildConfigMap);
412     } catch (ExecutionException e) {
413       throw new RuntimeException(e);
414     }
415 
416     Map<String, TargetInfo> targetInfoByLabel = new HashMap<>();
417     List<String> usedTargetNames = new ArrayList<>();
418     PBXFileReferences fileReferences = new PBXFileReferences();
419     LibraryObjects libraryObjects = new LibraryObjects(fileReferences);
420     PBXBuildFiles pbxBuildFiles = new PBXBuildFiles(fileReferences);
421     Resources resources =
422         Resources.fromTargetControls(fileSystem, pbxBuildFiles, control.getTargetList());
423     Xcdatamodels xcdatamodels =
424         Xcdatamodels.fromTargetControls(fileSystem, pbxBuildFiles, control.getTargetList());
425     // We use a hash set for the Project Navigator files so that the same PBXFileReference does not
426     // get added twice. Because PBXFileReference uses equality-by-identity semantics, this requires
427     // the PBXFileReferences cache to properly return the same reference for functionally-equivalent
428     // files.
429     Set<PBXReference> projectNavigatorFiles = new LinkedHashSet<>();
430     for (TargetControl targetControl : control.getTargetList()) {
431       checkArgument(targetControl.hasName(), "TargetControl requires a name: %s", targetControl);
432       checkArgument(targetControl.hasLabel(), "TargetControl requires a label: %s", targetControl);
433 
434       ProductType productType = productType(targetControl);
435       Preconditions.checkArgument(
436           (productType != ProductType.APPLICATION) || hasAtLeastOneCompilableSource(targetControl),
437           APP_NEEDS_SOURCE_ERROR);
438       PBXSourcesBuildPhase sourcesBuildPhase = new PBXSourcesBuildPhase();
439 
440       for (SourceFile source : SourceFile.allSourceFiles(fileSystem, targetControl)) {
441         PBXFileReference fileRef =
442             fileReferences.get(FileReference.of(source.path().toString(), SourceTree.GROUP));
443         projectNavigatorFiles.add(fileRef);
444         if (Equaling.of(source.buildType(), BuildType.NO_BUILD)) {
445           continue;
446         }
447         PBXBuildFile buildFile = new PBXBuildFile(fileRef);
448         if (Equaling.of(source.buildType(), BuildType.NON_ARC_BUILD)) {
449           buildFile.setSettings(Optional.of(nonArcCompileSettings()));
450         }
451         sourcesBuildPhase.getFiles().add(buildFile);
452       }
453       sourcesBuildPhase.getFiles().addAll(xcdatamodels.buildFiles().get(targetControl));
454 
455       PBXFileReference productReference = fileReferences.get(productReference(targetControl));
456       projectNavigatorFiles.add(productReference);
457 
458       NSDictionary targetBuildConfigMap = new NSDictionary();
459       // TODO(bazel-team): Stop adding the workspace root automatically once the
460       // released version of Bazel starts passing it.
461       targetBuildConfigMap.put("USER_HEADER_SEARCH_PATHS",
462           headerSearchPaths(
463               plus(targetControl.getUserHeaderSearchPathList(), "$(WORKSPACE_ROOT)")));
464       targetBuildConfigMap.put("HEADER_SEARCH_PATHS",
465           headerSearchPaths(
466               plus(targetControl.getHeaderSearchPathList(), "$(inherited)")));
467       targetBuildConfigMap.put("FRAMEWORK_SEARCH_PATHS",
468           frameworkSearchPaths(targetControl.getFrameworkList()));
469 
470       targetBuildConfigMap.put("WORKSPACE_ROOT", workspaceRoot.toString());
471 
472       if (targetControl.hasPchPath()) {
473         targetBuildConfigMap.put(
474             "GCC_PREFIX_HEADER", "$(WORKSPACE_ROOT)/" + targetControl.getPchPath());
475       }
476 
477       targetBuildConfigMap.put("PRODUCT_NAME", productName(targetControl));
478       if (targetControl.hasInfoplist()) {
479         targetBuildConfigMap.put(
480             "INFOPLIST_FILE", "$(WORKSPACE_ROOT)/" + targetControl.getInfoplist());
481       }
482 
483       // Double-quotes in copt strings need to be escaped for XCode.
484       if (targetControl.getCoptCount() > 0) {
485         List<String> escapedCopts = Lists.transform(
486             targetControl.getCoptList(), QUOTE_ESCAPER.asFunction());
487         targetBuildConfigMap.put("OTHER_CFLAGS", NSObject.wrap(escapedCopts));
488       }
489       targetBuildConfigMap.put("OTHER_LDFLAGS", NSObject.wrap(otherLdflags(targetControl)));
490       for (XcodeprojBuildSetting setting : targetControl.getBuildSettingList()) {
491         String name = setting.getName();
492         String value = setting.getValue();
493         // TODO(bazel-team): Remove this hack after next Bazel release.
494         if (name.equals("CODE_SIGN_ENTITLEMENTS") && !value.startsWith("$")) {
495           value = "$(WORKSPACE_ROOT)/" + value;
496         }
497         targetBuildConfigMap.put(name, value);
498       }
499 
500       String targetName = targetControl.getName();
501       if (usedTargetNames.contains(targetName)) {
502         // Use the label in the odd case where we have two targets with the same name.
503         targetName = targetControl.getLabel();
504       }
505       checkState(!usedTargetNames.contains(targetName),
506           "Name already exists for target with label/name %s/%s in list: %s",
507           targetControl.getLabel(), targetControl.getName(), usedTargetNames);
508       usedTargetNames.add(targetName);
509       PBXNativeTarget target = new PBXNativeTarget(targetName, productType);
510       try {
511         target
512             .getBuildConfigurationList()
513             .getBuildConfigurationsByName()
514             .get(DEFAULT_OPTIONS_NAME)
515             .setBuildSettings(targetBuildConfigMap);
516       } catch (ExecutionException e) {
517         throw new RuntimeException(e);
518       }
519       target.setProductReference(productReference);
520 
521       PBXFrameworksBuildPhase frameworksPhase = buildLibraryInfo(libraryObjects, targetControl);
522       PBXResourcesBuildPhase resourcesPhase = resources.resourcesBuildPhase(targetControl);
523 
524       for (String importedArchive : targetControl.getImportedLibraryList()) {
525         PBXFileReference fileReference = fileReferences.get(
526             FileReference.of(importedArchive, SourceTree.GROUP)
527                 .withExplicitFileType(FILE_TYPE_ARCHIVE_LIBRARY));
528         projectNavigatorFiles.add(fileReference);
529       }
530 
531       project.getTargets().add(target);
532 
533       target.getBuildPhases().add(frameworksPhase);
534       target.getBuildPhases().add(sourcesBuildPhase);
535       target.getBuildPhases().add(resourcesPhase);
536 
537       checkState(!Mapping.of(targetInfoByLabel, targetControl.getLabel()).isPresent(),
538           "Mapping already exists for target with label %s in map: %s",
539           targetControl.getLabel(), targetInfoByLabel);
540       targetInfoByLabel.put(
541           targetControl.getLabel(),
542           new TargetInfo(
543               targetControl,
544               target,
545               frameworksPhase,
546               resourcesPhase,
547               new PBXBuildFile(productReference),
548               new LocalPBXTargetDependency(
549                   new LocalPBXContainerItemProxy(
550                       project, target, ProxyType.TARGET_REFERENCE)),
551               targetBuildConfigMap));
552     }
553 
554     for (HasProjectNavigatorFiles references : ImmutableList.of(pbxBuildFiles, libraryObjects)) {
555       Iterables.addAll(projectNavigatorFiles, references.mainGroupReferences());
556     }
557 
558     Iterable<PBXReference> processedProjectFiles = projectNavigatorFiles;
559     for (PbxReferencesProcessor postProcessor : postProcessors) {
560       processedProjectFiles = postProcessor.process(processedProjectFiles);
561     }
562 
563     Iterables.addAll(project.getMainGroup().getChildren(), processedProjectFiles);
564     for (TargetInfo targetInfo : targetInfoByLabel.values()) {
565       for (DependencyControl dependency : targetInfo.control.getDependencyList()) {
566         targetInfo.addDependencyInfo(dependency, targetInfoByLabel);
567       }
568     }
569 
570     return project;
571   }
