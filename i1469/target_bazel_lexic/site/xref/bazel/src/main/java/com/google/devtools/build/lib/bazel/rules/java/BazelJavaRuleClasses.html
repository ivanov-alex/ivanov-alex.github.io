67    /**
68     * Common attributes for rules that depend on ijar.
69     */
70    public static final class IjarBaseRule implements RuleDefinition {
71      @Override
72      public RuleClass build(Builder builder, RuleDefinitionEnvironment env) {
73        return builder
74            .add(attr("$ijar", LABEL).cfg(HOST).exec().value(env.getLabel("//tools/defaults:ijar")))
75            .setPreferredDependencyPredicate(JavaSemantics.JAVA_SOURCE)
76            .build();
77      }
78  
79      @Override
80      public Metadata getMetadata() {
81        return RuleDefinition.Metadata.builder()
82            .name("$ijar_base_rule")
83            .type(RuleClassType.ABSTRACT)
84            .build();
85      }
86    }

134   /**
135    * Common attributes for Java rules.
136    */
137   public static final class JavaRule implements RuleDefinition {
138     @Override
139     public RuleClass build(Builder builder, RuleDefinitionEnvironment env) {
140       return builder
141           /* <!-- #BLAZE_RULE($java_rule).ATTRIBUTE(deps) -->
142           The list of other libraries to be linked in to the target.
143           ${SYNOPSIS}
144           See general comments about <code>deps</code> at <a href="#common-attributes">Attributes
145           common to all build rules</a>.
146           <!-- #END_BLAZE_RULE.ATTRIBUTE --> */
147           .override(builder.copy("deps")
148               .allowedFileTypes(JavaSemantics.JAR)
149               .allowedRuleClasses(ALLOWED_RULES_IN_DEPS)
150               .skipAnalysisTimeFileTypeCheck())
151           /* <!-- #BLAZE_RULE($java_rule).ATTRIBUTE(runtime_deps) -->
152           Libraries to make available to the final binary or test at runtime only.
153           ${SYNOPSIS}
154           Like ordinary <code>deps</code>, these will appear on the runtime classpath, but unlike
155           them, not on the compile-time classpath. Dependencies needed only at runtime should be
156           listed here. Dependency-analysis tools should ignore targets that appear in both
157           <code>runtime_deps</code> and <code>deps</code>.
158           <!-- #END_BLAZE_RULE.ATTRIBUTE --> */
159           .add(attr("runtime_deps", LABEL_LIST)
160               .allowedFileTypes(JavaSemantics.JAR)
161               .allowedRuleClasses(ALLOWED_RULES_IN_DEPS)
162               .skipAnalysisTimeFileTypeCheck())
163 
164           /* <!-- #BLAZE_RULE($java_rule).ATTRIBUTE(srcs) -->
165           The list of source files that are processed to create the target.
166           ${SYNOPSIS}
167           This attribute is almost always required; see exceptions below.
168           <p>
169             Source files of type <code>.java</code> are compiled. In case of generated
170             <code>.java</code> files it is generally advisable to put the generating rule's name
171             here instead of the name of the file itself. This not only improves readability but
172             makes the rule more resilient to future changes: if the generating rule generates
173             different files in the future, you only need to fix one place: the <code>outs</code> of
174             the generating rule. You should not list the generating rule in <code>deps</code>
175             because it is a no-op.
176           </p>
177           <p>
178             Source files of type <code>.srcjar</code> are unpacked and compiled. (This is useful if
179             you need to generate a set of <code>.java</code> files with a genrule.)
180           </p>
181           <p>
182             Source files of type <code>.jar</code> are linked in. <em class="harmful">This is
183             discouraged, use <a href="#java_import"><code>java_import</code></a></em> if you need to
184             link against existing (or generated) jar files (which is useful if you have a
185             <code>.jar</code> file without sources).
186           </p>
187           <p>
188             Rules: if the rule (typically <code>genrule</code> or <code>filegroup</code>) generates
189             any of the files listed above, they will be used the same way as described for source
190             files.
191           </p>
192 
193           <p>
194             This argument is almost always required, except if a
195             <a href="#java_binary.main_class"><code>main_class</code></a> attribute specifies a
196             class on the runtime classpath or you specify the <code>runtime_deps</code> argument.
197           </p>
198           <!-- #END_BLAZE_RULE.ATTRIBUTE --> */
199           .add(attr("srcs", LABEL_LIST)
200               .orderIndependent()
201               .direct_compile_time_input()
202               .allowedFileTypes(JavaSemantics.JAVA_SOURCE, JavaSemantics.JAR,
203                   JavaSemantics.SOURCE_JAR, JavaSemantics.PROPERTIES))
204           /* <!-- #BLAZE_RULE($java_rule).ATTRIBUTE(resources) -->
205           A list of data files to include in a Java jar.
206           ${SYNOPSIS}
207           <p>
208             If resources are specified, they will be bundled in the jar along with the usual
209             <code>.class</code> files produced by compilation. The location of the resources inside
210             of the jar file is determined by the project structure. Bazel first looks for Maven's
211             <a href="https://maven.apache.org/guides/introduction/introduction-to-the-standard-directory-layout.html">standard directory layout</a>,
212             (a "src" directory followed by a "resources" directory grandchild). If that is not
213             found, Bazel then looks for the topmost directory named "java" or "javatests" (so, for
214             example, if a resource is at &lt;workspace root&gt;/x/java/y/java/z, Bazel will use the
215              path y/java/z. This heuristic cannot be overridden.
216           </p>
217 
218           <p>
219             Resources may be source files or generated files.
220           </p>
221           <!-- #END_BLAZE_RULE.ATTRIBUTE --> */
222           .add(attr("resources", LABEL_LIST).orderIndependent()
223               .allowedFileTypes(FileTypeSet.ANY_FILE))
224           /* <!-- #BLAZE_RULE($java_rule).ATTRIBUTE(plugins) -->
225           Java compiler plugins to run at compile-time.
226           ${SYNOPSIS}
227           Every <code>java_plugin</code> specified in this attribute will be run whenever this rule
228           is built. A library may also inherit plugins from dependencies that use
229           <code><a href="#java_library.exported_plugins">exported_plugins</a></code>. Resources
230           generated by the plugin will be included in the resulting jar of this rule.
231           <!-- #END_BLAZE_RULE.ATTRIBUTE --> */
232           .add(attr("plugins", LABEL_LIST).cfg(HOST).allowedRuleClasses("java_plugin")
233               .legacyAllowAnyFileType())
234           .add(attr(":java_plugins", LABEL_LIST)
235               .cfg(HOST)
236               .allowedRuleClasses("java_plugin")
237               .silentRuleClassFilter()
238               .value(JavaSemantics.JAVA_PLUGINS))
239           /* <!-- #BLAZE_RULE($java_rule).ATTRIBUTE(javacopts) -->
240           Extra compiler options for this library.
241           ${SYNOPSIS}
242           Subject to <a href="#make_variables">"Make variable"</a> substitution and
243           <a href="#sh-tokenization">Bourne shell tokenization</a>.
244           <p>These compiler options are passed to javac after the global compiler options.</p>
245           <!-- #END_BLAZE_RULE.ATTRIBUTE --> */
246           .add(attr("javacopts", STRING_LIST))
247           .build();
248     }
249 
250     @Override
251     public Metadata getMetadata() {
252       return RuleDefinition.Metadata.builder()
253           .name("$java_rule")
254           .type(RuleClassType.ABSTRACT)
255           .ancestors(BaseRuleClasses.RuleBase.class, JavaBaseRule.class)
256           .build();
257     }

260   /**
261    * Base class for rule definitions producing Java binaries.
262    */
263   public static final class BaseJavaBinaryRule implements RuleDefinition {
264     @Override
265     public RuleClass build(Builder builder, final RuleDefinitionEnvironment env) {
266       return builder
267           /* <!-- #BLAZE_RULE($base_java_binary).ATTRIBUTE(classpath_resources) -->
268           ${SYNOPSIS}
269           <em class="harmful">DO NOT USE THIS OPTION UNLESS THERE IS NO OTHER WAY)</em>
270           <p>
271             A list of resources that must be located at the root of the java tree. This attribute's
272             only purpose is to support third-party libraries that require that their resources be
273             found on the classpath as exactly <code>"myconfig.xml"</code>. It is only allowed on
274             binaries and not libraries, due to the danger of namespace conflicts.
275           </p>
276           <!-- #END_BLAZE_RULE.ATTRIBUTE --> */
277           .add(attr("classpath_resources", LABEL_LIST).legacyAllowAnyFileType())
278           /* <!-- #BLAZE_RULE($base_java_binary).ATTRIBUTE(jvm_flags) -->
279           A list of flags to embed in the wrapper script generated for running this binary.
280           ${SYNOPSIS}
281           Subject to <a href="#make_variables">"Make variable"</a> substitution and
282           <a href="#sh-tokenization">Bourne shell tokenization</a>.
283           <p>
284             The wrapper script for a Java binary includes a <code>CLASSPATH</code> definition (to
285             find all the dependent jars) and invokes the right Java interpreter. The command line
286             generated by the wrapper script includes the name of the main class followed by a
287             <code>"$@"</code> so you can pass along other arguments after the classname.  However,
288             arguments intended for parsing by the JVM must be specified <i>before</i> the classname
289             on the command line. The contents of <code>jvm_flags</code> are added to the wrapper
290             script before the classname is listed.
291           </p>
292           <!-- #END_BLAZE_RULE.ATTRIBUTE --> */
293           .add(attr("jvm_flags", STRING_LIST))
294           /* <!-- #BLAZE_RULE($base_java_binary).ATTRIBUTE(main_class) -->
295           Name of class with <code>main()</code> method to use as entry point.
296           ${SYNOPSIS}
297           If a rule uses this option, it does not need a <code>srcs=[...]</code> list.
298           Thus, with this attribute one can make an executable from a Java library that already
299           contains one or more <code>main()</code> methods.
300           <p>
301             The value of this attribute is a class name, not a source file. The class must be
302             available at runtime: it may be compiled by this rule (from <code>srcs</code>) or
303             provided by direct or transitive dependencies (through <code>runtime_deps</code> or
304             <code>deps</code>). If the class is unavailable, the binary will fail at runtime; there
305             is no build-time check.
306           </p>
307           <!-- #END_BLAZE_RULE.ATTRIBUTE --> */
308           .add(attr("main_class", STRING))
309           /* <!-- #BLAZE_RULE($base_java_binary).ATTRIBUTE(create_executable) -->
310           Whether to build the executable wrapper script or not.
311           ${SYNOPSIS}
312           If this option is present, no executable wrapper script is built around the
313           <code>.jar</code> file. Incompatible with <code>main_class</code> attribute.
314           <!-- #END_BLAZE_RULE.ATTRIBUTE --> */
315           .add(attr("create_executable", BOOLEAN)
316               .nonconfigurable("internal")
317               .value(true))
318           /* <!-- #BLAZE_RULE($base_java_binary).ATTRIBUTE(deploy_manifest_lines) -->
319           ${SYNOPSIS}
320           A list of lines to add to the <code>META-INF/manifest.mf</code> file generated for the
321           <code>*_deploy.jar</code> target. The contents of this attribute are <em>not</em> subject
322           to <a href="#make_variables">"Make variable"</a> substitution.
323           <!-- #END_BLAZE_RULE.ATTRIBUTE --> */
324           .add(attr("deploy_manifest_lines", STRING_LIST))
325           /* <!-- #BLAZE_RULE($base_java_binary).ATTRIBUTE(stamp) -->
326           Enable link stamping.
327           ${SYNOPSIS}
328           Whether to encode build information into the binary. Possible values:
329           <ul>
330             <li><code>stamp = 1</code>: Stamp the build information into the binary. Stamped
331               binaries are only rebuilt when their dependencies change. Use this if there are tests
332               that depend on the build information.</li>
333             <li><code>stamp = 0</code>: Always replace build information by constant values. This
334               gives good build result caching.</li>
335             <li><code>stamp = -1</code>: Embedding of build information is controlled by the
336               <a href="blaze-user-manual.html#flag--stamp">--[no]stamp</a> flag.</li>
337           </ul>
338           <!-- #END_BLAZE_RULE.ATTRIBUTE --> */
339           // TODO(bazel-team): describe how to access this data at runtime
340           .add(attr("stamp", TRISTATE).value(TriState.AUTO))
341           .add(attr(":java_launcher", LABEL).value(JavaSemantics.JAVA_LAUNCHER))  // blaze flag
342           .build();
343     }
344     @Override
345     public Metadata getMetadata() {
346       return RuleDefinition.Metadata.builder()
347           .name("$base_java_binary")
348           .type(RuleClassType.ABSTRACT)
349           .ancestors(JavaRule.class,
350               // java_binary and java_test require the crosstool C++ runtime
351               // libraries (libstdc++.so, libgcc_s.so).
352               // TODO(bazel-team): Add tests for Java+dynamic runtime.
353               BazelCppRuleClasses.CcLinkingRule.class)
354           .build();
355     }
