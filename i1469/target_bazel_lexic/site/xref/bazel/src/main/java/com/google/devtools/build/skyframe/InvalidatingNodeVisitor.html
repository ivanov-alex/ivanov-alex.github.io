34  /**
35   * A visitor that is useful for invalidating transitive dependencies of Skyframe nodes.
36   *
37   * <p>Interruptibility: It is safe to interrupt the invalidation process at any time. Consider a
38   * graph and a set of modified nodes. Then the reverse transitive closure of the modified nodes is
39   * the set of dirty nodes. We provide interruptibility by making sure that the following invariant
40   * holds at any time:
41   *
42   * <p>If a node is dirty, but not removed (or marked as dirty) yet, then either it or any of its
43   * transitive dependencies must be in the {@link #pendingVisitations} set. Furthermore, reverse dep
44   * pointers must always point to existing nodes.
45   *
46   * <p>Thread-safety: This class should only be instantiated and called on a single thread, but
47   * internally it spawns many worker threads to process the graph. The thread-safety of the workers
48   * on the graph can be delicate, and is documented below. Moreover, no other modifications to the
49   * graph can take place while invalidation occurs.
50   *
51   * <p>This is intended only for use in alternative {@code MemoizingEvaluator} implementations.
52   */
53  public abstract class InvalidatingNodeVisitor extends AbstractQueueVisitor {
54  
55    // Default thread count is equal to the number of cores to exploit
56    // that level of hardware parallelism, since invalidation should be CPU-bound.
57    // We may consider increasing this in the future.
58    private static final int DEFAULT_THREAD_COUNT = Runtime.getRuntime().availableProcessors();
59  
60    private static final boolean MUST_EXIST = true;
61  
62    protected final DirtiableGraph graph;
63    @Nullable protected final EvaluationProgressReceiver invalidationReceiver;
64    protected final DirtyKeyTracker dirtyKeyTracker;
65    // Aliased to InvalidationState.pendingVisitations.
66    protected final Set<Pair<SkyKey, InvalidationType>> pendingVisitations;
67  
68    protected InvalidatingNodeVisitor(
69        DirtiableGraph graph, @Nullable EvaluationProgressReceiver invalidationReceiver,
70        InvalidationState state, DirtyKeyTracker dirtyKeyTracker) {
71      this(graph, invalidationReceiver, state, dirtyKeyTracker, EXECUTOR_FACTORY);
72    }
73  
74    protected InvalidatingNodeVisitor(
75        DirtiableGraph graph, @Nullable EvaluationProgressReceiver invalidationReceiver,
76        InvalidationState state, DirtyKeyTracker dirtyKeyTracker,
77        Function<ThreadPoolExecutorParams, ThreadPoolExecutor> executorFactory) {
78      super(/*concurrent=*/true,
79          /*corePoolSize=*/DEFAULT_THREAD_COUNT,
80          /*maxPoolSize=*/DEFAULT_THREAD_COUNT,
81          /*keepAliveTime=*/1,
82          /*units=*/TimeUnit.SECONDS,
83          /*failFastOnException=*/true,
84          /*failFastOnInterrupt=*/true,
85          "skyframe-invalidator",
86          executorFactory);
87      this.graph = Preconditions.checkNotNull(graph);
88      this.invalidationReceiver = invalidationReceiver;
89      this.dirtyKeyTracker = Preconditions.checkNotNull(dirtyKeyTracker);
90      this.pendingVisitations = state.pendingValues;
91    }
92  
93    /**
94     * Initiates visitation and waits for completion.
95     */
96    void run() throws InterruptedException {
97      // Make a copy to avoid concurrent modification confusing us as to which nodes were passed by
98      // the caller, and which are added by other threads during the run. Since no tasks have been
99      // started yet (the queueDirtying calls start them), this is thread-safe.
100     for (Pair<SkyKey, InvalidationType> visitData : ImmutableList.copyOf(pendingVisitations)) {
101       // The caller may have specified non-existent SkyKeys, or there may be stale SkyKeys in
102       // pendingVisitations that have already been deleted. In both these cases, the nodes will not
103       // exist in the graph, so we must be tolerant of that case.
104       visit(visitData.first, visitData.second, !MUST_EXIST);
105     }
106     work(/*failFastOnInterrupt=*/true);
107     Preconditions.checkState(pendingVisitations.isEmpty(),
108         "All dirty nodes should have been processed: %s", pendingVisitations);
109   }
110 
111   protected abstract long count();
112 
113   protected void informInvalidationReceiver(SkyKey key,
114       EvaluationProgressReceiver.InvalidationState state) {
115     if (invalidationReceiver != null) {
116       invalidationReceiver.invalidated(key, state);
117     }
118   }
119 
120   /**
121    * Enqueues a node for invalidation.
122    */
123   @ThreadSafe
124   abstract void visit(SkyKey key, InvalidationType second, boolean mustExist);
125 
126   @VisibleForTesting
127   enum InvalidationType {
128     /**
129      * The node is dirty and must be recomputed.
130      */
131     CHANGED,
132     /**
133      * The node is dirty, but may be marked clean later during change pruning.
134      */
135     DIRTIED,
136     /**
137      * The node is deleted.
138      */
139     DELETED;
140   }
141 
142   /**
143    * Invalidation state object that keeps track of which nodes need to be invalidated, but have not
144    * been dirtied/deleted yet. This supports interrupts - by only deleting a node from this set
145    * when all its parents have been invalidated, we ensure that no information is lost when an
146    * interrupt comes in.
147    */
148   static class InvalidationState {
149     private final Set<Pair<SkyKey, InvalidationType>> pendingValues = Sets.newConcurrentHashSet();
150     private final InvalidationType defaultUpdateType;
151 
152     private InvalidationState(InvalidationType defaultUpdateType) {
153       this.defaultUpdateType = Preconditions.checkNotNull(defaultUpdateType);
154     }
155 
156     void update(Iterable<SkyKey> diff) {
157       Iterables.addAll(pendingValues, Iterables.transform(diff,
158           new Function<SkyKey, Pair<SkyKey, InvalidationType>>() {
159             @Override
160             public Pair<SkyKey, InvalidationType> apply(SkyKey skyKey) {
161               return Pair.of(skyKey, defaultUpdateType);
162             }
163           }));
164     }
165 
166     @VisibleForTesting
167     boolean isEmpty() {
168       return pendingValues.isEmpty();
169     }
170 
171     @VisibleForTesting
172     Set<Pair<SkyKey, InvalidationType>> getInvalidationsForTesting() {
173       return ImmutableSet.copyOf(pendingValues);
174     }
175   }
176 
177   public static class DirtyingInvalidationState extends InvalidationState {
178     public DirtyingInvalidationState() {
179       super(InvalidationType.CHANGED);
180     }
181   }
182 
183   static class DeletingInvalidationState extends InvalidationState {
184     public DeletingInvalidationState() {
185       super(InvalidationType.DELETED);
186     }
187   }

189   /**
190    * A node-deleting implementation.
191    */
192   static class DeletingNodeVisitor extends InvalidatingNodeVisitor {
193 
194     private final Set<SkyKey> visitedValues = Sets.newConcurrentHashSet();
195     private final boolean traverseGraph;
196 
197     protected DeletingNodeVisitor(DirtiableGraph graph,
198         EvaluationProgressReceiver invalidationReceiver, InvalidationState state,
199         boolean traverseGraph, DirtyKeyTracker dirtyKeyTracker) {
200       super(graph, invalidationReceiver, state, dirtyKeyTracker);
201       this.traverseGraph = traverseGraph;
202     }
203 
204     @Override
205     protected long count() {
206       return visitedValues.size();
207     }
208 
209     @Override
210     public void visit(final SkyKey key, InvalidationType invalidationType, boolean mustExist) {
211       Preconditions.checkState(invalidationType == InvalidationType.DELETED, key);
212       if (!visitedValues.add(key)) {
213         return;
214       }
215       final Pair<SkyKey, InvalidationType> invalidationPair = Pair.of(key, invalidationType);
216       pendingVisitations.add(invalidationPair);
217       enqueue(new Runnable() {
218         @Override
219         public void run() {
220           NodeEntry entry = graph.get(key);
221           if (entry == null) {
222             pendingVisitations.remove(invalidationPair);
223             return;
224           }
225 
226           if (traverseGraph) {
227             // Propagate deletion upwards.
228             for (SkyKey reverseDep : entry.getReverseDeps()) {
229               visit(reverseDep, InvalidationType.DELETED, !MUST_EXIST);
230             }
231           }
232 
233           if (entry.isDone()) {
234             // Only process this node's value and children if it is done, since dirty nodes have
235             // no awareness of either.
236 
237             // Unregister this node from direct deps, since reverse dep edges cannot point to
238             // non-existent nodes.
239             if (traverseGraph) {
240               for (SkyKey directDep : entry.getDirectDeps()) {
241                 NodeEntry dep = graph.get(directDep);
242                 if (dep != null) {
243                   dep.removeReverseDep(key);
244                 }
245               }
246             }
247             // Allow custom key-specific logic to update dirtiness status.
248             informInvalidationReceiver(key, EvaluationProgressReceiver.InvalidationState.DELETED);
249           }
250           if (traverseGraph) {
251             // Force reverseDeps consolidation (validates that attempts to remove reverse deps were
252             // really successful.
253             entry.getReverseDeps();
254           }
255           // Actually remove the node.
256           graph.remove(key);
257           dirtyKeyTracker.notDirty(key);
258 
259           // Remove the node from the set as the last operation.
260           pendingVisitations.remove(invalidationPair);
261         }
262       });
263     }
264   }

266   /**
267    * A node-dirtying implementation.
268    */
269   static class DirtyingNodeVisitor extends InvalidatingNodeVisitor {
270 
271     private final Set<Pair<SkyKey, InvalidationType>> visited = Sets.newConcurrentHashSet();
272 
273     protected DirtyingNodeVisitor(DirtiableGraph graph,
274         EvaluationProgressReceiver invalidationReceiver, InvalidationState state,
275         DirtyKeyTracker dirtyKeyTracker,
276         Function<ThreadPoolExecutorParams, ThreadPoolExecutor> executorFactory) {
277       super(graph, invalidationReceiver, state, dirtyKeyTracker, executorFactory);
278     }
279 
280     @Override
281     protected long count() {
282       return visited.size();
283     }
284 
285     /**
286      * Queues a task to dirty the node named by {@code key}. May be called from multiple threads.
287      * It is possible that the same node is enqueued many times. However, we require that a node
288      * is only actually marked dirty/changed once, with two exceptions:
289      *
290      * (1) If a node is marked dirty, it can subsequently be marked changed. This can occur if, for
291      * instance, FileValue workspace/foo/foo.cc is marked dirty because FileValue workspace/foo is
292      * marked changed (and every FileValue depends on its parent). Then FileValue
293      * workspace/foo/foo.cc is itself changed (this can even happen on the same build).
294      *
295      * (2) If a node is going to be marked both dirty and changed, as, for example, in the previous
296      * case if both workspace/foo/foo.cc and workspace/foo have been changed in the same build, the
297      * thread marking workspace/foo/foo.cc dirty may race with the one marking it changed, and so
298      * try to mark it dirty after it has already been marked changed. In that case, the
299      * {@link NodeEntry} ignores the second marking.
300      *
301      * The invariant that we do not process a (SkyKey, InvalidationType) pair twice is enforced by
302      * the {@link #visited} set.
303      *
304      * The "invariant" is also enforced across builds by checking to see if the entry is already
305      * marked changed, or if it is already marked dirty and we are just going to mark it dirty
306      * again.
307      *
308      * If either of the above tests shows that we have already started a task to mark this entry
309      * dirty/changed, or that it is already marked dirty/changed, we do not continue this task.
310      */
311     @Override
312     @ThreadSafe
313     public void visit(final SkyKey key, final InvalidationType invalidationType,
314         final boolean mustExist) {
315       Preconditions.checkState(invalidationType != InvalidationType.DELETED, key);
316       final boolean isChanged = (invalidationType == InvalidationType.CHANGED);
317       final Pair<SkyKey, InvalidationType> invalidationPair = Pair.of(key, invalidationType);
318       if (!visited.add(invalidationPair)) {
319         return;
320       }
321       pendingVisitations.add(invalidationPair);
322       enqueue(new Runnable() {
323         @Override
324         public void run() {
325           NodeEntry entry = graph.get(key);
326 
327           if (entry == null) {
328             Preconditions.checkState(!mustExist,
329                 "%s does not exist in the graph but was enqueued for dirtying by another node",
330                 key);
331             pendingVisitations.remove(invalidationPair);
332             return;
333           }
334 
335           if (entry.isChanged() || (!isChanged && entry.isDirty())) {
336             // If this node is already marked changed, or we are only marking this node dirty, and
337             // it already is, move along.
338             pendingVisitations.remove(invalidationPair);
339             return;
340           }
341 
342           // This entry remains in the graph in this dirty state until it is re-evaluated.
343           Iterable<SkyKey> deps = entry.markDirty(isChanged);
344           // It is not safe to interrupt the logic from this point until the end of the method.
345           // Any exception thrown should be unrecoverable.
346           if (deps == null) {
347             // Another thread has already dirtied this node. Don't do anything in this thread.
348             pendingVisitations.remove(invalidationPair);
349             return;
350           }
351           // Propagate dirtiness upwards and mark this node dirty/changed. Reverse deps should only
352           // be marked dirty (because only a dependency of theirs has changed).
353           for (SkyKey reverseDep : entry.getReverseDeps()) {
354             visit(reverseDep, InvalidationType.DIRTIED, MUST_EXIST);
355           }
356 
357           // Remove this node as a reverse dep from its children, since we have reset it and it no
358           // longer lists its children as direct deps.
359           Map<SkyKey, NodeEntry> children = graph.getBatch(deps);
360           if (children.size() != Iterables.size(deps)) {
361             Set<SkyKey> depsSet = ImmutableSet.copyOf(deps);
362             throw new IllegalStateException("Mismatch in getBatch: " + key + ", " + entry + "\n"
363                 + Sets.difference(depsSet, children.keySet()) + "\n"
364                 + Sets.difference(children.keySet(), depsSet));
365           }
366           for (NodeEntry child : children.values()) {
367             child.removeReverseDep(key);
368           }
369 
370           informInvalidationReceiver(key, EvaluationProgressReceiver.InvalidationState.DIRTY);
371           dirtyKeyTracker.dirty(key);
372           // Remove the node from the set as the last operation.
373           pendingVisitations.remove(invalidationPair);
374         }
375       });
376     }
377   }
