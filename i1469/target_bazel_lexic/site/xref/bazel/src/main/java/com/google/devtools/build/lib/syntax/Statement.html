14  package com.google.devtools.build.lib.syntax;
15  
16  /**
17   * Base class for all statements nodes in the AST.
18   */
19  public abstract class Statement extends ASTNode {
20  
21    /**
22     * Executes the statement in the specified build environment, which may be
23     * modified.
24     *
25     * @throws EvalException if execution of the statement could not be completed.
26     */
27    abstract void exec(Environment env) throws EvalException, InterruptedException;
28  
29    /**
30     * Checks the semantics of the Statement using the SkylarkEnvironment according to
31     * the rules of the Skylark language. The SkylarkEnvironment can be used e.g. to check
32     * variable type collision, read only variables, detecting recursion, existence of
33     * built-in variables, functions, etc.
34     *
35     * <p>The semantical check should be performed after the Skylark extension is loaded
36     * (i.e. is syntactically correct) and before is executed. The point of the semantical check
37     * is to make sure (as much as possible) that no error can occur during execution (Skylark
38     * programmers get a "compile time" error). It should also check execution branches (e.g. in
39     * if statements) that otherwise might never get executed.
40     *
41     * @throws EvalException if the Statement has a semantical error.
42     */
43    abstract void validate(ValidationEnvironment env) throws EvalException;
44  }
