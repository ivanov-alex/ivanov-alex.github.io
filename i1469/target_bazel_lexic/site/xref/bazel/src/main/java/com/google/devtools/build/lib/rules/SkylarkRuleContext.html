118   /**
119    * Creates a new SkylarkRuleContext using ruleContext.
120    */
121   public SkylarkRuleContext(RuleContext ruleContext) throws EvalException {
122     this.ruleContext = Preconditions.checkNotNull(ruleContext);
123     fragments = new FragmentCollection(ruleContext);
124 
125     HashMap<String, Object> outputsBuilder = new HashMap<>();
126     if (ruleContext.getRule().getRuleClassObject().outputsDefaultExecutable()) {
127       addOutput(outputsBuilder, "executable", ruleContext.createOutputArtifact());
128     }
129     ImplicitOutputsFunction implicitOutputsFunction =
130         ruleContext.getRule().getRuleClassObject().getImplicitOutputsFunction();
131 
132     if (implicitOutputsFunction instanceof SkylarkImplicitOutputsFunction) {
133       SkylarkImplicitOutputsFunction func = (SkylarkImplicitOutputsFunction)
134           ruleContext.getRule().getRuleClassObject().getImplicitOutputsFunction();
135       for (Map.Entry<String, String> entry : func.calculateOutputs(
136           RawAttributeMapper.of(ruleContext.getRule())).entrySet()) {
137         addOutput(outputsBuilder, entry.getKey(),
138             ruleContext.getImplicitOutputArtifact(entry.getValue()));
139       }
140     }
141 
142     ImmutableMap.Builder<Artifact, Label> artifactLabelMapBuilder =
143         ImmutableMap.builder();
144     for (Attribute a : ruleContext.getRule().getAttributes()) {
145       String attrName = a.getName();
146       Type<?> type = a.getType();
147       if (type != Type.OUTPUT && type != Type.OUTPUT_LIST) {
148         continue;
149       }
150       ImmutableList.Builder<Artifact> artifactsBuilder = ImmutableList.builder();
151       for (OutputFile outputFile : ruleContext.getRule().getOutputFileMap().get(attrName)) {
152         Artifact artifact = ruleContext.createOutputArtifact(outputFile);
153         artifactsBuilder.add(artifact);
154         artifactLabelMapBuilder.put(artifact, outputFile.getLabel());
155       }
156       ImmutableList<Artifact> artifacts = artifactsBuilder.build();
157 
158       if (type == Type.OUTPUT) {
159         if (artifacts.size() == 1) {
160           addOutput(outputsBuilder, attrName, Iterables.getOnlyElement(artifacts));
161         } else {
162           addOutput(outputsBuilder, attrName, Environment.NONE);
163         }
164       } else if (type == Type.OUTPUT_LIST) {
165         addOutput(outputsBuilder, attrName,
166             SkylarkList.list(artifacts, Artifact.class));
167       } else {
168         throw new IllegalArgumentException(
169             "Type of " + attrName + "(" + type + ") is not output type ");
170       }
171     }
172     artifactLabelMap = artifactLabelMapBuilder.build();
173     outputsObject = new SkylarkClassObject(outputsBuilder, "No such output '%s'");
174 
175     ImmutableMap.Builder<String, Object> attrBuilder = new ImmutableMap.Builder<>();
176     ImmutableMap.Builder<String, Object> executableBuilder = new ImmutableMap.Builder<>();
177     ImmutableMap.Builder<Artifact, FilesToRunProvider> executableRunfilesbuilder =
178         new ImmutableMap.Builder<>();
179     ImmutableMap.Builder<String, Object> fileBuilder = new ImmutableMap.Builder<>();
180     ImmutableMap.Builder<String, Object> filesBuilder = new ImmutableMap.Builder<>();
181     for (Attribute a : ruleContext.getRule().getAttributes()) {
182       Type<?> type = a.getType();
183       Object val = ruleContext.attributes().get(a.getName(), type);
184       if (type != Type.LABEL && type != Type.LABEL_LIST) {
185         attrBuilder.put(a.getPublicName(), val == null ? Environment.NONE
186             // Attribute values should be type safe
187             : SkylarkType.convertToSkylark(val, null));
188         continue;
189       }
190       String skyname = a.getPublicName();
191       Mode mode = getMode(a.getName());
192       if (a.isExecutable()) {
193         // In Skylark only label (not label list) type attributes can have the Executable flag.
194         FilesToRunProvider provider = ruleContext.getExecutablePrerequisite(a.getName(), mode);
195         if (provider != null && provider.getExecutable() != null) {
196           Artifact executable = provider.getExecutable();
197           executableBuilder.put(skyname, executable);
198           executableRunfilesbuilder.put(executable, provider);
199         } else {
200           executableBuilder.put(skyname, Environment.NONE);
201         }
202       }
203       if (a.isSingleArtifact()) {
204         // In Skylark only label (not label list) type attributes can have the SingleArtifact flag.
205         Artifact artifact = ruleContext.getPrerequisiteArtifact(a.getName(), mode);
206         if (artifact != null) {
207           fileBuilder.put(skyname, artifact);
208         } else {
209           fileBuilder.put(skyname, Environment.NONE);
210         }
211       }
212       filesBuilder.put(skyname, ruleContext.getPrerequisiteArtifacts(a.getName(), mode).list());
213       List<?> allPrereq = ruleContext.getPrerequisites(a.getName(), mode);
214       if (type == Type.LABEL) {
215         Object prereq = ruleContext.getPrerequisite(a.getName(), mode);
216         if (prereq == null) {
217           prereq = Environment.NONE;
218         }
219         attrBuilder.put(skyname, prereq);
220       } else {
221         // Type.LABEL_LIST
222         attrBuilder.put(skyname, SkylarkList.list(allPrereq, TransitiveInfoCollection.class));
223       }
224     }
225     attrObject = new SkylarkClassObject(attrBuilder.build(), "No such attribute '%s'");
226     executableObject = new SkylarkClassObject(executableBuilder.build(), "No such executable. "
227         + "Make sure there is a '%s' label type attribute marked as 'executable'");
228     fileObject = new SkylarkClassObject(fileBuilder.build(),
229         "No such file. Make sure there is a '%s' label type attribute marked as 'single_file'");
230     filesObject = new SkylarkClassObject(filesBuilder.build(),
231         "No such files. Make sure there is a '%s' label or label_list type attribute");
232     executableRunfilesMap = executableRunfilesbuilder.build();
233 
234     makeVariables = ruleContext.getConfigurationMakeVariableContext().collectMakeVariables();
235   }

237   private void addOutput(HashMap<String, Object> outputsBuilder, String key, Object value)
238       throws EvalException {
239     if (outputsBuilder.containsKey(key)) {
240       throw new EvalException(null, "Multiple outputs with the same key: " + key);
241     }
242     outputsBuilder.put(key, value);
243   }

369   @SkylarkCallable(doc = "Splits a shell command to a list of tokens.", documented = false)
370   public List<String> tokenize(String optionString) throws FuncallException {
371     List<String> options = new ArrayList<>();
372     try {
373       ShellUtils.tokenize(options, optionString);
374     } catch (TokenizationException e) {
375       throw new FuncallException(e.getMessage() + " while tokenizing '" + optionString + "'");
376     }
377     return ImmutableList.copyOf(options);
378   }

380   @SkylarkCallable(doc =
381       "Expands all references to labels embedded within a string for all files using a mapping "
382     + "from definition labels (i.e. the label in the output type attribute) to files. Deprecated.",
383       documented = false)
384   public String expand(@Nullable String expression,
385       List<Artifact> artifacts, Label labelResolver) throws FuncallException {
386     try {
387       Map<Label, Iterable<Artifact>> labelMap = new HashMap<>();
388       for (Artifact artifact : artifacts) {
389         labelMap.put(artifactLabelMap.get(artifact), ImmutableList.of(artifact));
390       }
391       return LabelExpander.expand(expression, labelMap, labelResolver);
392     } catch (NotUniqueExpansionException e) {
393       throw new FuncallException(e.getMessage() + " while expanding '" + expression + "'");
394     }
395   }

439   @SkylarkCallable(documented = false)
440   public boolean checkPlaceholders(String template, List<String> allowedPlaceholders) {
441     List<String> actualPlaceHolders = new LinkedList<>();
442     Set<String> allowedPlaceholderSet = ImmutableSet.copyOf(allowedPlaceholders);
443     ImplicitOutputsFunction.createPlaceholderSubstitutionFormatString(template, actualPlaceHolders);
444     for (String placeholder : actualPlaceHolders) {
445       if (!allowedPlaceholderSet.contains(placeholder)) {
446         return false;
447       }
448     }
449     return true;
450   }

452   @SkylarkCallable(doc =
453         "Returns a string after expanding all references to \"Make variables\". The variables "
454       + "must have the following format: <code>$(VAR_NAME)</code>. Also, <code>$$VAR_NAME"
455       + "</code> expands to <code>$VAR_NAME</code>. Parameters:"
456       + "<ul><li>The name of the attribute (<code>string</code>). It's only used for error "
457       + "reporting.</li>\n"
458       + "<li>The expression to expand (<code>string</code>). It can contain references to "
459       + "\"Make variables\".</li>\n"
460       + "<li>A mapping of additional substitutions (<code>dict</code> of <code>string</code> : "
461       + "<code>string</code>).</li></ul>\n"
462       + "Examples:"
463       + "<pre class=language-python>\n"
464       + "ctx.expand_make_variables(\"cmd\", \"$(MY_VAR)\", {\"MY_VAR\": \"Hi\"})  # == \"Hi\"\n"
465       + "ctx.expand_make_variables(\"cmd\", \"$$PWD\", {})  # == \"$PWD\"\n"
466       + "</pre>"
467       + "Additional variables may come from other places, such as configurations. Note that "
468       + "this function is experimental.")
469   public String expandMakeVariables(String attributeName, String command,
470       final Map<String, String> additionalSubstitutions) {
471     return ruleContext.expandMakeVariables(attributeName,
472         command, new ConfigurationMakeVariableContext(ruleContext.getRule().getPackage(),
473             ruleContext.getConfiguration()) {
474           @Override
475           public String lookupMakeVariable(String name) throws ExpansionException {
476             if (additionalSubstitutions.containsKey(name)) {
477               return additionalSubstitutions.get(name);
478             } else {
479               return super.lookupMakeVariable(name);
480             }
481           }
482         });
483   }
