34  /**
35   * Represents a target pattern. Target patterns are a generalization of labels to include
36   * wildcards for finding all packages recursively beneath some root, and for finding all targets
37   * within a package.
38   *
39   * <p>Note that this class does not handle negative patterns ("-//foo/bar"); these must be handled
40   * one level up. In particular, the query language comes with built-in support for negative
41   * patterns.
42   *
43   * <p>In order to resolve target patterns, you need an implementation of {@link
44   * TargetPatternResolver}. This class is thread-safe if the corresponding instance is thread-safe.
45   *
46   * <p>See lib/blaze/commands/target-syntax.txt for details.
47   */
48  public abstract class TargetPattern implements Serializable {
49  
50    private static final Splitter SLASH_SPLITTER = Splitter.on('/');
51    private static final Joiner SLASH_JOINER = Joiner.on('/');
52  
53    private static final Parser DEFAULT_PARSER = new Parser("");
54  
55    private final Type type;
56    private final String originalPattern;

59     * Returns a parser with no offset. Note that the Parser class is immutable, so this method may
60     * return the same instance on subsequent calls.

63      return DEFAULT_PARSER;
64    }

66    private static String removeSuffix(String s, String suffix) {
67      if (s.endsWith(suffix)) {
68        return s.substring(0, s.length() - suffix.length());
69      } else {
70        throw new IllegalArgumentException(s + ", " + suffix);
71      }
72    }

74    /**
75     * Normalizes the given relative path by resolving {@code //}, {@code /./} and {@code x/../}
76     * pieces. Note that leading {@code ".."} segments are not removed, so the returned string can
77     * have leading {@code ".."} segments.
78     *
79     * @throws IllegalArgumentException if the path is absolute, i.e. starts with a @{code '/'}
80     */
81    @VisibleForTesting
82    static String normalize(String path) {
83      Preconditions.checkArgument(!path.startsWith("/"));
84      Preconditions.checkArgument(!path.startsWith("@"));
85      Iterator<String> it = SLASH_SPLITTER.split(path).iterator();
86      List<String> pieces = new ArrayList<>();
87      while (it.hasNext()) {
88        String piece = it.next();
89        if (".".equals(piece) || piece.isEmpty()) {
90          continue;
91        }
92        if ("..".equals(piece)) {
93          if (pieces.isEmpty()) {
94            pieces.add(piece);
95            continue;
96          }
97          String predecessor = pieces.remove(pieces.size() - 1);
98          if ("..".equals(predecessor)) {
99            pieces.add(piece);
100           pieces.add(piece);
101         }
102         continue;
103       }
104       pieces.add(piece);
105     }
106     return SLASH_JOINER.join(pieces);
107   }

236   private static final class InterpretPathAsTarget extends TargetPattern {
237 
238     private final String path;
239 
240     private InterpretPathAsTarget(String path, String originalPattern) {
241       super(Type.PATH_AS_TARGET, originalPattern);
242       this.path = normalize(Preconditions.checkNotNull(path));
243     }
244 
245     @Override
246     public <T> ResolvedTargets<T> eval(TargetPatternResolver<T> resolver,
247         ImmutableSet<String> excludedSubdirectories)
248         throws TargetParsingException, InterruptedException {
249       Preconditions.checkArgument(excludedSubdirectories.isEmpty(),
250           "Target pattern \"%s\" of type %s cannot be evaluated with excluded subdirectories: %s.",
251           getOriginalPattern(), getType(), excludedSubdirectories);
252       if (resolver.isPackage(path)) {
253         // User has specified a package name. lookout for default target.
254         return resolver.getExplicitTarget("//" + path);
255       }
256 
257       List<String> pieces = SLASH_SPLITTER.splitToList(path);
258 
259       // Interprets the label as a file target.  This loop stops as soon as the
260       // first BUILD file is found (i.e. longest prefix match).
261       for (int i = pieces.size() - 1; i > 0; i--) {
262         String packageName = SLASH_JOINER.join(pieces.subList(0, i));
263         if (resolver.isPackage(packageName)) {
264           String targetName = SLASH_JOINER.join(pieces.subList(i, pieces.size()));
265           return resolver.getExplicitTarget("//" + packageName + ":" + targetName);
266         }
267       }
268 
269       throw new TargetParsingException("couldn't determine target from filename '" + path + "'");
270     }
271 
272     @Override
273     public boolean containsBelowDirectory(String directory) {
274       return false;
275     }
276 
277     @Override
278     public String getDirectory() {
279       int lastSlashIndex = path.lastIndexOf('/');
280       return lastSlashIndex < 0 ? "" : path.substring(0, lastSlashIndex);
281     }
282 
283     @Override
284     public boolean getRulesOnly() {
285       return false;
286     }
287 
288     @Override
289     public boolean equals(Object o) {
290       if (this == o) {
291         return true;
292       }
293       if (!(o instanceof InterpretPathAsTarget)) {
294         return false;
295       }
296       InterpretPathAsTarget that = (InterpretPathAsTarget) o;
297       return path.equals(that.path);
298     }
299 
300     @Override
301     public int hashCode() {
302       return Objects.hash(getType(), path);
303     }
304   }

306   private static final class TargetsInPackage extends TargetPattern {
307 
308     private final String pattern;
309     private final String suffix;
310     private final boolean isAbsolute;
311     private final boolean rulesOnly;
312     private final boolean checkWildcardConflict;
313 
314     private TargetsInPackage(String originalPattern, String pattern, String suffix,
315         boolean isAbsolute, boolean rulesOnly, boolean checkWildcardConflict) {
316       super(Type.TARGETS_IN_PACKAGE, originalPattern);
317       this.pattern = Preconditions.checkNotNull(pattern);
318       this.suffix = Preconditions.checkNotNull(suffix);
319       this.isAbsolute = isAbsolute;
320       this.rulesOnly = rulesOnly;
321       this.checkWildcardConflict = checkWildcardConflict;
322     }
323 
324     @Override
325     public <T> ResolvedTargets<T> eval(TargetPatternResolver<T> resolver,
326         ImmutableSet<String> excludedSubdirectories)
327         throws TargetParsingException, InterruptedException {
328       Preconditions.checkArgument(excludedSubdirectories.isEmpty(),
329           "Target pattern \"%s\" of type %s cannot be evaluated with excluded subdirectories: %s.",
330           getOriginalPattern(), getType(), excludedSubdirectories);
331       if (checkWildcardConflict) {
332         ResolvedTargets<T> targets = getWildcardConflict(resolver);
333         if (targets != null) {
334           return targets;
335         }
336       }
337       return resolver.getTargetsInPackage(getOriginalPattern(), removeSuffix(pattern, suffix),
338           rulesOnly);
339     }
340 
341     @Override
342     public boolean containsBelowDirectory(String directory) {
343       return false;
344     }
345 
346     @Override
347     public String getDirectory() {
348       return removeSuffix(pattern, suffix);
349     }
350 
351     @Override
352     public boolean getRulesOnly() {
353       return rulesOnly;
354     }
355 
356     @Override
357     public boolean equals(Object o) {
358       if (this == o) {
359         return true;
360       }
361       if (!(o instanceof TargetsInPackage)) {
362         return false;
363       }
364       TargetsInPackage that = (TargetsInPackage) o;
365       return isAbsolute == that.isAbsolute && rulesOnly == that.rulesOnly
366           && checkWildcardConflict == that.checkWildcardConflict
367           && getOriginalPattern().equals(that.getOriginalPattern())
368           && pattern.equals(that.pattern) && suffix.equals(that.suffix);
369     }
370 
371     @Override
372     public int hashCode() {
373       return Objects.hash(getType(), getOriginalPattern(), pattern, suffix, isAbsolute, rulesOnly,
374           checkWildcardConflict);
375     }
376 
377     /**
378      * There's a potential ambiguity if '//foo/bar:all' refers to an actual target. In this case, we
379      * use the the target but print a warning.
380      *
381      * @return the Target corresponding to the given pattern, if the pattern is absolute and there
382      *         is such a target. Otherwise, return null.
383      */
384     private <T> ResolvedTargets<T> getWildcardConflict(TargetPatternResolver<T> resolver)
385         throws InterruptedException {
386       if (!isAbsolute) {
387         return null;
388       }
389 
390       T target = resolver.getTargetOrNull("//" + pattern);
391       if (target != null) {
392         String name = pattern.lastIndexOf(':') != -1
393             ? pattern.substring(pattern.lastIndexOf(':') + 1)
394             : pattern.substring(pattern.lastIndexOf('/') + 1);
395         resolver.warn(String.format("The Blaze target pattern '%s' is ambiguous: '%s' is " +
396                                     "both a wildcard, and the name of an existing %s; " +
397                                     "using the latter interpretation",
398                                     "//" + pattern, ":" + name,
399                                     resolver.getTargetKind(target)));
400         try {
401           return resolver.getExplicitTarget("//" + pattern);
402         } catch (TargetParsingException e) {
403           throw new IllegalStateException(
404               "getTargetOrNull() returned non-null, so target should exist", e);
405         }
406       }
407       return null;
408     }
409   }

411   private static final class TargetsBelowDirectory extends TargetPattern {
412 
413     private final String directory;
414     private final boolean rulesOnly;
415 
416     private TargetsBelowDirectory(String originalPattern, String directory, boolean rulesOnly) {
417       super(Type.TARGETS_BELOW_DIRECTORY, originalPattern);
418       this.directory = Preconditions.checkNotNull(directory);
419       this.rulesOnly = rulesOnly;
420     }
421 
422     @Override
423     public <T> ResolvedTargets<T> eval(TargetPatternResolver<T> resolver,
424         ImmutableSet<String> excludedSubdirectories)
425         throws TargetParsingException, InterruptedException {
426       return resolver.findTargetsBeneathDirectory(getOriginalPattern(), directory, rulesOnly,
427           excludedSubdirectories);
428     }
429 
430     @Override
431     public boolean containsBelowDirectory(String containedDirectory) {
432       // Note that merely checking to see if the directory startsWith the TargetsBelowDirectory's
433       // directory is insufficient. "food" begins with "foo", but "//foo/..." does not contain
434       // "//food/...".
435       return directory.isEmpty() || (containedDirectory + "/").startsWith(directory + "/");
436     }
437 
438     @Override
439     public String getDirectory() {
440       return directory;
441     }
442 
443     @Override
444     public boolean getRulesOnly() {
445       return rulesOnly;
446     }
447 
448     @Override
449     public boolean equals(Object o) {
450       if (this == o) {
451         return true;
452       }
453       if (!(o instanceof TargetsBelowDirectory)) {
454         return false;
455       }
456       TargetsBelowDirectory that = (TargetsBelowDirectory) o;
457       return rulesOnly == that.rulesOnly && getOriginalPattern().equals(that.getOriginalPattern())
458           && directory.equals(that.directory);
459     }
460 
461     @Override
462     public int hashCode() {
463       return Objects.hash(getType(), getOriginalPattern(), directory, rulesOnly);
464     }

467   @Immutable
468   public static final class Parser {
469     // TODO(bazel-team): Merge the Label functionality that requires similar constants into this
470     // class.
471     /**
472      * The set of target-pattern suffixes which indicate wildcards over all <em>rules</em> in a
473      * single package.
474      */
475     private static final List<String> ALL_RULES_IN_SUFFIXES = ImmutableList.of(
476         "all");
477 
478     /**
479      * The set of target-pattern suffixes which indicate wildcards over all <em>targets</em> in a
480      * single package.
481      */
482     private static final List<String> ALL_TARGETS_IN_SUFFIXES = ImmutableList.of(
483         "*",
484         "all-targets");
485 
486     private static final List<String> SUFFIXES;
487 
488     static {
489       SUFFIXES = ImmutableList.<String>builder()
490           .addAll(ALL_RULES_IN_SUFFIXES)
491           .addAll(ALL_TARGETS_IN_SUFFIXES)
492           .add("/...")
493           .build();
494     }
495 
496     /**
497      * Returns whether the given pattern is simple, i.e., not starting with '-' and using none of
498      * the target matching suffixes.
499      */
500     public static boolean isSimpleTargetPattern(String pattern) {
501       if (pattern.startsWith("-")) {
502         return false;
503       }
504 
505       for (String suffix : SUFFIXES) {
506         if (pattern.endsWith(":" + suffix)) {
507           return false;
508         }
509       }
510       return true;
511     }
512 
513     /**
514      * Directory prefix to use when resolving relative labels (rather than absolute ones). For
515      * example, if the working directory is "<workspace root>/foo", then this should be "foo",
516      * which will make patterns such as "bar:bar" be resolved as "//foo/bar:bar". This makes the
517      * command line a bit more convenient to use.
518      */
519     private final String relativeDirectory;
520 
521     /**
522      * Creates a new parser with the given offset for relative patterns.
523      */
524     public Parser(String relativeDirectory) {
525       this.relativeDirectory = relativeDirectory;
526     }
527 
528     /**
529      * Parses the given pattern, and throws an exception if the pattern is invalid.
530      *
531      * @return a target pattern corresponding to the pattern parsed
532      * @throws TargetParsingException if the pattern is invalid
533      */
534     public TargetPattern parse(String pattern) throws TargetParsingException {
535       // The structure of this method is by cases, according to the usage string
536       // constant (see lib/blaze/commands/target-syntax.txt).
537 
538       String originalPattern = pattern;
539       final boolean includesRepo = pattern.startsWith("@");
540       String repoName = "";
541       if (includesRepo) {
542         int pkgStart = pattern.indexOf("//");
543         if (pkgStart < 0) {
544           throw new TargetParsingException("Couldn't find package in target " + pattern);
545         }
546         repoName = pattern.substring(0, pkgStart);
547         pattern = pattern.substring(pkgStart);
548       }
549       final boolean isAbsolute = pattern.startsWith("//");
550 
551       // We now absolutize non-absolute target patterns.
552       pattern = isAbsolute ? pattern.substring(2) : absolutize(pattern);
553       // Check for common errors.
554       if (pattern.startsWith("/")) {
555         throw new TargetParsingException("not a relative path or label: '" + pattern + "'");
556       }
557       if (pattern.isEmpty()) {
558         throw new TargetParsingException("the empty string is not a valid target");
559       }
560 
561       // Transform "/BUILD" suffix into ":BUILD" to accept //foo/bar/BUILD
562       // syntax as a synonym to //foo/bar:BUILD.
563       if (pattern.endsWith("/BUILD")) {
564         pattern = pattern.substring(0, pattern.length() - 6) + ":BUILD";
565       }
566 
567       int colonIndex = pattern.lastIndexOf(':');
568       String packagePart = colonIndex < 0 ? pattern : pattern.substring(0, colonIndex);
569       String targetPart = colonIndex < 0 ? "" : pattern.substring(colonIndex + 1);
570 
571       if (packagePart.equals("...")) {
572         packagePart = "/...";  // special case this for easier parsing
573       }
574 
575       if (packagePart.endsWith("/")) {
576         throw new TargetParsingException("The package part of '" + originalPattern
577             + "' should not end in a slash");
578       }
579 
580       if (packagePart.endsWith("/...")) {
581         String realPackagePart = removeSuffix(packagePart, "/...");
582         if (targetPart.isEmpty() || ALL_RULES_IN_SUFFIXES.contains(targetPart)) {
583           return new TargetsBelowDirectory(originalPattern, realPackagePart, true);
584         } else if (ALL_TARGETS_IN_SUFFIXES.contains(targetPart)) {
585           return new TargetsBelowDirectory(originalPattern, realPackagePart, false);
586         }
587       }
588 
589       if (ALL_RULES_IN_SUFFIXES.contains(targetPart)) {
590         return new TargetsInPackage(
591             originalPattern, pattern, ":" + targetPart, isAbsolute, true, true);
592       }
593 
594       if (ALL_TARGETS_IN_SUFFIXES.contains(targetPart)) {
595         return new TargetsInPackage(
596             originalPattern, pattern, ":" + targetPart, isAbsolute, false, true);
597       }
598 
599 
600       if (includesRepo || isAbsolute || pattern.contains(":")) {
601         PackageAndTarget packageAndTarget;
602         String fullLabel = repoName + "//" + pattern;
603         try {
604           packageAndTarget = LabelValidator.validateAbsoluteLabel(fullLabel);
605         } catch (BadLabelException e) {
606           String error = "invalid target format '" + originalPattern + "': " + e.getMessage();
607           throw new TargetParsingException(error);
608         }
609         return new SingleTarget(fullLabel, packageAndTarget.getPackageName(), originalPattern);
610       }
611 
612       // This is a stripped-down version of interpretPathAsTarget that does no I/O.  We have a basic
613       // relative path. e.g. "foo/bar/Wiz.java". The strictest correct check we can do here (without
614       // I/O) is just to ensure that there is *some* prefix that is a valid package-name. It's
615       // sufficient to test the first segment. This is really a rather weak check; perhaps we should
616       // just eliminate it.
617       int slashIndex = pattern.indexOf('/');
618       String packageName = pattern;
619       if (slashIndex > 0) {
620         packageName = pattern.substring(0, slashIndex);
621       }
622       String errorMessage = LabelValidator.validatePackageName(packageName);
623       if (errorMessage != null) {
624         throw new TargetParsingException("Bad target pattern '" + originalPattern + "': " +
625             errorMessage);
626       }
627       return new InterpretPathAsTarget(pattern, originalPattern);
628     }
629 
630     /**
631      * Absolutizes the target pattern to the offset.
632      * Patterns starting with "/" are absolute and not modified.
633      *
634      * If the offset is "foo":
635      *   absolutize(":bar") --> "foo:bar"
636      *   absolutize("bar") --> "foo/bar"
637      *   absolutize("/biz/bar") --> "biz/bar" (absolute)
638      *   absolutize("biz:bar") --> "foo/biz:bar"
639      *
640      * @param pattern The target pattern to parse.
641      * @return the pattern, absolutized to the offset if approprate.
642      */
643     private String absolutize(String pattern) {
644       if (relativeDirectory.isEmpty() || pattern.startsWith("/")) {
645         return pattern;
646       }
647 
648       // It seems natural to use {@link PathFragment#getRelative()} here,
649       // but it doesn't work when the pattern starts with ":".
650       // "foo".getRelative(":all") would return "foo/:all", where we
651       // really want "foo:all".
652       return pattern.startsWith(":")
653           ? relativeDirectory + pattern
654           : relativeDirectory + "/" + pattern;
655     }
656   }

658   /**
659    * The target pattern type (targets below package, in package, explicit target, etc.)
660    */
661   public enum Type {
662     /** A path interpreted as a target, eg "foo/bar/baz" */
663     PATH_AS_TARGET,
664     /** An explicit target, eg "//foo:bar." */
665     SINGLE_TARGET,
666     /** Targets below a directory, eg "foo/...". */
667     TARGETS_BELOW_DIRECTORY,
668     /** Target in a package, eg "foo:all". */
669     TARGETS_IN_PACKAGE;
670   }
