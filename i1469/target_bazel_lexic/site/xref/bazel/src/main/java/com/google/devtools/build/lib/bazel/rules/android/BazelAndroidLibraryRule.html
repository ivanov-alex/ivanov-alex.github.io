15  package com.google.devtools.build.lib.bazel.rules.android;

17  import com.google.devtools.build.lib.analysis.RuleDefinition;
18  import com.google.devtools.build.lib.analysis.RuleDefinitionEnvironment;
19  import com.google.devtools.build.lib.bazel.rules.java.BazelJavaRuleClasses;
20  import com.google.devtools.build.lib.packages.RuleClass;
21  import com.google.devtools.build.lib.packages.RuleClass.Builder;
22  import com.google.devtools.build.lib.rules.android.AndroidLibraryBaseRule;
23  import com.google.devtools.build.lib.rules.android.AndroidRuleClasses;
24  
25  /**
26   * Definition of the {@code android_library} rule for Bazel.
27   */
28  public class BazelAndroidLibraryRule implements RuleDefinition {
29  
30    @Override
31    public RuleClass build(Builder builder, RuleDefinitionEnvironment env) {
32      return builder
33          /* <!-- #BLAZE_RULE(android_library).IMPLICIT_OUTPUTS -->
34          <ul>
35            <li><code>lib<var>name</var>.jar</code>: A Java archive.</li>
36            <li><code>lib<var>name</var>-src.jar</code>: An archive containing the
37            sources ("source jar").</li>
38            <li><code>lib<var>name</var>-gen.jar</code>: An archive containing the
39              class files for source files generated by annotation processors
40              ("gen jar").</li>
41          </ul>
42          <!-- #END_BLAZE_RULE.IMPLICIT_OUTPUTS --> */
43          .setImplicitOutputsFunction(AndroidRuleClasses.ANDROID_LIBRARY_IMPLICIT_OUTPUTS)
44          .build();
45  
46    }
47  
48    @Override
49    public Metadata getMetadata() {
50      return RuleDefinition.Metadata.builder()
51          .name("android_library")
52          .ancestors(
53              BazelJavaRuleClasses.JavaBaseRule.class,
54              AndroidLibraryBaseRule.class)
55          .factoryClass(BazelAndroidLibrary.class)
56          .build();
57    }
58  }
59
