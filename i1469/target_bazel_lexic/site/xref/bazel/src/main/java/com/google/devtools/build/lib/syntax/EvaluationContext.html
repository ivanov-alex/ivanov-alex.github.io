21  import com.google.devtools.build.lib.events.EventHandler;

28  import java.util.List;

30  import javax.annotation.Nullable;

32  /**
33   * Context for the evaluation of programs.
34   */
35  public final class EvaluationContext {
36  
37    @Nullable private EventHandler eventHandler;
38    private Environment env;
39    @Nullable private ValidationEnvironment validationEnv;
40    private boolean parsePython;
41  
42    private EvaluationContext(EventHandler eventHandler, Environment env,
43        @Nullable ValidationEnvironment validationEnv, boolean parsePython) {
44      this.eventHandler = eventHandler;
45      this.env = env;
46      this.validationEnv = validationEnv;
47      this.parsePython = parsePython;
48    }
49  
50    /**
51     * The fail fast handler, which throws a runtime exception whenever we encounter an error event.
52     */
53    public static final EventHandler FAIL_FAST_HANDLER = new EventHandler() {
54        @Override
55        public void handle(Event event) {
56          Preconditions.checkArgument(
57              !EventKind.ERRORS_AND_WARNINGS.contains(event.getKind()), event);
58        }
59      };
60  
61    public static EvaluationContext newBuildContext(EventHandler eventHandler, Environment env,
62        boolean parsePython) {
63      return new EvaluationContext(eventHandler, env, null, parsePython);
64    }
65  
66    public static EvaluationContext newBuildContext(EventHandler eventHandler, Environment env) {
67      return newBuildContext(eventHandler, env, false);
68    }
69  
70    public static EvaluationContext newBuildContext(EventHandler eventHandler) {
71      return newBuildContext(eventHandler, new Environment());
72    }
73  
74    public static EvaluationContext newSkylarkContext(
75        Environment env, ValidationEnvironment validationEnv) {
76      return new EvaluationContext(env.getEventHandler(), env, validationEnv, false);
77    }
78  
79    public static EvaluationContext newSkylarkContext(EventHandler eventHandler) {
80      return newSkylarkContext(new SkylarkEnvironment(eventHandler), new ValidationEnvironment());
81    }
82  
83    /** Base context for Skylark evaluation for internal use only, while initializing builtins */
84    static final EvaluationContext SKYLARK_INITIALIZATION = newSkylarkContext(FAIL_FAST_HANDLER);
85  
86    @VisibleForTesting
87    public Environment getEnvironment() {
88      return env;
89    }
90  
91    /** Mock package locator */
92    private static final class EmptyPackageLocator implements CachingPackageLocator {
93      @Override
94      public Path getBuildFileForPackage(PackageIdentifier packageName) {
95        return null;
96      }
97    }
98  
99    /** An empty package locator */
100   private static final CachingPackageLocator EMPTY_PACKAGE_LOCATOR = new EmptyPackageLocator();
101 
102   /** Create a Lexer without a supporting file */
103   @VisibleForTesting
104   Lexer createLexer(String... input) {
105     return new Lexer(ParserInputSource.create(Joiner.on("\n").join(input), null),
106         eventHandler);
107   }
108 
109   /** Is this a Skylark evaluation context? */
110   public boolean isSkylark() {
111     return env.isSkylark();
112   }
113 
114   /** Parse a string without a supporting file, returning statements and comments */
115   @VisibleForTesting
116   Parser.ParseResult parseFileWithComments(String... input) {
117     return isSkylark()
118         ? Parser.parseFileForSkylark(createLexer(input), eventHandler, null, validationEnv)
119         : Parser.parseFile(createLexer(input), eventHandler, EMPTY_PACKAGE_LOCATOR, parsePython);
120   }
121 
122   /** Parse a string without a supporting file, returning statements only */
123   @VisibleForTesting
124   List<Statement> parseFile(String... input) {
125     return parseFileWithComments(input).statements;
126   }
127 
128   /** Parse an Expression from string without a supporting file */
129   @VisibleForTesting
130   Expression parseExpression(String... input) {
131     return Parser.parseExpression(createLexer(input), eventHandler);
132   }
133 
134   /** Evaluate an Expression */
135   @VisibleForTesting
136   Object evalExpression(Expression expression) throws EvalException, InterruptedException {
137     return expression.eval(env);
138   }
139 
140   /** Evaluate an Expression as parsed from String-s */
141   Object evalExpression(String... input) throws EvalException, InterruptedException {
142     return evalExpression(parseExpression(input));
143   }
144 
145   /** Parse a build (not Skylark) Statement from string without a supporting file */
146   @VisibleForTesting
147   Statement parseStatement(String... input) {
148     return Parser.parseStatement(createLexer(input), eventHandler);
149   }
150 
151   /**
152    * Evaluate a Statement
153    * @param statement the Statement
154    * @return the value of the evaluation, if it's an Expression, or else null
155    */
156   @Nullable private Object eval(Statement statement) throws EvalException, InterruptedException {
157     if (statement instanceof ExpressionStatement) {
158       return evalExpression(((ExpressionStatement) statement).getExpression());
159     }
160     statement.exec(env);
161     return null;
162   }
163 
164   /**
165    * Evaluate a list of Statement-s
166    * @return the value of the last statement if it's an Expression or else null
167    */
168   @Nullable private Object eval(List<Statement> statements)
169       throws EvalException, InterruptedException {
170     Object last = null;
171     for (Statement statement : statements) {
172       last = eval(statement);
173     }
174     return last;
175   }
176 
177   /** Update a variable in the environment, in fluent style */
178   public EvaluationContext update(String varname, Object value) throws EvalException {
179     env.update(varname, value);
180     if (validationEnv != null) {
181       validationEnv.declare(varname, null);
182     }
183     return this;
184   }
185 
186   /** Lookup a variable in the environment */
187   public Object lookup(String varname) throws NoSuchVariableException {
188     return env.lookup(varname);
189   }
190 
191   /** Evaluate a series of statements */
192   public Object eval(String... input) throws EvalException, InterruptedException {
193     return eval(parseFile(input));
194   }
195 }
