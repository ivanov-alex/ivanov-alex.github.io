218   /**
219    * Executes the action without handling ExecException errors.
220    *
221    * <p>Called by {@link #execute}.
222    */
223   protected void internalExecute(
224       ActionExecutionContext actionExecutionContext) throws ExecException, InterruptedException {
225     getContext(actionExecutionContext.getExecutor()).exec(getSpawn(), actionExecutionContext);
226   }
227 
228   @Override
229   public void execute(ActionExecutionContext actionExecutionContext)
230       throws ActionExecutionException, InterruptedException {
231     Executor executor = actionExecutionContext.getExecutor();
232     try {
233       internalExecute(actionExecutionContext);
234     } catch (ExecException e) {
235       String failMessage = progressMessage;
236       if (isShellCommand()) {
237         // The possible reasons it could fail are: shell executable not found, shell
238         // exited non-zero, or shell died from signal.  The first is impossible
239         // and the second two aren't very interesting, so in the interests of
240         // keeping the noise-level down, we don't print a reason why, just the
241         // command that failed.
242         //
243         // 0=shell executable, 1=shell command switch, 2=command
244         failMessage = "error executing shell command: " + "'"
245             + truncate(Iterables.get(argv.arguments(), 2), 200) + "'";
246       }
247       throw e.toActionExecutionException(failMessage, executor.getVerboseFailures(), this);
248     }
249   }

255   /**
256    * Returns s, truncated to no more than maxLen characters, appending an
257    * ellipsis if truncation occurred.
258    */
259   private static String truncate(String s, int maxLen) {
260     return s.length() > maxLen
261         ? s.substring(0, maxLen - "...".length()) + "..."
262         : s;
263   }

273   @Override
274   protected String computeKey() {
275     Fingerprint f = new Fingerprint();
276     f.addString(GUID);
277     f.addStrings(argv.arguments());
278     f.addString(getMnemonic());
279     f.addInt(inputManifests.size());
280     for (Map.Entry<PathFragment, Artifact> input : inputManifests.entrySet()) {
281       f.addString(input.getKey().getPathString() + "/");
282       f.addPath(input.getValue().getExecPath());
283     }
284     f.addStringMap(getEnvironment());
285     f.addStringMap(getExecutionInfo());
286     return f.hexDigestAndReset();
287   }

289   @Override
290   public String describeKey() {
291     StringBuilder message = new StringBuilder();
292     message.append(getProgressMessage());
293     message.append('\n');
294     for (Map.Entry<String, String> entry : getEnvironment().entrySet()) {
295       message.append("  Environment variable: ");
296       message.append(ShellEscaper.escapeString(entry.getKey()));
297       message.append('=');
298       message.append(ShellEscaper.escapeString(entry.getValue()));
299       message.append('\n');
300     }
301     for (String argument : ShellEscaper.escapeAll(argv.arguments())) {
302       message.append("  Argument: ");
303       message.append(argument);
304       message.append('\n');
305     }
306     return message.toString();
307   }

474     /**
475      * Builds the SpawnAction using the passed in action configuration and returns it and all
476      * dependent actions. The first item of the returned array is always the SpawnAction itself.
477      *
478      * <p>This method makes a copy of all the collections, so it is safe to reuse the builder after
479      * this method returns.
480      *
481      * <p>This is annotated with @CheckReturnValue, which causes a compiler error when you call this
482      * method and ignore its return value. This is because some time ago, calling .build() had the
483      * side-effect of registering it with the RuleContext that was passed in to the constructor.
484      * This logic was removed, but if people don't notice and still rely on the side-effect, things
485      * may break.
486      *
487      * @return the SpawnAction and any actions required by it, with the first item always being the
488      *      SpawnAction itself.
489      */
490     @CheckReturnValue
491     public Action[] build(ActionConstructionContext context) {
492       return build(context.getActionOwner(), context.getAnalysisEnvironment(),
493           context.getConfiguration());
494     }

496     @VisibleForTesting @CheckReturnValue
497     public Action[] build(ActionOwner owner, AnalysisEnvironment analysisEnvironment,
498         BuildConfiguration configuration) {
499       if (isShellCommand && executable == null) {
500         executable = configuration.getShExecutable();
501       }
502       Preconditions.checkNotNull(executable);
503       Preconditions.checkNotNull(executableArgs);
504 
505       if (useDefaultShellEnvironment) {
506         this.environment = configuration.getDefaultShellEnvironment();
507       }
508 
509       ImmutableList<String> argv = ImmutableList.<String>builder()
510           .add(executable.getPathString())
511           .addAll(executableArgs)
512           .build();
513 
514       Iterable<String> arguments = argumentsBuilder.build();
515 
516       Artifact paramsFile = ParamFileHelper.getParamsFileMaybe(argv, arguments, commandLine,
517           paramFileInfo, configuration, analysisEnvironment, outputs);
518 
519       List<Action> actions = new ArrayList<>();
520       CommandLine actualCommandLine;
521       if (paramsFile != null) {
522         actualCommandLine = ParamFileHelper.createWithParamsFile(argv, arguments, commandLine,
523             isShellCommand, owner, actions, paramFileInfo, paramsFile);
524       } else {
525         actualCommandLine = ParamFileHelper.createWithoutParamsFile(argv, arguments, commandLine,
526             isShellCommand);
527       }
528 
529       Iterable<Artifact> actualInputs = collectActualInputs(paramsFile);
530 
531       actions.add(0, new SpawnAction(owner, actualInputs, ImmutableList.copyOf(outputs),
532           resourceSet, actualCommandLine, environment, executionInfo, progressMessage,
533           ImmutableMap.copyOf(inputManifests), mnemonic, executeUnconditionally,
534           extraActionInfoSupplier));
535       return actions.toArray(new Action[actions.size()]);
536     }

539       if (parameterFile != null) {
540         inputsBuilder.add(parameterFile);
541       }
542       return inputsBuilder.build();
543     }

689         List<String> jvmArgs, String... launchArgs) {
690       this.executable = javaExecutable;
691       this.executableArgs = Lists.newArrayList();
692       executableArgs.add("-Xverify:none");
693       executableArgs.addAll(jvmArgs);
694       Collections.addAll(executableArgs, launchArgs);
695       inputsBuilder.add(deployJar);
696       this.isShellCommand = false;
697       return this;
698     }

731      * Sets the executable to be the shell and adds the given command as the
732      * command to be executed.
733      *
734      * <p>Note that this will not clear the arguments, so any arguments will
735      * be passed in addition to the command given here.
736      *
737      * <p>Calling this method overrides any previous values set via calls to
738      * {@link #setExecutable(Artifact)}, {@link #setJavaExecutable}, or
739      * {@link #setShellCommand(String)}.
740      */
741     public Builder setShellCommand(String command) {
742       this.executable = null;
743       // 0=shell command switch, 1=command
744       this.executableArgs = Lists.newArrayList("-c", command);
745       this.isShellCommand = true;
746       return this;
747     }

749     /**
750      * Sets the executable to be the shell and adds the given interned commands as the
751      * commands to be executed.
752      */
753     public Builder setShellCommand(Iterable<String> command) {
754       this.executable = new PathFragment(Iterables.getFirst(command, null));
755       // The first item of the commands is the shell executable that should be used.
756       this.executableArgs = ImmutableList.copyOf(Iterables.skip(command, 1));
757       this.isShellCommand = true;
758       return this;
