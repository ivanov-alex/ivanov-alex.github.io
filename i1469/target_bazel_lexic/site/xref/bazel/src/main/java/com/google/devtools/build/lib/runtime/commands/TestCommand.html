47  /**
48   * Handles the 'test' command on the Blaze command line.
49   */
50  @Command(name = "test",
51           builds = true,
52           inherits = { BuildCommand.class },
53           options = { TestSummaryOptions.class },
54           shortDescription = "Builds and runs the specified test targets.",
55           help = "resource:test.txt",
56           completion = "label-test",
57           allowResidue = true)
58  public class TestCommand implements BlazeCommand {
59    private AnsiTerminalPrinter printer;
60  
61    /** Returns the name of the command to ask the project file for. */
62    // TODO(hdm): move into BlazeRuntime?  It feels odd to duplicate the annotation here.
63    protected String commandName() {
64      return "test";
65    }
66  
67    @Override
68    public void editOptions(BlazeRuntime runtime, OptionsParser optionsParser)
69        throws AbruptExitException {
70      ProjectFileSupport.handleProjectFiles(runtime, optionsParser, commandName());
71  
72      TestOutputFormat testOutput = optionsParser.getOptions(ExecutionOptions.class).testOutput;
73  
74      try {
75        if (testOutput == TestStrategy.TestOutputFormat.STREAMED) {
76          runtime.getReporter().handle(Event.warn(
77              "Streamed test output requested so all tests will be run locally, without sharding, " +
78               "one at a time"));
79          optionsParser.parse(OptionPriority.SOFTWARE_REQUIREMENT,
80              "streamed output requires locally run tests, without sharding",
81              ImmutableList.of("--test_sharding_strategy=disabled", "--test_strategy=exclusive"));
82        }
83  
84        if (optionsParser.getOptions(BuildConfiguration.Options.class).collectCodeCoverage) {
85          optionsParser.parse(OptionPriority.SOFTWARE_REQUIREMENT,
86              "baseline coverage artifacts are built with running tests with coverage collection",
87              ImmutableList.of("--output_groups=" + OutputGroupProvider.BASELINE_COVERAGE));
88        }
89      } catch (OptionsParsingException e) {
90        throw new IllegalStateException("Known options failed to parse", e);
91      }
92    }
93  
94    @Override
95    public ExitCode exec(BlazeRuntime runtime, OptionsProvider options) {
96      TestResultAnalyzer resultAnalyzer = new TestResultAnalyzer(
97          runtime.getExecRoot(),
98          options.getOptions(TestSummaryOptions.class),
99          options.getOptions(ExecutionOptions.class),
100         runtime.getEventBus());
101 
102     printer = new AnsiTerminalPrinter(runtime.getReporter().getOutErr().getOutputStream(),
103         options.getOptions(BlazeCommandEventHandler.Options.class).useColor());
104 
105     // Initialize test handler.
106     AggregatingTestListener testListener = new AggregatingTestListener(
107         resultAnalyzer, runtime.getEventBus(), runtime.getReporter());
108 
109     runtime.getEventBus().register(testListener);
110     return doTest(runtime, options, testListener);
111   }
112 
113   private ExitCode doTest(BlazeRuntime runtime,
114       OptionsProvider options,
115       AggregatingTestListener testListener) {
116     // Run simultaneous build and test.
117     List<String> targets = ProjectFileSupport.getTargets(runtime, options);
118     BuildRequest request = BuildRequest.create(
119         getClass().getAnnotation(Command.class).name(), options,
120         runtime.getStartupOptionsProvider(), targets,
121         runtime.getReporter().getOutErr(), runtime.getCommandId(), runtime.getCommandStartTime());
122     request.setRunTests();
123 
124     BuildResult buildResult = runtime.getBuildTool().processRequest(request, null);
125 
126     Collection<ConfiguredTarget> testTargets = buildResult.getTestTargets();
127     // TODO(bazel-team): don't handle isEmpty here or fix up a bunch of tests
128     if (buildResult.getSuccessfulTargets() == null) {
129       // This can happen if there were errors in the target parsing or loading phase
130       // (original exitcode=BUILD_FAILURE) or if there weren't but --noanalyze was given
131       // (original exitcode=SUCCESS).
132       runtime.getReporter().handle(Event.error("Couldn't start the build. Unable to run tests"));
133       return buildResult.getSuccess() ? ExitCode.PARSING_FAILURE : buildResult.getExitCondition();
134     }
135     // TODO(bazel-team): the check above shadows NO_TESTS_FOUND, but switching the conditions breaks
136     // more tests
137     if (testTargets.isEmpty()) {
138       runtime.getReporter().handle(Event.error(
139           null, "No test targets were found, yet testing was requested"));
140       return buildResult.getSuccess() ? ExitCode.NO_TESTS_FOUND : buildResult.getExitCondition();
141     }
142 
143     boolean buildSuccess = buildResult.getSuccess();
144     boolean testSuccess = analyzeTestResults(testTargets, testListener, options);
145 
146     if (testSuccess && !buildSuccess) {
147       // If all tests run successfully, test summary should include warning if
148       // there were build errors not associated with the test targets.
149       printer.printLn(AnsiTerminalPrinter.Mode.ERROR
150           + "One or more non-test targets failed to build.\n"
151           + AnsiTerminalPrinter.Mode.DEFAULT);
152     }
153 
154     return buildSuccess ?
155            (testSuccess ? ExitCode.SUCCESS : ExitCode.TESTS_FAILED)
156            : buildResult.getExitCondition();
157   }
158 
159   /**
160    * Analyzes test results and prints summary information.
161    * Returns true if and only if all tests were successful.
162    */
163   private boolean analyzeTestResults(Collection<ConfiguredTarget> testTargets,
164                                      AggregatingTestListener listener,
165                                      OptionsProvider options) {
166     TestResultNotifier notifier = new TerminalTestResultNotifier(printer, options);
167     return listener.getAnalyzer().differentialAnalyzeAndReport(
168         testTargets, listener, notifier);
169   }
