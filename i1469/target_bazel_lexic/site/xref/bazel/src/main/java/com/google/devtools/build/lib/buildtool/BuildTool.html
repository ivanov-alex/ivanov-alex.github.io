89  /**
90   * Provides the bulk of the implementation of the 'blaze build' command.
91   *
92   * <p>The various concrete build command classes handle the command options and request
93   * setup, then delegate the handling of the request (the building of targets) to this class.
94   *
95   * <p>The main entry point is {@link #buildTargets}.
96   *
97   * <p>This class is always instantiated and managed as a singleton, being constructed and held by
98   * {@link BlazeRuntime}. This is so multiple kinds of build commands can share this single
99   * instance.
100  *
101  * <p>Most of analysis is handled in {@link BuildView}, and execution in {@link ExecutionTool}.
102  */
103 public class BuildTool {

118   /**
119    * The crux of the build system. Builds the targets specified in the request using the specified
120    * Executor.
121    *
122    * <p>Performs loading, analysis and execution for the specified set of targets, honoring the
123    * configuration options in the BuildRequest. Returns normally iff successful, throws an exception
124    * otherwise.
125    *
126    * <p>Callers must ensure that {@link #stopRequest} is called after this method, even if it
127    * throws.
128    *
129    * <p>The caller is responsible for setting up and syncing the package cache.
130    *
131    * <p>During this function's execution, the actualTargets and successfulTargets
132    * fields of the request object are set.
133    *
134    * @param request the build request that this build tool is servicing, which specifies various
135    *        options; during this method's execution, the actualTargets and successfulTargets fields
136    *        of the request object are populated
137    * @param result the build result that is the mutable result of this build
138    * @param validator target validator
139    */
140   public void buildTargets(BuildRequest request, BuildResult result, TargetValidator validator)
141       throws BuildFailedException, LocalEnvironmentException,
142              InterruptedException, ViewCreationFailedException,
143              TargetParsingException, LoadingFailedException, ExecutorInitException,
144              AbruptExitException, InvalidConfigurationException, TestExecException {
145     validateOptions(request);
146     BuildOptions buildOptions = runtime.createBuildOptions(request);
147     // Sync the package manager before sending the BuildStartingEvent in runLoadingPhase()
148     runtime.setupPackageCache(request.getPackageCacheOptions(),
149         DefaultsPackage.getDefaultsPackageContent(buildOptions));
150 
151     ExecutionTool executionTool = null;
152     LoadingResult loadingResult = null;
153     BuildConfigurationCollection configurations = null;
154     try {
155       getEventBus().post(new BuildStartingEvent(runtime.getOutputFileSystem(), request));
156       LOG.info("Build identifier: " + request.getId());
157       executionTool = new ExecutionTool(runtime, request);
158       if (needsExecutionPhase(request.getBuildOptions())) {
159         // Initialize the execution tool early if we need it. This hides the latency of setting up
160         // the execution backends.
161         executionTool.init();
162       }
163 
164       // Loading phase.
165       loadingResult = runLoadingPhase(request, validator);
166 
167       // Create the build configurations.
168       if (!request.getMultiCpus().isEmpty()) {
169         getReporter().handle(Event.warn(
170             "The --experimental_multi_cpu option is _very_ experimental and only intended for "
171             + "internal testing at this time. If you do not work on the build tool, then you "
172             + "should stop now!"));
173         if (!"build".equals(request.getCommandName()) && !"test".equals(request.getCommandName())) {
174           throw new InvalidConfigurationException(
175               "The experimental setting to select multiple CPUs is only supported for 'build' and "
176               + "'test' right now!");
177         }
178       }
179       configurations = getConfigurations(buildOptions, request.getMultiCpus(),
180           request.getViewOptions().keepGoing);
181 
182       getEventBus().post(new ConfigurationsCreatedEvent(configurations));
183       runtime.throwPendingException();
184       if (configurations.getTargetConfigurations().size() == 1) {
185         // TODO(bazel-team): This is not optimal - we retain backwards compatibility in the case
186         // where there's only a single configuration, but we don't send an event in the multi-config
187         // case. Can we do better? [multi-config]
188         getEventBus().post(new MakeEnvironmentEvent(
189             configurations.getTargetConfigurations().get(0).getMakeEnvironment()));
190       }
191       LOG.info("Configurations created");
192 
193       // Analysis phase.
194       AnalysisResult analysisResult = runAnalysisPhase(request, loadingResult, configurations);
195       result.setActualTargets(analysisResult.getTargetsToBuild());
196       result.setTestTargets(analysisResult.getTargetsToTest());
197 
198       checkTargetEnvironmentRestrictions(analysisResult.getTargetsToBuild(),
199           runtime.getPackageManager());
200       reportTargets(analysisResult);
201 
202       // Execution phase.
203       if (needsExecutionPhase(request.getBuildOptions())) {
204         runtime.getSkyframeExecutor().injectTopLevelContext(request.getTopLevelArtifactContext());
205         executionTool.executeBuild(request.getId(), analysisResult, result,
206             runtime.getSkyframeExecutor(), configurations,
207             mergePackageRoots(loadingResult.getPackageRoots(),
208             runtime.getSkyframeExecutor().getPackageRoots()));
209       }
210 
211       String delayedErrorMsg = analysisResult.getError();
212       if (delayedErrorMsg != null) {
213         throw new BuildFailedException(delayedErrorMsg);
214       }
215     } catch (RuntimeException e) {
216       // Print an error message for unchecked runtime exceptions. This does not concern Error
217       // subclasses such as OutOfMemoryError.
218       request.getOutErr().printErrLn("Unhandled exception thrown during build; message: " +
219           e.getMessage());
220       throw e;
221     } finally {
222       // Delete dirty nodes to ensure that they do not accumulate indefinitely.
223       long versionWindow = request.getViewOptions().versionWindowForDirtyNodeGc;
224       if (versionWindow != -1) {
225         runtime.getSkyframeExecutor().deleteOldNodes(versionWindow);
226       }
227 
228       if (executionTool != null) {
229         executionTool.shutdown();
230       }
231       // The workspace status actions will not run with certain flags, or if an error
232       // occurs early in the build. Tell a lie so that the event is not missing.
233       // If multiple build_info events are sent, only the first is kept, so this does not harm
234       // successful runs (which use the workspace status action).
235       getEventBus().post(new BuildInfoEvent(
236           runtime.getworkspaceStatusActionFactory().createDummyWorkspaceStatus()));
237     }
238 
239     if (loadingResult != null && loadingResult.hasTargetPatternError()) {
240       throw new BuildFailedException("execution phase successful, but there were errors " +
241                                      "parsing the target pattern");
242     }
243   }

245   /**
246    * Checks that if this is an environment-restricted build, all top-level targets support
247    * the expected environments.
248    *
249    * @param topLevelTargets the build's top-level targets
250    * @throws ViewCreationFailedException if constraint enforcement is on, the build declares
251    *     environment-restricted top level configurations, and any top-level target doesn't
252    *     support the expected environments
253    */
254   private void checkTargetEnvironmentRestrictions(Iterable<ConfiguredTarget> topLevelTargets,
255       PackageManager packageManager) throws ViewCreationFailedException {
256     for (ConfiguredTarget topLevelTarget : topLevelTargets) {
257       BuildConfiguration config = topLevelTarget.getConfiguration();
258       if (config == null) {
259         // TODO(bazel-team): support file targets (they should apply package-default constraints).
260         continue;
261       } else if (!config.enforceConstraints() || config.getTargetEnvironments().isEmpty()) {
262         continue;
263       }
264 
265       // Parse and collect this configuration's environments.
266       EnvironmentCollection.Builder builder = new EnvironmentCollection.Builder();
267       for (Label envLabel : config.getTargetEnvironments()) {
268         try {
269           Target env = packageManager.getLoadedTarget(envLabel);
270           builder.put(ConstraintSemantics.getEnvironmentGroup(env), envLabel);
271         } catch (NoSuchPackageException | NoSuchTargetException
272             | ConstraintSemantics.EnvironmentLookupException e) {
273           throw new ViewCreationFailedException("invalid target environment", e);
274         }
275       }
276       EnvironmentCollection expectedEnvironments = builder.build();
277 
278       // Now check the target against those environments.
279       SupportedEnvironmentsProvider provider =
280           Verify.verifyNotNull(topLevelTarget.getProvider(SupportedEnvironmentsProvider.class));
281         Collection<Label> missingEnvironments = ConstraintSemantics.getUnsupportedEnvironments(
282             provider.getEnvironments(), expectedEnvironments);
283         if (!missingEnvironments.isEmpty()) {
284           throw new ViewCreationFailedException(
285               String.format("This is a restricted-environment build. %s does not support"
286                   + " required environment%s %s",
287                   topLevelTarget.getLabel(),
288                   missingEnvironments.size() == 1 ? "" : "s",
289                   Joiner.on(", ").join(missingEnvironments)));
290         }
291     }
292   }

294   private ImmutableMap<PathFragment, Path> mergePackageRoots(
295       ImmutableMap<PackageIdentifier, Path> first,
296       ImmutableMap<PackageIdentifier, Path> second) {
297     Map<PathFragment, Path> builder = Maps.newHashMap();
298     for (Map.Entry<PackageIdentifier, Path> entry : first.entrySet()) {
299       builder.put(entry.getKey().getPathFragment(), entry.getValue());
300     }
301     for (Map.Entry<PackageIdentifier, Path> entry : second.entrySet()) {
302       if (first.containsKey(entry.getKey())) {
303         Preconditions.checkState(first.get(entry.getKey()).equals(entry.getValue()));
304       } else {
305         // This could overwrite entries from first in other repositories.
306         builder.put(entry.getKey().getPackageFragment(), entry.getValue());
307       }
308     }
309     return ImmutableMap.copyOf(builder);
310   }

317   /**
318    * The crux of the build system. Builds the targets specified in the request using the specified
319    * Executor.
320    *
321    * <p>Performs loading, analysis and execution for the specified set of targets, honoring the
322    * configuration options in the BuildRequest. Returns normally iff successful, throws an exception
323    * otherwise.
324    *
325    * <p>The caller is responsible for setting up and syncing the package cache.
326    *
327    * <p>During this function's execution, the actualTargets and successfulTargets
328    * fields of the request object are set.
329    *
330    * @param request the build request that this build tool is servicing, which specifies various
331    *        options; during this method's execution, the actualTargets and successfulTargets fields
332    *        of the request object are populated
333    * @param validator target validator
334    * @return the result as a {@link BuildResult} object
335    */
336   public BuildResult processRequest(BuildRequest request, TargetValidator validator) {
337     BuildResult result = new BuildResult(request.getStartTime());
338     runtime.getEventBus().register(result);
339     Throwable catastrophe = null;
340     ExitCode exitCode = ExitCode.BLAZE_INTERNAL_ERROR;
341     try {
342       buildTargets(request, result, validator);
343       exitCode = ExitCode.SUCCESS;
344     } catch (BuildFailedException e) {
345       if (e.isErrorAlreadyShown()) {
346         // The actual error has already been reported by the Builder.
347       } else {
348         reportExceptionError(e);
349       }
350       if (e.isCatastrophic()) {
351         result.setCatastrophe();
352       }
353       exitCode = ExitCode.BUILD_FAILURE;
354     } catch (InterruptedException e) {
355       exitCode = ExitCode.INTERRUPTED;
356       getReporter().handle(Event.error("build interrupted"));
357       getEventBus().post(new BuildInterruptedEvent());
358     } catch (TargetParsingException | LoadingFailedException | ViewCreationFailedException e) {
359       exitCode = ExitCode.PARSING_FAILURE;
360       reportExceptionError(e);
361     } catch (TestExecException e) {
362       // ExitCode.SUCCESS means that build was successful. Real return code of program
363       // is going to be calculated in TestCommand.doTest().
364       exitCode = ExitCode.SUCCESS;
365       reportExceptionError(e);
366     } catch (InvalidConfigurationException e) {
367       exitCode = ExitCode.COMMAND_LINE_ERROR;
368       reportExceptionError(e);
369     } catch (AbruptExitException e) {
370       exitCode = e.getExitCode();
371       reportExceptionError(e);
372       result.setCatastrophe();
373     } catch (Throwable throwable) {
374       catastrophe = throwable;
375       Throwables.propagate(throwable);
376     } finally {
377       stopRequest(request, result, catastrophe, exitCode);
378     }
379 
380     return result;
381   }

394   @VisibleForTesting
395   protected final LoadingResult runLoadingPhase(final BuildRequest request,
396                                                 final TargetValidator validator)
397           throws LoadingFailedException, TargetParsingException, InterruptedException,
398           AbruptExitException {
399     Profiler.instance().markPhase(ProfilePhase.LOAD);
400     runtime.throwPendingException();
401 
402     initializeOutputFilter(request);
403 
404     final boolean keepGoing = request.getViewOptions().keepGoing;
405 
406     Callback callback = new Callback() {
407       @Override
408       public void notifyTargets(Collection<Target> targets) throws LoadingFailedException {
409         if (validator != null) {
410           validator.validateTargets(targets, keepGoing);
411         }
412       }
413 
414       @Override
415       public void notifyVisitedPackages(Set<PackageIdentifier> visitedPackages) {
416         runtime.getSkyframeExecutor().updateLoadedPackageSet(visitedPackages);
417       }
418     };
419 
420     LoadingResult result = runtime.getLoadingPhaseRunner().execute(getReporter(),
421         getEventBus(), request.getTargets(), request.getLoadingOptions(),
422         runtime.createBuildOptions(request).getAllLabels(), keepGoing,
423         request.shouldRunTests(), callback);
424     runtime.throwPendingException();
425     return result;
426   }

438   /**
439    * Performs the initial phases 0-2 of the build: Setup, Loading and Analysis.
440    * <p>
441    * Postcondition: On success, populates the BuildRequest's set of targets to
442    * build.
443    *
444    * @return null if loading / analysis phases were successful; a useful error
445    *         message if loading or analysis phase errors were encountered and
446    *         request.keepGoing.
447    * @throws InterruptedException if the current thread was interrupted.
448    * @throws ViewCreationFailedException if analysis failed for any reason.
449    */
450   private AnalysisResult runAnalysisPhase(BuildRequest request, LoadingResult loadingResult,
451       BuildConfigurationCollection configurations)
452       throws InterruptedException, ViewCreationFailedException {
453     Stopwatch timer = Stopwatch.createStarted();
454     if (!request.getBuildOptions().performAnalysisPhase) {
455       getReporter().handle(Event.progress("Loading complete."));
456       LOG.info("No analysis requested, so finished");
457       return AnalysisResult.EMPTY;
458     }
459 
460     getReporter().handle(Event.progress("Loading complete.  Analyzing..."));
461     Profiler.instance().markPhase(ProfilePhase.ANALYZE);
462 
463     AnalysisResult analysisResult =
464         getView()
465             .update(
466                 loadingResult,
467                 configurations,
468                 request.getAspects(),
469                 request.getViewOptions(),
470                 request.getTopLevelArtifactContext(),
471                 getReporter(),
472                 getEventBus());
473 
474     // TODO(bazel-team): Merge these into one event.
475     getEventBus().post(new AnalysisPhaseCompleteEvent(analysisResult.getTargetsToBuild(),
476         getView().getTargetsVisited(), timer.stop().elapsed(TimeUnit.MILLISECONDS)));
477     getEventBus().post(new TestFilteringCompleteEvent(analysisResult.getTargetsToBuild(),
478         analysisResult.getTargetsToTest()));
479 
480     // Check licenses.
481     // We check licenses if the first target configuration has license checking enabled. Right now,
482     // it is not possible to have multiple target configurations with different settings for this
483     // flag, which allows us to take this short cut.
484     boolean checkLicenses = configurations.getTargetConfigurations().get(0).checkLicenses();
485     if (checkLicenses) {
486       Profiler.instance().markPhase(ProfilePhase.LICENSE);
487       validateLicensingForTargets(analysisResult.getTargetsToBuild(),
488           request.getViewOptions().keepGoing);
489     }
490 
491     return analysisResult;
492   }

494   private static boolean needsExecutionPhase(BuildRequestOptions options) {

498   /**
499    * Stops processing the specified request.
500    *
501    * <p>This logs the build result, cleans up and stops the clock.
502    *
503    * @param request the build request that this build tool is servicing
504    * @param crash Any unexpected RuntimeException or Error. May be null
505    * @param exitCondition A suggested exit condition from either the build logic or
506    *        a thrown exception somewhere along the way.
507    */
508   public void stopRequest(BuildRequest request, BuildResult result, Throwable crash,
509       ExitCode exitCondition) {
510     Preconditions.checkState((crash == null) || (exitCondition != ExitCode.SUCCESS));
511     result.setUnhandledThrowable(crash);
512     result.setExitCondition(exitCondition);
513     // The stop time has to be captured before we send the BuildCompleteEvent.
514     result.setStopTime(runtime.getClock().currentTimeMillis());
515     getEventBus().post(new BuildCompleteEvent(request, result));
516   }

518   private void reportTargets(AnalysisResult analysisResult) {
519     Collection<ConfiguredTarget> targetsToBuild = analysisResult.getTargetsToBuild();
520     Collection<ConfiguredTarget> targetsToTest = analysisResult.getTargetsToTest();
521     if (targetsToTest != null) {
522       int testCount = targetsToTest.size();
523       int targetCount = targetsToBuild.size() - testCount;
524       if (targetCount == 0) {
525         getReporter().handle(Event.info("Found "
526             + testCount + (testCount == 1 ? " test target..." : " test targets...")));
527       } else {
528         getReporter().handle(Event.info("Found "
529             + targetCount + (targetCount == 1 ? " target and " : " targets and ")
530             + testCount + (testCount == 1 ? " test target..." : " test targets...")));
531       }
532     } else {
533       int targetCount = targetsToBuild.size();
534       getReporter().handle(Event.info("Found "
535           + targetCount + (targetCount == 1 ? " target..." : " targets...")));
536     }
537   }

552   /**
553    * Takes a set of configured targets, and checks if the distribution methods
554    * declared for the targets are compatible with the constraints imposed by
555    * their prerequisites' licenses.
556    *
557    * @param configuredTargets the targets to check
558    * @param keepGoing if false, and a licensing error is encountered, both
559    *        generates an error message on the reporter, <em>and</em> throws an
560    *        exception. If true, then just generates a message on the reporter.
561    * @throws ViewCreationFailedException if the license checking failed (and not
562    *         --keep_going)
563    */
564   private void validateLicensingForTargets(Iterable<ConfiguredTarget> configuredTargets,
565       boolean keepGoing) throws ViewCreationFailedException {
566     for (ConfiguredTarget configuredTarget : configuredTargets) {
567       final Target target = configuredTarget.getTarget();
568 
569       if (TargetUtils.isTestRule(target)) {
570         continue;  // Tests are exempt from license checking
571       }
572 
573       final Set<DistributionType> distribs = target.getDistributions();
574       BuildConfiguration config = configuredTarget.getConfiguration();
575       boolean staticallyLinked = (config != null) && config.performsStaticLink();
576       staticallyLinked |= (config != null) && (target instanceof Rule)
577           && ((Rule) target).getRuleClassObject().hasAttr("linkopts", Type.STRING_LIST)
578           && ConfiguredAttributeMapper.of((RuleConfiguredTarget) configuredTarget)
579               .get("linkopts", Type.STRING_LIST).contains("-static");
580 
581       LicensesProvider provider = configuredTarget.getProvider(LicensesProvider.class);
582       if (provider != null) {
583         NestedSet<TargetLicense> licenses = provider.getTransitiveLicenses();
584         for (TargetLicense targetLicense : licenses) {
585           if (!targetLicense.getLicense().checkCompatibility(
586               distribs, target, targetLicense.getLabel(), getReporter(), staticallyLinked)) {
587             if (!keepGoing) {
588               throw new ViewCreationFailedException("Build aborted due to licensing error");
589             }
590           }
591         }
592       } else if (configuredTarget.getTarget() instanceof InputFile) {
593         // Input file targets do not provide licenses because they do not
594         // depend on the rule where their license is taken from. This is usually
595         // not a problem, because the transitive collection of licenses always
596         // hits the rule they come from, except when the input file is a
597         // top-level target. Thus, we need to handle that case specially here.
598         //
599         // See FileTarget#getLicense for more information about the handling of
600         // license issues with File targets.
601         License license = configuredTarget.getTarget().getLicense();
602         if (!license.checkCompatibility(distribs, target, configuredTarget.getLabel(),
603             getReporter(), staticallyLinked)) {
604           if (!keepGoing) {
605             throw new ViewCreationFailedException("Build aborted due to licensing error");
606           }
607        }
608       }
609     }
610   }
