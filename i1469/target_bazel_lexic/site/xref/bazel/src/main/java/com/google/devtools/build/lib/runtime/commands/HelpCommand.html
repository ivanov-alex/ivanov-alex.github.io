44  /**
45   * The 'blaze help' command, which prints all available commands as well as
46   * specific help pages.
47   */
48  @Command(name = "help",
49           options = { HelpCommand.Options.class },
50           allowResidue = true,
51           mustRunInWorkspace = false,
52           shortDescription = "Prints help for commands, or the index.",
53           completion = "command|{startup_options,target-syntax,info-keys}",
54           help = "resource:help.txt")
55  public final class HelpCommand implements BlazeCommand {
56    private static final Joiner SPACE_JOINER = Joiner.on(" ");
57  
58    public static class Options extends OptionsBase {
59  
60      @Option(name = "help_verbosity",
61              category = "help",
62              defaultValue = "medium",
63              converter = Converters.HelpVerbosityConverter.class,
64              help = "Select the verbosity of the help command.")
65      public OptionsParser.HelpVerbosity helpVerbosity;
66  
67      @Option(name = "long",
68              abbrev = 'l',
69              defaultValue = "null",
70              category = "help",
71              expansion = {"--help_verbosity", "long"},
72              help = "Show full description of each option, instead of just its name.")
73      public Void showLongFormOptions;
74  
75      @Option(name = "short",
76              defaultValue = "null",
77              category = "help",
78              expansion = {"--help_verbosity", "short"},
79              help = "Show only the names of the options, not their types or meanings.")
80      public Void showShortFormOptions;
81    }

83    /**
84     * Returns a map that maps option categories to descriptive help strings for categories that
85     * are not part of the Bazel core.
86     */
87    private ImmutableMap<String, String> getOptionCategories(BlazeRuntime runtime) {
88      ImmutableMap.Builder<String, String> optionCategoriesBuilder = ImmutableMap.builder();
89      String name = Constants.PRODUCT_NAME;
90      optionCategoriesBuilder
91          .put("checking", String.format(
92               "Checking options, which control %s's error checking and/or warnings", name))
93          .put("coverage", String.format(
94               "Options that affect how %s generates code coverage information", name))
95          .put("experimental",
96               "Experimental options, which control experimental (and potentially risky) features")
97          .put("flags",
98               "Flags options, for passing options to other tools")
99          .put("help",
100              "Help options")
101         .put("host jvm startup", String.format(
102             "Options that affect the startup of the %s server's JVM", name))
103         .put("misc",
104              "Miscellaneous options")
105         .put("package loading",
106              "Options that specify how to locate packages")
107         .put("query", String.format(
108             "Options affecting the '%s query' dependency query command", name))
109         .put("run", String.format(
110             "Options specific to '%s run'", name))
111         .put("semantics",
112              "Semantics options, which affect the build commands and/or output file contents")
113         .put("server startup", String.format(
114             "Startup options, which affect the startup of the %s server", name))
115         .put("strategy", String.format(
116             "Strategy options, which affect how %s will execute the build", name))
117         .put("testing", String.format(
118             "Options that affect how %s runs tests", name))
119         .put("verbosity", String.format(
120             "Verbosity options, which control what %s prints", name))
121         .put("version",
122              "Version options, for selecting which version of other tools will be used")
123         .put("what",
124              "Output selection options, for determining what to build/test");
125     for (BlazeModule module : runtime.getBlazeModules()) {
126       optionCategoriesBuilder.putAll(module.getOptionCategories());
127     }
128     return optionCategoriesBuilder.build();
129   }

134   @Override
135   public ExitCode exec(BlazeRuntime runtime, OptionsProvider options) {
136     OutErr outErr = runtime.getReporter().getOutErr();
137     Options helpOptions = options.getOptions(Options.class);
138     if (options.getResidue().isEmpty()) {
139       emitBlazeVersionInfo(outErr);
140       emitGenericHelp(runtime, outErr);
141       return ExitCode.SUCCESS;
142     }
143     if (options.getResidue().size() != 1) {
144       runtime.getReporter().handle(Event.error("You must specify exactly one command"));
145       return ExitCode.COMMAND_LINE_ERROR;
146     }
147     String helpSubject = options.getResidue().get(0);
148     if (helpSubject.equals("startup_options")) {
149       emitBlazeVersionInfo(outErr);
150       emitStartupOptions(outErr, helpOptions.helpVerbosity, runtime, getOptionCategories(runtime));
151       return ExitCode.SUCCESS;
152     } else if (helpSubject.equals("target-syntax")) {
153       emitBlazeVersionInfo(outErr);
154       emitTargetSyntaxHelp(outErr, getOptionCategories(runtime));
155       return ExitCode.SUCCESS;
156     } else if (helpSubject.equals("info-keys")) {
157       emitInfoKeysHelp(runtime, outErr);
158       return ExitCode.SUCCESS;
159     } else if (helpSubject.equals("completion")) {
160       emitCompletionHelp(runtime, outErr);
161       return ExitCode.SUCCESS;
162     }
163 
164     BlazeCommand command = runtime.getCommandMap().get(helpSubject);
165     if (command == null) {
166       ConfiguredRuleClassProvider provider = runtime.getRuleClassProvider();
167       RuleClass ruleClass = provider.getRuleClassMap().get(helpSubject);
168       if (ruleClass != null && ruleClass.isDocumented()) {
169         // There is a rule with a corresponding name
170         outErr.printOut(BlazeRuleHelpPrinter.getRuleDoc(helpSubject, provider));
171         return ExitCode.SUCCESS;
172       } else {
173         runtime.getReporter().handle(Event.error(
174             null, "'" + helpSubject + "' is neither a command nor a build rule"));
175         return ExitCode.COMMAND_LINE_ERROR;
176       }
177     }
178     emitBlazeVersionInfo(outErr);
179     outErr.printOut(BlazeCommandUtils.getUsage(
180         command.getClass(),
181         getOptionCategories(runtime),
182         helpOptions.helpVerbosity,
183         runtime.getBlazeModules(),
184         runtime.getRuleClassProvider()));
185     return ExitCode.SUCCESS;
186   }

188   private void emitBlazeVersionInfo(OutErr outErr) {
189     String releaseInfo = BlazeVersionInfo.instance().getReleaseName();
190     String line = String.format("[%s %s]", Constants.PRODUCT_NAME, releaseInfo);
191     outErr.printOut(String.format("%80s\n", line));
192   }

194   @SuppressWarnings("unchecked") // varargs generic array creation
195   private void emitStartupOptions(OutErr outErr, OptionsParser.HelpVerbosity helpVerbosity,
196       BlazeRuntime runtime, ImmutableMap<String, String> optionCategories) {
197     outErr.printOut(
198         BlazeCommandUtils.expandHelpTopic("startup_options",
199             "resource:startup_options.txt",
200             getClass(),
201             BlazeCommandUtils.getStartupOptions(runtime.getBlazeModules()),
202             optionCategories,
203         helpVerbosity));
204   }

206   private void emitCompletionHelp(BlazeRuntime runtime, OutErr outErr) {
207     // First startup_options
208     Iterable<BlazeModule> blazeModules = runtime.getBlazeModules();
209     ConfiguredRuleClassProvider ruleClassProvider = runtime.getRuleClassProvider();
210     Map<String, BlazeCommand> commandsByName = runtime.getCommandMap();
211     Set<String> commands = commandsByName.keySet();
212 
213     outErr.printOutLn("BAZEL_COMMAND_LIST=\"" + SPACE_JOINER.join(commands) + "\"");
214 
215     outErr.printOutLn("BAZEL_INFO_KEYS=\"");
216     for (InfoKey key : InfoKey.values()) {
217         outErr.printOutLn(key.getName());
218     }
219     outErr.printOutLn("\"");
220 
221     outErr.printOutLn("BAZEL_STARTUP_OPTIONS=\"");
222     Iterable<Class<? extends OptionsBase>> options =
223         BlazeCommandUtils.getStartupOptions(blazeModules);
224     outErr.printOut(OptionsParser.newOptionsParser(options).getOptionsCompletion());
225     outErr.printOutLn("\"");
226 
227     for (String name : commands) {
228       BlazeCommand command = commandsByName.get(name);
229       String varName = name.toUpperCase().replace("-", "_");
230       Command annotation = command.getClass().getAnnotation(Command.class);
231       if (!annotation.completion().isEmpty()) {
232         outErr.printOutLn("BAZEL_COMMAND_" + varName + "_ARGUMENT=\""
233             + annotation.completion() + "\"");
234       }
235       options = BlazeCommandUtils.getOptions(command.getClass(), blazeModules, ruleClassProvider);
236       outErr.printOutLn("BAZEL_COMMAND_" + varName + "_FLAGS=\"");
237       outErr.printOut(OptionsParser.newOptionsParser(options).getOptionsCompletion());
238       outErr.printOutLn("\"");
239     }
240   }

242   private void emitTargetSyntaxHelp(OutErr outErr, ImmutableMap<String, String> optionCategories) {
243     outErr.printOut(BlazeCommandUtils.expandHelpTopic("target-syntax",
244                                     "resource:target-syntax.txt",
245                                     getClass(),
246                                     ImmutableList.<Class<? extends OptionsBase>>of(),
247                                     optionCategories,
248                                     OptionsParser.HelpVerbosity.MEDIUM));
249   }
250 
251   private void emitInfoKeysHelp(BlazeRuntime runtime, OutErr outErr) {
252     for (BlazeModule.InfoItem item : InfoCommand.getInfoItemMap(runtime,
253         OptionsParser.newOptionsParser(
254             ImmutableList.<Class<? extends OptionsBase>>of())).values()) {
255       outErr.printOut(String.format("%-23s %s\n", item.getName(), item.getDescription()));
256     }
257   }

259   private void emitGenericHelp(BlazeRuntime runtime, OutErr outErr) {
260     outErr.printOut(String.format("Usage: %s <command> <options> ...\n\n",
261             Constants.PRODUCT_NAME));
262 
263     outErr.printOut("Available commands:\n");
264 
265     Map<String, BlazeCommand> commandsByName = runtime.getCommandMap();
266     List<String> namesInOrder = new ArrayList<>(commandsByName.keySet());
267     Collections.sort(namesInOrder);
268 
269     for (String name : namesInOrder) {
270       BlazeCommand command = commandsByName.get(name);
271       Command annotation = command.getClass().getAnnotation(Command.class);
272       if (annotation.hidden()) {
273         continue;
274       }
275 
276       String shortDescription = annotation.shortDescription().
277           replace("%{product}", Constants.PRODUCT_NAME);
278       outErr.printOut(String.format("  %-19s %s\n", name, shortDescription));
279     }
280 
281     outErr.printOut("\n");
282     outErr.printOut("Getting more help:\n");
283     outErr.printOut(String.format("  %s help <command>\n", Constants.PRODUCT_NAME));
284     outErr.printOut("                   Prints help and options for <command>.\n");
285     outErr.printOut(String.format("  %s help startup_options\n", Constants.PRODUCT_NAME));
286     outErr.printOut(String.format("                   Options for the JVM hosting %s.\n",
287             Constants.PRODUCT_NAME));
288     outErr.printOut(String.format("  %s help target-syntax\n", Constants.PRODUCT_NAME));
289     outErr.printOut("                   Explains the syntax for specifying targets.\n");
290     outErr.printOut(String.format("  %s help info-keys\n", Constants.PRODUCT_NAME));
291     outErr.printOut("                   Displays a list of keys used by the info command.\n");
292   }
