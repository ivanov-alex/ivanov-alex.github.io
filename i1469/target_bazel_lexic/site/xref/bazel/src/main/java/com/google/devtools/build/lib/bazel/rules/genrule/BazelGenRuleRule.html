37  public final class BazelGenRuleRule implements RuleDefinition {
38    public static final String GENRULE_SETUP_LABEL = "//tools/genrule:genrule-setup.sh";
39  
40    @Override
41    public RuleClass build(Builder builder, RuleDefinitionEnvironment env) {
42      /* <!-- #BLAZE_RULE(genrule).NAME -->
43      <br/>You may refer to this rule by name in the <code>srcs</code> or <code>deps</code> section of
44      other <code>BUILD</code> rules. If the rule generates source files, you should use the
45      <code>srcs</code> attribute.
46      <!-- #END_BLAZE_RULE.NAME --> */
47      return builder
48          .setOutputToGenfiles()
49          /* <!-- #BLAZE_RULE(genrule).ATTRIBUTE(srcs) -->
50          A list of inputs for this rule, such as source files to process.
51          ${SYNOPSIS}
52          <p>
53            <em>This attributes is not suitable to list tools executed by the <code>cmd</code>; use
54            the <a href="#genrule.tools"><code>tools</code></a> attribute for them instead.</em>
55          </p>
56          <p>
57            The build system ensures these prerequisites are built before running the genrule
58            command; they are built using the same configuration as the original build request. The
59            names of the files of these prerequisites are available to the command as a
60            space-separated list in <code>$(SRCS)</code>; alternatively the path of an individual
61            <code>srcs</code> target <code>//x:y</code> can be obtained using <code>$(location
62            //x:y)</code>, or using <code>$&lt;</code> provided it's the only entry in
63            //<code>srcs</code>.
64          </p>
65          <!-- #END_BLAZE_RULE.ATTRIBUTE --> */
66          .add(attr("srcs", LABEL_LIST)
67              .direct_compile_time_input()
68              .legacyAllowAnyFileType())
69  
70          /* <!-- #BLAZE_RULE(genrule).ATTRIBUTE(tools) -->
71          A list of <i>tool</i> dependencies for this rule.
72          ${SYNOPSIS}
73          <p>
74            The build system ensures these prerequisites are built before running the genrule command;
75            they are built using the <a href='blaze-user-manual.html#configurations'><i>host</i>
76            configuration</a>, since these tools are executed as part of the build. The path of an
77            individual <code>tools</code> target <code>//x:y</code> can be obtained using
78            <code>$(location //x:y)</code>.
79          </p>
80          <p>
81            Any <code>*_binary</code> or tool to be executed by <code>cmd</code> must appear in this
82            list, not in <code>srcs</code>, to ensure they are built in the correct configuration.
83          </p>
84          <!-- #END_BLAZE_RULE.ATTRIBUTE --> */
85          .add(attr("tools", LABEL_LIST).cfg(HOST).legacyAllowAnyFileType())
86          .add(attr("$genrule_setup", LABEL).cfg(HOST).value(env.getLabel(GENRULE_SETUP_LABEL)))
87  
88          /* <!-- #BLAZE_RULE(genrule).ATTRIBUTE(outs) -->
89          A list of files generated by this rule.
90          ${SYNOPSIS}
91          <p>
92            All output files must be generated into the same directory and must not cross package
93            boundaries. Output filenames are interpreted as relative to the package.
94          </p>
95          <p>
96            If the <code>executable</code> flag is set, <code>outs</code> must contain exactly one
97            label.
98          </p>
99          <!-- #END_BLAZE_RULE.ATTRIBUTE --> */
100         .add(attr("outs", OUTPUT_LIST).mandatory())
101 
102         /* <!-- #BLAZE_RULE(genrule).ATTRIBUTE(cmd) -->
103         The command to run.
104         ${SYNOPSIS}
105         Subject to <a href="#location">$(location)</a> and <a href="#make_variables">"Make"
106         variable</a> substitution.
107         <ol>
108           <li>
109             First <a href="#location">$(location)</a> substitution is applied, replacing all
110             occurrences of <code>$(location <i>label</i>)</code> and of <code>$(locations
111             <i>label</i>)</code>.
112           </li>
113           <li>
114             <p>
115               Note that <code>outs</code> are <i>not</i> included in this substitution. Output files
116               are always generated into a predictable location (available via <code>$(@D)</code>,
117               <code>$@</code>, <code>$(OUTS)</code> or <code>$(location <i>output_name</i>)</code>;
118               see below).
119             </p>
120           </li>
121           <li>
122             Next, <a href="#make_variables">"Make" variables</a> are expanded. Note that predefined
123             variables <code>$(JAVA)</code>, <code>$(JAVAC)</code> and <code>$(JAVABASE)</code>
124             expand under the <i>host</i> configuration, so Java invocations that run as part of a
125             build step can correctly load shared libraries and other dependencies.
126           </li>
127           <li>
128             Finally, the resulting command is executed using the Bash shell. If its exit code is
129             non-zero the command is considered to have failed.
130           </li>
131         </ol>
132         <p>
133           The command may refer to binaries that were declared as
134           <a href="#genrule.tools"><code>tools</code></a>; it should use a
135           <a href="build-ref.html#labels">label</a> and <code>$(location)</code> expansion for this.
136           The <a href="#predefined_variables">Predefined "Make" Variables</a> and the
137           <a href="#predefined_variables.genrule.cmd">genrule-specific "Make" variables</a> are also
138           available for <code>cmd</code>.
139         </p>
140         <!-- #END_BLAZE_RULE.ATTRIBUTE --> */
141         .add(attr("cmd", STRING).mandatory())
142 
143         /* <!-- #BLAZE_RULE(genrule).ATTRIBUTE(output_to_bindir) -->
144         ${SYNOPSIS}
145         <p>
146           If set to 1, this option causes output files to be written into the <code>bin</code>
147           directory instead of the <code>genfiles</code> directory.
148         </p>
149         <!-- #END_BLAZE_RULE.ATTRIBUTE --> */
150         // TODO(bazel-team): find a location to document genfiles/binfiles, link to them from here.
151         .add(attr("output_to_bindir", BOOLEAN).value(false)
152             .nonconfigurable("policy decision: no reason for this to depend on the configuration"))
153 
154         // TODO(bazel-team): remove?
155         .add(attr("local", BOOLEAN).value(false))
156 
157         /* <!-- #BLAZE_RULE(genrule).ATTRIBUTE(message) -->
158         A progress message.
159         ${SYNOPSIS}
160         <p>
161           A progress message that will be printed as this build step is executed. By default, the
162           message is "Generating <i>output</i>" (or something equally bland) but you may provide a
163           more specific one. Use this attribute instead of <code>echo</code> or other print
164           statements in your <code>cmd</code> command, as this allows the build tool to control
165           whether such progress messages are printed or not.
166         </p>
167         <!-- #END_BLAZE_RULE.ATTRIBUTE --> */
168         .add(attr("message", STRING))
169         .add(attr("output_licenses", LICENSE))
170 
171         /* <!-- #BLAZE_RULE(genrule).ATTRIBUTE(executable) -->
172         Declare output to be executable.
173         ${SYNOPSIS}
174         <p>
175           Setting this flag to 1 means the output is an executable file and can be run using the
176           <code>bazel run</code> command. The genrule must produce exactly one output in this case.
177           If this attribute is set, <code>bazel run</code> will try executing the file regardless of
178           its content.
179         </p>
180         <p>Declaring data dependencies for the generated executable is not supported.</p>
181         <!-- #END_BLAZE_RULE.ATTRIBUTE --> */
182         .add(attr("executable", BOOLEAN).value(false))
183 
184         // TODO(bazel-team): stamping doesn't seem to work. Fix it or remove attribute.
185         .add(attr("stamp", BOOLEAN).value(false))
186         // This is a misfeature, so don't document it. We would like to get rid of it, but that
187         // would require a cleanup of existing rules.
188         .add(attr("heuristic_label_expansion", BOOLEAN).value(false))
189         .add(attr("$is_executable", BOOLEAN)
190             .nonconfigurable("Called from RunCommand.isExecutable, which takes a Target")
191             .value(
192             new Attribute.ComputedDefault("outs", "executable") {
193               @Override
194               public Object getDefault(AttributeMap rule) {
195                 return (rule.get("outs", Type.OUTPUT_LIST).size() == 1)
196                     && rule.get("executable", BOOLEAN);
197               }
198             }))
199         .removeAttribute("data")
200         .removeAttribute("deps")
201         .build();
202   }

206     return RuleDefinition.Metadata.builder()
207         .name("genrule")
208         .ancestors(BaseRuleClasses.RuleBase.class)
209         .factoryClass(GenRule.class)
210         .build();
211   }

214 /*<!-- #BLAZE_RULE (NAME = genrule, TYPE = OTHER, FAMILY = General)[GENERIC_RULE] -->

216 ${ATTRIBUTE_SIGNATURE}
217 
218 <p>A <code>genrule</code> generates one or more files using a user-defined Bash command.</p>
219 
220 <p>
221   Genrules are generic build rules that you can use if there's no specific rule for the task. If for
222   example you want to minify JavaScript files then you can use a genrule to do so. If however you
223   need to compile C++ files, stick to the existing <code>cc_*</code> rules, because all the heavy
224   lifting has already been done for you.
225 </p>
226 <p>
227   Do not use a genrule for running tests. There are special dispensations for tests and test
228   results, including caching policies and environment variables. Tests generally need to be run
229   after the build is complete and on the target architecture, whereas genrules are executed during
230   the build and on the host architecture (the two may be different). If you need a general purpose
231   testing rule, use <a href="#sh_test"><code>sh_test</code></a>.
232 </p>
233 
234 <h4>Cross-compilation Considerations</h4>
235 
236 <p>
237   <em>See <a href="bazel-user-manual.html#configurations">the user manual</a> for more info about
238   cross-compilation.</em>
239 </p>
240 <p>
241   While genrules run during a build, their outputs are often used after the build, for deployment or
242   testing. Consider the example of compiling C code for a microcontroller: the compiler accepts C
243   source files and generates code that runs on a microcontroller. The generated code obviously
244   cannot run on the CPU that was used for building it, but the C compiler (if compiled from source)
245   itself has to.
246 </p>
247 <p>
248   The build system uses the host configuration to describe the machine(s) on which the build runs
249   and the target configuration to describe the machine(s) on which the output of the build is
250   supposed to run. It provides options to configure each of these and it segregates the
251   corresponding files into separate directories to avoid conflicts.
252 </p>
253 <p>
254   For genrules, the build system ensures that dependencies are built appropriately:
255   <code>srcs</code> are built (if necessary) for the <em>target</em> configuration,
256   <code>tools</code> are built for the <em>host</em> configuration, and the output is considered to
257   be for the <em>target</em> configuration. It also provides <a href="#make_variables">"Make"
258   variables</a> that genrule commands can pass to the corresponding tools.
259 </p>
260 <p>
261   It is intentional that genrule defines no <code>deps</code> attribute: other built-in rules use
262   language-dependent meta information passed between the rules to automatically determine how to
263   handle dependent rules, but this level of automation is not possible for genrules. Genrules work
264   purely at the file and runfiles level.
265 </p>
266 
267 <h4>Special Cases</h4>
268 
269 <p>
270   <i>Host-host compilation</i>: in some cases, the build system needs to run genrules such that the
271   output can also be executed during the build. If for example a genrule builds some custom compiler
272   which is subsequently used by another genrule, the first one has to produce its output for the
273   host configuration, because that's where the compiler will run in the other genrule. In this case,
274   the build system does the right thing automatically: it builds the <code>srcs</code> and
275   <code>outs</code> of the first genrule for the host configuration instead of the target
276   configuration. See <a href="bazel-user-manual.html#configurations">the user manual</a> for more
277   info.
278 </p>
279 <p>
280   <i>JDK & C++ Tooling</i>: to use a tool from the JDK or the C++ compiler suite, the build system
281   provides a set of variables to use. See <a href="#make_variables">"Make" variable</a> for details.
282 </p>
283 
284 <h4>Genrule Environment</h4>
285 
286 <p>
287   The genrule command is executed in a Bash shell, configured to fail when a command or a pipeline
288   fails (<code>set -e -o pipefail</code>). Genrules should not access the network (except to create
289   connections between processes running within the same genrule on the same machine), though this is
290   not currently enforced.
291 </p>
292 <p>
293   The build system automatically deletes any existing output files, but creates any necessary parent
294   directories before it runs a genrule. It also removes any output files in case of a failure.
295 </p>
296 
297 <h4>General Advice</h4>
298 
299 <ul>
300   <li>Do ensure that tools run by a genrule are deterministic and hermetic. They should not write
301     timestamps to their output, and they should use stable ordering for sets and maps, as well as
302     write only relative file paths to the output, no absolute paths. Not following this rule will
303     lead to unexpected build behavior (Bazel not rebuilding a genrule you thought it would) and
304     degrade cache performance.</li>
305   <li>Do use <code>$(location)</code> extensively, for outputs, tools and sources. Due to the
306     segregation of output files for different configurations, genrules cannot rely on hard-coded
307     and/or absolute paths.</li>
308   <li>Do write a common Skylark macro in case the same or very similar genrules are used in multiple
309     places. If the genrule is complex, consider implementing it in a script or as a Skylark rule.
310     This improves readability as well as testability.</li>
311   <li>Do make sure that the exit code correctly indicates success or failure of the genrule.</li>
312   <li>Do not write informational messages to stdout or stderr. While useful for debugging, this can
313     easily become noise; a successful genrule should be silent. On the other hand, a failing genrule
314     should emit good error messages.</li>
315   <li><code>$$</code> evaluates to a <code>$</code>, a literal dollar-sign, so in order to invoke a
316     shell command containing dollar-signs such as <code>ls $(dirname $x)</code>, one must escape it
317     thus: <code>ls $$(dirname $$x)</code>.</li>
318   <li>Avoid creating symlinks and directories. Bazel doesn't copy over the directory/symlink
319     structure created by genrules and its dependency checking of directories is unsound.</li>
320   <li>When referencing the genrule in other rules, you can use either the genrule's label or the
321     labels of individual output files. Sometimes the one approach is more readable, sometimes the
322     other: referencing outputs by name in a consuming rule's <code>srcs</code> will avoid
323     unintentionally picking up other outputs of the genrule, but can be tedious if the genrule
324     produces many outputs.</li>
325 </ul>

329 <h4 id="genrule_examples">Examples</h4>
330 
331 <p>
332   This example generates <code>foo.h</code>. There are no sources, because the command doesn't take
333   any input. The "binary" run by the command is a perl script in the same package as the genrule.
334 </p>
335 <pre class="code">
336 genrule(
337     name = "foo",
338     srcs = [],
339     outs = ["foo.h"],
340     cmd = "$(location create_foo.pl) &gt; \"$@\"",
341     tools = ["create_foo.pl"],
342 )
343 </pre>
344 
345 <p>
346   The following example shows how to use a <a href="#filegroup"><code>filegroup</code></a> and
347   the outputs of another <code>genrule</code>. Note that using <code>$(SRCS)</code> instead of
348   explicit <code>$(location)</code> directives would also work; this example uses the latter for
349   sake of demonstration.
350 </p>
351 <pre class="code">
352 genrule(
353     name = "concat_all_files",
354     srcs = [
355         "//some:files",  # a filegroup with multiple files in it ==> $(location<b>s</b>)
356         "//other:gen",   # a genrule with a single output ==> $(location)
357     ],
358     outs = ["concatenated.txt"],
359     cmd = "cat $(locations //some:files) $(location //other:gen) > $@",
360 )
361 </pre>
