14  package com.google.devtools.build.lib.concurrent;
15  
16  import com.google.common.base.Preconditions;
17  import com.google.common.collect.ConcurrentHashMultiset;
18  import com.google.common.util.concurrent.Striped;
19  
20  import java.util.concurrent.ConcurrentHashMap;
21  import java.util.concurrent.ConcurrentMap;
22  import java.util.concurrent.atomic.AtomicBoolean;
23  import java.util.concurrent.locks.Lock;
24  import java.util.concurrent.locks.ReentrantLock;
25  
26  /**
27   * An implementation of {@link KeyedLocker} that uses ref counting to efficiently only store locks
28   * that are live.
29   */
30  public class RefCountedMultisetKeyedLocker<K> implements KeyedLocker<K> {
31    // Multiset of keys that have threads waiting on a lock or using a lock.
32    private final ConcurrentHashMultiset<K> waiters = ConcurrentHashMultiset.<K>create();
33  
34    private static final int NUM_STRIPES = 2048;
35    // For each key, gives the striped lock to use for atomically managing the waiters on that key
36    // internally.
37    private final Striped<Lock> waitersLocks = Striped.lazyWeakLock(NUM_STRIPES);
38  
39    // Map of key to current lock, for keys that have at least one waiting or live thread.
40    private final ConcurrentMap<K, ReentrantLock> locks =
41        new ConcurrentHashMap<>(1024, .75f, NUM_STRIPES);
42  
43    public RefCountedMultisetKeyedLocker() {
44    }
45  
46    private class RefCountedAutoUnlocker implements AutoUnlocker {
47      private final K key;
48      private final ReentrantLock lock;
49      private final AtomicBoolean closeCalled = new AtomicBoolean(false);
50  
51      private RefCountedAutoUnlocker(K key, ReentrantLock lock) {
52        this.key = key;
53        this.lock = lock;
54      }
55  
56      @Override
57      public void close() {
58        if (closeCalled.getAndSet(true)) {
59          String msg = String.format("For key %s, 'close' can be called at most once per "
60                                         + "AutoUnlocker instance", key);
61          throw new IllegalUnlockException(msg);
62        }
63        if (!lock.isHeldByCurrentThread()) {
64          String msg = String.format("For key %s, the calling thread to 'close' must be the one "
65              + "that acquired the AutoUnlocker", key);
66          throw new IllegalUnlockException(msg);
67        }
68        try {
69          Lock waitersLock = waitersLocks.get(key);
70          try {
71            waitersLock.lock();
72            // Note that ConcurrentHashMultiset automatically removes removes entries for keys whose
73            // count is 0.
74            if (waiters.remove(key, 1) == 1) {
75              // No other thread is waiting to access this key, nor does the current thread have
76              // another AutoUnlocker instance, so we garbage collect the lock.
77              Preconditions.checkState(locks.remove(key, lock), key);
78            }
79          } finally {
80            waitersLock.unlock();
81          }
82        } finally {
83          lock.unlock();
84        }
85      }
86    }
87  
88    @Override
89    public AutoUnlocker lock(K key) {
90      ReentrantLock newLock = new ReentrantLock();
91      // Pre-lock our fresh lock, in case we win the race to get access to 'key'.
92      newLock.lock();
93      Lock waitersLock = waitersLocks.get(key);
94      try {
95        waitersLock.lock();
96        // Add us to the set of waiters, so that in case we lose the race to access 'key', the winner
97        // will know that we are waiting. If we already have access to 'key', this simply bumps up
98        // the ref count.
99        waiters.add(key);
100     } finally {
101       waitersLock.unlock();
102     }
103     ReentrantLock lock;
104     lock = locks.putIfAbsent(key, newLock);
105     if (lock != null) {
106       Preconditions.checkState(lock != newLock);
107       newLock.unlock();
108       // Either another thread won the race to get access to 'key', or we already have exclusive
109       // access to 'key'. Either way, we lock; in the former case we wait for our turn and in the
110       // latter case the lock's implicit counter is incremented.
111       lock.lock();
112       return new RefCountedAutoUnlocker(key, lock);
113     }
114     // We won the race, so the current lock for 'key' is the one we locked and inserted.
115     return new RefCountedAutoUnlocker(key, newLock);
116   }
117 }
