40  /**
41   * This class provides a complete in-memory file system.
42   *
43   * <p>Naming convention: we use "path" for all {@link Path} variables, since these
44   * represent *names* and we use "node" or "inode" for InMemoryContentInfo
45   * variables, since these correspond to inodes in the UNIX file system.
46   *
47   * <p>The code is structured to be as similar to the implementation of UNIX "namei"
48   * as is reasonably possibly. This provides a firm reference point for many
49   * concepts and makes compatibility easier to achieve.
50   *
51   * <p>As a scope-escapable file system, this class supports re-delegation of symbolic links
52   * that escape its root. This is done through the use of {@link OutOfScopeFileStatus}
53   * and {@link OutOfScopeDirectoryStatus} objects, which may be returned by
54   * getDirectory, pathWalk, and scopeLimitedStat. Any code that calls one of these
55   * methods (either directly or indirectly) is obligated to check the possibility
56   * that its info represents an out-of-scope path. Lack of such a check will result
57   * in unchecked runtime exceptions upon any request for status data (as well as
58   * possible logical errors).
59   */
60  @ThreadSafe
61  public class InMemoryFileSystem extends ScopeEscapableFileSystem {
62

64

66    private final InMemoryDirectoryInfo rootInode;

87    /**

89     * scopeRoot, i.e. any path that's not below scopeRoot is considered
90     * to be out of scope.
91     */
92    protected InMemoryFileSystem(Clock clock, PathFragment scopeRoot) {
93      super(scopeRoot);
94      this.clock = clock;
95      this.rootInode = new InMemoryDirectoryInfo(clock);
96      rootInode.addChild(".", rootInode);
97      rootInode.addChild("..", rootInode);

129     /**
130      * The exceptions below extend their parent classes in order to additionally store the error
131      * that caused them. However, they must impersonate their parents to any outside callers,
132      * including in their toString() method, which prints the class name followed by the exception
133      * method. This method returns the same value as the toString() method of a {@link Throwable}'s
134      * parent would, so that the child class can have the same toString() value.

136     private static String parentThrowableToString(Throwable obj) {
137       String s = obj.getClass().getSuperclass().getName();
138       String message = obj.getLocalizedMessage();
139       return (message != null) ? (s + ": " + message) : s;
140     }

203     /**
204      * Returns a new IOException for the error. The exception message
205      * contains 'path', and is consistent with the messages returned by
206      * c.g.common.unix.FilesystemUtils.
207      */
208     public IOException exception(Path path) throws IOException {
209       String m = path + " (" + message + ")";
210       if (this == EACCES) {
211         throw new FileAccessExceptionWithError(m, this);
212       } else if (this == ENOENT) {
213         throw new FileNotFoundExceptionWithError(m, this);
214       } else {
215         throw new IOExceptionWithError(m, this);
216       }

235   /**
236    * Unlinks the entry 'child' from its existing parent directory 'dir'. Dual to
237    * insert. This succeeds even if 'child' names a non-empty directory; we need
238    * that for renameTo. 'child' must be a member of its parent directory,
239    * however. Fails if the directory was read-only.
240    */
241   private void unlink(InMemoryDirectoryInfo dir, String child, Path errorPath)
242       throws IOException {
243     if (!dir.isWritable()) { throw Error.EACCES.exception(errorPath); }
244     dir.removeChild(child);
245   }
246 
247   /**
248    * Inserts inode 'childInode' into the existing directory 'dir' under the
249    * specified 'name'.  Dual to unlink.  Fails if the directory was read-only.
250    */
251   private void insert(InMemoryDirectoryInfo dir, String child,
252                       InMemoryContentInfo childInode, Path errorPath)
253       throws IOException {
254     if (!dir.isWritable()) { throw Error.EACCES.exception(errorPath); }
255     dir.addChild(child, childInode);
256   }

258   /**
259    * Given an existing directory 'dir', looks up 'name' within it and returns
260    * its inode. Assumes the file exists, unless 'create', in which case it will
261    * try to create it. May fail with ENOTDIR, EACCES, ENOENT. Error messages
262    * will be reported against file 'path'.
263    */
264   private InMemoryContentInfo directoryLookup(InMemoryContentInfo dir,
265                                               String name,
266                                               boolean create,
267                                               Path path) throws IOException {
268     if (!dir.isDirectory()) { throw Error.ENOTDIR.exception(path); }
269     InMemoryDirectoryInfo imdi = (InMemoryDirectoryInfo) dir;
270     if (!imdi.isExecutable()) { throw Error.EACCES.exception(path); }
271     InMemoryContentInfo child = imdi.getChild(name);
272     if (child == null) {
273       if (!create)  {
274         throw Error.ENOENT.exception(path);
275       } else {
276         child = makeFileInfo(clock, path.asFragment());
277         insert(imdi, name, child, path);
278       }
279     }
280     return child;
281   }

283   /**
284    * Low-level path-to-inode lookup routine. Analogous to path_walk() in many
285    * UNIX kernels. Given 'path', walks the directory tree from the root,
286    * resolving all symbolic links, and returns the designated inode.
287    *
288    * <p>If 'create' is false, the inode must exist; otherwise, it will be created
289    * and added to its parent directory, which must exist.
290    *
291    * <p>Iff the given path escapes this file system's scope, the returned value
292    * is an {@link OutOfScopeFileStatus} instance. Any code that calls this method
293    * needs to check for that possibility (via {@link ScopeEscapableStatus#outOfScope}).
294    *
295    * <p>May fail with ENOTDIR, ENOENT, EACCES, ELOOP.
296    */
297   private synchronized InMemoryContentInfo pathWalk(Path path, boolean create)
298       throws IOException {
299     // Implementation note: This is where we check for out-of-scope symlinks and
300     // trigger re-delegation to another file system accordingly. This code handles
301     // both absolute and relative symlinks. Some assumptions we make: First, only
302     // symlink targets as read from getNormalizedLinkContent() can escape our scope.
303     // This is because Path objects are all canonicalized (see {@link Path#getRelative},
304     // etc.) and symlink target segments that get added to the stack are in-scope by
305     // definition. Second, symlink targets with relative segments must have the form
306     // [".."]*[standard segment]+, i.e. only the ".." non-standard segment is allowed
307     // and it may only appear as part of a contiguous prefix sequence.
308 
309     Stack<String> stack = new Stack<>();
310     PathFragment rootPathFragment = rootPath.asFragment();
311     for (Path p = path; !p.asFragment().equals(rootPathFragment); p = p.getParentDirectory()) {
312       stack.push(p.getBaseName());
313     }
314 
315     InMemoryContentInfo inode = rootInode;
316     int parentDepth = -1;
317     int traversals = 0;
318 
319     while (!stack.isEmpty()) {
320       traversals++;
321 
322       String name = stack.pop();
323       parentDepth += name.equals("..") ? -1 : 1;
324 
325       // ENOENT on last segment with 'create' => create a new file.
326       InMemoryContentInfo child = directoryLookup(inode, name, create && stack.isEmpty(), path);
327       if (child.isSymbolicLink()) {
328         PathFragment linkTarget = ((InMemoryLinkInfo) child).getNormalizedLinkContent();
329         if (!inScope(parentDepth, linkTarget)) {
330           return outOfScopeStatus(linkTarget, parentDepth, stack);
331         }
332         if (linkTarget.isAbsolute()) {
333           inode = rootInode;
334           parentDepth = -1;
335         }
336         if (traversals > MAX_TRAVERSALS) {
337           throw Error.ELOOP.exception(path);
338         }
339         for (int ii = linkTarget.segmentCount() - 1; ii >= 0; --ii) {
340           stack.push(linkTarget.getSegment(ii)); // Note this may include ".." segments.
341         }
342       } else {
343         inode = child;
344       }
345     }
346     return inode;
347   }

349   /**
350    * Helper routine for pathWalk: given a symlink target known to escape this file system's
351    * scope (and that has the form [".."]*[standard segment]+), the number of segments
352    * in the directory containing the symlink, and the remaining path segments following
353    * the symlink in the original input to pathWalk, returns an OutofScopeFileStatus
354    * initialized with an appropriate out-of-scope reformulation of pathWalk's original
355    * input.
356    */
357   private OutOfScopeFileStatus outOfScopeStatus(PathFragment linkTarget, int parentDepth,
358       Stack<String> descendantSegments) {
359 
360     PathFragment escapingPath;
361     if (linkTarget.isAbsolute()) {
362       escapingPath = linkTarget;
363     } else {
364       // Relative out-of-scope paths must look like "../../../a/b/c". Find the target's
365       // parent path depth by subtracting one from parentDepth for each ".." reference.
366       // Then use that to retrieve a prefix of the scope root, which is the target's
367       // canonicalized parent path.
368       int leadingParentRefs = leadingParentReferences(linkTarget);
369       int baseDepth = parentDepth - leadingParentRefs;
370       Preconditions.checkState(baseDepth < scopeRoot.segmentCount());
371       escapingPath = baseDepth > 0
372           ? scopeRoot.subFragment(0, baseDepth)
373           : scopeRoot.subFragment(0, 0);
374       // Now add in everything that comes after the ".." sequence.
375       for (int i = leadingParentRefs; i < linkTarget.segmentCount(); i++) {
376         escapingPath = escapingPath.getRelative(linkTarget.getSegment(i));
377       }
378     }
379 
380     // We've now converted the symlink to its target in canonicalized absolute path
381     // form. Since the symlink wasn't necessarily the final segment in the original
382     // input sent to pathWalk, now add in every segment that came after.
383     while (!descendantSegments.empty()) {
384       escapingPath = escapingPath.getRelative(descendantSegments.pop());
385     }
386 
387     return new OutOfScopeFileStatus(escapingPath);
388   }

390   /**
391    * Given 'path', returns the existing directory inode it designates,
392    * following symbolic links.
393    *
394    * <p>May fail with ENOTDIR, or any exception from pathWalk.
395    *
396    * <p>Iff the given path escapes this file system's scope, this method skips
397    * ENOTDIR checking and returns an OutOfScopeDirectoryStatus instance. Any
398    * code that calls this method needs to check for that possibility
399    * (via {@link ScopeEscapableStatus#outOfScope}).
400    */
401   private InMemoryDirectoryInfo getDirectory(Path path) throws IOException {
402     InMemoryContentInfo dirInfo = pathWalk(path, false);
403     if (dirInfo.outOfScope()) {
404       return new OutOfScopeDirectoryStatus(dirInfo.getEscapingPath());
405     } else if (!dirInfo.isDirectory()) {
406       throw Error.ENOTDIR.exception(path);
407     } else {
408       return (InMemoryDirectoryInfo) dirInfo;
409     }
410   }

412   /**
413    * Helper method for stat, scopeLimitedStat: lock the internal state and return the
414    * path's (no symlink-followed) stat if the path's parent directory is within scope,
415    * else return an "out of scope" reference to the path's parent directory (which will
416    * presumably be re-delegated to another FS).
417    */
418   private synchronized InMemoryContentInfo getNoFollowStatOrOutOfScopeParent(Path path)
419       throws IOException  {
420     InMemoryDirectoryInfo dirInfo = getDirectory(path.getParentDirectory());
421     return dirInfo.outOfScope()
422         ? dirInfo
423         : directoryLookup(dirInfo, path.getBaseName(), /*create=*/false, path);
424   }

426   /**
427    * Given 'path', returns the existing inode it designates, optionally
428    * following symbolic links.  Analogous to UNIX stat(2)/lstat(2), except that
429    * it returns a mutable inode we can modify directly.
430    */
431   @Override
432   public FileStatus stat(Path path, boolean followSymlinks) throws IOException {
433     if (followSymlinks) {
434       InMemoryContentInfo status = scopeLimitedStat(path, true);
435       return status.outOfScope()
436           ? statWithDelegator(status.getEscapingPath(), true)
437           : status;
438     } else {
439       if (path.equals(rootPath)) {
440         return rootInode;
441       } else {
442         InMemoryContentInfo status = getNoFollowStatOrOutOfScopeParent(path);
443         // If out of scope, status references the path's parent directory. Else it references the
444         // path itself.
445         return status.outOfScope()
446             ? getDelegatedPath(status.getEscapingPath().getRelative(
447                   path.getBaseName())).stat(Symlinks.NOFOLLOW)
448             : status;
449       }
450     }
451   }

453   @Override
454   @Nullable
455   public FileStatus statIfFound(Path path, boolean followSymlinks) throws IOException {
456     try {
457       return stat(path, followSymlinks);
458     } catch (IOException e) {
459       if (e instanceof Error.WithError) {
460         Error errorCode = ((Error.WithError) e).getError();
461         if  (errorCode == Error.ENOENT || errorCode == Error.ENOTDIR) {
462           return null;
463         }
464       }
465       throw e;
466     }
467   }

469   /**
470    * Version of stat that returns an inode if the input path stays entirely within
471    * this file system's scope, otherwise an {@link OutOfScopeFileStatus}.
472    *
473    * <p>Any code that calls this method needs to check for either possibility via
474    * {@link ScopeEscapableStatus#outOfScope}.
475    */
476   protected InMemoryContentInfo scopeLimitedStat(Path path, boolean followSymlinks)
477       throws IOException {
478     if (followSymlinks) {
479       return pathWalk(path, false);
480     } else {
481       if (path.equals(rootPath)) {
482         return rootInode;
483       } else {
484         InMemoryContentInfo status = getNoFollowStatOrOutOfScopeParent(path);
485         // If out of scope, status references the path's parent directory. Else it references the
486         // path itself.
487         return status.outOfScope()
488             ? new OutOfScopeFileStatus(status.getEscapingPath().getRelative(path.getBaseName()))
489             : status;
490       }
491     }
492   }

498   /**
499    * This is a helper routing for {@link #resolveSymbolicLinks(Path)}, i.e.
500    * the "user-mode" routing for canonicalising paths. It is analogous to the
501    * code in glibc's realpath(3).
502    *
503    * <p>Just like realpath, resolveSymbolicLinks requires a quadratic number of
504    * directory lookups: n path segments are statted, and each stat requires a
505    * linear amount of work in the "kernel" routine.
506    */
507   @Override
508   protected PathFragment resolveOneLink(Path path) throws IOException {
509     // Beware, this seemingly simple code belies the complex specification of
510     // FileSystem.resolveOneLink().
511     InMemoryContentInfo status = scopeLimitedStat(path, false);
512     if (status.outOfScope()) {
513       return resolveOneLinkWithDelegator(status.getEscapingPath());
514     } else {
515       return status.isSymbolicLink()
516           ? ((InMemoryLinkInfo) status).getLinkContent()
517           : null;
518     }
519   }

656   /**
657    * Returns a new path constructed by appending the child's base name to the
658    * escaped parent path. For example, assume our file system root is /foo
659    * and /foo/link1 -> /bar. This method can be used on child = /foo/link1/link2/name

667   @Override
668   protected boolean createDirectory(Path path) throws IOException {
669     if (path.equals(rootPath)) { throw Error.EACCES.exception(path); }
670 
671     InMemoryDirectoryInfo parent;
672     synchronized (this) {
673       parent = getDirectory(path.getParentDirectory());
674       if (!parent.outOfScope()) {
675         InMemoryContentInfo child = parent.getChild(path.getBaseName());
676         if (child != null) { // already exists
677           if (child.isDirectory()) {
678             return false;
679           } else {
680             throw Error.EEXIST.exception(path);
681           }
682         }
683 
684         InMemoryDirectoryInfo newDir = new InMemoryDirectoryInfo(clock);
685         newDir.addChild(".", newDir);
686         newDir.addChild("..", parent);
687         insert(parent, path.getBaseName(), newDir, path);
688 
689         return true;
690       }
691     }

694     return getDelegatedPath(parent.getEscapingPath(), path).createDirectory();

698   protected void createSymbolicLink(Path path, PathFragment targetFragment)
699       throws IOException {
700     if (path.equals(rootPath)) { throw Error.EACCES.exception(path); }
701 
702     InMemoryDirectoryInfo parent;
703     synchronized (this) {
704       parent = getDirectory(path.getParentDirectory());
705       if (!parent.outOfScope()) {
706         if (parent.getChild(path.getBaseName()) != null) { throw Error.EEXIST.exception(path); }
707         insert(parent, path.getBaseName(), new InMemoryLinkInfo(clock, targetFragment), path);
708         return;
709       }
710     }
711 
712     // If we get here, we're out of scope.
713     getDelegatedPath(parent.getEscapingPath(), path).createSymbolicLink(targetFragment);
714   }

737     InMemoryDirectoryInfo dirInfo;
738     synchronized (this) {
739       dirInfo = getDirectory(path);
740       if (!dirInfo.outOfScope()) {
741         FileStatus status = stat(path, false);
742         Preconditions.checkState(status instanceof InMemoryContentInfo);
743         if (!((InMemoryContentInfo) status).isReadable()) {
744           throw new IOException("Directory is not readable");
745         }
746 
747         Set<String> allChildren = dirInfo.getAllChildren();
748         List<Path> result = new ArrayList<>(allChildren.size());
749         for (String child : allChildren) {
750           if (!(child.equals(".") || child.equals(".."))) {
751             result.add(path.getChild(child));
752           }
753         }
754         return result;
755       }
756     }
757 
758     // If we get here, we're out of scope.
759     return getDelegatedPath(dirInfo.getEscapingPath()).getDirectoryEntries();
760   }

762   @Override
763   protected boolean delete(Path path) throws IOException {
764     if (path.equals(rootPath)) { throw Error.EBUSY.exception(path); }
765     if (!exists(path, false)) { return false; }
766 
767     InMemoryDirectoryInfo parent;
768     synchronized (this) {
769       parent = getDirectory(path.getParentDirectory());
770       if (!parent.outOfScope()) {
771         InMemoryContentInfo child = parent.getChild(path.getBaseName());
772         if (child.isDirectory() && child.getSize() > 2) { throw Error.ENOTEMPTY.exception(path); }
773         unlink(parent, path.getBaseName(), path);
774         return true;
775       }
776     }
777 
778     // If we get here, we're out of scope.
779     return getDelegatedPath(parent.getEscapingPath(), path).delete();
780   }

822   /**
823    * Creates a new file at the given path and returns its inode. If the path
824    * escapes this file system's scope, trivially returns an "out of scope" status.
825    * Calling code should check for both possibilities via
826    * {@link ScopeEscapableStatus#outOfScope}.
827    */
828   protected InMemoryContentInfo getOrCreateWritableInode(Path path)
829       throws IOException {
830     // open(WR_ONLY) of a dangling link writes through the link.  That means
831     // that the usual path lookup operations have to behave differently when
832     // resolving a path with the intent to create it: instead of failing with
833     // ENOENT they have to return an open file.  This is exactly how UNIX
834     // kernels do it, which is what we're trying to emulate.
835     InMemoryContentInfo child = pathWalk(path, /*create=*/true);
836     Preconditions.checkNotNull(child);
837     if (child.outOfScope()) {
838       return child;
839     } else if (child.isDirectory()) {
840       throw Error.EISDIR.exception(path);
841     } else { // existing or newly-created file
842       if (!child.isWritable()) { throw Error.EACCES.exception(path); }
843       return child;
844     }
845   }

847   @Override
848   protected OutputStream getOutputStream(Path path, boolean append)
849       throws IOException {
850     InMemoryContentInfo status;
851     synchronized (this) {
852       status = getOrCreateWritableInode(path);
853       if (!status.outOfScope()) {
854         return ((FileInfo) getOrCreateWritableInode(path)).getOutputStream(append);
855       }
856     }
857     // If we get here, we're out of scope.
858     return getDelegatedPath(status.getEscapingPath()).getOutputStream(append);
859   }

861   @Override
862   protected void renameTo(Path sourcePath, Path targetPath)
863       throws IOException {
864     if (sourcePath.equals(rootPath)) { throw Error.EACCES.exception(sourcePath); }
865     if (targetPath.equals(rootPath)) { throw Error.EACCES.exception(targetPath); }
866 
867     InMemoryDirectoryInfo sourceParent;
868     InMemoryDirectoryInfo targetParent;
869 
870     synchronized (this) {
871       sourceParent = getDirectory(sourcePath.getParentDirectory());
872       targetParent = getDirectory(targetPath.getParentDirectory());
873 
874       // Handle the rename if both paths are within our scope.
875       if (!sourceParent.outOfScope() && !targetParent.outOfScope()) {
876         InMemoryContentInfo sourceInode = sourceParent.getChild(sourcePath.getBaseName());
877         if (sourceInode == null) { throw Error.ENOENT.exception(sourcePath); }
878         InMemoryContentInfo targetInode = targetParent.getChild(targetPath.getBaseName());
879 
880         unlink(sourceParent, sourcePath.getBaseName(), sourcePath);
881         try {
882           // TODO(bazel-team): (2009) test with symbolic links.
883 
884           // Precondition checks:
885           if (targetInode != null) { // already exists
886             if (targetInode.isDirectory()) {
887               if (!sourceInode.isDirectory()) {
888                 throw new IOException(sourcePath + " -> " + targetPath + " (" + Error.EISDIR + ")");
889               }
890               if (targetInode.getSize() > 2) {
891                 throw Error.ENOTEMPTY.exception(targetPath);
892               }
893             } else if (sourceInode.isDirectory()) {
894               throw new IOException(sourcePath + " -> " + targetPath + " (" + Error.ENOTDIR + ")");
895             }
896             unlink(targetParent, targetPath.getBaseName(), targetPath);
897           }
898           sourceInode.movedTo(targetPath);
899           insert(targetParent, targetPath.getBaseName(), sourceInode, targetPath);
900           return;
901 
902         } catch (IOException e) {
903           sourceInode.movedTo(sourcePath);
904           insert(sourceParent, sourcePath.getBaseName(), sourceInode, sourcePath); // restore source
905           throw e;
906         }
907       }
908     }
909 
910     // If we get here, either one or both paths is out of scope.
911     if (sourceParent.outOfScope() && targetParent.outOfScope()) {
912       Path delegatedSource = getDelegatedPath(sourceParent.getEscapingPath(), sourcePath);
913       Path delegatedTarget = getDelegatedPath(targetParent.getEscapingPath(), targetPath);
914       delegatedSource.renameTo(delegatedTarget);
915     } else {
916       // We don't support cross-file system renaming.
917       throw Error.EACCES.exception(targetPath);
918     }
919   }
