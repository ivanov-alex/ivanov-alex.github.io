46  /**
47   * Runs TestRunnerAction actions.
48   */
49  @ExecutionStrategy(contextType = TestActionContext.class, name = { "standalone" })
50  public class StandaloneTestStrategy extends TestStrategy {
51    // TODO(bazel-team) - add tests for this strategy.
52    // TODO(bazel-team) - add support for test timeouts.
53  
54    private final Path workspace;
55  
56    public StandaloneTestStrategy(OptionsClassProvider requestOptions,
57        OptionsClassProvider startupOptions, BinTools binTools, Map<String, String> clientEnv,
58        Path workspace) {
59      super(requestOptions, startupOptions, binTools, clientEnv);
60      this.workspace = workspace;
61    }
62  
63    private static final String TEST_SETUP = "tools/test/test-setup.sh";
64  
65    @Override
66    public void exec(TestRunnerAction action, ActionExecutionContext actionExecutionContext)
67        throws ExecException, InterruptedException {
68      Path runfilesDir = null;
69      try {
70        runfilesDir = TestStrategy.getLocalRunfilesDirectory(
71            action, actionExecutionContext, binTools);
72      } catch (ExecException e) {
73        throw new TestExecException(e.getMessage());
74      }
75  
76      Path testTmpDir = TestStrategy.getTmpRoot(
77          workspace, actionExecutionContext.getExecutor().getExecRoot(), executionOptions)
78          .getChild(getTmpDirName(action.getExecutionSettings().getExecutable().getExecPath()));
79      Path workingDirectory = runfilesDir.getRelative(action.getRunfilesPrefix());
80  
81      TestRunnerAction.ResolvedPaths resolvedPaths =
82          action.resolve(actionExecutionContext.getExecutor().getExecRoot());
83      Map<String, String> env = getEnv(action, runfilesDir, testTmpDir, resolvedPaths);
84  
85      Map<String, String> info = new HashMap<>();
86  
87      // This key is only understood by StandaloneSpawnStrategy.
88      info.put("timeout", "" + getTimeout(action));
89      info.putAll(action.getTestProperties().getExecutionInfo());
90  
91      Spawn spawn =
92          new BaseSpawn(
93              // Bazel lacks much of the tooling for coverage, so we don't attempt to pass a coverage
94              // script here.
95              getArgs(TEST_SETUP, "", action),
96              env,
97              info,
98              new RunfilesSupplierImpl(
99                  runfilesDir.asFragment(), action.getExecutionSettings().getRunfiles()),
100             action,
101             action
102                 .getTestProperties()
103                 .getLocalResourceUsage(executionOptions.usingLocalTestJobs()));
104 
105     Executor executor = actionExecutionContext.getExecutor();
106 
107     try {
108       FileSystemUtils.createDirectoryAndParents(testTmpDir);
109     } catch (IOException e) {
110       executor.getEventHandler().handle(Event.error("Could not create TEST_TMPDIR: " + e));
111       throw new EnvironmentalExecException("Could not create TEST_TMPDIR " + testTmpDir, e);
112     }
113 
114     ResourceSet resources = null;
115     FileOutErr fileOutErr = null;
116     try {
117       FileSystemUtils.createDirectoryAndParents(workingDirectory);
118       fileOutErr = new FileOutErr(action.getTestLog().getPath(),
119           action.resolve(actionExecutionContext.getExecutor().getExecRoot()).getTestStderr());
120 
121       resources = action.getTestProperties()
122           .getLocalResourceUsage(executionOptions.usingLocalTestJobs());
123       ResourceManager.instance().acquireResources(action, resources);
124       TestResultData data = execute(
125           actionExecutionContext.withFileOutErr(fileOutErr), spawn, action);
126       appendStderr(fileOutErr.getOutputFile(), fileOutErr.getErrorFile());
127       finalizeTest(actionExecutionContext, action, data);
128     } catch (IOException e) {
129       executor.getEventHandler().handle(Event.error("Caught I/O exception: " + e));
130       throw new EnvironmentalExecException("unexpected I/O exception", e);
131     } finally {
132       if (resources != null) {
133         ResourceManager.instance().releaseResources(action, resources);
134       }
135       try {
136         if (fileOutErr != null) {
137           fileOutErr.close();
138         }
139       } catch (IOException e) {
140         // If the close fails, there is little we can do.
141       }
142     }
143   }

145   private Map<String, String> getEnv(
146       TestRunnerAction action,
147       Path runfilesDir,
148       Path tmpDir,
149       TestRunnerAction.ResolvedPaths resolvedPaths) {
150     Map<String, String> vars = getDefaultTestEnvironment(action);
151     BuildConfiguration config = action.getConfiguration();
152 
153     vars.putAll(config.getDefaultShellEnvironment());
154     vars.putAll(action.getTestEnv());
155 
156     /*
157      * TODO(bazel-team): the paths below are absolute,
158      * making test actions impossible to cache remotely.
159      */
160     vars.put("TEST_SRCDIR", runfilesDir.getPathString());
161     vars.put("TEST_TMPDIR", tmpDir.getPathString());
162     vars.put("XML_OUTPUT_FILE", resolvedPaths.getXmlOutputPath().getPathString());
163 
164     return vars;
165   }

167   private TestResultData execute(
168       ActionExecutionContext actionExecutionContext, Spawn spawn, TestRunnerAction action)
169       throws TestExecException, InterruptedException {
170     Executor executor = actionExecutionContext.getExecutor();
171     Closeable streamed = null;
172     Path testLogPath = action.getTestLog().getPath();
173     TestResultData.Builder builder = TestResultData.newBuilder();
174 
175     try {
176       try {
177         if (executionOptions.testOutput.equals(TestOutputFormat.STREAMED)) {
178           streamed = new StreamedTestOutput(
179               Reporter.outErrForReporter(
180                   actionExecutionContext.getExecutor().getEventHandler()), testLogPath);
181         }
182         executor.getSpawnActionContext(action.getMnemonic()).exec(spawn, actionExecutionContext);
183 
184         builder.setTestPassed(true)
185             .setStatus(BlazeTestStatus.PASSED)
186             .setCachable(true);
187       } catch (ExecException e) {
188         // Execution failed, which we consider a test failure.
189 
190         // TODO(bazel-team): set cachable==true for relevant statuses (failure, but not for
191         // timeout, etc.)
192         builder.setTestPassed(false)
193             .setStatus(BlazeTestStatus.FAILED);
194       } finally {
195         if (streamed != null) {
196           streamed.close();
197         }
198       }
199 
200       TestCase details = parseTestResult(
201           action.resolve(actionExecutionContext.getExecutor().getExecRoot()).getXmlOutputPath());
202       if (details != null) {
203         builder.setTestCase(details);
204       }
205 
206       return builder.build();
207     } catch (IOException e) {
208       throw new TestExecException(e.getMessage());
209     }
210   }

212   /**
213    * Outputs test result to the stdout after test has finished (e.g. for --test_output=all or
214    * --test_output=errors). Will also try to group output lines together (up to 10000 lines) so
215    * parallel test outputs will not get interleaved.
216    */
217   protected void processTestOutput(Executor executor, FileOutErr outErr, TestResult result)
218       throws IOException {
219     Path testOutput = executor.getExecRoot().getRelative(result.getTestLogPath().asFragment());
220     boolean isPassed = result.getData().getTestPassed();
221     try {
222       if (TestLogHelper.shouldOutputTestLog(executionOptions.testOutput, isPassed)) {
223         TestLogHelper.writeTestLog(testOutput, result.getTestName(), outErr.getOutputStream());
224       }
225     } finally {
226       if (isPassed) {
227         executor.getEventHandler().handle(new Event(EventKind.PASS, null, result.getTestName()));
228       } else {
229         if (result.getData().getStatus() == BlazeTestStatus.TIMEOUT) {
230           executor.getEventHandler().handle(
231               new Event(EventKind.TIMEOUT, null, result.getTestName()
232                   + " (see " + testOutput + ")"));
233         } else {
234           executor.getEventHandler().handle(
235               new Event(EventKind.FAIL, null, result.getTestName() + " (see " + testOutput + ")"));
236         }
237       }
238     }
239   }

241   private final void finalizeTest(ActionExecutionContext actionExecutionContext,
242       TestRunnerAction action, TestResultData data) throws IOException, ExecException {
243     TestResult result = new TestResult(action, data, false);
244     postTestResult(actionExecutionContext.getExecutor(), result);
245 
246     processTestOutput(actionExecutionContext.getExecutor(),
247         actionExecutionContext.getFileOutErr(), result);
248     // TODO(bazel-team): handle --test_output=errors, --test_output=all.
249 
250     if (!executionOptions.testKeepGoing && data.getStatus() != BlazeTestStatus.PASSED) {
251       throw new TestExecException("Test failed: aborting");
252     }
253   }

255   @Override
256   public String strategyLocality(TestRunnerAction action) { return "standalone"; }

258   @Override
259   public TestResult newCachedTestResult(
260       Path execRoot, TestRunnerAction action, TestResultData data) {
261     return new TestResult(action, data, /*cached*/ true);
262   }
