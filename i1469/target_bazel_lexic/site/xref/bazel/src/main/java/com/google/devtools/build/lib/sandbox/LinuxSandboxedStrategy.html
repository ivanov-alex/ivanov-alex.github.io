63  /**
64   * Strategy that uses sandboxing to execute a process.
65   */
66  @ExecutionStrategy(name = {"sandboxed"},
67                     contextType = SpawnActionContext.class)
68  public class LinuxSandboxedStrategy implements SpawnActionContext {
69    private final ExecutorService backgroundWorkers;
70  
71    private final BlazeRuntime blazeRuntime;
72    private final BlazeDirectories blazeDirs;
73    private final Path execRoot;
74    private final boolean verboseFailures;
75    private final StandaloneSpawnStrategy standaloneStrategy;
76    private final UUID uuid = UUID.randomUUID();
77    private final AtomicInteger execCounter = new AtomicInteger();
78  
79    public LinuxSandboxedStrategy(
80        BlazeRuntime blazeRuntime, boolean verboseFailures, ExecutorService backgroundWorkers) {
81      this.blazeRuntime = blazeRuntime;
82      this.blazeDirs = blazeRuntime.getDirectories();
83      this.execRoot = blazeDirs.getExecRoot();
84      this.verboseFailures = verboseFailures;
85      this.backgroundWorkers = backgroundWorkers;
86      this.standaloneStrategy = new StandaloneSpawnStrategy(blazeDirs.getExecRoot(), verboseFailures);
87    }

89    /**
90     * Executes the given {@code spawn}.
91     */
92    @Override
93    public void exec(Spawn spawn, ActionExecutionContext actionExecutionContext)
94        throws ExecException {
95      Executor executor = actionExecutionContext.getExecutor();
96  
97      // Certain actions can't run remotely or in a sandbox - pass them on to the standalone strategy.
98      if (!spawn.isRemotable()) {
99        standaloneStrategy.exec(spawn, actionExecutionContext);
100       return;
101     }
102 
103     if (executor.reportsSubcommands()) {
104       executor.reportSubcommand(
105           Label.print(spawn.getOwner().getLabel()) + " [" + spawn.getResourceOwner().prettyPrint()
106               + "]", spawn.asShellCommand(executor.getExecRoot()));
107     }
108 
109     FileOutErr outErr = actionExecutionContext.getFileOutErr();
110 
111     // The execId is a unique ID just for this invocation of "exec".
112     String execId = uuid + "-" + Integer.toString(execCounter.getAndIncrement());
113 
114     // Each invocation of "exec" gets its own sandbox.
115     Path sandboxPath =
116         execRoot.getRelative(Constants.PRODUCT_NAME + "-sandbox").getRelative(execId);
117 
118     ImmutableMultimap<Path, Path> mounts;
119     try {
120       // Gather all necessary mounts for the sandbox.
121       mounts = getMounts(spawn, sandboxPath, actionExecutionContext);
122     } catch (IOException e) {
123       throw new UserExecException("Could not prepare mounts for sandbox execution", e);
124     }
125 
126     int timeout = getTimeout(spawn);
127 
128     try {
129       final NamespaceSandboxRunner runner =
130           new NamespaceSandboxRunner(execRoot, sandboxPath, mounts, verboseFailures);
131       try {
132         runner.run(
133             spawn.getArguments(),
134             spawn.getEnvironment(),
135             blazeDirs.getExecRoot().getPathFile(),
136             outErr,
137             spawn.getOutputFiles(),
138             timeout);
139       } finally {
140         // Due to the Linux kernel behavior, if we try to remove the sandbox too quickly after the
141         // process has exited, we get "Device busy" errors because some of the mounts have not yet
142         // been undone. A second later it usually works. We will just clean the old sandboxes up
143         // using a background worker.
144         backgroundWorkers.execute(
145             new Runnable() {
146               @Override
147               public void run() {
148                 try {
149                   while (!Thread.currentThread().isInterrupted()) {
150                     try {
151                       runner.cleanup();
152                       return;
153                     } catch (IOException e2) {
154                       // Sleep & retry.
155                       Thread.sleep(250);
156                     }
157                   }
158                 } catch (InterruptedException e) {
159                   // Exit.
160                 }
161               }
162             });
163       }
164     } catch (CommandException e) {
165       EventHandler handler = actionExecutionContext.getExecutor().getEventHandler();
166       handler.handle(
167           Event.error("Sandboxed execution failed: " + spawn.getOwner().getLabel() + "."));
168       throw new UserExecException("Error during execution of spawn", e);
169     } catch (IOException e) {
170       EventHandler handler = actionExecutionContext.getExecutor().getEventHandler();
171       handler.handle(
172           Event.error(
173               "I/O error during sandboxed execution:\n" + Throwables.getStackTraceAsString(e)));
174       throw new UserExecException("Could not execute spawn", e);
175     }
176   }

178   private int getTimeout(Spawn spawn) throws UserExecException {
179     String timeoutStr = spawn.getExecutionInfo().get("timeout");
180     if (timeoutStr != null) {
181       try {
182         return Integer.parseInt(timeoutStr);
183       } catch (NumberFormatException e) {
184         throw new UserExecException("could not parse timeout: " + e);
185       }
186     }
187     return -1;
188   }

190   private ImmutableMultimap<Path, Path> getMounts(
191       Spawn spawn, Path sandboxPath, ActionExecutionContext actionExecutionContext)
192       throws IOException {
193     ImmutableMultimap.Builder<Path, Path> mounts = ImmutableMultimap.builder();
194     mounts.putAll(mountUsualUnixDirs(sandboxPath));
195     mounts.putAll(setupBlazeUtils(sandboxPath));
196     mounts.putAll(mountRunfilesFromManifests(spawn, sandboxPath));
197     mounts.putAll(mountRunfilesFromSuppliers(spawn, sandboxPath));
198     mounts.putAll(mountRunfilesForTests(spawn, sandboxPath));
199     mounts.putAll(mountInputs(spawn, sandboxPath, actionExecutionContext));
200     mounts.putAll(mountRunUnderCommand(spawn, sandboxPath));
201 
202     SetMultimap<Path, Path> fixedMounts = LinkedHashMultimap.create();
203     for (Entry<Path, Path> mount : mounts.build().entries()) {
204       Path source = mount.getKey();
205       Path target = mount.getValue();
206       validateAndAddMount(sandboxPath, fixedMounts, source, target);
207 
208       // Iteratively resolve symlinks and mount the whole chain. Take care not to run into a cyclic
209       // symlink - when we already processed the source once, we can exit the loop. Skyframe will
210       // catch cyclic symlinks for declared inputs, but this won't help if there is one in the parts
211       // of the host system that we mount.
212       Set<Path> seenSources = new HashSet<>();
213       while (source.isSymbolicLink() && seenSources.add(source)) {
214         source = source.getParentDirectory().getRelative(source.readSymbolicLink());
215         target = sandboxPath.getRelative(source.asFragment().relativeTo("/"));
216 
217         validateAndAddMount(sandboxPath, fixedMounts, source, target);
218       }
219     }
220     return ImmutableMultimap.copyOf(fixedMounts);
221   }

223   /**
224    * Adds the new mount ("source" -> "target") to "mounts" after doing some validations on it.
225    *
226    * @return true if the mount was added to the multimap, or false if the multimap already contained
227    *         the mount.
228    */
229   private static boolean validateAndAddMount(
230       Path sandboxPath, SetMultimap<Path, Path> mounts, Path source, Path target) {
231     // The source must exist.
232     Preconditions.checkArgument(source.exists(), source.toString() + " does not exist");
233 
234     // We cannot mount two different things onto the same target.
235     if (!mounts.containsEntry(source, target) && mounts.containsValue(target)) {
236       // There is a conflicting entry, find it and error out.
237       for (Entry<Path, Path> mount : mounts.entries()) {
238         if (mount.getValue().equals(target)) {
239           throw new IllegalStateException(
240               String.format(
241                   "Cannot mount both '%s' and '%s' onto '%s'", mount.getKey(), source, target));
242         }
243       }
244     }
245 
246     // Mounts must always mount into the sandbox, otherwise they might corrupt the host system.
247     Preconditions.checkArgument(
248         target.startsWith(sandboxPath),
249         String.format("(%s -> %s) does not mount into sandbox", source, target));
250 
251     return mounts.put(source, target);
252   }

254   /**
255    * Mount a certain set of unix directories to make the usual tools and libraries available to the
256    * spawn that runs.
257    */
258   private ImmutableMultimap<Path, Path> mountUsualUnixDirs(Path sandboxPath) throws IOException {
259     ImmutableMultimap.Builder<Path, Path> mounts = ImmutableMultimap.builder();
260     FileSystem fs = blazeDirs.getFileSystem();
261     mounts.put(fs.getPath("/bin"), sandboxPath.getRelative("bin"));
262     mounts.put(fs.getPath("/etc"), sandboxPath.getRelative("etc"));
263     for (String entry : FilesystemUtils.readdir("/")) {
264       if (entry.startsWith("lib")) {
265         mounts.put(fs.getRootDirectory().getRelative(entry), sandboxPath.getRelative(entry));
266       }
267     }
268     for (String entry : FilesystemUtils.readdir("/usr")) {
269       if (!entry.equals("local")) {
270         mounts.put(
271             fs.getPath("/usr").getRelative(entry),
272             sandboxPath.getRelative("usr").getRelative(entry));
273       }
274     }
275     return mounts.build();
276   }

278   /**
279    * Mount the embedded tools.
280    */
281   private ImmutableMultimap<Path, Path> setupBlazeUtils(Path sandboxPath) throws IOException {
282     ImmutableMultimap.Builder<Path, Path> mounts = ImmutableMultimap.builder();
283     Path source = blazeDirs.getEmbeddedBinariesRoot().getRelative("build-runfiles");
284     Path target = sandboxPath.getRelative(source.asFragment().relativeTo("/"));
285     mounts.put(source, target);
286     return mounts.build();
287   }

289   /**
290    * Mount all runfiles that the spawn needs as specified in its runfiles manifests.
291    */
292   private ImmutableMultimap<Path, Path> mountRunfilesFromManifests(Spawn spawn, Path sandboxPath)
293       throws IOException {
294     ImmutableMultimap.Builder<Path, Path> mounts = ImmutableMultimap.builder();
295     FileSystem fs = blazeDirs.getFileSystem();
296     for (Entry<PathFragment, Artifact> manifest : spawn.getRunfilesManifests().entrySet()) {
297       String manifestFilePath = manifest.getValue().getPath().getPathString();
298       Preconditions.checkState(!manifest.getKey().isAbsolute());
299       Path targetDirectory = execRoot.getRelative(manifest.getKey());
300       for (String line : Files.readLines(new File(manifestFilePath), Charset.defaultCharset())) {
301         String[] fields = line.split(" ");
302         Preconditions.checkState(
303             fields.length == 2, "'" + line + "' does not split into exactly 2 parts");
304         Path source = fs.getPath(fields[1]);
305         Path targetPath = targetDirectory.getRelative(fields[0]);
306         Path targetInSandbox = sandboxPath.getRelative(targetPath.asFragment().relativeTo("/"));
307         mounts.put(source, targetInSandbox);
308       }
309     }
310     return mounts.build();
311   }

313   /**
314    * Mount all runfiles that the spawn needs as specified via its runfiles suppliers.
315    */
316   private ImmutableMultimap<Path, Path> mountRunfilesFromSuppliers(Spawn spawn, Path sandboxPath)
317       throws IOException {
318     ImmutableMultimap.Builder<Path, Path> mounts = ImmutableMultimap.builder();
319     FileSystem fs = blazeDirs.getFileSystem();
320     Map<PathFragment, Map<PathFragment, Artifact>> rootsAndMappings =
321         spawn.getRunfilesSupplier().getMappings();
322     for (Entry<PathFragment, Map<PathFragment, Artifact>> rootAndMappings :
323         rootsAndMappings.entrySet()) {
324       PathFragment root = rootAndMappings.getKey();
325       if (root.isAbsolute()) {
326         root = root.relativeTo("/");
327       }
328       for (Entry<PathFragment, Artifact> mapping : rootAndMappings.getValue().entrySet()) {
329         Artifact sourceArtifact = mapping.getValue();
330         Path source = (sourceArtifact != null) ? sourceArtifact.getPath() : fs.getPath("/dev/null");
331 
332         Preconditions.checkArgument(!mapping.getKey().isAbsolute());
333         Path target = sandboxPath.getRelative(root.getRelative(mapping.getKey()));
334         mounts.put(source, target);
335       }
336     }
337     return mounts.build();
338   }

340   /**
341    * Tests are a special case and we have to mount the TEST_SRCDIR where the test expects it to be
342    * and also provide a TEST_TMPDIR to the test where it can store temporary files.
343    */
344   private ImmutableMultimap<Path, Path> mountRunfilesForTests(Spawn spawn, Path sandboxPath)
345       throws IOException {
346     ImmutableMultimap.Builder<Path, Path> mounts = ImmutableMultimap.builder();
347     FileSystem fs = blazeDirs.getFileSystem();
348     if (spawn.getEnvironment().containsKey("TEST_TMPDIR")) {
349       Path source = fs.getPath(spawn.getEnvironment().get("TEST_TMPDIR"));
350       Path target = sandboxPath.getRelative(source.asFragment().relativeTo("/"));
351       FileSystemUtils.createDirectoryAndParents(target);
352     }
353     return mounts.build();
354   }

356   /**
357    * Mount all inputs of the spawn.
358    */
359   private ImmutableMultimap<Path, Path> mountInputs(
360       Spawn spawn, Path sandboxPath, ActionExecutionContext actionExecutionContext)
361       throws IOException {
362     ImmutableMultimap.Builder<Path, Path> mounts = ImmutableMultimap.builder();
363 
364     List<ActionInput> inputs =
365         ActionInputHelper.expandMiddlemen(
366             spawn.getInputFiles(), actionExecutionContext.getMiddlemanExpander());
367 
368     if (spawn.getResourceOwner() instanceof CppCompileAction) {
369       CppCompileAction action = (CppCompileAction) spawn.getResourceOwner();
370       if (action.shouldScanIncludes()) {
371         inputs.addAll(action.getAdditionalInputs());
372       }
373     }
374 
375     for (ActionInput input : inputs) {
376       if (input.getExecPathString().contains("internal/_middlemen/")) {
377         continue;
378       }
379       Path source = execRoot.getRelative(input.getExecPathString());
380       Path target = sandboxPath.getRelative(source.asFragment().relativeTo("/"));
381       mounts.put(source, target);
382     }
383     return mounts.build();
384   }

386   /**
387    * If a --run_under= option is set and refers to a command via its path (as opposed to via its
388    * label), we have to mount this. Note that this is best effort and works fine for shell scripts
389    * and small binaries, but we can't track any further dependencies of this command.
390    *
391    * <p>If --run_under= refers to a label, it is automatically provided in the spawn's input files,
392    * so mountInputs() will catch that case.
393    */
394   private ImmutableMultimap<Path, Path> mountRunUnderCommand(Spawn spawn, Path sandboxPath) {
395     ImmutableMultimap.Builder<Path, Path> mounts = ImmutableMultimap.builder();
396 
397     if (spawn.getResourceOwner() instanceof TestRunnerAction) {
398       TestRunnerAction testRunnerAction = ((TestRunnerAction) spawn.getResourceOwner());
399       RunUnder runUnder = testRunnerAction.getExecutionSettings().getRunUnder();
400       if (runUnder != null && runUnder.getCommand() != null) {
401         PathFragment sourceFragment = new PathFragment(runUnder.getCommand());
402         Path source;
403         if (sourceFragment.isAbsolute()) {
404           source = blazeDirs.getFileSystem().getPath(sourceFragment);
405         } else if (blazeDirs.getExecRoot().getRelative(sourceFragment).exists()) {
406           source = blazeDirs.getExecRoot().getRelative(sourceFragment);
407         } else {
408           List<Path> searchPath =
409               SearchPath.parse(blazeDirs.getFileSystem(), blazeRuntime.getClientEnv().get("PATH"));
410           source = SearchPath.which(searchPath, runUnder.getCommand());
411         }
412         if (source != null) {
413           Path target = sandboxPath.getRelative(source.asFragment().relativeTo("/"));
414           mounts.put(source, target);
415         }
416       }
417     }
418     return mounts.build();
419   }
