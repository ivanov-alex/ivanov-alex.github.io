55  /**
56   * The implementation of AnalysisEnvironment used for analysis. It tracks metadata for each
57   * configured target, such as the errors and warnings emitted by that target. It is intended that
58   * a separate instance is used for each configured target, so that these don't mix up.
59   */
60  public class CachingAnalysisEnvironment implements AnalysisEnvironment {
61    private final ArtifactFactory artifactFactory;
62  
63    private final ArtifactOwner owner;
64    /**
65     * If this is the system analysis environment, then errors and warnings are directly reported
66     * to the global reporter, rather than stored, i.e., we don't track here whether there are any
67     * errors.
68     */
69    private final boolean isSystemEnv;
70    private final boolean extendedSanityChecks;
71  
72    /**
73     * If false, no actions will be registered, they'll all be just dropped.
74     *
75     * <p>Usually, an analysis environment should register all actions. However, in some scenarios we
76     * analyze some targets twice, but the first one only serves the purpose of collecting information
77     * for the second analysis. In this case we don't register actions created by the first pass in
78     * order to avoid action conflicts.
79     */
80    private final boolean allowRegisteringActions;
81  
82    private boolean enabled = true;
83    private MiddlemanFactory middlemanFactory;
84    private EventHandler errorEventListener;
85    private SkyFunction.Environment skyframeEnv;
86    private Map<Artifact, String> artifacts;
87    private final BinTools binTools;
88  
89    /**
90     * The list of actions registered by the configured target this analysis environment is
91     * responsible for. May get cleared out at the end of the analysis of said target.
92     */
93    final List<Action> actions = new ArrayList<>();
94  
95    public CachingAnalysisEnvironment(ArtifactFactory artifactFactory,
96        ArtifactOwner owner, boolean isSystemEnv, boolean extendedSanityChecks,
97        EventHandler errorEventListener, SkyFunction.Environment env, boolean allowRegisteringActions,
98        BinTools binTools) {
99      this.artifactFactory = artifactFactory;
100     this.owner = Preconditions.checkNotNull(owner);
101     this.isSystemEnv = isSystemEnv;
102     this.extendedSanityChecks = extendedSanityChecks;
103     this.errorEventListener = errorEventListener;
104     this.skyframeEnv = env;
105     this.allowRegisteringActions = allowRegisteringActions;
106     this.binTools = binTools;
107     middlemanFactory = new MiddlemanFactory(artifactFactory, this);
108     artifacts = new HashMap<>();
109   }
110 
111   public void disable(Target target) {
112     if (!hasErrors() && allowRegisteringActions) {
113       verifyGeneratedArtifactHaveActions(target);
114     }
115     artifacts = null;
116     middlemanFactory = null;
117     enabled = false;
118     errorEventListener = null;
119     skyframeEnv = null;
120   }

122   private static StringBuilder shortDescription(Action action) {
123     if (action == null) {
124       return new StringBuilder("null Action");
125     }
126     return new StringBuilder()
127       .append(action.getClass().getName())
128       .append(' ')
129       .append(action.getMnemonic());
130   }
131 
132   /**
133    * Sanity checks that all generated artifacts have a generating action.
134    * @param target for error reporting
135    */
136   public void verifyGeneratedArtifactHaveActions(Target target) {
137     Collection<String> orphanArtifacts = getOrphanArtifactMap().values();
138     List<String> checkedActions = null;
139     if (!orphanArtifacts.isEmpty()) {
140       checkedActions = Lists.newArrayListWithCapacity(actions.size());
141       for (Action action : actions) {
142         StringBuilder sb = shortDescription(action);
143         for (Artifact o : action.getOutputs()) {
144           sb.append("\n    ");
145           sb.append(o.getExecPathString());
146         }
147         checkedActions.add(sb.toString());
148       }
149       throw new IllegalStateException(
150           String.format(
151               "%s %s : These artifacts do not have a generating action:\n%s\n"
152               + "These actions were checked:\n%s\n",
153               target.getTargetKind(), target.getLabel(),
154               Joiner.on('\n').join(orphanArtifacts), Joiner.on('\n').join(checkedActions)));
155     }
156   }

158   @Override
159   public ImmutableSet<Artifact> getOrphanArtifacts() {
160     return ImmutableSet.copyOf(getOrphanArtifactMap().keySet());
161   }
162 
163   private Map<Artifact, String> getOrphanArtifactMap() {
164     // Construct this set to avoid poor performance under large --runs_per_test.
165     Set<Artifact> artifactsWithActions = new HashSet<>();
166     for (Action action : actions) {
167       // Don't bother checking that every Artifact only appears once; that test is performed
168       // elsewhere (see #testNonUniqueOutputs in ActionListenerIntegrationTest).
169       artifactsWithActions.addAll(action.getOutputs());
170     }
171     // The order of the artifacts.entrySet iteration is unspecified - we use a TreeMap here to
172     // guarantee that the return value of this method is deterministic.
173     Map<Artifact, String> orphanArtifacts = new TreeMap<>(Artifact.EXEC_PATH_COMPARATOR);
174     for (Map.Entry<Artifact, String> entry : artifacts.entrySet()) {
175       Artifact a = entry.getKey();
176       if (!a.isSourceArtifact() && !artifactsWithActions.contains(a)) {
177         orphanArtifacts.put(a, String.format("%s\n%s",
178             a.getExecPathString(),  // uncovered artifact
179             entry.getValue()));  // origin of creation
180       }
181     }
182     return orphanArtifacts;
183   }

190   @Override
191   public boolean hasErrors() {
192     // The system analysis environment never has errors.
193     if (isSystemEnv) {
194       return false;
195     }
196     Preconditions.checkState(enabled);
197     return ((StoredEventHandler) errorEventListener).hasErrors();
198   }

206   /**
207    * Keeps track of artifacts. We check that all of them have an owner when the environment is
208    * sealed (disable()). For performance reasons we only track the originating stacktrace when
209    * running with --experimental_extended_sanity_checks.
210    */
211   private Artifact trackArtifactAndOrigin(Artifact a, @Nullable Throwable e) {
212     if ((e != null) && !artifacts.containsKey(a)) {
213       StringWriter sw = new StringWriter();
214       e.printStackTrace(new PrintWriter(sw));
215       artifacts.put(a, sw.toString());
216     } else {
217       artifacts.put(a, "No origin, run with --experimental_extended_sanity_checks");
218     }
219     return a;
220   }

222   @Override
223   public Artifact getDerivedArtifact(PathFragment rootRelativePath, Root root) {
224     Preconditions.checkState(enabled);
225     return trackArtifactAndOrigin(
226         artifactFactory.getDerivedArtifact(rootRelativePath, root, getOwner()),
227         extendedSanityChecks ? new Throwable() : null);
228   }

249   @Override
250   public void registerAction(Action... actions) {
251     Preconditions.checkState(enabled);
252     if (allowRegisteringActions) {
253       Collections.addAll(this.actions, actions);
254     }
255   }
256 
257   @Override
258   public Action getLocalGeneratingAction(Artifact artifact) {
259     Preconditions.checkState(allowRegisteringActions);
260     for (Action action : actions) {
261       if (action.getOutputs().contains(artifact)) {
262         return action;
263       }
264     }
265     return null;
266   }

290   // See SkyframeBuildView#getWorkspaceStatusValues for the code that this method is attempting to
291   // verify.
292   private NullPointerException collectDebugInfoAndCrash(
293       BuildInfoKey key, BuildConfiguration config) {
294     String debugInfo = key + " " + config;
295     Preconditions.checkState(skyframeEnv.valuesMissing(), debugInfo);
296     Map<BuildInfoKey, BuildInfoFactory> buildInfoFactories = Preconditions.checkNotNull(
297         PrecomputedValue.BUILD_INFO_FACTORIES.get(skyframeEnv), debugInfo);
298     BuildInfoFactory buildInfoFactory =
299         Preconditions.checkNotNull(buildInfoFactories.get(key), debugInfo);
300     Preconditions.checkState(buildInfoFactory.isEnabled(config), debugInfo);
301     throw new NullPointerException("BuildInfoCollectionValue shouldn't have been null");
302   }
303 
304   @Override
305   public ImmutableList<Artifact> getBuildInfo(RuleContext ruleContext, BuildInfoKey key) {
306     boolean stamp = AnalysisUtils.isStampingEnabled(ruleContext);
307     BuildInfoCollectionValue collectionValue =
308         (BuildInfoCollectionValue) skyframeEnv.getValue(BuildInfoCollectionValue.key(
309             new BuildInfoCollectionValue.BuildInfoKeyAndConfig(
310                 key, ruleContext.getConfiguration())));
311     if (collectionValue == null) {
312       throw collectDebugInfoAndCrash(key, ruleContext.getConfiguration());
313     }
314     BuildInfoCollection collection = collectionValue.getCollection();
315    return stamp ? collection.getStampedBuildInfo() : collection.getRedactedBuildInfo();
316   }
