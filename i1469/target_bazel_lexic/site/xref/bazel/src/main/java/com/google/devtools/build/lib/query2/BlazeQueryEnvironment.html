69    /**
70     * Note that the correct operation of this class critically depends on the Reporter being a
71     * singleton object, shared by all cooperating classes contributing to Query.
72     * @param strictScope if true, fail the whole query if a label goes out of scope.
73     * @param loadingPhaseThreads the number of threads to use during loading
74     *     the packages for the query.
75     * @param labelFilter a predicate that determines if a specific label is
76     *     allowed to be visited during query execution. If it returns false,
77     *     the query execution is stopped with an error message.
78     * @param settings a set of enabled settings
79     */
80    BlazeQueryEnvironment(TransitivePackageLoader transitivePackageLoader,
81        PackageProvider packageProvider,
82        TargetPatternEvaluator targetPatternEvaluator,
83        boolean keepGoing,
84        boolean strictScope,
85        int loadingPhaseThreads,
86        Predicate<Label> labelFilter,
87        EventHandler eventHandler,
88        Set<Setting> settings,
89        Iterable<QueryFunction> extraFunctions) {
90      super(keepGoing, strictScope, labelFilter, eventHandler, settings, extraFunctions);
91      this.targetPatternEvaluator = targetPatternEvaluator;
92      this.transitivePackageLoader = transitivePackageLoader;
93      this.targetProvider = packageProvider;
94      this.errorObserver = new ErrorPrintingTargetEdgeErrorObserver(this.eventHandler);
95      this.loadingPhaseThreads = loadingPhaseThreads;
96      this.labelVisitor = new LabelVisitor(packageProvider, dependencyFilter);
97    }

99    @Override
100   public BlazeQueryEvalResult<Target> evaluateQuery(QueryExpression expr)
101       throws QueryException, InterruptedException {
102     // Some errors are reported as QueryExceptions and others as ERROR events (if --keep_going). The
103     // result is set to have an error iff there were errors emitted during the query, so we reset
104     // errors here.
105     eventHandler.resetErrors();
106     QueryEvalResult<Target> queryEvalResult = super.evaluateQuery(expr);
107     return new BlazeQueryEvalResult<>(queryEvalResult.getSuccess(), queryEvalResult.getResultSet(),
108         graph);
109   }

111   @Override
112   public Set<Target> getTargetsMatchingPattern(QueryExpression caller,
113       String pattern) throws QueryException {
114     // We can safely ignore the boolean error flag. The evaluateQuery() method above wraps the
115     // entire query computation in an error sensor.
116 
117     Set<Target> targets = new LinkedHashSet<>(resolvedTargetPatterns.get(pattern));
118 
119     // Sets.filter would be more convenient here, but can't deal with exceptions.
120     Iterator<Target> targetIterator = targets.iterator();
121     while (targetIterator.hasNext()) {
122       Target target = targetIterator.next();
123       if (!validateScope(target.getLabel(), strictScope)) {
124         targetIterator.remove();
125       }
126     }
127 
128     Set<PathFragment> packages = new HashSet<>();
129     for (Target target : targets) {
130       packages.add(target.getLabel().getPackageFragment());
131     }
132 
133     Set<Target> result = new LinkedHashSet<>();
134     for (Target target : targets) {
135       result.add(getOrCreate(target));
136 
137       // Preservation of graph order: it is important that targets obtained via
138       // a wildcard such as p:* are correctly ordered w.r.t. each other, so to
139       // ensure this, we add edges between any pair of directly connected
140       // targets in this set.
141       if (target instanceof OutputFile) {
142         OutputFile outputFile = (OutputFile) target;
143         if (targets.contains(outputFile.getGeneratingRule())) {
144           makeEdge(outputFile, outputFile.getGeneratingRule());
145         }
146       } else if (target instanceof Rule) {
147         Rule rule = (Rule) target;
148         for (Label label : rule.getLabels(dependencyFilter)) {
149           if (!packages.contains(label.getPackageFragment())) {
150             continue;  // don't cause additional package loading
151           }
152           try {
153             if (!validateScope(label, strictScope)) {
154               continue;  // Don't create edges to targets which are out of scope.
155             }
156             Target to = getTargetOrThrow(label);
157             if (targets.contains(to)) {
158               makeEdge(rule, to);
159             }
160           } catch (NoSuchThingException e) {
161             /* ignore */
162           } catch (InterruptedException e) {
163             throw new QueryException("interrupted");
164           }
165         }
166       }
167     }
168     return result;
169   }

171   public Target getTarget(Label label) throws TargetNotFoundException, QueryException {
172     // Can't use strictScope here because we are expecting a target back.
173     validateScope(label, true);
174     try {
175       return getNode(getTargetOrThrow(label)).getLabel();
176     } catch (NoSuchThingException e) {
177       throw new TargetNotFoundException(e);
178     } catch (InterruptedException e) {
179       throw new QueryException("interrupted");
180     }
181   }

187   private Collection<Node<Target>> getNodes(Iterable<Target> target) {
188     Set<Node<Target>> result = new LinkedHashSet<>();
189     for (Target t : target) {
190       result.add(getNode(t));

192     return result;

218   @Override
219   public Set<Target> getTransitiveClosure(Set<Target> targetNodes) {
220     for (Target node : targetNodes) {
221       checkBuilt(node);
222     }
223     return getTargetsFromNodes(graph.getFwdReachable(getNodes(targetNodes)));
224   }

226   /**
227    * Checks that the graph rooted at 'targetNode' has been completely built;
228    * fails if not.  Callers of {@link #getTransitiveClosure} must ensure that
229    * {@link #buildTransitiveClosure} has been called before.
230    *
231    * <p>It would be inefficient and failure-prone to make getTransitiveClosure
232    * call buildTransitiveClosure directly.  Also, it would cause
233    * nondeterministic behavior of the operators, since the set of packages
234    * loaded (and hence errors reported) would depend on the ordering details of
235    * the query operators' implementations.
236    */
237   private void checkBuilt(Target targetNode) {
238     Preconditions.checkState(
239         labelVisitor.hasVisited(targetNode.getLabel()),
240         "getTransitiveClosure(%s) called without prior call to buildTransitiveClosure()",
241         targetNode);
242   }
243 
244   @Override
245   public void buildTransitiveClosure(QueryExpression caller,
246                                      Set<Target> targetNodes,
247                                      int maxDepth) throws QueryException, InterruptedException {
248     Set<Target> targets = targetNodes;
249     preloadTransitiveClosure(targets, maxDepth);
250     labelVisitor.syncWithVisitor(eventHandler, targets, keepGoing,
251         loadingPhaseThreads, maxDepth, errorObserver, new GraphBuildingObserver());
252 
253     if (errorObserver.hasErrors()) {
254       reportBuildFileError(caller, "errors were encountered while computing transitive closure");
255     }
256   }

258   @Override

260     return getTargetsFromNodes(graph.getShortestPath(getNode(from), getNode(to)));
261   }

263   private void preloadTransitiveClosure(Set<Target> targets, int maxDepth) throws QueryException {
264     if (maxDepth >= MAX_DEPTH_FULL_SCAN_LIMIT && transitivePackageLoader != null) {
265       // Only do the full visitation if "maxDepth" is large enough. Otherwise, the benefits of
266       // preloading will be outweighed by the cost of doing more work than necessary.
267       try {
268         transitivePackageLoader.sync(eventHandler, targets, ImmutableSet.<Label>of(), keepGoing,
269             loadingPhaseThreads, -1);
270       } catch (InterruptedException e) {
271         throw new QueryException("interrupted");
272       }
273     }
274   }

276   /**
277    * It suffices to synchronize the modifications of this.graph from within the
278    * GraphBuildingObserver, because that's the only concurrent part.
279    * Concurrency is always encapsulated within the evaluation of a single query
280    * operator (e.g. deps(), somepath(), etc).
281    */
282   private class GraphBuildingObserver implements TargetEdgeObserver {
283 
284     @Override
285     public synchronized void edge(Target from, Attribute attribute, Target to) {
286       Preconditions.checkState(attribute == null ||
287           dependencyFilter.apply(((Rule) from), attribute),
288           "Disallowed edge from LabelVisitor: %s --> %s", from, to);
289       makeEdge(from, to);
290     }
291 
292     @Override
293     public synchronized void node(Target node) {
294       graph.createNode(node);
295     }
296 
297     @Override
298     public void missingEdge(Target target, Label to, NoSuchThingException e) {
299       // No - op.
300     }
301   }
302 
303   private void makeEdge(Target from, Target to) {
304     graph.addEdge(from, to);
305   }

307   private Target getTargetOrThrow(Label label)
308       throws NoSuchThingException, SkyframeRestartQueryException, InterruptedException {
309     Target target = targetProvider.getTarget(eventHandler, label);
310     if (target == null) {
311       throw new SkyframeRestartQueryException();
312     }
313     return target;
314   }

316   // TODO(bazel-team): rename this to getDependentFiles when all implementations
317   // of QueryEnvironment is fixed.
318   @Override
319   public Set<Target> getBuildFiles(final QueryExpression caller, Set<Target> nodes)
320       throws QueryException {
321     Set<Target> dependentFiles = new LinkedHashSet<>();
322     Set<Package> seenPackages = new HashSet<>();
323     // Keep track of seen labels, to avoid adding a fake subinclude label that also exists as a
324     // real target.
325     Set<Label> seenLabels = new HashSet<>();
326 
327     // Adds all the package definition files (BUILD files and build
328     // extensions) for package "pkg", to "buildfiles".
329     for (Target x : nodes) {
330       Package pkg = x.getPackage();
331       if (seenPackages.add(pkg)) {
332         addIfUniqueLabel(getNode(pkg.getBuildFile()), seenLabels, dependentFiles);
333         for (Label subinclude
334             : Iterables.concat(pkg.getSubincludeLabels(), pkg.getSkylarkFileDependencies())) {
335           addIfUniqueLabel(getSubincludeTarget(subinclude, pkg), seenLabels, dependentFiles);
336 
337           // Also add the BUILD file of the subinclude.
338           try {
339             addIfUniqueLabel(getSubincludeTarget(
340                 subinclude.getLocalTargetLabel("BUILD"), pkg), seenLabels, dependentFiles);
341           } catch (Label.SyntaxException e) {
342             throw new AssertionError("BUILD should always parse as a target name", e);
343           }
344         }
345       }
346     }
347     return dependentFiles;
348   }

350   private static final Function<ResolvedTargets<Target>, Set<Target>> RESOLVED_TARGETS_TO_TARGETS =
351       new Function<ResolvedTargets<Target>, Set<Target>>() {
352         @Override
353         public Set<Target> apply(ResolvedTargets<Target> resolvedTargets) {
354           return resolvedTargets.getTargets();
355         }
356       };
357 
358   protected Map<String, Set<Target>> preloadOrThrow(
359       QueryExpression caller, Collection<String> patterns) throws TargetParsingException {
360     try {
361       // Note that this may throw a RuntimeException if deps are missing in Skyframe and this is
362       // being called from within a SkyFunction.
363       return Maps.transformValues(
364           targetPatternEvaluator.preloadTargetPatterns(eventHandler, patterns, keepGoing),
365           RESOLVED_TARGETS_TO_TARGETS);
366     } catch (InterruptedException e) {
367       // TODO(bazel-team): Propagate the InterruptedException from here [skyframe-loading].
368       throw new TargetParsingException("interrupted");
369     }
370   }
371 
372   private static void addIfUniqueLabel(Node<Target> node, Set<Label> labels, Set<Target> nodes) {
373     if (labels.add(node.getLabel().getLabel())) {
374       nodes.add(node.getLabel());
375     }
376   }

378   private Node<Target> getSubincludeTarget(final Label label, Package pkg) {
379     return getNode(new FakeSubincludeTarget(label, pkg));
380   }

387   /** Given a set of target nodes, returns the targets. */
388   private static Set<Target> getTargetsFromNodes(Iterable<Node<Target>> input) {
389     Set<Target> result = new LinkedHashSet<>();
390     for (Node<Target> node : input) {
391       result.add(node.getLabel());
392     }
393     return result;
