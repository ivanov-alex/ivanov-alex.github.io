85    /**
86     * Merges several plist files into a single {@code NSDictionary}. Each file should be a plist (of
87     * one of these formats: ASCII, Binary, or XML) that contains an NSDictionary.
88     */
89    @VisibleForTesting
90    static NSDictionary merge(Iterable<? extends Path> sourceFilePaths) throws IOException {
91      NSDictionary result = new NSDictionary();
92      for (Path sourceFilePath : sourceFilePaths) {
93        result.putAll(readPlistFile(sourceFilePath));
94      }
95      return result;
96    }
97  
98    public static NSDictionary readPlistFile(final Path sourceFilePath) throws IOException {
99      ByteSource rawBytes = new Utf8BomSkippingByteSource(sourceFilePath);
100 
101     try {
102       try (InputStream in = rawBytes.openStream()) {
103         return (NSDictionary) PropertyListParser.parse(in);
104       } catch (PropertyListFormatException | ParseException e) {
105         // If we failed to parse, the plist may implicitly be a map. To handle this, wrap the plist
106         // with {}.
107         // TODO(bazel-team): Do this in a cleaner way.
108         ByteSource concatenated = ByteSource.concat(
109             ByteSource.wrap(new byte[] {'{'}),
110             rawBytes,
111             ByteSource.wrap(new byte[] {'}'}));
112         try (InputStream in = concatenated.openStream()) {
113           return (NSDictionary) PropertyListParser.parse(in);
114         }
115       }
116     } catch (PropertyListFormatException | ParseException | ParserConfigurationException
117         | SAXException e) {
118       throw new IOException(e);
119     }
120   }

122   /**
123    * Writes the results of a merge operation to a plist file.
124    * @param plistPath the path of the plist to write in binary format
125    */
126   public void writePlist(Path plistPath) throws IOException {
127     try (OutputStream out = Files.newOutputStream(plistPath)) {
128       BinaryPropertyListWriter.write(out, merged);
129     }
130   }

132   /**
133    * Writes a PkgInfo file based on certain keys in the merged plist.
134    * @param pkgInfoPath the path of the PkgInfo file to write. In many iOS apps, this file just
135    *     contains the raw string {@code APPL????}.
136    */
137   public void writePkgInfo(Path pkgInfoPath) throws IOException {
138     String pkgInfo =
139         Mapping.of(merged, "CFBundlePackageType").or(NSObject.wrap("APPL")).toString()
140         + Mapping.of(merged, "CFBundleSignature").or(NSObject.wrap("????")).toString();
141     Files.write(pkgInfoPath, pkgInfo.getBytes(StandardCharsets.UTF_8));
142   }
143 
144   /** Invokes {@link #writePlist(Path)} and {@link #writePkgInfo(Path)}. */
145   public void write(Path plistPath, Path pkgInfoPath) throws IOException {
146     writePlist(plistPath);
147     writePkgInfo(pkgInfoPath);
148   }

150   /**
151    * Returns a map containing entries that should be added to the merged plist. These are usually
152    * generated by Xcode automatically during the build process.
153    */
154   public static Map<String, NSObject> automaticEntries(
155       Collection<String> targetedDeviceFamily, Platform platform, String sdkVersion,
156       String minimumOsVersion) {
157     ImmutableMap.Builder<String, NSObject> result = new ImmutableMap.Builder<>();
158     List<Integer> uiDeviceFamily = FluentIterable.from(targetedDeviceFamily)
159         .transform(Maps.asConverter(DEVICE_FAMILIES))
160         .toSortedList(Ordering.natural());
161 
162     result.put("UIDeviceFamily", NSObject.wrap(uiDeviceFamily.toArray()));
163     result.put("DTPlatformName", NSObject.wrap(platform.getLowerCaseNameInPlist()));
164     result.put("DTSDKName", NSObject.wrap(platform.getLowerCaseNameInPlist() + sdkVersion));
165     result.put("CFBundleSupportedPlatforms", new NSArray(NSObject.wrap(platform.getNameInPlist())));
166     result.put("MinimumOSVersion", NSObject.wrap(minimumOsVersion));
167 
168     // Adding these here by brute force to override values inserted by
169     // release bundling support which are wrong.
170     // These shall be removed once the errors in release bundling support are fixed.
171     // These values are taken from an Xcode 6.2 device build.
172     // Adding them to a simulator build causes no problems when tested with Xcode 6.2.
173     result.put("DTPlatformBuild", NSObject.wrap("12D508"));
174     result.put("DTSDKBuild", NSObject.wrap("12D508"));
175     result.put("DTXcode", NSObject.wrap("0620"));
176     result.put("DTXcodeBuild", NSObject.wrap("6C131e"));
177     return result.build();
178   }

180   /**
181    * Generates final merged Plist file and PkgInfo file in the specified locations, and includes the
182    * "automatic" entries in the Plist.
183    */
184   public static PlistMerging from(List<Path> sourceFiles, Map<String, NSObject> automaticEntries,
185       Map<String, String> substitutions, KeysToRemoveIfEmptyString keysToRemoveIfEmptyString)
186           throws IOException {
187     NSDictionary merged = PlistMerging.merge(sourceFiles);
188 
189     // Remove the entries that we have added by brute force in automaticEntries so that our
190     // check for conflictingEntries still works.
191     // This shall be removed once the errors in release bundling support are fixed.
192     Set<String> conflictingEntries = Sets.difference(
193         Sets.intersection(automaticEntries.keySet(), merged.keySet()),
194         ImmutableSet.of("DTPlatformBuild", "DTSDKBuild", "DTXcode", "DTXcodeBuild"));
195 
196     Preconditions.checkArgument(conflictingEntries.isEmpty(),
197         "The following plist entries are generated automatically, but are present in more than one "
198             + "of the input lists: %s", conflictingEntries);
199     merged.putAll(automaticEntries);
200 
201     for (Map.Entry<String, NSObject> entry : merged.entrySet()) {
202       if (entry.getValue().toJavaObject() instanceof String) {
203         String newValue = substituteEnvironmentVariable(
204             substitutions, (String) entry.getValue().toJavaObject());
205         merged.put(entry.getKey(), newValue);
206       }
207     }
208 
209     for (String key : keysToRemoveIfEmptyString) {
210       if (Equaling.of(Mapping.of(merged, key), Optional.<NSObject>of(new NSString("")))) {
211         merged.remove(key);
212       }
213     }
214 
215     return new PlistMerging(merged);
216   }

218   // Assume that if an RFC 1034 format string is specified, the value is RFC 1034 compliant.
219   private static String substituteEnvironmentVariable(
220       Map<String, String> substitutions, String string) {
221     // The substitution is *not* performed recursively.
222     for (Map.Entry<String, String> variable : substitutions.entrySet()) {
223       for (String variableNameWithFormatString : withFormatStrings(variable.getKey())) {
224         string = string
225             .replace("${" + variableNameWithFormatString + "}", variable.getValue())
226             .replace("$(" + variableNameWithFormatString + ")", variable.getValue());
227       }
228     }
229 
230     return string;
231   }
232 
233   private static ImmutableSet<String> withFormatStrings(String variableName) {
234     return ImmutableSet.of(variableName, variableName + ":rfc1034identifier");
235   }

242   /**
243    * Sets the given executable name on this merged plist in the {@code CFBundleExecutable}
244    * attribute.
245    *
246    * @param executableName name of the bundle executable
247    * @return this plist merging
248    * @throws ValidationException if the plist already contains an incompatible
249    *    {@code CFBundleExecutable} entry
250    */
251   public PlistMerging setExecutableName(String executableName) {
252     NSString bundleExecutable = (NSString) merged.get("CFBundleExecutable");
253 
254     if (bundleExecutable == null) {
255       merged.put("CFBundleExecutable", executableName);
256     } else if (!executableName.equals(bundleExecutable.getContent())) {
257       throw new ValidationException(String.format(
258           "Blaze generated the executable %s but the Plist CFBundleExecutable is %s",
259           executableName, bundleExecutable));
260     }
261 
262     return this;
263   }
264   
265   /**
266    * Sets the given identifier on this merged plist in the {@code CFBundleIdentifier}
267    * attribute.
268    *
269    * @param primaryIdentifier used to set the bundle identifier or override the existing one from
270    *     plist file, can be null
271    * @param fallbackIdentifier used to set the bundle identifier if it is not set by plist file or
272    *     primary identifier, can be null
273    * @return this plist merging
274    */
275   public PlistMerging setBundleIdentifier(String primaryIdentifier, String fallbackIdentifier) {
276     NSString bundleIdentifier = (NSString) merged.get(BUNDLE_IDENTIFIER_PLIST_KEY);
277         
278     if (primaryIdentifier != null) {
279       merged.put(BUNDLE_IDENTIFIER_PLIST_KEY, primaryIdentifier);
280     } else if (bundleIdentifier == null && fallbackIdentifier != null) {
281       merged.put(BUNDLE_IDENTIFIER_PLIST_KEY, fallbackIdentifier);
282     }
283 
284     return this;
285   }

287   private static class Utf8BomSkippingByteSource extends ByteSource {
288 
289     private static final byte[] UTF8_BOM =
290         new byte[] { (byte) 0xEF, (byte) 0xBB, (byte) 0xBF };
291 
292     private final Path path;
293 
294     public Utf8BomSkippingByteSource(Path path) {
295       this.path = path;
296     }
297 
298     @Override
299     public InputStream openStream() throws IOException {
300       InputStream stream = new BufferedInputStream(Files.newInputStream(path));
301       stream.mark(UTF8_BOM.length);
302       byte[] buffer = new byte[UTF8_BOM.length];
303       int read = stream.read(buffer);
304       stream.reset();
305 
306       if (UTF8_BOM.length == read && Arrays.equals(buffer, UTF8_BOM)) {
307         stream.skip(UTF8_BOM.length);
308       }
309 
310       return stream;
311     }
312   }
