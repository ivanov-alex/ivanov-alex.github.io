58  /**
59   * Implements the loading phase; responsible for:
60   * <ul>
61   *   <li>target pattern evaluation
62   *   <li>test suite expansion
63   *   <li>loading the labels needed to construct the build configuration
64   *   <li>loading the labels needed for the analysis with the build configuration
65   *   <li>loading the transitive closure of the targets and the configuration labels
66   * </ul>
67   *
68   * <p>In order to ensure correctness of incremental loading and of full cache hits, this class is
69   * very restrictive about access to its internal state and to its collaborators. In particular, none
70   * of the collaborators of this class may change in incompatible ways, such as changing the relative
71   * working directory for the target pattern parser, without notifying this class.
72   *
73   * <p>For full caching, this class tracks the exact values of all inputs to the loading phase. To
74   * maximize caching, it is vital that these change as rarely as possible.
75   */
76  public class LoadingPhaseRunner {

274   public LoadedPackageProvider loadForConfigurations(EventHandler eventHandler,
275       Set<Label> labelsToLoad, boolean keepGoing) throws InterruptedException {
276     // Use a new Label Visitor here to avoid erasing the cache on the existing one.
277     TransitivePackageLoader transitivePackageLoader = packageManager.newTransitiveLoader();

282     return loadingSuccessful ? packageManager : null;

285   /**
286    * Performs target pattern evaluation, test suite expansion (if requested), and loads the
287    * transitive closure of the resulting targets as well as of the targets needed to use the
288    * given build configuration provider.
289    */
290   public LoadingResult execute(EventHandler eventHandler, EventBus eventBus,
291       List<String> targetPatterns, Options options,
292       ListMultimap<String, Label> labelsToLoadUnconditionally, boolean keepGoing,
293       boolean determineTests, @Nullable Callback callback)
294           throws TargetParsingException, LoadingFailedException, InterruptedException {
295     LOG.info("Starting pattern evaluation");
296     Stopwatch timer = Stopwatch.createStarted();
297     if (options.buildTestsOnly && options.compileOneDependency) {
298       throw new LoadingFailedException("--compile_one_dependency cannot be used together with "
299           + "the --build_tests_only option or the 'bazel test' command ");
300     }
301 
302     EventHandler parseFailureListener = new ParseFailureListenerImpl(eventHandler, eventBus);
303     // Determine targets to build:
304     ResolvedTargets<Target> targets = getTargetsToBuild(parseFailureListener,
305         targetPatterns, options.compileOneDependency, keepGoing);
306 
307     ImmutableSet<Target> filteredTargets = targets.getFilteredTargets();
308 
309     boolean buildTestsOnly = options.buildTestsOnly;
310     ImmutableSet<Target> testsToRun = null;
311     ImmutableSet<Target> testFilteredTargets = ImmutableSet.of();
312 
313     // Now we have a list of targets to build. If the --build_tests_only option was specified or we
314     // want to run tests, we need to determine the list of targets to test. For that, we remove
315     // manual tests and apply the command line filters. Also, if --build_tests_only is specified,
316     // then the list of filtered targets will be set as build list as well.
317     if (determineTests || buildTestsOnly) {
318       // Parse the targets to get the tests.
319       ResolvedTargets<Target> testTargets = determineTests(parseFailureListener,
320           targetPatterns, options, keepGoing);
321       if (testTargets.getTargets().isEmpty() && !testTargets.getFilteredTargets().isEmpty()) {
322         eventHandler.handle(Event.warn("All specified test targets were excluded by filters"));
323       }
324 
325       if (buildTestsOnly) {
326         // Replace original targets to build with test targets, so that only targets that are
327         // actually going to be built are loaded in the loading phase. Note that this has a side
328         // effect that any test_suite target requested to be built is replaced by the set of *_test
329         // targets it represents; for example, this affects the status and the summary reports.
330         Set<Target> allFilteredTargets = new HashSet<>();
331         allFilteredTargets.addAll(targets.getTargets());
332         allFilteredTargets.addAll(targets.getFilteredTargets());
333         allFilteredTargets.removeAll(testTargets.getTargets());
334         allFilteredTargets.addAll(testTargets.getFilteredTargets());
335         testFilteredTargets = ImmutableSet.copyOf(allFilteredTargets);
336         filteredTargets = ImmutableSet.of();
337 
338         targets = ResolvedTargets.<Target>builder()
339             .merge(testTargets)
340             .mergeError(targets.hasError())
341             .build();
342         if (determineTests) {
343           testsToRun = testTargets.getTargets();
344         }
345       } else /*if (determineTests)*/ {
346         testsToRun = testTargets.getTargets();
347         targets = ResolvedTargets.<Target>builder()
348             .merge(targets)
349             // Avoid merge() here which would remove the filteredTargets from the targets.
350             .addAll(testsToRun)
351             .mergeError(testTargets.hasError())
352             .build();
353         // filteredTargets is correct in this case - it cannot contain tests that got back in
354         // through test_suite expansion, because the test determination would also filter those out.
355         // However, that's not obvious, and it might be better to explicitly recompute it.
356       }
357       if (testsToRun != null) {
358         // Note that testsToRun can still be null here, if buildTestsOnly && !shouldRunTests.
359         Preconditions.checkState(targets.getTargets().containsAll(testsToRun));
360       }
361     }
362 
363     eventBus.post(new TargetParsingCompleteEvent(targets.getTargets(),
364         filteredTargets, testFilteredTargets,
365         timer.stop().elapsed(TimeUnit.MILLISECONDS)));
366 
367     if (targets.hasError()) {
368       eventHandler.handle(Event.warn("Target pattern parsing failed. Continuing anyway"));
369     }
370     if (callback != null) {
371       callback.notifyTargets(targets.getTargets());
372     }
373     maybeReportDeprecation(eventHandler, targets.getTargets());
374 
375     BaseLoadingResult result = doLoadingPhase(eventHandler, eventBus, targets.getTargets(),
376         labelsToLoadUnconditionally, keepGoing, options.loadingPhaseThreads);
377     LoadingResult loadingResult = new LoadingResult(targets.hasError(), !result.isSuccesful(),
378         result.getTargets(), testsToRun,
379         collectPackageRoots(pkgLoader.getErrorFreeVisitedPackages()));
380     freeMemoryAfterLoading(callback, pkgLoader.getVisitedPackageNames());
381     return loadingResult;
382   }
383 
384   private void freeMemoryAfterLoading(Callback callback, Set<PackageIdentifier> visitedPackages) {
385     if (callback != null) {
386       callback.notifyVisitedPackages(visitedPackages);
387     }
388     // Clear some targets from the cache to free memory.
389     packageManager.partiallyClear();
390   }

392   /**
393    * Visit the transitive closure of the targets, populating the package cache
394    * and ensuring that all labels can be resolved and all rules were free from
395    * errors.
396    *
397    * @param targetsToLoad the list of command-line target patterns specified by the user
398    * @param labelsToLoadUnconditionally the labels to load unconditionally (presumably for the build
399    *                                    configuration)
400    * @param keepGoing if true, don't throw ViewCreationFailedException if some
401    *                  targets could not be loaded, just skip thm.
402    */
403   private BaseLoadingResult doLoadingPhase(EventHandler eventHandler, EventBus eventBus,
404       ImmutableSet<Target> targetsToLoad, ListMultimap<String, Label> labelsToLoadUnconditionally,
405       boolean keepGoing, int loadingPhaseThreads)
406           throws InterruptedException, LoadingFailedException {
407     eventHandler.handle(Event.progress("Loading..."));
408     Stopwatch timer = Stopwatch.createStarted();
409     LOG.info("Starting loading phase");
410 
411     BaseLoadingResult baseResult = performLoadingOfTargets(eventHandler, eventBus, targetsToLoad,
412         labelsToLoadUnconditionally, keepGoing, loadingPhaseThreads);
413     BaseLoadingResult expandedResult = expandTestSuites(eventHandler, baseResult.getTargets(),
414         keepGoing);
415 
416     Set<Target> testSuiteTargets = Sets.difference(baseResult.getTargets(),
417         expandedResult.getTargets());
418     eventBus.post(new LoadingPhaseCompleteEvent(expandedResult.getTargets(), testSuiteTargets,
419         packageManager.getStatistics(), timer.stop().elapsed(TimeUnit.MILLISECONDS)));
420     LOG.info("Loading phase finished");
421 
422     return new BaseLoadingResult(expandedResult.getTargets(),
423         baseResult.isSuccesful() && expandedResult.isSuccesful());
424   }

426   private BaseLoadingResult performLoadingOfTargets(EventHandler eventHandler, EventBus eventBus,
427       ImmutableSet<Target> targetsToLoad, ListMultimap<String, Label> labelsToLoadUnconditionally,
428       boolean keepGoing, int loadingPhaseThreads) throws InterruptedException,
429       LoadingFailedException {
430     Set<Label> labelsToLoad = ImmutableSet.copyOf(labelsToLoadUnconditionally.values());
431 
432     // For each label in {@code targetsToLoad}, ensure that the target to which
433     // it refers exists, and also every target in its transitive closure of label
434     // dependencies. Success guarantees that a call to
435     // {@code getConfiguredTarget} for the same targets will not fail; the
436     // configuration process is intolerant of missing packages/targets. Before
437     // calling getConfiguredTarget(), clients must ensure that all necessary
438     // packages/targets have been visited since the last sync/clear.
439     boolean loadingSuccessful = pkgLoader.sync(eventHandler, targetsToLoad, labelsToLoad,
440           keepGoing, loadingPhaseThreads, Integer.MAX_VALUE);
441 
442     ImmutableSet<Target> targetsToAnalyze;
443     if (loadingSuccessful) {
444       // Success: all loaded targets will be analyzed.
445       targetsToAnalyze = targetsToLoad;
446     } else if (keepGoing) {
447       // Keep going: filter out the error-free targets and only continue with those.
448       targetsToAnalyze = filterErrorFreeTargets(eventBus, targetsToLoad,
449           labelsToLoadUnconditionally);
450       reportAboutPartiallySuccesfulLoading(targetsToLoad, targetsToAnalyze, eventHandler);
451     } else {
452       throw new LoadingFailedException("Loading failed; build aborted");
453     }
454     return new BaseLoadingResult(targetsToAnalyze, loadingSuccessful);
455   }

457   private void reportAboutPartiallySuccesfulLoading(ImmutableSet<Target> requestedTargets,
458       ImmutableSet<Target> loadedTargets, EventHandler eventHandler) {
459     // Tell the user about the subset of successful targets.
460     int requested = requestedTargets.size();
461     int loaded = loadedTargets.size();
462     if (0 < loaded) {
463       String message = String.format("Loading succeeded for only %d of %d targets", loaded,
464           requested);
465       eventHandler.handle(Event.info(message));
466       LOG.info(message);
467     }
468   }

470   private BaseLoadingResult expandTestSuites(EventHandler eventHandler,
471       ImmutableSet<Target> targets, boolean keepGoing) throws LoadingFailedException {
472     try {
473       // We use strict test_suite expansion here to match the analysis-time checks.
474       ResolvedTargets<Target> expandedResult = TestTargetUtils.expandTestSuites(
475           packageManager, eventHandler, targets, /*strict=*/true, /*keepGoing=*/true);
476       if (expandedResult.hasError() && !keepGoing) {
477         throw new LoadingFailedException("Could not expand test suite target");
478       }
479       return new BaseLoadingResult(expandedResult.getTargets(), !expandedResult.hasError());
480     } catch (TargetParsingException e) {
481       // This shouldn't happen, because we've already loaded the targets successfully.
482       throw (AssertionError) (new AssertionError("Unexpected target failure").initCause(e));
483     }
484   }

504   private Collection<Target> getTargetsForLabels(Collection<Label> labels) {
505     Set<Target> result = new HashSet<>();
506 
507     for (Label label : labels) {
508       try {
509         result.add(packageManager.getLoadedTarget(label));
510       } catch (NoSuchThingException e) {
511         throw new IllegalStateException(e);  // The target should have been loaded
512       }
513     }
514 
515     return result;
516   }

518   private ImmutableSet<Target> filterErrorFreeTargets(
519       EventBus eventBus, Collection<Target> targetsToLoad,
520       ListMultimap<String, Label> labelsToLoadUnconditionally) throws LoadingFailedException {
521     // Error out if any of the labels needed for the configuration could not be loaded.
522     Collection<Label> labelsToLoad = new ArrayList<>(labelsToLoadUnconditionally.values());
523     for (Target target : targetsToLoad) {
524       labelsToLoad.add(target.getLabel());
525     }
526     Multimap<Label, Label> rootCauses = pkgLoader.getRootCauses(labelsToLoad);
527     for (Map.Entry<String, Label> entry : labelsToLoadUnconditionally.entries()) {
528       if (rootCauses.containsKey(entry.getValue())) {
529         throw new LoadingFailedException("Failed to load required " + entry.getKey()
530             + " target: '" + entry.getValue() + "'");
531       }
532     }
533 
534     // Post root causes for command-line targets that could not be loaded.
535     for (Map.Entry<Label, Label> entry : rootCauses.entries()) {
536       eventBus.post(new LoadingFailureEvent(entry.getKey(), entry.getValue()));
537     }
538 
539     return ImmutableSet.copyOf(Sets.difference(ImmutableSet.copyOf(targetsToLoad),
540         ImmutableSet.copyOf(getTargetsForLabels(rootCauses.keySet()))));
541   }

543   /**
544    * Returns a map of collected package names to root paths.
545    */
546   private static ImmutableMap<PackageIdentifier, Path> collectPackageRoots(
547       Collection<Package> packages) {
548     // Make a map of the package names to their root paths.
549     ImmutableMap.Builder<PackageIdentifier, Path> packageRoots = ImmutableMap.builder();
550     for (Package pkg : packages) {
551       packageRoots.put(pkg.getPackageIdentifier(), pkg.getSourceRoot());
552     }
553     return packageRoots.build();
554   }

556   /**
557    * Interpret the command-line arguments.
558    *
559    * @param targetPatterns the list of command-line target patterns specified by the user
560    * @param compileOneDependency if true, enables alternative interpretation of targetPatterns; see
561    *     {@link Options#compileOneDependency}
562    * @throws TargetParsingException if parsing failed and !keepGoing
563    */
564   private ResolvedTargets<Target> getTargetsToBuild(EventHandler eventHandler,
565       List<String> targetPatterns, boolean compileOneDependency,
566       boolean keepGoing) throws TargetParsingException, InterruptedException {
567     ResolvedTargets<Target> result =
568         targetPatternEvaluator.parseTargetPatternList(eventHandler, targetPatterns,
569             FilteringPolicies.FILTER_MANUAL, keepGoing);
570     if (compileOneDependency) {
571       return new CompileOneDependencyTransformer(packageManager)
572           .transformCompileOneDependency(eventHandler, result);
573     }
574     return result;
575   }

577   /**
578    * Interpret test target labels from the command-line arguments and return the corresponding set
579    * of targets, handling the filter flags, and expanding test suites.
580    *
581    * @param eventHandler the error event eventHandler
582    * @param targetPatterns the list of command-line target patterns specified by the user
583    * @param options the loading phase options
584    * @param keepGoing value of the --keep_going flag
585    */
586   private ResolvedTargets<Target> determineTests(EventHandler eventHandler,
587       List<String> targetPatterns, Options options, boolean keepGoing)
588           throws TargetParsingException, InterruptedException {
589     // Parse the targets to get the tests.
590     ResolvedTargets<Target> testTargetsBuilder = targetPatternEvaluator.parseTargetPatternList(
591         eventHandler, targetPatterns, FilteringPolicies.FILTER_TESTS, keepGoing);
592 
593     ResolvedTargets.Builder<Target> finalBuilder = ResolvedTargets.builder();
594     finalBuilder.merge(testTargetsBuilder);
595     finalBuilder.filter(TestFilter.forOptions(options, eventHandler, ruleNames));
596     return finalBuilder.build();
597   }
598 
599   /**
600    * Emit a warning when a deprecated target is mentioned on the command line.
601    *
602    * <p>Note that this does not stop us from emitting "target X depends on deprecated target Y"
603    * style warnings for the same target and it is a good thing; <i>depending</i> on a target and
604    * <i>wanting</i> to build it are different things.
605    */
606   private void maybeReportDeprecation(EventHandler eventHandler, Collection<Target> targets) {
607     for (Rule rule : Iterables.filter(targets, Rule.class)) {
608       if (rule.isAttributeValueExplicitlySpecified("deprecation")) {
609         eventHandler.handle(Event.warn(rule.getLocation(), String.format(
610             "target '%s' is deprecated: %s", rule.getLabel(),
611             NonconfigurableAttributeMapper.of(rule).get("deprecation", Type.STRING))));
612       }
613     }
614   }
