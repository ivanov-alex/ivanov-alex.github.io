93    private LinkCommandLine(
94        BuildConfiguration configuration,
95        ActionOwner owner,
96        Artifact output,
97        @Nullable Artifact interfaceOutput,
98        @Nullable Artifact symbolCountsOutput,
99        ImmutableList<Artifact> buildInfoHeaderArtifacts,
100       Iterable<? extends LinkerInput> linkerInputs,
101       Iterable<? extends LinkerInput> runtimeInputs,
102       LinkTargetType linkTargetType,
103       LinkStaticness linkStaticness,
104       ImmutableList<String> linkopts,
105       ImmutableSet<String> features,
106       ImmutableMap<Artifact, Artifact> linkstamps,
107       ImmutableList<String> linkstampCompileOptions,
108       @Nullable PathFragment runtimeSolibDir,
109       boolean nativeDeps,
110       boolean useTestOnlyFlags,
111       boolean needWholeArchive,
112       @Nullable Iterable<LTOBackendArtifacts> allLTOArtifacts,
113       @Nullable Artifact paramFile,
114       Artifact interfaceSoBuilder,
115       CcToolchainFeatures.Variables variables,
116       @Nullable FeatureConfiguration featureConfiguration) {
117     Preconditions.checkArgument(linkTargetType != LinkTargetType.INTERFACE_DYNAMIC_LIBRARY,
118         "you can't link an interface dynamic library directly");
119     if (linkTargetType != LinkTargetType.DYNAMIC_LIBRARY) {
120       Preconditions.checkArgument(interfaceOutput == null,
121           "interface output may only be non-null for dynamic library links");
122     }
123     if (linkTargetType.isStaticLibraryLink()) {
124       Preconditions.checkArgument(linkstamps.isEmpty(),
125           "linkstamps may only be present on dynamic library or executable links");
126       Preconditions.checkArgument(linkStaticness == LinkStaticness.FULLY_STATIC,
127           "static library link must be static");
128       Preconditions.checkArgument(buildInfoHeaderArtifacts.isEmpty(),
129           "build info headers may only be present on dynamic library or executable links");
130       Preconditions.checkArgument(symbolCountsOutput == null,
131           "the symbol counts output must be null for static links");
132       Preconditions.checkArgument(runtimeSolibDir == null,
133           "the runtime solib directory must be null for static links");
134       Preconditions.checkArgument(!nativeDeps,
135           "the native deps flag must be false for static links");
136       Preconditions.checkArgument(!needWholeArchive,
137           "the need whole archive flag must be false for static links");
138     }
139 
140     this.configuration = Preconditions.checkNotNull(configuration);
141     this.cppConfiguration = configuration.getFragment(CppConfiguration.class);
142     this.variables = variables;
143     this.featureConfiguration = featureConfiguration;
144     this.owner = Preconditions.checkNotNull(owner);
145     this.output = output;
146     this.interfaceOutput = interfaceOutput;
147     if (interfaceOutput != null) {
148       Preconditions.checkNotNull(this.output);
149     }
150 
151     this.symbolCountsOutput = symbolCountsOutput;
152     this.buildInfoHeaderArtifacts = Preconditions.checkNotNull(buildInfoHeaderArtifacts);
153     this.linkerInputs = Preconditions.checkNotNull(linkerInputs);
154     this.runtimeInputs = Preconditions.checkNotNull(runtimeInputs);
155     this.linkTargetType = Preconditions.checkNotNull(linkTargetType);
156     this.linkStaticness = Preconditions.checkNotNull(linkStaticness);
157     // For now, silently ignore linkopts if this is a static library link.
158     this.linkopts = linkTargetType.isStaticLibraryLink()
159         ? ImmutableList.<String>of()
160         : Preconditions.checkNotNull(linkopts);
161     this.features = Preconditions.checkNotNull(features);
162     this.linkstamps = Preconditions.checkNotNull(linkstamps);
163     this.linkstampCompileOptions = linkstampCompileOptions;
164     this.runtimeSolibDir = runtimeSolibDir;
165     this.nativeDeps = nativeDeps;
166     this.useTestOnlyFlags = useTestOnlyFlags;
167     this.needWholeArchive = needWholeArchive;
168     this.allLTOArtifacts = allLTOArtifacts;
169     this.paramFile = paramFile;
170 
171     // For now, silently ignore interfaceSoBuilder if we don't build an interface dynamic library.
172     this.interfaceSoBuilder =
173         ((linkTargetType == LinkTargetType.DYNAMIC_LIBRARY) && (interfaceOutput != null))
174         ? Preconditions.checkNotNull(interfaceSoBuilder,
175             "cannot build interface dynamic library without builder")
176         : null;
177   }

281   /**
282    * Splits the link command-line into a part to be written to a parameter file, and the remaining
283    * actual command line to be executed (which references the parameter file). Should only be used
284    * if getParamFile() is not null.
285    *
286    * @throws IllegalStateException if the command-line cannot be split
287    */
288   @VisibleForTesting
289   final Pair<List<String>, List<String>> splitCommandline() {
290     List<String> args = getRawLinkArgv();
291     if (linkTargetType.isStaticLibraryLink()) {
292       // Ar link commands can also generate huge command lines.
293       List<String> paramFileArgs = args.subList(1, args.size());
294       List<String> commandlineArgs = new ArrayList<>();
295       commandlineArgs.add(args.get(0));
296 
297       commandlineArgs.add("@" + paramFile.getExecPath().getPathString());
298       return Pair.of(commandlineArgs, paramFileArgs);
299     } else {
300       // Gcc link commands tend to generate humongous commandlines for some targets, which may
301       // not fit on some remote execution machines. To work around this we will employ the help of
302       // a parameter file and pass any linker options through it.
303       List<String> paramFileArgs = new ArrayList<>();
304       List<String> commandlineArgs = new ArrayList<>();
305       extractArgumentsForParamFile(args, commandlineArgs, paramFileArgs);
306 
307       commandlineArgs.add("-Wl,@" + paramFile.getExecPath().getPathString());
308       return Pair.of(commandlineArgs, paramFileArgs);
309     }
310   }

328   private static void extractArgumentsForParamFile(List<String> args, List<String> commandlineArgs,
329       List<String> paramFileArgs) {
330     // Note, that it is not important that all linker arguments are extracted so that
331     // they can be moved into a parameter file, but the vast majority should.
332     commandlineArgs.add(args.get(0));   // gcc command, must not be moved!
333     int argsSize = args.size();
334     for (int i = 1; i < argsSize; i++) {
335       String arg = args.get(i);
336       if (arg.equals("-Wl,-no-whole-archive")) {
337         paramFileArgs.add("-no-whole-archive");
338       } else if (arg.equals("-Wl,-whole-archive")) {
339         paramFileArgs.add("-whole-archive");
340       } else if (arg.equals("-Wl,--start-group")) {
341         paramFileArgs.add("--start-group");
342       } else if (arg.equals("-Wl,--end-group")) {
343         paramFileArgs.add("--end-group");
344       } else if (arg.equals("-Wl,--start-lib")) {
345         paramFileArgs.add("--start-lib");
346       } else if (arg.equals("-Wl,--end-lib")) {
347         paramFileArgs.add("--end-lib");
348       } else if (arg.equals("--incremental-unchanged")) {
349         paramFileArgs.add(arg);
350       } else if (arg.equals("--incremental-changed")) {
351         paramFileArgs.add(arg);
352       } else if (arg.charAt(0) == '-') {
353         if (arg.startsWith("-l")) {
354           paramFileArgs.add(arg);
355         } else {
356           // Anything else starting with a '-' can stay on the commandline.
357           commandlineArgs.add(arg);
358           if (arg.equals("-o")) {
359             // Special case for '-o': add the following argument as well - it is the output file!
360             commandlineArgs.add(args.get(++i));
361           }
362         }
363       } else if (arg.endsWith(".a") || arg.endsWith(".lo") || arg.endsWith(".so")
364           || arg.endsWith(".ifso") || arg.endsWith(".o")
365           || CppFileTypes.VERSIONED_SHARED_LIBRARY.matches(arg)) {
366         // All objects of any kind go into the linker parameters.
367         paramFileArgs.add(arg);
368       } else {
369         // Everything that's left stays conservatively on the commandline.
370         commandlineArgs.add(arg);
371       }
372     }
373   }

375   /**
376    * Returns a raw link command for the given link invocation, including both command and
377    * arguments (argv). After any further usage-specific processing, this can be passed to
378    * {@link #finalizeWithLinkstampCommands} to give the final command line.
379    *
380    * @return raw link command line.
381    */
382   public List<String> getRawLinkArgv() {
383     List<String> argv = new ArrayList<>();
384     switch (linkTargetType) {
385       case EXECUTABLE:
386         addCppArgv(argv);
387         break;
388 
389       case DYNAMIC_LIBRARY:
390         if (interfaceOutput != null) {
391           argv.add(configuration.getShExecutable().getPathString());
392           argv.add("-c");
393           argv.add("build_iface_so=\"$0\"; impl=\"$1\"; iface=\"$2\"; cmd=\"$3\"; shift 3; "
394               + "\"$cmd\" \"$@\" && \"$build_iface_so\" \"$impl\" \"$iface\"");
395           argv.add(interfaceSoBuilder.getExecPathString());
396           argv.add(output.getExecPathString());
397           argv.add(interfaceOutput.getExecPathString());
398         }
399         addCppArgv(argv);
400         // -pie is not compatible with -shared and should be
401         // removed when the latter is part of the link command. Should we need to further
402         // distinguish between shared libraries and executables, we could add additional
403         // command line / CROSSTOOL flags that distinguish them. But as long as this is
404         // the only relevant use case we're just special-casing it here.
405         Iterables.removeIf(argv, Predicates.equalTo("-pie"));
406         break;
407 
408       case STATIC_LIBRARY:
409       case PIC_STATIC_LIBRARY:
410       case ALWAYS_LINK_STATIC_LIBRARY:
411       case ALWAYS_LINK_PIC_STATIC_LIBRARY:
412         // The static library link command follows this template:
413         // ar <cmd> <output_archive> <input_files...>
414         argv.add(cppConfiguration.getArExecutable().getPathString());
415         argv.addAll(
416             cppConfiguration.getArFlags(cppConfiguration.archiveType() == Link.ArchiveType.THIN));
417         argv.add(output.getExecPathString());
418         addInputFileLinkOptions(argv, /*needWholeArchive=*/false);
419         break;
420 
421       default:
422         throw new IllegalArgumentException();
423     }
424 
425     // Fission mode: debug info is in .dwo files instead of .o files. Inform the linker of this.
426     if (!linkTargetType.isStaticLibraryLink() && cppConfiguration.useFission()) {
427       argv.add("-Wl,--gdb-index");
428     }
429 
430     return argv;
431   }

433   List<String> getCommandLine() {
434     List<String> commandlineArgs;
435     // Try to shorten the command line by use of a parameter file.
436     // This makes the output with --subcommands (et al) more readable.
437     if (paramFile != null) {
438       Pair<List<String>, List<String>> split = splitCommandline();
439       commandlineArgs = split.first;
440     } else {
441       commandlineArgs = getRawLinkArgv();
442     }
443     return finalizeWithLinkstampCommands(commandlineArgs);
444   }

478   /**
479    * Adds linkstamp compilation to the (otherwise) fully specified link
480    * command if {@link #getLinkstamps} is non-empty.
481    *
482    * <p>Linkstamps were historically compiled implicitly as part of the link
483    * command, but implicit compilation doesn't guarantee consistent outputs.
484    * For example, the command "gcc input.o input.o foo/linkstamp.cc -o myapp"
485    * causes gcc to implicitly run "gcc foo/linkstamp.cc -o /tmp/ccEtJHDB.o",
486    * for some internally decided output path /tmp/ccEtJHDB.o, then add that path
487    * to the linker's command line options. The name of this path can change
488    * even between equivalently specified gcc invocations.
489    *
490    * <p>So now we explicitly compile these files in their own command
491    * invocations before running the link command, thus giving us direct
492    * control over the naming of their outputs. This method adds those extra
493    * steps as necessary.
494    * @param linkstampCommands individual linkstamp compilation commands
495    * @param linkCommand the complete list of link command arguments (after
496    *        .params file compacting) for an invocation
497    * @param escapeArgs if true, linkCommand arguments are shell escaped. if
498    *        false, arguments are returned as-is
499    *
500    * @return The original argument list if no linkstamps compilation commands
501    *         are given, otherwise an expanded list that adds the linkstamp
502    *         compilation commands and funnels their outputs into the link step.
503    *         Note that these outputs only need to persist for the duration of
504    *         the link step.
505    */
506   private static List<String> addLinkstampingToCommand(
507       List<String> linkstampCommands,
508       List<String> linkCommand,
509       boolean escapeArgs) {
510     if (linkstampCommands.isEmpty()) {
511       return linkCommand;
512     } else {
513       List<String> batchCommand = Lists.newArrayListWithCapacity(3);
514       batchCommand.add("/bin/bash");
515       batchCommand.add("-c");
516       batchCommand.add(
517           Joiner.on(" && ").join(linkstampCommands) + " && "
518           + (escapeArgs
519               ? ShellEscaper.escapeJoinAll(linkCommand)
520               : Joiner.on(" ").join(linkCommand)));
521       return ImmutableList.copyOf(batchCommand);
522     }
523   }

525   /**
526    * Computes, for each C++ source file in
527    * {@link #getLinkstamps}, the command necessary to compile
528    * that file such that the output is correctly fed into the link command.
529    *
530    * <p>As these options (as well as all others) are taken into account when
531    * computing the action key, they do not directly contain volatile build
532    * information to avoid unnecessary relinking. Instead this information is
533    * passed as an additional header generated by
534    * {@link com.google.devtools.build.lib.rules.cpp.WriteBuildInfoHeaderAction}.
535    *
536    * @param outputPrefix prefix to add before the linkstamp outputs' exec paths
537    * @return a list of shell-escaped compiler commmands, one for each entry
538    *         in {@link #getLinkstamps}
539    */
540   public List<String> getLinkstampCompileCommands(String outputPrefix) {
541     if (linkstamps.isEmpty()) {
542       return ImmutableList.of();
543     }
544 
545     String compilerCommand = cppConfiguration.getCppExecutable().getPathString();
546     List<String> commands = Lists.newArrayListWithCapacity(linkstamps.size());
547 
548     for (Map.Entry<Artifact, Artifact> linkstamp : linkstamps.entrySet()) {
549       List<String> optionList = new ArrayList<>();
550 
551       // Defines related to the build info are read from generated headers.
552       for (Artifact header : buildInfoHeaderArtifacts) {
553         optionList.add("-include");
554         optionList.add(header.getExecPathString());
555       }
556 
557       String labelReplacement = Matcher.quoteReplacement(
558           isSharedNativeLibrary() ? output.getExecPathString() : Label.print(owner.getLabel()));
559       String outputPathReplacement = Matcher.quoteReplacement(
560           output.getExecPathString());
561       for (String option : linkstampCompileOptions) {
562         optionList.add(option
563             .replaceAll(Pattern.quote("${LABEL}"), labelReplacement)
564             .replaceAll(Pattern.quote("${OUTPUT_PATH}"), outputPathReplacement));
565       }
566 
567       optionList.add("-DGPLATFORM=\"" + cppConfiguration + "\"");
568 
569       // Needed to find headers included from linkstamps.
570       optionList.add("-I.");
571 
572       // Add sysroot.
573       PathFragment sysroot = cppConfiguration.getSysroot();
574       if (sysroot != null) {
575         optionList.add("--sysroot=" + sysroot.getPathString());
576       }
577 
578       // Add toolchain compiler options.
579       optionList.addAll(cppConfiguration.getCompilerOptions(features));
580       optionList.addAll(cppConfiguration.getCOptions());
581       optionList.addAll(cppConfiguration.getUnfilteredCompilerOptions(features));
582       if (CppFileTypes.CPP_SOURCE.matches(linkstamp.getKey().getExecPath())) {
583         optionList.addAll(cppConfiguration.getCxxOptions(features));
584       }
585 
586       // For dynamic libraries, produce position independent code.
587       if (linkTargetType == LinkTargetType.DYNAMIC_LIBRARY
588           && cppConfiguration.toolchainNeedsPic()) {
589         optionList.add("-fPIC");
590       }
591 
592       // Stamp FDO builds with FDO subtype string
593       String fdoBuildStamp = CppHelper.getFdoBuildStamp(cppConfiguration);
594       if (fdoBuildStamp != null) {
595         optionList.add("-D" + CppConfiguration.FDO_STAMP_MACRO + "=\"" + fdoBuildStamp + "\"");
596       }
597 
598       // Add the compilation target.
599       optionList.add("-c");
600       optionList.add(linkstamp.getKey().getExecPathString());
601 
602       // Assemble the final command, exempting outputPrefix from shell escaping.
603       commands.add(compilerCommand + " "
604           + ShellEscaper.escapeJoinAll(optionList)
605           + " -o "
606           + outputPrefix
607           + ShellEscaper.escapeString(linkstamp.getValue().getExecPathString()));
608     }
609 
610     return commands;
611   }

615    * Add them to the {@code argv} parameter.
616    */
617   private void addCppArgv(List<String> argv) {
618     argv.add(cppConfiguration.getCppExecutable().getPathString());
619 
620     // When using gold to link an executable, output the number of used and unused symbols.
621     if (symbolCountsOutput != null) {
622       argv.add("-Wl,--print-symbol-counts=" + symbolCountsOutput.getExecPathString());
623     }
624 
625     if (linkTargetType == LinkTargetType.DYNAMIC_LIBRARY) {
626       argv.add("-shared");
627     }
628 
629     // Add the outputs of any associated linkstamp compilations.
630     for (Artifact linkstampOutput : linkstamps.values()) {
631       argv.add(linkstampOutput.getExecPathString());
632     }
633 
634     boolean fullyStatic = (linkStaticness == LinkStaticness.FULLY_STATIC);
635     boolean mostlyStatic = (linkStaticness == LinkStaticness.MOSTLY_STATIC);
636     boolean sharedLinkopts =
637         linkTargetType == LinkTargetType.DYNAMIC_LIBRARY
638         || linkopts.contains("-shared")
639         || cppConfiguration.getLinkOptions().contains("-shared");
640 
641     if (output != null) {
642       argv.add("-o");
643       String execpath = output.getExecPathString();
644       if (mostlyStatic
645           && linkTargetType == LinkTargetType.EXECUTABLE
646           && cppConfiguration.skipStaticOutputs()) {
647         // Linked binary goes to /dev/null; bogus dependency info in its place.
648         Collections.addAll(argv, "/dev/null", "-MMD", "-MF", execpath);
649       } else {
650         argv.add(execpath);
651       }
652     }
653 
654     addInputFileLinkOptions(argv, needWholeArchive);
655 
656     /*
657      * For backwards compatibility, linkopts come _after_ inputFiles.
658      * This is needed to allow linkopts to contain libraries and
659      * positional library-related options such as
660      *    -Wl,--begin-group -lfoo -lbar -Wl,--end-group
661      * or
662      *    -Wl,--as-needed -lfoo -Wl,--no-as-needed
663      *
664      * As for the relative order of the three different flavours of linkopts
665      * (global defaults, per-target linkopts, and command-line linkopts),
666      * we have no idea what the right order should be, or if anyone cares.
667      */
668     argv.addAll(linkopts);
669 
670     // Extra toolchain link options based on the output's link staticness.
671     if (fullyStatic) {
672       argv.addAll(cppConfiguration.getFullyStaticLinkOptions(features, sharedLinkopts));
673     } else if (mostlyStatic) {
674       argv.addAll(cppConfiguration.getMostlyStaticLinkOptions(features, sharedLinkopts));
675     } else {
676       argv.addAll(cppConfiguration.getDynamicLinkOptions(features, sharedLinkopts));
677     }
678 
679     // Extra test-specific link options.
680     if (useTestOnlyFlags) {
681       argv.addAll(cppConfiguration.getTestOnlyLinkOptions());
682     }
683 
684     if (linkTargetType == LinkTargetType.EXECUTABLE && cppConfiguration.forcePic()) {
685       argv.add("-pie");
686     }
687 
688     argv.addAll(cppConfiguration.getLinkOptions());
689     // The feature config can be null for tests.
690     if (featureConfiguration != null) {
691       argv.addAll(featureConfiguration.getCommandLine(CPP_LINK, variables));
692     }
693   }

695   private static boolean isDynamicLibrary(LinkerInput linkInput) {
696     Artifact libraryArtifact = linkInput.getArtifact();
697     String name = libraryArtifact.getFilename();
698     return Link.SHARED_LIBRARY_FILETYPES.matches(name) && name.startsWith("lib");
699   }

701   private boolean isSharedNativeLibrary() {
702     return nativeDeps && cppConfiguration.shareNativeDeps();
703   }

705   /**
706    * When linking a shared library fully or mostly static then we need to link in
707    * *all* dependent files, not just what the shared library needs for its own
708    * code. This is done by wrapping all objects/libraries with
709    * -Wl,-whole-archive and -Wl,-no-whole-archive. For this case the
710    * globalNeedWholeArchive parameter must be set to true.  Otherwise only
711    * library objects (.lo) need to be wrapped with -Wl,-whole-archive and
712    * -Wl,-no-whole-archive.
713    */
714   private void addInputFileLinkOptions(List<String> argv, boolean globalNeedWholeArchive) {
715     // The Apple ld doesn't support -whole-archive/-no-whole-archive. It
716     // does have -all_load/-noall_load, but -all_load is a global setting
717     // that affects all subsequent files, and -noall_load is simply ignored.
718     // TODO(bazel-team): Not sure what the implications of this are, other than
719     // bloated binaries.
720     boolean macosx = cppConfiguration.getTargetLibc().equals("macosx");
721     if (globalNeedWholeArchive) {
722       argv.add(macosx ? "-Wl,-all_load" : "-Wl,-whole-archive");
723     }
724 
725     // Used to collect -L and -Wl,-rpath options, ensuring that each used only once.
726     Set<String> libOpts = new LinkedHashSet<>();
727 
728     // List of command line parameters to link input files (either directly or using -l).
729     List<String> linkerInputs = new ArrayList<>();
730 
731     // List of command line parameters that need to be placed *outside* of
732     // --whole-archive ... --no-whole-archive.
733     List<String> noWholeArchiveInputs = new ArrayList<>();
734 
735     PathFragment solibDir = configuration.getBinDirectory().getExecPath()
736         .getRelative(cppConfiguration.getSolibDirectory());
737     String runtimeSolibName = runtimeSolibDir != null ? runtimeSolibDir.getBaseName() : null;
738     boolean runtimeRpath = runtimeSolibDir != null
739         && (linkTargetType == LinkTargetType.DYNAMIC_LIBRARY
740         || (linkTargetType == LinkTargetType.EXECUTABLE
741         && linkStaticness == LinkStaticness.DYNAMIC));
742 
743     String rpathRoot = null;
744     List<String> runtimeRpathEntries = new ArrayList<>();
745 
746     if (output != null) {
747       String origin =
748           useTestOnlyFlags && cppConfiguration.supportsExecOrigin() ? "$EXEC_ORIGIN/" : "$ORIGIN/";
749       if (runtimeRpath) {
750         runtimeRpathEntries.add("-Wl,-rpath," + origin + runtimeSolibName + "/");
751       }
752 
753       // Calculate the correct relative value for the "-rpath" link option (which sets
754       // the search path for finding shared libraries).
755       if (isSharedNativeLibrary()) {
756         // For shared native libraries, special symlinking is applied to ensure C++
757         // runtimes are available under $ORIGIN/_solib_[arch]. So we set the RPATH to find
758         // them.
759         //
760         // Note that we have to do this because $ORIGIN points to different paths for
761         // different targets. In other words, blaze-bin/d1/d2/d3/a_shareddeps.so and
762         // blaze-bin/d4/b_shareddeps.so have different path depths. The first could
763         // reference a standard blaze-bin/_solib_[arch] via $ORIGIN/../../../_solib[arch],
764         // and the second could use $ORIGIN/../_solib_[arch]. But since this is a shared
765         // artifact, both are symlinks to the same place, so
766         // there's no *one* RPATH setting that fits all targets involved in the sharing.
767         rpathRoot = "-Wl,-rpath," + origin + ":"
768             + origin + cppConfiguration.getSolibDirectory() + "/";
769         if (runtimeRpath) {
770           runtimeRpathEntries.add("-Wl,-rpath," + origin + "../" + runtimeSolibName + "/");
771         }
772       } else {
773         // For all other links, calculate the relative path from the output file to _solib_[arch]
774         // (the directory where all shared libraries are stored, which resides under the blaze-bin
775         // directory. In other words, given blaze-bin/my/package/binary, rpathRoot would be
776         // "../../_solib_[arch]".
777         if (runtimeRpath) {
778           runtimeRpathEntries.add("-Wl,-rpath," + origin
779               + Strings.repeat("../", output.getRootRelativePath().segmentCount() - 1)
780               + runtimeSolibName + "/");
781         }
782 
783         rpathRoot = "-Wl,-rpath,"
784             + origin + Strings.repeat("../", output.getRootRelativePath().segmentCount() - 1)
785             + cppConfiguration.getSolibDirectory() + "/";
786 
787         if (nativeDeps) {
788           // We also retain the $ORIGIN/ path to solibs that are in _solib_<arch>, as opposed to
789           // the package directory)
790           if (runtimeRpath) {
791             runtimeRpathEntries.add("-Wl,-rpath," + origin + "../" + runtimeSolibName + "/");
792           }
793           rpathRoot += ":" + origin;
794         }
795       }
796     }
797 
798     boolean includeSolibDir = false;
799 
800 
801     Map<Artifact, Artifact> ltoMap = null;
802     if (allLTOArtifacts != null) {
803       // TODO(bazel-team): The LTO final link can only work if there are individual .o files on the
804       // command line. Rather than crashing, this should issue a nice error. We will get this by
805       // 1) moving supports_start_end_lib to a toolchain feature
806       // 2) having thin_lto require start_end_lib
807       // As a bonus, we can rephrase --nostart_end_lib as --features=-start_end_lib and get rid
808       // of a command line option.
809 
810       Preconditions.checkState(cppConfiguration.useStartEndLib());
811       ltoMap = new HashMap<>();
812       for (LTOBackendArtifacts l : allLTOArtifacts) {
813         ltoMap.put(l.getBitcodeFile(), l.getObjectFile());
814       }
815     }
816 
817     for (LinkerInput input : getLinkerInputs()) {
818       if (isDynamicLibrary(input)) {
819         PathFragment libDir = input.getArtifact().getExecPath().getParentDirectory();
820         Preconditions.checkState(
821             libDir.startsWith(solibDir),
822             "Artifact '%s' is not under directory '%s'.", input.getArtifact(), solibDir);
823         if (libDir.equals(solibDir)) {
824           includeSolibDir = true;
825         }
826         addDynamicInputLinkOptions(input, linkerInputs, libOpts, solibDir, rpathRoot);
827       } else {
828         addStaticInputLinkOptions(input, linkerInputs, ltoMap);
829       }
830     }
831 
832     boolean includeRuntimeSolibDir = false;
833 
834     for (LinkerInput input : runtimeInputs) {
835       List<String> optionsList = globalNeedWholeArchive
836           ? noWholeArchiveInputs
837           : linkerInputs;
838 
839       if (isDynamicLibrary(input)) {
840         PathFragment libDir = input.getArtifact().getExecPath().getParentDirectory();
841         Preconditions.checkState(runtimeSolibDir != null && libDir.equals(runtimeSolibDir),
842             "Artifact '%s' is not under directory '%s'.", input.getArtifact(), solibDir);
843         includeRuntimeSolibDir = true;
844         addDynamicInputLinkOptions(input, optionsList, libOpts, solibDir, rpathRoot);
845       } else {
846         addStaticInputLinkOptions(input, optionsList, ltoMap);
847       }
848     }
849 
850     // rpath ordering matters for performance; first add the one where most libraries are found.
851     if (includeSolibDir && rpathRoot != null) {
852       argv.add(rpathRoot);
853     }
854     if (includeRuntimeSolibDir) {
855       argv.addAll(runtimeRpathEntries);
856     }
857     argv.addAll(libOpts);
858 
859     // Need to wrap static libraries with whole-archive option
860     for (String option : linkerInputs) {
861       if (!globalNeedWholeArchive && Link.LINK_LIBRARY_FILETYPES.matches(option)) {
862         argv.add(macosx ? "-Wl,-all_load" : "-Wl,-whole-archive");
863         argv.add(option);
864         argv.add(macosx ? "-Wl,-noall_load" : "-Wl,-no-whole-archive");
865       } else {
866         argv.add(option);
867       }
868     }
869 
870     if (globalNeedWholeArchive) {
871       argv.add(macosx ? "-Wl,-noall_load" : "-Wl,-no-whole-archive");
872       argv.addAll(noWholeArchiveInputs);
873     }
874 
875     if (ltoMap != null) {
876       Preconditions.checkState(
877           ltoMap.size() == 0, "Still have LTO objects left: %s, command-line: %s", ltoMap, argv);
878     }
879   }

881   /**
882    * Adds command-line options for a dynamic library input file into
883    * options and libOpts.
884    */
885   private void addDynamicInputLinkOptions(LinkerInput input, List<String> options,
886       Set<String> libOpts, PathFragment solibDir, String rpathRoot) {
887     Preconditions.checkState(isDynamicLibrary(input));
888     Preconditions.checkState(
889         !Link.useStartEndLib(input, cppConfiguration.archiveType()));
890 
891     Artifact inputArtifact = input.getArtifact();
892     PathFragment libDir = inputArtifact.getExecPath().getParentDirectory();
893     if (rpathRoot != null
894         && !libDir.equals(solibDir)
895         && (runtimeSolibDir == null || !runtimeSolibDir.equals(libDir))) {
896       String dotdots = "";
897       PathFragment commonParent = solibDir;
898       while (!libDir.startsWith(commonParent)) {
899         dotdots += "../";
900         commonParent = commonParent.getParentDirectory();
901       }
902 
903       libOpts.add(rpathRoot + dotdots + libDir.relativeTo(commonParent).getPathString());
904     }
905 
906     libOpts.add("-L" + inputArtifact.getExecPath().getParentDirectory().getPathString());
907 
908     String name = inputArtifact.getFilename();
909     if (CppFileTypes.SHARED_LIBRARY.matches(name)) {
910       String libName = name.replaceAll("(^lib|\\.(so|dylib)$)", "");
911       options.add("-l" + libName);
912     } else {
913       // Interface shared objects have a non-standard extension
914       // that the linker won't be able to find.  So use the
915       // filename directly rather than a -l option.  Since the
916       // library has an SONAME attribute, this will work fine.
917       options.add(inputArtifact.getExecPathString());
918     }
919   }

921   /**
922    * Adds command-line options for a static library or non-library input
923    * into options.
924    *
925    * @param ltoMap is a mutable list of exec paths that should be on the command-line, which
926    *    must be supplied for LTO final links.
927    */
928   private void addStaticInputLinkOptions(
929       LinkerInput input, List<String> options, @Nullable Map<Artifact, Artifact> ltoMap) {
930     Preconditions.checkState(!isDynamicLibrary(input));
931 
932     // start-lib/end-lib library: adds its input object files.
933     if (Link.useStartEndLib(input, cppConfiguration.archiveType())) {
934       Iterable<Artifact> archiveMembers = input.getObjectFiles();
935       if (!Iterables.isEmpty(archiveMembers)) {
936         options.add("-Wl,--start-lib");
937         for (Artifact member : archiveMembers) {
938           if (ltoMap != null) {
939             Artifact backend = ltoMap.remove(member);
940 
941             if (backend == null) {
942               System.err.println(
943                   "LTO backend file missing for " + member + " already did: " + options);
944               backend = member;
945             }
946             options.add(backend.getExecPathString());
947           } else {
948             options.add(member.getExecPathString());
949           }
950         }
951         options.add("-Wl,--end-lib");
952       }
953     } else {
954       // For anything else, add the input directly.
955       Artifact inputArtifact = input.getArtifact();
956 
957       if (ltoMap != null) {
958         Artifact ltoArtifact = ltoMap.remove(inputArtifact);
959         if (ltoArtifact != null) {
960           inputArtifact = ltoArtifact;
961         }
962       }
963 
964       if (input.isFake()) {
965         options.add(Link.FAKE_OBJECT_PREFIX + inputArtifact.getExecPathString());
966       } else {
967         options.add(inputArtifact.getExecPathString());
968       }
969     }
970   }
