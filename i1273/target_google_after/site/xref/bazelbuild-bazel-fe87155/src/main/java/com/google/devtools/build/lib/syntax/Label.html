32  /**
33   * A class to identify a BUILD target. All targets belong to exactly one package.
34   * The name of a target is called its label. A typical label looks like this:
35   * //dir1/dir2:target_name where 'dir1/dir2' identifies the package containing a BUILD file,
36   * and 'target_name' identifies the target within the package.
37   *
38   * <p>Parsing is robust against bad input, for example, from the command line.
39   */
40  @SkylarkModule(name = "Label", doc = "A BUILD target identifier.")
41  @Immutable @ThreadSafe
42  public final class Label implements Comparable<Label>, Serializable {
43  
44    /**
45     * Factory for Labels from absolute string form. e.g.
46     * <pre>
47     * //foo/bar
48     * //foo/bar:quux
49     * {@literal @}foo//bar
50     * {@literal @}foo//bar:baz
51     * </pre>
52     */
53    public static Label parseAbsolute(String absName) throws LabelSyntaxException {
54      String repo = PackageIdentifier.DEFAULT_REPOSITORY;
55      int packageStartPos = absName.indexOf("//");
56      if (packageStartPos > 0) {
57        repo = absName.substring(0, packageStartPos);
58        absName = absName.substring(packageStartPos);
59      }
60      try {
61        LabelValidator.PackageAndTarget labelParts = LabelValidator.parseAbsoluteLabel(absName);
62        validate(labelParts.getPackageName(), labelParts.getTargetName());
63        return new Label(new PackageIdentifier(repo, new PathFragment(labelParts.getPackageName())),
64            labelParts.getTargetName());
65      } catch (BadLabelException e) {
66        throw new LabelSyntaxException(e.getMessage());
67      }
68    }

70    /**
71     * Alternate factory method for Labels from absolute strings. This is a convenience method for
72     * cases when a Label needs to be initialized statically, so the declared exception is
73     * inconvenient.
74     *
75     * <p>Do not use this when the argument is not hard-wired.
76     */
77    public static Label parseAbsoluteUnchecked(String absName) {
78      try {
79        return parseAbsolute(absName);
80      } catch (LabelSyntaxException e) {
81        throw new IllegalArgumentException(e);
82      }
83    }

108   /**
109    * Resolves a relative label using a workspace-relative path to the current working directory. The
110    * method handles these cases:
111    * <ul>
112    *   <li>The label is absolute.
113    *   <li>The label starts with a colon.
114    *   <li>The label consists of a relative path, a colon, and a local part.
115    *   <li>The label consists only of a local part.
116    * </ul>
117    *
118    * <p>Note that this method does not support any of the special syntactic constructs otherwise
119    * supported on the command line, like ":all", "/...", and so on.
120    *
121    * <p>It would be cleaner to use the TargetPatternEvaluator for this resolution, but that is not
122    * possible, because it is sometimes necessary to resolve a relative label before the package path
123    * is setup; in particular, before the tools/defaults package is created.
124    *
125    * @throws LabelSyntaxException if the resulting label is not valid
126    */
127   public static Label parseCommandLineLabel(String label, PathFragment workspaceRelativePath)
128       throws LabelSyntaxException {
129     Preconditions.checkArgument(!workspaceRelativePath.isAbsolute());
130     if (LabelValidator.isAbsolute(label)) {
131       return parseAbsolute(label);
132     }
133     int index = label.indexOf(':');
134     if (index < 0) {
135       index = 0;
136       label = ":" + label;
137     }
138     PathFragment path = workspaceRelativePath.getRelative(label.substring(0, index));
139     // Use the String, String constructor, to make sure that the package name goes through the
140     // validity check.
141     return new Label(path.getPathString(), label.substring(index + 1));
142   }

144   /**
145    * Validates the given target name and returns a canonical String instance if it is valid.
146    * Otherwise it throws a SyntaxException.
147    */
148   private static String canonicalizeTargetName(String name) throws LabelSyntaxException {
149     String error = LabelValidator.validateTargetName(name);
150     if (error != null) {
151       error = "invalid target name '" + StringUtilities.sanitizeControlChars(name) + "': " + error;
152       throw new LabelSyntaxException(error);
153     }
154 
155     // TODO(bazel-team): This should be an error, but we can't make it one for legacy reasons.
156     if (name.endsWith("/.")) {
157       name = name.substring(0, name.length() - 2);
158     }
159 
160     return StringCanonicalizer.intern(name);
161   }
162 
163   /**
164    * Validates the given package name and returns a canonical PathFragment instance if it is valid.
165    * Otherwise it throws a SyntaxException.
166    */
167   private static PackageIdentifier validate(String packageIdentifier, String name)
168       throws LabelSyntaxException {
169     String error = null;
170     try {
171       return PackageIdentifier.parse(packageIdentifier);
172     } catch (LabelSyntaxException e) {
173       error = e.getMessage();
174       error = "invalid package name '" + packageIdentifier + "': " + error;
175       // This check is just for a more helpful error message
176       // i.e. valid target name, invalid package name, colon-free label form
177       // used => probably they meant "//foo:bar.c" not "//foo/bar.c".
178       if (packageIdentifier.endsWith("/" + name)) {
179         error += " (perhaps you meant \":" + name + "\"?)";
180       }
181       throw new LabelSyntaxException(error);
182     }
183   }

185   /** The name and repository of the package. */
186   private final PackageIdentifier packageIdentifier;

188   /** The name of the target within the package. Canonical. */
189   private final String name;

191   /**
192    * Constructor from a package name, target name. Both are checked for validity
193    * and a SyntaxException is thrown if either is invalid.
194    * TODO(bazel-team): move the validation to {@link PackageIdentifier}. Unfortunately, there are a
195    * bazillion tests that use invalid package names (taking advantage of the fact that calling
196    * Label(PathFragment, String) doesn't validate the package name).
197    */
198   private Label(String packageIdentifier, String name) throws LabelSyntaxException {
199     this(validate(packageIdentifier, name), name);
200   }
201 
202   private Label(PackageIdentifier packageIdentifier, String name)
203       throws LabelSyntaxException {
204     Preconditions.checkNotNull(packageIdentifier);
205     Preconditions.checkNotNull(name);
206 
207     try {
208       this.packageIdentifier = packageIdentifier;
209       this.name = canonicalizeTargetName(name);
210     } catch (LabelSyntaxException e) {
211       // This check is just for a more helpful error message
212       // i.e. valid target name, invalid package name, colon-free label form
213       // used => probably they meant "//foo:bar.c" not "//foo/bar.c".
214       if (packageIdentifier.getPackageFragment().getPathString().endsWith("/" + name)) {
215         throw new LabelSyntaxException(e.getMessage() + " (perhaps you meant \":" + name + "\"?)");
216       }
217       throw e;
218     }
219   }

283    * <p>invariant: {@code parseAbsolute(x.toString()).equals(x)}

286   public String toString() {

288         + ":" + name;

292    * Renders this label in shorthand form.
293    *
294    * <p>Labels with canonical form {@code //foo/bar:bar} have the shorthand form {@code //foo/bar}.

298     return packageIdentifier.getRepository() + (getPackageFragment().getBaseName().equals(name)

312   /**
313    * Resolves a relative or absolute label name. If given name is absolute, then this method calls
314    * {@link #parseAbsolute}. Otherwise, it calls {@link #getLocalTargetLabel}.
315    *
316    * <p>For example:
317    * {@code :quux} relative to {@code //foo/bar:baz} is {@code //foo/bar:quux};
318    * {@code //wiz:quux} relative to {@code //foo/bar:baz} is {@code //wiz:quux}.
319    *
320    * @param relName the relative label name; must be non-empty.
321    */
322   @SkylarkCallable(name = "relative", doc =
323         "Resolves a label that is either absolute (starts with <code>//</code>) or relative to the"
324       + " current package.<br>"
325       + "For example:<br><ul>"
326       + "<pre class=language-python>\n"
327       + "Label(\"//foo/bar:baz\").relative(\":quux\") == Label(\"//foo/bar:quux\")\n"
328       + "Label(\"//foo/bar:baz\").relative(\"//wiz:quux\") == Label(\"//wiz:quux\")\n"
329       + "</pre>")
330   public Label getRelative(String relName) throws LabelSyntaxException {
331     if (relName.length() == 0) {
332       throw new LabelSyntaxException("empty package-relative label");
333     }
334     if (LabelValidator.isAbsolute(relName)) {
335       return parseAbsolute(relName);
336     } else if (relName.equals(":")) {
337       throw new LabelSyntaxException("':' is not a valid package-relative label");
338     } else if (relName.charAt(0) == ':') {
339       return getLocalTargetLabel(relName.substring(1));
340     } else {
341       return getLocalTargetLabel(relName);
342     }
343   }

345   /**
346    * Resolves the repository of a label in the context of another label.
347    *
348    * <p>This is necessary so that dependency edges in remote repositories do not need to explicitly
349    * mention their repository name. Otherwise, referring to e.g. <code>//a:b</code> in a remote
350    * repository would point back to the main repository, which is usually not what is intended.
351    */
352   public Label resolveRepositoryRelative(Label relative) {
353     if (packageIdentifier.getRepository().isDefault()
354         || !relative.packageIdentifier.getRepository().isDefault()) {
355       return relative;
356     } else {
357       try {
358         return new Label(
359             new PackageIdentifier(packageIdentifier.getRepository(), relative.getPackageFragment()),
360             relative.getName());
361       } catch (LabelSyntaxException e) {
362         // We are creating the new label from an existing one which is guaranteed to be valid, so
363         // this can't happen
364         throw new IllegalStateException(e);
365       }
366     }
367   }

369   @Override
370   public int hashCode() {
371     return name.hashCode() ^ packageIdentifier.hashCode();
372   }

374   /**
375    * Two labels are equal iff both their name and their package name are equal.
376    */
377   @Override
378   public boolean equals(Object other) {
379     if (!(other instanceof Label)) {
380       return false;
381     }
382     Label otherLabel = (Label) other;
383     return name.equals(otherLabel.name) // least likely one first
384         && packageIdentifier.equals(otherLabel.packageIdentifier);
385   }
386 
387   /**
388    * Defines the order between labels.
389    *
390    * <p>Labels are ordered primarily by package name and secondarily by target name. Both components
391    * are ordered lexicographically. Thus {@code //a:b/c} comes before {@code //a/b:a}, i.e. the
392    * position of the colon is significant to the order.
393    */
394   @Override
395   public int compareTo(Label other) {
396     return ComparisonChain.start()
397         .compare(packageIdentifier, other.packageIdentifier)
398         .compare(name, other.name)
399         .result();
400   }

402   /**

406   public static String print(Label label) {
407     return label == null ? "(unknown)" : label.toString();
