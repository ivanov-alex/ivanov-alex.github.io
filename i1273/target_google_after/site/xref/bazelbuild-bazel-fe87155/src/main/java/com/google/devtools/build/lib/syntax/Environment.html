45  /**
46   * An Environment is the main entry point to evaluating code in the BUILD language or Skylark.
47   * It embodies all the state that is required to evaluate such code,
48   * except for the current instruction pointer, which is an {@link ASTNode}
49   * whose {@link Statement#exec exec} or {@link Expression#eval eval} method is invoked with
50   * this Environment, in a straightforward direct-style AST-walking interpreter.
51   * {@link Continuation}-s are explicitly represented, but only partly, with another part being
52   * implicit in a series of try-catch statements, to maintain the direct style. One notable trick
53   * is how a {@link UserDefinedFunction} implements returning values as the function catching a
54   * {@link ReturnStatement.ReturnException} thrown by a {@link ReturnStatement} in the body.
55   *
56   * <p>Every Environment has a {@link Mutability} field, and must be used within a function that
57   * creates and closes this {@link Mutability} with the try-with-resource pattern.
58   * This {@link Mutability} is also used when initializing mutable objects within that Environment;
59   * when closed at the end of the computation freezes the Environment and all those objects that
60   * then become forever immutable. The pattern enforces the discipline that there should be no
61   * dangling mutable Environment, or concurrency between interacting Environment-s.
62   * It is also an error to try to mutate an Environment and its objects from another Environment,
63   * before the {@link Mutability} is closed.
64   *
65   * <p>One creates an Environment using the {@link #builder} function, then
66   * populates it with {@link #setup}, {@link #setupDynamic} and sometimes {@link #setupOverride},
67   * before to evaluate code in it with {@link #eval}, or with {@link BuildFileAST#exec}
68   * (where the AST was obtained by passing a {@link ValidationEnvironment} constructed from the
69   * Environment to {@link BuildFileAST#parseBuildFile} or {@link BuildFileAST#parseSkylarkFile}).
70   * When the computation is over, the frozen Environment can still be queried with {@link #lookup}.
71   *
72   * <p>Final fields of an Environment represent its dynamic state, i.e. state that remains the same
73   * throughout a given evaluation context, and don't change with source code location,
74   * while mutable fields embody its static state, that change with source code location.
75   * The seeming paradox is that the words "dynamic" and "static" refer to the point of view
76   * of the source code, and here we have a dual point of view.
77   */
78  public final class Environment implements Freezable {
79  
80    /**
81     * A Frame is a Map of bindings, plus a {@link Mutability} and a parent Frame
82     * from which to inherit bindings.
83     *
84     * <p>A Frame contains bindings mapping variable name to variable value in a given scope.
85     * It may also inherit bindings from a parent Frame corresponding to a parent scope,
86     * which in turn may inherit bindings from its own parent, etc., transitively.
87     * Bindings may shadow bindings from the parent. In Skylark, you may only mutate
88     * bindings from the current Frame, which always got its {@link Mutability} with the
89     * current {@link Environment}; but future extensions may make it more like Python
90     * and allow mutation of bindings in outer Frame-s (or then again may not).
91     *
92     * <p>A Frame inherits the {@link Mutability} from the {@link Environment} in which it was
93     * originally created. When that {@link Environment} is finalized and its {@link Mutability}
94     * is closed, it becomes immutable, including the Frame, which can be shared in other
95     * {@link Environment}-s. Indeed, a {@link UserDefinedFunction} will close over the global
96     * Frame of its definition {@link Environment}, which will thus be reused (immutably)
97     * in all any {@link Environment} in which this function is called, so it's important to
98     * preserve the {@link Mutability} to make sure no Frame is modified after it's been finalized.
99     */
100   public static final class Frame implements Freezable {
101 
102     private final Mutability mutability;
103     final Frame parent;
104     final Map<String, Object> bindings = new HashMap<>();
105 
106     Frame(Mutability mutability, Frame parent) {
107       this.mutability = mutability;
108       this.parent = parent;
109     }
110 
111     @Override
112     public final Mutability mutability() {
113       return mutability;
114     }
115 
116     /**
117      * Gets a binding from the current frame or if not found its parent.
118      * @param varname the name of the variable to be bound
119      * @return the value bound to variable
120      */
121     public Object get(String varname) {
122       if (bindings.containsKey(varname)) {
123         return bindings.get(varname);
124       }
125       if (parent != null) {
126         return parent.get(varname);
127       }
128       return null;
129     }
130 
131     /**
132      * Modifies a binding in the current Frame.
133      * Does not try to modify an inherited binding.
134      * This will shadow any inherited binding, which may be an error
135      * that you want to guard against before calling this function.
136      * @param env the Environment attempting the mutation
137      * @param varname the name of the variable to be bound
138      * @param value the value to bind to the variable
139      */
140     public void put(Environment env, String varname, Object value)
141         throws MutabilityException {
142       Mutability.checkMutable(this, env);
143       bindings.put(varname, value);
144     }
145 
146     /**
147      * Adds the variable names of this Frame and its transitive parents to the given set.
148      * This provides a O(n) way of extracting the list of all variables visible in an Environment.
149      * @param vars the set of visible variables in the Environment, being computed.
150      */
151     public void addVariableNamesTo(Set<String> vars) {
152       vars.addAll(bindings.keySet());
153       if (parent != null) {
154         parent.addVariableNamesTo(vars);
155       }
156     }
157 
158     public Set<String> getDirectVariableNames() {
159       return bindings.keySet();
160     }
161 
162     @Override
163     public String toString() {
164       String prefix = "Frame";
165       StringBuilder sb = new StringBuilder();
166       for (Frame f = this; f != null; f = f.parent) {
167         Printer.formatTo(sb, "%s%s%r",
168             ImmutableList.<Object>of(prefix, f.mutability(), f.bindings));
169         prefix = "=>";
170       }
171       return sb.toString();
172     }
173   }

338   /**
339    * Enters a scope by saving state to a new Continuation
340    * @param function the function whose scope to enter
341    * @param caller the source AST node for the caller
342    * @param globals the global Frame that this function closes over from its definition Environment

345     continuation = new Continuation(
346         continuation, function, caller, lexicalFrame, globalFrame, knownGlobalVariables, isSkylark);

462   /**
463    * Returns the FuncallExpression and the BaseFunction for the top-level call being evaluated.
464    */
465   public Pair<FuncallExpression, BaseFunction> getTopCall() {
466     Continuation continuation = this.continuation;
467     if (continuation == null) {
468       return null;
469     }
470     while (continuation.continuation != null) {
471       continuation = continuation.continuation;
472     }
473     return new Pair<>(continuation.caller, continuation.function);
474   }

563     /** Builds the Environment. */

575           globalFrame,
576           dynamicFrame,
577           eventHandler,
578           importedExtensions,
579           isSkylark,
580           fileContentHashCode,
581           isLoadingPhase);
582       return env;
583     }

590   /**
591    * Sets a binding for a special dynamic variable in this Environment.
592    * This is not for end-users, and will throw an AssertionError in case of conflict.
593    * @param varname the name of the dynamic variable to be bound
594    * @param value a value to bind to the variable
595    * @return this Environment, in fluid style
596    */
597   public Environment setupDynamic(String varname, Object value) {
598     if (dynamicFrame.get(varname) != null) {
599       throw new AssertionError(
600           String.format("Trying to bind dynamic variable '%s' but it is already bound",
601               varname));
602     }
603     if (lexicalFrame != null && lexicalFrame.get(varname) != null) {
604       throw new AssertionError(
605           String.format("Trying to bind dynamic variable '%s' but it is already bound lexically",
606               varname));
607     }
608     if (globalFrame.get(varname) != null) {
609       throw new AssertionError(
610           String.format("Trying to bind dynamic variable '%s' but it is already bound globally",
611               varname));
612     }
613     try {
614       dynamicFrame.put(this, varname, value);
615     } catch (MutabilityException e) {
616       // End users don't have access to setupDynamic, and it is an implementation error
617       // if we encounter a mutability exception.
618       throw new AssertionError(
619           Printer.format(
620               "Trying to bind dynamic variable '%s' in frozen environment %r", varname, this),
621           e);
622     }
623     return this;
624   }

627   /**
628    * Modifies a binding in the current Frame of this Environment, as would an
629    * {@link AssignmentStatement}. Does not try to modify an inherited binding.
630    * This will shadow any inherited binding, which may be an error
631    * that you want to guard against before calling this function.
632    * @param varname the name of the variable to be bound
633    * @param value the value to bind to the variable
634    * @return this Environment, in fluid style
635    */
636   public Environment update(String varname, Object value) throws EvalException {
637     Preconditions.checkNotNull(value, "update(value == null)");
638     // prevents clashes between static and dynamic variables.
639     if (dynamicFrame.get(varname) != null) {
640       throw new EvalException(
641           null, String.format("Trying to update special read-only global variable '%s'", varname));
642     }
643     if (isKnownGlobalVariable(varname)) {
644       throw new EvalException(
645           null, String.format("Trying to update read-only global variable '%s'", varname));
646     }
647     try {
648       currentFrame().put(this, varname, Preconditions.checkNotNull(value));
649     } catch (MutabilityException e) {
650       // Note that since at this time we don't accept the global keyword, and don't have closures,
651       // end users should never be able to mutate a frozen Environment, and a MutabilityException
652       // is therefore a failed assertion for Bazel. However, it is possible to shadow a binding
653       // imported from a parent Environment by updating the current Environment, which will not
654       // trigger a MutabilityException.
655       throw new AssertionError(
656           Printer.format("Can't update %s to %r in frozen environment", varname, value),
657           e);
658     }
659     return this;
660   }

674    * @param varname the name of the variable to be bound

700   /**
701    * @return the value from the environment whose name is "varname".
702    * @throws NoSuchVariableException if the variable is not defined in the Environment.
703    */
704   public Object lookup(String varname) throws NoSuchVariableException {
705     // Which Frame to lookup first doesn't matter because update prevents clashes.
706     if (lexicalFrame != null) {
707       Object lexicalValue = lexicalFrame.get(varname);
708       if (lexicalValue != null) {
709         return lexicalValue;
710       }
711     }
712     Object globalValue = globalFrame.get(varname);
713     Object dynamicValue = dynamicFrame.get(varname);
714     if (globalValue == null && dynamicValue == null) {
715       throw new NoSuchVariableException(varname);
716     }
717     if (knownGlobalVariables != null) {
718       knownGlobalVariables.add(varname);
719     }
720     if (globalValue != null) {
721       return globalValue;
722     }
723     return dynamicValue;
724   }

726   /**
727    * Like {@link #lookup(String)}, but instead of throwing an exception in the case
728    * where <code>varname</code> is not defined, <code>defaultValue</code> is returned instead.
729    */
730   public Object lookup(String varname, Object defaultValue) {
731     Preconditions.checkState(!isSkylark);
732     try {
733       return lookup(varname);
734     } catch (NoSuchVariableException e) {
735       return defaultValue;
736     }
737   }

810    * An Exception thrown when an attempt is made to import a symbol from a file
811    * that was not properly loaded.
812    */
813   public static class LoadFailedException extends Exception {
814     LoadFailedException(PathFragment extension) {
815       super(String.format("file '%s' was not correctly loaded. "
816               + "Make sure the 'load' statement appears in the global scope in your file",
817               extension));
818     }

821   public void importSymbol(PathFragment extension, Identifier symbol, String nameInLoadedFile)
822       throws NoSuchVariableException, LoadFailedException {
823     Preconditions.checkState(isGlobal()); // loading is only allowed at global scope.
824 
825     if (!importedExtensions.containsKey(extension)) {
826       throw new LoadFailedException(extension);
827     }
828 
829     Extension ext = importedExtensions.get(extension);
830 
831     // TODO(bazel-team): Throw a LoadFailedException instead, with an appropriate message.
832     // Throwing a NoSuchVariableException is backward compatible, but backward.
833     if (!ext.containsKey(nameInLoadedFile)) {
834       throw new NoSuchVariableException(nameInLoadedFile);
835     }
836 
837     Object value = ext.get(nameInLoadedFile);
838     // TODO(bazel-team): Unify data structures between Skylark and BUILD,
839     // and stop doing the conversions below:
840     if (!isSkylark) {
841       value = SkylarkType.convertFromSkylark(value);
842     }
843 
844     try {
845       update(symbol.getName(), value);
846     } catch (EvalException e) {
847       throw new LoadFailedException(extension);
848     }
849   }

851   /**
852    * Returns a hash code calculated from the hash code of this Environment and the
853    * transitive closure of other Environments it loads.
854    */
855   public String getTransitiveContentHashCode() {
856     // Calculate a new hash from the hash of the loaded Extension-s.
857     Fingerprint fingerprint = new Fingerprint();
858     fingerprint.addString(Preconditions.checkNotNull(fileContentHashCode));
859     TreeSet<PathFragment> paths = new TreeSet<>(importedExtensions.keySet());
860     for (PathFragment path : paths) {
861       fingerprint.addString(importedExtensions.get(path).getTransitiveContentHashCode());
862     }
863     return fingerprint.hexDigestAndReset();
864   }

964   /**
965    * Evaluates code some String input without a supporting file.
966    * @param input a list of lines of code to evaluate
967    * @return the value of the last statement if it's an Expression or else null
968    */
969   @Nullable public Object eval(String... input) throws EvalException, InterruptedException {
970     Object last = null;
971     for (Statement statement : parseFile(input)) {
972       if (statement instanceof ExpressionStatement) {
973         last = ((ExpressionStatement) statement).getExpression().eval(this);
974       } else {
975         statement.exec(this);
976         last = null;
977       }
978     }
979     return last;
980   }
