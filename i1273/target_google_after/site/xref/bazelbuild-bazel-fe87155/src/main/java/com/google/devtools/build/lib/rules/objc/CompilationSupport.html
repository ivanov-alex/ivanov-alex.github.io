141   /**
142    * Returns information about the given rule's compilation artifacts.

144   // TODO(bazel-team): Remove this information from ObjcCommon and move it internal to this class.
145   static CompilationArtifacts compilationArtifacts(RuleContext ruleContext) {
146     PrerequisiteArtifacts srcs = ruleContext.getPrerequisiteArtifacts("srcs", Mode.TARGET)
147         .errorsForNonMatching(SRCS_TYPE);
148     return new CompilationArtifacts.Builder()
149         .addSrcs(srcs.filter(COMPILABLE_SRCS_TYPE).list())
150         .addNonArcSrcs(ruleContext.getPrerequisiteArtifacts("non_arc_srcs", Mode.TARGET)
151             .errorsForNonMatching(NON_ARC_SRCS_TYPE)
152             .list())
153         .addPrivateHdrs(srcs.filter(HEADERS).list())
154         .setIntermediateArtifacts(ObjcRuleClasses.intermediateArtifacts(ruleContext))
155         .setPchFile(Optional.fromNullable(ruleContext.getPrerequisiteArtifact("pch", Mode.TARGET)))
156         .build();
157   }

170   /**
171    * Registers all actions necessary to compile this rule's sources and archive them.
172    *
173    * @param common common information about this rule and its dependencies
174    * @return this compilation support
175    */
176   CompilationSupport registerCompileAndArchiveActions(ObjcCommon common) {
177     if (common.getCompilationArtifacts().isPresent()) {
178       registerGenerateModuleMapAction(common.getCompilationArtifacts());
179       IntermediateArtifacts intermediateArtifacts =
180           ObjcRuleClasses.intermediateArtifacts(ruleContext);
181       Optional<CppModuleMap> moduleMap;
182       if (ObjcRuleClasses.objcConfiguration(ruleContext).moduleMapsEnabled()) {
183         moduleMap = Optional.of(intermediateArtifacts.moduleMap());
184       } else {
185         moduleMap = Optional.<CppModuleMap>absent();
186       }
187       registerCompileAndArchiveActions(
188           common.getCompilationArtifacts().get(),
189           intermediateArtifacts,
190           common.getObjcProvider(),
191           moduleMap,
192           ruleContext.getConfiguration().isCodeCoverageEnabled());
193     }
194     return this;
195   }

197   /**
198    * Creates actions to compile each source file individually, and link all the compiled object
199    * files into a single archive library.
200    */
201   private void registerCompileAndArchiveActions(
202       CompilationArtifacts compilationArtifacts,
203       IntermediateArtifacts intermediateArtifacts,
204       ObjcProvider objcProvider,
205       Optional<CppModuleMap> moduleMap,
206       boolean isCodeCoverageEnabled) {
207     ImmutableList.Builder<Artifact> objFiles = new ImmutableList.Builder<>();
208     for (Artifact sourceFile : compilationArtifacts.getSrcs()) {
209       Artifact objFile = intermediateArtifacts.objFile(sourceFile);
210       objFiles.add(objFile);
211       if (ObjcRuleClasses.SWIFT_SOURCES.matches(sourceFile.getFilename())) {
212         registerSwiftCompileAction(sourceFile, objFile, intermediateArtifacts, objcProvider);
213       } else {
214         registerCompileAction(
215             sourceFile,
216             objFile,
217             objcProvider,
218             moduleMap,
219             intermediateArtifacts,
220             compilationArtifacts,
221             ImmutableList.of("-fobjc-arc"),
222             isCodeCoverageEnabled);
223       }
224     }
225     for (Artifact nonArcSourceFile : compilationArtifacts.getNonArcSrcs()) {
226       Artifact objFile = intermediateArtifacts.objFile(nonArcSourceFile);
227       objFiles.add(objFile);
228       registerCompileAction(
229           nonArcSourceFile,
230           objFile,
231           objcProvider,
232           moduleMap,
233           intermediateArtifacts,
234           compilationArtifacts,
235           ImmutableList.of("-fno-objc-arc"),
236           isCodeCoverageEnabled);
237     }
238 
239     if (compilationArtifacts.hasSwiftSources()) {
240       registerSwiftModuleMergeAction(intermediateArtifacts, compilationArtifacts);
241     }
242 
243     for (Artifact archive : compilationArtifacts.getArchive().asSet()) {
244       registerArchiveActions(intermediateArtifacts, objFiles, archive);
245     }
246   }

248   private void registerCompileAction(
249       Artifact sourceFile,
250       Artifact objFile,
251       ObjcProvider objcProvider,
252       Optional<CppModuleMap> moduleMap,
253       IntermediateArtifacts intermediateArtifacts,
254       CompilationArtifacts compilationArtifacts,
255       Iterable<String> otherFlags,
256       boolean isCodeCoverageEnabled) {
257     ObjcConfiguration objcConfiguration = ObjcRuleClasses.objcConfiguration(ruleContext);
258     ImmutableList.Builder<String> coverageFlags = new ImmutableList.Builder<>();
259     ImmutableList.Builder<Artifact> gcnoFiles = new ImmutableList.Builder<>();
260     ImmutableList.Builder<Artifact> additionalInputs = new ImmutableList.Builder<>();
261     if (isCodeCoverageEnabled && ObjcRuleClasses.isInstrumentable(sourceFile)) {
262       coverageFlags.addAll(CLANG_COVERAGE_FLAGS);
263       gcnoFiles.add(intermediateArtifacts.gcnoFile(sourceFile));
264     }
265     CustomCommandLine.Builder commandLine = new CustomCommandLine.Builder();
266     if (ObjcRuleClasses.CPP_SOURCES.matches(sourceFile.getExecPath())) {
267       commandLine.add("-stdlib=libc++");
268     }
269 
270     if (compilationArtifacts.hasSwiftSources()) {
271       // Add the directory that contains merged TargetName-Swift.h header to search path, in case
272       // any of ObjC files use it.
273       commandLine.add("-I");
274       commandLine.addPath(intermediateArtifacts.swiftHeader().getExecPath().getParentDirectory());
275       additionalInputs.add(intermediateArtifacts.swiftHeader());
276     }
277 
278     // The linker needs full debug symbol information to perform binary dead-code stripping.
279     if (objcConfiguration.shouldStripBinary()) {
280       commandLine.add("-g");
281     }
282 
283     Artifact dotdFile = intermediateArtifacts.dotdFile(sourceFile);
284     commandLine
285         .add(IosSdkCommands.compileFlagsForClang(objcConfiguration))
286         .add(IosSdkCommands.commonLinkAndCompileFlagsForClang(objcProvider, objcConfiguration))
287         .add(objcConfiguration.getCoptsForCompilationMode())
288         .addBeforeEachPath(
289             "-iquote", ObjcCommon.userHeaderSearchPaths(ruleContext.getConfiguration()))
290         .addBeforeEachExecPath("-include", compilationArtifacts.getPchFile().asSet())
291         .addBeforeEachPath("-I", objcProvider.get(INCLUDE))
292         .addBeforeEachPath("-isystem", objcProvider.get(INCLUDE_SYSTEM))
293         .add(otherFlags)
294         .addFormatEach("-D%s", objcProvider.get(DEFINE))
295         .add(coverageFlags.build())
296         .add(objcConfiguration.getCopts())
297         .add(attributes.copts())
298         .add(attributes.optionsCopts())
299         .addExecPath("-c", sourceFile)
300         .addExecPath("-o", objFile)
301         .add("-MD")
302         .addExecPath("-MF", dotdFile);
303 
304     if (moduleMap.isPresent()) {
305       // -fmodule-map-file only loads the module in Xcode 7, so we add the module maps's directory
306       // to the include path instead.
307       // TODO(bazel-team): Use -fmodule-map-file when Xcode 6 support is dropped.
308       commandLine
309           .add(attributes.enableModules() ? "-fmodules" : "-fmodule-maps")
310           .add("-iquote")
311           .add(
312               moduleMap
313                   .get()
314                   .getArtifact()
315                   .getExecPath()
316                   .getParentDirectory()
317                   .toString())
318           .add("-fmodule-name=" + moduleMap.get().getName());
319       if (attributes.enableModules()) {
320         String cachePath =
321             ruleContext.getConfiguration().getGenfilesFragment() + "/_objc_module_cache";
322         commandLine.add("-fmodules-cache-path=" + cachePath);
323       }
324     }
325 
326     // TODO(bazel-team): Remote private headers from inputs once they're added to the provider.
327     ruleContext.registerAction(ObjcRuleClasses.spawnOnDarwinActionBuilder(ruleContext)
328         .setMnemonic("ObjcCompile")
329         .setExecutable(CLANG)
330         .setCommandLine(commandLine.build())
331         .addInput(sourceFile)
332         .addInputs(additionalInputs.build())
333         .addOutput(objFile)
334         .addOutputs(gcnoFiles.build())
335         .addOutput(dotdFile)
336         .addTransitiveInputs(objcProvider.get(HEADER))
337         .addTransitiveInputs(objcProvider.get(MODULE_MAP))
338         .addInputs(compilationArtifacts.getPrivateHdrs())
339         .addTransitiveInputs(objcProvider.get(FRAMEWORK_FILE))
340         .addInputs(compilationArtifacts.getPchFile().asSet())
341         .build(ruleContext));
342   }

344   /**
345    * Compiles a single swift file.
346    *
347    * @param sourceFile the artifact to compile
348    * @param objFile the resulting object artifact
349    */
350   private void registerSwiftCompileAction(
351       Artifact sourceFile,
352       Artifact objFile,
353       IntermediateArtifacts intermediateArtifacts,
354       ObjcProvider objcProvider) {
355     ObjcConfiguration objcConfiguration = ObjcRuleClasses.objcConfiguration(ruleContext);
356 
357     // Compiling a single swift file requires knowledge of all of the other
358     // swift files in the same module. The primary file ({@code sourceFile}) is
359     // compiled to an object file, while the remaining files are used to resolve
360     // symbols (they behave like c/c++ headers in this context).
361     ImmutableSet.Builder<Artifact> otherSwiftSourcesBuilder = ImmutableSet.builder();
362     for (Artifact otherSourceFile : compilationArtifacts(ruleContext).getSrcs()) {
363       if (ObjcRuleClasses.SWIFT_SOURCES.matches(otherSourceFile.getFilename())
364           && otherSourceFile != sourceFile) {
365         otherSwiftSourcesBuilder.add(otherSourceFile);
366       }
367     }
368     ImmutableSet<Artifact> otherSwiftSources = otherSwiftSourcesBuilder.build();
369 
370     CustomCommandLine.Builder commandLine = new CustomCommandLine.Builder()
371         .add("-frontend")
372         .add("-emit-object")
373         .add("-target").add(IosSdkCommands.swiftTarget(objcConfiguration))
374         .add("-sdk").add(IosSdkCommands.sdkDir(objcConfiguration))
375         .add("-enable-objc-interop");
376 
377     if (objcConfiguration.generateDebugSymbols()) {
378       commandLine.add("-g");
379     }
380 
381     commandLine
382       .add("-Onone")
383       .add("-module-name").add(getModuleName())
384       .add("-parse-as-library")
385       .addExecPath("-primary-file", sourceFile)
386       .addExecPaths(otherSwiftSources)
387       .addExecPath("-o", objFile)
388       .addExecPath("-emit-module-path", intermediateArtifacts.swiftModuleFile(sourceFile));
389 
390 
391     ImmutableList.Builder<Artifact> inputHeaders = ImmutableList.builder();
392     inputHeaders.addAll(attributes.hdrs());
393 
394     Optional<Artifact> bridgingHeader = attributes.bridgingHeader();
395     if (bridgingHeader.isPresent()) {
396       commandLine.addExecPath("-import-objc-header", bridgingHeader.get());
397       inputHeaders.add(bridgingHeader.get());
398     }
399 
400     // Import the Objective-C module map.
401     // TODO(bazel-team): Find a way to import the module map directly, instead of the parent
402     // directory?
403     if (objcConfiguration.moduleMapsEnabled()) {
404       PathFragment moduleMapPath = intermediateArtifacts.moduleMap().getArtifact().getExecPath();
405       commandLine.add("-I").add(moduleMapPath.getParentDirectory().toString());
406     }
407 
408     ruleContext.registerAction(
409         ObjcRuleClasses.spawnOnDarwinActionBuilder(ruleContext)
410             .setMnemonic("SwiftCompile")
411             .setExecutable(SWIFT)
412             .setCommandLine(commandLine.build())
413             .addInput(sourceFile)
414             .addInputs(otherSwiftSources)
415             .addInputs(inputHeaders.build())
416             .addTransitiveInputs(objcProvider.get(HEADER))
417             .addTransitiveInputs(objcProvider.get(MODULE_MAP))
418             .addOutput(objFile)
419             .addOutput(intermediateArtifacts.swiftModuleFile(sourceFile))
420             .build(ruleContext));
421   }

423   /**
424    * Merges multiple .partial_swiftmodule files together. Also produces a swift header that can be
425    * used by Objective-C code.
426    */
427   private void registerSwiftModuleMergeAction(
428       IntermediateArtifacts intermediateArtifacts,
429       CompilationArtifacts compilationArtifacts) {
430     ObjcConfiguration objcConfiguration = ObjcRuleClasses.objcConfiguration(ruleContext);
431 
432     ImmutableList.Builder<Artifact> moduleFiles = new ImmutableList.Builder<>();
433     for (Artifact src : compilationArtifacts.getSrcs()) {
434       if (ObjcRuleClasses.SWIFT_SOURCES.matches(src.getFilename())) {
435         moduleFiles.add(intermediateArtifacts.swiftModuleFile(src));
436       }
437     }
438 
439     CustomCommandLine.Builder commandLine = new CustomCommandLine.Builder();
440     commandLine.add("-frontend");
441     commandLine.add("-emit-module");
442     commandLine.add("-sdk").add(IosSdkCommands.sdkDir(objcConfiguration));
443     commandLine.add("-target").add(IosSdkCommands.swiftTarget(objcConfiguration));
444     if (objcConfiguration.generateDebugSymbols()) {
445       commandLine.add("-g");
446     }
447 
448     commandLine.add("-module-name").add(getModuleName());
449     commandLine.add("-parse-as-library");
450     commandLine.addExecPaths(moduleFiles.build());
451     commandLine.addExecPath("-o", intermediateArtifacts.swiftModule());
452     commandLine.addExecPath("-emit-objc-header-path", intermediateArtifacts.swiftHeader());
453 
454     ruleContext.registerAction(ObjcRuleClasses.spawnOnDarwinActionBuilder(ruleContext)
455         .setMnemonic("SwiftModuleMerge")
456         .setExecutable(SWIFT)
457         .setCommandLine(commandLine.build())
458         .addInputs(moduleFiles.build())
459         .addOutput(intermediateArtifacts.swiftModule())
460         .addOutput(intermediateArtifacts.swiftHeader())
461         .build(ruleContext));
462   }

464   private void registerArchiveActions(IntermediateArtifacts intermediateArtifacts,
465       ImmutableList.Builder<Artifact> objFiles, Artifact archive) {
466     for (Action action : archiveActions(ruleContext, objFiles.build(), archive,
467         ObjcRuleClasses.objcConfiguration(ruleContext),
468         intermediateArtifacts.objList())) {
469       ruleContext.registerAction(action);
470     }
471   }

473   private Iterable<Action> archiveActions(
474       ActionConstructionContext context,
475       Iterable<Artifact> objFiles,
476       Artifact archive,
477       ObjcConfiguration objcConfiguration,
478       Artifact objList) {
479 
480     ImmutableList.Builder<Action> actions = new ImmutableList.Builder<>();
481 
482     actions.add(new FileWriteAction(
483         context.getActionOwner(),
484         objList,
485         Artifact.joinExecPaths("\n", objFiles),
486         /*makeExecutable=*/ false));
487 
488     actions.add(ObjcRuleClasses.spawnOnDarwinActionBuilder(ruleContext)
489         .setMnemonic("ObjcLink")
490         .setExecutable(ObjcRuleClasses.LIBTOOL)
491         .setCommandLine(new CustomCommandLine.Builder()
492             .add("-static")
493             .add("-filelist").add(objList.getExecPathString())
494             .add("-arch_only").add(objcConfiguration.getIosCpu())
495             .add("-syslibroot").add(IosSdkCommands.sdkDir(objcConfiguration))
496             .add("-o").add(archive.getExecPathString())
497             .build())
498         .addInputs(objFiles)
499         .addInput(objList)
500         .addOutput(archive)
501         .build(context));
502 
503     return actions.build();
504   }

506   /**
507    * Registers any actions necessary to link this rule and its dependencies.
508    *
509    * <p>Dsym bundle and breakpad files are generated if
510    * {@link ObjcConfiguration#generateDebugSymbols()} is set.
511    *
512    * <p>When Bazel flags {@code --compilation_mode=opt} and {@code --objc_enable_binary_stripping}
513    * are specified, additional optimizations will be performed on the linked binary: all-symbol
514    * stripping (using {@code /usr/bin/strip}) and dead-code stripping (using linker flags:
515    * {@code -dead_strip} and {@code -no_dead_strip_inits_and_terms}).
516    *
517    * @param objcProvider common information about this rule's attributes and its dependencies
518    * @param extraLinkArgs any additional arguments to pass to the linker
519    * @param extraLinkInputs any additional input artifacts to pass to the link action
520    *
521    * @return this compilation support
522    */
523   CompilationSupport registerLinkActions(ObjcProvider objcProvider, ExtraLinkArgs extraLinkArgs,
524       Iterable<Artifact> extraLinkInputs) {
525     IntermediateArtifacts intermediateArtifacts =
526         ObjcRuleClasses.intermediateArtifacts(ruleContext);
527     Optional<Artifact> dsymBundle;
528     if (ObjcRuleClasses.objcConfiguration(ruleContext).generateDebugSymbols()) {
529       registerDsymActions();
530       dsymBundle = Optional.of(intermediateArtifacts.dsymBundle());
531     } else {
532       dsymBundle = Optional.absent();
533     }
534 
535     registerLinkAction(objcProvider, extraLinkArgs, extraLinkInputs, dsymBundle);
536     return this;
537   }

539   /**
540    * Registers an action that will generate a clang module map for this target, using the hdrs
541    * attribute of this rule.
542    */
543   public CompilationSupport registerGenerateModuleMapAction(
544       Optional<CompilationArtifacts> compilationArtifacts) {
545     if (ObjcRuleClasses.objcConfiguration(ruleContext).moduleMapsEnabled()) {
546       Iterable<Artifact> publicHeaders = attributes.hdrs();
547       Iterable<Artifact> privateHeaders = ImmutableList.<Artifact>of();
548       if (compilationArtifacts.isPresent()) {
549         CompilationArtifacts artifacts = compilationArtifacts.get();
550         publicHeaders = Iterables.concat(publicHeaders, artifacts.getAdditionalHdrs());
551         privateHeaders = Iterables.concat(privateHeaders, artifacts.getPrivateHdrs());
552       }
553       CppModuleMap moduleMap = ObjcRuleClasses.intermediateArtifacts(ruleContext).moduleMap();
554       registerGenerateModuleMapAction(moduleMap, publicHeaders, privateHeaders);
555     }
556     return this;
557   }

560    * Registers an action that will generate a clang module map.
561    *
562    * @param moduleMap the module map to generate
563    * @param publicHeaders the headers that should be directly accessible by dependers
564    * @param privateHeaders the headers that should only be directly accessible by this module
565    */
566   private void registerGenerateModuleMapAction(
567       CppModuleMap moduleMap, Iterable<Artifact> publicHeaders, Iterable<Artifact> privateHeaders) {
568     // The current clang (clang-600.0.57) on Darwin doesn't support 'textual', so we can't have
569     // '.inc' files in the module map (since they're implictly textual).
570     // TODO(bazel-team): Remove filtering once clang-700 is the base clang we support.
571     publicHeaders = Iterables.filter(publicHeaders, NON_INC_FILES);
572     privateHeaders = Iterables.filter(privateHeaders, NON_INC_FILES);
573     ruleContext.registerAction(
574         new CppModuleMapAction(
575             ruleContext.getActionOwner(),
576             moduleMap,
577             privateHeaders,
578             publicHeaders,
579             attributes.moduleMapsForDirectDeps(),
580             ImmutableList.<PathFragment>of(),
581             /*compiledModule=*/ true,
582             /*moduleMapHomeIsCwd=*/ false,
583             /*generateSubModules=*/ true));

586   private void registerLinkAction(ObjcProvider objcProvider, ExtraLinkArgs extraLinkArgs,
587       Iterable<Artifact> extraLinkInputs, Optional<Artifact> dsymBundle) {
588     ObjcConfiguration objcConfiguration = ObjcRuleClasses.objcConfiguration(ruleContext);
589     IntermediateArtifacts intermediateArtifacts =
590         ObjcRuleClasses.intermediateArtifacts(ruleContext);
591 
592     // When compilation_mode=opt and objc_enable_binary_stripping are specified, the unstripped
593     // binary containing debug symbols is generated by the linker, which also needs the debug
594     // symbols for dead-code removal. The binary is also used to generate dSYM bundle if
595     // --objc_generate_debug_symbol is specified. A symbol strip action is later registered to strip
596     // the symbol table from the unstripped binary.
597     Artifact binaryToLink =
598         objcConfiguration.shouldStripBinary()
599             ? intermediateArtifacts.unstrippedSingleArchitectureBinary()
600             : intermediateArtifacts.strippedSingleArchitectureBinary();
601 
602     ImmutableList<Artifact> ccLibraries = ccLibraries(objcProvider);
603     ruleContext.registerAction(
604         ObjcRuleClasses.spawnOnDarwinActionBuilder(ruleContext)
605             .setMnemonic("ObjcLink")
606             .setShellCommand(ImmutableList.of("/bin/bash", "-c"))
607             .setCommandLine(
608                 linkCommandLine(extraLinkArgs, objcProvider, binaryToLink, dsymBundle, ccLibraries))
609             .addOutput(binaryToLink)
610             .addOutputs(dsymBundle.asSet())
611             .addTransitiveInputs(objcProvider.get(LIBRARY))
612             .addTransitiveInputs(objcProvider.get(IMPORTED_LIBRARY))
613             .addTransitiveInputs(objcProvider.get(FRAMEWORK_FILE))
614             .addInputs(ccLibraries)
615             .addInputs(extraLinkInputs)
616             .build(ruleContext));
617 
618     if (objcConfiguration.shouldStripBinary()) {
619       // For test targets, only debug symbols are stripped off, since /usr/bin/strip is not able
620       // to strip off all symbols in XCTest bundle.
621       boolean isTestTarget = TargetUtils.isTestRule(ruleContext.getRule());
622       Iterable<String> stripArgs =
623           isTestTarget ? ImmutableList.of("-S") : ImmutableList.<String>of();
624       Artifact strippedBinary = intermediateArtifacts.strippedSingleArchitectureBinary();
625 
626       ruleContext.registerAction(
627           ObjcRuleClasses.spawnOnDarwinActionBuilder(ruleContext)
628               .setMnemonic("ObjcBinarySymbolStrip")
629               .setExecutable(STRIP)
630               .setCommandLine(symbolStripCommandLine(stripArgs, binaryToLink, strippedBinary))
631               .addOutput(strippedBinary)
632               .addInput(binaryToLink)
633               .build(ruleContext));
634     }
635   }

640       ccLibraryBuilder.add(libraryToLink.getArtifact());

654   private CommandLine linkCommandLine(ExtraLinkArgs extraLinkArgs,
655       ObjcProvider objcProvider, Artifact linkedBinary, Optional<Artifact> dsymBundle,
656       ImmutableList<Artifact> ccLibraries) {
657     ObjcConfiguration objcConfiguration = ObjcRuleClasses.objcConfiguration(ruleContext);
658 
659     CustomCommandLine.Builder commandLine = CustomCommandLine.builder();
660 
661     if (objcProvider.is(USES_CPP)) {
662       commandLine
663           .addPath(CLANG_PLUSPLUS)
664           .add("-stdlib=libc++");
665     } else {
666       commandLine.addPath(CLANG);
667     }
668 
669     // Do not perform code stripping on tests because XCTest binary is linked not as an executable
670     // but as a bundle without any entry point.
671     boolean isTestTarget = TargetUtils.isTestRule(ruleContext.getRule());
672     if (objcConfiguration.shouldStripBinary() && !isTestTarget) {
673       commandLine.add("-dead_strip").add("-no_dead_strip_inits_and_terms");
674     }
675 
676     commandLine
677         .add(IosSdkCommands.commonLinkAndCompileFlagsForClang(objcProvider, objcConfiguration))
678         .add("-Xlinker")
679         .add("-objc_abi_version")
680         .add("-Xlinker")
681         .add("2")
682         .add("-fobjc-link-runtime")
683         .add(IosSdkCommands.DEFAULT_LINKER_FLAGS)
684         .addBeforeEach("-framework", frameworkNames(objcProvider))
685         .addBeforeEach("-weak_framework", SdkFramework.names(objcProvider.get(WEAK_SDK_FRAMEWORK)))
686         .addExecPath("-o", linkedBinary)
687         .addExecPaths(objcProvider.get(LIBRARY))
688         .addExecPaths(objcProvider.get(IMPORTED_LIBRARY))
689         .addExecPaths(ccLibraries)
690         .add(dylibPaths(objcProvider))
691         .addBeforeEach("-force_load", Artifact.toExecPaths(objcProvider.get(FORCE_LOAD_LIBRARY)))
692         .add(extraLinkArgs)
693         .build();
694 
695     if (ruleContext.getConfiguration().isCodeCoverageEnabled()) {
696       commandLine.add(LINKER_COVERAGE_FLAGS);
697     }
698 
699     if (objcProvider.is(USES_SWIFT)) {
700       commandLine.add("-L").add(IosSdkCommands.swiftLibDir(objcConfiguration));
701     }
702 
703     if (objcProvider.is(USES_SWIFT) || objcProvider.is(USES_FRAMEWORKS)) {
704       // Enable loading bundled frameworks.
705       commandLine
706           .add("-Xlinker").add("-rpath")
707           .add("-Xlinker").add("@executable_path/Frameworks");
708     }
709 
710     // Call to dsymutil for debug symbol generation must happen in the link action.
711     // All debug symbol information is encoded in object files inside archive files. To generate
712     // the debug symbol bundle, dsymutil will look inside the linked binary for the encoded
713     // absolute paths to archive files, which are only valid in the link action.
714     if (dsymBundle.isPresent()) {
715       PathFragment dsymPath = FileSystemUtils.removeExtension(dsymBundle.get().getExecPath());
716       commandLine
717           .add("&&")
718           .addPath(DSYMUTIL)
719           .add(linkedBinary.getExecPathString())
720           .add("-o " + dsymPath)
721           .add("&& zipped_bundle=${PWD}/" + dsymBundle.get().getShellEscapedExecPathString())
722           .add("&& cd " + dsymPath)
723           .add("&& /usr/bin/zip -q -r \"${zipped_bundle}\" .");
724     }
725 
726     return new SingleArgCommandLine(commandLine.build());
727   }

730    * Command line that converts its input's arg array to a single input.
731    *
732    * <p>Required as a hack to the link command line because that may contain two commands, which are
733    * then passed to {@code /bin/bash -c}, and accordingly need to be a single argument.
734    */

759   /**
760    * All framework names to pass to the linker using {@code -framework} flags. For a framework in
761    * the directory foo/bar.framework, the name is "bar". Each framework is found without using the
762    * full path by means of the framework search paths. The search paths are added by
763    * {@link IosSdkCommands#commonLinkAndCompileFlagsForClang(ObjcProvider, ObjcConfiguration)}).
764    *
765    * <p>It's awful that we can't pass the full path to the framework and avoid framework search
766    * paths, but this is imposed on us by clang. clang does not support passing the full path to the
767    * framework, so Bazel cannot do it either.
768    */
769   private Iterable<String> frameworkNames(ObjcProvider provider) {
770     List<String> names = new ArrayList<>();
771     Iterables.addAll(names, SdkFramework.names(provider.get(SDK_FRAMEWORK)));
772     for (PathFragment frameworkDir : provider.get(FRAMEWORK_DIR)) {
773       String segment = frameworkDir.getBaseName();
774       Preconditions.checkState(segment.endsWith(FRAMEWORK_SUFFIX),
775           "expect %s to end with %s, but it does not", segment, FRAMEWORK_SUFFIX);
776       names.add(segment.substring(0, segment.length() - FRAMEWORK_SUFFIX.length()));
777     }
778     return names;
779   }

781   /**
782    * Registers actions that compile and archive j2Objc dependencies of this rule.
783    *
784    * @param objcProvider common information about this rule's attributes and its dependencies
785    *
786    * @return this compilation support
787    */
788   CompilationSupport registerJ2ObjcCompileAndArchiveActions(ObjcProvider objcProvider) {
789     J2ObjcSrcsProvider provider = J2ObjcSrcsProvider.buildFrom(ruleContext);
790     Iterable<J2ObjcSource> j2ObjcSources = provider.getSrcs();
791     J2ObjcConfiguration j2objcConfiguration = ruleContext.getFragment(J2ObjcConfiguration.class);
792 
793     // Only perform J2ObjC dead code stripping if flag --j2objc_dead_code_removal is specified and
794     // users have specified entry classes.
795     boolean stripJ2ObjcDeadCode = j2objcConfiguration.removeDeadCode()
796         && !provider.getEntryClasses().isEmpty();
797 
798     if (stripJ2ObjcDeadCode) {
799       registerJ2ObjcDeadCodeRemovalActions(j2ObjcSources, provider.getEntryClasses());
800     }
801 
802     for (J2ObjcSource j2ObjcSource : j2ObjcSources) {
803       if (j2ObjcSource.hasSourceFiles()) {
804         J2ObjcSource sourceToCompile =
805             j2ObjcSource.getSourceType() == SourceType.JAVA && stripJ2ObjcDeadCode
806                 ? j2ObjcSource.toPrunedSource(ruleContext)
807                 : j2ObjcSource;
808         IntermediateArtifacts intermediateArtifacts =
809             ObjcRuleClasses.j2objcIntermediateArtifacts(ruleContext, sourceToCompile);
810         CompilationArtifacts compilationArtifact = new CompilationArtifacts.Builder()
811             .addNonArcSrcs(sourceToCompile.getObjcSrcs())
812             .setIntermediateArtifacts(intermediateArtifacts)
813             .setPchFile(Optional.<Artifact>absent())
814             .build();
815         // The module map that the above intermediateArtifacts would point to is a combination of
816         // the current target and the java target. Instead, we want the one generated for that
817         // source, but there is no easy way to get that. So for now, compile the source with the
818         // module map (and therefore module name) for this rule.
819         // TODO(bazel-team): Generate module maps along with the J2Objc sources, and use that here.
820         Optional<CppModuleMap> moduleMap;
821         if (ObjcRuleClasses.objcConfiguration(ruleContext).moduleMapsEnabled()) {
822           moduleMap = Optional.of(ObjcRuleClasses.intermediateArtifacts(ruleContext).moduleMap());
823         } else {
824           moduleMap = Optional.<CppModuleMap>absent();
825         }
826         registerCompileAndArchiveActions(
827             compilationArtifact,
828             intermediateArtifacts,
829             objcProvider,
830             moduleMap,
831             ruleContext.getConfiguration().isCodeCoverageEnabled());
832       }
833     }
834 
835     return this;
836   }

856   private void registerJ2ObjcDeadCodeRemovalActions(Iterable<J2ObjcSource> j2ObjcSources,
857       Iterable<String> entryClasses) {
858     Artifact pruner = ruleContext.getPrerequisiteArtifact("$j2objc_dead_code_pruner", Mode.HOST);
859     J2ObjcMappingFileProvider provider = ObjcRuleClasses.j2ObjcMappingFileProvider(ruleContext);
860     NestedSet<Artifact> j2ObjcDependencyMappingFiles = provider.getDependencyMappingFiles();
861     NestedSet<Artifact> j2ObjcHeaderMappingFiles = provider.getHeaderMappingFiles();
862 
863     for (J2ObjcSource j2ObjcSource : j2ObjcSources) {
864       if (j2ObjcSource.getSourceType() == SourceType.JAVA
865           && j2ObjcSource.hasSourceFiles()) {
866         Iterable<Artifact> sourceArtifacts = j2ObjcSource.getObjcSrcs();
867         Iterable<Artifact> prunedSourceArtifacts =
868             j2ObjcSource.toPrunedSource(ruleContext).getObjcSrcs();
869         PathFragment paramFilePath = FileSystemUtils.replaceExtension(
870             j2ObjcSource.getTargetLabel().toPathFragment(), ".param.j2objc");
871         Artifact paramFile = ruleContext.getUniqueDirectoryArtifact(
872             "_j2objc_pruned",
873             paramFilePath,
874             ruleContext.getBinOrGenfilesDirectory());
875         PathFragment objcFilePath = j2ObjcSource.getObjcFilePath();
876         CustomCommandLine commandLine = CustomCommandLine.builder()
877             .addJoinExecPaths("--input_files", ",", sourceArtifacts)
878             .addJoinExecPaths("--output_files", ",", prunedSourceArtifacts)
879             .addJoinExecPaths("--dependency_mapping_files", ",", j2ObjcDependencyMappingFiles)
880             .addJoinExecPaths("--header_mapping_files", ",", j2ObjcHeaderMappingFiles)
881             .add("--entry_classes").add(Joiner.on(",").join(entryClasses))
882             .add("--objc_file_path").add(objcFilePath.getPathString())
883             .build();
884 
885         ruleContext.registerAction(new ParameterFileWriteAction(
886             ruleContext.getActionOwner(),
887             paramFile,
888             commandLine,
889             ParameterFile.ParameterFileType.UNQUOTED, ISO_8859_1));
890         ruleContext.registerAction(new SpawnAction.Builder()
891             .setMnemonic("DummyPruner")
892             .setExecutable(pruner)
893             .addInput(pruner)
894             .addInput(paramFile)
895             .addInputs(sourceArtifacts)
896             .addTransitiveInputs(j2ObjcDependencyMappingFiles)
897             .addTransitiveInputs(j2ObjcHeaderMappingFiles)
898             .setCommandLine(CustomCommandLine.builder()
899                 .addPaths("@%s", paramFile.getExecPath())
900                 .build())
901             .addOutputs(prunedSourceArtifacts)
902             .build(ruleContext));
903       }
904     }
905   }

913   CompilationSupport addXcodeSettings(Builder xcodeProviderBuilder, ObjcCommon common) {
914     ObjcConfiguration objcConfiguration = ObjcRuleClasses.objcConfiguration(ruleContext);
915     for (CompilationArtifacts artifacts : common.getCompilationArtifacts().asSet()) {
916       xcodeProviderBuilder.setCompilationArtifacts(artifacts);
917     }
918 
919     // The include directory options ("-I") are parsed out of copts. The include directories are
920     // added as non-propagated header search paths local to the associated Xcode target.
921     Iterable<String> copts = Iterables.concat(
922         objcConfiguration.getCopts(), attributes.copts(), attributes.optionsCopts());
923     Iterable<String> includeDirOptions = Iterables.filter(copts, INCLUDE_DIR_OPTION_IN_COPTS);
924     Iterable<String> coptsWithoutIncludeDirs = Iterables.filter(
925         copts, Predicates.not(INCLUDE_DIR_OPTION_IN_COPTS));
926     ImmutableList.Builder<PathFragment> nonPropagatedHeaderSearchPaths =
927         new ImmutableList.Builder<>();
928     for (String includeDirOption : includeDirOptions) {
929       nonPropagatedHeaderSearchPaths.add(new PathFragment(includeDirOption.substring(2)));
930     }
931 
932     xcodeProviderBuilder
933         .addHeaders(attributes.hdrs())
934         .addUserHeaderSearchPaths(ObjcCommon.userHeaderSearchPaths(ruleContext.getConfiguration()))
935         .addHeaderSearchPaths("$(WORKSPACE_ROOT)", attributes.headerSearchPaths())
936         .addHeaderSearchPaths("$(SDKROOT)/usr/include", attributes.sdkIncludes())
937         .addNonPropagatedHeaderSearchPaths(
938             "$(WORKSPACE_ROOT)", nonPropagatedHeaderSearchPaths.build())
939         .addCompilationModeCopts(objcConfiguration.getCoptsForCompilationMode())
940         .addCopts(coptsWithoutIncludeDirs);
941 
942     return this;

971   private CompilationSupport registerDsymActions() {
972     IntermediateArtifacts intermediateArtifacts =
973         ObjcRuleClasses.intermediateArtifacts(ruleContext);
974     ObjcConfiguration objcConfiguration = ObjcRuleClasses.objcConfiguration(ruleContext);
975 
976     Artifact dsymBundle = intermediateArtifacts.dsymBundle();
977     Artifact linkedBinary =
978         objcConfiguration.shouldStripBinary()
979             ? intermediateArtifacts.unstrippedSingleArchitectureBinary()
980             : intermediateArtifacts.strippedSingleArchitectureBinary();
981     Artifact debugSymbolFile = intermediateArtifacts.dsymSymbol();
982     Artifact dsymPlist = intermediateArtifacts.dsymPlist();
983 
984     PathFragment dsymOutputDir =
985         replaceSuffix(
986             dsymBundle.getExecPath(), IntermediateArtifacts.TMP_DSYM_BUNDLE_SUFFIX, ".app.dSYM");
987     PathFragment dsymPlistZipEntry = dsymPlist.getExecPath().relativeTo(dsymOutputDir);
988     PathFragment debugSymbolFileZipEntry =
989         debugSymbolFile
990             .getExecPath()
991             .replaceName(linkedBinary.getFilename())
992             .relativeTo(dsymOutputDir);
993 
994     StringBuilder unzipDsymCommand = new StringBuilder();
995     unzipDsymCommand
996         .append(
997             String.format(
998                 "unzip -p %s %s > %s",
999                 dsymBundle.getExecPathString(),
1000                 dsymPlistZipEntry,
1001                 dsymPlist.getExecPathString()))
1002         .append(
1003             String.format(
1004                 " && unzip -p %s %s > %s",
1005                 dsymBundle.getExecPathString(),
1006                 debugSymbolFileZipEntry,
1007                 debugSymbolFile.getExecPathString()));
1008 
1009     ruleContext.registerAction(
1010         new SpawnAction.Builder()
1011             .setMnemonic("UnzipDsym")
1012             .setShellCommand(unzipDsymCommand.toString())
1013             .addInput(dsymBundle)
1014             .addOutput(dsymPlist)
1015             .addOutput(debugSymbolFile)
1016             .build(ruleContext));
1017 
1018     Artifact dumpsyms = ruleContext.getPrerequisiteArtifact(":dumpsyms", Mode.HOST);
1019     Artifact breakpadFile = intermediateArtifacts.breakpadSym();
1020     ruleContext.registerAction(ObjcRuleClasses.spawnOnDarwinActionBuilder(ruleContext)
1021         .setMnemonic("GenBreakpad")
1022         .setProgressMessage("Generating breakpad file: " + ruleContext.getLabel())
1023         .setShellCommand(ImmutableList.of("/bin/bash", "-c"))
1024         .addInput(dumpsyms)
1025         .addInput(debugSymbolFile)
1026         .addArgument(String.format("%s %s > %s",
1027             ShellUtils.shellEscape(dumpsyms.getExecPathString()),
1028             ShellUtils.shellEscape(debugSymbolFile.getExecPathString()),
1029             ShellUtils.shellEscape(breakpadFile.getExecPathString())))
1030         .addOutput(breakpadFile)
1031         .build(ruleContext));
1032     return this;
1033   }
