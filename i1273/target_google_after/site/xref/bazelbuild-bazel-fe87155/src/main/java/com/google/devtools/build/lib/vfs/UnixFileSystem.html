41    /**
42     * Eager implementation of FileStatus for file systems that have an atomic
43     * stat(2) syscall. A proxy for {@link com.google.devtools.build.lib.unix.FileStatus}.
44     * Note that isFile and getLastModifiedTime have slightly different meanings
45     * between UNIX and VFS.
46     */
47    @VisibleForTesting
48    protected static class UnixFileStatus implements FileStatus {
49  
50      private final com.google.devtools.build.lib.unix.FileStatus status;
51  
52      UnixFileStatus(com.google.devtools.build.lib.unix.FileStatus status) {
53        this.status = status;
54      }
55  
56      @Override
57      public boolean isFile() { return !isDirectory() && !isSymbolicLink(); }
58  
59      @Override
60      public boolean isDirectory() { return status.isDirectory(); }
61  
62      @Override
63      public boolean isSymbolicLink() { return status.isSymbolicLink(); }
64  
65      @Override
66      public long getSize() { return status.getSize(); }
67  
68      @Override
69      public long getLastModifiedTime() {
70        return (status.getLastModifiedTime() * 1000)
71            + (status.getFractionalLastModifiedTime() / 1000000);
72      }
73  
74      @Override
75      public long getLastChangeTime() {
76        return (status.getLastChangeTime() * 1000)
77            + (status.getFractionalLastChangeTime() / 1000000);
78      }
79  
80      @Override
81      public long getNodeId() {
82        // Note that we may want to include more information in this id number going forward,
83        // especially the device number.
84        return status.getInodeNumber();
85      }
86  
87      int getPermissions() { return status.getPermissions(); }
88  
89      @Override
90      public String toString() { return status.toString(); }
91    }

93    @Override
94    protected Collection<Path> getDirectoryEntries(Path path) throws IOException {
95      String name = path.getPathString();
96      String[] entries;
97      long startTime = Profiler.nanoTimeMaybe();
98      try {
99        entries = FilesystemUtils.readdir(name);
100     } finally {
101       profiler.logSimpleTask(startTime, ProfilerTask.VFS_DIR, name);
102     }
103     Collection<Path> result = new ArrayList<>(entries.length);
104     for (String entry : entries) {
105       result.add(path.getChild(entry));
106     }
107     return result;
108   }

110   @Override
111   protected PathFragment resolveOneLink(Path path) throws IOException {
112     // Beware, this seemingly simple code belies the complex specification of
113     // FileSystem.resolveOneLink().
114     return stat(path, false).isSymbolicLink()
115         ? readSymbolicLink(path)
116         : null;
117   }

119   /**
120    * Converts from {@link com.google.devtools.build.lib.unix.FilesystemUtils.Dirents.Type} to
121    * {@link com.google.devtools.build.lib.vfs.Dirent.Type}.
122    */
123   private static Dirent.Type convertToDirentType(Dirents.Type type) {
124     switch (type) {
125       case FILE:
126         return Dirent.Type.FILE;
127       case DIRECTORY:
128         return Dirent.Type.DIRECTORY;
129       case SYMLINK:
130         return Dirent.Type.SYMLINK;
131       case UNKNOWN:
132         return Dirent.Type.UNKNOWN;
133       default:
134         throw new IllegalArgumentException("Unknown type " + type);
135     }
136   }

138   @Override
139   protected Collection<Dirent> readdir(Path path, boolean followSymlinks) throws IOException {
140     String name = path.getPathString();
141     long startTime = Profiler.nanoTimeMaybe();
142     try {
143       Dirents unixDirents = FilesystemUtils.readdir(name,
144           followSymlinks ? ReadTypes.FOLLOW : ReadTypes.NOFOLLOW);
145       Preconditions.checkState(unixDirents.hasTypes());
146       List<Dirent> dirents = Lists.newArrayListWithCapacity(unixDirents.size());
147       for (int i = 0; i < unixDirents.size(); i++) {
148         dirents.add(new Dirent(unixDirents.getName(i),
149             convertToDirentType(unixDirents.getType(i))));
150       }
151       return dirents;
152     } finally {
153       profiler.logSimpleTask(startTime, ProfilerTask.VFS_DIR, name);
154     }
155   }

162   @VisibleForTesting
163   protected UnixFileStatus statInternal(Path path, boolean followSymlinks) throws IOException {
164     String name = path.getPathString();
165     long startTime = Profiler.nanoTimeMaybe();
166     try {
167       return new UnixFileStatus(followSymlinks
168                                       ? FilesystemUtils.stat(name)
169                                       : FilesystemUtils.lstat(name));
170     } finally {
171       profiler.logSimpleTask(startTime, ProfilerTask.VFS_STAT, name);
172     }
173   }
174 
175   // Like stat(), but returns null instead of throwing.
176   // This is a performance optimization in the case where clients
177   // catch and don't re-throw.
178   @Override
179   protected FileStatus statNullable(Path path, boolean followSymlinks) {
180     String name = path.getPathString();
181     long startTime = Profiler.nanoTimeMaybe();
182     try {
183       ErrnoFileStatus stat = followSymlinks
184           ? FilesystemUtils.errnoStat(name)
185           : FilesystemUtils.errnoLstat(name);
186       return stat.hasError() ? null : new UnixFileStatus(stat);
187     } finally {
188       profiler.logSimpleTask(startTime, ProfilerTask.VFS_STAT, name);
189     }
190   }

192   @Override
193   protected boolean exists(Path path, boolean followSymlinks) {
194     return statNullable(path, followSymlinks) != null;
195   }

197   /**
198    * Return true iff the {@code stat} of {@code path} resulted in an {@code ENOENT}
199    * or {@code ENOTDIR} error.
200    */
201   @Override
202   protected FileStatus statIfFound(Path path, boolean followSymlinks) throws IOException {
203     String name = path.getPathString();
204     long startTime = Profiler.nanoTimeMaybe();
205     try {
206       ErrnoFileStatus stat = followSymlinks
207           ? FilesystemUtils.errnoStat(name)
208           : FilesystemUtils.errnoLstat(name);
209       if (!stat.hasError()) {
210         return new UnixFileStatus(stat);
211       }
212       int errno = stat.getErrno();
213       if (errno == ErrnoFileStatus.ENOENT || errno == ErrnoFileStatus.ENOTDIR) {
214         return null;
215       }
216       // This should not return -- we are calling stat here just to throw the proper exception.
217       // However, since there may be transient IO errors, we cannot guarantee that an exception will
218       // be thrown.
219       // TODO(bazel-team): Extract the exception-construction code and make it visible separately in
220       // FilesystemUtils to avoid having to do a duplicate stat call.
221       return stat(path, followSymlinks);
222     } finally {
223       profiler.logSimpleTask(startTime, ProfilerTask.VFS_STAT, name);
224     }
225   }

227   @Override
228   protected boolean isDirectory(Path path, boolean followSymlinks) {
229     FileStatus stat = statNullable(path, followSymlinks);
230     return stat != null && stat.isDirectory();
231   }

233   @Override
234   protected boolean isFile(Path path, boolean followSymlinks) {
235     // Note, FileStatus.isFile means *regular* file whereas Path.isFile may
236     // mean special file too, so we don't return FileStatus.isFile here.
237     FileStatus status = statNullable(path, followSymlinks);
238     return status != null && !(status.isSymbolicLink() || status.isDirectory());
239   }

256   /**
257    * Adds or remove the bits specified in "permissionBits" to the permission
258    * mask of the file specified by {@code path}. If the argument {@code add} is
259    * true, the specified permissions are added, otherwise they are removed.
260    *
261    * @throws IOException if there was an error writing the file's metadata
262    */
263   private void modifyPermissionBits(Path path, int permissionBits, boolean add)
264     throws IOException {
265     synchronized (path) {
266       int oldMode = statInternal(path, true).getPermissions();
267       int newMode = add ? (oldMode | permissionBits) : (oldMode & ~permissionBits);
268       FilesystemUtils.chmod(path.toString(), newMode);
269     }
270   }

274     modifyPermissionBits(path, 0400, readable);

304   @Override
305   protected boolean createDirectory(Path path) throws IOException {
306     synchronized (path) {
307       // Note: UNIX mkdir(2), FilesystemUtils.mkdir() and createDirectory all
308       // have different ways of representing failure!
309       if (FilesystemUtils.mkdir(path.toString(), 0777)) {
310         return true; // successfully created
311       }
312 
313       // false => EEXIST: something is already in the way (file/dir/symlink)
314       if (isDirectory(path, false)) {
315         return false; // directory already existed
316       } else {
317         throw new IOException(path + " (File exists)");
318       }
319     }
320   }

330   @Override
331   protected PathFragment readSymbolicLink(Path path) throws IOException {
332     String name = path.toString();
333     long startTime = Profiler.nanoTimeMaybe();
334     try {
335       return new PathFragment(FilesystemUtils.readlink(name));
336     } catch (IOException e) {
337       // EINVAL => not a symbolic link.  Anything else is a real error.
338       throw e.getMessage().endsWith("(Invalid argument)") ? new NotASymlinkException(path) : e;
339     } finally {
340       profiler.logSimpleTask(startTime, ProfilerTask.VFS_READLINK, name);
341     }
342   }

380   @Override
381   protected void setLastModifiedTime(Path path, long newTime) throws IOException {
382     synchronized (path) {
383       if (newTime == -1L) { // "now"
384         FilesystemUtils.utime(path.toString(), true, 0, 0);
385       } else {
386         // newTime > MAX_INT => -ve unixTime
387         int unixTime = (int) (newTime / 1000);
388         FilesystemUtils.utime(path.toString(), false, unixTime, unixTime);
389       }
390     }
391   }

393   @Override
394   protected byte[] getxattr(Path path, String name, boolean followSymlinks) throws IOException {
395     String pathName = path.toString();
396     long startTime = Profiler.nanoTimeMaybe();
397     try {
398       return followSymlinks
399           ? FilesystemUtils.getxattr(pathName, name) : FilesystemUtils.lgetxattr(pathName, name);
400     } catch (UnsupportedOperationException e) {
401       // getxattr() syscall is not supported by the underlying filesystem (it returned ENOTSUP).
402       // Per method contract, treat this as ENODATA.
403       return null;
404     } finally {
405       profiler.logSimpleTask(startTime, ProfilerTask.VFS_XATTR, pathName);
406     }
407   }
