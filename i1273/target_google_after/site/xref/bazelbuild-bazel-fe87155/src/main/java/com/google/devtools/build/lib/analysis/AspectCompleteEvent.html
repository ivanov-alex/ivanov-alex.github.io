14  package com.google.devtools.build.lib.analysis;
15  
16  import com.google.common.base.Preconditions;
17  import com.google.common.collect.Iterables;
18  import com.google.devtools.build.lib.collect.nestedset.NestedSet;
19  import com.google.devtools.build.lib.collect.nestedset.NestedSetBuilder;
20  import com.google.devtools.build.lib.collect.nestedset.Order;
21  import com.google.devtools.build.lib.skyframe.AspectValue;
22  import com.google.devtools.build.lib.syntax.Label;
23  import com.google.devtools.build.skyframe.SkyValue;
24  
25  /**
26   * This event is fired as soon as a top-level aspect is either built or fails.
27   */
28  public class AspectCompleteEvent implements SkyValue {
29    private final AspectValue aspectValue;
30    private final NestedSet<Label> rootCauses;
31  
32    private AspectCompleteEvent(AspectValue aspectValue, NestedSet<Label> rootCauses) {
33      this.aspectValue = aspectValue;
34      this.rootCauses =
35          (rootCauses == null) ? NestedSetBuilder.<Label>emptySet(Order.STABLE_ORDER) : rootCauses;
36    }
37  
38    /**
39     * Construct a successful target completion event.
40     */
41    public static AspectCompleteEvent createSuccessful(AspectValue value) {
42      return new AspectCompleteEvent(value, null);
43    }
44  
45    /**
46     * Construct a target completion event for a failed target, with the given non-empty root causes.
47     */
48    public static AspectCompleteEvent createFailed(AspectValue value, NestedSet<Label> rootCauses) {
49      Preconditions.checkArgument(!Iterables.isEmpty(rootCauses));
50      return new AspectCompleteEvent(value, rootCauses);
51    }
52  
53    /**
54     * Returns the target associated with the event.
55     */
56    public AspectValue getAspectValue() {
57      return aspectValue;
58    }
59  
60    /**
61     * Determines whether the target has failed or succeeded.
62     */
63    public boolean failed() {
64      return !rootCauses.isEmpty();
65    }
66  
67    /**
68     * Get the root causes of the target. May be empty.
69     */
70    public Iterable<Label> getRootCauses() {
71      return rootCauses;
72    }
73  }
