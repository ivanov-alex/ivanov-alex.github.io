41  /**
42   * Abstract implementation of Action which implements basic functionality: the
43   * inputs, outputs, and toString method.  Both input and output sets are
44   * immutable.
45   */
46  @Immutable @ThreadSafe
47  public abstract class AbstractAction implements Action {
48  
49    /**
50     * An arbitrary default resource set. Currently 250MB of memory, 50% CPU and 0% of total I/O.
51     */
52    public static final ResourceSet DEFAULT_RESOURCE_SET =
53        ResourceSet.createWithRamCpuIo(250, 0.5, 0);

55    // owner/inputs/outputs attributes below should never be directly accessed even
56    // within AbstractAction itself. The appropriate getter methods should be used
57    // instead. This has to be done due to the fact that the getter methods can be
58    // overridden in subclasses.
59    private final ActionOwner owner;
60    // The variable inputs is non-final only so that actions that discover their inputs can modify it.
61    private Iterable<Artifact> inputs;
62    private final RunfilesSupplier runfilesSupplier;
63    private final ImmutableSet<Artifact> outputs;
64  
65    private String cachedKey;
66  
67    /**
68     * Construct an abstract action with the specified inputs and outputs;
69     */
70    protected AbstractAction(ActionOwner owner,
71                             Iterable<Artifact> inputs,
72                             Iterable<Artifact> outputs) {
73      this(owner, inputs, EmptyRunfilesSupplier.INSTANCE, outputs);
74    }
75  
76    protected AbstractAction(ActionOwner owner,
77        Iterable<Artifact> inputs,
78        RunfilesSupplier runfilesSupplier,
79        Iterable<Artifact> outputs) {
80      Preconditions.checkNotNull(owner);
81      // TODO(bazel-team): Use RuleContext.actionOwner here instead
82      this.owner = new ActionOwnerDescription(owner);
83      this.inputs = CollectionUtils.makeImmutable(inputs);
84      this.outputs = ImmutableSet.copyOf(outputs);
85      this.runfilesSupplier = Preconditions.checkNotNull(runfilesSupplier,
86          "runfilesSupplier may not be null");
87      Preconditions.checkArgument(!this.outputs.isEmpty(), "action outputs may not be empty");
88    }
89

92      return owner;
93    }

97      return true;
98    }

216     String message = getRawProgressMessage();
217     if (message == null) {
218       return null;

221     return additionalInfo == null ? message : message + " [" + additionalInfo + "]";

224   /**
225    * Returns a progress message string that is specific for this action. This is
226    * then annotated with additional information, currently the string '[for host]'
227    * for actions in the host configurations.
228    *
229    * <p>A return value of null indicates no message should be reported.
230    */
231   protected String getRawProgressMessage() {
232     // A cheesy default implementation.  Subclasses are invited to do something
233     // more meaningful.
234     return defaultProgressMessage();
235   }

237   private String defaultProgressMessage() {
238     return getMnemonic() + " " + getPrimaryOutput().prettyPrint();
239   }

241   @Override

243     return "action '" + describe() + "'";
244   }

246   /**
247    * Deletes all of the action's output files, if they exist. If any of the
248    * Artifacts refers to a directory recursively removes the contents of the
249    * directory.
250    *
251    * @param execRoot the exec root in which this action is executed
252    */
253   protected void deleteOutputs(Path execRoot) throws IOException {
254     for (Artifact output : getOutputs()) {
255       deleteOutput(output);
256     }
257   }
258 
259   /**
260    * Helper method to remove an Artifact. If the Artifact refers to a directory
261    * recursively removes the contents of the directory.
262    */
263   protected void deleteOutput(Artifact output) throws IOException {
264     Path path = output.getPath();
265     try {
266       // Optimize for the common case: output artifacts are files.
267       path.delete();
268     } catch (IOException e) {
269       // Only try to recursively delete a directory if the output root is known. This is just a
270       // sanity check so that we do not start deleting random files on disk.
271       // TODO(bazel-team): Strengthen this test by making sure that the output is part of the
272       // output tree.
273       if (path.isDirectory(Symlinks.NOFOLLOW) && output.getRoot() != null) {
274         FileSystemUtils.deleteTree(path);
275       } else {
276         throw e;
277       }
278     }
279   }

281   /**
282    * If the action might read directories as inputs in a way that is unsound wrt dependency
283    * checking, this method must be called.
284    */
285   protected void checkInputsForDirectories(EventHandler eventHandler,
286                                            MetadataHandler metadataHandler) {
287     // Report "directory dependency checking" warning only for non-generated directories (generated
288     // ones will be reported earlier).
289     for (Artifact input : getMandatoryInputs()) {
290       // Assume that if the file did not exist, we would not have gotten here.
291       if (input.isSourceArtifact() && !metadataHandler.isRegularFile(input)) {
292         eventHandler.handle(Event.warn(getOwner().getLocation(), "input '"
293             + input.prettyPrint() + "' to " + getOwner().getLabel()
294             + " is a directory; dependency checking of directories is unsound"));
295       }
296     }
297   }

299   @Override
300   public MiddlemanType getActionType() {
301     return MiddlemanType.NORMAL;
302   }
303 
304   /**
305    * If the action might create directories as outputs this method must be called.
306    */
307   protected void checkOutputsForDirectories(EventHandler eventHandler) {
308     for (Artifact output : getOutputs()) {
309       Path path = output.getPath();
310       String ownerString = Label.print(getOwner().getLabel());
311       if (path.isDirectory()) {
312         eventHandler.handle(new Event(EventKind.WARNING, getOwner().getLocation(),
313             "output '" + output.prettyPrint() + "' of " + ownerString
314                   + " is a directory; dependency checking of directories is unsound",
315                   ownerString));
316       }
317     }
318   }

352   /**
353    * Returns input files that need to be present to allow extra_action rules to shadow this action
354    * correctly when run remotely. This is at least the normal inputs of the action, but may include
355    * other files as well. For example C(++) compilation may perform include file header scanning.
356    * This needs to be mirrored by the extra_action rule. Called by
357    * {@link com.google.devtools.build.lib.rules.extra.ExtraAction} at execution time.
358    *
359    * <p>As this method is called from the ExtraAction, make sure it is ok to call
360    * this method from a different thread than the one this action is executed on.
361    *
362    * @param actionExecutionContext Services in the scope of the action, like the Out/Err streams.
363    * @throws ActionExecutionException only when code called from this method
364    *     throws that exception.
365    * @throws InterruptedException if interrupted
366    */
367   public Iterable<Artifact> getInputFilesForExtraAction(
368       ActionExecutionContext actionExecutionContext)
369       throws ActionExecutionException, InterruptedException {
370     return getInputs();
371   }

373   /**
374    * A copying implementation of {@link ActionOwner}.
375    *
376    * <p>ConfiguredTargets implement ActionOwner themselves, but we do not want actions
377    * to keep direct references to configured targets just for a label and a few strings.
378    */
379   @Immutable
380   private static class ActionOwnerDescription implements ActionOwner {
381 
382     private final Location location;
383     private final Label label;
384     private final String configurationMnemonic;
385     private final String configurationChecksum;
386     private final String targetKind;
387     private final String additionalProgressInfo;
388 
389     private ActionOwnerDescription(ActionOwner originalOwner) {
390       this.location = originalOwner.getLocation();
391       this.label = originalOwner.getLabel();
392       this.configurationMnemonic = originalOwner.getConfigurationMnemonic();
393       this.configurationChecksum = originalOwner.getConfigurationChecksum();
394       this.targetKind = originalOwner.getTargetKind();
395       this.additionalProgressInfo = originalOwner.getAdditionalProgressInfo();
396     }
397 
398     @Override
399     public Location getLocation() {
400       return location;
401     }
402 
403     @Override
404     public Label getLabel() {
405       return label;
406     }
407 
408     @Override
409     public String getConfigurationMnemonic() {
410       return configurationMnemonic;
411     }
412 
413     @Override
414     public String getConfigurationChecksum() {
415       return configurationChecksum;
416     }
417 
418     @Override
419     public String getTargetKind() {
420       return targetKind;
421     }
422 
423     @Override
424     public String getAdditionalProgressInfo() {
425       return additionalProgressInfo;
426     }
427   }
