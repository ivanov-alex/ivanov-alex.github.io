60  /**
61   * Provides information about the workspace (e.g. source control context, current machine, current
62   * user, etc).
63   *
64   * <p>Note that the <code>equals()</code> method is necessary so that Skyframe knows when to
65   * invalidate the node representing the workspace status action.
66   */
67  public class BazelWorkspaceStatusModule extends BlazeModule {
68    private static class BazelWorkspaceStatusAction extends WorkspaceStatusAction {
69      private final Artifact stableStatus;
70      private final Artifact volatileStatus;
71      private final Options options;
72      private final String username;
73      private final String hostname;
74      private final long timestamp;
75      private final com.google.devtools.build.lib.shell.Command getWorkspaceStatusCommand;
76  
77      private BazelWorkspaceStatusAction(
78          WorkspaceStatusAction.Options options,
79          BlazeRuntime runtime,
80          Artifact stableStatus,
81          Artifact volatileStatus) {
82        super(BuildInfoHelper.BUILD_INFO_ACTION_OWNER, Artifact.NO_ARTIFACTS,
83            ImmutableList.of(stableStatus, volatileStatus));
84        this.options = Preconditions.checkNotNull(options);
85        this.stableStatus = stableStatus;
86        this.volatileStatus = volatileStatus;
87        this.username = USER_NAME.value();
88        this.hostname = NetUtil.findShortHostName();
89        this.timestamp = System.currentTimeMillis();
90        this.getWorkspaceStatusCommand =
91            options.workspaceStatusCommand.equals(PathFragment.EMPTY_FRAGMENT)
92                ? null
93                : new CommandBuilder()
94                    .addArgs(options.workspaceStatusCommand.toString())
95                    // Pass client env, because certain SCM client(like
96                    // perforce, git) relies on environment variables to work
97                    // correctly.
98                    .setEnv(runtime.getClientEnv())
99                    .setWorkingDir(runtime.getWorkspace())
100                   .useShell(true)
101                   .build();
102     }
103 
104     @Override
105     public String describeStrategy(Executor executor) {
106       return "";
107     }
108 
109     private String getAdditionalWorkspaceStatus(ActionExecutionContext actionExecutionContext)
110         throws ActionExecutionException {
111       try {
112         if (this.getWorkspaceStatusCommand != null) {
113           actionExecutionContext
114               .getExecutor()
115               .getEventHandler()
116               .handle(
117                   Event.progress(
118                       "Getting additional workspace status by running "
119                           + options.workspaceStatusCommand));
120           CommandResult result = this.getWorkspaceStatusCommand.execute();
121           if (result.getTerminationStatus().success()) {
122             return new String(result.getStdout());
123           }
124           throw new ActionExecutionException(
125               "workspace status command failed: " + result.getTerminationStatus(), this, true);
126         }
127       } catch (CommandException e) {
128         throw new ActionExecutionException(e, this, true);
129       }
130       return "";
131     }
132 
133     @Override
134     public void execute(ActionExecutionContext actionExecutionContext)
135         throws ActionExecutionException {
136       try {
137         Joiner joiner = Joiner.on('\n');
138         String info =
139             joiner.join(
140                 BuildInfo.BUILD_EMBED_LABEL + " " + options.embedLabel,
141                 BuildInfo.BUILD_HOST + " " + hostname,
142                 BuildInfo.BUILD_USER + " " + username);
143         FileSystemUtils.writeContent(stableStatus.getPath(), info.getBytes(StandardCharsets.UTF_8));
144         String volatileInfo =
145             joiner.join(
146                 BuildInfo.BUILD_TIMESTAMP + " " + timestamp,
147                 getAdditionalWorkspaceStatus(actionExecutionContext));
148 
149         FileSystemUtils.writeContent(
150             volatileStatus.getPath(), volatileInfo.getBytes(StandardCharsets.UTF_8));
151       } catch (IOException e) {
152         throw new ActionExecutionException(
153             "Failed to run workspace status command " + options.workspaceStatusCommand,
154             e,
155             this,
156             true);
157       }
158     }
159 
160     @Override
161     public boolean equals(Object o) {
162       if (!(o instanceof BazelWorkspaceStatusAction)) {
163         return false;
164       }
165 
166       BazelWorkspaceStatusAction that = (BazelWorkspaceStatusAction) o;
167       return this.stableStatus.equals(that.stableStatus)
168           && this.volatileStatus.equals(that.volatileStatus)
169           && this.options.equals(that.options);
170     }
171

173     public int hashCode() {
174       return Objects.hash(stableStatus, volatileStatus, options);
175     }

213   private class BazelStatusActionFactory implements WorkspaceStatusAction.Factory {
214     @Override
215     public Map<String, String> createDummyWorkspaceStatus() {
216       return ImmutableMap.of();
217     }
218 
219     @Override
220     public WorkspaceStatusAction createWorkspaceStatusAction(
221         ArtifactFactory factory, ArtifactOwner artifactOwner, Supplier<UUID> buildId) {
222       Root root = runtime.getDirectories().getBuildDataDirectory();
223 
224       Artifact stableArtifact = factory.getDerivedArtifact(
225           new PathFragment("stable-status.txt"), root, artifactOwner);
226       Artifact volatileArtifact = factory.getConstantMetadataArtifact(
227           new PathFragment("volatile-status.txt"), root, artifactOwner);
228 
229       return new BazelWorkspaceStatusAction(options, runtime, stableArtifact, volatileArtifact);
230     }
231   }
232 
233   @ExecutionStrategy(contextType = WorkspaceStatusAction.Context.class)
234   private class BazelWorkspaceStatusActionContext implements WorkspaceStatusAction.Context {
235     @Override
236     public ImmutableMap<String, Key> getStableKeys() {
237       return ImmutableMap.of(
238           BuildInfo.BUILD_EMBED_LABEL,
239           Key.of(KeyType.STRING, options.embedLabel, "redacted"),
240           BuildInfo.BUILD_HOST,
241           Key.of(KeyType.STRING, "hostname", "redacted"),
242           BuildInfo.BUILD_USER,
243           Key.of(KeyType.STRING, "username", "redacted"));
244     }
245 
246     @Override
247     public ImmutableMap<String, Key> getVolatileKeys() {
248       return ImmutableMap.of(
249           BuildInfo.BUILD_TIMESTAMP,
250           Key.of(KeyType.INTEGER, "0", "0"),
251           BuildInfo.BUILD_SCM_REVISION,
252           Key.of(KeyType.STRING, "0", "0"),
253           BuildInfo.BUILD_SCM_STATUS,
254           Key.of(KeyType.STRING, "", "redacted"));
255     }
256   }

277   }
