63  /**
64   * Instances of RuleClass encapsulate the set of attributes of a given "class" of rule, such as
65   * <code>cc_binary</code>.
66   *
67   * <p>This is an instance of the "meta-class" pattern for Rules: we achieve using <i>values</i>
68   * what subclasses achieve using <i>types</i>.  (The "Design Patterns" book doesn't include this
69   * pattern, so think of it as something like a cross between a Flyweight and a State pattern. Like
70   * Flyweight, we avoid repeatedly storing data that belongs to many instances. Like State, we
71   * delegate from Rule to RuleClass for the specific behavior of that rule (though unlike state, a
72   * Rule object never changes its RuleClass).  This avoids the need to declare one Java class per
73   * class of Rule, yet achieves the same behavior.)
74   *
75   * <p>The use of a metaclass also allows us to compute a mapping from Attributes to small integers
76   * and share this between all rules of the same metaclass.  This means we can save the attribute
77   * dictionary for each rule instance using an array, which is much more compact than a hashtable.
78   *
79   * <p>Rule classes whose names start with "$" are considered "abstract"; since they are not valid
80   * identifiers, they cannot be named in the build language. However, they are useful for grouping
81   * related attributes which are inherited.
82   *
83   * <p>The exact values in this class are important.  In particular:
84   * <ul>
85   * <li>Changing an attribute from MANDATORY to OPTIONAL creates the potential for null-pointer
86   *     exceptions in code that expects a value.
87   * <li>Attributes whose names are preceded by a "$" or a ":" are "hidden", and cannot be redefined
88   *     in a BUILD file.  They are a useful way of adding a special dependency. By convention,
89   *     attributes starting with "$" are implicit dependencies, and those starting with a ":" are
90   *     late-bound implicit dependencies, i.e. dependencies that can only be resolved when the
91   *     configuration is known.
92   * <li>Attributes should not be introduced into the hierarchy higher then necessary.
93   * <li>The 'deps' and 'data' attributes are treated specially by the code that builds the runfiles
94   *     tree.  All targets appearing in these attributes appears beneath the ".runfiles" tree; in
95   *     addition, "deps" may have rule-specific semantics.
96   * </ul>
97   */
98  @Immutable
99  public final class RuleClass {

102   /**
103    * A constraint for the package name of the Rule instances.
104    */
105   public static class PackageNameConstraint implements PredicateWithMessage<Rule> {
106 
107     public static final int ANY_SEGMENT = 0;
108 
109     private final int pathSegment;
110 
111     private final Set<String> values;
112 
113     /**
114      * The pathSegment-th segment of the package must be one of the specified values.
115      * The path segment indexing starts from 1.
116      */
117     public PackageNameConstraint(int pathSegment, String... values) {
118       this.values = ImmutableSet.copyOf(values);
119       this.pathSegment = pathSegment;
120     }
121 
122     @Override
123     public boolean apply(Rule input) {
124       PathFragment path = input.getLabel().getPackageFragment();
125       if (pathSegment == ANY_SEGMENT) {
126         return path.getFirstSegment(values) != PathFragment.INVALID_SEGMENT;
127       } else {
128         return path.segmentCount() >= pathSegment
129             && values.contains(path.getSegment(pathSegment - 1));
130       }
131     }
132 
133     @Override
134     public String getErrorReason(Rule param) {
135       if (pathSegment == ANY_SEGMENT) {
136         return param.getRuleClass() + " rules have to be under a "
137             + StringUtil.joinEnglishList(values, "or", "'") + " directory";
138       } else if (pathSegment == 1) {
139         return param.getRuleClass() + " rules are only allowed in "
140             + StringUtil.joinEnglishList(StringUtil.append(values, "//", ""), "or");
141       } else {
142           return param.getRuleClass() + " rules are only allowed in packages which "
143               + StringUtil.ordinal(pathSegment) + " is " + StringUtil.joinEnglishList(values, "or");
144       }
145     }
146 
147     @VisibleForTesting
148     public int getPathSegment() {
149       return pathSegment;
150     }
151 
152     @VisibleForTesting
153     public Collection<String> getValues() {
154       return values;
155     }
156   }

513     /**
514      * Constructs a new {@code RuleClassBuilder} using all attributes from all
515      * parent rule classes. An attribute cannot exist in more than one parent.
516      *
517      * <p>The rule type affects the the allowed names and the required
518      * attributes (see {@link RuleClassType}).
519      *
520      * @throws IllegalArgumentException if an attribute with the same name exists
521      * in more than one parent
522      */
523     public Builder(String name, RuleClassType type, boolean skylark, RuleClass... parents) {
524       this.name = name;
525       this.skylark = skylark;
526       this.type = type;
527       this.documented = type != RuleClassType.ABSTRACT;
528       for (RuleClass parent : parents) {
529         if (parent.getValidityPredicate() != PredicatesWithMessage.<Rule>alwaysTrue()) {
530           setValidityPredicate(parent.getValidityPredicate());
531         }
532         if (parent.preferredDependencyPredicate != Predicates.<String>alwaysFalse()) {
533           setPreferredDependencyPredicate(parent.preferredDependencyPredicate);
534         }
535         configurationFragments.addAll(parent.requiredConfigurationFragments);
536         missingFragmentPolicy = parent.missingFragmentPolicy;
537         supportsConstraintChecking = parent.supportsConstraintChecking;
538 
539         for (Attribute attribute : parent.getAttributes()) {
540           String attrName = attribute.getName();
541           Preconditions.checkArgument(
542               !attributes.containsKey(attrName) || attributes.get(attrName) == attribute,
543               "Attribute %s is inherited multiple times in %s ruleclass",
544               attrName,
545               name);
546           attributes.put(attrName, attribute);
547         }
548 
549         advertisedProviders.addAll(parent.getAdvertisedProviders());
550       }
551       // TODO(bazel-team): move this testonly attribute setting to somewhere else
552       // preferably to some base RuleClass implementation.
553       if (this.type.equals(RuleClassType.TEST)) {
554         Attribute.Builder<Boolean> testOnlyAttr = attr("testonly", BOOLEAN).value(true)
555             .nonconfigurable("policy decision: this shouldn't depend on the configuration");
556         if (attributes.containsKey("testonly")) {
557           override(testOnlyAttr);
558         } else {
559           add(testOnlyAttr);
560         }
561       }
562     }

574     /**

576      */
577     public RuleClass build(String name) {
578       Preconditions.checkArgument(this.name.isEmpty() || this.name.equals(name));
579       type.checkName(name);
580       type.checkAttributes(attributes);
581       boolean skylarkExecutable =
582           skylark && (type == RuleClassType.NORMAL || type == RuleClassType.TEST);
583       Preconditions.checkState(
584           (type == RuleClassType.ABSTRACT)
585           == (configuredTargetFactory == null && configuredTargetFunction == null));
586       Preconditions.checkState(skylarkExecutable == (configuredTargetFunction != null));
587       Preconditions.checkState(skylarkExecutable == (ruleDefinitionEnvironment != null));
588       Preconditions.checkState(workspaceOnly || externalBindingsFunction == NO_EXTERNAL_BINDINGS);
589 
590       return new RuleClass(name, skylarkExecutable, documented, publicByDefault, binaryOutput,
591           workspaceOnly, outputsDefaultExecutable, implicitOutputsFunction, configurator,
592           configuredTargetFactory, validityPredicate, preferredDependencyPredicate,
593           ImmutableSet.copyOf(advertisedProviders), configuredTargetFunction,
594           externalBindingsFunction, ruleDefinitionEnvironment, configurationFragments,
595           ImmutableMap.copyOf(requiredFragmentNames), fragmentNameResolver, missingFragmentPolicy,
596           supportsConstraintChecking, attributes.values().toArray(new Attribute[0]));

1359    * Creates a rule with the attribute values that are already parsed.
1360    *
1361    * <p><b>WARNING:</b> This assumes that the attribute values here have the right type and
1362    * bypasses some sanity checks. If they are of the wrong type, everything will come down burning.
1363    */
1364   @SuppressWarnings("unchecked")
1365   Rule createRuleWithParsedAttributeValues(Label label,
1366       Package.Builder pkgBuilder, Location ruleLocation,
1367       Map<String, ParsedAttributeValue> attributeValues, EventHandler eventHandler)
1368           throws LabelSyntaxException, InterruptedException {
1369     Rule rule = pkgBuilder.newRuleWithLabel(label, this, null, ruleLocation);
1370     rule.checkValidityPredicate(eventHandler);
1371 
1372     for (Attribute attribute : rule.getRuleClassObject().getAttributes()) {
1373       ParsedAttributeValue value = attributeValues.get(attribute.getName());
1374       if (attribute.isMandatory()) {
1375         Preconditions.checkState(value != null);
1376       }
1377 
1378       if (value == null) {
1379         continue;
1380       }
1381 
1382       rule.setAttributeValue(attribute, value.getValue(), value.getExplicitlySpecified());
1383       rule.setAttributeLocation(attribute, value.getLocation());
1384       checkAllowedValues(rule, attribute, eventHandler);
1385 
1386       if (attribute.getName().equals("visibility")) {
1387         // TODO(bazel-team): Verify that this cast works
1388         rule.setVisibility(PackageFactory.getVisibility((List<Label>) value.getValue()));
1389       }
1390     }
1391 
1392     rule.populateOutputFiles(eventHandler, pkgBuilder);
1393     Preconditions.checkState(!rule.containsErrors());
1394     return rule;
1395   }

1397   /**
1398    * Populates the attributes table of new rule "rule" from the
1399    * "attributeValues" mapping from attribute names to values in the build
1400    * language.  Errors are reported on "reporter".  "ast" is used to associate
1401    * location information with each rule attribute.
1402    */
1403   private void populateRuleAttributeValues(Rule rule,
1404                                            Package.Builder pkgBuilder,
1405                                            Map<String, Object> attributeValues,
1406                                            EventHandler eventHandler,
1407                                            FuncallExpression ast)
1408                                                throws InterruptedException {
1409     BitSet definedAttrs = new BitSet(); //  set of attr indices
1410 
1411     for (Map.Entry<String, Object> entry : attributeValues.entrySet()) {
1412       String attributeName = entry.getKey();
1413       Object attributeValue = entry.getValue();
1414       if (attributeValue == Runtime.NONE) {  // Ignore all None values.
1415         continue;
1416       }
1417       Integer attrIndex = setRuleAttributeValue(rule, eventHandler, attributeName, attributeValue);
1418       if (attrIndex != null) {
1419         definedAttrs.set(attrIndex);
1420         checkAttrValNonEmpty(rule, eventHandler, attributeValue, attrIndex);
1421       }
1422     }
1423 
1424     // Save the location of each non-default attribute definition:
1425     if (ast != null) {
1426       for (Argument.Passed arg : ast.getArguments()) {
1427         if (arg.isKeyword()) {
1428           String name = arg.getName();
1429           Integer attrIndex = getAttributeIndex(name);
1430           if (attrIndex != null) {
1431             rule.setAttributeLocation(attrIndex, arg.getValue().getLocation());
1432           }
1433         }
1434       }
1435     }
1436 
1437     List<Attribute> attrsWithComputedDefaults = new ArrayList<>();
1438 
1439     // Set defaults; ensure that every mandatory attribute has a value.  Use
1440     // the default if none is specified.
1441     int numAttributes = getAttributeCount();
1442     for (int attrIndex = 0; attrIndex < numAttributes; ++attrIndex) {
1443       if (!definedAttrs.get(attrIndex)) {
1444         Attribute attr = getAttribute(attrIndex);
1445         if (attr.isMandatory()) {
1446           rule.reportError(rule.getLabel() + ": missing value for mandatory "
1447                            + "attribute '" + attr.getName() + "' in '"
1448                            + name + "' rule", eventHandler);
1449         }
1450 
1451         if (attr.hasComputedDefault()) {
1452           attrsWithComputedDefaults.add(attr);
1453         } else {
1454           Object defaultValue = getAttributeNoncomputedDefaultValue(attr, pkgBuilder);
1455           checkAttrValNonEmpty(rule, eventHandler, defaultValue, attrIndex);
1456           rule.setAttributeValue(attr, defaultValue, /*explicit=*/false);
1457           checkAllowedValues(rule, attr, eventHandler);
1458         }
1459       }
1460     }
1461 
1462     // Evaluate and set any computed defaults now that all non-computed
1463     // TODO(bazel-team): remove this special casing. Thanks to configurable attributes refactoring,
1464     // computed defaults don't get bound to their final values at this point, so we no longer
1465     // have to wait until regular attributes have been initialized.
1466     for (Attribute attr : attrsWithComputedDefaults) {
1467       rule.setAttributeValue(attr, attr.getDefaultValue(rule), /*explicit=*/false);
1468     }
1469 
1470     // Now that all attributes are bound to values, collect and store configurable attribute keys.
1471     populateConfigDependenciesAttribute(rule);
1472     checkForDuplicateLabels(rule, eventHandler);
1473     checkThirdPartyRuleHasLicense(rule, pkgBuilder, eventHandler);
1474     checkForValidSizeAndTimeoutValues(rule, eventHandler);
1475   }

1477   /**
1478    * Collects all labels used as keys for configurable attributes and places them into
1479    * the special implicit attribute that tracks them.
1480    */
1481   private static void populateConfigDependenciesAttribute(Rule rule) {
1482     RawAttributeMapper attributes = RawAttributeMapper.of(rule);
1483     Attribute configDepsAttribute = attributes.getAttributeDefinition("$config_dependencies");
1484     if (configDepsAttribute == null) {
1485       // Not currently compatible with Skylark rules.
1486       return;
1487     }
1488 
1489     Set<Label> configLabels = new LinkedHashSet<>();
1490     for (Attribute attr : rule.getAttributes()) {
1491       Type.SelectorList<?> selectors = attributes.getSelectorList(attr.getName(), attr.getType());
1492       if (selectors != null) {
1493         configLabels.addAll(selectors.getKeyLabels());
1494       }
1495     }
1496 
1497     rule.setAttributeValue(configDepsAttribute, ImmutableList.copyOf(configLabels),
1498         /*explicit=*/false);
1499   }

1615   /**
1616    * Sets the value of attribute "attrName" in rule "rule", by converting the
1617    * build-language value "attrVal" to the appropriate type for the attribute.
1618    * Returns the attribute index iff successful, null otherwise.
1619    *
1620    * <p>In case of failure, error messages are reported on "handler", and "rule"
1621    * is marked as containing errors.
1622    */
1623   @SuppressWarnings("unchecked")
1624   private Integer setRuleAttributeValue(Rule rule,
1625                                         EventHandler eventHandler,
1626                                         String attrName,
1627                                         Object attrVal) {
1628     if (attrName.equals("name")) {
1629       return null; // "name" is handled specially
1630     }
1631 
1632     Integer attrIndex = getAttributeIndex(attrName);
1633     if (attrIndex == null) {
1634       rule.reportError(rule.getLabel() + ": no such attribute '" + attrName
1635           + "' in '" + name + "' rule", eventHandler);
1636       return null;
1637     }
1638 
1639     Attribute attr = getAttribute(attrIndex);
1640     Object converted;
1641     try {
1642       String what = "attribute '" + attrName + "' in '" + name + "' rule";
1643       converted = attr.getType().selectableConvert(attrVal, what, rule.getLabel());
1644 
1645       if ((converted instanceof Type.SelectorList<?>) && !attr.isConfigurable()) {
1646         rule.reportError(rule.getLabel() + ": attribute \"" + attr.getName()
1647             + "\" is not configurable", eventHandler);
1648         return null;
1649       }
1650 
1651       if ((converted instanceof List<?>) && !(converted instanceof GlobList<?>)) {
1652         if (attr.isOrderIndependent()) {
1653           converted = Ordering.natural().sortedCopy((List<? extends Comparable<?>>) converted);
1654         }
1655         converted = ImmutableList.copyOf((List<?>) converted);
1656       }
1657     } catch (Type.ConversionException e) {
1658       rule.reportError(rule.getLabel() + ": " + e.getMessage(), eventHandler);
1659       return null;
1660     }
1661 
1662     if (attrName.equals("visibility")) {
1663       List<Label> attrList = (List<Label>) converted;
1664       if (!attrList.isEmpty()
1665           && ConstantRuleVisibility.LEGACY_PUBLIC_LABEL.equals(attrList.get(0))) {
1666         rule.reportError(rule.getLabel() + ": //visibility:legacy_public only allowed in package "
1667             + "declaration", eventHandler);
1668       }
1669       rule.setVisibility(PackageFactory.getVisibility(attrList));
1670     }
1671 
1672     rule.setAttributeValue(attr, converted, /*explicit=*/true);
1673     checkAllowedValues(rule, attr, eventHandler);
1674     return attrIndex;
1675   }

1677   /**
1678    * Verifies that the rule has a valid value for the attribute according to its allowed values.
1679    *
1680    * <p>If the value for the given attribute on the given rule is invalid, an error will be recorded
1681    * in the given EventHandler.
1682    *
1683    * <p>If the rule is configurable, all of its potential values are evaluated, and errors for each
1684    * of the invalid values are reported.
1685    */
1686   private void checkAllowedValues(Rule rule, Attribute attribute, EventHandler eventHandler) {
1687     if (attribute.checkAllowedValues()) {
1688       PredicateWithMessage<Object> allowedValues = attribute.getAllowedValues();
1689       Iterable<?> values =
1690           AggregatingAttributeMapper.of(rule).visitAttribute(
1691               attribute.getName(), attribute.getType());
1692       for (Object value : values) {
1693         if (!allowedValues.apply(value)) {
1694           rule.reportError(String.format(rule.getLabel() + ": invalid value in '%s' attribute: %s",
1695               attribute.getName(),
1696               allowedValues.getErrorReason(value)), eventHandler);
1697         }
1698       }
1699     }
