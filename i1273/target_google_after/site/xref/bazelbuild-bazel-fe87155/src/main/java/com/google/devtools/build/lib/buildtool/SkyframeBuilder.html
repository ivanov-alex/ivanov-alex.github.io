85    @VisibleForTesting
86    public SkyframeBuilder(SkyframeExecutor skyframeExecutor, ActionCacheChecker actionCacheChecker,
87        boolean keepGoing, int numJobs, boolean checkOutputFiles,
88        ActionInputFileCache fileCache, int progressReportInterval) {
89      this.skyframeExecutor = skyframeExecutor;
90      this.actionCacheChecker = actionCacheChecker;
91      this.keepGoing = keepGoing;
92      this.numJobs = numJobs;
93      this.checkOutputFiles = checkOutputFiles;
94      this.fileCache = fileCache;
95      this.progressReportInterval = progressReportInterval;
96    }

98    @Override
99    public void buildArtifacts(
100       Set<Artifact> artifacts,
101       Set<ConfiguredTarget> parallelTests,
102       Set<ConfiguredTarget> exclusiveTests,
103       Collection<ConfiguredTarget> targetsToBuild,
104       Collection<AspectValue> aspects,
105       Executor executor,
106       Set<ConfiguredTarget> builtTargets,
107       boolean explain,
108       @Nullable Range<Long> lastExecutionTimeRange)
109       throws BuildFailedException, AbruptExitException, TestExecException, InterruptedException {
110     skyframeExecutor.prepareExecution(checkOutputFiles, lastExecutionTimeRange);
111     skyframeExecutor.setFileCache(fileCache);
112     // Note that executionProgressReceiver accesses builtTargets concurrently (after wrapping in a
113     // synchronized collection), so unsynchronized access to this variable is unsafe while it runs.
114     ExecutionProgressReceiver executionProgressReceiver =
115         new ExecutionProgressReceiver(Preconditions.checkNotNull(builtTargets),
116             countTestActions(exclusiveTests), skyframeExecutor.getEventBus());
117     ResourceManager.instance().setEventBus(skyframeExecutor.getEventBus());
118 
119     boolean success = false;
120     EvaluationResult<?> result;
121 
122     ActionExecutionStatusReporter statusReporter = ActionExecutionStatusReporter.create(
123         skyframeExecutor.getReporter(), executor, skyframeExecutor.getEventBus());
124 
125     AtomicBoolean isBuildingExclusiveArtifacts = new AtomicBoolean(false);
126     ActionExecutionInactivityWatchdog watchdog = new ActionExecutionInactivityWatchdog(
127         executionProgressReceiver.createInactivityMonitor(statusReporter),
128         executionProgressReceiver.createInactivityReporter(statusReporter,
129             isBuildingExclusiveArtifacts), progressReportInterval);
130 
131     skyframeExecutor.setActionExecutionProgressReportingObjects(executionProgressReceiver,
132         executionProgressReceiver, statusReporter);
133     watchdog.start();
134 
135     try {
136       result =
137           skyframeExecutor.buildArtifacts(
138               executor,
139               artifacts,
140               targetsToBuild,
141               aspects,
142               parallelTests,
143               /*exclusiveTesting=*/ false,
144               keepGoing,
145               explain,
146               numJobs,
147               actionCacheChecker,
148               executionProgressReceiver);
149       // progressReceiver is finished, so unsynchronized access to builtTargets is now safe.
150       success = processResult(result, keepGoing, skyframeExecutor);
151 
152       Preconditions.checkState(
153           !success
154               || result.keyNames().size()
155                   == (artifacts.size()
156                       + targetsToBuild.size()
157                       + aspects.size()
158                       + parallelTests.size()),
159           "Build reported as successful but not all artifacts and targets built: %s, %s",
160           result,
161           artifacts);
162 
163       // Run exclusive tests: either tagged as "exclusive" or is run in an invocation with
164       // --test_output=streamed.
165       isBuildingExclusiveArtifacts.set(true);
166       for (ConfiguredTarget exclusiveTest : exclusiveTests) {
167         // Since only one artifact is being built at a time, we don't worry about an artifact being
168         // built and then the build being interrupted.
169         result =
170             skyframeExecutor.buildArtifacts(
171                 executor,
172                 ImmutableSet.<Artifact>of(),
173                 targetsToBuild,
174                 aspects,
175                 ImmutableSet.of(exclusiveTest), /*exclusiveTesting=*/
176                 true,
177                 keepGoing,
178                 explain,
179                 numJobs,
180                 actionCacheChecker,
181                 null);
182         boolean exclusiveSuccess = processResult(result, keepGoing, skyframeExecutor);
183         Preconditions.checkState(!exclusiveSuccess || !result.keyNames().isEmpty(),
184             "Build reported as successful but test %s not executed: %s",
185             exclusiveTest, result);
186         success &= exclusiveSuccess;
187       }
188     } finally {
189       watchdog.stop();
190       ResourceManager.instance().unsetEventBus();
191       skyframeExecutor.setActionExecutionProgressReportingObjects(null, null, null);
192       statusReporter.unregisterFromEventBus();
193     }
194 
195     if (!success) {
196       throw new BuildFailedException();
197     }
198   }

200   private static boolean resultHasCatastrophicError(EvaluationResult<?> result) {
201     for (ErrorInfo errorInfo : result.errorMap().values()) {
202       if (errorInfo.isCatastrophic()) {
203         return true;
204       }
205     }
206     // An unreported catastrophe manifests with hasError() being true but no errors visible.
207     return result.hasError() && result.errorMap().isEmpty();
208   }

210   /**
211    * Process the Skyframe update, taking into account the keepGoing setting.
212    *
213    * <p>Returns false if the update() failed, but we should continue. Returns true on success.
214    * Throws on fail-fast failures.
215    */
216   private static boolean processResult(EvaluationResult<?> result, boolean keepGoing,
217       SkyframeExecutor skyframeExecutor) throws BuildFailedException, TestExecException {
218     if (result.hasError()) {
219       boolean hasCycles = false;
220       for (Map.Entry<SkyKey, ErrorInfo> entry : result.errorMap().entrySet()) {
221         Iterable<CycleInfo> cycles = entry.getValue().getCycleInfo();
222         skyframeExecutor.reportCycles(cycles, entry.getKey());
223         hasCycles |= !Iterables.isEmpty(cycles);
224       }
225       if (keepGoing && !resultHasCatastrophicError(result)) {
226         return false;
227       }
228       if (hasCycles || result.errorMap().isEmpty()) {
229         // error map may be empty in the case of a catastrophe.
230         throw new BuildFailedException();
231       } else {
232         rethrow(Preconditions.checkNotNull(result.getError().getException()));
233       }
234     }
235     return true;
236   }

238   /** Figure out why an action's execution failed and rethrow the right kind of exception. */
239   @VisibleForTesting
240   public static void rethrow(Throwable cause) throws BuildFailedException, TestExecException {
241     Throwable innerCause = cause.getCause();
242     if (innerCause instanceof TestExecException) {
243       throw (TestExecException) innerCause;
244     }
245     if (cause instanceof ActionExecutionException) {
246       ActionExecutionException actionExecutionCause = (ActionExecutionException) cause;
247       // Sometimes ActionExecutionExceptions are caused by Actions with no owner.
248       String message =
249           (actionExecutionCause.getLocation() != null)
250               ? (actionExecutionCause.getLocation().print() + " " + cause.getMessage())
251               : cause.getMessage();
252       throw new BuildFailedException(
253           message,
254           actionExecutionCause.isCatastrophe(),
255           actionExecutionCause.getAction(),
256           actionExecutionCause.getRootCauses(),
257           /*errorAlreadyShown=*/ !actionExecutionCause.showError());
258     } else if (cause instanceof MissingInputFileException) {
259       throw new BuildFailedException(cause.getMessage());
260     } else if (cause instanceof BuildFileNotFoundException) {
261       // Sadly, this can happen because we may load new packages during input discovery. Any
262       // failures reading those packages shouldn't terminate the build, but in Skyframe they do.
263       LoggingUtil.logToRemote(Level.WARNING, "undesirable loading exception", cause);
264       throw new BuildFailedException(cause.getMessage());
265     } else if (cause instanceof RuntimeException) {
266       throw (RuntimeException) cause;
267     } else if (cause instanceof Error) {
268       throw (Error) cause;
269     } else {
270       // We encountered an exception we don't think we should have encountered. This can indicate
271       // a bug in our code, such as lower level exceptions not being properly handled, or in our
272       // expectations in this method.
273       throw new IllegalArgumentException(
274           "action terminated with " + "unexpected exception: " + cause.getMessage(), cause);
275     }
276   }

278   private static int countTestActions(Iterable<ConfiguredTarget> testTargets) {
279     int count = 0;
280     for (ConfiguredTarget testTarget : testTargets) {
281       count += TestProvider.getTestStatusArtifacts(testTarget).size();
282     }
283     return count;
284   }

286   /**
287    * Listener for executed actions and built artifacts. We use a listener so that we have an
288    * accurate set of successfully run actions and built artifacts, even if the build is interrupted.
289    */
290   private static final class ExecutionProgressReceiver implements EvaluationProgressReceiver,
291       SkyframeActionExecutor.ProgressSupplier, SkyframeActionExecutor.ActionCompletedReceiver {
292     private static final NumberFormat PROGRESS_MESSAGE_NUMBER_FORMATTER;
293 
294     // Must be thread-safe!
295     private final Set<ConfiguredTarget> builtTargets;
296     private final Set<SkyKey> enqueuedActions = Sets.newConcurrentHashSet();
297     private final Set<Action> completedActions = Sets.newConcurrentHashSet();
298     private final Object activityIndicator = new Object();
299     /** Number of exclusive tests. To be accounted for in progress messages. */
300     private final int exclusiveTestsCount;
301     private final EventBus eventBus;
302 
303     static {
304       PROGRESS_MESSAGE_NUMBER_FORMATTER = NumberFormat.getIntegerInstance(Locale.ENGLISH);
305       PROGRESS_MESSAGE_NUMBER_FORMATTER.setGroupingUsed(true);
306     }
307 
308     /**
309      * {@code builtTargets} is accessed through a synchronized set, and so no other access to it
310      * is permitted while this receiver is active.
311      */
312     ExecutionProgressReceiver(Set<ConfiguredTarget> builtTargets, int exclusiveTestsCount,
313                               EventBus eventBus) {
314       this.builtTargets = Collections.synchronizedSet(builtTargets);
315       this.exclusiveTestsCount = exclusiveTestsCount;
316       this.eventBus = eventBus;
317     }
318 
319     @Override
320     public void invalidated(SkyKey skyKey, InvalidationState state) {}
321 
322     @Override
323     public void enqueueing(SkyKey skyKey) {
324       if (ActionExecutionValue.isReportWorthyAction(skyKey)) {
325         // Remember all enqueued actions for the benefit of progress reporting.
326         // We discover most actions early in the build, well before we start executing them.
327         // Some of these will be cache hits and won't be executed, so we'll need to account for them
328         // in the evaluated method too.
329         enqueuedActions.add(skyKey);
330       }
331     }
332 
333     @Override
334     public void computed(SkyKey skyKey, long elapsedTimeNanos) {}
335 
336     @Override
337     public void evaluated(SkyKey skyKey, Supplier<SkyValue> skyValueSupplier,
338         EvaluationState state) {
339       SkyFunctionName type = skyKey.functionName();
340       if (type.equals(SkyFunctions.TARGET_COMPLETION)) {
341         TargetCompletionValue value = (TargetCompletionValue) skyValueSupplier.get();
342         if (value != null) {
343           ConfiguredTarget target = value.getConfiguredTarget();
344           builtTargets.add(target);
345           eventBus.post(TargetCompleteEvent.createSuccessful(target));
346         }
347       } else if (type.equals(SkyFunctions.ASPECT_COMPLETION)) {
348         AspectCompletionValue value = (AspectCompletionValue) skyValueSupplier.get();
349         if (value != null) {
350           eventBus.post(AspectCompleteEvent.createSuccessful(value.getAspectValue()));
351         }
352       } else if (type.equals(SkyFunctions.ACTION_EXECUTION)) {
353         // Remember all completed actions, even those in error, regardless of having been cached or
354         // really executed.
355         actionCompleted((Action) skyKey.argument());
356       }
357     }
358 
359     /**
360      * {@inheritDoc}
361      *
362      * <p>This method adds the action to {@link #completedActions} and notifies the
363      * {@link #activityIndicator}.
364      *
365      * <p>We could do this only in the {@link #evaluated} method too, but as it happens the action
366      * executor tells the reporter about the completed action before the node is inserted into the
367      * graph, so the reporter would find out about the completed action sooner than we could
368      * have updated {@link #completedActions}, which would result in incorrect numbers on the
369      * progress messages. However we have to store completed actions in {@link #evaluated} too,
370      * because that's the only place we get notified about completed cached actions.
371      */
372     @Override
373     public void actionCompleted(Action a) {
374       if (ActionExecutionValue.isReportWorthyAction(a)) {
375         completedActions.add(a);
376         synchronized (activityIndicator) {
377           activityIndicator.notifyAll();
378         }
379       }
380     }
381 
382     @Override
383     public String getProgressString() {
384       return String.format("[%s / %s]",
385           PROGRESS_MESSAGE_NUMBER_FORMATTER.format(completedActions.size()),
386           PROGRESS_MESSAGE_NUMBER_FORMATTER.format(exclusiveTestsCount + enqueuedActions.size()));
387     }

389     ActionExecutionInactivityWatchdog.InactivityMonitor createInactivityMonitor(
390         final ActionExecutionStatusReporter statusReporter) {
391       return new ActionExecutionInactivityWatchdog.InactivityMonitor() {
392 
393         @Override
394         public boolean hasStarted() {
395           return !enqueuedActions.isEmpty();
396         }
397 
398         @Override
399         public int getPending() {
400           return statusReporter.getCount();
401         }
402 
403         @Override
404         public int waitForNextCompletion(int timeoutMilliseconds) throws InterruptedException {
405           synchronized (activityIndicator) {
406             int before = completedActions.size();
407             long startTime = BlazeClock.instance().currentTimeMillis();
408             while (true) {
409               activityIndicator.wait(timeoutMilliseconds);
410 
411               int completed = completedActions.size() - before;
412               long now = 0;
413               if (completed > 0 || (startTime + timeoutMilliseconds) <= (now = BlazeClock.instance()
414                   .currentTimeMillis())) {
415                 // Some actions completed, or timeout fully elapsed.
416                 return completed;
417               } else {
418                 // Spurious Wakeup -- no actions completed and there's still time to wait.
419                 timeoutMilliseconds -= now - startTime;  // account for elapsed wait time
420                 startTime = now;
421               }
422             }
423           }
424         }
425       };
426     }
427 
428     ActionExecutionInactivityWatchdog.InactivityReporter createInactivityReporter(
429         final ActionExecutionStatusReporter statusReporter,
430         final AtomicBoolean isBuildingExclusiveArtifacts) {
431       return new ActionExecutionInactivityWatchdog.InactivityReporter() {
432         @Override
433         public void maybeReportInactivity() {
434           // Do not report inactivity if we are currently running an exclusive test or a streaming
435           // action (in practice only tests can stream and it implicitly makes them exclusive).
436           if (!isBuildingExclusiveArtifacts.get()) {
437             statusReporter.showCurrentlyExecutingActions(
438                 ExecutionProgressReceiver.this.getProgressString() + " ");
439           }
440         }
441       };
442     }
