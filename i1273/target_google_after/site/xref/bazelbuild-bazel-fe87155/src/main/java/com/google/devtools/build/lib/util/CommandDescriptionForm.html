15  package com.google.devtools.build.lib.util;
16  
17  /**
18   * Forms in which a command can be described by {@link CommandFailureUtils#describeCommand}.
19   */
20  public enum CommandDescriptionForm {
21    /**
22     * A form that is usually suitable for identifying the command but not for
23     * re-executing it.  The working directory and environment are not shown, and
24     * the arguments are truncated to a maximum of a few hundred bytes.
25     */
26    ABBREVIATED,
27  
28    /**
29     * A form that is complete and suitable for a user to copy and paste into a
30     * shell.  On Linux, the command is placed in a subshell so it has no side
31     * effects on the user's shell.  On Windows, this is not implemented, but the
32     * side effects in question are less severe (no "exec").
33     */
34    COMPLETE,
35  
36    /**
37     * A form that is complete and does not isolate side effects.  Suitable for
38     * launch scripts, i.e., "blaze run --script_path".
39     */
40    COMPLETE_UNISOLATED,
41  }
