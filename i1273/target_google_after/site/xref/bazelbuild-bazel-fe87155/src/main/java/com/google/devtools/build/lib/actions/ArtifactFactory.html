39  /**
40   * A cache of Artifacts, keyed by Path.
41   */
42  @ThreadSafe
43  public class ArtifactFactory implements ArtifactResolver, ArtifactSerializer, ArtifactDeserializer {
44  
45    private final Path execRoot;
46  
47    /**
48     * Cache of source artifacts.
49     */
50    private final SourceArtifactCache sourceArtifactCache = new SourceArtifactCache();
51  
52    /**
53     * Map of package names to source root paths so that we can create source
54     * artifact paths given execPaths in the symlink forest.
55     */
56    private ImmutableMap<PackageIdentifier, Root> packageRoots;
57  
58    /**
59     * Reverse-ordered list of derived roots for use in looking up or (in rare cases) creating
60     * derived artifacts from execPaths. The reverse order is only significant for overlapping roots
61     * so that the longest is found first.
62     */
63    private ImmutableCollection<Root> derivedRoots = ImmutableList.of();

65    private ArtifactIdRegistry artifactIdRegistry = new ArtifactIdRegistry();

67    private static class SourceArtifactCache {
68  
69      private class Entry {
70        private final Artifact artifact;
71        private final int idOfBuild;
72  
73        Entry(Artifact artifact) {
74          this.artifact = artifact;
75          idOfBuild = buildId;
76        }
77  
78        Artifact getArtifact() {
79          return artifact;
80        }
81  
82        int getIdOfBuild() {
83          return idOfBuild;
84        }
85      }
86  
87      /**
88       * The main Path to source artifact cache. There will always be exactly one canonical
89       * artifact for a given source path.
90       */
91      private final Map<PathFragment, Entry> pathToSourceArtifact = new HashMap<>();
92  
93      /** Id of current build. Has to be increased every time before execution phase starts. */
94      private int buildId = 0;
95  
96      /** Returns artifact if it present in the cache, otherwise null. */
97      Artifact getArtifact(PathFragment execPath) {
98        Entry cacheEntry = pathToSourceArtifact.get(execPath);
99        return cacheEntry == null ? null : cacheEntry.getArtifact();
100     }
101 
102     /**
103      * Returns artifact if it present in the cache and was created during this build,
104      * otherwise null.
105      */
106     Artifact getArtifactIfValid(PathFragment execPath) {
107       Entry cacheEntry = pathToSourceArtifact.get(execPath);
108       if (cacheEntry != null && cacheEntry.getIdOfBuild() == buildId) {
109         return cacheEntry.getArtifact();
110       }
111       return null;
112     }
113 
114     void markEntryAsValid(PathFragment execPath) {
115       Artifact oldValue = Preconditions.checkNotNull(getArtifact(execPath));
116       pathToSourceArtifact.put(execPath, new Entry(oldValue));
117     }
118 
119     void newBuild() {
120       buildId++;
121     }
122 
123     void clear() {
124       pathToSourceArtifact.clear();
125       buildId = 0;
126     }
127 
128     void putArtifact(PathFragment execPath, Artifact artifact) {
129       pathToSourceArtifact.put(execPath, new Entry(artifact));
130     }
131   }

146   public synchronized void clear() {

154   /**
155    * Set the set of known packages and their corresponding source artifact
156    * roots. Must be called exactly once after construction or clear().
157    *
158    * @param packageRoots the map of package names to source artifact roots to
159    *        use.

161   public synchronized void setPackageRoots(Map<PackageIdentifier, Root> packageRoots) {
162     this.packageRoots = ImmutableMap.copyOf(packageRoots);
163     sourceArtifactCache.newBuild();
164   }

166   /**
167    * Set the set of known derived artifact roots. Must be called exactly once
168    * after construction or clear().
169    *
170    * @param roots the set of derived artifact roots to use
171    */
172   public synchronized void setDerivedArtifactRoots(Collection<Root> roots) {
173     derivedRoots = ImmutableSortedSet.<Root>reverseOrder().addAll(roots).build();
174   }

176   @Override
177   public Artifact getSourceArtifact(PathFragment execPath, Root root, ArtifactOwner owner) {
178     Preconditions.checkArgument(!execPath.isAbsolute(), "%s %s %s", execPath, root, owner);
179     Preconditions.checkNotNull(owner, "%s %s", execPath, root);
180     execPath = execPath.normalize();
181     return getArtifact(root.getPath().getRelative(execPath), root, execPath, owner, null);
182   }

189   /**
190    * Only for use by BinTools! Returns an artifact for a tool at the given path
191    * fragment, relative to the exec root, creating it if not found. This method
192    * only works for normalized, relative paths.
193    */
194   public Artifact getDerivedArtifact(PathFragment execPath) {
195     Preconditions.checkArgument(!execPath.isAbsolute(), execPath);
196     Preconditions.checkArgument(execPath.isNormalized(), execPath);
197     // TODO(bazel-team): Check that either BinTools do not change over the life of the Blaze server,
198     // or require that a legitimate ArtifactOwner be passed in here to allow for ownership.
199     return getArtifact(execRoot.getRelative(execPath), Root.execRootAsDerivedRoot(execRoot),
200         execPath, ArtifactOwner.NULL_OWNER, null);
201   }

203   private void validatePath(PathFragment rootRelativePath, Root root) {
204     Preconditions.checkArgument(!rootRelativePath.isAbsolute(), rootRelativePath);
205     Preconditions.checkArgument(rootRelativePath.isNormalized(), rootRelativePath);
206     Preconditions.checkArgument(root.getPath().startsWith(execRoot), "%s %s", root, execRoot);
207     Preconditions.checkArgument(!root.getPath().equals(execRoot), "%s %s", root, execRoot);
208     // TODO(bazel-team): this should only accept roots from derivedRoots.
209     //Preconditions.checkArgument(derivedRoots.contains(root), "%s not in %s", root, derivedRoots);
210   }

212   /**
213    * Returns an artifact for a tool at the given root-relative path under the given root, creating
214    * it if not found. This method only works for normalized, relative paths.
215    *
216    * <p>The root must be below the execRoot, and the execPath of the resulting Artifact is computed
217    * as {@code root.getRelative(rootRelativePath).relativeTo(execRoot)}.
218    */
219   // TODO(bazel-team): Don't allow root == execRoot.
220   public Artifact getDerivedArtifact(PathFragment rootRelativePath, Root root,
221       ArtifactOwner owner) {
222     validatePath(rootRelativePath, root);
223     Path path = root.getPath().getRelative(rootRelativePath);
224     return getArtifact(path, root, path.relativeTo(execRoot), owner, null);
225   }

250   /**
251    * Returns the Artifact for the specified path, creating one if not found and
252    * setting the <code>root</code> and <code>execPath</code> to the
253    * specified values.
254    */
255   private synchronized Artifact getArtifact(Path path, Root root, PathFragment execPath,
256       ArtifactOwner owner, @Nullable SpecialArtifactType type) {
257     Preconditions.checkNotNull(root);
258     Preconditions.checkNotNull(execPath);
259 
260     if (!root.isSourceRoot()) {
261       return createArtifact(path, root, execPath, owner, type);
262     }
263 
264     Artifact artifact = sourceArtifactCache.getArtifact(execPath);
265     if (artifact == null || !Objects.equals(artifact.getArtifactOwner(), owner)
266         || !root.equals(artifact.getRoot())) {
267       // There really should be a safety net that makes it impossible to create two Artifacts
268       // with the same exec path but a different Owner, but we also need to reuse Artifacts from
269       // previous builds.
270       artifact = createArtifact(path, root, execPath, owner, type);
271       sourceArtifactCache.putArtifact(execPath, artifact);
272     }
273     return artifact;
274   }
275 
276   private Artifact createArtifact(Path path, Root root, PathFragment execPath, ArtifactOwner owner,
277       @Nullable SpecialArtifactType type) {
278     Preconditions.checkNotNull(owner, path);
279     if (type == null) {
280       return new Artifact(path, root, execPath, owner);
281     } else {
282       return new Artifact.SpecialArtifact(path, root, execPath, owner, type);
283     }
284   }

286   /**
287    * Returns an {@link Artifact} with exec path formed by composing {@code baseExecPath} and
288    * {@code relativePath} (via {@code baseExecPath.getRelative(relativePath)} if baseExecPath is
289    * not null). That Artifact will have root determined by the package roots of this factory if it
290    * lives in a subpackage distinct from that of baseExecPath, and {@code baseRoot} otherwise.
291    */
292   public synchronized Artifact resolveSourceArtifactWithAncestor(
293       PathFragment relativePath, PathFragment baseExecPath, Root baseRoot) {
294     Preconditions.checkState(
295         (baseExecPath == null) == (baseRoot == null),
296         "%s %s %s",
297         relativePath,
298         baseExecPath,
299         baseRoot);
300     Preconditions.checkState(
301         relativePath.segmentCount() > 0, "%s %s %s", relativePath, baseExecPath, baseRoot);
302     PathFragment execPath =
303         baseExecPath == null ? relativePath : baseExecPath.getRelative(relativePath);
304     execPath = execPath.normalize();
305     if (execPath.containsUplevelReferences()) {
306       // Source exec paths cannot escape the source root.
307       return null;
308     }
309     Artifact artifact = sourceArtifactCache.getArtifactIfValid(execPath);
310     if (artifact != null) {
311       return artifact;
312     }
313     // Don't create an artifact if it's derived.
314     if (findDerivedRoot(execRoot.getRelative(execPath)) != null) {
315       return null;
316     }
317 
318     return createArtifactIfNotValid(findSourceRoot(execPath, baseExecPath, baseRoot), execPath);
319   }
320 
321   @Nullable
322   private Root findSourceRoot(
323       PathFragment execPath, @Nullable PathFragment baseExecPath, @Nullable Root baseRoot) {
324     // Probe the known packages to find the longest package prefix up until the base, or until the
325     // root directory if our execPath doesn't start with baseExecPath due to uplevel references.
326     PathFragment dir;
327     for (dir = execPath.getParentDirectory();
328         dir != null && !dir.equals(baseExecPath);
329         dir = dir.getParentDirectory()) {
330       Root sourceRoot = packageRoots.get(PackageIdentifier.createInDefaultRepo(dir));
331       if (sourceRoot != null) {
332         return sourceRoot;
333       }
334     }
335     return dir != null && dir.equals(baseExecPath) ? baseRoot : null;
336   }

338   @Override
339   public Artifact resolveSourceArtifact(PathFragment execPath) {
340     return resolveSourceArtifactWithAncestor(execPath, null, null);
341   }

343   @Override
344   public synchronized Map<PathFragment, Artifact> resolveSourceArtifacts(
345       Iterable<PathFragment> execPaths, PackageRootResolver resolver)
346           throws PackageRootResolutionException {
347     Map<PathFragment, Artifact> result = new HashMap<>();
348     ArrayList<PathFragment> unresolvedPaths = new ArrayList<>();
349 
350     for (PathFragment execPath : execPaths) {
351       PathFragment execPathNormalized = execPath.normalize();
352       if (execPathNormalized.containsUplevelReferences()) {
353         // Source exec paths cannot escape the source root.
354         result.put(execPath, null);
355         continue;
356       }
357       // First try a quick map lookup to see if the artifact already exists.
358       Artifact a = sourceArtifactCache.getArtifactIfValid(execPathNormalized);
359       if (a != null) {
360         result.put(execPath, a);
361       } else if (findDerivedRoot(execRoot.getRelative(execPathNormalized)) != null) {
362         // Don't create an artifact if it's derived.
363         result.put(execPath, null);
364       } else {
365         // Remember this path, maybe we can resolve it with the help of PackageRootResolver.
366         unresolvedPaths.add(execPath);
367       }
368     }
369     Map<PathFragment, Root> sourceRoots = resolver.findPackageRoots(unresolvedPaths);
370     // We are missing some dependencies. We need to rerun this method later.
371     if (sourceRoots == null) {
372       return null;
373     }
374     for (PathFragment path : unresolvedPaths) {
375       result.put(path, createArtifactIfNotValid(sourceRoots.get(path), path));
376     }
377     return result;
378   }
379 
380   private Artifact createArtifactIfNotValid(Root sourceRoot, PathFragment execPath) {
381     if (sourceRoot == null) {
382       return null;  // not a path that we can find...
383     }
384     Artifact artifact = sourceArtifactCache.getArtifact(execPath);
385     if (artifact != null && sourceRoot.equals(artifact.getRoot())) {
386       // Source root of existing artifact hasn't changed so we should mark corresponding entry in
387       // the cache as valid.
388       sourceArtifactCache.markEntryAsValid(execPath);
389     } else {
390       // Must be a new artifact or artifact in the cache is stale, so create a new one.
391       artifact = getSourceArtifact(execPath, sourceRoot, ArtifactOwner.NULL_OWNER); 
392     }
393     return artifact;
394   }

396   /**
397    * Finds the derived root for a full path by comparing against the known
398    * derived artifact roots.
399    *
400    * @param path a Path to resolve the root for
401    * @return the root for the path or null if no root can be determined
402    */
403   @VisibleForTesting  // for our own unit tests only.
404   synchronized Root findDerivedRoot(Path path) {
405     for (Root prefix : derivedRoots) {
406       if (path.startsWith(prefix.getPath())) {
407         return prefix;
408       }
409     }
410     return null;
411   }

413   // Non-final only because clear()ing a map does not actually free the memory it took up, so we
414   // assign it to a new map in lieu of clearing.
415   private ConcurrentMap<PathFragment, Artifact> deserializedArtifacts =
416       new ConcurrentHashMap<>();

418   /**
419    * Returns the map of all artifacts that were deserialized this build. The caller should process
420    * them and then call {@link #clearDeserializedArtifacts}.
421    */
422   public Map<PathFragment, Artifact> getDeserializedArtifacts() {
423     return deserializedArtifacts;
424   }

426   /** Clears the map of deserialized artifacts. */
427   public void clearDeserializedArtifacts() {
428     deserializedArtifacts = new ConcurrentHashMap<>();
429   }

431   /**
432    * Resolves an artifact based on its deserialized representation. The artifact can be either a
433    * source or a derived one.
434    *
435    * <p>Note: this method represents a hole in the usual contract that artifacts with a random path
436    * cannot be created. Unfortunately, we currently need this in some cases.
437    *
438    * @param execPath the exec path of the artifact
439    * @throws PackageRootResolutionException on failure to determine the package roots of
440    *    {@code execPath}
441    */
442   public Artifact deserializeArtifact(PathFragment execPath, PackageRootResolver resolver)
443       throws PackageRootResolutionException {
444     Preconditions.checkArgument(!execPath.isAbsolute(), execPath);
445     Path path = execRoot.getRelative(execPath);
446     Root root = findDerivedRoot(path);
447 
448     if (root != null) {
449       Artifact result = getDerivedArtifact(path.relativeTo(root.getPath()), root,
450           Artifact.DESERIALIZED_MARKER_OWNER);
451       Artifact oldResult = deserializedArtifacts.putIfAbsent(execPath, result);
452       if (oldResult != null) {
453         result = oldResult;
454       }
455       return result;
456     } else {
457       Map<PathFragment, Root> sourceRoots = resolver.findPackageRoots(Lists.newArrayList(execPath));
458       if (sourceRoots == null || sourceRoots.get(execPath) == null) {
459         return null;
460       }
461       return getSourceArtifact(execPath, sourceRoots.get(execPath), ArtifactOwner.NULL_OWNER);
462     }
463   }
