108   // We keep track of actions already executed this build in order to avoid executing a shared
109   // action twice. Note that we may still unnecessarily re-execute the action on a subsequent
110   // build: say actions A and B are shared. If A is requested on the first build and then B is
111   // requested on the second build, we will execute B even though its output files are up to date.
112   // However, we will not re-execute A on a subsequent build.
113   // We do not allow the shared action to re-execute in the same build, even after the first
114   // action has finished execution, because a downstream action might be reading the output file
115   // at the same time as the shared action was writing to it.
116   // This map is also used for Actions that try to execute twice because they have discovered
117   // headers -- the SkyFunction tries to declare a dep on the missing headers and has to restart.
118   // We don't want to execute the action again on the second entry to the SkyFunction.
119   // In both cases, we store the already-computed ActionExecutionValue to avoid having to compute it
120   // again.
121   private ConcurrentMap<Artifact, Pair<Action, FutureTask<ActionExecutionValue>>> buildActionMap;

143   /**
144    * A typed union of {@link ActionConflictException}, which indicates two actions that generate
145    * the same {@link Artifact}, and {@link ArtifactPrefixConflictException}, which indicates that
146    * the path of one {@link Artifact} is a prefix of another.
147    */
148   public static class ConflictException extends Exception {
149     @Nullable private final ActionConflictException ace;
150     @Nullable private final ArtifactPrefixConflictException apce;
151 
152     public ConflictException(ActionConflictException e) {
153       super(e);
154       this.ace = e;
155       this.apce = null;
156     }
157 
158     public ConflictException(ArtifactPrefixConflictException e) {
159       super(e);
160       this.ace = null;
161       this.apce = e;
162     }
163 
164     void rethrowTyped() throws ActionConflictException, ArtifactPrefixConflictException {
165       if (ace == null) {
166         throw Preconditions.checkNotNull(apce);
167       }
168       if (apce == null) {
169         throw Preconditions.checkNotNull(ace);
170       }
171       throw new IllegalStateException();
172     }
173   }

185   /**
186    * Find conflicts between generated artifacts. There are two ways to have conflicts. First, if
187    * two (unshareable) actions generate the same output artifact, this will result in an {@link
188    * ActionConflictException}. Second, if one action generates an artifact whose path is a prefix of
189    * another artifact's path, those two artifacts cannot exist simultaneously in the output tree.
190    * This causes an {@link ArtifactPrefixConflictException}. The relevant exceptions are stored in
191    * the executor in {@code badActionMap}, and will be thrown immediately when that action is
192    * executed. Those exceptions persist, so that even if the action is not executed this build, the
193    * first time it is executed, the correct exception will be thrown.
194    *
195    * <p>This method must be called if a new action was added to the graph this build, so
196    * whenever a new configured target was analyzed this build. It is somewhat expensive (~1s
197    * range for a medium build as of 2014), so it should only be called when necessary.
198    *
199    * <p>Conflicts found may not be requested this build, and so we may overzealously throw an error.
200    * For instance, if actions A and B generate the same artifact foo, and the user first requests
201    * A' depending on A, and then in a subsequent build B' depending on B, we will fail the second
202    * build, even though it would have succeeded if it had been the only build. However, since
203    * Skyframe does not know the transitive dependencies of the request, we err on the conservative
204    * side.
205    *
206    * <p>If the user first runs one action on the first build, and on the second build adds a
207    * conflicting action, only the second action's error may be reported (because the first action
208    * will be cached), whereas if both actions were requested for the first time, both errors would
209    * be reported. However, the first time an action is added to the build, we are guaranteed to find
210    * any conflicts it has, since this method will compare it against all other actions. So there is
211    * no sequence of builds that can evade the error.
212    */
213   void findAndStoreArtifactConflicts(Iterable<ActionLookupValue> actionLookupValues)
214       throws InterruptedException {
215     ConcurrentMap<Action, ConflictException> temporaryBadActionMap = new ConcurrentHashMap<>();
216     Pair<ActionGraph, SortedMap<PathFragment, Artifact>> result;
217     result = constructActionGraphAndPathMap(actionLookupValues, temporaryBadActionMap);
218     ActionGraph actionGraph = result.first;
219     SortedMap<PathFragment, Artifact> artifactPathMap = result.second;
220 
221     // Report an error for every derived artifact which is a prefix of another.
222     // If x << y << z (where x << y means "y starts with x"), then we only report (x,y), (x,z), but
223     // not (y,z).
224     Iterator<PathFragment> iter = artifactPathMap.keySet().iterator();
225     if (!iter.hasNext()) {
226       // No actions in graph -- currently happens only in tests. Special-cased because .next() call
227       // below is unconditional.
228       this.badActionMap = ImmutableMap.of();
229       return;
230     }
231     for (PathFragment pathJ = iter.next(); iter.hasNext(); ) {
232       // For each comparison, we have a prefix candidate (pathI) and a suffix candidate (pathJ).
233       // At the beginning of the loop, we set pathI to the last suffix candidate, since it has not
234       // yet been tested as a prefix candidate, and then set pathJ to the paths coming after pathI,
235       // until we come to one that does not contain pathI as a prefix. pathI is then verified not to
236       // be the prefix of any path, so we start the next run of the loop.
237       PathFragment pathI = pathJ;
238       // Compare pathI to the paths coming after it.
239       while (iter.hasNext()) {
240         pathJ = iter.next();
241         if (pathJ.startsWith(pathI)) { // prefix conflict.
242           Artifact artifactI = Preconditions.checkNotNull(artifactPathMap.get(pathI), pathI);
243           Artifact artifactJ = Preconditions.checkNotNull(artifactPathMap.get(pathJ), pathJ);
244           Action actionI =
245               Preconditions.checkNotNull(actionGraph.getGeneratingAction(artifactI), artifactI);
246           Action actionJ =
247               Preconditions.checkNotNull(actionGraph.getGeneratingAction(artifactJ), artifactJ);
248           if (actionI.shouldReportPathPrefixConflict(actionJ)) {
249             ArtifactPrefixConflictException exception = new ArtifactPrefixConflictException(pathI,
250                 pathJ, actionI.getOwner().getLabel(), actionJ.getOwner().getLabel());
251             temporaryBadActionMap.put(actionI, new ConflictException(exception));
252             temporaryBadActionMap.put(actionJ, new ConflictException(exception));
253           }
254         } else { // pathJ didn't have prefix pathI, so no conflict possible for pathI.
255           break;
256         }
257       }
258     }
259     this.badActionMap = ImmutableMap.copyOf(temporaryBadActionMap);
260   }

262   /**
263    * Simultaneously construct an action graph for all the actions in Skyframe and a map from
264    * {@link PathFragment}s to their respective {@link Artifact}s. We do this in a threadpool to save
265    * around 1.5 seconds on a mid-sized build versus a single-threaded operation.
266    */
267   private static Pair<ActionGraph, SortedMap<PathFragment, Artifact>>
268       constructActionGraphAndPathMap(
269           Iterable<ActionLookupValue> values,
270           ConcurrentMap<Action, ConflictException> badActionMap) throws InterruptedException {
271     MutableActionGraph actionGraph = new MapBasedActionGraph();
272     ConcurrentNavigableMap<PathFragment, Artifact> artifactPathMap = new ConcurrentSkipListMap<>();
273     // Action graph construction is CPU-bound.
274     int numJobs = Runtime.getRuntime().availableProcessors();
275     // No great reason for expecting 5000 action lookup values, but not worth counting size of
276     // values.
277     Sharder<ActionLookupValue> actionShards = new Sharder<>(numJobs, 5000);
278     for (ActionLookupValue value : values) {
279       actionShards.add(value);
280     }
281 
282     ThrowableRecordingRunnableWrapper wrapper = new ThrowableRecordingRunnableWrapper(
283         "SkyframeActionExecutor#constructActionGraphAndPathMap");
284 
285     ExecutorService executor = Executors.newFixedThreadPool(
286         numJobs,
287         new ThreadFactoryBuilder().setNameFormat("ActionLookupValue Processor %d").build());
288     for (List<ActionLookupValue> shard : actionShards) {
289       executor.execute(
290           wrapper.wrap(actionRegistration(shard, actionGraph, artifactPathMap, badActionMap)));
291     }
292     boolean interrupted = ExecutorUtil.interruptibleShutdown(executor);
293     Throwables.propagateIfPossible(wrapper.getFirstThrownError());
294     if (interrupted) {
295       throw new InterruptedException();
296     }
297     return Pair.<ActionGraph, SortedMap<PathFragment, Artifact>>of(actionGraph, artifactPathMap);
298   }

300   private static Runnable actionRegistration(
301       final List<ActionLookupValue> values,
302       final MutableActionGraph actionGraph,
303       final ConcurrentMap<PathFragment, Artifact> artifactPathMap,
304       final ConcurrentMap<Action, ConflictException> badActionMap) {
305     return new Runnable() {
306       @Override
307       public void run() {
308         for (ActionLookupValue value : values) {
309           Set<Action> registeredActions = new HashSet<>();
310           for (Map.Entry<Artifact, Action> entry : value.getMapForConsistencyCheck().entrySet()) {
311             Action action = entry.getValue();
312             // We have an entry for each <action, artifact> pair. Only try to register each action
313             // once.
314             if (registeredActions.add(action)) {
315               try {
316                 actionGraph.registerAction(action);
317               } catch (ActionConflictException e) {
318                 Exception oldException = badActionMap.put(action, new ConflictException(e));
319                 Preconditions.checkState(oldException == null,
320                   "%s | %s | %s", action, e, oldException);
321                 // We skip the rest of the loop, and do not add the path->artifact mapping for this
322                 // artifact below -- we don't need to check it since this action is already in
323                 // error.
324                 continue;
325               }
326             }
327             artifactPathMap.put(entry.getKey().getExecPath(), entry.getKey());
328           }
329         }
330       }
331     };
332   }

368   /**
369    * Executes the provided action on the current thread. Returns the ActionExecutionValue with the
370    * result, either computed here or already computed on another thread.
371    *
372    * <p>For use from {@link ArtifactFunction} only.
373    */
374   ActionExecutionValue executeAction(Action action, ActionMetadataHandler metadataHandler,
375       long actionStartTime,
376       ActionExecutionContext actionExecutionContext)
377       throws ActionExecutionException, InterruptedException {
378     Exception exception = badActionMap.get(action);
379     if (exception != null) {
380       // If action had a conflict with some other action in the graph, report it now.
381       reportError(exception.getMessage(), exception, action, null);
382     }
383     Artifact primaryOutput = action.getPrimaryOutput();
384     FutureTask<ActionExecutionValue> actionTask =
385         new FutureTask<>(new ActionRunner(action, metadataHandler,
386             actionStartTime, actionExecutionContext));
387     // Check to see if another action is already executing/has executed this value.
388     Pair<Action, FutureTask<ActionExecutionValue>> oldAction =
389         buildActionMap.putIfAbsent(primaryOutput, Pair.of(action, actionTask));
390 
391     if (oldAction == null) {
392       actionTask.run();
393     } else if (action == oldAction.first) {
394       // We only allow the same action to be executed twice if it discovers inputs. We allow that
395       // because we need to declare additional dependencies on those new inputs.
396       Preconditions.checkState(action.discoversInputs(),
397           "Same action shouldn't execute twice in build: %s", action);
398       actionTask = oldAction.second;
399     } else {
400       Preconditions.checkState(Actions.canBeShared(oldAction.first, action),
401           "Actions cannot be shared: %s %s", oldAction.first, action);
402       // Wait for other action to finish, so any actions that depend on its outputs can execute.
403       actionTask = oldAction.second;
404     }
405     try {
406       return actionTask.get();
407     } catch (ExecutionException e) {
408       Throwables.propagateIfPossible(e.getCause(),
409           ActionExecutionException.class, InterruptedException.class);
410       throw new IllegalStateException(e);
411     } finally {
412       String message = action.getProgressMessage();
413       if (message != null) {
414         // Tell the receiver that the action has completed *before* telling the reporter.
415         // This way the latter will correctly show the number of completed actions when task
416         // completion messages are enabled (--show_task_finish).
417         if (completionReceiver != null) {
418           completionReceiver.actionCompleted(action);
419         }
420         reporter.finishTask(null, prependExecPhaseStats(message));
421       }
422     }
423   }

461   /**
462    * Checks the action cache to see if {@code action} needs to be executed, or is up to date.
463    * Returns a token with the semantics of {@link ActionCacheChecker#getTokenIfNeedToExecute}: null
464    * if the action is up to date, and non-null if it needs to be executed, in which case that token
465    * should be provided to the ActionCacheChecker after execution.
466    */
467   Token checkActionCache(Action action, MetadataHandler metadataHandler,
468       long actionStartTime, Iterable<Artifact> resolvedCacheArtifacts) {
469     profiler.startTask(ProfilerTask.ACTION_CHECK, action);
470     Token token = actionCacheChecker.getTokenIfNeedToExecute(
471         action, resolvedCacheArtifacts, explain ? reporter : null, metadataHandler);
472     profiler.completeTask(ProfilerTask.ACTION_CHECK);
473     if (token == null) {
474       boolean eventPosted = false;
475       // Notify BlazeRuntimeStatistics about the action middleman 'execution'.
476       if (action.getActionType().isMiddleman()) {
477         postEvent(new ActionMiddlemanEvent(action, actionStartTime));
478         eventPosted = true;
479       }
480 
481       if (action instanceof NotifyOnActionCacheHit) {
482         NotifyOnActionCacheHit notify = (NotifyOnActionCacheHit) action;
483         notify.actionCacheHit(executorEngine);
484       }
485 
486       // We still need to check the outputs so that output file data is available to the value.
487       checkOutputs(action, metadataHandler);
488       if (!eventPosted) {
489         postEvent(new CachedActionEvent(action, actionStartTime));
490       }
491     }
492     return token;
493   }

495   void afterExecution(Action action, MetadataHandler metadataHandler, Token token) {
496     if (!actionReallyExecuted(action)) {
497       // If an action shared with this one executed, then we need not update the action cache, since
498       // the other action will do it. Moreover, this action is not aware of metadata acquired
499       // during execution, so its metadata handler is likely unusable anyway.
500       return;
501     }
502     try {
503       actionCacheChecker.afterExecution(action, token, metadataHandler);
504     } catch (IOException e) {
505       // Skyframe has already done all the filesystem access needed for outputs and swallows
506       // IOExceptions for inputs. So an IOException is impossible here.
507       throw new IllegalStateException(
508           "failed to update action cache for " + action.prettyPrint()
509               + ", but all outputs should already have been checked", e);
510     }
511   }

525   Collection<Artifact> discoverInputs(Action action, PerActionFileCache graphFileCache,

537     } catch (ActionExecutionException e) {
538       throw processAndThrow(e, action, actionExecutionContext.getFileOutErr());
539     }

569   private class ActionRunner implements Callable<ActionExecutionValue> {
570     private final Action action;
571     private final ActionMetadataHandler metadataHandler;
572     private long actionStartTime;
573     private ActionExecutionContext actionExecutionContext;
574 
575     ActionRunner(Action action, ActionMetadataHandler metadataHandler,
576         long actionStartTime,
577         ActionExecutionContext actionExecutionContext) {
578       this.action = action;
579       this.metadataHandler = metadataHandler;
580       this.actionStartTime = actionStartTime;
581       this.actionExecutionContext = actionExecutionContext;
582     }
583 
584     @Override
585     public ActionExecutionValue call() throws ActionExecutionException, InterruptedException {
586       profiler.startTask(ProfilerTask.ACTION, action);
587       try {
588         if (actionCacheChecker.isActionExecutionProhibited(action)) {
589           // We can't execute an action (e.g. because --check_???_up_to_date option was used). Fail
590           // the build instead.
591           synchronized (reporter) {
592             TargetOutOfDateException e = new TargetOutOfDateException(action);
593             reporter.handle(Event.error(e.getMessage()));
594             recordExecutionError();
595             throw e;
596           }
597         }
598 
599         String message = action.getProgressMessage();
600         if (message != null) {
601           reporter.startTask(null, prependExecPhaseStats(message));
602         }
603         statusReporterRef.get().setPreparing(action);
604 
605         createOutputDirectories(action);
606 
607         Preconditions.checkState(actionExecutionContext.getMetadataHandler() == metadataHandler,
608             "%s %s", actionExecutionContext.getMetadataHandler(), metadataHandler);
609         prepareScheduleExecuteAndCompleteAction(action, actionExecutionContext, actionStartTime);
610         return new ActionExecutionValue(
611             metadataHandler.getOutputData(), metadataHandler.getAdditionalOutputData());
612       } finally {
613         profiler.completeTask(ProfilerTask.ACTION);
614       }
615     }
616   }

618   private void createOutputDirectories(Action action) throws ActionExecutionException {
619     try {
620       Set<Path> done = new HashSet<>(); // avoid redundant calls for the same directory.
621       for (Artifact outputFile : action.getOutputs()) {
622         Path outputDir = outputFile.getPath().getParentDirectory();
623         if (done.add(outputDir)) {
624           try {
625             createDirectoryAndParents(outputDir);
626             continue;
627           } catch (IOException e) {
628             /* Fall through to plan B. */
629           }
630 
631           // Possibly some direct ancestors are not directories.  In that case, we unlink all the
632           // ancestors until we reach a directory, then try again. This handles the case where a
633           // file becomes a directory, either from one build to another, or within a single build.
634           //
635           // Symlinks should not be followed so in order to clean up symlinks pointing to Fileset
636           // outputs from previous builds. See bug [incremental build of Fileset fails if
637           // Fileset.out was changed to be a subdirectory of the old value].
638           try {
639             for (Path p = outputDir; !p.isDirectory(Symlinks.NOFOLLOW);
640                 p = p.getParentDirectory()) {
641               // p may be a file or dangling symlink, or a symlink to an old Fileset output
642               p.delete(); // throws IOException
643             }
644             createDirectoryAndParents(outputDir);
645           } catch (IOException e) {
646             throw new ActionExecutionException(
647                 "failed to create output directory '" + outputDir + "'", e, action, false);
648           }
649         }
650       }
651     } catch (ActionExecutionException ex) {
652       printError(ex.getMessage(), action, null);
653       throw ex;
654     }
655   }

668   /**
669    * Prepare, schedule, execute, and then complete the action.
670    * When this function is called, we know that this action needs to be executed.
671    * This function will prepare for the action's execution (i.e. delete the outputs);
672    * schedule its execution; execute the action;
673    * and then do some post-execution processing to complete the action:
674    * set the outputs readonly and executable, and insert the action results in the
675    * action cache.
676    *
677    * @param action  The action to execute
678    * @param context services in the scope of the action
679    * @param actionStartTime time when we started the first phase of the action execution.
680    * @throws ActionExecutionException if the execution of the specified action
681    *   failed for any reason.
682    * @throws InterruptedException if the thread was interrupted.
683    */
684   private void prepareScheduleExecuteAndCompleteAction(Action action,
685       ActionExecutionContext context, long actionStartTime)
686       throws ActionExecutionException, InterruptedException {
687     // Delete the metadataHandler's cache of the action's outputs, since they are being deleted.
688     context.getMetadataHandler().discardOutputMetadata();
689     // Delete the outputs before executing the action, just to ensure that
690     // the action really does produce the outputs.
691     try {
692       action.prepare(context.getExecutor().getExecRoot());
693     } catch (IOException e) {
694       reportError("failed to delete output files before executing action", e, action, null);
695     }
696 
697     postEvent(new ActionStartedEvent(action, actionStartTime));
698     ResourceSet estimate = action.estimateResourceConsumption(executorEngine);
699     ActionExecutionStatusReporter statusReporter = statusReporterRef.get();
700     try {
701       if (estimate == null || estimate == ResourceSet.ZERO) {
702         statusReporter.setRunningFromBuildData(action);
703       } else {
704         // If estimated resource consumption is null, action will manually call
705         // resource manager when it knows what resources are needed.
706         resourceManager.acquireResources(action, estimate);
707       }
708       boolean outputDumped = executeActionTask(action, context);
709       completeAction(action, context.getMetadataHandler(),
710           context.getFileOutErr(), outputDumped);
711     } finally {
712       if (estimate != null) {
713         resourceManager.releaseResources(action, estimate);
714       }
715       statusReporter.remove(action);
716       postEvent(new ActionCompletionEvent(actionStartTime, action));
717     }
718   }

720   private ActionExecutionException processAndThrow(
721       ActionExecutionException e, Action action, FileOutErr outErrBuffer)
722       throws ActionExecutionException {
723     reportActionExecution(action, e, outErrBuffer);
724     boolean reported = reportErrorIfNotAbortingMode(e, outErrBuffer);
725 
726     ActionExecutionException toThrow = e;
727     if (reported){
728       // If we already printed the error for the exception we mark it as already reported
729       // so that we do not print it again in upper levels.
730       // Note that we need to report it here since we want immediate feedback of the errors
731       // and in some cases the upper-level printing mechanism only prints one of the errors.
732       toThrow = new AlreadyReportedActionExecutionException(e);
733     }
734 
735     // Now, rethrow the exception.
736     // This can have two effects:
737     // If we're still building, the exception will get retrieved by the
738     // completor and rethrown.
739     // If we're aborting, the exception will never be retrieved from the
740     // completor, since the completor is waiting for all outstanding jobs
741     // to finish. After they have finished, it will only rethrow the
742     // exception that initially caused it to abort will and not check the
743     // exit status of any actions that had finished in the meantime.
744     throw toThrow;
745   }

747   /**
748    * Execute the specified action, in a profiler task.
749    * The caller is responsible for having already checked that we need to
750    * execute it and for acquiring/releasing any scheduling locks needed.
751    *
752    * <p>This is thread-safe so long as you don't try to execute the same action
753    * twice at the same time (or overlapping times).
754    * May execute in a worker thread.
755    *
756    * @throws ActionExecutionException if the execution of the specified action
757    *   failed for any reason.
758    * @throws InterruptedException if the thread was interrupted.
759    * @return true if the action output was dumped, false otherwise.
760    */
761   private boolean executeActionTask(Action action, ActionExecutionContext actionExecutionContext)
762       throws ActionExecutionException, InterruptedException {
763     profiler.startTask(ProfilerTask.ACTION_EXECUTE, action);
764     // ActionExecutionExceptions that occur as the thread is interrupted are
765     // assumed to be a result of that, so we throw InterruptedException
766     // instead.
767     FileOutErr outErrBuffer = actionExecutionContext.getFileOutErr();
768     try {
769       action.execute(actionExecutionContext);
770 
771       // Action terminated fine, now report the output.
772       // The .showOutput() method is not necessarily a quick check: in its
773       // current implementation it uses regular expression matching.
774       if (outErrBuffer.hasRecordedOutput()
775           && (action.showsOutputUnconditionally()
776           || reporter.showOutput(Label.print(action.getOwner().getLabel())))) {
777         dumpRecordedOutErr(action, outErrBuffer);
778         return true;
779       }
780       // Defer reporting action success until outputs are checked
781     } catch (ActionExecutionException e) {
782       processAndThrow(e, action, outErrBuffer);
783     } finally {
784       profiler.completeTask(ProfilerTask.ACTION_EXECUTE);
785     }
786     return false;
787   }

789   private void completeAction(Action action, MetadataHandler metadataHandler,
790       FileOutErr fileOutErr, boolean outputAlreadyDumped) throws ActionExecutionException {
791     try {
792       Preconditions.checkState(action.inputsKnown(),
793           "Action %s successfully executed, but inputs still not known", action);
794 
795       profiler.startTask(ProfilerTask.ACTION_COMPLETE, action);
796       try {
797         if (!checkOutputs(action, metadataHandler)) {
798           reportError("not all outputs were created", null, action,
799               outputAlreadyDumped ? null : fileOutErr);
800         }
801         // Prevent accidental stomping on files.
802         // This will also throw a FileNotFoundException
803         // if any of the output files doesn't exist.
804         try {
805           setOutputsReadOnlyAndExecutable(action, metadataHandler);
806         } catch (IOException e) {
807           reportError("failed to set outputs read-only", e, action, null);
808         }
809       } finally {
810         profiler.completeTask(ProfilerTask.ACTION_COMPLETE);
811       }
812       reportActionExecution(action, null, fileOutErr);
813     } catch (ActionExecutionException actionException) {
814       // Success in execution but failure in completion.
815       reportActionExecution(action, actionException, fileOutErr);
816       throw actionException;
817     } catch (IllegalStateException exception) {
818       // More serious internal error, but failure still reported.
819       reportActionExecution(action,
820           new ActionExecutionException(exception, action, true), fileOutErr);
821       throw exception;
822     }
823   }

825   /**
826    * For each of the action's outputs that is a regular file (not a symbolic
827    * link or directory), make it read-only and executable.
828    *
829    * <p>Making the outputs read-only helps preventing accidental editing of
830    * them (e.g. in case of generated source code), while making them executable
831    * helps running generated files (such as generated shell scripts) on the
832    * command line.
833    *
834    * <p>May execute in a worker thread.
835    *
836    * <p>Note: setting these bits maintains transparency regarding the locality of the build;
837    * because the remote execution engine sets them, they should be set for local builds too.
838    *
839    * @throws IOException if an I/O error occurred.
840    */
841   private final void setOutputsReadOnlyAndExecutable(Action action, MetadataHandler metadataHandler)
842       throws IOException {
843     Preconditions.checkState(!action.getActionType().isMiddleman());
844 
845     for (Artifact output : action.getOutputs()) {
846       Path path = output.getPath();
847       if (metadataHandler.isInjected(output)) {
848         // We trust the files created by the execution-engine to be non symlinks with expected
849         // chmod() settings already applied. The follow stanza implies a total of 6 system calls,
850         // since the UnixFileSystem implementation of setWritable() and setExecutable() both
851         // do a stat() internally.
852         continue;
853       }
854       if (path.isFile(Symlinks.NOFOLLOW)) { // i.e. regular files only.
855         path.setWritable(false);
856         path.setExecutable(true);
857       }
858     }
859   }

861   private void reportMissingOutputFile(Action action, Artifact output, Reporter reporter,
862       boolean isSymlink) {
863     boolean genrule = action.getMnemonic().equals("Genrule");
864     String prefix = (genrule ? "declared output '" : "output '") + output.prettyPrint() + "' ";
865     if (isSymlink) {
866       reporter.handle(Event.error(
867           action.getOwner().getLocation(), prefix + "is a dangling symbolic link"));
868     } else {
869       String suffix = genrule ? " by genrule. This is probably "
870           + "because the genrule actually didn't create this output, or because the output was a "
871           + "directory and the genrule was run remotely (note that only the contents of "
872           + "declared file outputs are copied from genrules run remotely)" : "";
873       reporter.handle(Event.error(
874           action.getOwner().getLocation(), prefix + "was not created" + suffix));
875     }
876   }

878   /**
879    * Validates that all action outputs were created or intentionally omitted.
880    *
881    * @return false if some outputs are missing, true - otherwise.
882    */
883   private boolean checkOutputs(Action action, MetadataHandler metadataHandler) {
884     boolean success = true;
885     for (Artifact output : action.getOutputs()) {
886       // artifactExists has the side effect of potentially adding the artifact to the cache,
887       // therefore we only call it if we know the artifact is indeed not omitted to avoid any
888       // unintended side effects.
889       if (!(metadataHandler.artifactOmitted(output) || metadataHandler.artifactExists(output))) {
890         reportMissingOutputFile(action, output, reporter, output.getPath().isSymbolicLink());
891         success = false;
892       }
893     }
894     return success;
895   }

904   /**
905    * Convenience function for reporting that the action failed due to a
906    * the exception cause, if there is an additional explanatory message that
907    * clarifies the message of the exception. Combines the user-provided message
908    * and the exceptions' message and reports the combination as error.
909    * Then, throws an ActionExecutionException with the reported error as
910    * message and the provided exception as the cause.
911    *
912    * @param message A small text that explains why the action failed
913    * @param cause The exception that caused the action to fail
914    * @param action The action that failed
915    * @param actionOutput The output of the failed Action.
916    *     May be null, if there is no output to display
917    */
918   private void reportError(String message, Throwable cause, Action action, FileOutErr actionOutput)
919       throws ActionExecutionException {
920     ActionExecutionException ex;
921     if (cause == null) {
922       ex = new ActionExecutionException(message, action, false);
923     } else {
924       ex = new ActionExecutionException(message, cause, action, false);
925     }
926     printError(ex.getMessage(), action, actionOutput);
927     throw ex;
928   }

930   /**
931    * For the action 'action' that failed due to 'ex' with the output
932    * 'actionOutput', notify the user about the error. To notify the user, the
933    * method first displays the output of the action and then reports an error
934    * via the reporter. The method ensures that the two messages appear next to
935    * each other by locking the outErr object where the output is displayed.
936    *
937    * @param message The reason why the action failed
938    * @param action The action that failed, must not be null.
939    * @param actionOutput The output of the failed Action.
940    *     May be null, if there is no output to display
941    */
942   private void printError(String message, Action action, FileOutErr actionOutput) {
943     synchronized (reporter) {
944       if (actionOutput != null && actionOutput.hasRecordedOutput()) {
945         dumpRecordedOutErr(action, actionOutput);
946       }
947       if (keepGoing) {
948         message = "Couldn't " + describeAction(action) + ": " + message;
949       }
950       reporter.handle(Event.error(action.getOwner().getLocation(), message));
951       recordExecutionError();
952     }
953   }

955   /** Describe an action, for use in error messages. */
956   private static String describeAction(Action action) {
957     if (action.getOutputs().isEmpty()) {
958       return "run " + action.prettyPrint();
959     } else if (action.getActionType().isMiddleman()) {
960       return "build " + action.prettyPrint();
961     } else {
962       return "build file " + action.getPrimaryOutput().prettyPrint();
963     }

966   /**
967    * Dump the output from the action.
968    *
969    * @param action The action whose output is being dumped
970    * @param outErrBuffer The OutErr that recorded the actions output
971    */
972   private void dumpRecordedOutErr(Action action, FileOutErr outErrBuffer) {
973     StringBuilder message = new StringBuilder("");
974     message.append("From ");
975     message.append(action.describe());
976     message.append(":");
977 
978     // Synchronize this on the reporter, so that the output from multiple
979     // actions will not be interleaved.
980     synchronized (reporter) {
981       // Only print the output if we're not winding down.
982       if (isBuilderAborting()) {
983         return;
984       }
985       reporter.handle(Event.info(message.toString()));
986 
987       OutErr outErr = this.reporter.getOutErr();
988       outErrBuffer.dumpOutAsLatin1(outErr.getOutputStream());
989       outErrBuffer.dumpErrAsLatin1(outErr.getErrorStream());
990     }
991   }

993   private void reportActionExecution(Action action,
994       ActionExecutionException exception, FileOutErr outErr) {
995     String stdout = null;
996     String stderr = null;
997 
998     if (outErr.hasRecordedStdout()) {
999       stdout = outErr.getOutputFile().toString();
1000     }
1001     if (outErr.hasRecordedStderr()) {
1002       stderr = outErr.getErrorFile().toString();
1003     }
1004     postEvent(new ActionExecutedEvent(action, exception, stdout, stderr));
1005   }

1007   /**
1008    * Returns true if the exception was reported. False otherwise. Currently this is a copy of what
1009    * we did in pre-Skyframe execution. The main implication is that we are printing the error to the
1010    * top level reporter instead of the action reporter. Because of that Skyframe values do not know
1011    * about the errors happening in the execution phase. Even if we change in the future to log to
1012    * the action reporter (that would be done in ActionExecutionFunction.compute() when we get an
1013    * ActionExecutionException), we probably do not want to also store the StdErr output, so
1014    * dumpRecordedOutErr() should still be called here.
1015    */
1016   private boolean reportErrorIfNotAbortingMode(ActionExecutionException ex,
1017       FileOutErr outErrBuffer) {
1018     // For some actions (e.g. many local actions) the pollInterruptedStatus()
1019     // won't notice that we had an interrupted job. It will continue.
1020     // For that reason we must take care to NOT report errors if we're
1021     // in the 'aborting' mode: Any cancelled action would show up here.
1022     // For some actions (e.g. many local actions) the pollInterruptedStatus()
1023     // won't notice that we had an interrupted job. It will continue.
1024     // For that reason we must take care to NOT report errors if we're
1025     // in the 'aborting' mode: Any cancelled action would show up here.
1026     synchronized (this.reporter) {
1027       if (!isBuilderAborting()) {
1028         // Oops. The action aborted. Report the problem.
1029         printError(ex.getMessage(), ex.getAction(), outErrBuffer);
1030         return true;
1031       }
1032     }
1033     return false;
1034   }
