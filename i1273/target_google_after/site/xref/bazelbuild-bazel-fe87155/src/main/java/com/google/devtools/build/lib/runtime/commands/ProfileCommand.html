126     return new InfoListener() {

144   @Override
145   public ExitCode exec(final CommandEnvironment env, OptionsProvider options) {
146     final BlazeRuntime runtime = env.getRuntime();
147     ProfileOptions opts =
148         options.getOptions(ProfileOptions.class);
149 
150     if (!opts.vfsStats) {
151       opts.vfsStatsLimit = 0;
152     }
153 
154     currentPathMapping = new Function<String, String>() {
155       @Override
156       public String apply(String input) {
157         if (runtime.getWorkspaceName().isEmpty()) {
158           return input;
159         } else {
160           return input.substring(input.lastIndexOf("/" + runtime.getWorkspaceName()) + 1);
161         }
162       }
163     };
164 
165     PrintStream out = new PrintStream(env.getReporter().getOutErr().getOutputStream());
166     try {
167       env.getReporter().handle(Event.warn(
168           null, "This information is intended for consumption by Blaze developers"
169               + " only, and may change at any time.  Script against it at your own risk"));
170 
171       for (String name : options.getResidue()) {
172         Path profileFile = runtime.getWorkingDirectory().getRelative(name);
173         try {
174           ProfileInfo info = ProfileInfo.loadProfileVerbosely(
175               profileFile, getInfoListener(runtime));
176           if (opts.dumpMode != null) {
177             dumpProfile(runtime, info, out, opts.dumpMode);
178           } else if (opts.html) {
179             Path htmlFile =
180                 profileFile.getParentDirectory().getChild(profileFile.getBaseName() + ".html");
181 
182             env.getReporter().handle(Event.info("Creating HTML output in " + htmlFile));
183 
184             HtmlCreator.createHtml(
185                 info,
186                 htmlFile,
187                 getStatistics(runtime, info, opts),
188                 opts.htmlDetails,
189                 opts.htmlPixelsPerSecond);
190           } else {
191             createText(runtime, info, out, opts);
192           }
193         } catch (IOException e) {
194           env.getReporter().handle(Event.error(
195               null, "Failed to process file " + name + ": " + e.getMessage()));
196         }
197       }
198     } finally {
199       out.flush();
200     }
201     return ExitCode.SUCCESS;
202   }

204   private void createText(BlazeRuntime runtime, ProfileInfo info, PrintStream out,
205       ProfileOptions opts) {
206     List<ProfilePhaseStatistics> statistics = getStatistics(runtime, info, opts);
207 
208     for (ProfilePhaseStatistics stat : statistics) {
209       String title = stat.getTitle();
210 
211       if (!title.isEmpty()) {
212         out.println("\n=== " + title.toUpperCase() + " ===\n");
213       }
214       out.print(stat.getStatistics());
215     }
216   }

218   private List<ProfilePhaseStatistics> getStatistics(
219       BlazeRuntime runtime, ProfileInfo info, ProfileOptions opts) {
220     try {
221       ProfileInfo.aggregateProfile(info, getInfoListener(runtime));
222       runtime.getReporter().handle(Event.info("Analyzing relationships"));
223 
224       info.analyzeRelationships();
225 
226       List<ProfilePhaseStatistics> statistics = new ArrayList<>();
227 
228       // Print phase durations and total execution time
229       ByteArrayOutputStream byteOutput = new ByteArrayOutputStream();
230       PrintStream out = new PrintStream(byteOutput, false, "UTF-8");
231       long duration = 0;
232       for (ProfilePhase phase : ProfilePhase.values()) {
233         ProfileInfo.Task phaseTask = info.getPhaseTask(phase);
234         if (phaseTask != null) {
235           duration += info.getPhaseDuration(phaseTask);
236         }
237       }
238       for (ProfilePhase phase : ProfilePhase.values()) {
239         ProfileInfo.Task phaseTask = info.getPhaseTask(phase);
240         if (phaseTask != null) {
241           long phaseDuration = info.getPhaseDuration(phaseTask);
242           out.printf(THREE_COLUMN_FORMAT, "Total " + phase.nick + " phase time",
243               TimeUtilities.prettyTime(phaseDuration), prettyPercentage(phaseDuration, duration));
244         }
245       }
246       out.printf(THREE_COLUMN_FORMAT, "Total run time", TimeUtilities.prettyTime(duration),
247           "100.00%");
248       statistics.add(new ProfilePhaseStatistics("Phase Summary Information",
249           new String(byteOutput.toByteArray(), "UTF-8")));
250 
251       // Print details of major phases
252       if (duration > 0) {
253         statistics.add(formatInitPhaseStatistics(info, opts));
254         statistics.add(formatLoadingPhaseStatistics(info, opts));
255         statistics.add(formatAnalysisPhaseStatistics(info, opts));
256         ProfilePhaseStatistics stat = formatExecutionPhaseStatistics(info, opts);
257         if (stat != null) {
258           statistics.add(stat);
259         }
260       }
261 
262       return statistics;
263     } catch (UnsupportedEncodingException e) {
264       throw new AssertionError("Should not happen since, UTF8 is available on all JVMs");
265     }
266   }

268   private void dumpProfile(
269       BlazeRuntime runtime, ProfileInfo info, PrintStream out, String dumpMode) {
270     if (!dumpMode.contains("unsorted")) {
271       ProfileInfo.aggregateProfile(info, getInfoListener(runtime));
272     }
273     if (dumpMode.contains("raw")) {
274       for (ProfileInfo.Task task : info.allTasksById) {
275         dumpRaw(task, out);
276       }
277     } else if (dumpMode.contains("unsorted")) {
278       for (ProfileInfo.Task task : info.allTasksById) {
279         dumpTask(task, out, 0);
280       }
281     } else {
282       for (ProfileInfo.Task task : info.rootTasksById) {
283         dumpTask(task, out, 0);
284       }
285     }
286   }

288   private void dumpTask(ProfileInfo.Task task, PrintStream out, int indent) {
289     StringBuilder builder = new StringBuilder(String.format(
290         "\n%s %s\nThread: %-6d  Id: %-6d  Parent: %d\nStart time: %-12s   Duration: %s",
291         task.type, task.getDescription(), task.threadId, task.id, task.parentId,
292         TimeUtilities.prettyTime(task.startTime), TimeUtilities.prettyTime(task.duration)));
293     if (task.hasStats()) {
294       builder.append("\n");
295       ProfileInfo.AggregateAttr[] stats = task.getStatAttrArray();
296       for (ProfilerTask type : ProfilerTask.values()) {
297         ProfileInfo.AggregateAttr attr = stats[type.ordinal()];
298         if (attr != null) {
299           builder.append(type.toString().toLowerCase()).append("=(").
300               append(attr.count).append(", ").
301               append(TimeUtilities.prettyTime(attr.totalTime)).append(") ");
302         }
303       }
304     }
305     out.println(StringUtil.indent(builder.toString(), indent));
306     for (ProfileInfo.Task subtask : task.subtasks) {
307       dumpTask(subtask, out, indent + 1);
308     }
309   }

311   private void dumpRaw(ProfileInfo.Task task, PrintStream out) {
312     StringBuilder aggregateString = new StringBuilder();
313     ProfileInfo.AggregateAttr[] stats = task.getStatAttrArray();
314     for (ProfilerTask type : ProfilerTask.values()) {
315       ProfileInfo.AggregateAttr attr = stats[type.ordinal()];
316       if (attr != null) {
317         aggregateString.append(type.toString().toLowerCase()).append(",").
318             append(attr.count).append(",").append(attr.totalTime).append(" ");
319       }
320     }
321     out.println(
322         task.threadId + "|" + task.id + "|" + task.parentId + "|"
323         + task.startTime + "|" + task.duration + "|"
324         + aggregateString.toString().trim() + "|"
325         + task.type + "|" + task.getDescription());
326   }

328   /**
329    * Converts relative duration to the percentage string
330    * @return formatted percentage string or "N/A" if result is undefined.
331    */
332   private static String prettyPercentage(long duration, long total) {
333     if (total == 0) {
334       // Return "not available" string if total is 0 and result is undefined.
335       return "N/A";
336     }
337     return String.format("%5.2f%%", duration*100.0/total);
338   }

340   private void printCriticalPath(String title, PrintStream out, CriticalPathEntry path) {
341     out.printf("\n%s (%s):%n", title, TimeUtilities.prettyTime(path.cumulativeDuration));
342 
343     boolean lightCriticalPath = isLightCriticalPath(path);
344     out.println(lightCriticalPath ?
345         String.format("%6s %11s %8s   %s", "Id", "Time", "Percentage", "Description")
346         : String.format("%6s %11s %8s %8s   %s", "Id", "Time", "Share", "Critical", "Description"));
347 
348     long totalPathTime = path.cumulativeDuration;
349     int middlemanCount = 0;
350     long middlemanDuration = 0L;
351     long middlemanCritTime = 0L;
352 
353     for (; path != null ; path = path.next) {
354       if (path.task.id < 0) {
355         // Ignore fake actions.
356         continue;
357       } else if (path.task.getDescription().startsWith(MiddlemanAction.MIDDLEMAN_MNEMONIC + " ")
358           || path.task.getDescription().startsWith("TargetCompletionMiddleman")) {
359         // Aggregate middleman actions.
360         middlemanCount++;
361         middlemanDuration += path.duration;
362         middlemanCritTime += path.getCriticalTime();
363       } else {
364         String desc = path.task.getDescription().replace(':', ' ');
365         if (lightCriticalPath) {
366           out.printf("%6d %11s %8s   %s%n", path.task.id, TimeUtilities.prettyTime(path.duration),
367               prettyPercentage(path.duration, totalPathTime), desc);
368         } else {
369           out.printf("%6d %11s %8s %8s   %s%n", path.task.id,
370               TimeUtilities.prettyTime(path.duration),
371               prettyPercentage(path.duration, totalPathTime),
372               prettyPercentage(path.getCriticalTime(), totalPathTime), desc);
373         }
374       }
375     }
376     if (middlemanCount > 0) {
377       if (lightCriticalPath) {
378         out.printf("       %11s %8s   [%d middleman actions]%n",
379             TimeUtilities.prettyTime(middlemanDuration),
380             prettyPercentage(middlemanDuration, totalPathTime), middlemanCount);
381       } else {
382         out.printf("       %11s %8s %8s   [%d middleman actions]%n",
383             TimeUtilities.prettyTime(middlemanDuration),
384             prettyPercentage(middlemanDuration, totalPathTime),
385             prettyPercentage(middlemanCritTime, totalPathTime), middlemanCount);
386       }
387     }
388   }

394   private void printShortPhaseAnalysis(ProfileInfo info, PrintStream out, ProfilePhase phase) {
395     ProfileInfo.Task phaseTask = info.getPhaseTask(phase);
396     if (phaseTask != null) {
397       long phaseDuration = info.getPhaseDuration(phaseTask);
398       out.printf(TWO_COLUMN_FORMAT, "Total " + phase.nick + " phase time",
399           TimeUtilities.prettyTime(phaseDuration));
400       printTimeDistributionByType(info, out, phaseTask);
401     }

404   private void printTimeDistributionByType(ProfileInfo info, PrintStream out,
405       ProfileInfo.Task phaseTask) {
406     List<ProfileInfo.Task> taskList = info.getTasksForPhase(phaseTask);
407     long phaseDuration = info.getPhaseDuration(phaseTask);
408     long totalDuration = phaseDuration;
409     for (ProfileInfo.Task task : taskList) {
410       // Tasks on the phaseTask thread already accounted for in the phaseDuration.
411       if (task.threadId != phaseTask.threadId) {
412         totalDuration += task.duration;
413       }
414     }
415     boolean headerNeeded = true;
416     for (ProfilerTask type : ProfilerTask.values()) {
417       ProfileInfo.AggregateAttr stats = info.getStatsForType(type, taskList);
418       if (stats.count > 0 && stats.totalTime > 0) {
419         if (headerNeeded) {
420           out.println("\nTotal time (across all threads) spent on:");
421           out.printf("%18s %8s %8s %11s%n", "Type", "Total", "Count", "Average");
422           headerNeeded = false;
423         }
424         out.printf("%18s %8s %8d %11s%n", type.toString(),
425             prettyPercentage(stats.totalTime, totalDuration), stats.count,
426             TimeUtilities.prettyTime(stats.totalTime / stats.count));
427       }
428     }
429   }

431   static class Stat implements Comparable<Stat> {
432     public long duration;
433     public long frequency;
434 
435     @Override
436     public int compareTo(Stat o) {
437       return this.duration == o.duration ? Long.compare(this.frequency, o.frequency)
438           : Long.compare(this.duration, o.duration);
439     }
440   }

442   /**
443    * Print the time spent on VFS operations on each path. Output is grouped by operation and sorted
444    * by descending duration. If multiple of the same VFS operation were logged for the same path,
445    * print the total duration.
446    *
447    * @param info profiling data.
448    * @param out output stream.
449    * @param phase build phase.
450    * @param limit maximum number of statistics to print, or -1 for no limit.
451    */
452   private void printVfsStatistics(ProfileInfo info, PrintStream out,
453                                   ProfilePhase phase, int limit) {
454     ProfileInfo.Task phaseTask = info.getPhaseTask(phase);
455     if (phaseTask == null) {
456       return;
457     }
458 
459     if (limit == 0) {
460       return;
461     }
462 
463     // Group into VFS operations and build maps from path to duration.
464 
465     List<ProfileInfo.Task> taskList = info.getTasksForPhase(phaseTask);
466     EnumMap<ProfilerTask, Map<String, Stat>> stats = Maps.newEnumMap(ProfilerTask.class);
467 
468     collectVfsEntries(stats, taskList);
469 
470     if (!stats.isEmpty()) {
471       out.printf("\nVFS path statistics:\n");
472       out.printf("%15s %10s %10s %s\n", "Type", "Frequency", "Duration", "Path");
473     }
474 
475     // Reverse the maps to get maps from duration to path. We use a TreeMultimap to sort by duration
476     // and because durations are not unique.
477 
478     for (ProfilerTask type : stats.keySet()) {
479       Map<String, Stat> statsForType = stats.get(type);
480       TreeMultimap<Stat, String> sortedStats =
481           TreeMultimap.create(Ordering.natural().reverse(), Ordering.natural());
482 
483       Multimaps.invertFrom(Multimaps.forMap(statsForType), sortedStats);
484 
485       int numPrinted = 0;
486       for (Map.Entry<Stat, String> stat : sortedStats.entries()) {
487         if (limit != -1 && numPrinted++ == limit) {
488           out.printf("... %d more ...\n", sortedStats.size() - limit);
489           break;
490         }
491         out.printf("%15s %10d %10s %s\n",
492             type.name(), stat.getKey().frequency, TimeUtilities.prettyTime(stat.getKey().duration),
493             stat.getValue());
494       }
495     }
496   }
497 
498   private void collectVfsEntries(EnumMap<ProfilerTask, Map<String, Stat>> stats,
499       List<ProfileInfo.Task> taskList) {
500     for (ProfileInfo.Task task : taskList) {
501       collectVfsEntries(stats, Arrays.asList(task.subtasks));
502       if (!task.type.name().startsWith("VFS_")) {
503         continue;
504       }
505 
506       Map<String, Stat> statsForType = stats.get(task.type);
507       if (statsForType == null) {
508         statsForType = Maps.newHashMap();
509         stats.put(task.type, statsForType);
510       }
511 
512       String path = currentPathMapping.apply(task.getDescription());
513 
514       Stat stat = statsForType.get(path);
515       if (stat == null) {
516         stat = new Stat();
517       }
518 
519       stat.duration += task.duration;
520       stat.frequency++;
521       statsForType.put(path, stat);
522     }
523   }

525   /**
526    * Returns set of profiler tasks to be filtered from critical path.
527    * Also always filters out ACTION_LOCK and WAIT tasks to simulate
528    * unlimited resource critical path (see comments inside formatExecutionPhaseStatistics()
529    * method).
530    */
531   private EnumSet<ProfilerTask> getTypeFilter(ProfilerTask... tasks) {
532     EnumSet<ProfilerTask> filter = EnumSet.of(ProfilerTask.ACTION_LOCK, ProfilerTask.WAIT);
533     Collections.addAll(filter, tasks);
534     return filter;
535   }

567   private ProfilePhaseStatistics formatExecutionPhaseStatistics(ProfileInfo info,
568                                                                 ProfileOptions opts)
569       throws UnsupportedEncodingException {
570     ByteArrayOutputStream byteOutput = new ByteArrayOutputStream();
571     PrintStream out = new PrintStream(byteOutput, false, "UTF-8");
572 
573     ProfileInfo.Task prepPhase = info.getPhaseTask(ProfilePhase.PREPARE);
574     ProfileInfo.Task execPhase = info.getPhaseTask(ProfilePhase.EXECUTE);
575     ProfileInfo.Task finishPhase = info.getPhaseTask(ProfilePhase.FINISH);
576     if (execPhase == null) {
577       return null;
578     }
579 
580     List<ProfileInfo.Task> execTasks = info.getTasksForPhase(execPhase);
581     long graphTime = info.getStatsForType(ProfilerTask.ACTION_GRAPH, execTasks).totalTime;
582     long execTime = info.getPhaseDuration(execPhase) - graphTime;
583 
584     if (prepPhase != null) {
585       out.printf(TWO_COLUMN_FORMAT, "Total preparation time",
586           TimeUtilities.prettyTime(info.getPhaseDuration(prepPhase)));
587     }
588     out.printf(TWO_COLUMN_FORMAT, "Total execution phase time",
589         TimeUtilities.prettyTime(info.getPhaseDuration(execPhase)));
590     if (finishPhase != null) {
591       out.printf(TWO_COLUMN_FORMAT, "Total time finalizing build",
592           TimeUtilities.prettyTime(info.getPhaseDuration(finishPhase)));
593     }
594     out.println();
595     out.printf(TWO_COLUMN_FORMAT, "Action dependency map creation",
596         TimeUtilities.prettyTime(graphTime));
597     out.printf(TWO_COLUMN_FORMAT, "Actual execution time",
598         TimeUtilities.prettyTime(execTime));
599 
600     EnumSet<ProfilerTask> typeFilter = EnumSet.noneOf(ProfilerTask.class);
601     CriticalPathEntry totalPath = info.getCriticalPath(typeFilter);
602     info.analyzeCriticalPath(typeFilter, totalPath);
603 
604     typeFilter = getTypeFilter();
605     CriticalPathEntry optimalPath = info.getCriticalPath(typeFilter);
606     info.analyzeCriticalPath(typeFilter, optimalPath);
607 
608     if (totalPath != null) {
609       printCriticalPathTimingBreakdown(info, totalPath, optimalPath, execTime, out);
610     } else {
611       out.println("\nCritical path not available because no action graph was generated.");
612     }
613 
614     printTimeDistributionByType(info, out, execPhase);
615 
616     if (totalPath != null) {
617       printCriticalPath("Critical path", out, totalPath);
618       // In light critical path we do not record scheduling delay data so it does not make sense
619       // to differentiate it.
620       if (!isLightCriticalPath(totalPath)) {
621         printCriticalPath("Critical path excluding scheduling delays", out, optimalPath);
622       }
623     }
624 
625     if (info.getMissingActionsCount() > 0) {
626       out.println("\n" + info.getMissingActionsCount() + " action(s) are present in the"
627           + " action graph but missing instrumentation data. Most likely profile file"
628           + " has been created for the failed or aborted build.");
629     }
630 
631     printVfsStatistics(info, out, ProfilePhase.EXECUTE, opts.vfsStatsLimit);
632 
633     return new ProfilePhaseStatistics("Execution Phase Information",
634         new String(byteOutput.toByteArray(), "UTF-8"));
635   }

637   void printCriticalPathTimingBreakdown(ProfileInfo info, CriticalPathEntry totalPath,
638       CriticalPathEntry optimalPath, long execTime, PrintStream out) {
639     Preconditions.checkNotNull(totalPath);
640     Preconditions.checkNotNull(optimalPath);
641     // TODO(bazel-team): Print remote vs build stats recorded by CriticalPathStats
642     if (isLightCriticalPath(totalPath)) {
643       return;
644     }
645     out.println(totalPath.task.type);
646     // Worker thread pool scheduling delays for the actual critical path.
647     long workerWaitTime = 0;
648     long mainThreadWaitTime = 0;
649     for (ProfileInfo.CriticalPathEntry entry = totalPath; entry != null; entry = entry.next) {
650       workerWaitTime += info.getActionWaitTime(entry.task);
651       mainThreadWaitTime += info.getActionQueueTime(entry.task);
652     }
653     out.printf(TWO_COLUMN_FORMAT, "Worker thread scheduling delays",
654         TimeUtilities.prettyTime(workerWaitTime));
655     out.printf(TWO_COLUMN_FORMAT, "Main thread scheduling delays",
656         TimeUtilities.prettyTime(mainThreadWaitTime));
657 
658     out.println("\nCritical path time:");
659     // Actual critical path.
660     long totalTime = totalPath.cumulativeDuration;
661     out.printf("%-37s %10s (%s of execution time)\n", "Actual time",
662         TimeUtilities.prettyTime(totalTime),
663         prettyPercentage(totalTime, execTime));
664     // Unlimited resource critical path. Essentially, we assume that if we
665     // remove all scheduling delays caused by resource semaphore contention,
666     // each action execution time would not change (even though load now would
667     // be substantially higher - so this assumption might be incorrect but it is
668     // still useful for modeling). Given those assumptions we calculate critical
669     // path excluding scheduling delays.
670     long optimalTime = optimalPath.cumulativeDuration;
671     out.printf("%-37s %10s (%s of execution time)\n", "Time excluding scheduling delays",
672         TimeUtilities.prettyTime(optimalTime),
673         prettyPercentage(optimalTime, execTime));
674 
675     // Artificial critical path if we ignore all the time spent in all tasks,
676     // except time directly attributed to the ACTION tasks.
677     out.println("\nTime related to:");
678 
679     EnumSet<ProfilerTask> typeFilter = EnumSet.allOf(ProfilerTask.class);
680     ProfileInfo.CriticalPathEntry path = info.getCriticalPath(typeFilter);
681     out.printf(TWO_COLUMN_FORMAT, "the builder overhead",
682         prettyPercentage(path.cumulativeDuration, totalTime));
683 
684     typeFilter = getTypeFilter();
685     for (ProfilerTask task : ProfilerTask.values()) {
686       if (task.name().startsWith("VFS_")) {
687         typeFilter.add(task);
688       }
689     }
690     path = info.getCriticalPath(typeFilter);
691     out.printf(TWO_COLUMN_FORMAT, "the VFS calls",
692         prettyPercentage(optimalTime - path.cumulativeDuration, optimalTime));
693 
694     typeFilter = getTypeFilter(ProfilerTask.ACTION_CHECK);
695     path = info.getCriticalPath(typeFilter);
696     out.printf(TWO_COLUMN_FORMAT, "the dependency checking",
697         prettyPercentage(optimalTime - path.cumulativeDuration, optimalTime));
698 
699     typeFilter = getTypeFilter(ProfilerTask.ACTION_EXECUTE);
700     path = info.getCriticalPath(typeFilter);
701     out.printf(TWO_COLUMN_FORMAT, "the execution setup",
702         prettyPercentage(optimalTime - path.cumulativeDuration, optimalTime));
703 
704     typeFilter = getTypeFilter(ProfilerTask.SPAWN, ProfilerTask.LOCAL_EXECUTION);
705     path = info.getCriticalPath(typeFilter);
706     out.printf(TWO_COLUMN_FORMAT, "local execution",
707         prettyPercentage(optimalTime - path.cumulativeDuration, optimalTime));
708 
709     typeFilter = getTypeFilter(ProfilerTask.SCANNER);
710     path = info.getCriticalPath(typeFilter);
711     out.printf(TWO_COLUMN_FORMAT, "the include scanner",
712         prettyPercentage(optimalTime - path.cumulativeDuration, optimalTime));
713 
714     typeFilter = getTypeFilter(ProfilerTask.REMOTE_EXECUTION, ProfilerTask.PROCESS_TIME,
715         ProfilerTask.LOCAL_PARSE,  ProfilerTask.UPLOAD_TIME,
716         ProfilerTask.REMOTE_QUEUE,  ProfilerTask.REMOTE_SETUP, ProfilerTask.FETCH);
717     path = info.getCriticalPath(typeFilter);
718     out.printf(TWO_COLUMN_FORMAT, "Remote execution (cumulative)",
719         prettyPercentage(optimalTime - path.cumulativeDuration, optimalTime));
720 
721     typeFilter = getTypeFilter( ProfilerTask.UPLOAD_TIME, ProfilerTask.REMOTE_SETUP);
722     path = info.getCriticalPath(typeFilter);
723     out.printf(TWO_COLUMN_FORMAT, "  file uploads",
724         prettyPercentage(optimalTime - path.cumulativeDuration, optimalTime));
725 
726     typeFilter = getTypeFilter(ProfilerTask.FETCH);
727     path = info.getCriticalPath(typeFilter);
728     out.printf(TWO_COLUMN_FORMAT, "  file fetching",
729         prettyPercentage(optimalTime - path.cumulativeDuration, optimalTime));
730 
731     typeFilter = getTypeFilter(ProfilerTask.PROCESS_TIME);
732     path = info.getCriticalPath(typeFilter);
733     out.printf(TWO_COLUMN_FORMAT, "  process time",
734         prettyPercentage(optimalTime - path.cumulativeDuration, optimalTime));
735 
736     typeFilter = getTypeFilter(ProfilerTask.REMOTE_QUEUE);
737     path = info.getCriticalPath(typeFilter);
738     out.printf(TWO_COLUMN_FORMAT, "  remote queueing",
739         prettyPercentage(optimalTime - path.cumulativeDuration, optimalTime));
740 
741     typeFilter = getTypeFilter(ProfilerTask.LOCAL_PARSE);
742     path = info.getCriticalPath(typeFilter);
743     out.printf(TWO_COLUMN_FORMAT, "  remote execution parse",
744         prettyPercentage(optimalTime - path.cumulativeDuration, optimalTime));
745 
746     typeFilter = getTypeFilter(ProfilerTask.REMOTE_EXECUTION);
747     path = info.getCriticalPath(typeFilter);
748     out.printf(TWO_COLUMN_FORMAT, "  other remote activities",
749         prettyPercentage(optimalTime - path.cumulativeDuration, optimalTime));
750   }
