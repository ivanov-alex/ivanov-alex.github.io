103     /**
104      * Adds any info plists specified in the given rule's {@code infoplist} attribute as well as
105      * from its {@code options} as inputs to this bundle's {@code Info.plist} (which is merged from
106      * any such added plists plus some additional information).
107      */
108     public Builder addInfoplistInputFromRule(RuleContext ruleContext) {
109       if (ruleContext.attributes().has("options", Type.LABEL)) {
110         OptionsProvider optionsProvider = ruleContext
111             .getPrerequisite("options", Mode.TARGET, OptionsProvider.class);
112         if (optionsProvider != null) {
113           infoplists.addAll(optionsProvider.getInfoplists());
114         }
115       }
116       Artifact infoplist = ruleContext.getPrerequisiteArtifact("infoplist", Mode.TARGET);
117       if (infoplist != null) {
118         infoplists.add(infoplist);
119       }
120       return this;
121     }

147     private static NestedSet<Artifact> nestedBundleContentArtifacts(Iterable<Bundling> bundles) {
148       NestedSetBuilder<Artifact> artifacts = NestedSetBuilder.stableOrder();
149       for (Bundling bundle : bundles) {
150         artifacts.addTransitive(bundle.getBundleContentArtifacts());
151       }
152       return artifacts.build();
153     }
154 
155     private NestedSet<Artifact> mergeZips(Optional<Artifact> actoolzipOutput) {
156       NestedSetBuilder<Artifact> mergeZipBuilder =
157           NestedSetBuilder.<Artifact>stableOrder()
158               .addAll(actoolzipOutput.asSet())
159               .addAll(
160                   Xcdatamodel.outputZips(
161                       Xcdatamodels.xcdatamodels(
162                           intermediateArtifacts, objcProvider.get(XCDATAMODEL))))
163               .addTransitive(objcProvider.get(MERGE_ZIP));
164       for (Artifact xibFile : objcProvider.get(XIB)) {
165         mergeZipBuilder.add(intermediateArtifacts.compiledXibFileZip(xibFile));
166       }
167       for (Artifact storyboard : objcProvider.get(STORYBOARD)) {
168         mergeZipBuilder.add(intermediateArtifacts.compiledStoryboardZip(storyboard));
169       }
170 
171       if (objcProvider.is(USES_SWIFT)) {
172         mergeZipBuilder.add(intermediateArtifacts.swiftFrameworksFileZip());
173       }
174 
175       return mergeZipBuilder.build();
176     }

178     private NestedSet<Artifact> bundleInfoplistInputs() {
179       if (objcProvider.hasAssetCatalogs()) {
180         infoplists.add(intermediateArtifacts.actoolPartialInfoplist());
181       }
182       return infoplists.build();
183     }
184 
185     private Optional<Artifact> bundleInfoplist(NestedSet<Artifact> bundleInfoplistInputs) {
186       if (bundleInfoplistInputs.isEmpty()) {
187         return Optional.absent();
188       }
189       if (needsToMerge(bundleInfoplistInputs, primaryBundleId, fallbackBundleId)) {
190         return Optional.of(intermediateArtifacts.mergedInfoplist());
191       }
192       return Optional.of(Iterables.getOnlyElement(bundleInfoplistInputs));
193     }
194 
195     private Optional<Artifact> combinedArchitectureBinary() {
196       Optional<Artifact> combinedArchitectureBinary = Optional.absent();
197       if (!Iterables.isEmpty(objcProvider.get(LIBRARY))
198           || !Iterables.isEmpty(objcProvider.get(IMPORTED_LIBRARY))) {
199         combinedArchitectureBinary =
200             Optional.of(intermediateArtifacts.combinedArchitectureBinary());
201       }
202       return combinedArchitectureBinary;
203     }
204 
205     private Optional<Artifact> actoolzipOutput() {
206       Optional<Artifact> actoolzipOutput = Optional.absent();
207       if (!Iterables.isEmpty(objcProvider.get(ASSET_CATALOG))) {
208         actoolzipOutput = Optional.of(intermediateArtifacts.actoolzipOutput());
209       }
210       return actoolzipOutput;
211     }

213     private NestedSet<BundleableFile> binaryStringsFiles() {
214       NestedSetBuilder<BundleableFile> binaryStringsBuilder = NestedSetBuilder.stableOrder();
215       for (Artifact stringsFile : objcProvider.get(STRINGS)) {
216         BundleableFile bundleFile =
217             new BundleableFile(
218                 intermediateArtifacts.convertedStringsFile(stringsFile),
219                 BundleableFile.flatBundlePath(stringsFile.getExecPath()));
220         binaryStringsBuilder.add(bundleFile);
221       }
222       return binaryStringsBuilder.build();
223     }

225     /**
226      * Filters files that would map to the same location in the bundle, adding only one copy to the
227      * set of files returned.
228      *
229      * <p>Files can have the same bundle path for various illegal reasons and errors are raised for
230      * that separately (see {@link BundleSupport#validateResources}). There are situations though
231      * where the same file exists multiple times (for example in multi-architecture builds) and
232      * would conflict when creating the bundle. In all these cases it shouldn't matter which one is
233      * included and this class will select the first one.
234      */
235     ImmutableList<BundleableFile> deduplicateByBundlePaths(
236         ImmutableList<BundleableFile> bundleFiles) {
237       ImmutableList.Builder<BundleableFile> deduplicated = ImmutableList.builder();
238       Set<String> bundlePaths = new HashSet<>();
239       for (BundleableFile bundleFile : bundleFiles) {
240         if (bundlePaths.add(bundleFile.getBundlePath())) {
241           deduplicated.add(bundleFile);
242         }
243       }
244       return deduplicated.build();
245     }

247     public Bundling build() {
248       Preconditions.checkNotNull(intermediateArtifacts, "intermediateArtifacts");
249 
250       NestedSet<Artifact> bundleInfoplistInputs = bundleInfoplistInputs();
251       Optional<Artifact> bundleInfoplist = bundleInfoplist(bundleInfoplistInputs);
252       Optional<Artifact> actoolzipOutput = actoolzipOutput();
253       Optional<Artifact> combinedArchitectureBinary = combinedArchitectureBinary();
254       NestedSet<BundleableFile> binaryStringsFiles = binaryStringsFiles();
255       NestedSet<Artifact> mergeZips = mergeZips(actoolzipOutput);
256 
257       bundleFilesBuilder.addAll(binaryStringsFiles).addAll(objcProvider.get(BUNDLE_FILE));
258       ImmutableList<BundleableFile> bundleFiles =
259           deduplicateByBundlePaths(bundleFilesBuilder.build());
260 
261       NestedSetBuilder<Artifact> bundleContentArtifactsBuilder =
262           NestedSetBuilder.<Artifact>stableOrder()
263               .addTransitive(nestedBundleContentArtifacts(objcProvider.get(NESTED_BUNDLE)))
264               .addAll(combinedArchitectureBinary.asSet())
265               .addAll(bundleInfoplist.asSet())
266               .addTransitive(mergeZips)
267               .addAll(BundleableFile.toArtifacts(binaryStringsFiles))
268               .addAll(BundleableFile.toArtifacts(bundleFiles));
269 
270       return new Bundling(
271           name,
272           bundleDirFormat,
273           combinedArchitectureBinary,
274           bundleFiles,
275           bundleInfoplist,
276           actoolzipOutput,
277           bundleContentArtifactsBuilder.build(),
278           mergeZips,
279           primaryBundleId,
280           fallbackBundleId,
281           architecture,
282           minimumOsVersion,
283           bundleInfoplistInputs,
284           objcProvider.get(NESTED_BUNDLE));
285     }

288   private static boolean needsToMerge(
289       NestedSet<Artifact> bundleInfoplistInputs, String primaryBundleId, String fallbackBundleId) {
290     return primaryBundleId != null || fallbackBundleId != null
291         || Iterables.size(bundleInfoplistInputs) > 1;
292   }

309   private Bundling(
310       String name,
311       String bundleDirFormat,
312       Optional<Artifact> combinedArchitectureBinary,
313       ImmutableList<BundleableFile> bundleFiles,
314       Optional<Artifact> bundleInfoplist,
315       Optional<Artifact> actoolzipOutput,
316       NestedSet<Artifact> bundleContentArtifacts,
317       NestedSet<Artifact> mergeZips,
318       String primaryBundleId,
319       String fallbackBundleId,
320       String architecture,
321       String minimumOsVersion,
322       NestedSet<Artifact> bundleInfoplistInputs,
323       NestedSet<Bundling> nestedBundlings) {
324     this.nestedBundlings = Preconditions.checkNotNull(nestedBundlings);
325     this.name = Preconditions.checkNotNull(name);
326     this.bundleDirFormat = Preconditions.checkNotNull(bundleDirFormat);
327     this.combinedArchitectureBinary = Preconditions.checkNotNull(combinedArchitectureBinary);
328     this.bundleFiles = Preconditions.checkNotNull(bundleFiles);
329     this.bundleInfoplist = Preconditions.checkNotNull(bundleInfoplist);
330     this.actoolzipOutput = Preconditions.checkNotNull(actoolzipOutput);
331     this.bundleContentArtifacts = Preconditions.checkNotNull(bundleContentArtifacts);
332     this.mergeZips = Preconditions.checkNotNull(mergeZips);
333     this.fallbackBundleId = fallbackBundleId;
334     this.primaryBundleId = primaryBundleId;
335     this.architecture = Preconditions.checkNotNull(architecture);
336     this.minimumOsVersion = Preconditions.checkNotNull(minimumOsVersion);
337     this.bundleInfoplistInputs = Preconditions.checkNotNull(bundleInfoplistInputs);
338   }

340   /**
341    * The bundle directory. For apps, this would be {@code "Payload/TARGET_NAME.app"}, which is where
342    * in the bundle zip archive every file is found, including the linked binary, nested bundles, and
343    * everything returned by {@link #getBundleFiles()}.
344    */

346     return String.format(bundleDirFormat, name);
347   }
