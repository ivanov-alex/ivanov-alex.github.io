42  /**
43   * A Skyframe function to look up and import a single Skylark extension.
44   */
45  public class SkylarkImportLookupFunction implements SkyFunction {
46  
47    private final RuleClassProvider ruleClassProvider;
48    private final PackageFactory packageFactory;
49  
50    public SkylarkImportLookupFunction(
51      RuleClassProvider ruleClassProvider, PackageFactory packageFactory) {
52      this.ruleClassProvider = ruleClassProvider;
53      this.packageFactory = packageFactory;
54    }
55  
56    @Override
57    public SkyValue compute(SkyKey skyKey, Environment env) throws SkyFunctionException,
58        InterruptedException {
59      PackageIdentifier arg = (PackageIdentifier) skyKey.argument();
60      PathFragment file = arg.getPackageFragment();
61      ASTFileLookupValue astLookupValue = null;
62      try {
63        SkyKey astLookupKey = ASTFileLookupValue.key(arg);
64        astLookupValue = (ASTFileLookupValue) env.getValueOrThrow(astLookupKey,
65            ErrorReadingSkylarkExtensionException.class, InconsistentFilesystemException.class);
66      } catch (ErrorReadingSkylarkExtensionException e) {
67        throw new SkylarkImportLookupFunctionException(SkylarkImportFailedException.errorReadingFile(
68            file, e.getMessage()));
69      } catch (InconsistentFilesystemException e) {
70        throw new SkylarkImportLookupFunctionException(e, Transience.PERSISTENT);
71      }
72      if (astLookupValue == null) {
73        return null;
74      }
75      if (astLookupValue.getAST() == null) {
76        // Skylark import files have to exist.
77        throw new SkylarkImportLookupFunctionException(SkylarkImportFailedException.noFile(file));
78      }
79  
80      Map<PathFragment, Extension> importMap = new HashMap<>();
81      ImmutableList.Builder<SkylarkFileDependency> fileDependencies = ImmutableList.builder();
82      BuildFileAST ast = astLookupValue.getAST();
83      // TODO(bazel-team): Refactor this code and PackageFunction to reduce code duplications.
84      for (Map.Entry<Location, PathFragment> entry : ast.getImports().entrySet()) {
85        try {
86          PathFragment importFile = entry.getValue();
87          // HACK: The prelude sometimes contains load() statements, which need to be resolved
88          // relative to the prelude file. However, we don't have a good way to tell "this should come
89          // from the main repository" in a load() statement, and we don't have a good way to tell if
90          // a load() statement comes from the prelude, since we just prepend those statements before
91          // the actual BUILD file. So we use this evil .endsWith() statement to figure it out.
92          RepositoryName repository =
93              entry.getKey().getPath().endsWith(ruleClassProvider.getPreludePath())
94                  ? PackageIdentifier.DEFAULT_REPOSITORY_NAME : arg.getRepository();
95          SkyKey importsLookupKey = SkylarkImportLookupValue.key(repository, file, importFile);
96          SkylarkImportLookupValue importsLookupValue;
97          importsLookupValue = (SkylarkImportLookupValue) env.getValueOrThrow(
98              importsLookupKey, ASTLookupInputException.class);
99          if (importsLookupValue != null) {
100           importMap.put(importFile, importsLookupValue.getEnvironmentExtension());
101           fileDependencies.add(importsLookupValue.getDependency());
102         }
103       } catch (ASTLookupInputException e) {
104         throw new SkylarkImportLookupFunctionException(e, Transience.PERSISTENT);
105       }
106     }
107     Label label = pathFragmentToLabel(arg.getRepository(), file, env);
108     if (env.valuesMissing()) {
109       // This means some imports are unavailable.
110       return null;
111     }
112 
113     if (ast.containsErrors()) {
114       throw new SkylarkImportLookupFunctionException(SkylarkImportFailedException.skylarkErrors(
115           file));
116     }
117 
118     // Skylark UserDefinedFunction-s in that file will share this function definition Environment,
119     // which will be frozen by the time it is returned by createExtension.
120     Extension extension =
121         createExtension(ast, file, importMap, env);
122 
123     return new SkylarkImportLookupValue(
124         extension, new SkylarkFileDependency(label, fileDependencies.build()));
125   }

127   /**
128    * Converts the PathFragment of the Skylark file to a Label using the BUILD file closest to the
129    * Skylark file in its directory hierarchy - finds the package to which the Skylark file belongs.
130    * Throws an exception if no such BUILD file exists.
131    */
132   private Label pathFragmentToLabel(RepositoryName repo, PathFragment file, Environment env)
133       throws SkylarkImportLookupFunctionException {
134     ContainingPackageLookupValue containingPackageLookupValue = null;
135     try {
136       PackageIdentifier newPkgId = new PackageIdentifier(repo, file.getParentDirectory());
137       containingPackageLookupValue = (ContainingPackageLookupValue) env.getValueOrThrow(
138           ContainingPackageLookupValue.key(newPkgId),
139           BuildFileNotFoundException.class, InconsistentFilesystemException.class);
140     } catch (BuildFileNotFoundException e) {
141       // Thrown when there are IO errors looking for BUILD files.
142       throw new SkylarkImportLookupFunctionException(e, Transience.PERSISTENT);
143     } catch (InconsistentFilesystemException e) {
144       throw new SkylarkImportLookupFunctionException(e, Transience.PERSISTENT);
145     }
146 
147     if (containingPackageLookupValue == null) {
148       return null;
149     }
150 
151     if (!containingPackageLookupValue.hasContainingPackage()) {
152       throw new SkylarkImportLookupFunctionException(
153           SkylarkImportFailedException.noBuildFile(file));
154     }
155 
156     PathFragment pkgName =
157         containingPackageLookupValue.getContainingPackageName().getPackageFragment();
158     PathFragment fileInPkg = file.relativeTo(pkgName);
159 
160     try {
161       // This code relies on PackageIdentifier.RepositoryName.toString()
162       return Label.parseAbsolute(repo + "//" + pkgName.getPathString() + ":" + fileInPkg);
163     } catch (LabelSyntaxException e) {
164       throw new IllegalStateException(e);
165     }
166   }

168   /**
169    * Creates the Extension to be imported.
170    */
171   private Extension createExtension(
172       BuildFileAST ast,
173       PathFragment file,
174       Map<PathFragment, Extension> importMap,
175       Environment env)
176           throws InterruptedException, SkylarkImportLookupFunctionException {
177     StoredEventHandler eventHandler = new StoredEventHandler();
178     // TODO(bazel-team): this method overestimates the changes which can affect the
179     // Skylark RuleClass. For example changes to comments or unused functions can modify the hash.
180     // A more accurate - however much more complicated - way would be to calculate a hash based on
181     // the transitive closure of the accessible AST nodes.
182     try (Mutability mutability = Mutability.create("importing %s", file)) {
183       com.google.devtools.build.lib.syntax.Environment extensionEnv =
184           ruleClassProvider.createSkylarkRuleClassEnvironment(
185               mutability, eventHandler, ast.getContentHashCode(), importMap)
186           .setupOverride("native", packageFactory.getNativeModule());
187       ast.exec(extensionEnv, eventHandler);
188       SkylarkRuleClassFunctions.exportRuleFunctions(extensionEnv, file);
189 
190       Event.replayEventsOn(env.getListener(), eventHandler.getEvents());
191       if (eventHandler.hasErrors()) {
192         throw new SkylarkImportLookupFunctionException(SkylarkImportFailedException.errors(file));
193       }
194       return new Extension(extensionEnv);
195     }
196   }

203   static final class SkylarkImportFailedException extends Exception {
204     private SkylarkImportFailedException(String errorMessage) {
205       super(errorMessage);
206     }
207 
208     static SkylarkImportFailedException errors(PathFragment file) {
209       return new SkylarkImportFailedException(
210           String.format("Extension file '%s' has errors", file));
211     }
212 
213     static SkylarkImportFailedException errorReadingFile(PathFragment file, String error) {
214       return new SkylarkImportFailedException(
215           String.format("Encountered error while reading extension file '%s': %s", file, error));
216     }
217 
218     static SkylarkImportFailedException noFile(PathFragment file) {
219       return new SkylarkImportFailedException(
220           String.format("Extension file not found: '%s'", file));
221     }
222 
223     static SkylarkImportFailedException noBuildFile(PathFragment file) {
224       return new SkylarkImportFailedException(
225           String.format("Every .bzl file must have a corresponding package, but '%s' "
226               + "does not have one. Please create a BUILD file in the same or any parent directory."
227               + " Note that this BUILD file does not need to do anything except exist.", file));
228     }
229 
230     static SkylarkImportFailedException skylarkErrors(PathFragment file) {
231       return new SkylarkImportFailedException(String.format("Extension '%s' has errors", file));
232     }
233   }
234 
235   private static final class SkylarkImportLookupFunctionException extends SkyFunctionException {
236     private SkylarkImportLookupFunctionException(SkylarkImportFailedException cause) {
237       super(cause, Transience.PERSISTENT);
238     }
239 
240     private SkylarkImportLookupFunctionException(InconsistentFilesystemException e,
241         Transience transience) {
242       super(e, transience);
243     }
244 
245     private SkylarkImportLookupFunctionException(ASTLookupInputException e,
246         Transience transience) {
247       super(e, transience);
248     }
249 
250     private SkylarkImportLookupFunctionException(BuildFileNotFoundException e,
251         Transience transience) {
252       super(e, transience);
253     }
254   }
