174   /**
175    * In native code, private values start with $.
176    * In Skylark, private values start with _, because of the grammar.
177    */
178   private static String attributeToNative(String oldName, Location loc, boolean isLateBound)
179       throws EvalException {
180     if (oldName.isEmpty()) {
181       throw new EvalException(loc, "Attribute name cannot be empty");
182     }
183     if (isLateBound) {
184       if (oldName.charAt(0) != '_') {
185         throw new EvalException(loc, "When an attribute value is a function, "
186             + "the attribute must be private (start with '_')");
187       }
188       return ":" + oldName.substring(1);
189     }
190     if (oldName.charAt(0) == '_') {
191       return "$" + oldName.substring(1);
192     }
193     return oldName;
194   }

198   @SkylarkSignature(name = "rule", doc =
199       "Creates a new rule. Store it in a global value, so that it can be loaded and called "
200       + "from BUILD files.",
201       returnType = BaseFunction.class,
202       mandatoryPositionals = {
203         @Param(name = "implementation", type = BaseFunction.class,
204             doc = "the function implementing this rule, must have exactly one parameter: "
205             + "<a href=\"ctx.html\">ctx</a>. The function is called during the analysis phase "
206             + "for each instance of the rule. It can access the attributes provided by the user. "
207             + "It must create actions to generate all the declared outputs.")
208       },
209       optionalPositionals = {
210         @Param(name = "test", type = Boolean.class, defaultValue = "False",
211             doc = "Whether this rule is a test rule. "
212             + "If True, the rule must end with <code>_test</code> (otherwise it must not), "
213             + "and there must be an action that generates <code>ctx.outputs.executable</code>."),
214         @Param(name = "attrs", type = Map.class, noneable = true, defaultValue = "None", doc =
215             "dictionary to declare all the attributes of the rule. It maps from an attribute name "
216             + "to an attribute object (see <a href=\"attr.html\">attr</a> module). "
217             + "Attributes starting with <code>_</code> are private, and can be used to add "
218             + "an implicit dependency on a label. The attribute <code>name</code> is implicitly "
219             + "added and must not be specified. Attributes <code>visibility</code>, "
220             + "<code>deprecation</code>, <code>tags</code>, <code>testonly</code>, and "
221             + "<code>features</code> are implicitly added and might be overriden."),
222             // TODO(bazel-team): need to give the types of these builtin attributes
223         @Param(name = "outputs", type = Map.class, callbackEnabled = true, noneable = true,
224             defaultValue = "None", doc = "outputs of this rule. "
225             + "It is a dictionary mapping from string to a template name. "
226             + "For example: <code>{\"ext\": \"%{name}.ext\"}</code>. <br>"
227             + "The dictionary key becomes an attribute in <code>ctx.outputs</code>. "
228             // TODO(bazel-team): Make doc more clear, wrt late-bound attributes.
229             + "It may also be a function (which receives <code>ctx.attr</code> as argument) "
230             + "returning such a dictionary."),
231         @Param(name = "executable", type = Boolean.class, defaultValue = "False",
232             doc = "whether this rule is marked as executable or not. If True, "
233             + "there must be an action that generates <code>ctx.outputs.executable</code>."),
234         @Param(name = "output_to_genfiles", type = Boolean.class, defaultValue = "False",
235             doc = "If true, the files will be generated in the genfiles directory instead of the "
236             + "bin directory. This is used for compatibility with existing rules."),
237        @Param(name = "fragments", type = SkylarkList.class, generic1 = String.class,
238            defaultValue = "[]",
239            doc = "List of names of configuration fragments that the rule requires "
240            + "in target configuration."),
241        @Param(name = "host_fragments", type = SkylarkList.class, generic1 = String.class,
242            defaultValue = "[]",
243            doc = "List of names of configuration fragments that the rule requires "
244            + "in host configuration.")},
245       useAst = true, useEnvironment = true)
246   private static final BuiltinFunction rule = new BuiltinFunction("rule") {
247     @SuppressWarnings({"rawtypes", "unchecked"}) // castMap produces
248     // an Attribute.Builder instead of a Attribute.Builder<?> but it's OK.
249     public BaseFunction invoke(BaseFunction implementation, Boolean test, Object attrs,
250         Object implicitOutputs, Boolean executable, Boolean outputToGenfiles, SkylarkList fragments,
251         SkylarkList hostFragments, FuncallExpression ast, Environment funcallEnv)
252         throws EvalException, ConversionException {
253       funcallEnv.checkLoadingPhase("rule", ast.getLocation());
254       RuleClassType type = test ? RuleClassType.TEST : RuleClassType.NORMAL;
255       RuleClass parent = test ? testBaseRule : (executable ? binaryBaseRule : baseRule);
256 
257       // We'll set the name later, pass the empty string for now.
258       RuleClass.Builder builder = new RuleClass.Builder("", type, true, parent);
259 
260       if (attrs != Runtime.NONE) {
261         for (Map.Entry<String, Attribute.Builder> attr :
262             castMap(attrs, String.class, Attribute.Builder.class, "attrs").entrySet()) {
263           Attribute.Builder<?> attrBuilder = (Attribute.Builder<?>) attr.getValue();
264           String attrName =
265               attributeToNative(attr.getKey(), ast.getLocation(), attrBuilder.hasLateBoundValue());
266           builder.addOrOverrideAttribute(attrBuilder.build(attrName));
267         }
268       }
269       if (executable || test) {
270         builder.addOrOverrideAttribute(
271             attr("$is_executable", BOOLEAN)
272                 .value(true)
273                 .nonconfigurable("Called from RunCommand.isExecutable, which takes a Target")
274                 .build());
275         builder.setOutputsDefaultExecutable();
276       }
277 
278       if (implicitOutputs != Runtime.NONE) {
279         if (implicitOutputs instanceof BaseFunction) {
280           BaseFunction func = (BaseFunction) implicitOutputs;
281           SkylarkCallbackFunction callback =
282               new SkylarkCallbackFunction(func, ast, funcallEnv);
283           builder.setImplicitOutputsFunction(
284               new SkylarkImplicitOutputsFunctionWithCallback(callback, ast.getLocation()));
285         } else {
286           builder.setImplicitOutputsFunction(
287               new SkylarkImplicitOutputsFunctionWithMap(
288                   ImmutableMap.copyOf(castMap(
289                       implicitOutputs, String.class, String.class,
290                       "implicit outputs of the rule class"))));
291         }
292       }
293 
294       if (outputToGenfiles) {
295         builder.setOutputToGenfiles();
296       }
297 
298       registerRequiredFragments(fragments, hostFragments, builder);
299 
300       builder.setConfiguredTargetFunction(implementation);
301       builder.setRuleDefinitionEnvironment(funcallEnv);
302       return new RuleFunction(builder, type);
303     }
304 
305     private void registerRequiredFragments(
306         SkylarkList fragments, SkylarkList hostFragments, RuleClass.Builder builder) {
307       Map<ConfigurationTransition, ImmutableSet<String>> map = new HashMap<>();
308       addFragmentsToMap(map, fragments, NONE); // NONE represents target configuration
309       addFragmentsToMap(map, hostFragments, HOST);
310 
311       builder.requiresConfigurationFragments(new SkylarkModuleNameResolver(), map);
312     }
313 
314     private void addFragmentsToMap(Map<ConfigurationTransition, ImmutableSet<String>> map,
315         SkylarkList fragments, ConfigurationTransition config) {
316       if (!fragments.isEmpty()) {
317         map.put(config, ImmutableSet.copyOf(castList(fragments, String.class)));
318       }
319     }

322   // This class is needed for testing
323   static final class RuleFunction extends BaseFunction {
324     // Note that this means that we can reuse the same builder.
325     // This is fine since we don't modify the builder from here.
326     private final RuleClass.Builder builder;
327     private final RuleClassType type;
328     private PathFragment skylarkFile;
329     private String ruleClassName;
330 
331     public RuleFunction(Builder builder, RuleClassType type) {
332       super("rule", FunctionSignature.KWARGS);
333       this.builder = builder;
334       this.type = type;
335     }
336 
337     @Override
338     @SuppressWarnings("unchecked") // the magic hidden $pkg_context variable is guaranteed
339     // to be a PackageContext
340     public Object call(Object[] args, FuncallExpression ast, Environment env)
341         throws EvalException, InterruptedException, ConversionException {
342       env.checkLoadingPhase(getName(), ast.getLocation());
343       try {
344         if (ruleClassName == null || skylarkFile == null) {
345           throw new EvalException(ast.getLocation(),
346               "Invalid rule class hasn't been exported by a Skylark file");
347         }
348         if (type == RuleClassType.TEST != TargetUtils.isTestRuleName(ruleClassName)) {
349           throw new EvalException(ast.getLocation(), "Invalid rule class name '" + ruleClassName
350               + "', test rule class names must end with '_test' and other rule classes must not");
351         }
352         RuleClass ruleClass = builder.build(ruleClassName);
353         PackageContext pkgContext = (PackageContext) env.lookup(PackageFactory.PKG_CONTEXT);
354         return RuleFactory.createAndAddRule(
355             pkgContext, ruleClass, (Map<String, Object>) args[0], ast, env);
356       } catch (InvalidRuleException | NameConflictException | NoSuchVariableException e) {
357         throw new EvalException(ast.getLocation(), e.getMessage());
358       }
359     }
360 
361     /**
362      * Export a RuleFunction from a Skylark file with a given name.
363      */
364     void export(PathFragment skylarkFile, String ruleClassName) {
365       this.skylarkFile = skylarkFile;
366       this.ruleClassName = ruleClassName;
367     }
368 
369     @VisibleForTesting
370     RuleClass.Builder getBuilder() {
371       return builder;
372     }
373   }

375   public static void exportRuleFunctions(Environment env, PathFragment skylarkFile) {
376     for (String name : env.getGlobals().getDirectVariableNames()) {
377       try {
378         Object value = env.lookup(name);
379         if (value instanceof RuleFunction) {
380           RuleFunction function = (RuleFunction) value;
381           if (function.skylarkFile == null) {
382             function.export(skylarkFile, name);
383           }
384         }
385       } catch (NoSuchVariableException e) {
386         throw new AssertionError(e);
387       }
388     }
389   }

423   @SkylarkSignature(name = "to_proto",
424       doc = "Creates a text message from the struct parameter. This method only works if all "
425           + "struct elements (recursively) are strings, ints, booleans, other structs or a "
426           + "list of these types. Quotes and new lines in strings are escaped. "
427           + "Examples:<br><pre class=language-python>"
428           + "struct(key=123).to_proto()\n# key: 123\n\n"
429           + "struct(key=True).to_proto()\n# key: true\n\n"
430           + "struct(key=[1, 2, 3]).to_proto()\n# key: 1\n# key: 2\n# key: 3\n\n"
431           + "struct(key='text').to_proto()\n# key: \"text\"\n\n"
432           + "struct(key=struct(inner_key='text')).to_proto()\n"
433           + "# key {\n#   inner_key: \"text\"\n# }\n\n"
434           + "struct(key=[struct(inner_key=1), struct(inner_key=2)]).to_proto()\n"
435           + "# key {\n#   inner_key: 1\n# }\n# key {\n#   inner_key: 2\n# }\n\n"
436           + "struct(key=struct(inner_key=struct(inner_inner_key='text'))).to_proto()\n"
437           + "# key {\n#    inner_key {\n#     inner_inner_key: \"text\"\n#   }\n# }\n</pre>",
438       objectType = SkylarkClassObject.class, returnType = String.class,
439       mandatoryPositionals = {
440         // TODO(bazel-team): shouldn't we accept any ClassObject?
441         @Param(name = "self", type = SkylarkClassObject.class,
442             doc = "this struct")},
443       useLocation = true)
444   private static final BuiltinFunction toProto = new BuiltinFunction("to_proto") {
445       public String invoke(SkylarkClassObject self, Location loc) throws EvalException {
446         StringBuilder sb = new StringBuilder();
447         printTextMessage(self, sb, 0, loc);
448         return sb.toString();
449       }
450 
451       private void printTextMessage(ClassObject object, StringBuilder sb,
452           int indent, Location loc) throws EvalException {
453         for (String key : object.getKeys()) {
454           printTextMessage(key, object.getValue(key), sb, indent, loc);
455         }
456       }
457 
458       private void printSimpleTextMessage(String key, Object value, StringBuilder sb,
459           int indent, Location loc, String container) throws EvalException {
460         if (value instanceof ClassObject) {
461           print(sb, key + " {", indent);
462           printTextMessage((ClassObject) value, sb, indent + 1, loc);
463           print(sb, "}", indent);
464         } else if (value instanceof String) {
465           print(sb, key + ": \"" + escape((String) value) + "\"", indent);
466         } else if (value instanceof Integer) {
467           print(sb, key + ": " + value, indent);
468         } else if (value instanceof Boolean) {
469           // We're relying on the fact that Java converts Booleans to Strings in the same way
470           // as the protocol buffers do.
471           print(sb, key + ": " + value, indent);
472         } else {
473           throw new EvalException(loc,
474               "Invalid text format, expected a struct, a string, a bool, or an int but got a "
475               + EvalUtils.getDataTypeName(value) + " for " + container + " '" + key + "'");
476         }
477       }
478 
479       private void printTextMessage(String key, Object value, StringBuilder sb,
480           int indent, Location loc) throws EvalException {
481         if (value instanceof SkylarkList) {
482           for (Object item : ((SkylarkList) value)) {
483             // TODO(bazel-team): There should be some constraint on the fields of the structs
484             // in the same list but we ignore that for now.
485             printSimpleTextMessage(key, item, sb, indent, loc, "list element in struct field");
486           }
487         } else {
488         printSimpleTextMessage(key, value, sb, indent, loc, "struct field");
489         }
490       }
491 
492       private String escape(String string) {
493         // TODO(bazel-team): use guava's SourceCodeEscapers when it's released.
494         return string.replace("\"", "\\\"").replace("\n", "\\n");
495       }
496 
497       private void print(StringBuilder sb, String text, int indent) {
498         for (int i = 0; i < indent; i++) {
499           sb.append("  ");
500         }
501       sb.append(text);
502       sb.append("\n");
503       }
504     };
