35  /**
36   * {@link AttributeMap} implementation that provides the ability to retrieve <i>all possible</i>
37   * values an attribute might take.
38   */
39  public class AggregatingAttributeMapper extends AbstractAttributeMapper {
40  
41    /**
42     * Store for all of this rule's attributes that are non-configurable. These are
43     * unconditionally  available to computed defaults no matter what dependencies
44     * they've declared.
45     */
46    private final List<String> nonconfigurableAttributes;
47  
48    private AggregatingAttributeMapper(Rule rule) {
49      super(rule.getPackage(), rule.getRuleClassObject(), rule.getLabel(),
50          rule.getAttributeContainer());
51  
52      ImmutableList.Builder<String> nonconfigurableAttributesBuilder = ImmutableList.builder();
53      for (Attribute attr : rule.getAttributes()) {
54        if (!attr.isConfigurable()) {
55          nonconfigurableAttributesBuilder.add(attr.getName());
56        }
57      }
58      nonconfigurableAttributes = nonconfigurableAttributesBuilder.build();
59    }
60  
61    public static AggregatingAttributeMapper of(Rule rule) {
62      return new AggregatingAttributeMapper(rule);
63    }

65    /**
66     * Override that also visits the rule's configurable attribute keys (which are
67     * themselves labels).
68     *
69     * <p>Note that we directly parse the selectors rather than just calling {@link #visitAttribute}
70     * to iterate over all possible values. That's because {@link #visitAttribute} can grow
71     * exponentially with respect to the number of selects (e.g. if an attribute uses three selects
72     * with three conditions each, it can take nine possible values). So we want to avoid that code
73     * path whenever actual value iteration isn't specifically needed.
74     */
75    @Override
76    protected void visitLabels(Attribute attribute, AcceptsLabelAttribute observer) {
77      visitLabels(attribute, true, observer);
78    }
79  
80    private void visitLabels(Attribute attribute, boolean includeSelectKeys,
81      AcceptsLabelAttribute observer) {
82      Type<?> type = attribute.getType();
83      Type.SelectorList<?> selectorList = getSelectorList(attribute.getName(), type);
84      if (selectorList == null) {
85        if (getComputedDefault(attribute.getName(), attribute.getType()) != null) {
86          // Computed defaults are a special pain: we have no choice but to iterate through their
87          // (computed) values and look for labels.
88          for (Object value : visitAttribute(attribute.getName(), attribute.getType())) {
89            if (value != null) {
90              for (Label label : type.getLabels(value)) {
91                observer.acceptLabelAttribute(label, attribute);
92              }
93            }
94          }
95        } else {
96          super.visitLabels(attribute, observer);
97        }
98      } else {
99        for (Type.Selector<?> selector : selectorList.getSelectors()) {
100         for (Map.Entry<Label, ?> selectorEntry : selector.getEntries().entrySet()) {
101           if (includeSelectKeys && !Type.Selector.isReservedLabel(selectorEntry.getKey())) {
102             observer.acceptLabelAttribute(selectorEntry.getKey(), attribute);
103           }
104           for (Label value : type.getLabels(selectorEntry.getValue())) {
105             observer.acceptLabelAttribute(value, attribute);
106           }
107         }
108       }
109     }
110   }

112   /**
113    * Returns all labels reachable via the given attribute. If a label is listed multiple times,
114    * each instance appears in the returned list.
115    *
116    * @param includeSelectKeys whether to include config_setting keys for configurable attributes
117    */
118   public List<Label> getReachableLabels(String attributeName, boolean includeSelectKeys) {
119     final ImmutableList.Builder<Label> builder = ImmutableList.builder();
120     visitLabels(getAttributeDefinition(attributeName), includeSelectKeys,
121         new AcceptsLabelAttribute() {
122           @Override
123           public void acceptLabelAttribute(Label label, Attribute attribute) {
124             builder.add(label);
125           }
126         });
127     return builder.build();
128   }

130   /**
131    * Returns the labels that might appear multiple times in the same attribute value.
132    */
133   public Set<Label> checkForDuplicateLabels(Attribute attribute) {
134     String attrName = attribute.getName();
135     Type<?> attrType = attribute.getType();
136     ImmutableSet.Builder<Label> duplicates = ImmutableSet.builder();
137 
138     Type.SelectorList<?> selectorList = getSelectorList(attribute.getName(), attrType);
139     if (selectorList == null || selectorList.getSelectors().size() == 1) {
140       // Three possible scenarios:
141       //  1) Plain old attribute (no selects). Without selects, visitAttribute runs efficiently.
142       //  2) Computed default, possibly depending on other attributes using select. In this case,
143       //     visitAttribute might be inefficient. But we have no choice but to iterate over all
144       //     possible values (since we have to compute them), so we take the efficiency hit.
145       //  3) "attr = select({...})". With just a single select, visitAttribute runs efficiently.
146       for (Object value : visitAttribute(attrName, attrType)) {
147         if (value != null) {
148           duplicates.addAll(CollectionUtils.duplicatedElementsOf(
149               ImmutableList.copyOf(attrType.getLabels(value))));
150         }
151       }
152     } else {
153       // Multiple selects concatenated together. It's expensive to iterate over every possible
154       // value, so instead collect all labels across all the selects and check for duplicates.
155       // This is overly strict, since this counts duplicates across values. We can presumably
156       // relax this if necessary, but doing so would incur the value iteration expense this
157       // code path avoids.
158       List<Label> combinedLabels = new LinkedList<>(); // Labels that appear across all selectors.
159       for (Type.Selector<?> selector : selectorList.getSelectors()) {
160         // Labels within a single selector. It's okay for there to be duplicates as long as
161         // they're in different selector paths (since only one path can actually get chosen).
162         Set<Label> selectorLabels = new LinkedHashSet<>();
163         for (Object selectorValue : selector.getEntries().values()) {
164           Collection<Label> labelsInSelectorValue = attrType.getLabels(selectorValue);
165           // Duplicates within a single path are not okay.
166           duplicates.addAll(CollectionUtils.duplicatedElementsOf(labelsInSelectorValue));
167           selectorLabels.addAll(labelsInSelectorValue);
168         }
169         combinedLabels.addAll(selectorLabels);
170       }
171       duplicates.addAll(CollectionUtils.duplicatedElementsOf(combinedLabels));
172     }
173 
174     return duplicates.build();
175   }

177   /**
178    * Returns a list of all possible values an attribute can take for this rule.
179    *
180    * <p>Note that when an attribute uses multiple selects, it can potentially take on many
181    * values. So be cautious about unnecessarily relying on this method.
182    */
183   public <T> Iterable<T> visitAttribute(String attributeName, Type<T> type) {
184     // If this attribute value is configurable, visit all possible values.
185     Type.SelectorList<T> selectorList = getSelectorList(attributeName, type);
186     if (selectorList != null) {
187       ImmutableList.Builder<T> builder = ImmutableList.builder();
188       visitConfigurableAttribute(selectorList.getSelectors(), new BoundSelectorPaths(), type,
189           null, builder);
190       return builder.build();
191     }
192 
193     // If this attribute is a computed default, feed it all possible value combinations of
194     // its declared dependencies and return all computed results. For example, if this default
195     // uses attributes x and y, x can configurably be x1 or x2, and y can configurably be y1
196     // or y1, then compute default values for the (x1,y1), (x1,y2), (x2,y1), and (x2,y2) cases.
197     Attribute.ComputedDefault computedDefault = getComputedDefault(attributeName, type);
198     if (computedDefault != null) {
199       // This will hold every (value1, value2, ..) combination of the declared dependencies.
200       List<Map<String, Object>> depMaps = new LinkedList<>();
201       // Collect those combinations.
202       mapDepsForComputedDefault(computedDefault.dependencies(), depMaps,
203           ImmutableMap.<String, Object>of());
204       List<T> possibleValues = new ArrayList<>(); // Not ImmutableList.Builder: values may be null.
205       // For each combination, call getDefault on a specialized AttributeMap providing those values.
206       for (Map<String, Object> depMap : depMaps) {
207         possibleValues.add(type.cast(computedDefault.getDefault(mapBackedAttributeMap(depMap))));
208       }
209       return possibleValues;
210     }
211 
212     // For any other attribute, just return its direct value.
213     T value = get(attributeName, type);
214     return value == null ? ImmutableList.<T>of() : ImmutableList.of(value);
215   }

217   /**
218    * Determines all possible values a configurable attribute can take. Do not call this method
219    * unless really necessary (see TODO comment inside).
220    *
221    * @param selectors the selectors that make up this attribute assignment (in order)
222    * @param boundSelectorPaths paths that have already been chosen from previous selectors in an
223    *     earlier recursive call of this method. For example, given
224    *     <pre>cmd = select({':a': 'w', ':b': 'x'}) + select({':a': 'y', ':b': 'z'})</pre>
225    *     the only possible values for <code>cmd</code> are <code>"wy"</code> and <code>"xz"</code>.
226    *     This is because the selects have the same conditions, so whatever matches the first also
227    *     matches the second. Note that this doesn't work for selects with overlapping but
228    *     <i>different</i> key sets. That's because of key specialization (see
229    *     {@link com.google.devtools.build.lib.analysis.ConfiguredAttributeMapper} - if the
230    *     second select also included a condition <code>':c'</code> that includes both the flags
231    *     in <code>':a'</code> and <code>':b'</code>, <code>':c'</code> would be chosen over
232    *     them both.
233    * @param type the type of this attribute
234    * @param currentValueSoFar the partial value produced so far from earlier calls to this method
235    * @param valuesBuilder output container for full values this attribute can take
236    */
237   private <T> void visitConfigurableAttribute(List<Type.Selector<T>> selectors,
238       BoundSelectorPaths boundSelectorPaths, Type<T> type, T currentValueSoFar,
239       ImmutableList.Builder<T> valuesBuilder) {
240     // TODO(bazel-team): minimize or eliminate uses of this interface. It necessarily grows
241     // exponentially with the number of selects in the attribute. Is that always necessary?
242     // For example, dependency resolution just needs to know every possible label an attribute
243     // might reference, but it doesn't need to know the exact combination of labels that make
244     // up a value. This may be even less important for non-label values (e.g. strings), which
245     // have no impact on the dependency structure.
246 
247     if (selectors.isEmpty()) {
248       valuesBuilder.add(Preconditions.checkNotNull(currentValueSoFar));
249     } else {
250       Type.Selector<T> firstSelector = selectors.get(0);
251       List<Type.Selector<T>> remainingSelectors = selectors.subList(1, selectors.size());
252 
253       Map<Label, T> firstSelectorEntries = firstSelector.getEntries();
254       Label boundKey = boundSelectorPaths.getChosenKey(firstSelectorEntries.keySet());
255       if (boundKey != null) {
256         // If we've already followed some path from a previous selector with the same exact
257         // conditions as this one, we only need to visit that path (since the same key will
258         // match both selectors).
259         T boundValue = firstSelectorEntries.get(boundKey);
260         visitConfigurableAttribute(remainingSelectors, boundSelectorPaths, type,
261                     currentValueSoFar == null
262                         ? boundValue
263                         : type.concat(ImmutableList.of(currentValueSoFar, boundValue)),
264                     valuesBuilder);
265       } else {
266         // Otherwise, we need to iterate over all possible paths.
267         for (Map.Entry<Label, T> selectorBranch : firstSelectorEntries.entrySet()) {
268           // Bind this particular path for later selectors using the same conditions.
269           boundSelectorPaths.bind(firstSelectorEntries.keySet(), selectorBranch.getKey());
270           visitConfigurableAttribute(remainingSelectors, boundSelectorPaths, type,
271               currentValueSoFar == null
272                   ? selectorBranch.getValue()
273                   : type.concat(ImmutableList.of(currentValueSoFar, selectorBranch.getValue())),
274               valuesBuilder);
275           // Unbind the path (so when we pop back up the recursive stack we can rebind it to new
276           // values if we visit this selector again).
277           boundSelectorPaths.unbind(firstSelectorEntries.keySet());
278         }
279       }
280     }
281   }

283   /**
284    * Helper class for {@link #visitConfigurableAttribute}. See that method's comments for more
285    * details.
286    */
287   private static class BoundSelectorPaths {
288     private final Map<Set<Label>, Label> bindings = new HashMap<>();
289 
290     /**
291      * Binds the given config key set to the specified path. There should be no previous binding
292      * for this key set.
293      */
294     public void bind(Set<Label> allKeys, Label chosenKey) {
295       Preconditions.checkState(allKeys.contains(chosenKey));
296       Verify.verify(bindings.put(allKeys, chosenKey) == null);
297     }
298 
299     /**
300      * Unbinds the given config key set.
301      */
302     public void unbind(Set<Label> allKeys) {
303       Verify.verifyNotNull(bindings.remove(allKeys));
304     }
305 
306     /**
307      * Returns the key this config key set is bound to or null if no binding.
308      */
309     public Label getChosenKey(Set<Label> allKeys) {
310       return bindings.get(allKeys);
311     }
312   }

314   /**
315    * Given (possibly configurable) attributes that a computed default depends on, creates an
316    * {attrName -> attrValue} map for every possible combination of those attribute values and
317    * returns a list of all the maps. This defines the complete dependency space that can affect
318    * the computed default's values.
319    *
320    * <p>For example, given dependencies x and y, which might respectively have values x1, x2 and
321    * y1, y2, this returns:
322    * <pre>
323    *   [
324    *    {x: x1, y: y1},
325    *    {x: x1, y: y2},
326    *    {x: x2, y: y1},
327    *    {x: x2, y: y2}
328    *   ]
329    * </pre>
330    *
331    * @param depAttributes the names of the attributes this computed default depends on
332    * @param mappings the list of {attrName --> attrValue} maps defining the computed default's
333    *                 dependency space. This is where this method's results are written.
334    * @param currentMap a (possibly non-empty) map to add {attrName --> attrValue}
335    *                   entries to. Outside callers can just pass in an empty map.
336    */
337   private void mapDepsForComputedDefault(List<String> depAttributes,
338       List<Map<String, Object>> mappings, Map<String, Object> currentMap) {
339     // Because this method uses exponential time/space on the number of inputs, keep the
340     // maximum number of inputs conservatively small.
341     Preconditions.checkState(depAttributes.size() <= 2);
342 
343     if (depAttributes.isEmpty()) {
344       // Recursive base case: store whatever's already been populated in currentMap.
345       mappings.add(currentMap);
346       return;
347     }
348 
349     // Take the first attribute in the dependency list and iterate over all its values. For each
350     // value x, copy currentMap with the additional entry { firstAttrName: x }, then feed
351     // this recursively into a subcall over all remaining dependencies. This recursively
352     // continues until we run out of values.
353     String firstAttribute = depAttributes.get(0);
354     for (Object value : visitAttribute(firstAttribute, getAttributeType(firstAttribute))) {
355       Map<String, Object> newMap = new HashMap<>();
356       newMap.putAll(currentMap);
357       newMap.put(firstAttribute, value);
358       mapDepsForComputedDefault(depAttributes.subList(1, depAttributes.size()), mappings, newMap);
359     }
360   }

362   /**
363    * A custom {@link AttributeMap} that reads attribute values from the given Map. All
364    * non-configurable attributes are also readable. Any attempt to read an attribute
365    * that's not in one of these two cases triggers an IllegalArgumentException.
366    */
367   private AttributeMap mapBackedAttributeMap(final Map<String, Object> directMap) {
368     final AggregatingAttributeMapper owner = AggregatingAttributeMapper.this;
369     return new AttributeMap() {
370 
371       @Override
372       public <T> T get(String attributeName, Type<T> type) {
373         owner.checkType(attributeName, type);
374         if (nonconfigurableAttributes.contains(attributeName)) {
375           return owner.get(attributeName, type);
376         }
377         if (!directMap.containsKey(attributeName)) {
378           throw new IllegalArgumentException("attribute \"" + attributeName
379               + "\" isn't available in this computed default context");
380         }
381         return type.cast(directMap.get(attributeName));
382       }
383 
384       @Override
385       public <T> boolean isConfigurable(String attributeName, Type<T> type) {
386         return owner.isConfigurable(attributeName, type);
387       }
388 
389       @Override public String getName() { return owner.getName(); }
390       @Override public Label getLabel() { return owner.getLabel(); }
391       @Override public Iterable<String> getAttributeNames() {
392         return ImmutableList.<String>builder()
393             .addAll(directMap.keySet()).addAll(nonconfigurableAttributes).build();
394       }
395       @Override
396       public void visitLabels(AcceptsLabelAttribute observer) { owner.visitLabels(observer); }
397       @Override
398       public String getPackageDefaultHdrsCheck() { return owner.getPackageDefaultHdrsCheck(); }
399       @Override
400       public Boolean getPackageDefaultTestOnly() { return owner.getPackageDefaultTestOnly(); }
401       @Override
402       public String getPackageDefaultDeprecation() { return owner.getPackageDefaultDeprecation(); }
403       @Override
404       public ImmutableList<String> getPackageDefaultCopts() {
405         return owner.getPackageDefaultCopts();
406       }
407       @Nullable @Override
408       public Type<?> getAttributeType(String attrName) { return owner.getAttributeType(attrName); }
409       @Nullable @Override  public Attribute getAttributeDefinition(String attrName) {
410         return owner.getAttributeDefinition(attrName);
411       }
412       @Override public boolean isAttributeValueExplicitlySpecified(String attributeName) {
413         return owner.isAttributeValueExplicitlySpecified(attributeName);
414       }
415       @Override
416       public boolean has(String attrName, Type<?> type) { return owner.has(attrName, type); }
417     };
418   }
