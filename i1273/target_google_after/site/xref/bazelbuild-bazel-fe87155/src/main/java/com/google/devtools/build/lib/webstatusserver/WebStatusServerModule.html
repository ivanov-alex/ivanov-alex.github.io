41  /**
42   * Web server for monitoring blaze commands status.
43   */
44  public class WebStatusServerModule extends BlazeModule {
45    static final String LAST_TEST_URI = "/tests/last";
46    // 100 is an arbitrary limit; it seems like a reasonable size for history and it's okay to change
47    // it
48    private static final int MAX_TESTS_STORED = 100;
49  
50    private HttpServer server;
51    private boolean running = false;
52    private BlazeServerStartupOptions serverOptions;
53    private static final Logger LOG =
54        Logger.getLogger(WebStatusServerModule.class.getCanonicalName());
55    private int port;
56    private LinkedList<TestStatusHandler> testsRan = new LinkedList<>();
57    @SuppressWarnings("unused")
58    private WebStatusEventCollector collector;
59    @SuppressWarnings("unused")
60    private IndexPageHandler indexHandler;
61  
62    @Override
63    public Iterable<Class<? extends OptionsBase>> getStartupOptions() {
64      return ImmutableList.<Class<? extends OptionsBase>>of(BlazeServerStartupOptions.class);
65    }
66  
67    @Override
68    public void blazeStartup(OptionsProvider startupOptions, BlazeVersionInfo versionInfo,
69        UUID instanceId, BlazeDirectories directories, Clock clock) throws AbruptExitException {
70      serverOptions = startupOptions.getOptions(BlazeServerStartupOptions.class);
71      if (serverOptions.useWebStatusServer <= 0) {
72        LOG.info("web status server disabled");
73        return;
74      }
75      port = serverOptions.useWebStatusServer;
76      try {
77        server = HttpServer.create(new InetSocketAddress(port), 0);
78        serveStaticContent();
79        TextHandler lastCommandHandler = new TextHandler("No commands ran yet.");
80        server.createContext("/last", lastCommandHandler);
81        server.setExecutor(null);
82        server.start();
83        indexHandler = new IndexPageHandler(server, this.testsRan);
84        running = true;
85        LOG.info("Running web status server on port " + port);
86      } catch (IOException e) {
87        // TODO(bazel-team): Display information about why it failed
88        running = false;
89        LOG.warning("Unable to run web status server on port " + port);
90      }
91    }
92  
93    @Override
94    public void beforeCommand(Command command, CommandEnvironment env)
95        throws AbruptExitException {
96      if (!running) {
97        return;
98      }
99      collector =
100         new WebStatusEventCollector(env.getEventBus(), env.getReporter(), this);
101   }
102 
103   public void commandStarted() {
104     WebStatusBuildLog currentBuild = collector.getBuildLog();
105 
106     if (testsRan.size() == MAX_TESTS_STORED) {
107       TestStatusHandler oldestTest = testsRan.removeLast();
108       oldestTest.deregister();
109     }
110 
111     TestStatusHandler lastTest = new TestStatusHandler(server, currentBuild);
112     testsRan.add(lastTest);
113 
114     lastTest.overrideURI(LAST_TEST_URI);
115   }
116 
117   private void serveStaticContent() {
118     StaticResourceHandler testjs =
119         StaticResourceHandler.createFromRelativePath("static/test.js", "application/javascript");
120     StaticResourceHandler indexjs =
121         StaticResourceHandler.createFromRelativePath("static/index.js", "application/javascript");
122     StaticResourceHandler style =
123         StaticResourceHandler.createFromRelativePath("static/style.css", "text/css");
124     StaticResourceHandler d3 = StaticResourceHandler.createFromAbsolutePath(
125         "third_party/javascript/d3/d3-js.js", "application/javascript");
126     StaticResourceHandler jquery = StaticResourceHandler.createFromAbsolutePath(
127         "third_party/javascript/jquery/v2_0_3/jquery_uncompressed.jslib",
128         "application/javascript");
129     StaticResourceHandler testFrontend =
130         StaticResourceHandler.createFromRelativePath("static/test.html", "text/html");
131 
132     server.createContext("/css/style.css", style);
133     server.createContext("/js/test.js", testjs);
134     server.createContext("/js/index.js", indexjs);
135     server.createContext("/js/lib/d3.js", d3);
136     server.createContext("/js/lib/jquery.js", jquery);
137     server.createContext(LAST_TEST_URI, testFrontend);
138   }
139 
140   private static class TextHandler implements HttpHandler {
141     private String response;
142 
143     private TextHandler(String response) {
144       this.response = response;
145     }
146 
147     @Override
148     public void handle(HttpExchange exchange) throws IOException {
149       exchange.getResponseHeaders().put("Content-Type", ImmutableList.of("text/plain"));
150       exchange.sendResponseHeaders(200, response.length());
151       try (OutputStream os = exchange.getResponseBody()) {
152         os.write(response.getBytes(StandardCharsets.UTF_8));
153       }
154     }
155   }
156 
157   public int getPort() {
158     return port;
159   }
160 }
