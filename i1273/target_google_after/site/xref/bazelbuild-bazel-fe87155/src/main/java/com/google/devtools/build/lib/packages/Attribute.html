48  /**
49   * Metadata of a rule attribute. Contains the attribute name and type, and an
50   * default value to be used if none is provided in a rule declaration in a BUILD
51   * file. Attributes are immutable, and may be shared by more than one rule (for
52   * example, <code>foo_binary</code> and <code>foo_library</code> may share many
53   * attributes in common).
54   */
55  @Immutable
56  public final class Attribute implements Comparable<Attribute> {
57  
58    public static final Predicate<RuleClass> ANY_RULE = Predicates.alwaysTrue();
59  
60    public static final Predicate<RuleClass> NO_RULE = Predicates.alwaysFalse();

62    private static final class RuleAspect {
63      private final Class<? extends AspectFactory<?, ?, ?>> aspectFactory;
64      private final Function<Rule, AspectParameters> parametersExtractor;
65  
66      RuleAspect(
67          Class<? extends AspectFactory<?, ?, ?>> aspectFactory,
68          Function<Rule, AspectParameters> parametersExtractor) {
69        this.aspectFactory = aspectFactory;
70        this.parametersExtractor = parametersExtractor;
71      }
72  
73      Class<? extends AspectFactory<?, ?, ?>> getAspectFactory() {
74        return aspectFactory;
75      }
76      
77      Function<Rule, AspectParameters> getParametersExtractor() {
78        return parametersExtractor;
79      }
80    }

82    /**
83     * A configuration transition.
84     */
85    public interface Transition {
86      /**
87       * Usually, a non-existent entry in the configuration transition table indicates an error.
88       * Unfortunately, that means that we need to always build the full table. This method allows a
89       * transition to indicate that a non-existent entry indicates a self transition, i.e., that the
90       * resulting configuration is the same as the current configuration. This can simplify the code
91       * needed to set up the transition table.
92       */
93      boolean defaultsToSelf();
94    }

96    /**
97     * A configuration split transition; this should be used to transition to multiple configurations
98     * simultaneously. Note that the corresponding rule implementations must have special support to
99     * handle this.
100    */
101   // TODO(bazel-team): Serializability constraints?
102   public interface SplitTransition<T> extends Transition {
103     /**
104      * Return the list of {@code BuildOptions} after splitting; empty if not applicable.
105      */
106     List<T> split(T buildOptions);
107   }

109   /**
110    * Declaration how the configuration should change when following a label or
111    * label list attribute.
112    */
113   @SkylarkModule(name = "ConfigurationTransition", doc =
114       "Declares how the configuration should change when following a dependency. "
115     + "It can be either <a href=\"globals.html#DATA_CFG\">DATA_CFG</a> or "
116     + "<a href=\"globals.html#HOST_CFG\">HOST_CFG</a>.")
117   public enum ConfigurationTransition implements Transition {
118     /** No transition, i.e., the same configuration as the current. */
119     NONE,
120 
121     /** Transition to the host configuration. */
122     HOST,
123 
124     /** Transition to a null configuration (applies to, e.g., input files). */
125     NULL,
126 
127     /** Transition from the target configuration to the data configuration. */
128     // TODO(bazel-team): Move this elsewhere.
129     DATA;
130 
131     @Override
132     public boolean defaultsToSelf() {
133       return false;
134     }
135   }

169     NON_EMPTY,

196      */
197     CHECK_ALLOWED_VALUES,

202      */

240   /**
241    * A predicate class to check if the value of the attribute comes from a predefined set.
242    */
243   public static class AllowedValueSet implements PredicateWithMessage<Object> {
244 
245     private final Set<Object> allowedValues;
246 
247     public AllowedValueSet(Iterable<?> values) {
248       Preconditions.checkNotNull(values);
249       Preconditions.checkArgument(!Iterables.isEmpty(values));
250       allowedValues = ImmutableSet.copyOf(values);
251     }
252 
253     @Override
254     public boolean apply(Object input) {
255       return allowedValues.contains(input);
256     }
257 
258     @Override
259     public String getErrorReason(Object value) {
260       return String.format("has to be one of %s instead of '%s'",
261           StringUtil.joinEnglishList(allowedValues, "or", "'"), value);
262     }
263 
264     @VisibleForTesting
265     public Collection<Object> getAllowedValues() {
266       return allowedValues;
267     }
268   }

306     /**
307      * Creates an attribute builder with given name and type. This attribute is optional, uses
308      * target configuration and has a default value the same as its type default value. This
309      * attribute will be marked as undocumented if its name starts with the dollar sign ({@code $})
310      * or colon ({@code :}).
311      *
312      * @param name attribute name
313      * @param type attribute type
314      */
315     public Builder(String name, Type<TYPE> type) {
316       this.name = Preconditions.checkNotNull(name);
317       this.type = Preconditions.checkNotNull(type);
318       if (isImplicit(name) || isLateBound(name)) {
319         setPropertyFlag(PropertyFlag.UNDOCUMENTED, "undocumented");
320       }
321     }
322 
323     private Builder<TYPE> setPropertyFlag(PropertyFlag flag, String propertyName) {
324       Preconditions.checkState(
325           !propertyFlags.contains(flag), "%s flag is already set", propertyName);
326       propertyFlags.add(flag);
327       return this;
328     }
329 
330     /**
331      * Sets the property flag of the corresponding name if exists, otherwise throws an Exception.
332      * Only meant to use from Skylark, do not use from Java.
333      */
334     public Builder<TYPE> setPropertyFlag(String propertyName) {
335       PropertyFlag flag = null;
336       try {
337         flag = PropertyFlag.valueOf(propertyName);
338       } catch (IllegalArgumentException e) {
339         throw new IllegalArgumentException("unknown attribute flag " + propertyName);
340       }
341       setPropertyFlag(flag, propertyName);
342       return this;
343     }

779   /**
780    * A computed default is a default value for a Rule attribute that is a
781    * function of other attributes of the rule.
782    *
783    * <p>Attributes whose defaults are computed are first initialized to the default
784    * for their type, and then the computed defaults are evaluated after all
785    * non-computed defaults have been initialized. There is no defined order
786    * among computed defaults, so they must not depend on each other.
787    *
788    * <p>If a computed default reads the value of another attribute, at least one of
789    * the following must be true:
790    *
791    * <ol>
792    *   <li>The other attribute must be declared in the computed default's constructor</li>
793    *   <li>The other attribute must be non-configurable ({@link Builder#nonconfigurable}</li>
794    * </ol>
795    *
796    * <p>The reason for enforced declarations is that, since attribute values might be
797    * configurable, a computed default that depends on them may itself take multiple
798    * values. Since we have no access to a target's configuration at the time these values
799    * are computed, we need the ability to probe the default's *complete* dependency space.
800    * Declared dependencies allow us to do so sanely. Non-configurable attributes don't have
801    * this problem because their value is fixed and known even without configuration information.
802    *
803    * <p>Implementations of this interface must be immutable.
804    */
805   public abstract static class ComputedDefault {
806     private final List<String> dependencies;
807     List<String> dependencies() { return dependencies; }
808 
809     /**
810      * Create a computed default that can read all non-configurable attribute values and no
811      * configurable attribute values.
812      */
813     public ComputedDefault() {
814       dependencies = ImmutableList.of();
815     }
816 
817     /**
818      * Create a computed default that can read all non-configurable attributes values and one
819      * explicitly specified configurable attribute value
820      */
821     public ComputedDefault(String depAttribute) {
822       dependencies = ImmutableList.of(depAttribute);
823     }
824 
825     /**
826      * Create a computed default that can read all non-configurable attributes values and two
827      * explicitly specified configurable attribute values.
828      */
829     public ComputedDefault(String depAttribute1, String depAttribute2) {
830       dependencies = ImmutableList.of(depAttribute1, depAttribute2);
831     }
832 
833     public abstract Object getDefault(AttributeMap rule);
834   }

1340    * <p>During population of the rule's attribute dictionary, all non-computed
1341    * defaults must be set before all computed ones.
1342    *
1343    * @param rule the rule to which this attribute belongs; non-null if
1344    *   {@code hasComputedDefault()}; ignored otherwise.
1345    */
1346   public Object getDefaultValue(Rule rule) {
1347     if (!getCondition().apply(rule == null ? null : NonconfigurableAttributeMapper.of(rule))) {
1348       return null;
1349     } else if (defaultValue instanceof LateBoundDefault<?>) {
1350       return ((LateBoundDefault<?>) defaultValue).getDefault();
1351     } else {
1352       return defaultValue;
1353     }
1354   }
