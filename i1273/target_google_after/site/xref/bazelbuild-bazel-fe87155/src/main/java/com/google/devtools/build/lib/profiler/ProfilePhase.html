15  package com.google.devtools.build.lib.profiler;
16  
17  /**
18   * Build phase markers. Used as a separators between different build phases.
19   */
20  public enum ProfilePhase {
21    LAUNCH("launch", "Launch Blaze", 0x3F9FCF9F),                // 9C9
22    INIT("init", "Initialize command", 0x3F9F9FCF),              // 99C
23    LOAD("loading", "Load packages", 0x3FCFFFCF),                // CFC
24    ANALYZE("analysis", "Analyze dependencies", 0x3FCFCFFF),     // CCF
25    LICENSE("license checking", "Analyze licenses", 0x3FCFFFFF), // CFF
26    PREPARE("preparation", "Prepare for build", 0x3FFFFFCF),     // FFC
27    EXECUTE("execution", "Build artifacts", 0x3FFFCFCF),         // FCC
28    FINISH("finish", "Complete build",0x3FFFCFFF);               // FCF
29  
30    /** Short name for the phase */
31    public final String nick;
32    /** Human readable description for the phase. */
33    public final String description;
34    /** Default color of the task, when rendered in a chart. */
35    public final int color;
36  
37    ProfilePhase(String nick, String description, int color) {
38      this.nick = nick;
39      this.description = description;
40      this.color = color;
41    }
42  }
