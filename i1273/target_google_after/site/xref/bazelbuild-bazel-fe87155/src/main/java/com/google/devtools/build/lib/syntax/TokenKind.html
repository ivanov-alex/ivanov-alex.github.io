15  package com.google.devtools.build.lib.syntax;
16  
17  /**
18   * A TokenKind is an enumeration of each different kind of lexical symbol.
19   */
20  public enum TokenKind {
21  
22    ASSERT("assert"),
23    AND("and"),
24    AS("as"),
25    BREAK("break"),
26    CLASS("class"),
27    COLON(":"),
28    COMMA(","),
29    COMMENT("comment"),
30    CONTINUE("continue"),
31    DEF("def"),
32    DEL("del"),
33    DOT("."),
34    ELIF("elif"),
35    ELSE("else"),
36    EOF("EOF"),
37    EQUALS("="),
38    EQUALS_EQUALS("=="),
39    EXCEPT("except"),
40    FINALLY("finally"),
41    FOR("for"),
42    FROM("from"),
43    GLOBAL("global"),
44    GREATER(">"),
45    GREATER_EQUALS(">="),
46    IDENTIFIER("identifier"),
47    IF("if"),
48    ILLEGAL("illegal character"),
49    IMPORT("import"),
50    IN("in"),
51    INDENT("indent"),
52    INT("integer"),
53    IS("is"),
54    LAMBDA("lambda"),
55    LBRACE("{"),
56    LBRACKET("["),
57    LESS("<"),
58    LESS_EQUALS("<="),
59    LPAREN("("),
60    MINUS("-"),
61    NEWLINE("newline"),
62    NONLOCAL("nonlocal"),
63    NOT("not"),
64    NOT_EQUALS("!="),
65    NOT_IN("not in"),
66    OR("or"),
67    OUTDENT("outdent"),
68    PASS("pass"),
69    PERCENT("%"),
70    PIPE("|"),
71    PLUS("+"),
72    PLUS_EQUALS("+="),
73    RAISE("raise"),
74    RBRACE("}"),
75    RBRACKET("]"),
76    RETURN("return"),
77    RPAREN(")"),
78    SEMI(";"),
79    SLASH("/"),
80    STAR("*"),
81    STAR_STAR("**"),
82    STRING("string"),
83    TRY("try"),
84    WHILE("while"),
85    WITH("with"),
86    YIELD("yield");
87  
88    private final String prettyName;
89  
90    private TokenKind(String prettyName) {
91      this.prettyName = prettyName;
92    }
93  
94    /**
95     * Returns the pretty name for this token, for use in error messages for the user.
96     */
97    public String getPrettyName() {
98      return prettyName;
99    }
100 }
