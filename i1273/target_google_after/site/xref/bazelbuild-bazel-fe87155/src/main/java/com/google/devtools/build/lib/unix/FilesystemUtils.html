36    /**
37     * Returns true iff the file identified by 'path' is a symbolic link. Has
38     * similar semantics to POSIX stat(2) syscall, with all errors being mapped to
39     * a false return.
40     *
41     * @param path the file of interest
42     * @return true iff path exists, is accessible and is a symlink
43     */
44    public static boolean isSymbolicLink(File path) {
45      try {
46        return lstat(path.toString()).isSymbolicLink();
47      } catch (IOException e) {
48        return false;
49      }
50    }

56     * a false return.
57     *
58     * @param path the file of interest
59     * @return true iff path exists, is accessible and is a symlink

70    /**
71     * Marks the file or directory {@code path} as executable. (Non-atomic)
72     *
73     * @see File#setReadOnly
74     *
75     * @param path the file of interest
76     * @throws FileAccessException if path can't be accessed
77     * @throws FileNotFoundException if path doesn't exist
78     * @throws IOException for other filesystem or path errors
79     */
80    public static void setExecutable(File path) throws IOException {
81      String p = path.toString();
82      chmod(p, stat(p).getPermissions() | FileStatus.S_IEXEC);
83    }
84  
85    /**
86     * Marks the file or directory {@code path} as owner writable. (Non-atomic)
87     *
88     * @see File#setReadOnly
89     *
90     * @param path the file of interest
91     * @throws FileAccessException if path can't be accessed
92     * @throws FileNotFoundException if path doesn't exist
93     * @throws IOException for other filesystem or path errors
94     */
95    public static void setWritable(File path) throws IOException {
96      String p = path.toString();
97      chmod(p, stat(p).getPermissions() | FileStatus.S_IWUSR);
98    }

100   /**
101    * Changes permissions of a file.
102    *
103    * @param path the file whose mode to change.
104    * @param mode the mode bits within 07777, interpreted according to
105    *   long-standing UNIX tradition.
106    * @throws IOException if the chmod() syscall failed.
107    */
108   public static void chmod(File path, int mode) throws IOException {
109     int mask = FileStatus.S_ISUID |
110                FileStatus.S_ISGID |
111                FileStatus.S_ISVTX |
112                FileStatus.S_IRWXA;
113     chmod(path.toString(), mode & mask);
114   }

249   /**
250    * An enum for specifying now the types of the individual entries returned by
251    * {@link #readdir(String, ReadTypes)} is to be returned.
252    */
253   public enum ReadTypes {
254     NONE('n'),      // Do not read types
255     NOFOLLOW('d'),  // Do not follow symlinks
256     FOLLOW('f');    // Follow symlinks; never returns "SYMLINK" and returns "UNKNOWN" when dangling
257 
258     private final char code;
259 
260     private ReadTypes(char code) {
261       this.code = code;
262     }
263 
264     private char getCode() {
265       return code;
266     }
267   }

269   /**
270    * A compound return type for readdir(), analogous to struct dirent[] in C. A low memory profile
271    * is critical for this class, as instances are expected to be kept around for caching for
272    * potentially a long time.
273    */
274   public static final class Dirents {
275 
276   /**
277    * The type of the directory entry.
278    */
279   public enum Type {
280     FILE,
281     DIRECTORY,
282     SYMLINK,
283     UNKNOWN;
284 
285     private static Type forChar(char c) {
286       if (c == 'f') {
287         return Type.FILE;
288       } else if (c == 'd') {
289         return Type.DIRECTORY;
290       } else if (c == 's') {
291         return Type.SYMLINK;
292       } else {
293         return Type.UNKNOWN;
294       }
295     }
296   }
297 
298     /** The names of the entries in a directory. */
299     private final String[] names;
300     /**
301      * An optional (nullable) array of entry types, corresponding positionally
302      * to the "names" field.  The types are:
303      *   'd': a subdirectory
304      *   'f': a regular file
305      *   's': a symlink (only returned with {@code NOFOLLOW})
306      *   '?': anything else
307      * Note that unlike libc, this implementation of readdir() follows
308      * symlinks when determining these types.
309      *
310      * <p>This is intentionally a byte array rather than a array of enums to save memory.
311      */
312     private final byte[] types;
313 
314     /** called from JNI */
315     public Dirents(String[] names, byte[] types) {
316       this.names = names;
317       this.types = types;
318     }
319 
320     public int size() {
321       return names.length;
322     }
323 
324     public boolean hasTypes() {
325       return types != null;
326     }
327 
328     public String getName(int i) {
329       return names[i];
330     }
331 
332     public Type getType(int i) {
333       return Type.forChar((char) types[i]);
334     }
335   }
336 
337   /**
338    * Native wrapper around POSIX opendir(2)/readdir(3)/closedir(3) syscall.
339    *
340    * @param path the directory to read.
341    * @param readTypes How the types of individual entries should be returned. If {@code NONE},
342    *   the "types" field in the result will be null.
343    * @return a Dirents object, containing "names", the list of directory entries
344    *   (excluding "." and "..") in the order they were returned by the system,
345    *   and "types", an array of entry types (file, directory, etc) corresponding
346    *   positionally to "names".
347    * @throws IOException if the call to opendir failed for any reason.
348    */
349   public static Dirents readdir(String path, ReadTypes readTypes) throws IOException {
350     // Passing enums to native code is possible, but onerous; we use a char instead.
351     return readdir(path, readTypes.getCode());
352   }

427   /**
428    * Removes entire directory tree. Doesn't follow symlinks.
429    *
430    * @param path the file or directory to remove.
431    * @throws IOException if the remove failed.
432    */
433   public static void rmTree(String path) throws IOException { 
434     if (isDirectory(path)) {
435       String[] contents = readdir(path);
436       for (String entry : contents) {
437         rmTree(path + "/" + entry);
438       }
439     }
440     remove(path.toString());
441   }
