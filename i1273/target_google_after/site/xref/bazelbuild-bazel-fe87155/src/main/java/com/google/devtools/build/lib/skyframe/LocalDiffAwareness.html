43  /**
44   * File system watcher for local filesystems. It's able to provide a list of changed
45   * files between two consecutive calls. Uses the standard Java WatchService, which uses
46   * 'inotify' on Linux.
47   */
48  public class LocalDiffAwareness implements DiffAwareness {
49  
50    /** Factory for creating {@link LocalDiffAwareness} instances. */
51    public static class Factory implements DiffAwareness.Factory {
52      private final ImmutableList<String> prefixBlacklist;
53  
54      /**
55       * Creates a new factory; the file system watcher may not work on all file systems, particularly
56       * for network file systems. The prefix blacklist can be used to blacklist known paths that
57       * point to network file systems.
58       */
59      public Factory(ImmutableList<String> prefixBlacklist) {
60        this.prefixBlacklist = prefixBlacklist;
61      }
62  
63      @Override
64      public DiffAwareness maybeCreate(com.google.devtools.build.lib.vfs.Path pathEntry) {
65        com.google.devtools.build.lib.vfs.Path resolvedPathEntry;
66        try {
67          resolvedPathEntry = pathEntry.resolveSymbolicLinks();
68        } catch (IOException e) {
69          return null;
70        }
71        PathFragment resolvedPathEntryFragment = resolvedPathEntry.asFragment();
72        // There's no good way to automatically detect network file systems. We rely on a blacklist
73        // for now (and maybe add a command-line option in the future?).
74        for (String prefix : prefixBlacklist) {
75          if (resolvedPathEntryFragment.startsWith(new PathFragment(prefix))) {
76            return null;
77          }
78        }
79  
80        WatchService watchService;
81        try {
82          watchService = FileSystems.getDefault().newWatchService();
83        } catch (IOException e) {
84          return null;
85        }
86        return new LocalDiffAwareness(resolvedPathEntryFragment.toString(),
87            watchService);
88      }
89    }
90  
91    private int numGetCurrentViewCalls = 0;
92  
93    /**
94     * Bijection from WatchKey to the (absolute) Path being watched. WatchKeys don't have this
95     * functionality built-in so we do it ourselves.
96     */
97    private final HashBiMap<WatchKey, Path> watchKeyToDirBiMap = HashBiMap.create();
98  
99    /** Root directory to watch. This is an absolute path. */
100   private final Path watchRootPath;
101 
102   /** Every directory is registered under this watch service. */
103   private WatchService watchService;
104 
105   private LocalDiffAwareness(String watchRoot, WatchService watchService) {
106     this.watchRootPath = FileSystems.getDefault().getPath(watchRoot);
107     this.watchService = watchService;
108   }

110   /**
111    * The WatchService is inherently sequential and side-effectful, so we enforce this by only
112    * supporting {@link #getDiff} calls that happen to be sequential.
113    */
114   private static class SequentialView implements DiffAwareness.View {
115     private final LocalDiffAwareness owner;
116     private final int position;
117     private final Set<Path> modifiedAbsolutePaths;
118 
119     public SequentialView(LocalDiffAwareness owner, int position, Set<Path> modifiedAbsolutePaths) {
120       this.owner = owner;
121       this.position = position;
122       this.modifiedAbsolutePaths = modifiedAbsolutePaths;
123     }
124 
125     public static boolean areInSequence(SequentialView oldView, SequentialView newView) {
126       return oldView.owner == newView.owner && (oldView.position + 1) == newView.position;
127     }
128 
129     @Override
130     public String toString() {
131       return String.format("SequentialView[owner=%s, position=%d, modifiedAbsolutePaths=%s]", owner,
132           position, modifiedAbsolutePaths);
133     }
134   }
135 
136   @Override
137   public SequentialView getCurrentView() throws BrokenDiffAwarenessException {
138     Set<Path> modifiedAbsolutePaths;
139     if (numGetCurrentViewCalls++ == 0) {
140       try {
141         registerSubDirectoriesAndReturnContents(watchRootPath);
142       } catch (IOException e) {
143         close();
144         throw new BrokenDiffAwarenessException(
145             "Error encountered with local file system watcher " + e);
146       }
147       modifiedAbsolutePaths = ImmutableSet.of();
148     } else {
149       try {
150         modifiedAbsolutePaths = collectChanges();
151       } catch (BrokenDiffAwarenessException e) {
152         close();
153         throw e;
154       } catch (IOException e) {
155         close();
156         throw new BrokenDiffAwarenessException(
157             "Error encountered with local file system watcher " + e);
158       } catch (ClosedWatchServiceException e) {
159         throw new BrokenDiffAwarenessException(
160             "Internal error with the local file system watcher " + e);
161       }
162     }
163     return new SequentialView(this, numGetCurrentViewCalls, modifiedAbsolutePaths);
164   }
165 
166   @Override
167   public ModifiedFileSet getDiff(View oldView, View newView)
168       throws IncompatibleViewException, BrokenDiffAwarenessException {
169     SequentialView oldSequentialView;
170     SequentialView newSequentialView;
171     try {
172       oldSequentialView = (SequentialView) oldView;
173       newSequentialView = (SequentialView) newView;
174     } catch (ClassCastException e) {
175       throw new IncompatibleViewException("Given views are not from LocalDiffAwareness");
176     }
177     if (!SequentialView.areInSequence(oldSequentialView, newSequentialView)) {
178       return ModifiedFileSet.EVERYTHING_MODIFIED;
179     }
180     return ModifiedFileSet.builder()
181         .modifyAll(Iterables.transform(newSequentialView.modifiedAbsolutePaths,
182             nioAbsolutePathToPathFragment))
183             .build();
184   }
185 
186   @Override
187   public String name() {
188     return "local";
189   }
190 
191   @Override
192   public void close() {
193     try {
194       watchService.close();
195     } catch (IOException ignored) {
196       // Nothing we can do here.
197     }
198   }

200   /** Converts java.nio.file.Path objects to vfs.PathFragment. */
201   private final Function<Path, PathFragment> nioAbsolutePathToPathFragment =
202       new Function<Path, PathFragment>() {
203     @Override
204     public PathFragment apply(Path input) {
205       Preconditions.checkArgument(input.startsWith(watchRootPath), "%s %s", input,
206           watchRootPath);
207       return new PathFragment(watchRootPath.relativize(input).toString());
208     }
209   };

211   /** Returns the changed files caught by the watch service. */
212   private Set<Path> collectChanges() throws BrokenDiffAwarenessException, IOException {
213     Set<Path> createdFilesAndDirectories = new HashSet<>();
214     Set<Path> deletedOrModifiedFilesAndDirectories = new HashSet<>();
215     Set<Path> deletedTrackedDirectories = new HashSet<>();
216 
217     WatchKey watchKey;
218     while ((watchKey = watchService.poll()) != null) {
219       Path dir = watchKeyToDirBiMap.get(watchKey);
220       Preconditions.checkArgument(dir != null);
221 
222       // We replay all the events for this watched directory in chronological order and
223       // construct the diff of this directory since the last #collectChanges call.
224       for (WatchEvent<?> event : watchKey.pollEvents()) {
225         Kind<?> kind = event.kind();
226         if (kind == StandardWatchEventKinds.OVERFLOW) {
227           // TODO(bazel-team): find out when an overflow might happen, and maybe handle it more
228           // gently.
229           throw new BrokenDiffAwarenessException("Overflow when watching local filesystem for "
230               + "changes");
231         }
232         if (event.context() == null) {
233           // The WatchService documentation mentions that WatchEvent#context may return null, but
234           // doesn't explain how/why it would do so. Looking at the implementation, it only
235           // happens on an overflow event. But we make no assumptions about that implementation
236           // detail here.
237           throw new BrokenDiffAwarenessException("Insufficient information from local file system "
238               + "watcher");
239         }
240         // For the events we've registered, the context given is a relative path.
241         Path relativePath = (Path) event.context();
242         Path path = dir.resolve(relativePath);
243         Preconditions.checkState(path.isAbsolute(), path);
244         if (kind == StandardWatchEventKinds.ENTRY_CREATE) {
245           createdFilesAndDirectories.add(path);
246           deletedOrModifiedFilesAndDirectories.remove(path);
247         } else if (kind == StandardWatchEventKinds.ENTRY_DELETE) {
248           createdFilesAndDirectories.remove(path);
249           deletedOrModifiedFilesAndDirectories.add(path);
250           WatchKey deletedDirectoryKey = watchKeyToDirBiMap.inverse().get(path);
251           if (deletedDirectoryKey != null) {
252             // If the deleted directory has children, then there will also be events for the
253             // WatchKey of the directory itself. WatchService#poll doesn't specify the order in
254             // which WatchKeys are returned, so the key for the directory itself may be processed
255             // *after* the current key (the parent of the deleted directory), and so we don't want
256             // to remove the deleted directory from our bimap just yet.
257             //
258             // For example, suppose we have the file '/root/a/foo.txt' and are watching the
259             // directories '/root' and '/root/a'. If the directory '/root/a' gets deleted then the
260             // following is a valid sequence of events by key.
261             //
262             //  WatchKey '/root/'
263             //    WatchEvent EVENT_MODIFY 'a'
264             //    WatchEvent EVENT_DELETE 'a'
265             //  WatchKey '/root/a'
266             //    WatchEvent EVENT_DELETE 'foo.txt'
267             deletedTrackedDirectories.add(path);
268             // Since inotify uses inodes under the covers we cancel our registration on this key to
269             // avoid getting WatchEvents from a new directory that happens to have the same inode.
270             deletedDirectoryKey.cancel();
271           }
272         } else if (kind == StandardWatchEventKinds.ENTRY_MODIFY) {
273           // If a file was created and then modified, then the net diff is that it was
274           // created.
275           if (!createdFilesAndDirectories.contains(path)) {
276             deletedOrModifiedFilesAndDirectories.add(path);
277           }
278         }
279       }
280 
281       if (!watchKey.reset()) {
282         // Watcher got deleted, directory no longer valid.
283         watchKeyToDirBiMap.remove(watchKey);
284       }
285     }
286 
287     for (Path path : deletedTrackedDirectories) {
288       WatchKey staleKey = watchKeyToDirBiMap.inverse().get(path);
289       watchKeyToDirBiMap.remove(staleKey);
290     }
291     if (watchKeyToDirBiMap.isEmpty()) {
292       // No more directories to watch, something happened the root directory being watched.
293       throw new IOException("Root directory " + watchRootPath + " became inaccessible.");
294     }
295 
296     Set<Path> changedPaths = new HashSet<>();
297     for (Path path : createdFilesAndDirectories) {
298       if (Files.isDirectory(path, LinkOption.NOFOLLOW_LINKS)) {
299         // This is a new directory, so changes to it since its creation have not been watched.
300         // We manually traverse the directory tree to register all the new subdirectories and find
301         // all the new subdirectories and files.
302         changedPaths.addAll(registerSubDirectoriesAndReturnContents(path));
303       } else {
304         changedPaths.add(path);
305       }
306     }
307     changedPaths.addAll(deletedOrModifiedFilesAndDirectories);
308     return changedPaths;
309   }

311   /**
312    * Traverses directory tree to register subdirectories. Returns all paths traversed (as absolute
313    * paths).
314    */
315   private Set<Path> registerSubDirectoriesAndReturnContents(Path rootDir) throws IOException {
316     Set<Path> visitedAbsolutePaths = new HashSet<>();
317     // Note that this does not follow symlinks.
318     Files.walkFileTree(rootDir, new WatcherFileVisitor(visitedAbsolutePaths));
319     return visitedAbsolutePaths;
320   }
321 
322   /** File visitor used by Files.walkFileTree() upon traversing subdirectories. */
323   private class WatcherFileVisitor extends SimpleFileVisitor<Path> {
324 
325     private final Set<Path> visitedAbsolutePaths;
326 
327     private WatcherFileVisitor(Set<Path> visitedPaths) {
328       this.visitedAbsolutePaths = visitedPaths;
329     }
330 
331     @Override
332     public FileVisitResult visitFile(Path path, BasicFileAttributes attrs) {
333       Preconditions.checkState(path.isAbsolute(), path);
334       visitedAbsolutePaths.add(path);
335       return FileVisitResult.CONTINUE;
336     }
337 
338     @Override
339     public FileVisitResult preVisitDirectory(Path path, BasicFileAttributes attrs)
340         throws IOException {
341       // It's important that we register the directory before we visit its children. This way we
342       // are guaranteed to see new files/directories either on this #getDiff or the next one.
343       // Otherwise, e.g., an intra-build creation of a child directory will be forever missed if it
344       // happens before the directory is listed as part of the visitation.
345       WatchKey key = path.register(watchService,
346           StandardWatchEventKinds.ENTRY_CREATE,
347           StandardWatchEventKinds.ENTRY_MODIFY,
348           StandardWatchEventKinds.ENTRY_DELETE);
349       Preconditions.checkState(path.isAbsolute(), path);
350       visitedAbsolutePaths.add(path);
351       watchKeyToDirBiMap.put(key, path);
352       return FileVisitResult.CONTINUE;
353     }
354   }
