14  package com.google.devtools.build.lib.sandbox;
15  
16  import com.google.common.base.Preconditions;
17  import com.google.common.collect.ImmutableList;
18  import com.google.common.eventbus.Subscribe;
19  import com.google.devtools.build.lib.actions.ActionContextConsumer;
20  import com.google.devtools.build.lib.actions.ActionContextProvider;
21  import com.google.devtools.build.lib.buildtool.BuildRequest;
22  import com.google.devtools.build.lib.buildtool.buildevent.BuildStartingEvent;
23  import com.google.devtools.build.lib.events.Event;
24  import com.google.devtools.build.lib.runtime.BlazeModule;
25  import com.google.devtools.build.lib.runtime.BlazeRuntime;
26  import com.google.devtools.build.lib.runtime.Command;
27  import com.google.devtools.build.lib.runtime.CommandEnvironment;
28  import com.google.devtools.build.lib.util.OS;
29  import com.google.devtools.common.options.OptionsBase;
30  
31  import java.util.concurrent.ExecutorService;
32  import java.util.concurrent.Executors;
33  import java.util.concurrent.TimeUnit;
34  
35  /**
36   * This module provides the Sandbox spawn strategy.
37   */
38  public class SandboxModule extends BlazeModule {
39    private final ExecutorService backgroundWorkers = Executors.newCachedThreadPool();
40    private BuildRequest buildRequest;
41    private CommandEnvironment env;
42    private BlazeRuntime runtime;
43    private Boolean sandboxingSupported = null;
44  
45    public static final String SANDBOX_NOT_SUPPORTED_MESSAGE =
46        "Sandboxed execution is not supported on your system and thus hermeticity of actions cannot "
47            + "be guaranteed. See http://bazel.io/docs/bazel-user-manual.html#sandboxing for more "
48            + "information. You can turn off this warning via --ignore_unsupported_sandboxing";
49  
50    @Override
51    public Iterable<ActionContextProvider> getActionContextProviders() {
52      Preconditions.checkNotNull(buildRequest);
53      Preconditions.checkNotNull(env);
54  
55      // Cache
56      if (sandboxingSupported == null) {
57        sandboxingSupported = NamespaceSandboxRunner.isSupported(runtime);
58      }
59  
60      if (sandboxingSupported) {
61        return ImmutableList.<ActionContextProvider>of(
62            new SandboxActionContextProvider(runtime, buildRequest, backgroundWorkers));
63      }
64  
65      // For now, sandboxing is only supported on Linux and there's not much point in showing a scary
66      // warning to the user if they can't do anything about it.
67      if (!buildRequest.getOptions(SandboxOptions.class).ignoreUnsupportedSandboxing
68          && OS.getCurrent() == OS.LINUX) {
69        env.getReporter().handle(Event.warn(SANDBOX_NOT_SUPPORTED_MESSAGE));
70      }
71  
72      return ImmutableList.of();
73    }
74  
75    @Override
76    public Iterable<ActionContextConsumer> getActionContextConsumers() {
77      Preconditions.checkNotNull(env);
78  
79      if (sandboxingSupported == null) {
80        sandboxingSupported = NamespaceSandboxRunner.isSupported(runtime);
81      }
82  
83      if (sandboxingSupported) {
84        return ImmutableList.<ActionContextConsumer>of(new SandboxActionContextConsumer());
85      }
86  
87      return ImmutableList.of();
88    }
89  
90    @Override
91    public Iterable<Class<? extends OptionsBase>> getCommandOptions(Command command) {
92      return command.builds()
93          ? ImmutableList.<Class<? extends OptionsBase>>of(SandboxOptions.class)
94          : ImmutableList.<Class<? extends OptionsBase>>of();
95    }
96  
97    @Override
98    public void beforeCommand(Command command, CommandEnvironment env) {
99      this.env = env;
100     this.runtime = env.getRuntime();
101     env.getEventBus().register(this);
102   }
103 
104   @Subscribe
105   public void buildStarting(BuildStartingEvent event) {
106     buildRequest = event.getRequest();
107   }
108 
109   /**
110    * Shut down the background worker pool in the canonical way.
111    *
112    * <p>See https://docs.oracle.com/javase/8/docs/api/java/util/concurrent/ExecutorService.html
113    */
114   @Override
115   public void blazeShutdown() {
116     // Disable new tasks from being submitted
117     backgroundWorkers.shutdown();
118 
119     try {
120       // Wait a while for existing tasks to terminate
121       if (!backgroundWorkers.awaitTermination(5, TimeUnit.SECONDS)) {
122         backgroundWorkers.shutdownNow(); // Cancel currently executing tasks
123 
124         // Wait a while for tasks to respond to being cancelled and force-kill them if necessary
125         // after the timeout.
126         backgroundWorkers.awaitTermination(5, TimeUnit.SECONDS);
127       }
128     } catch (InterruptedException ie) {
129       // (Re-)Cancel if current thread also interrupted
130       backgroundWorkers.shutdownNow();
131 
132       // Preserve interrupt status
133       Thread.currentThread().interrupt();
134     }
135   }
136 }
