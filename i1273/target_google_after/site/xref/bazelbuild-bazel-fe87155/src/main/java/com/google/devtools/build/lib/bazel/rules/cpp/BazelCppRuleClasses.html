116   private static final RuleClass.Configurator<BuildConfiguration, Rule> LIPO_ON_DEMAND =
117       new RuleClass.Configurator<BuildConfiguration, Rule>() {
118     @Override
119     public BuildConfiguration apply(Rule rule, BuildConfiguration configuration) {
120       if (configuration.useDynamicConfigurations()) {
121         // Dynamic configurations don't currently work with LIPO. partially because of lack of
122         // support for TARGET_CONFIG_FOR_LIPO. We can't check for LIPO here because we have
123         // to apply TARGET_CONFIG_FOR_LIPO to determine it, So we just assume LIPO is disabled.
124         // This is safe because Bazel errors out if the two options are combined.
125         return configuration;
126       }
127       BuildConfiguration toplevelConfig =
128           configuration.getConfiguration(CppTransition.TARGET_CONFIG_FOR_LIPO);
129       // If LIPO is enabled, override the default configuration.
130       if (toplevelConfig != null
131           && toplevelConfig.getFragment(CppConfiguration.class).isLipoOptimization()
132           && !configuration.isHostConfiguration()
133           && !configuration.getFragment(CppConfiguration.class).isLipoContextCollector()) {
134         // Switch back to data when the cc_binary is not the LIPO context.
135         return (rule.getLabel().equals(
136             toplevelConfig.getFragment(CppConfiguration.class).getLipoContextLabel()))
137             ? toplevelConfig
138             : configuration.getTransitions().getStaticConfiguration(ConfigurationTransition.DATA);
139       }
140       return configuration;
141     }
142 
143     @Override
144     public String getCategory() {
145       return "lipo";
146     }
147   };

196   /**
197    * Returns the STL prerequisite of the rule.
198    *
199    * <p>If rule has an implicit $stl attribute returns STL version set on the
200    * command line or if not set, the value of the $stl attribute. Returns
201    * {@code null} otherwise.
202    */
203   private static Label getStl(Rule rule, BuildConfiguration original) {
204     Label stl = null;
205     if (rule.getRuleClassObject().hasAttr("$stl", Type.LABEL)) {
206       Label stlConfigLabel = original.getFragment(CppConfiguration.class).getStl();
207       Label stlRuleLabel = RawAttributeMapper.of(rule).get("$stl", Type.LABEL);
208       if (stlConfigLabel == null) {
209         stl = stlRuleLabel;
210       } else if (!stlConfigLabel.equals(rule.getLabel()) && stlRuleLabel != null) {
211         // prevents self-reference and a cycle through standard STL in the dependency graph
212         stl = stlConfigLabel;
213       }
214     }
215     return stl;
216   }

277   public static final class CcDeclRule implements RuleDefinition {
278     @Override
279     public RuleClass build(Builder builder, RuleDefinitionEnvironment env) {
280       return builder
281           /*<!-- #BLAZE_RULE($cc_decl_rule).ATTRIBUTE(abi)[DEPRECATED] -->
282            Platform-specific information string which is used in combination
283             with <code>abi_deps</code>.
284             ${SYNOPSIS}
285             Subject to <a href="#make_variables">"Make" variable</a> substitution.
286             <p>
287               This string typically includes references to one or more "Make" variables of the form
288               <code>"$(VAR)"</code>. The default value is <code>"$(ABI)"</code>.
289             </p>
290             <p>
291               With <code>abi_deps</code>, the regular expression <code>patterns</code> will be
292               matched against this string.
293             </p>
294           <!-- #END_BLAZE_RULE.ATTRIBUTE -->*/
295           // Deprecated in favor of configurable attributes.
296           .add(attr("abi", STRING).value("$(ABI)"))
297           /*<!-- #BLAZE_RULE($cc_decl_rule).ATTRIBUTE(abi_deps)[DEPRECATED] -->
298           The list of architecture-specific dependencies that are processed to create the target.
299           <i>(Dictionary mapping strings to lists of
300              <a href="build-ref.html#labels">labels</a>; optional)</i>
301           <p><i><a href="#configurable-attributes">Configurable attributes</a> is a generalization
302             of the same concept that works for most rules and attributes. It mostly deprecates
303             <code>abi_deps</code>, which we expect to remove it as soon as possible. Use
304             configurable attributes over <code>abi_deps</code> whenever possible. If the former is
305             insufficient for you, please let us know why.</i>
306           </p>
307           <p>Each entry in this dictionary follows the form of
308              <code>'pattern' : ['label1', 'label2', ...]</code>.  If the library's
309              <code>abi</code> attribute is an unanchored match for the regular
310              expression defined in <code>pattern</code>, the corresponding
311              labels are used as dependencies as if they had appeared in the
312              <code>deps</code> list.
313           </p>
314           <p>All pairs with a matching <code>pattern</code> will have their labels
315              used.  If no matches are found, no dependencies will be used.  The
316              ordering is irrelevant.
317           </p>
318           <p>If you want a <code>pattern</code> to not match a particular
319              <code>abi</code>, for example adding a dep on all non-k8 platforms, you
320              can use a negative lookahead pattern.  This would look like
321              <code>(?!k8).*</code>.
322           </p>
323           <p>If using <code>abi_deps</code>, do not provide <code>deps</code>.
324              Instead, use an entry with a <code>pattern</code> of <code>'.*'</code>
325              because that matches everything.  This is also how to share
326              dependencies across multiple different <code>abi</code> values.
327           </p>
328           <p>Typically, this mechanism is used to specify the appropriate set of
329              paths to pre-compiled libraries for the target architecture of the
330              current build.  Such paths are parameterized over "Make" variables
331              such as <code>$(ABI)</code>, <code>$(TARGET_CPU)</code>,
332              <code>$(C_COMPILER)</code>, etc, but since "Make" variables are not
333              allowed in <a href="build-ref.html#labels">labels</a>, the
334              architecture-specific files cannot be specified via the normal
335              <code>srcs</code> attribute. Instead, this mechanism can be used
336              to declare architecture-specific dependent rules for the current
337              target that can specify the correct libraries in their own
338              <code>srcs</code>.
339           </p>
340           <p>This mechanism is also used to specify the appropriate set of
341              dependencies when some targets can't compile for the target architecture
342              of the current build.  In most cases, uses an <code>#ifdef</code>.
343              Only use <code>abi_deps</code> for more significant dependency changes.
344           </p>
345           <!-- #END_BLAZE_RULE.ATTRIBUTE -->*/
346           // Deprecated in favor of configurable attributes.
347           .add(attr("abi_deps", LABEL_LIST_DICT))
348           /*<!-- #BLAZE_RULE($cc_decl_rule).ATTRIBUTE(defines) -->
349           List of defines to add to the compile line.
350           ${SYNOPSIS}
351           Subject to <a href="#make_variables">"Make" variable</a> substitution and
352           <a href="#sh-tokenization">Bourne shell tokenization</a>.
353           Each string, which must consist of a single Bourne shell token,
354           is prepended with <code>-D</code> and added to
355           <code>COPTS</code>.
356           Unlike <a href="#cc_binary.copts"><code>copts</code></a>, these flags are added for the
357           target and every rule that depends on it!  Be very careful, since this may have
358           far-reaching effects.  When in doubt, add "-D" flags to
359           <a href="#cc_binary.copts"><code>copts</code></a> instead.
360           <!-- #END_BLAZE_RULE.ATTRIBUTE -->*/
361           .add(attr("defines", STRING_LIST))
362           /*<!-- #BLAZE_RULE($cc_decl_rule).ATTRIBUTE(includes) -->
363           List of include dirs to be added to the compile line.
364           ${SYNOPSIS}
365           Subject to <a href="#make_variables">"Make variable"</a> substitution.
366           Each string is prepended with <code>-iquote</code> and added to <code>COPTS</code>. Unlike
367           <a href="#cc_binary.copts">COPTS</a>, these flags are added for this rule
368           and every rule that depends on it. (Note: not the rules it depends upon!) Be
369           very careful, since this may have far-reaching effects.  When in doubt, add
370           "-I" flags to <a href="#cc_binary.copts">COPTS</a> instead.
371           <!-- #END_BLAZE_RULE.ATTRIBUTE -->*/
372           .add(attr("includes", STRING_LIST))
373           .add(attr(":lipo_context_collector", LABEL)
374               .cfg(CppTransition.LIPO_COLLECTOR)
375               .value(LIPO_CONTEXT_COLLECTOR)
376               .skipConstraintsCheck())
377           .build();
378     }

380     @Override
381     public Metadata getMetadata() {
382       return RuleDefinition.Metadata.builder()
383           .name("$cc_decl_rule")
384           .type(RuleClassType.ABSTRACT)
385           .ancestors(BaseRuleClasses.RuleBase.class)
386           .build();
387     }

393   public static final class CcRule implements RuleDefinition {
394     @Override
395     public RuleClass build(Builder builder, final RuleDefinitionEnvironment env) {
396       return builder
397           /*<!-- #BLAZE_RULE($cc_rule).ATTRIBUTE(srcs) -->
398           The list of C and C++ files that are processed to create the target.
399           ${SYNOPSIS}
400           These are C/C++ source and header files, either non-generated (normal source
401           code) or generated.
402           <p>All <code>.cc</code>, <code>.c</code>, and <code>.cpp</code> files will
403              be compiled. These might be generated files: if a named file is in
404              the <code>outs</code> of some other rule, this rule
405              will automatically depend on that other rule.
406           </p>
407           <p>A <code>.h</code> file will not be compiled, but will be available for
408              inclusion by sources in this rule. Both <code>.cc</code> and
409              <code>.h</code> files can directly include headers listed in
410              these <code>srcs</code> or in the <code>hdrs</code> of any rule listed in
411              the <code>deps</code> argument. For compatibility with older
412              implementations, and depending on the checking strictness specified by
413              each library, additional headers may be available for inclusion. See
414              <a href="#cc_library.hdrs_check"><code>cc_library.hdrs_check</code></a>.
415           </p>
416           <p>All <code>#include</code>d files must be mentioned in the
417              <code>srcs</code> attribute of this rule, or in the
418              <code>hdrs</code> attribute of referenced <code>cc_library()</code>s.
419              The recommended style is for headers associated with a library to be
420              listed in that library's <code>hdrs</code> attribute, and any remaining
421              headers associated with this rule's sources to be listed in
422              <code>srcs</code>. See <a href="#hdrs">"Header inclusion checking"</a>
423              for a more detailed description.
424           </p>
425           <p>If a rule's name is in the <code>srcs</code>,
426              then this rule automatically depends on that one.
427              If the named rule's <code>outs</code> are C or C++
428              source files, they are compiled into this rule;
429              if they are library files, they are linked in.
430           </p>
431           <p>
432             Permitted <code>srcs</code> file types:
433           </p>
434           <ul>
435             <li>C and C++ source files: <code>.c</code>, <code>.cc</code>, <code>.cpp</code>,
436               <code>.cxx</code>, <code>.c++</code, <code>.C</code></li>
437             <li>C and C++ header files: <code>.h</code>, <code>.hh</code>, <code>.hpp</code>,
438               <code>.hxx</code>, <code>.inc</code></li>
439             <li>Assembler with C preprocessor: <code>.S</code></li>
440             <li>Archive: <code>.a</code>, <code>.pic.a</code></li>
441             <li>"Always link" library: <code>.lo</code>, <code>.pic.lo</code></li>
442             <li>Shared library, versioned or unversioned: <code>.so</code>,
443               <code>.so.<i>version</i></code></li>
444             <li>Object file: <code>.o</code>, <code>.pic.o</code></li>
445           </ul>
446           <p>
447             ...and any rules that produce those files.
448             Different extensions denote different programming languages in
449             accordance with gcc convention.
450           </p>
451           <!-- #END_BLAZE_RULE.ATTRIBUTE -->*/
452           .add(attr("srcs", LABEL_LIST)
453               .direct_compile_time_input()
454               .allowedFileTypes(ALLOWED_SRC_FILES))
455           /*<!-- #BLAZE_RULE($cc_rule).ATTRIBUTE(deps) -->
456           The list of other libraries to be linked in to the binary target.
457           ${SYNOPSIS}
458           <p>These are always <code>cc_library</code> rules.</p>
459           <!-- #END_BLAZE_RULE.ATTRIBUTE -->*/
460           .override(attr("deps", LABEL_LIST)
461               .allowedRuleClasses(DEPS_ALLOWED_RULES)
462               .allowedFileTypes(CppFileTypes.LINKER_SCRIPT)
463               .skipAnalysisTimeFileTypeCheck())
464           /*<!-- #BLAZE_RULE($cc_rule).ATTRIBUTE(linkopts) -->
465           Add these flags to the C++ linker command.
466           ${SYNOPSIS}
467           Subject to <a href="#make_variables">"Make" variable</a> substitution,
468           <a href="#sh-tokenization">Bourne shell tokenization</a>
469           and <a href="#label-expansion">label expansion</a>.
470           Each string in this attribute is added to <code>LINKOPTS</code> before
471           linking the binary target.
472           <p>
473             Each element of this list that does not start with <code>$</code> or <code>-</code> is
474             assumed to be the label of a target in <code>deps</code>.  The
475             list of files generated by that target is appended to the linker
476             options.  An error is reported if the label is invalid, or is
477             not declared in <code>deps</code>.
478           </p>
479           <!-- #END_BLAZE_RULE.ATTRIBUTE -->*/
480           .add(attr("linkopts", STRING_LIST))
481           /*<!-- #BLAZE_RULE($cc_rule).ATTRIBUTE(nocopts) -->
482           Remove matching options from the C++ compilation command.
483           ${SYNOPSIS}
484           Subject to <a href="#make_variables">"Make" variable</a> substitution.
485           The value of this attribute is interpreted as a regular expression.
486           Any preexisting <code>COPTS</code> that match this regular expression
487           (not including values explicitly specified in the rule's <a
488           href="#cc_binary.copts">copts</a> attribute) will be removed from
489           <code>COPTS</code> for purposes of compiling this rule.
490           This attribute should rarely be needed.
491           <!-- #END_BLAZE_RULE.ATTRIBUTE -->*/
492           .add(attr("nocopts", STRING))
493           // cc_library.hdrs_check is an alias for cc_binary.hdrs_check
494           // because many external links depend on it.
495           /*<!-- #BLAZE_RULE($cc_rule).ATTRIBUTE(hdrs_check)[DEPRECATED] -->
496           Deprecated: we will roll out support for C++ modules, which are strict by design.
497           <a id="cc_library.hdrs_check"></a>
498           The strictness of the header file check to be applied to dependents and the
499           source files of this library. Must be one of: <code>"loose"</code>,
500           <code>"warn"</code> or <code>"strict"</code>.
501           ${SYNOPSIS}
502           <p>
503              The three levels of checking operate as follows:
504           </p>
505           <ul>
506              <li><code>"loose"</code>: Any source header file in any of the
507              directories of the containing package may be <code>#include</code>d by
508              sources in this rule and dependent rules. Generated header files must be
509              explicitly listed in <code>hdrs</code> or <code>srcs</code>. This
510              check level is the current default, but it will be phased out.</li>
511              <li><code>"warn"</code>: Enforcement is the same as for
512              <code>"loose"</code>, but any case that would violate the
513              <code>"strict"</code> check will cause a warning to be emitted.
514              This check level is provided to help with the migration to strict.</li>
515              <li><code>"strict"</code>: All <code>#include</code>d files must be explicitly declared
516              somewhere in the <code>hdrs</code> or <code>srcs</code> of the rules providing the
517              libraries or the rules containing the including source. Note that this means that a
518              rule can include any header that is in the <code>hdrs</code> or <code>srcs</code> of
519              any rule it transitively depends on. A stricter layering check is currently being
520              rolled out.</li>
521           </ul>
522           <!-- #END_BLAZE_RULE.ATTRIBUTE -->*/
523           .add(attr("hdrs_check", STRING).value("strict"))
524           /*<!-- #BLAZE_RULE($cc_rule).ATTRIBUTE(linkstatic) -->
525            Link the binary in mostly-static mode.
526            ${SYNOPSIS}
527            By default this option is on for <code>cc_binary</code> and off for <code>cc_test</code>.
528            <p>
529              If enabled, this tells the build tool to link in <code>.a</code>'s instead of
530              <code>.so</code>'s for user libraries whenever possible. Some system libraries may
531              still be linked dynamically, as are libraries for which there's no static library. So
532              the resulting binary will be dynamically linked, hence only <i>mostly</i> static.
533            </p>
534            <p>There are really three different ways to link an executable:</p>
535            <ul>
536            <li> FULLY STATIC, in which everything is linked statically; e.g. "<code>gcc -static
537              foo.o libbar.a libbaz.a -lm</code>".<br/>This mode is enabled by specifying
538              <code>-static</code> in the <a href="#cc_binary.linkopts"><code>linkopts</code></a>
539              attribute.</li>
540            <li> MOSTLY STATIC, in which all user libraries are linked statically (if a static
541              version is available), but where system libraries are linked dynamically, e.g.
542              "<code>gcc foo.o libfoo.a libbaz.a -lm</code>".<br/>This mode is enabled by specifying
543              <code>linkstatic=1</code>.</li>
544            <li> DYNAMIC, in which all libraries are linked dynamically (if a dynamic version is
545              available), e.g. "<code>gcc foo.o libfoo.so libbaz.so -lm</code>".<br/> This mode is
546              enabled by specifying <code>linkstatic=0</code>.</li>
547            </ul>
548            <p>
549            The <code>linkstatic</code> attribute has a different meaning if used on a
550            <a href="#cc_library"><code>cc_library()</code></a> rule.
551            For a C++ library, <code>linkstatic=1</code> indicates that only
552            static linking is allowed, so no <code>.so</code> will be produced.
553            </p>
554            <p>
555            If <code>linkstatic=0</code>, then the build tool will create symlinks to
556            depended-upon shared libraries in the <code>*.runfiles</code> area.
557            </p>
558           <!-- #END_BLAZE_RULE.ATTRIBUTE -->*/
559           .add(attr("linkstatic", BOOLEAN).value(true))
560           .override(attr("$stl", LABEL).value(new Attribute.ComputedDefault() {
561             @Override
562             public Object getDefault(AttributeMap rule) {
563               // Every cc_rule depends implicitly on STL to make
564               // sure that the correct headers are used for inclusion. The only exception is
565               // STL itself to avoid cycles in the dependency graph.
566               Label stl = env.getLabel("//tools/cpp:stl");
567               return rule.getLabel().equals(stl) ? null : stl;
568             }
569           }))
570           .build();
571     }
572     @Override
573     public Metadata getMetadata() {
574       return RuleDefinition.Metadata.builder()
575           .name("$cc_rule")
576           .type(RuleClassType.ABSTRACT)
577           .ancestors(CcDeclRule.class, CcBaseRule.class)
578           .build();
579     }

831 
832 <h4 id="hdrs">Header inclusion checking</h4>
833 
834 <p>
835   All header files that are used in the build must be declared in the <code>hdrs</code> or
836   <code>srcs</code> of <code>cc_*</code> rules. This is enforced by default (i.e.
837   <code><a href="#cc_library.hdrs_check">hdrs_check</a> = 'strict'</code>).
838 </p>
839 
840 <p>
841   For <code>cc_library</code> rules, headers in <code>hdrs</code> comprise the public interface of
842   the library and can be directly included both from the files in <code>hdrs</code> and
843   <code>srcs</code> of the library itself as well as from files in <code>hdrs</code> and
844   <code>srcs</code> of <code>cc_*</code> rules that list the library in their <code>deps</code>.
845   Headers in <code>srcs</code> must only be directly included from the files in <code>hdrs</code>
846   and <code>srcs</code> of the library itself. When deciding whether to put a header into
847   <code>hdrs</code> or <code>srcs</code>, you should ask whether you want consumers of this library
848   to be able to directly include it. This is roughly the same decision as between
849   <code>public</code> and <code>private</code> visibility in programming languages.
850 </p>
851 
852 <p>
853   <code>cc_binary</code> and <code>cc_test</code> rules do not have an exported interface, so they
854   also do not have a <code>hdrs</code> attribute. All headers that belong to the binary or test
855   directly should be listed in the <code>srcs</code>.
856 </p>
857 
858 <p>
859   To illustrate these rules, look at the following example.
860 </p>
861 
862 <pre class="code">
863 cc_binary(
864     name = "foo",
865     srcs = [
866         "foo.cc",
867         "foo.h",
868     ],
869     deps = [":bar"],
870 )
871 
872 cc_library(
873     name = "bar",
874     srcs = [
875         "bar.cc",
876         "bar-impl.h",
877     ],
878     hdrs = ["bar.h"],
879     deps = [":baz"],
880 )
881 
882 cc_library(
883     name = "baz",
884     srcs = [
885         "baz.cc",
886         "baz-impl.h",
887     ],
888     hdrs = ["baz.h"],
889 )
890 </pre>
891 
892 <p>
893   The allowed direct inclusions in this example are listed in the table below. For example
894   <code>foo.cc</code> is allowed to directly include <code>foo.h</code> and <code>bar.h</code>, but
895   not <code>baz.h</code>.
896 </p>
897 
898 <table class="table table-striped table-bordered table-condensed">
899   <thead>
900     <tr><th>Including file</th><th>Allowed inclusions</th></tr>
901   </thead>
902   <tbody>
903     <tr><td>foo.h</td><td>bar.h</td></tr>
904     <tr><td>foo.cc</td><td>foo.h bar.h</td></tr>
905     <tr><td>bar.h</td><td>bar-impl.h baz.h</td></tr>
906     <tr><td>bar-impl.h</td><td>bar.h baz.h</td></tr>
907     <tr><td>bar.cc</td><td>bar.h bar-impl.h baz.h</td></tr>
908     <tr><td>baz.h</td><td>baz-impl.h</td></tr>
909     <tr><td>baz-impl.h</td><td>baz.h</td></tr>
910     <tr><td>baz.cc</td><td>baz.h baz-impl.h</td></tr>
911   </tbody>
912 </table>
913 
914 <p>
915   The inclusion checking rules only apply to <em>direct</em>
916   inclusions. In the example above <code>foo.cc</code> is allowed to
917   include <code>bar.h</code>, which may include <code>baz.h</code>, which in
918   turn is allowed to include <code>baz-impl.h</code>. Technically, the
919   compilation of a <code>.cc</code> file may transitively include any header
920   file in the <code>hdrs</code> or <code>srcs</code> in
921   any <code>cc_library</code> in the transitive <code>deps</code> closure. In
922   this case the compiler may read <code>baz.h</code> and <code>baz-impl.h</code>
923   when compiling <code>foo.cc</code>, but <code>foo.cc</code> must not
924   contain <code>#include "baz.h"</code>. For that to be
925   allowed, <code>baz</code> must be added to the <code>deps</code>
926   of <code>foo</code>.
927 </p>
928 
929 <p>
930   Unfortunately Bazel currently cannot distinguish between direct and transitive
931   inclusions, so it cannot detect error cases where a file illegally includes a
932   header directly that is only allowed to be included transitively. For example,
933   Bazel would not complain if in the example above <code>foo.cc</code> directly
934   includes <code>baz.h</code>. This would be illegal, because <code>foo</code>
935   does not directly depend on <code>baz</code>. Currently, no error is produced
936   in that case, but such error checking may be added in the future.
937 </p>
