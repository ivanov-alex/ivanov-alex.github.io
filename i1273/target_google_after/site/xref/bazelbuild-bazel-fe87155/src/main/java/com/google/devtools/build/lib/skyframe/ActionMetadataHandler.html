46  /**
47   * Cache provided by an {@link ActionExecutionFunction}, allowing Blaze to obtain data from the
48   * graph and to inject data (e.g. file digests) back into the graph.
49   *
50   * <p>Data for the action's inputs is injected into this cache on construction, using the graph as
51   * the source of truth.
52   *
53   * <p>As well, this cache collects data about the action's output files, which is used in three
54   * ways. First, it is served as requested during action execution, primarily by the {@code
55   * ActionCacheChecker} when determining if the action must be rerun, and then after the action is
56   * run, to gather information about the outputs. Second, it is accessed by {@link
57   * ArtifactFunction}s in order to construct {@link ArtifactValue}s. Third, the {@link
58   * FilesystemValueChecker} uses it to determine the set of output files to check for inter-build
59   * modifications. Because all these use cases are slightly different, we must occasionally store two
60   * versions of the data for a value (see {@link #getAdditionalOutputData} for more.
61   */
62  @VisibleForTesting
63  public class ActionMetadataHandler implements MetadataHandler {
64    /** This should never be read directly. Use {@link #getInputFileArtifactValue} instead. */
65    private final Map<Artifact, FileArtifactValue> inputArtifactData;
66    private final ConcurrentMap<Artifact, FileValue> outputArtifactData =
67        new ConcurrentHashMap<>();
68    private final Set<Artifact> omittedOutputs = Sets.newConcurrentHashSet();
69    // See #getAdditionalOutputData for documentation of this field.
70    private final ConcurrentMap<Artifact, FileArtifactValue> additionalOutputData =
71        new ConcurrentHashMap<>();
72    private final Set<Artifact> injectedArtifacts = Sets.newConcurrentHashSet();
73    private final ImmutableSet<Artifact> outputs;
74    private final TimestampGranularityMonitor tsgm;
75  
76    @VisibleForTesting
77    public ActionMetadataHandler(Map<Artifact, FileArtifactValue> inputArtifactData,
78        Iterable<Artifact> outputs,
79        TimestampGranularityMonitor tsgm) {
80      this.inputArtifactData = Preconditions.checkNotNull(inputArtifactData);
81      this.outputs = ImmutableSet.copyOf(outputs);
82      this.tsgm = tsgm;
83    }

94    private static Metadata metadataFromValue(FileArtifactValue value) throws FileNotFoundException {
95      if (value == FileArtifactValue.MISSING_FILE_MARKER
96          || value == FileArtifactValue.OMITTED_FILE_MARKER) {
97        throw new FileNotFoundException();
98      }
99      // If the file is empty or a directory, we need to return the mtime because the action cache
100     // uses mtime to determine if this artifact has changed.  We do not optimize for this code
101     // path (by storing the mtime somewhere) because we eventually may be switching to use digests
102     // for empty files. We want this code path to go away somehow too for directories (maybe by
103     // implementing FileSet in Skyframe).
104     return value.getSize() > 0
105         ? new Metadata(value.getDigest())
106         : new Metadata(value.getModifiedTime());
107   }

109   @Override
110   public Metadata getMetadata(Artifact artifact) throws IOException {
111     Metadata metadata = getRealMetadata(artifact);
112     return artifact.isConstantMetadata() ? Metadata.CONSTANT_METADATA : metadata;
113   }
114 
115   @Nullable
116   private FileArtifactValue getInputFileArtifactValue(ActionInput input) {
117     if (outputs.contains(input) || !(input instanceof Artifact)) {
118       return null;
119     }
120     return Preconditions.checkNotNull(inputArtifactData.get(input), input);
121   }

123   /**
124    * We cache data for constant-metadata artifacts, even though it is technically unnecessary,
125    * because the data stored in this cache is consumed by various parts of Blaze via the {@link
126    * ActionExecutionValue} (for now, {@link FilesystemValueChecker} and {@link ArtifactFunction}).
127    * It is simpler for those parts if every output of the action is present in the cache. However,
128    * we must not return the actual metadata for a constant-metadata artifact.
129    */
130   private Metadata getRealMetadata(Artifact artifact) throws IOException {
131     FileArtifactValue value = getInputFileArtifactValue(artifact);
132     if (value != null) {
133       return metadataFromValue(value);
134     }
135     if (artifact.isSourceArtifact()) {
136       // A discovered input we didn't have data for.
137       // TODO(bazel-team): Change this to an assertion once Skyframe has native input discovery, so
138       // all inputs will already have metadata known.
139       return null;
140     } else if (artifact.isMiddlemanArtifact()) {
141       // A middleman artifact's data was either already injected from the action cache checker using
142       // #setDigestForVirtualArtifact, or it has the default middleman value.
143       value = additionalOutputData.get(artifact);
144       if (value != null) {
145         return metadataFromValue(value);
146       }
147       value = FileArtifactValue.DEFAULT_MIDDLEMAN;
148       FileArtifactValue oldValue = additionalOutputData.putIfAbsent(artifact, value);
149       checkInconsistentData(artifact, oldValue, value);
150       return metadataFromValue(value);
151     }
152     FileValue fileValue = outputArtifactData.get(artifact);
153     if (fileValue != null) {
154       // Non-middleman artifacts should only have additionalOutputData if they have
155       // outputArtifactData. We don't assert this because of concurrency possibilities, but at least
156       // we don't check additionalOutputData unless we expect that we might see the artifact there.
157       value = additionalOutputData.get(artifact);
158       // If additional output data is present for this artifact, we use it in preference to the
159       // usual calculation.
160       if (value != null) {
161         return metadataFromValue(value);
162       }
163       if (!fileValue.exists()) {
164         throw new FileNotFoundException(artifact.prettyPrint() + " does not exist");
165       }
166       return new Metadata(Preconditions.checkNotNull(fileValue.getDigest(), artifact));
167     }
168     // We do not cache exceptions besides nonexistence here, because it is unlikely that the file
169     // will be requested from this cache too many times.
170     fileValue = fileValueFromArtifact(artifact, null, tsgm);
171     FileValue oldFileValue = outputArtifactData.putIfAbsent(artifact, fileValue);
172     checkInconsistentData(artifact, oldFileValue, value);
173     return maybeStoreAdditionalData(artifact, fileValue, null);
174   }
175 
176   /**
177    * Check that the new {@code data} we just calculated for an {@code artifact} agrees with the
178    * {@code oldData} (presumably calculated concurrently), if it was present.
179    */
180   // Not private only because used by SkyframeActionExecutor's metadata handler.
181   static void checkInconsistentData(Artifact artifact,
182       @Nullable Object oldData, Object data) throws IOException {
183     if (oldData != null && !oldData.equals(data)) {
184       // Another thread checked this file since we looked at the map, and got a different answer
185       // than we did. Presumably the user modified the file between reads.
186       throw new IOException("Data for " + artifact.prettyPrint() + " changed to " + data
187           + " after it was calculated as " + oldData);
188     }
189   }

191   /**
192    * See {@link #getAdditionalOutputData} for why we sometimes need to store additional data, even
193    * for normal (non-middleman) artifacts.
194    */
195   @Nullable
196   private Metadata maybeStoreAdditionalData(Artifact artifact, FileValue data,
197       @Nullable byte[] injectedDigest) throws IOException {
198     if (!data.exists()) {
199       // Nonexistent files should only occur before executing an action.
200       throw new FileNotFoundException(artifact.prettyPrint() + " does not exist");
201     }
202     boolean isFile = data.isFile();
203     boolean useDigest = DigestUtils.useFileDigest(isFile, isFile ? data.getSize() : 0);
204     if (useDigest && data.getDigest() != null) {
205       // We do not need to store the FileArtifactValue separately -- the digest is in the file value
206       // and that is all that is needed for this file's metadata.
207       return new Metadata(data.getDigest());
208     }
209     // Unfortunately, the FileValue does not contain enough information for us to calculate the
210     // corresponding FileArtifactValue -- either the metadata must use the modified time, which we
211     // do not expose in the FileValue, or the FileValue didn't store the digest So we store the
212     // metadata separately.
213     // Use the FileValue's digest if no digest was injected, or if the file can't be digested.
214     injectedDigest = injectedDigest != null || !isFile ? injectedDigest : data.getDigest();
215     FileArtifactValue value =
216         FileArtifactValue.create(artifact, isFile, isFile ? data.getSize() : 0, injectedDigest);
217     FileArtifactValue oldValue = additionalOutputData.putIfAbsent(artifact, value);
218     checkInconsistentData(artifact, oldValue, value);
219     return metadataFromValue(value);
220   }

222   @Override
223   public void setDigestForVirtualArtifact(Artifact artifact, Digest digest) {
224     Preconditions.checkState(artifact.isMiddlemanArtifact(), artifact);
225     Preconditions.checkNotNull(digest, artifact);
226     additionalOutputData.put(artifact,
227         FileArtifactValue.createMiddleman(digest.asMetadata().digest));
228   }

230   @Override
231   public void injectDigest(ActionInput output, FileStatus statNoFollow, byte[] digest) {
232     if (output instanceof Artifact) {
233       Artifact artifact = (Artifact) output;
234       Preconditions.checkState(injectedArtifacts.add(artifact), artifact);
235       FileValue fileValue;
236       try {
237         // This call may do an unnecessary call to Path#getFastDigest to see if the digest is
238         // readily available. We cannot pass the digest in, though, because if it is not available
239         // from the filesystem, this FileValue will not compare equal to another one created for the
240         // same file, because the other one will be missing its digest.
241         fileValue = fileValueFromArtifact(artifact, FileStatusWithDigestAdapter.adapt(statNoFollow),
242             tsgm);
243         // Ensure the digest supplied matches the actual digest if it exists.
244         byte[] fileDigest = fileValue.getDigest();
245         if (fileDigest != null && !Arrays.equals(digest, fileDigest)) {
246           BaseEncoding base16 = BaseEncoding.base16();
247           String digestString = (digest != null) ? base16.encode(digest) : "null";
248           String fileDigestString = (fileDigest != null) ? base16.encode(fileDigest) : "null";
249           throw new IllegalStateException("Expected digest " + digestString + " for artifact "
250               + artifact + ", but got " + fileDigestString + " (" + fileValue + ")");
251         }
252         outputArtifactData.put(artifact, fileValue);
253       } catch (IOException e) {
254         // Do nothing - we just failed to inject metadata. Real error handling will be done later,
255         // when somebody will try to access that file.
256         return;
257       }
258       // If needed, insert additional data. Note that this can only be true if the file is empty or
259       // the filesystem does not support fast digests. Since we usually only inject digests when
260       // running with a filesystem that supports fast digests, this is fairly unlikely.
261       try {
262         maybeStoreAdditionalData(artifact, fileValue, digest);
263       } catch (IOException e) {
264         if (fileValue.getSize() != 0) {
265           // Empty files currently have their mtimes examined, and so could throw. No other files
266           // should throw, since all filesystem access has already been done.
267           throw new IllegalStateException(
268               "Filesystem should not have been accessed while injecting data for "
269                   + artifact.prettyPrint(), e);
270         }
271         // Ignore exceptions for empty files, as above.
272       }
273     }
274   }

276   @Override
277   public void markOmitted(ActionInput output) {
278     if (output instanceof Artifact) {
279       Artifact artifact = (Artifact) output;
280       Preconditions.checkState(omittedOutputs.add(artifact), artifact);
281       additionalOutputData.put(artifact, FileArtifactValue.OMITTED_FILE_MARKER);
282     }
283   }

285   @Override
286   public boolean artifactOmitted(Artifact artifact) {
287     return omittedOutputs.contains(artifact);
288   }

290   @Override
291   public void discardOutputMetadata() {
292     Preconditions.checkState(injectedArtifacts.isEmpty(),
293         "Artifacts cannot be injected before action execution: %s", injectedArtifacts);
294     Preconditions.checkState(omittedOutputs.isEmpty(),
295         "Artifacts cannot be marked omitted before action execution: %s", omittedOutputs);
296     outputArtifactData.clear();
297     additionalOutputData.clear();
298   }

300   @Override
301   public boolean artifactExists(Artifact artifact) {
302     Preconditions.checkState(!artifactOmitted(artifact), artifact);
303     return getMetadataMaybe(artifact) != null;
304   }

306   @Override
307   public boolean isRegularFile(Artifact artifact) {
308     // Currently this method is used only for genrule input directory checks. If we need to call
309     // this on output artifacts too, this could be more efficient.
310     FileArtifactValue value = getInputFileArtifactValue(artifact);
311     if (value != null && value.isFile()) {
312       return true;
313     }
314     return artifact.getPath().isFile();
315   }

330   /**
331    * Returns data for any output files whose metadata was not computable from the corresponding
332    * entry in {@link #getOutputData}.
333    *
334    * <p>There are three reasons why we might not be able to compute metadata for an artifact from
335    * the FileValue. First, middleman artifacts have no corresponding FileValues. Second, if
336    * computing a file's digest is not fast, the FileValue does not do so, so a file on a filesystem
337    * without fast digests has to have its metadata stored separately. Third, some files' metadata
338    * (directories, empty files) contain their mtimes, which the FileValue does not expose, so that
339    * has to be stored separately.
340    *
341    * <p>Note that for files that need digests, we can't easily inject the digest in the FileValue
342    * because it would complicate equality-checking on subsequent builds -- if our filesystem doesn't
343    * do fast digests, the comparison value would not have a digest.
344    */
345   Map<Artifact, FileArtifactValue> getAdditionalOutputData() {
346     return additionalOutputData;
347   }

349   static FileValue fileValueFromArtifact(Artifact artifact,
350       @Nullable FileStatusWithDigest statNoFollow, TimestampGranularityMonitor tsgm)
351       throws IOException {
352     Path path = artifact.getPath();
353     RootedPath rootedPath =
354         RootedPath.toRootedPath(artifact.getRoot().getPath(), artifact.getRootRelativePath());
355     if (statNoFollow == null) {
356       statNoFollow = FileStatusWithDigestAdapter.adapt(path.statIfFound(Symlinks.NOFOLLOW));
357       if (statNoFollow == null) {
358         return FileValue.value(rootedPath, FileStateValue.NONEXISTENT_FILE_STATE_NODE,
359             rootedPath, FileStateValue.NONEXISTENT_FILE_STATE_NODE);
360       }
361     }
362     Path realPath = path;
363     // We use FileStatus#isSymbolicLink over Path#isSymbolicLink to avoid the unnecessary stat
364     // done by the latter.
365     if (statNoFollow.isSymbolicLink()) {
366       realPath = path.resolveSymbolicLinks();
367       // We need to protect against symlink cycles since FileValue#value assumes it's dealing with a
368       // file that's not in a symlink cycle.
369       if (realPath.equals(path)) {
370         throw new IOException("symlink cycle");
371       }
372     }
373     RootedPath realRootedPath = RootedPath.toRootedPathMaybeUnderRoot(realPath,
374         ImmutableList.of(artifact.getRoot().getPath()));
375     FileStateValue fileStateValue;
376     FileStateValue realFileStateValue;
377     try {
378       fileStateValue = FileStateValue.createWithStatNoFollow(rootedPath, statNoFollow, tsgm);
379       // TODO(bazel-team): consider avoiding a 'stat' here when the symlink target hasn't changed
380       // and is a source file (since changes to those are checked separately).
381       realFileStateValue = realPath.equals(path) ? fileStateValue
382           : FileStateValue.create(realRootedPath, tsgm);
383     } catch (InconsistentFilesystemException e) {
384       throw new IOException(e);
385     }
386     return FileValue.value(rootedPath, fileStateValue, realRootedPath, realFileStateValue);
387   }
