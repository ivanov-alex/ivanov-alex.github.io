105   @Test
106   public void testGetVariableNames() throws Exception {
107     Environment outerEnv;
108     Environment innerEnv;
109     try (Mutability mut = Mutability.create("outer")) {
110       outerEnv = Environment.builder(mut)
111           .setGlobals(Environment.BUILD).build()
112           .update("foo", "bar")
113           .update("wiz", 3);
114     }
115     try (Mutability mut = Mutability.create("inner")) {
116       innerEnv = Environment.builder(mut)
117           .setGlobals(outerEnv.getGlobals()).build()
118           .update("foo", "bat")
119           .update("quux", 42);
120     }
121 
122     assertEquals(Sets.newHashSet("foo", "wiz",
123             "False", "None", "True",
124             "-", "bool", "dict", "enumerate", "int", "len", "list",
125             "range", "repr", "select", "sorted", "str", "zip"),
126         outerEnv.getVariableNames());
127     assertEquals(Sets.newHashSet("foo", "wiz", "quux",
128             "False", "None", "True",
129             "-", "bool", "dict", "enumerate", "int", "len", "list",
130             "range", "repr", "select", "sorted", "str", "zip"),
131         innerEnv.getVariableNames());
132   }

154   @Test
155   public void testFrozen() throws Exception {
156     Environment env;
157     try (Mutability mutability = Mutability.create("testFrozen")) {
158       env = Environment.builder(mutability)
159           .setGlobals(Environment.BUILD).setEventHandler(Environment.FAIL_FAST_HANDLER).build();
160       env.update("x", 1);
161       assertEquals(env.lookup("x"), 1);
162       env.update("y", 2);
163       assertEquals(env.lookup("y"), 2);
164       assertEquals(env.lookup("x"), 1);
165       env.update("x", 3);
166       assertEquals(env.lookup("x"), 3);
167     }
168     try {
169       // This update to an existing variable should fail because the environment was frozen.
170       env.update("x", 4);
171       throw new Exception("failed to fail"); // not an AssertionError like fail()
172     } catch (AssertionError e) {
173       assertThat(e).hasMessage("Can't update x to 4 in frozen environment");
174     }
175     try {
176       // This update to a new variable should also fail because the environment was frozen.
177       env.update("newvar", 5);
178       throw new Exception("failed to fail"); // not an AssertionError like fail()
179     } catch (AssertionError e) {
180       assertThat(e).hasMessage("Can't update newvar to 5 in frozen environment");
181     }
182   }

184   @Test
185   public void testReadOnly() throws Exception {
186     Environment env = newSkylarkEnvironment()
187         .setup("special_var", 42)
188         .update("global_var", 666);
189 
190     // We don't even get a runtime exception trying to modify these,
191     // because we get compile-time exceptions even before we reach runtime!
192     try {
193       env.eval("special_var = 41");
194       throw new AssertionError("failed to fail");
195     } catch (IllegalArgumentException e) {
196       assertThat(e).hasMessage("ERROR 1:1: Variable special_var is read only");
197     }
198 
199     try {
200       env.eval("def foo(x): x += global_var; global_var = 36; return x", "foo(1)");
201       throw new AssertionError("failed to fail");
202     } catch (EvalExceptionWithStackTrace e) {
203       assertThat(e.getMessage()).contains("Variable 'global_var' is referenced before assignment. "
204           + "The variable is defined in the global scope.");
205     }
206   }
