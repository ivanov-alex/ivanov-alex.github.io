68    @Before

132   @Test
133   public void testResolveArtifactWithUpLevelFailsCleanly() throws Exception {
134     // We need a package in the root directory to make every exec path (even one with up-level
135     // references) be in a package.
136     Map<PackageIdentifier, Root> packageRoots = ImmutableMap.of(
137         PackageIdentifier.createInDefaultRepo(new PathFragment("")), clientRoot);
138     artifactFactory.setPackageRoots(packageRoots);
139     PathFragment outsideWorkspace = new PathFragment("../foo");
140     PathFragment insideWorkspace =
141         new PathFragment("../" + clientRoot.getPath().getBaseName() + "/foo");
142     assertNull(artifactFactory.resolveSourceArtifact(outsideWorkspace));
143     assertNull("Up-level-containing paths that descend into the right workspace aren't allowed",
144             artifactFactory.resolveSourceArtifact(insideWorkspace));
145     MockPackageRootResolver packageRootResolver = new MockPackageRootResolver();
146     packageRootResolver.setPackageRoots(packageRoots);
147     Map<PathFragment, Artifact> result = new HashMap<>();
148     result.put(insideWorkspace, null);
149     result.put(outsideWorkspace, null);
150     assertThat(
151         artifactFactory.resolveSourceArtifacts(ImmutableList.of(insideWorkspace, outsideWorkspace),
152             packageRootResolver).entrySet()).containsExactlyElementsIn(result.entrySet());
153   }

171   @Test
172   public void testSetGeneratingActionIdempotenceNewActionGraph() throws Exception {
173     Artifact a = artifactFactory.getDerivedArtifact(fooRelative, outRoot, NULL_ARTIFACT_OWNER);
174     Artifact b = artifactFactory.getDerivedArtifact(barRelative, outRoot, NULL_ARTIFACT_OWNER);
175     MutableActionGraph actionGraph = new MapBasedActionGraph();
176     Action originalAction = new ActionsTestUtil.NullAction(NULL_ACTION_OWNER, a);
177     actionGraph.registerAction(originalAction);
178 
179     // Creating a second Action referring to the Artifact should create a conflict.
180     try {
181       Action action = new ActionsTestUtil.NullAction(NULL_ACTION_OWNER, a, b);
182       actionGraph.registerAction(action);
183       fail();
184     } catch (ActionConflictException e) {
185       assertSame(a, e.getArtifact());
186       assertSame(originalAction, actionGraph.getGeneratingAction(a));
187     }
188   }

210   private static class MockPackageRootResolver implements PackageRootResolver {
211     private Map<PathFragment, Root> packageRoots = Maps.newHashMap();
212 
213     public void setPackageRoots(Map<PackageIdentifier, Root> packageRoots) {
214       for (Entry<PackageIdentifier, Root> packageRoot : packageRoots.entrySet()) {
215         this.packageRoots.put(packageRoot.getKey().getPackageFragment(), packageRoot.getValue());
216       }
217     }
218 
219     @Override
220     public Map<PathFragment, Root> findPackageRoots(Iterable<PathFragment> execPaths) {
221       Map<PathFragment, Root> result = new HashMap<>();
222       for (PathFragment execPath : execPaths) {
223         for (PathFragment dir = execPath.getParentDirectory(); dir != null;
224             dir = dir.getParentDirectory()) {
225           if (packageRoots.get(dir) != null) {
226             result.put(execPath, packageRoots.get(dir));
227           }
228         }
229         if (result.get(execPath) == null) {
230           result.put(execPath, null);
231         }
232       }
233       return result;
234     }
235   }

237   @Test
238   public void testArtifactDeserializationWithoutReusedArtifacts() throws Exception {
239     PathFragment derivedPath = outRoot.getExecPath().getRelative("fruit/banana");
240     artifactFactory.clear();
241     artifactFactory.setDerivedArtifactRoots(ImmutableList.of(outRoot));
242     MockPackageRootResolver rootResolver = new MockPackageRootResolver();
243     rootResolver.setPackageRoots(
244         ImmutableMap.of(PackageIdentifier.createInDefaultRepo(""), clientRoot));
245     Artifact artifact1 = artifactFactory.deserializeArtifact(derivedPath, rootResolver);
246     Artifact artifact2 = artifactFactory.deserializeArtifact(derivedPath, rootResolver);
247     assertEquals(artifact1, artifact2);
248     assertNull(artifact1.getOwner());
249     assertNull(artifact2.getOwner());
250     assertEquals(derivedPath, artifact1.getExecPath());
251     assertEquals(derivedPath, artifact2.getExecPath());
252 
253     // Source artifacts are always reused
254     PathFragment sourcePath = clientRoot.getExecPath().getRelative("fruit/mango");
255     artifact1 = artifactFactory.deserializeArtifact(sourcePath, rootResolver);
256     artifact2 = artifactFactory.deserializeArtifact(sourcePath, rootResolver);
257     assertSame(artifact1, artifact2);
258     assertEquals(sourcePath, artifact1.getExecPath());
259   }
