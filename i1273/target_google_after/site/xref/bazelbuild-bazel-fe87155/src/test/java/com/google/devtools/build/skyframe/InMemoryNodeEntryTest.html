554   @Test
555   public void checkDepsOneByOne() {
556     NodeEntry entry = new InMemoryNodeEntry();
557     entry.addReverseDepAndCheckIfDone(null); // Start evaluation.
558     List<SkyKey> deps = new ArrayList<>();
559     for (int ii = 0; ii < 10; ii++) {
560       SkyKey dep = key(Integer.toString(ii));
561       deps.add(dep);
562       addTemporaryDirectDep(entry, dep);
563       entry.signalDep();
564     }
565     setValue(entry, new IntegerValue(5), /*errorInfo=*/null, /*graphVersion=*/0L);
566     entry.markDirty(/*isChanged=*/false);
567     entry.addReverseDepAndCheckIfDone(null); // Start new evaluation.
568     assertEquals(NodeEntry.DirtyState.CHECK_DEPENDENCIES, entry.getDirtyState());
569     for (int ii = 0; ii < 10; ii++) {
570       assertThat(entry.getNextDirtyDirectDeps()).containsExactly(deps.get(ii)).inOrder();
571       addTemporaryDirectDep(entry, deps.get(ii));
572       assertTrue(entry.signalDep(new IntVersion(0L)));
573       if (ii < 9) {

668   private static void addTemporaryDirectDeps(NodeEntry entry, SkyKey... keys) {

675     entry.addTemporaryDirectDeps(helper);
