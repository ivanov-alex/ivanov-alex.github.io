28  /**
29   * Tests for {@link BaseFunction}.
30   * This tests the argument processing by BaseFunction
31   * between the outer call(posargs, kwargs, ast, env) and the inner call(args, ast, env).
32   */
33  @RunWith(JUnit4.class)
34  public class BaseFunctionTest extends EvaluationTestCase {
35  
36    /**
37     * Handy implementation of {@link BaseFunction} that returns all its args as a list.
38     * (We'd use SkylarkList.tuple, but it can't handle null.)
39     */
40    private static class TestingBaseFunction extends BaseFunction {
41      TestingBaseFunction(FunctionSignature signature) {
42        super("mixed", signature);
43      }
44  
45      @Override
46      public Object call(Object[] arguments, FuncallExpression ast, Environment env) {
47        return Arrays.asList(arguments);
48      }
49    }
50  
51    private void checkBaseFunction(BaseFunction func, String callExpression, String expectedOutput)
52        throws Exception {
53      setUp();
54      update(func.getName(), func);
55  
56      if (expectedOutput.charAt(0) == '[') { // a tuple => expected to pass
57        assertEquals("Wrong output for " + callExpression,
58            expectedOutput, eval(callExpression).toString());
59  
60      } else { // expected to fail with an exception
61        try {
62          eval(callExpression);
63          fail();
64        } catch (EvalException e) {
65          assertEquals("Wrong exception for " + callExpression,
66              expectedOutput, e.getMessage());
67        }
68      }
69    }
70  
71    private static final String[] BASE_FUNCTION_EXPRESSIONS = {
72      "mixed()",
73      "mixed(1)",
74      "mixed(1, 2)",
75      "mixed(1, 2, 3)",
76      "mixed(1, 2, wiz=3, quux=4)",
77      "mixed(foo=1)",
78      "mixed(bar=2)",
79      "mixed(foo=1, bar=2)",
80      "mixed(bar=2, foo=1)",
81      "mixed(2, foo=1)",
82      "mixed(bar=2, foo=1, wiz=3)",
83    };
84  
85    public void checkBaseFunctions(boolean onlyNamedArguments,
86        String expectedSignature, String... expectedResults) throws Exception {
87      BaseFunction func = new TestingBaseFunction(
88          onlyNamedArguments
89          ? FunctionSignature.namedOnly(1, "foo", "bar")
90          : FunctionSignature.of(1, "foo", "bar"));
91  
92      assertThat(func.toString()).isEqualTo(expectedSignature);
93  
94      for (int i = 0; i < BASE_FUNCTION_EXPRESSIONS.length; i++) {
95        String expr = BASE_FUNCTION_EXPRESSIONS[i];
96        String expected = expectedResults[i];
97        checkBaseFunction(func, expr, expected);
98      }
99    }

101   @Test
102   public void testNoSurplusArguments() throws Exception {
103     checkBaseFunctions(false, "mixed(foo, bar = ?)",
104         "insufficient arguments received by mixed(foo, bar = ?) (got 0, expected at least 1)",
105         "[1, null]",
106         "[1, 2]",
107         "too many (3) positional arguments in call to mixed(foo, bar = ?)",
108         "unexpected keywords 'quux', 'wiz' in call to mixed(foo, bar = ?)",
109         "[1, null]",
110         "missing mandatory positional argument 'foo' while calling mixed(foo, bar = ?)",
111         "[1, 2]",
112         "[1, 2]",
113         "argument 'foo' passed both by position and by name in call to mixed(foo, bar = ?)",
114         "unexpected keyword 'wiz' in call to mixed(foo, bar = ?)");
115   }
116 
117   @Test
118   public void testOnlyNamedArguments() throws Exception {
119     checkBaseFunctions(true, "mixed(*, foo, bar = ?)",
120         "missing mandatory keyword arguments in call to mixed(*, foo, bar = ?)",
121         "mixed(*, foo, bar = ?) does not accept positional arguments, but got 1",
122         "mixed(*, foo, bar = ?) does not accept positional arguments, but got 2",
123         "mixed(*, foo, bar = ?) does not accept positional arguments, but got 3",
124         "mixed(*, foo, bar = ?) does not accept positional arguments, but got 2",
125         "[1, null]",
126         "missing mandatory named-only argument 'foo' while calling mixed(*, foo, bar = ?)",
127         "[1, 2]",
128         "[1, 2]",
129         "mixed(*, foo, bar = ?) does not accept positional arguments, but got 1",
130         "unexpected keyword 'wiz' in call to mixed(*, foo, bar = ?)");
131   }

133   @Test
134   @SuppressWarnings("unchecked")
135   public void testKwParam() throws Exception {
136     eval("def foo(a, b, c=3, d=4, *args, e, f, g=7, h=8, **kwargs):\n"
137         + "  return (a, b, c, d, e, f, g, h, args, kwargs)\n"
138         + "v1 = foo(1, 2, e=5, f=6)\n"
139         + "v2 = foo(1, *['x', 'y', 'z', 't'], h=9, e=5, f=6, i=0)\n"
140         + "def bar(**kwargs):\n"
141         + "  return kwargs\n"
142         + "b1 = bar(name='foo', type='jpg', version=42)\n"
143         + "b2 = bar()\n");
144 
145     assertThat(Printer.repr(lookup("v1")))
146         .isEqualTo("(1, 2, 3, 4, 5, 6, 7, 8, (), {})");
147     assertThat(Printer.repr(lookup("v2")))
148         .isEqualTo("(1, \"x\", \"y\", \"z\", 5, 6, 7, 9, (\"t\",), {\"i\": 0})");
149 
150     // NB: the conversion to a TreeMap below ensures the keys are sorted.
151     assertThat(Printer.repr(
152         new TreeMap<String, Object>((Map<String, Object>) lookup("b1"))))
153         .isEqualTo("{\"name\": \"foo\", \"type\": \"jpg\", \"version\": 42}");
154     assertThat(Printer.repr(lookup("b2"))).isEqualTo("{}");
155   }
