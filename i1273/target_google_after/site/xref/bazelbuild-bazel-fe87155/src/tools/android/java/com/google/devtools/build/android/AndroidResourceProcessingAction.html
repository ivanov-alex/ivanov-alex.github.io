52  /**
53   * Provides an entry point for the resource processing using the AOSP build tools.
54   *
55   * <pre>
56   * Example Usage:
57   *   java/com/google/build/android/AndroidResourceProcessingAction\
58   *      --sdkRoot path/to/sdk\
59   *      --aapt path/to/sdk/aapt\
60   *      --annotationJar path/to/sdk/annotationJar\
61   *      --adb path/to/sdk/adb\
62   *      --zipAlign path/to/sdk/zipAlign\
63   *      --androidJar path/to/sdk/androidJar\
64   *      --manifest path/to/manifest\
65   *      --primaryData path/to/resources:path/to/assets:path/to/manifest:path/to/R.txt
66   *      --data p/t/res1:p/t/assets1:p/t/1/AndroidManifest.xml:p/t/1/R.txt,\
67   *             p/t/res2:p/t/assets2:p/t/2/AndroidManifest.xml:p/t/2/R.txt
68   *      --generatedSourcePath path/to/write/generated/sources
69   *      --packagePath path/to/write/archive.ap_
70   *      --srcJarOutput path/to/write/archive.srcjar
71   * </pre>
72   */
73  public class AndroidResourceProcessingAction {

268   public static void main(String[] args) {
269     final Stopwatch timer = Stopwatch.createStarted();
270     OptionsParser optionsParser = OptionsParser.newOptionsParser(Options.class);
271     optionsParser.parseAndExitUponError(args);
272     options = optionsParser.getOptions(Options.class);
273     FileSystem fileSystem = FileSystems.getDefault();
274     Path working = fileSystem.getPath("").toAbsolutePath();
275     Path mergedAssets = working.resolve("merged_assets");
276     Path mergedResources = working.resolve("merged_resources");
277 
278     final AndroidResourceProcessor resourceProcessor =
279         new AndroidResourceProcessor(STD_LOGGER);
280 
281     final AndroidSdkTools sdkTools = new AndroidSdkTools(options.apiVersion,
282         options.aapt,
283         options.annotationJar,
284         options.adb,
285         options.zipAlign,
286         options.androidJar,
287         STD_LOGGER);
288 
289     try {
290 
291       Path expandedOut = Files.createTempDirectory("tmp-expanded");
292       expandedOut.toFile().deleteOnExit();
293       Path deduplicatedOut = Files.createTempDirectory("tmp-deduplicated");
294       deduplicatedOut.toFile().deleteOnExit();
295       
296       LOGGER.fine(String.format("Setup finished at %sms", timer.elapsed(TimeUnit.MILLISECONDS)));
297 
298       final ImmutableList<DirectoryModifier> modifiers = ImmutableList.of(
299           new PackedResourceTarExpander(expandedOut, working),
300           new FileDeDuplicator(Hashing.murmur3_128(), deduplicatedOut, working));
301 
302       final AndroidBuilder builder = sdkTools.createAndroidBuilder();
303 
304       final MergedAndroidData mergedData = resourceProcessor.mergeData(
305           options.primaryData,
306           options.data,
307           mergedResources,
308           mergedAssets,
309           modifiers,
310           useAaptCruncher() ? builder.getAaptCruncher() : null,
311           true);
312 
313       LOGGER.fine(String.format("Merging finished at %sms", timer.elapsed(TimeUnit.MILLISECONDS)));
314       final Path filteredResources = fileSystem.getPath("resources-filtered");
315       final Path densityManifest = fileSystem.getPath("manifest-filtered/AndroidManifest.xml");
316       final DensityFilteredAndroidData filteredData = mergedData.filter(
317           new DensitySpecificResourceFilter(options.densities, filteredResources, working),
318           new DensitySpecificManifestProcessor(options.densities, densityManifest));
319       LOGGER.fine(
320           String.format("Density filtering finished at %sms",
321               timer.elapsed(TimeUnit.MILLISECONDS)));
322       resourceProcessor.processResources(
323           builder,
324           options.packageType,
325           options.debug,
326           options.packageForR,
327           new FlagAaptOptions(),
328           options.resourceConfigs,
329           options.applicationId,
330           options.versionCode,
331           options.versionName,
332           filteredData,
333           options.data,
334           working.resolve("manifest"),
335           options.generatedSourcePath,
336           options.packagePath,
337           options.proguardOutput);
338       LOGGER.fine(String.format("appt finished at %sms", timer.elapsed(TimeUnit.MILLISECONDS)));
339       if (options.srcJarOutput != null) {
340         resourceProcessor.createSrcJar(options.generatedSourcePath, options.srcJarOutput);
341       }
342       if (options.rOutput != null) {
343         resourceProcessor.copyRToOutput(options.generatedSourcePath, options.rOutput);
344       }
345       if (options.symbolsTxtOut != null) {
346         resourceProcessor.copyRToOutput(options.generatedSourcePath, options.symbolsTxtOut);
347       }
348       LOGGER.fine(String.format("Packaging finished at %sms",
349           timer.elapsed(TimeUnit.MILLISECONDS)));
350     } catch (MergingException e) {
351       LOGGER.log(java.util.logging.Level.SEVERE, "Error during merging resources", e);
352       System.exit(1);
353     } catch (IOException | InterruptedException | LoggedErrorException e) {
354       LOGGER.log(java.util.logging.Level.SEVERE, "Error during processing resources", e);
355       System.exit(2);
356     } catch (Exception e) {
357       LOGGER.log(java.util.logging.Level.SEVERE, "Unexpected", e);
358       System.exit(3);
359     }
360     LOGGER.fine(String.format("Resources processed in %sms", timer.elapsed(TimeUnit.MILLISECONDS)));
361     // AOSP code can leave dangling threads.
362     System.exit(0);
363   }
364 
365   private static boolean useAaptCruncher() {
366     // If the value was set, use that.
367     if (options.useAaptCruncher != TriState.AUTO) {
368       return options.useAaptCruncher == TriState.YES;
369     }
370     // By default png cruncher shouldn't be invoked on a library -- the work is just thrown away.
371     return options.packageType != VariantConfiguration.Type.LIBRARY;
372   }
373 
374   private static final class FlagAaptOptions implements AaptOptions {
375     @Override
376     public boolean getUseAaptPngCruncher() {
377       return options.useAaptCruncher != TriState.NO;
378     }
379 
380     @Override
381     public Collection<String> getNoCompress() {
382       if (!options.uncompressedExtensions.isEmpty()) {
383         return options.uncompressedExtensions;
384       }
385       return null;
386     }
387 
388     @Override
389     public String getIgnoreAssets() {
390       if (!options.assetsToIgnore.isEmpty()) {
391         return Joiner.on(":").join(options.assetsToIgnore);
392       }
393       return null;
394     }
395 
396     @Override
397     public boolean getFailOnMissingConfigEntry() {
398       return false;
399     }
