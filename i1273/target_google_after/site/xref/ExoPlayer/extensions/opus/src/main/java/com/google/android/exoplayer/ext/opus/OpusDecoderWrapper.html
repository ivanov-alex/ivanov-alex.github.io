23  /**
24   * Wraps {@link OpusDecoder}, exposing a higher level decoder interface.
25   *
26   * @author vigneshv@google.com (Vignesh Venkatasubramanian)
27   */
28  /* package */ class OpusDecoderWrapper extends Thread {
29  
30    public static final int FLAG_END_OF_STREAM = 1;
31    public static final int FLAG_RESET_DECODER = 2;
32  
33    private static final int INPUT_BUFFER_SIZE = 960 * 6;
34    private static final int OUTPUT_BUFFER_SIZE = 960 * 6 * 2;
35    private static final int NUM_BUFFERS = 16;
36    private static final int DEFAULT_SEEK_PRE_ROLL = 3840;
37  
38    private final Object lock;
39    private final OpusHeader opusHeader;
40  
41    private final LinkedList<InputBuffer> queuedInputBuffers;
42    private final LinkedList<OutputBuffer> queuedOutputBuffers;
43    private final InputBuffer[] availableInputBuffers;
44    private final OutputBuffer[] availableOutputBuffers;
45    private int availableInputBufferCount;
46    private int availableOutputBufferCount;
47  
48    private int skipSamples;
49    private boolean flushDecodedOutputBuffer;
50    private boolean released;
51  
52    private int seekPreRoll;
53  
54    private OpusDecoderException decoderException;
55  
56    /**
57     * @param headerBytes Opus header data that is used to initialize the decoder. For WebM Container,
58     *    this comes from the CodecPrivate Track element.
59     * @param codecDelayNs Delay in nanoseconds added by the codec at the beginning. For WebM
60     *    Container, this comes from the CodecDelay Track Element. Can be -1 in which case the value
61     *    from the codec header will be used.
62     * @param seekPreRollNs Duration in nanoseconds of samples to discard when there is a
63     *    discontinuity. For WebM Container, this comes from the SeekPreRoll Track Element. Can be -1
64     *    in which case the default value of 80ns will be used.
65     * @throws OpusDecoderException if an exception occurs when initializing the decoder.
66     */
67    public OpusDecoderWrapper(byte[] headerBytes, long codecDelayNs,
68        long seekPreRollNs) throws OpusDecoderException {
69      lock = new Object();
70      opusHeader = parseOpusHeader(headerBytes);
71      skipSamples = (codecDelayNs == -1) ? opusHeader.skipSamples : nsToSamples(codecDelayNs);
72      seekPreRoll = (seekPreRoll == -1) ? DEFAULT_SEEK_PRE_ROLL : nsToSamples(seekPreRollNs);
73      queuedInputBuffers = new LinkedList<>();
74      queuedOutputBuffers = new LinkedList<>();
75      availableInputBuffers = new InputBuffer[NUM_BUFFERS];
76      availableOutputBuffers = new OutputBuffer[NUM_BUFFERS];
77      availableInputBufferCount = NUM_BUFFERS;
78      availableOutputBufferCount = NUM_BUFFERS;
79      for (int i = 0; i < NUM_BUFFERS; i++) {
80        availableInputBuffers[i] = new InputBuffer();
81        availableOutputBuffers[i] = new OutputBuffer();
82      }
83    }
84  
85    public InputBuffer getInputBuffer() throws OpusDecoderException {
86      synchronized (lock) {
87        maybeThrowDecoderError();
88        if (availableInputBufferCount == 0) {
89          return null;
90        }
91        InputBuffer inputBuffer = availableInputBuffers[--availableInputBufferCount];
92        inputBuffer.reset();
93        return inputBuffer;
94      }
95    }
96  
97    public void queueInputBuffer(InputBuffer inputBuffer) throws OpusDecoderException {
98      synchronized (lock) {
99        maybeThrowDecoderError();
100       queuedInputBuffers.addLast(inputBuffer);
101       maybeNotifyDecodeLoop();
102     }
103   }
104 
105   public OutputBuffer dequeueOutputBuffer() throws OpusDecoderException {
106     synchronized (lock) {
107       maybeThrowDecoderError();
108       if (queuedOutputBuffers.isEmpty()) {
109         return null;
110       }
111       return queuedOutputBuffers.removeFirst();
112     }
113   }
114 
115   public void releaseOutputBuffer(OutputBuffer outputBuffer) throws OpusDecoderException {
116     synchronized (lock) {
117       maybeThrowDecoderError();
118       outputBuffer.reset();
119       availableOutputBuffers[availableOutputBufferCount++] = outputBuffer;
120       maybeNotifyDecodeLoop();
121     }
122   }
123 
124   public void flush() {
125     synchronized (lock) {
126       flushDecodedOutputBuffer = true;
127       while (!queuedInputBuffers.isEmpty()) {
128         availableInputBuffers[availableInputBufferCount++] = queuedInputBuffers.removeFirst();
129       }
130       while (!queuedOutputBuffers.isEmpty()) {
131         availableOutputBuffers[availableOutputBufferCount++] = queuedOutputBuffers.removeFirst();
132       }
133     }
134   }
135 
136   public void release() {
137     synchronized (lock) {
138       released = true;
139       lock.notify();
140     }
141     try {
142       join();
143     } catch (InterruptedException e) {
144       Thread.currentThread().interrupt();
145     }
146   }
147 
148   private void maybeThrowDecoderError() throws OpusDecoderException {
149     if (decoderException != null) {
150       throw decoderException;
151     }
152   }
153 
154   /**
155    * Notifies the decode loop if there exists a queued input buffer and an available output buffer
156    * to decode into.
157    * <p>
158    * Should only be called whilst synchronized on the lock object.
159    */
160   private void maybeNotifyDecodeLoop() {
161     if (!queuedInputBuffers.isEmpty() && availableOutputBufferCount > 0) {
162       lock.notify();
163     }
164   }
165 
166   @Override
167   public void run() {
168     OpusDecoder decoder = null;
169     try {
170       decoder = new OpusDecoder(opusHeader);
171       while (decodeBuffer(decoder)) {
172         // Do nothing.
173       }
174     } catch (OpusDecoderException e) {
175       synchronized (lock) {
176         decoderException = e;
177       }
178     } catch (InterruptedException e) {
179       // Shouldn't ever happen.
180     } finally {
181       if (decoder != null) {
182         decoder.close();
183       }
184     }
185   }
186 
187   private boolean decodeBuffer(OpusDecoder decoder) throws InterruptedException,
188       OpusDecoderException {
189     InputBuffer inputBuffer;
190     OutputBuffer outputBuffer;
191 
192     // Wait until we have an input buffer to decode, and an output buffer to decode into.
193     synchronized (lock) {
194       while (!released && (queuedInputBuffers.isEmpty() || availableOutputBufferCount == 0)) {
195         lock.wait();
196       }
197       if (released) {
198         return false;
199       }
200       inputBuffer = queuedInputBuffers.removeFirst();
201       outputBuffer = availableOutputBuffers[--availableOutputBufferCount];
202       flushDecodedOutputBuffer = false;
203     }
204 
205     // Decode.
206     if (inputBuffer.getFlag(FLAG_END_OF_STREAM)) {
207       outputBuffer.setFlag(FLAG_END_OF_STREAM);
208     } else {
209       if (inputBuffer.getFlag(FLAG_RESET_DECODER)) {
210         decoder.reset();
211         // When seeking to 0, skip number of samples as specified in opus header. When seeking to
212         // any other time, skip number of samples as specified by seek preroll.
213         skipSamples = (inputBuffer.sampleHolder.timeUs == 0) ? opusHeader.skipSamples : seekPreRoll;
214       }
215       SampleHolder sampleHolder = inputBuffer.sampleHolder;
216       sampleHolder.data.position(sampleHolder.data.position() - sampleHolder.size);
217       outputBuffer.timestampUs = sampleHolder.timeUs;
218       outputBuffer.size = decoder.decode(sampleHolder.data, sampleHolder.size,
219           outputBuffer.data, outputBuffer.data.capacity());
220       outputBuffer.data.position(0);
221       if (skipSamples > 0) {
222         int bytesPerSample = opusHeader.channelCount * 2;
223         int skipBytes = skipSamples * bytesPerSample;
224         if (outputBuffer.size < skipBytes) {
225           skipSamples -= outputBuffer.size / bytesPerSample;
226           outputBuffer.size = 0;
227         } else {
228           skipSamples = 0;
229           outputBuffer.data.position(skipBytes);
230           outputBuffer.size -= skipBytes;
231         }
232       }
233     }
234 
235     synchronized (lock) {
236       if (flushDecodedOutputBuffer
237           || inputBuffer.sampleHolder.isDecodeOnly()
238           || outputBuffer.size == 0) {
239         // In the following cases, we make the output buffer available again rather than queuing it
240         // to be consumed:
241         // 1) A flush occured whilst we were decoding.
242         // 2) The input sample has decodeOnly flag set.
243         // 3) We skip the entire buffer due to skipSamples being greater than bytes decoded.
244         outputBuffer.reset();
245         availableOutputBuffers[availableOutputBufferCount++] = outputBuffer;
246       } else {
247         // Queue the decoded output buffer to be consumed.
248         queuedOutputBuffers.addLast(outputBuffer);
249       }
250       // Make the input buffer available again.
251       availableInputBuffers[availableInputBufferCount++] = inputBuffer;
252     }
253 
254     return true;
255   }

257   private OpusHeader parseOpusHeader(byte[] headerBytes) throws OpusDecoderException {
258     final int maxChannelCount = 8;
259     final int maxChannelCountWithDefaultLayout = 2;
260     final int headerSize = 19;
261     final int headerChannelCountOffset = 9;
262     final int headerSkipSamplesOffset = 10;
263     final int headerGainOffset = 16;
264     final int headerChannelMappingOffset = 18;
265     final int headerNumStreamsOffset = headerSize;
266     final int headerNumCoupledOffset = headerNumStreamsOffset + 1;
267     final int headerStreamMapOffset = headerNumStreamsOffset + 2;
268     OpusHeader opusHeader = new OpusHeader();
269     try {
270       // Opus streams are always decoded at 48000 hz.
271       opusHeader.sampleRate = 48000;
272       opusHeader.channelCount = headerBytes[headerChannelCountOffset];
273       if (opusHeader.channelCount > maxChannelCount) {
274         throw new OpusDecoderException("Invalid channel count: " + opusHeader.channelCount);
275       }
276       opusHeader.skipSamples = readLittleEndian16(headerBytes, headerSkipSamplesOffset);
277       opusHeader.gain = readLittleEndian16(headerBytes, headerGainOffset);
278       opusHeader.channelMapping = headerBytes[headerChannelMappingOffset];
279 
280       if (opusHeader.channelMapping == 0) {
281         // If there is no channel mapping, use the defaults.
282         if (opusHeader.channelCount > maxChannelCountWithDefaultLayout) {
283           throw new OpusDecoderException("Invalid Header, missing stream map.");
284         }
285         opusHeader.numStreams = 1;
286         opusHeader.numCoupled = (opusHeader.channelCount > 1) ? 1 : 0;
287         opusHeader.streamMap[0] = 0;
288         opusHeader.streamMap[1] = 1;
289       } else {
290         // Read the channel mapping.
291         opusHeader.numStreams = headerBytes[headerNumStreamsOffset];
292         opusHeader.numCoupled = headerBytes[headerNumCoupledOffset];
293         for (int i = 0; i < opusHeader.channelCount; i++) {
294           opusHeader.streamMap[i] = headerBytes[headerStreamMapOffset + i];
295         }
296       }
297       return opusHeader;
298     } catch (ArrayIndexOutOfBoundsException e) {
299       throw new OpusDecoderException("Header size is too small.");
300     }
301   }
302 
303   private int readLittleEndian16(byte[] input, int offset) {
304     int value = input[offset];
305     value |= input[offset + 1] << 8;
306     return value;
307   }
308 
309   private int nsToSamples(long ns) {
310     return (int) (ns * opusHeader.sampleRate / 1000000000);
311   }

313   /* package */ static final class InputBuffer {
314 
315     public final SampleHolder sampleHolder;
316 
317     public int flags;
318 
319     public InputBuffer() {
320       sampleHolder = new SampleHolder(SampleHolder.BUFFER_REPLACEMENT_MODE_DIRECT);
321       sampleHolder.data = ByteBuffer.allocateDirect(INPUT_BUFFER_SIZE);
322     }
323 
324     public void reset() {
325       sampleHolder.data.clear();
326       flags = 0;
327     }
328 
329     public void setFlag(int flag) {
330       flags |= flag;
331     }
332 
333     public boolean getFlag(int flag) {
334       return (flags & flag) == flag;
335     }
336 
337   }
338 
339   /* package */ static final class OutputBuffer {
340 
341     public ByteBuffer data;
342     public int size;
343     public long timestampUs;
344     public int flags;
345 
346     public OutputBuffer() {
347       data = ByteBuffer.allocateDirect(OUTPUT_BUFFER_SIZE);
348     }
349 
350     public void reset() {
351       data.clear();
352       size = 0;
353       flags = 0;
354     }
355 
356     public void setFlag(int flag) {
357       flags |= flag;
358     }
359 
360     public boolean getFlag(int flag) {
361       return (flags & flag) == flag;
362     }
363 
364   }
