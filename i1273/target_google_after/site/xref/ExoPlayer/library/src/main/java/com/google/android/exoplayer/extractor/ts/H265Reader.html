35    private static final String TAG = "H265Reader";

37    // nal_unit_type values from H.265/HEVC (2014) Table 7-1.
38    private static final int RASL_R = 9;
39    private static final int BLA_W_LP = 16;
40    private static final int BLA_W_RADL = 17;
41    private static final int BLA_N_LP = 18;
42    private static final int IDR_W_RADL = 19;
43    private static final int IDR_N_LP = 20;
44    private static final int CRA_NUT = 21;
45    private static final int VPS_NUT = 32;
46    private static final int SPS_NUT = 33;
47    private static final int PPS_NUT = 34;
48    private static final int PREFIX_SEI_NUT = 39;
49    private static final int SUFFIX_SEI_NUT = 40;
50  
51    // State that should not be reset on seek.
52    private boolean hasOutputFormat;
53  
54    // State that should be reset on seek.
55    private final SeiReader seiReader;
56    private final boolean[] prefixFlags;
57    private final NalUnitTargetBuffer vps;
58    private final NalUnitTargetBuffer sps;
59    private final NalUnitTargetBuffer pps;
60    private final NalUnitTargetBuffer prefixSei;
61    private final NalUnitTargetBuffer suffixSei; // TODO: Are both needed?
62    private boolean foundFirstSample;
63    private long totalBytesWritten;
64  
65    // Per sample state that gets reset at the start of each sample.
66    private boolean isKeyframe;
67    private long samplePosition;
68    private long sampleTimeUs;
69  
70    // Scratch variables to avoid allocations.
71    private final ParsableByteArray seiWrapper;
72  
73    public H265Reader(TrackOutput output, SeiReader seiReader) {
74      super(output);
75      this.seiReader = seiReader;
76      prefixFlags = new boolean[3];
77      vps = new NalUnitTargetBuffer(VPS_NUT, 128);
78      sps = new NalUnitTargetBuffer(SPS_NUT, 128);
79      pps = new NalUnitTargetBuffer(PPS_NUT, 128);
80      prefixSei = new NalUnitTargetBuffer(PREFIX_SEI_NUT, 128);
81      suffixSei = new NalUnitTargetBuffer(SUFFIX_SEI_NUT, 128);
82      seiWrapper = new ParsableByteArray();
83    }
84  
85    @Override
86    public void seek() {
87      seiReader.seek();
88      NalUnitUtil.clearPrefixFlags(prefixFlags);
89      vps.reset();
90      sps.reset();
91      pps.reset();
92      prefixSei.reset();
93      suffixSei.reset();
94      foundFirstSample = false;
95      totalBytesWritten = 0;
96    }
97  
98    @Override
99    public void consume(ParsableByteArray data, long pesTimeUs, boolean startOfPacket) {
100     while (data.bytesLeft() > 0) {
101       int offset = data.getPosition();
102       int limit = data.limit();
103       byte[] dataArray = data.data;
104 
105       // Append the data to the buffer.
106       totalBytesWritten += data.bytesLeft();
107       output.sampleData(data, data.bytesLeft());
108 
109       // Scan the appended data, processing NAL units as they are encountered
110       while (offset < limit) {
111         int nextNalUnitOffset = NalUnitUtil.findNalUnit(dataArray, offset, limit, prefixFlags);
112         if (nextNalUnitOffset < limit) {
113           // We've seen the start of a NAL unit.
114 
115           // This is the length to the start of the unit. It may be negative if the NAL unit
116           // actually started in previously consumed data.
117           int lengthToNalUnit = nextNalUnitOffset - offset;
118           if (lengthToNalUnit > 0) {
119             feedNalUnitTargetBuffersData(dataArray, offset, nextNalUnitOffset);
120           }
121 
122           int nalUnitType = NalUnitUtil.getH265NalUnitType(dataArray, nextNalUnitOffset);
123           int bytesWrittenPastNalUnit = limit - nextNalUnitOffset;
124           if (isFirstSliceSegmentInPic(dataArray, nextNalUnitOffset)) {
125             if (foundFirstSample) {
126               if (isKeyframe && !hasOutputFormat && vps.isCompleted() && sps.isCompleted()
127                   && pps.isCompleted()) {
128                 parseMediaFormat(vps, sps, pps);
129               }
130               int flags = isKeyframe ? C.SAMPLE_FLAG_SYNC : 0;
131               int size = (int) (totalBytesWritten - samplePosition) - bytesWrittenPastNalUnit;
132               output.sampleMetadata(sampleTimeUs, flags, size, bytesWrittenPastNalUnit, null);
133             }
134             foundFirstSample = true;
135             samplePosition = totalBytesWritten - bytesWrittenPastNalUnit;
136             sampleTimeUs = pesTimeUs;
137             isKeyframe = isRandomAccessPoint(nalUnitType);
138           }
139 
140           // If the length to the start of the unit is negative then we wrote too many bytes to the
141           // NAL buffers. Discard the excess bytes when notifying that the unit has ended.
142           feedNalUnitTargetEnd(pesTimeUs, lengthToNalUnit < 0 ? -lengthToNalUnit : 0);
143           // Notify the start of the next NAL unit.
144           feedNalUnitTargetBuffersStart(nalUnitType);
145           // Continue scanning the data.
146           offset = nextNalUnitOffset + 3;
147         } else {
148           feedNalUnitTargetBuffersData(dataArray, offset, limit);
149           offset = limit;
150         }
151       }
152     }
153   }
154 
155   @Override
156   public void packetFinished() {
157     // Do nothing.
158   }

160   private void feedNalUnitTargetBuffersStart(int nalUnitType) {
161     if (!hasOutputFormat) {
162       vps.startNalUnit(nalUnitType);
163       sps.startNalUnit(nalUnitType);
164       pps.startNalUnit(nalUnitType);
165     }
166     prefixSei.startNalUnit(nalUnitType);
167     suffixSei.startNalUnit(nalUnitType);
168   }
169 
170   private void feedNalUnitTargetBuffersData(byte[] dataArray, int offset, int limit) {
171     if (!hasOutputFormat) {
172       vps.appendToNalUnit(dataArray, offset, limit);
173       sps.appendToNalUnit(dataArray, offset, limit);
174       pps.appendToNalUnit(dataArray, offset, limit);
175     }
176     prefixSei.appendToNalUnit(dataArray, offset, limit);
177     suffixSei.appendToNalUnit(dataArray, offset, limit);
178   }
179 
180   private void feedNalUnitTargetEnd(long pesTimeUs, int discardPadding) {
181     vps.endNalUnit(discardPadding);
182     sps.endNalUnit(discardPadding);
183     pps.endNalUnit(discardPadding);
184     if (prefixSei.endNalUnit(discardPadding)) {
185       int unescapedLength = NalUnitUtil.unescapeStream(prefixSei.nalData, prefixSei.nalLength);
186       seiWrapper.reset(prefixSei.nalData, unescapedLength);
187 
188       // Skip the NAL prefix and type.
189       seiWrapper.skipBytes(5);
190       seiReader.consume(seiWrapper, pesTimeUs, true);
191     }
192     if (suffixSei.endNalUnit(discardPadding)) {
193       int unescapedLength = NalUnitUtil.unescapeStream(suffixSei.nalData, suffixSei.nalLength);
194       seiWrapper.reset(suffixSei.nalData, unescapedLength);
195 
196       // Skip the NAL prefix and type.
197       seiWrapper.skipBytes(5);
198       seiReader.consume(seiWrapper, pesTimeUs, true);
199     }
200   }

202   private void parseMediaFormat(NalUnitTargetBuffer vps, NalUnitTargetBuffer sps,
203       NalUnitTargetBuffer pps) {
204     // Build codec-specific data.
205     byte[] csd = new byte[vps.nalLength + sps.nalLength + pps.nalLength];
206     System.arraycopy(vps.nalData, 0, csd, 0, vps.nalLength);
207     System.arraycopy(sps.nalData, 0, csd, vps.nalLength, sps.nalLength);
208     System.arraycopy(pps.nalData, 0, csd, vps.nalLength + sps.nalLength, pps.nalLength);
209 
210     // Unescape and then parse the SPS NAL unit, as per H.265/HEVC (2014) 7.3.2.2.1.
211     NalUnitUtil.unescapeStream(sps.nalData, sps.nalLength);
212     ParsableBitArray bitArray = new ParsableBitArray(sps.nalData);
213     bitArray.skipBits(40 + 4); // NAL header, sps_video_parameter_set_id
214     int maxSubLayersMinus1 = bitArray.readBits(3);
215     bitArray.skipBits(1); // sps_temporal_id_nesting_flag
216 
217     // profile_tier_level(1, sps_max_sub_layers_minus1)
218     bitArray.skipBits(88); // if (profilePresentFlag) {...}
219     bitArray.skipBits(8); // general_level_idc
220     int toSkip = 0;
221     for (int i = 0; i < maxSubLayersMinus1; i++) {
222       if (bitArray.readBits(1) == 1) { // sub_layer_profile_present_flag[i]
223         toSkip += 89;
224       }
225       if (bitArray.readBits(1) == 1) { // sub_layer_level_present_flag[i]
226         toSkip += 8;
227       }
228     }
229     bitArray.skipBits(toSkip);
230     if (maxSubLayersMinus1 > 0) {
231       bitArray.skipBits(2 * (8 - maxSubLayersMinus1));
232     }
233 
234     bitArray.readUnsignedExpGolombCodedInt(); // sps_seq_parameter_set_id
235     int chromaFormatIdc = bitArray.readUnsignedExpGolombCodedInt();
236     if (chromaFormatIdc == 3) {
237       bitArray.skipBits(1); // separate_colour_plane_flag
238     }
239     int picWidthInLumaSamples = bitArray.readUnsignedExpGolombCodedInt();
240     int picHeightInLumaSamples = bitArray.readUnsignedExpGolombCodedInt();
241     if (bitArray.readBit()) { // conformance_window_flag
242       int confWinLeftOffset = bitArray.readUnsignedExpGolombCodedInt();
243       int confWinRightOffset = bitArray.readUnsignedExpGolombCodedInt();
244       int confWinTopOffset = bitArray.readUnsignedExpGolombCodedInt();
245       int confWinBottomOffset = bitArray.readUnsignedExpGolombCodedInt();
246       // H.265/HEVC (2014) Table 6-1
247       int subWidthC = chromaFormatIdc == 1 || chromaFormatIdc == 2 ? 2 : 1;
248       int subHeightC = chromaFormatIdc == 1 ? 2 : 1;
249       picWidthInLumaSamples -= subWidthC * (confWinLeftOffset + confWinRightOffset);
250       picHeightInLumaSamples -= subHeightC * (confWinTopOffset + confWinBottomOffset);
251     }
252     bitArray.readUnsignedExpGolombCodedInt(); // bit_depth_luma_minus8
253     bitArray.readUnsignedExpGolombCodedInt(); // bit_depth_chroma_minus8
254     int log2MaxPicOrderCntLsbMinus4 = bitArray.readUnsignedExpGolombCodedInt();
255     // for (i = sps_sub_layer_ordering_info_present_flag ? 0 : sps_max_sub_layers_minus1; ...)
256     for (int i = bitArray.readBit() ? 0 : maxSubLayersMinus1; i <= maxSubLayersMinus1; i++) {
257       bitArray.readUnsignedExpGolombCodedInt(); // sps_max_dec_pic_buffering_minus1[i]
258       bitArray.readUnsignedExpGolombCodedInt(); // sps_max_num_reorder_pics[i]
259       bitArray.readUnsignedExpGolombCodedInt(); // sps_max_latency_increase_plus1[i]
260     }
261     bitArray.readUnsignedExpGolombCodedInt(); // log2_min_luma_coding_block_size_minus3
262     bitArray.readUnsignedExpGolombCodedInt(); // log2_diff_max_min_luma_coding_block_size
263     bitArray.readUnsignedExpGolombCodedInt(); // log2_min_luma_transform_block_size_minus2
264     bitArray.readUnsignedExpGolombCodedInt(); // log2_diff_max_min_luma_transform_block_size
265     bitArray.readUnsignedExpGolombCodedInt(); // max_transform_hierarchy_depth_inter
266     bitArray.readUnsignedExpGolombCodedInt(); // max_transform_hierarchy_depth_intra
267     // if (scaling_list_enabled_flag) { if (sps_scaling_list_data_present_flag) {...}}
268     if (bitArray.readBit() && bitArray.readBit()) {
269       skipScalingList(bitArray);
270     }
271     bitArray.skipBits(2); // amp_enabled_flag (1), sample_adaptive_offset_enabled_flag (1)
272     if (bitArray.readBit()) { // pcm_enabled_flag
273       // pcm_sample_bit_depth_luma_minus1 (4), pcm_sample_bit_depth_chroma_minus1 (4)
274       bitArray.skipBits(4);
275       bitArray.readUnsignedExpGolombCodedInt(); // log2_min_pcm_luma_coding_block_size_minus3
276       bitArray.readUnsignedExpGolombCodedInt(); // log2_diff_max_min_pcm_luma_coding_block_size
277       bitArray.skipBits(1); // pcm_loop_filter_disabled_flag
278     }
279     // Skips all short term reference picture sets.
280     skipShortTermRefPicSets(bitArray);
281     if (bitArray.readBit()) { // long_term_ref_pics_present_flag
282       // num_long_term_ref_pics_sps
283       for (int i = 0; i < bitArray.readUnsignedExpGolombCodedInt(); i++) {
284         int ltRefPicPocLsbSpsLength = log2MaxPicOrderCntLsbMinus4 + 4;
285         // lt_ref_pic_poc_lsb_sps[i], used_by_curr_pic_lt_sps_flag[i]
286         bitArray.skipBits(ltRefPicPocLsbSpsLength + 1);
287       }
288     }
289     bitArray.skipBits(2); // sps_temporal_mvp_enabled_flag, strong_intra_smoothing_enabled_flag
290     float pixelWidthHeightRatio = 1;
291     if (bitArray.readBit()) { // vui_parameters_present_flag
292       if (bitArray.readBit()) { // aspect_ratio_info_present_flag
293         int aspectRatioIdc = bitArray.readBits(8);
294         if (aspectRatioIdc == NalUnitUtil.EXTENDED_SAR) {
295           int sarWidth = bitArray.readBits(16);
296           int sarHeight = bitArray.readBits(16);
297           if (sarWidth != 0 && sarHeight != 0) {
298             pixelWidthHeightRatio = (float) sarWidth / sarHeight;
299           }
300         } else if (aspectRatioIdc < NalUnitUtil.ASPECT_RATIO_IDC_VALUES.length) {
301           pixelWidthHeightRatio = NalUnitUtil.ASPECT_RATIO_IDC_VALUES[aspectRatioIdc];
302         } else {
303           Log.w(TAG, "Unexpected aspect_ratio_idc value: " + aspectRatioIdc);
304         }
305       }
306     }
307 
308     output.format(MediaFormat.createVideoFormat(MimeTypes.VIDEO_H265, MediaFormat.NO_VALUE,
309         C.UNKNOWN_TIME_US, picWidthInLumaSamples, picHeightInLumaSamples, pixelWidthHeightRatio,
310         Collections.singletonList(csd)));
311     hasOutputFormat = true;
312   }

314   /** Skips scaling_list_data(). See H.265/HEVC (2014) 7.3.4. */
315   private void skipScalingList(ParsableBitArray bitArray) {
316     for (int sizeId = 0; sizeId < 4; sizeId++) {
317       for (int matrixId = 0; matrixId < 6; matrixId += sizeId == 3 ? 3 : 1) {
318         if (!bitArray.readBit()) { // scaling_list_pred_mode_flag[sizeId][matrixId]
319           // scaling_list_pred_matrix_id_delta[sizeId][matrixId]
320           bitArray.readUnsignedExpGolombCodedInt();
321         } else {
322           int coefNum = Math.min(64, 1 << (4 + sizeId << 1));
323           if (sizeId > 1) {
324             // scaling_list_dc_coef_minus8[sizeId - 2][matrixId]
325             bitArray.readSignedExpGolombCodedInt();
326           }
327           for (int i = 0; i < coefNum; i++) {
328             bitArray.readSignedExpGolombCodedInt(); // scaling_list_delta_coef
329           }
330         }
331       }
332     }
333   }

335   /**
336    * Reads the number of short term reference picture sets in a SPS as ue(v), then skips all of
337    * them. See H.265/HEVC (2014) 7.3.7.
338    */
339   private static void skipShortTermRefPicSets(ParsableBitArray bitArray) {
340     int numShortTermRefPicSets = bitArray.readUnsignedExpGolombCodedInt();
341     boolean interRefPicSetPredictionFlag = false;
342     int numNegativePics = 0;
343     int numPositivePics = 0;
344     // As this method applies in a SPS, the only element of NumDeltaPocs accessed is the previous
345     // one, so we just keep track of that rather than storing the whole array.
346     // RefRpsIdx = stRpsIdx - (delta_idx_minus1 + 1) and delta_idx_minus1 is always zero in SPS.
347     int previousNumDeltaPocs = 0;
348     for (int stRpsIdx = 0; stRpsIdx < numShortTermRefPicSets; stRpsIdx++) {
349       if (stRpsIdx != 0) {
350         interRefPicSetPredictionFlag = bitArray.readBit();
351       }
352       if (interRefPicSetPredictionFlag) {
353         bitArray.skipBits(1); // delta_rps_sign
354         bitArray.readUnsignedExpGolombCodedInt(); // abs_delta_rps_minus1
355         for (int j = 0; j <= previousNumDeltaPocs; j++) {
356           if (bitArray.readBit()) { // used_by_curr_pic_flag[j]
357             bitArray.skipBits(1); // use_delta_flag[j]
358           }
359         }
360       } else {
361         numNegativePics = bitArray.readUnsignedExpGolombCodedInt();
362         numPositivePics = bitArray.readUnsignedExpGolombCodedInt();
363         previousNumDeltaPocs = numNegativePics + numPositivePics;
364         for (int i = 0; i < numNegativePics; i++) {
365           bitArray.readUnsignedExpGolombCodedInt(); // delta_poc_s0_minus1[i]
366           bitArray.skipBits(1); // used_by_curr_pic_s0_flag[i]
367         }
368         for (int i = 0; i < numPositivePics; i++) {
369           bitArray.readUnsignedExpGolombCodedInt(); // delta_poc_s1_minus1[i]
370           bitArray.skipBits(1); // used_by_curr_pic_s1_flag[i]
371         }
372       }
373     }
374   }

376   /**
377    * Returns whether the NAL unit is a random access point.
378    */
379   private static boolean isRandomAccessPoint(int nalUnitType) {
380     return nalUnitType == BLA_W_LP || nalUnitType == BLA_W_RADL || nalUnitType == BLA_N_LP
381         || nalUnitType == IDR_W_RADL || nalUnitType == IDR_N_LP || nalUnitType == CRA_NUT;
382   }

384   /**
385    * Returns whether the NAL unit in {@code data} starting at {@code offset} contains the first
386    * slice in a picture.
387    *
388    * @param data The data to read.
389    * @param offset The start offset of a NAL unit. Must lie between {@code -3} (inclusive) and
390    *     {@code data.length - 3} (exclusive).
391    * @return Whether the NAL unit contains the first slice in a picture.
392    */
393   public static boolean isFirstSliceSegmentInPic(byte[] data, int offset) {
394     int nalUnitType = NalUnitUtil.getH265NalUnitType(data, offset);
395     // Check the flag in NAL units that contain a slice_segment_layer_rbsp RBSP.
396     if ((nalUnitType <= RASL_R) || (nalUnitType >= BLA_W_LP && nalUnitType <= CRA_NUT)) {
397       return (data[offset + 5] & 0x80) != 0;
398     }
399     return false;
400   }
