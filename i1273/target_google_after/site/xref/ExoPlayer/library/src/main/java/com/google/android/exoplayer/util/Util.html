231    * Returns the index of the largest value in an array that is less than (or optionally equal to)
232    * a specified key.
233    * <p>
234    * The search is performed using a binary search algorithm, and so the array must be sorted.
235    *
236    * @param a The array to search.
237    * @param key The key being searched for.
238    * @param inclusive If the key is present in the array, whether to return the corresponding index.
239    *     If false then the returned index corresponds to the largest value in the array that is
240    *     strictly less than the key.
241    * @param stayInBounds If true, then 0 will be returned in the case that the key is smaller than
242    *     the smallest value in the array. If false then -1 will be returned.
243    */
244   public static int binarySearchFloor(long[] a, long key, boolean inclusive, boolean stayInBounds) {
245     int index = Arrays.binarySearch(a, key);
246     index = index < 0 ? -(index + 2) : (inclusive ? index : (index - 1));
247     return stayInBounds ? Math.max(0, index) : index;
248   }

256    * @param a The array to search.

262    *     key is greater than the largest value in the array. If false then {@code a.length} will be

314   /**
315    * Parses an xs:duration attribute value, returning the parsed duration in milliseconds.
316    *
317    * @param value The attribute value to parse.
318    * @return The parsed duration in milliseconds.
319    */
320   public static long parseXsDuration(String value) {
321     Matcher matcher = XS_DURATION_PATTERN.matcher(value);
322     if (matcher.matches()) {
323       boolean negated = !TextUtils.isEmpty(matcher.group(1));
324       // Durations containing years and months aren't completely defined. We assume there are
325       // 30.4368 days in a month, and 365.242 days in a year.
326       String years = matcher.group(3);
327       double durationSeconds = (years != null) ? Double.parseDouble(years) * 31556908 : 0;
328       String months = matcher.group(5);
329       durationSeconds += (months != null) ? Double.parseDouble(months) * 2629739 : 0;
330       String days = matcher.group(7);
331       durationSeconds += (days != null) ? Double.parseDouble(days) * 86400 : 0;
332       String hours = matcher.group(10);
333       durationSeconds += (hours != null) ? Double.parseDouble(hours) * 3600 : 0;
334       String minutes = matcher.group(12);
335       durationSeconds += (minutes != null) ? Double.parseDouble(minutes) * 60 : 0;
336       String seconds = matcher.group(14);
337       durationSeconds += (seconds != null) ? Double.parseDouble(seconds) : 0;
338       long durationMillis = (long) (durationSeconds * 1000);
339       return negated ? -durationMillis : durationMillis;
340     } else {
341       return (long) (Double.parseDouble(value) * 3600 * 1000);
342     }
343   }

345   /**
346    * Parses an xs:dateTime attribute value, returning the parsed timestamp in milliseconds since
347    * the epoch.
348    *
349    * @param value The attribute value to parse.
350    * @return The parsed timestamp in milliseconds since the epoch.
351    */
352   public static long parseXsDateTime(String value) throws ParseException {
353     Matcher matcher = XS_DATE_TIME_PATTERN.matcher(value);
354     if (!matcher.matches()) {
355       throw new ParseException("Invalid date/time format: " + value, 0);
356     }
357 
358     int timezoneShift;
359     if (matcher.group(9) == null) {
360       // No time zone specified.
361       timezoneShift = 0;
362     } else if (matcher.group(9).equalsIgnoreCase("Z")) {
363       timezoneShift = 0;
364     } else {
365       timezoneShift = ((Integer.parseInt(matcher.group(12)) * 60
366           + Integer.parseInt(matcher.group(13))));
367       if (matcher.group(11).equals("-")) {
368         timezoneShift *= -1;
369       }
370     }
371 
372     Calendar dateTime = new GregorianCalendar(TimeZone.getTimeZone("GMT"));
373 
374     dateTime.clear();
375     // Note: The month value is 0-based, hence the -1 on group(2)
376     dateTime.set(Integer.parseInt(matcher.group(1)),
377                  Integer.parseInt(matcher.group(2)) - 1,
378                  Integer.parseInt(matcher.group(3)),
379                  Integer.parseInt(matcher.group(4)),
380                  Integer.parseInt(matcher.group(5)),
381                  Integer.parseInt(matcher.group(6)));
382     if (!TextUtils.isEmpty(matcher.group(8))) {
383       final BigDecimal bd = new BigDecimal("0." + matcher.group(8));
384       // we care only for milliseconds, so movePointRight(3)
385       dateTime.set(Calendar.MILLISECOND, bd.movePointRight(3).intValue());
386     }
387 
388     long time = dateTime.getTimeInMillis();
389     if (timezoneShift != 0) {
390       time -= timezoneShift * 60000;
391     }
392 
393     return time;
394   }

396   /**
397    * Scales a large timestamp.
398    * <p>
399    * Logically, scaling consists of a multiplication followed by a division. The actual operations
400    * performed are designed to minimize the probability of overflow.
401    *
402    * @param timestamp The timestamp to scale.
403    * @param multiplier The multiplier.
404    * @param divisor The divisor.
405    * @return The scaled timestamp.
406    */
407   public static long scaleLargeTimestamp(long timestamp, long multiplier, long divisor) {
408     if (divisor >= multiplier && (divisor % multiplier) == 0) {
409       long divisionFactor = divisor / multiplier;
410       return timestamp / divisionFactor;
411     } else if (divisor < multiplier && (multiplier % divisor) == 0) {
412       long multiplicationFactor = multiplier / divisor;
413       return timestamp * multiplicationFactor;
414     } else {
415       double multiplicationFactor = (double) multiplier / divisor;
416       return (long) (timestamp * multiplicationFactor);
417     }
418   }

420   /**
421    * Applies {@link #scaleLargeTimestamp(long, long, long)} to a list of unscaled timestamps.
422    *
423    * @param timestamps The timestamps to scale.
424    * @param multiplier The multiplier.
425    * @param divisor The divisor.
426    * @return The scaled timestamps.
427    */
428   public static long[] scaleLargeTimestamps(List<Long> timestamps, long multiplier, long divisor) {
429     long[] scaledTimestamps = new long[timestamps.size()];
430     if (divisor >= multiplier && (divisor % multiplier) == 0) {
431       long divisionFactor = divisor / multiplier;
432       for (int i = 0; i < scaledTimestamps.length; i++) {
433         scaledTimestamps[i] = timestamps.get(i) / divisionFactor;
434       }
435     } else if (divisor < multiplier && (multiplier % divisor) == 0) {
436       long multiplicationFactor = multiplier / divisor;
437       for (int i = 0; i < scaledTimestamps.length; i++) {
438         scaledTimestamps[i] = timestamps.get(i) * multiplicationFactor;
439       }
440     } else {
441       double multiplicationFactor = (double) multiplier / divisor;
442       for (int i = 0; i < scaledTimestamps.length; i++) {
443         scaledTimestamps[i] = (long) (timestamps.get(i) * multiplicationFactor);
444       }
445     }
446     return scaledTimestamps;
447   }

493   /**
494    * On platform API levels 19 and 20, okhttp's implementation of {@link InputStream#close} can
495    * block for a long time if the stream has a lot of data remaining. Call this method before
496    * closing the input stream to make a best effort to cause the input stream to encounter an
497    * unexpected end of input, working around this issue. On other platform API levels, the method
498    * does nothing.
499    *
500    * @param connection The connection whose {@link InputStream} should be terminated.
501    * @param bytesRemaining The number of bytes remaining to be read from the input stream if its
502    *     length is known. {@link C#LENGTH_UNBOUNDED} otherwise.
503    */
504   public static void maybeTerminateInputStream(HttpURLConnection connection, long bytesRemaining) {
505     if (SDK_INT != 19 && SDK_INT != 20) {
506       return;
507     }
508 
509     try {
510       InputStream inputStream = connection.getInputStream();
511       if (bytesRemaining == C.LENGTH_UNBOUNDED) {
512         // If the input stream has already ended, do nothing. The socket may be re-used.
513         if (inputStream.read() == -1) {
514           return;
515         }
516       } else if (bytesRemaining <= MAX_BYTES_TO_DRAIN) {
517         // There isn't much data left. Prefer to allow it to drain, which may allow the socket to be
518         // re-used.
519         return;
520       }
521       String className = inputStream.getClass().getName();
522       if (className.equals("com.android.okhttp.internal.http.HttpTransport$ChunkedInputStream")
523           || className.equals(
524               "com.android.okhttp.internal.http.HttpTransport$FixedLengthInputStream")) {
525         Class<?> superclass = inputStream.getClass().getSuperclass();
526         Method unexpectedEndOfInput = superclass.getDeclaredMethod("unexpectedEndOfInput");
527         unexpectedEndOfInput.setAccessible(true);
528         unexpectedEndOfInput.invoke(inputStream);
529       }
530     } catch (IOException e) {
531       // The connection didn't ever have an input stream, or it was closed already.
532     } catch (Exception e) {
533       // Something went wrong. The device probably isn't using okhttp.
534     }
535   }

537   /**
538    * Given a {@link DataSpec} and a number of bytes already loaded, returns a {@link DataSpec}
539    * that represents the remainder of the data.
540    *
541    * @param dataSpec The original {@link DataSpec}.
542    * @param bytesLoaded The number of bytes already loaded.
543    * @return A {@link DataSpec} that represents the remainder of the data.
544    */
545   public static DataSpec getRemainderDataSpec(DataSpec dataSpec, int bytesLoaded) {
546     if (bytesLoaded == 0) {
547       return dataSpec;
548     } else {
549       long remainingLength = dataSpec.length == C.LENGTH_UNBOUNDED ? C.LENGTH_UNBOUNDED
550           : dataSpec.length - bytesLoaded;
551       return new DataSpec(dataSpec.uri, dataSpec.position + bytesLoaded, remainingLength,
552           dataSpec.key, dataSpec.flags);
553     }
554   }

556   /**
557    * Returns the integer equal to the big-endian concatenation of the characters in {@code string}
558    * as bytes. {@code string} must contain four or fewer characters.
559    */
560   public static int getIntegerCodeForString(String string) {
561     int length = string.length();
562     Assertions.checkArgument(length <= 4);
563     int result = 0;
564     for (int i = 0; i < length; i++) {
565       result <<= 8;
566       result |= string.charAt(i);
567     }
568     return result;
569   }

624   /**
625    * Executes a post request using {@link HttpURLConnection}.
626    *
627    * @param url The request URL.
628    * @param data The request body, or null.
629    * @param requestProperties Request properties, or null.
630    * @return The response body.
631    * @throws IOException If an error occurred making the request.
632    */
633   // TODO: Remove this and use HttpDataSource once DataSpec supports inclusion of a POST body.
634   public static byte[] executePost(String url, byte[] data, Map<String, String> requestProperties)
635       throws IOException {
636     HttpURLConnection urlConnection = null;
637     try {
638       urlConnection = (HttpURLConnection) new URL(url).openConnection();
639       urlConnection.setRequestMethod("POST");
640       urlConnection.setDoOutput(data != null);
641       urlConnection.setDoInput(true);
642       if (requestProperties != null) {
643         for (Map.Entry<String, String> requestProperty : requestProperties.entrySet()) {
644           urlConnection.setRequestProperty(requestProperty.getKey(), requestProperty.getValue());
645         }
646       }
647       // Write the request body, if there is one.
648       if (data != null) {
649         OutputStream out = urlConnection.getOutputStream();
650         try {
651           out.write(data);
652         } finally {
653           out.close();
654         }
655       }
656       // Read and return the response body.
657       InputStream inputStream = urlConnection.getInputStream();
658       try {
659         ByteArrayOutputStream byteArrayOutputStream = new ByteArrayOutputStream();
660         byte scratch[] = new byte[1024];
661         int bytesRead;
662         while ((bytesRead = inputStream.read(scratch)) != -1) {
663           byteArrayOutputStream.write(scratch, 0, bytesRead);
664         }
665         return byteArrayOutputStream.toByteArray();
666       } finally {
667         inputStream.close();
668       }
669     } finally {
670       if (urlConnection != null) {
671         urlConnection.disconnect();
672       }
673     }
674   }
