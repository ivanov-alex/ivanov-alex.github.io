65    private MimeTypes() {}

67    /**
68     * Returns the top-level type of {@code mimeType}.
69     *
70     * @param mimeType The mimeType whose top-level type is required.
71     * @return The top-level type.
72     */
73    public static String getTopLevelType(String mimeType) {
74      int indexOfSlash = mimeType.indexOf('/');
75      if (indexOfSlash == -1) {
76        throw new IllegalArgumentException("Invalid mime type: " + mimeType);
77      }
78      return mimeType.substring(0, indexOfSlash);
79    }
80  
81    /**
82     * Whether the top-level type of {@code mimeType} is audio.
83     *
84     * @param mimeType The mimeType to test.
85     * @return Whether the top level type is audio.
86     */
87    public static boolean isAudio(String mimeType) {
88      return getTopLevelType(mimeType).equals(BASE_TYPE_AUDIO);
89    }
90  
91    /**
92     * Whether the top-level type of {@code mimeType} is video.
93     *
94     * @param mimeType The mimeType to test.
95     * @return Whether the top level type is video.
96     */
97    public static boolean isVideo(String mimeType) {
98      return getTopLevelType(mimeType).equals(BASE_TYPE_VIDEO);
99    }
100

107   public static boolean isText(String mimeType) {

109   }

117   public static boolean isApplication(String mimeType) {

128     return mimeType.equals(APPLICATION_TTML);
129   }

131   /**
132    * Returns the output audio encoding that will result from processing input in {@code mimeType}.
133    * For non-passthrough audio formats, this is always {@link AudioFormat#ENCODING_PCM_16BIT}. For
134    * passthrough formats it will be one of {@link AudioFormat}'s other {@code ENCODING_*} constants.
135    * For non-audio formats, {@link AudioFormat#ENCODING_INVALID} will be returned.
136    *
137    * @param mimeType The MIME type of media that will be decoded (or passed through).
138    * @return The corresponding {@link AudioFormat} encoding.
139    */
140   public static int getEncodingForMimeType(String mimeType) {
141     if (AUDIO_AC3.equals(mimeType)) {
142       return C.ENCODING_AC3;
143     }
144     if (AUDIO_EC3.equals(mimeType)) {
145       return C.ENCODING_E_AC3;
146     }
147 
148     // All other audio formats will be decoded to 16-bit PCM.
149     return isAudio(mimeType) ? AudioFormat.ENCODING_PCM_16BIT : AudioFormat.ENCODING_INVALID;
150   }
151 
152   /**
153    * Returns whether the specified {@code mimeType} represents audio that can be played via
154    * passthrough if the device supports it.
155    *
156    * @param mimeType The MIME type of input media.
157    * @return Whether the audio can be played via passthrough. If this method returns {@code true},
158    *     it is still necessary to check the {@link AudioCapabilities} for device support.
159    */
160   public static boolean isPassthroughAudio(String mimeType) {
161     return AUDIO_AC3.equals(mimeType) || AUDIO_EC3.equals(mimeType);
162   }
