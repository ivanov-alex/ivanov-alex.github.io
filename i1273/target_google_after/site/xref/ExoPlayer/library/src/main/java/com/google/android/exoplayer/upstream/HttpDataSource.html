16  package com.google.android.exoplayer.upstream;
17  
18  import com.google.android.exoplayer.util.Predicate;
19  import com.google.android.exoplayer.util.Util;
20  
21  import android.text.TextUtils;
22  
23  import java.io.IOException;
24  import java.util.List;
25  import java.util.Map;
26  
27  /**
28   * An HTTP specific extension to {@link UriDataSource}.
29   */
30  public interface HttpDataSource extends UriDataSource {
31  
32    /**
33     * A {@link Predicate} that rejects content types often used for pay-walls.
34     */
35    public static final Predicate<String> REJECT_PAYWALL_TYPES = new Predicate<String>() {
36  
37      @Override
38      public boolean evaluate(String contentType) {
39        contentType = Util.toLowerInvariant(contentType);
40        return !TextUtils.isEmpty(contentType)
41            && (!contentType.contains("text") || contentType.contains("text/vtt"))
42            && !contentType.contains("html") && !contentType.contains("xml");
43      }
44  
45    };
46  
47    /**
48     * Thrown when an error is encountered when trying to read from a {@link HttpDataSource}.
49     */
50    public static class HttpDataSourceException extends IOException {
51  
52      /*
53       * The {@link DataSpec} associated with the current connection.
54       */
55      public final DataSpec dataSpec;
56  
57      public HttpDataSourceException(DataSpec dataSpec) {
58        super();
59        this.dataSpec = dataSpec;
60      }
61  
62      public HttpDataSourceException(String message, DataSpec dataSpec) {
63        super(message);
64        this.dataSpec = dataSpec;
65      }
66  
67      public HttpDataSourceException(IOException cause, DataSpec dataSpec) {
68        super(cause);
69        this.dataSpec = dataSpec;
70      }
71  
72      public HttpDataSourceException(String message, IOException cause, DataSpec dataSpec) {
73        super(message, cause);
74        this.dataSpec = dataSpec;
75      }
76  
77    }
78  
79    /**
80     * Thrown when the content type is invalid.
81     */
82    public static final class InvalidContentTypeException extends HttpDataSourceException {
83  
84      public final String contentType;
85  
86      public InvalidContentTypeException(String contentType, DataSpec dataSpec) {
87        super("Invalid content type: " + contentType, dataSpec);
88        this.contentType = contentType;
89      }
90  
91    }
92  
93    /**
94     * Thrown when an attempt to open a connection results in a response code not in the 2xx range.
95     */
96    public static final class InvalidResponseCodeException extends HttpDataSourceException {
97  
98      /**
99       * The response code that was outside of the 2xx range.
100      */
101     public final int responseCode;
102 
103     /**
104      * An unmodifiable map of the response header fields and values.
105      */
106     public final Map<String, List<String>> headerFields;
107 
108     public InvalidResponseCodeException(int responseCode, Map<String, List<String>> headerFields,
109         DataSpec dataSpec) {
110       super("Response code: " + responseCode, dataSpec);
111       this.responseCode = responseCode;
112       this.headerFields = headerFields;
113     }
114 
115   }
116 
117   @Override
118   long open(DataSpec dataSpec) throws HttpDataSourceException;
119 
120   @Override
121   void close() throws HttpDataSourceException;
122 
123   @Override
124   int read(byte[] buffer, int offset, int readLength) throws HttpDataSourceException;
125 
126   /**
127    * Sets the value of a request header field. The value will be used for subsequent connections
128    * established by the source.
129    *
130    * @param name The name of the header field.
131    * @param value The value of the field.
132    */
133   void setRequestProperty(String name, String value);
134 
135   /**
136    * Clears the value of a request header field. The change will apply to subsequent connections
137    * established by the source.
138    *
139    * @param name The name of the header field.
140    */
141   void clearRequestProperty(String name);
142 
143   /**
144    * Clears all request header fields that were set by {@link #setRequestProperty(String, String)}.
145    */
146   void clearAllRequestProperties();
147 
148   /**
149    * Gets the headers provided in the response.
150    *
151    * @return The response headers, or {@code null} if response headers are unavailable.
152    */
153   Map<String, List<String>> getResponseHeaders();
154 
155 }
