42    private static final int TS_SYNC_BYTE = 0x47; // First byte of each TS packet.

83    public TsExtractor(long firstSampleTimestampUs, AudioCapabilities audioCapabilities,
84        boolean idrKeyframesOnly) {
85      this.firstSampleTimestampUs = firstSampleTimestampUs;
86      this.idrKeyframesOnly = idrKeyframesOnly;
87      tsScratch = new ParsableBitArray(new byte[3]);
88      tsPacketBuffer = new ParsableByteArray(TS_PACKET_SIZE);
89      streamTypes = new SparseBooleanArray();
90      allowedPassthroughStreamTypes = getPassthroughStreamTypes(audioCapabilities);
91      tsPayloadReaders = new SparseArray<>();
92      tsPayloadReaders.put(TS_PAT_PID, new PatReader());
93      lastPts = Long.MIN_VALUE;
94    }

98    @Override
99    public boolean sniff(ExtractorInput input) throws IOException, InterruptedException {
100     byte[] scratch = new byte[1];
101     for (int i = 0; i < 5; i++) {
102       input.peekFully(scratch, 0, 1);
103       if ((scratch[0] & 0xFF) != 0x47) {
104         return false;
105       }
106       input.advancePeekPosition(TS_PACKET_SIZE - 1);
107     }
108     return true;
109   }

111   @Override
112   public void init(ExtractorOutput output) {
113     this.output = output;
114     output.seekMap(SeekMap.UNSEEKABLE);
115   }

117   @Override
118   public void seek() {
119     timestampOffsetUs = 0;
120     lastPts = Long.MIN_VALUE;
121     for (int i = 0; i < tsPayloadReaders.size(); i++) {
122       tsPayloadReaders.valueAt(i).seek();
123     }
124   }
125 
126   @Override
127   public int read(ExtractorInput input, PositionHolder seekPosition)
128       throws IOException, InterruptedException {
129     if (!input.readFully(tsPacketBuffer.data, 0, TS_PACKET_SIZE, true)) {
130       return RESULT_END_OF_INPUT;
131     }
132 
133     // Note: see ISO/IEC 13818-1, section 2.4.3.2 for detailed information on the format of
134     // the header.
135     tsPacketBuffer.setPosition(0);
136     tsPacketBuffer.setLimit(TS_PACKET_SIZE);
137     int syncByte = tsPacketBuffer.readUnsignedByte();
138     if (syncByte != TS_SYNC_BYTE) {
139       return RESULT_CONTINUE;
140     }
141 
142     tsPacketBuffer.readBytes(tsScratch, 3);
143     tsScratch.skipBits(1); // transport_error_indicator
144     boolean payloadUnitStartIndicator = tsScratch.readBit();
145     tsScratch.skipBits(1); // transport_priority
146     int pid = tsScratch.readBits(13);
147     tsScratch.skipBits(2); // transport_scrambling_control
148     boolean adaptationFieldExists = tsScratch.readBit();
149     boolean payloadExists = tsScratch.readBit();
150     // Last 4 bits of scratch are skipped: continuity_counter
151 
152     // Skip the adaptation field.
153     if (adaptationFieldExists) {
154       int adaptationFieldLength = tsPacketBuffer.readUnsignedByte();
155       tsPacketBuffer.skipBytes(adaptationFieldLength);
156     }
157 
158     // Read the payload.
159     if (payloadExists) {
160       TsPayloadReader payloadReader = tsPayloadReaders.get(pid);
161       if (payloadReader != null) {
162         payloadReader.consume(tsPacketBuffer, payloadUnitStartIndicator, output);
163       }
164     }
165 
166     return RESULT_CONTINUE;
167   }

171   /**
172    * Adjusts a PTS value to the corresponding time in microseconds, accounting for PTS wraparound.
173    *
174    * @param pts The raw PTS value.
175    * @return The corresponding time in microseconds.
176    */
177   /* package */ long ptsToTimeUs(long pts) {
178     if (lastPts != Long.MIN_VALUE) {
179       // The wrap count for the current PTS may be closestWrapCount or (closestWrapCount - 1),
180       // and we need to snap to the one closest to lastPts.
181       long closestWrapCount = (lastPts + (MAX_PTS / 2)) / MAX_PTS;
182       long ptsWrapBelow = pts + (MAX_PTS * (closestWrapCount - 1));
183       long ptsWrapAbove = pts + (MAX_PTS * closestWrapCount);
184       pts = Math.abs(ptsWrapBelow - lastPts) < Math.abs(ptsWrapAbove - lastPts)
185           ? ptsWrapBelow : ptsWrapAbove;
186     }
187     // Calculate the corresponding timestamp.
188     long timeUs = (pts * C.MICROS_PER_SECOND) / 90000;
189     // If we haven't done the initial timestamp adjustment, do it now.
190     if (lastPts == Long.MIN_VALUE) {
191       timestampOffsetUs = firstSampleTimestampUs - timeUs;
192     }
193     // Record the adjusted PTS to adjust for wraparound next time.
194     lastPts = pts;
195     return timeUs + timestampOffsetUs;
196   }

198   /**
199    * Returns a sparse boolean array of stream types that can be played back based on
200    * {@code audioCapabilities}.
201    */
202   private static SparseBooleanArray getPassthroughStreamTypes(AudioCapabilities audioCapabilities) {
203     SparseBooleanArray streamTypes = new SparseBooleanArray();
204     if (audioCapabilities != null) {
205       if (audioCapabilities.supportsEncoding(C.ENCODING_AC3)) {
206         streamTypes.put(TS_STREAM_TYPE_ATSC_AC3, true);
207       }
208       if (audioCapabilities.supportsEncoding(C.ENCODING_E_AC3)) {
209         // TODO: Uncomment when Ac3Reader supports enhanced AC-3.
210         // streamTypes.put(TS_STREAM_TYPE_ATSC_E_AC3, true);
211       }
212     }
213     return streamTypes;
214   }

216   /**
217    * Parses TS packet payload data.
218    */
219   private abstract static class TsPayloadReader {
220 
221     /**
222      * Notifies the reader that a seek has occurred.
223      * <p>
224      * Following a call to this method, the data passed to the next invocation of
225      * {@link #consume(ParsableByteArray, boolean, ExtractorOutput)} will not be a continuation of
226      * the data that was previously passed. Hence the reader should reset any internal state.
227      */
228     public abstract void seek();
229 
230     /**
231      * Consumes the payload of a TS packet.
232      *
233      * @param data The TS packet. The position will be set to the start of the payload.
234      * @param payloadUnitStartIndicator Whether payloadUnitStartIndicator was set on the TS packet.
235      * @param output The output to which parsed data should be written.
236      */
237     public abstract void consume(ParsableByteArray data, boolean payloadUnitStartIndicator,
238         ExtractorOutput output);
239 
240   }

242   /**
243    * Parses Program Association Table data.
244    */
245   private class PatReader extends TsPayloadReader {
246 
247     private final ParsableBitArray patScratch;
248 
249     public PatReader() {
250       patScratch = new ParsableBitArray(new byte[4]);
251     }
252 
253     @Override
254     public void seek() {
255       // Do nothing.
256     }
257 
258     @Override
259     public void consume(ParsableByteArray data, boolean payloadUnitStartIndicator,
260         ExtractorOutput output) {
261       // Skip pointer.
262       if (payloadUnitStartIndicator) {
263         int pointerField = data.readUnsignedByte();
264         data.skipBytes(pointerField);
265       }
266 
267       data.readBytes(patScratch, 3);
268       patScratch.skipBits(12); // table_id (8), section_syntax_indicator (1), '0' (1), reserved (2)
269       int sectionLength = patScratch.readBits(12);
270       // transport_stream_id (16), reserved (2), version_number (5), current_next_indicator (1),
271       // section_number (8), last_section_number (8)
272       data.skipBytes(5);
273 
274       int programCount = (sectionLength - 9) / 4;
275       for (int i = 0; i < programCount; i++) {
276         data.readBytes(patScratch, 4);
277         patScratch.skipBits(19); // program_number (16), reserved (3)
278         int pid = patScratch.readBits(13);
279         tsPayloadReaders.put(pid, new PmtReader());
280       }
281 
282       // Skip CRC_32.
283     }
284 
285   }

287   /**
288    * Parses Program Map Table.
289    */
290   private class PmtReader extends TsPayloadReader {
291 
292     private final ParsableBitArray pmtScratch;
293 
294     public PmtReader() {
295       pmtScratch = new ParsableBitArray(new byte[5]);
296     }
297 
298     @Override
299     public void seek() {
300       // Do nothing.
301     }
302 
303     @Override
304     public void consume(ParsableByteArray data, boolean payloadUnitStartIndicator,
305         ExtractorOutput output) {
306       // Skip pointer.
307       if (payloadUnitStartIndicator) {
308         int pointerField = data.readUnsignedByte();
309         data.skipBytes(pointerField);
310       }
311 
312       // Note: see ISO/IEC 13818-1, section 2.4.4.8 for detailed information on the format of
313       // the header.
314       data.readBytes(pmtScratch, 3);
315       pmtScratch.skipBits(12); // table_id (8), section_syntax_indicator (1), '0' (1), reserved (2)
316       int sectionLength = pmtScratch.readBits(12);
317 
318       // program_number (16), reserved (2), version_number (5), current_next_indicator (1),
319       // section_number (8), last_section_number (8), reserved (3), PCR_PID (13)
320       // Skip the rest of the PMT header.
321       data.skipBytes(7);
322 
323       data.readBytes(pmtScratch, 2);
324       pmtScratch.skipBits(4);
325       int programInfoLength = pmtScratch.readBits(12);
326 
327       // Skip the descriptors.
328       data.skipBytes(programInfoLength);
329 
330       if (id3Reader == null) {
331         // Setup an ID3 track regardless of whether there's a corresponding entry, in case one
332         // appears intermittently during playback. See b/20261500.
333         id3Reader = new Id3Reader(output.track(TS_STREAM_TYPE_ID3));
334       }
335 
336       int entriesSize = sectionLength - 9 /* Size of the rest of the fields before descriptors */
337           - programInfoLength - 4 /* CRC size */;
338       while (entriesSize > 0) {
339         data.readBytes(pmtScratch, 5);
340         int streamType = pmtScratch.readBits(8);
341         pmtScratch.skipBits(3); // reserved
342         int elementaryPid = pmtScratch.readBits(13);
343         pmtScratch.skipBits(4); // reserved
344         int esInfoLength = pmtScratch.readBits(12);
345 
346         // Skip the descriptors.
347         data.skipBytes(esInfoLength);
348         entriesSize -= esInfoLength + 5;
349 
350         if (streamTypes.get(streamType)) {
351           continue;
352         }
353 
354         // TODO: Detect and read DVB AC-3 streams with Ac3Reader.
355         ElementaryStreamReader pesPayloadReader = null;
356         switch (streamType) {
357           case TS_STREAM_TYPE_MPA:
358             pesPayloadReader = new MpegAudioReader(output.track(TS_STREAM_TYPE_MPA));
359             break;
360           case TS_STREAM_TYPE_MPA_LSF:
361             pesPayloadReader = new MpegAudioReader(output.track(TS_STREAM_TYPE_MPA_LSF));
362             break;
363           case TS_STREAM_TYPE_AAC:
364             pesPayloadReader = new AdtsReader(output.track(TS_STREAM_TYPE_AAC));
365             break;
366           case TS_STREAM_TYPE_ATSC_E_AC3:
367           case TS_STREAM_TYPE_ATSC_AC3:
368             if (!allowedPassthroughStreamTypes.get(streamType)) {
369               continue;
370             }
371             pesPayloadReader = new Ac3Reader(output.track(streamType));
372             break;
373           case TS_STREAM_TYPE_H264:
374             pesPayloadReader = new H264Reader(output.track(TS_STREAM_TYPE_H264),
375                 new SeiReader(output.track(TS_STREAM_TYPE_EIA608)), idrKeyframesOnly);
376             break;
377           case TS_STREAM_TYPE_H265:
378             pesPayloadReader = new H265Reader(output.track(TS_STREAM_TYPE_H265),
379                 new SeiReader(output.track(TS_STREAM_TYPE_EIA608)));
380             break;
381           case TS_STREAM_TYPE_ID3:
382             pesPayloadReader = id3Reader;
383             break;
384         }
385 
386         if (pesPayloadReader != null) {
387           streamTypes.put(streamType, true);
388           tsPayloadReaders.put(elementaryPid, new PesReader(pesPayloadReader));
389         }
390       }
391 
392       output.endTracks();
393     }
394 
395   }

397   /**
398    * Parses PES packet data and extracts samples.
399    */
400   private class PesReader extends TsPayloadReader {
401 
402     private static final int STATE_FINDING_HEADER = 0;
403     private static final int STATE_READING_HEADER = 1;
404     private static final int STATE_READING_HEADER_EXTENSION = 2;
405     private static final int STATE_READING_BODY = 3;
406 
407     private static final int HEADER_SIZE = 9;
408     private static final int MAX_HEADER_EXTENSION_SIZE = 5;
409 
410     private final ParsableBitArray pesScratch;
411     private final ElementaryStreamReader pesPayloadReader;
412 
413     private int state;
414     private int bytesRead;
415     private boolean bodyStarted;
416 
417     private boolean ptsFlag;
418     private int extendedHeaderLength;
419     private int payloadSize;
420     private long timeUs;
421 
422     public PesReader(ElementaryStreamReader pesPayloadReader) {
423       this.pesPayloadReader = pesPayloadReader;
424       pesScratch = new ParsableBitArray(new byte[HEADER_SIZE]);
425       state = STATE_FINDING_HEADER;
426     }
427 
428     @Override
429     public void seek() {
430       state = STATE_FINDING_HEADER;
431       bytesRead = 0;
432       bodyStarted = false;
433       pesPayloadReader.seek();
434     }
435 
436     @Override
437     public void consume(ParsableByteArray data, boolean payloadUnitStartIndicator,
438         ExtractorOutput output) {
439       if (payloadUnitStartIndicator) {
440         switch (state) {
441           case STATE_FINDING_HEADER:
442           case STATE_READING_HEADER:
443             // Expected.
444             break;
445           case STATE_READING_HEADER_EXTENSION:
446             Log.w(TAG, "Unexpected start indicator reading extended header");
447             break;
448           case STATE_READING_BODY:
449             // If payloadSize == -1 then the length of the previous packet was unspecified, and so
450             // we only know that it's finished now that we've seen the start of the next one. This
451             // is expected. If payloadSize != -1, then the length of the previous packet was known,
452             // but we didn't receive that amount of data. This is not expected.
453             if (payloadSize != -1) {
454               Log.w(TAG, "Unexpected start indicator: expected " + payloadSize + " more bytes");
455             }
456             // Either way, if the body was started, notify the reader that it has now finished.
457             if (bodyStarted) {
458               pesPayloadReader.packetFinished();
459             }
460             break;
461         }
462         setState(STATE_READING_HEADER);
463       }
464 
465       while (data.bytesLeft() > 0) {
466         switch (state) {
467           case STATE_FINDING_HEADER:
468             data.skipBytes(data.bytesLeft());
469             break;
470           case STATE_READING_HEADER:
471             if (continueRead(data, pesScratch.data, HEADER_SIZE)) {
472               setState(parseHeader() ? STATE_READING_HEADER_EXTENSION : STATE_FINDING_HEADER);
473             }
474             break;
475           case STATE_READING_HEADER_EXTENSION:
476             int readLength = Math.min(MAX_HEADER_EXTENSION_SIZE, extendedHeaderLength);
477             // Read as much of the extended header as we're interested in, and skip the rest.
478             if (continueRead(data, pesScratch.data, readLength)
479                 && continueRead(data, null, extendedHeaderLength)) {
480               parseHeaderExtension();
481               bodyStarted = false;
482               setState(STATE_READING_BODY);
483             }
484             break;
485           case STATE_READING_BODY:
486             readLength = data.bytesLeft();
487             int padding = payloadSize == -1 ? 0 : readLength - payloadSize;
488             if (padding > 0) {
489               readLength -= padding;
490               data.setLimit(data.getPosition() + readLength);
491             }
492             pesPayloadReader.consume(data, timeUs, !bodyStarted);
493             bodyStarted = true;
494             if (payloadSize != -1) {
495               payloadSize -= readLength;
496               if (payloadSize == 0) {
497                 pesPayloadReader.packetFinished();
498                 setState(STATE_READING_HEADER);
499               }
500             }
501             break;
502         }
503       }
504     }
505 
506     private void setState(int state) {
507       this.state = state;
508       bytesRead = 0;
509     }
510 
511     /**
512      * Continues a read from the provided {@code source} into a given {@code target}. It's assumed
513      * that the data should be written into {@code target} starting from an offset of zero.
514      *
515      * @param source The source from which to read.
516      * @param target The target into which data is to be read, or {@code null} to skip.
517      * @param targetLength The target length of the read.
518      * @return Whether the target length has been reached.
519      */
520     private boolean continueRead(ParsableByteArray source, byte[] target, int targetLength) {
521       int bytesToRead = Math.min(source.bytesLeft(), targetLength - bytesRead);
522       if (bytesToRead <= 0) {
523         return true;
524       } else if (target == null) {
525         source.skipBytes(bytesToRead);
526       } else {
527         source.readBytes(target, bytesRead, bytesToRead);
528       }
529       bytesRead += bytesToRead;
530       return bytesRead == targetLength;
531     }
532 
533     private boolean parseHeader() {
534       // Note: see ISO/IEC 13818-1, section 2.4.3.6 for detailed information on the format of
535       // the header.
536       pesScratch.setPosition(0);
537       int startCodePrefix = pesScratch.readBits(24);
538       if (startCodePrefix != 0x000001) {
539         Log.w(TAG, "Unexpected start code prefix: " + startCodePrefix);
540         payloadSize = -1;
541         return false;
542       }
543 
544       pesScratch.skipBits(8); // stream_id.
545       int packetLength = pesScratch.readBits(16);
546       // First 8 bits are skipped: '10' (2), PES_scrambling_control (2), PES_priority (1),
547       // data_alignment_indicator (1), copyright (1), original_or_copy (1)
548       pesScratch.skipBits(8);
549       ptsFlag = pesScratch.readBit();
550       // DTS_flag (1), ESCR_flag (1), ES_rate_flag (1), DSM_trick_mode_flag (1),
551       // additional_copy_info_flag (1), PES_CRC_flag (1), PES_extension_flag (1)
552       pesScratch.skipBits(7);
553       extendedHeaderLength = pesScratch.readBits(8);
554 
555       if (packetLength == 0) {
556         payloadSize = -1;
557       } else {
558         payloadSize = packetLength + 6 /* packetLength does not include the first 6 bytes */
559             - HEADER_SIZE - extendedHeaderLength;
560       }
561       return true;
562     }
563 
564     private void parseHeaderExtension() {
565       pesScratch.setPosition(0);
566       timeUs = 0;
567       if (ptsFlag) {
568         pesScratch.skipBits(4); // '0010' or '0011'
569         long pts = (long) pesScratch.readBits(3) << 30;
570         pesScratch.skipBits(1); // marker_bit
571         pts |= pesScratch.readBits(15) << 15;
572         pesScratch.skipBits(1); // marker_bit
573         pts |= pesScratch.readBits(15);
574         pesScratch.skipBits(1); // marker_bit
575         timeUs = ptsToTimeUs(pts);
576       }
577     }
