89     *
90     * @param manifestFetcher A fetcher for the manifest, which must have already successfully

93     *     the source.

95     *     the source. May be null if all tracks within the element should be considered.
96     * @param dataSource A {@link DataSource} suitable for loading the media data.
97     * @param formatEvaluator Selects from the available formats.
98     * @param liveEdgeLatencyMs For live streams, the number of milliseconds that the playback should
99     *     lag behind the "live edge" (i.e. the end of the most recently defined media in the
100    *     manifest). Choosing a small value will minimize latency introduced by the player, however

127   private SmoothStreamingChunkSource(ManifestFetcher<SmoothStreamingManifest> manifestFetcher,
128       SmoothStreamingManifest initialManifest, int streamElementIndex, int[] trackIndices,
129       DataSource dataSource, FormatEvaluator formatEvaluator, long liveEdgeLatencyMs) {
130     this.manifestFetcher = manifestFetcher;
131     this.streamElementIndex = streamElementIndex;
132     this.currentManifest = initialManifest;
133     this.dataSource = dataSource;
134     this.formatEvaluator = formatEvaluator;
135     this.liveEdgeLatencyUs = liveEdgeLatencyMs * 1000;
136 
137     StreamElement streamElement = getElement(initialManifest);
138     trackInfo = new TrackInfo(streamElement.tracks[0].format.mimeType, initialManifest.durationUs);
139     evaluation = new Evaluation();
140 
141     TrackEncryptionBox[] trackEncryptionBoxes = null;
142     ProtectionElement protectionElement = initialManifest.protectionElement;
143     if (protectionElement != null) {
144       byte[] keyId = getKeyId(protectionElement.data);
145       trackEncryptionBoxes = new TrackEncryptionBox[1];
146       trackEncryptionBoxes[0] = new TrackEncryptionBox(true, INITIALIZATION_VECTOR_SIZE, keyId);
147       DrmInitData.Mapped drmInitData = new DrmInitData.Mapped(MimeTypes.VIDEO_MP4);
148       drmInitData.put(protectionElement.uuid, protectionElement.data);
149       this.drmInitData = drmInitData;
150     } else {
151       drmInitData = null;
152     }
153 
154     int trackCount = trackIndices != null ? trackIndices.length : streamElement.tracks.length;
155     formats = new Format[trackCount];
156     extractorWrappers = new SparseArray<>();
157     mediaFormats = new SparseArray<>();
158     int maxWidth = 0;
159     int maxHeight = 0;
160     for (int i = 0; i < trackCount; i++) {
161       int trackIndex = trackIndices != null ? trackIndices[i] : i;
162       formats[i] = streamElement.tracks[trackIndex].format;
163       maxWidth = Math.max(maxWidth, formats[i].width);
164       maxHeight = Math.max(maxHeight, formats[i].height);
165 
166       MediaFormat mediaFormat = getMediaFormat(streamElement, trackIndex);
167       int trackType = streamElement.type == StreamElement.TYPE_VIDEO ? Track.TYPE_VIDEO
168           : Track.TYPE_AUDIO;
169       FragmentedMp4Extractor extractor = new FragmentedMp4Extractor(
170           FragmentedMp4Extractor.WORKAROUND_EVERY_VIDEO_FRAME_IS_SYNC_FRAME);
171       extractor.setTrack(new Track(trackIndex, trackType, streamElement.timescale,
172           initialManifest.durationUs, mediaFormat, trackEncryptionBoxes,
173           trackType == Track.TYPE_VIDEO ? 4 : -1));
174       extractorWrappers.put(trackIndex, new ChunkExtractorWrapper(extractor));
175       mediaFormats.put(trackIndex, mediaFormat);
176     }
177     this.maxWidth = maxWidth;
178     this.maxHeight = maxHeight;
179     Arrays.sort(formats, new DecreasingBandwidthComparator());
180   }

211   @Override
212   public void continueBuffering(long playbackPositionUs) {
213     if (manifestFetcher == null || !currentManifest.isLive || fatalError != null) {
214       return;
215     }
216 
217     SmoothStreamingManifest newManifest = manifestFetcher.getManifest();
218     if (currentManifest != newManifest && newManifest != null) {
219       StreamElement currentElement = getElement(currentManifest);
220       int currentElementChunkCount = currentElement.chunkCount;
221       StreamElement newElement = getElement(newManifest);
222       if (currentElementChunkCount == 0 || newElement.chunkCount == 0) {
223         // There's no overlap between the old and new elements because at least one is empty.
224         currentManifestChunkOffset += currentElementChunkCount;
225       } else {
226         long currentElementEndTimeUs = currentElement.getStartTimeUs(currentElementChunkCount - 1)
227             + currentElement.getChunkDurationUs(currentElementChunkCount - 1);
228         long newElementStartTimeUs = newElement.getStartTimeUs(0);
229         if (currentElementEndTimeUs <= newElementStartTimeUs) {
230           // There's no overlap between the old and new elements.
231           currentManifestChunkOffset += currentElementChunkCount;
232         } else {
233           // The new element overlaps with the old one.
234           currentManifestChunkOffset += currentElement.getChunkIndex(newElementStartTimeUs);
235         }
236       }
237       currentManifest = newManifest;
238       finishedCurrentManifest = false;
239     }
240 
241     if (finishedCurrentManifest && (SystemClock.elapsedRealtime()
242         > manifestFetcher.getManifestLoadStartTimestamp() + MINIMUM_MANIFEST_REFRESH_PERIOD_MS)) {
243       manifestFetcher.requestRefresh();
244     }
245   }

247   @Override
248   public final void getChunkOperation(List<? extends MediaChunk> queue, long seekPositionUs,
249       long playbackPositionUs, ChunkOperationHolder out) {
250     if (fatalError != null) {
251       out.chunk = null;
252       return;
253     }
254 
255     evaluation.queueSize = queue.size();
256     formatEvaluator.evaluate(queue, playbackPositionUs, formats, evaluation);
257     Format selectedFormat = evaluation.format;
258     out.queueSize = evaluation.queueSize;
259 
260     if (selectedFormat == null) {
261       out.chunk = null;
262       return;
263     } else if (out.queueSize == queue.size() && out.chunk != null
264         && out.chunk.format.equals(selectedFormat)) {
265       // We already have a chunk, and the evaluation hasn't changed either the format or the size
266       // of the queue. Leave unchanged.
267       return;
268     }
269 
270     // In all cases where we return before instantiating a new chunk, we want out.chunk to be null.
271     out.chunk = null;
272 
273     StreamElement streamElement = getElement(currentManifest);
274     if (streamElement.chunkCount == 0) {
275       // The manifest is currently empty for this stream.
276       finishedCurrentManifest = true;
277       return;
278     }
279 
280     int chunkIndex;
281     if (queue.isEmpty()) {
282       if (currentManifest.isLive) {
283         seekPositionUs = getLiveSeekPosition();
284       }
285       chunkIndex = streamElement.getChunkIndex(seekPositionUs);
286     } else {
287       MediaChunk previous = queue.get(out.queueSize - 1);
288       chunkIndex = previous.isLastChunk ? -1 : previous.chunkIndex + 1 - currentManifestChunkOffset;
289     }
290 
291     if (currentManifest.isLive) {
292       if (chunkIndex < 0) {
293         // This is before the first chunk in the current manifest.
294         fatalError = new BehindLiveWindowException();
295         return;
296       } else if (chunkIndex >= streamElement.chunkCount) {
297         // This is beyond the last chunk in the current manifest.
298         finishedCurrentManifest = true;
299         return;
300       } else if (chunkIndex == streamElement.chunkCount - 1) {
301         // This is the last chunk in the current manifest. Mark the manifest as being finished,
302         // but continue to return the final chunk.
303         finishedCurrentManifest = true;
304       }
305     }
306 
307     if (chunkIndex == -1) {
308       // We've reached the end of the stream.
309       return;
310     }
311 
312     boolean isLastChunk = !currentManifest.isLive && chunkIndex == streamElement.chunkCount - 1;
313     long chunkStartTimeUs = streamElement.getStartTimeUs(chunkIndex);
314     long chunkEndTimeUs = isLastChunk ? -1
315         : chunkStartTimeUs + streamElement.getChunkDurationUs(chunkIndex);
316     int currentAbsoluteChunkIndex = chunkIndex + currentManifestChunkOffset;
317 
318     int trackIndex = getTrackIndex(selectedFormat);
319     Uri uri = streamElement.buildRequestUri(trackIndex, chunkIndex);
320     Chunk mediaChunk = newMediaChunk(selectedFormat, uri, null, extractorWrappers.get(trackIndex),
321         drmInitData, dataSource, currentAbsoluteChunkIndex, isLastChunk, chunkStartTimeUs,
322         chunkEndTimeUs, evaluation.trigger, mediaFormats.get(trackIndex));
323     out.chunk = mediaChunk;
324   }

345   /**
346    * For live playbacks, determines the seek position that snaps playback to be
347    * {@link #liveEdgeLatencyUs} behind the live edge of the current manifest
348    *
349    * @return The seek position in microseconds.
350    */
351   private long getLiveSeekPosition() {
352     long liveEdgeTimestampUs = Long.MIN_VALUE;
353     for (int i = 0; i < currentManifest.streamElements.length; i++) {
354       StreamElement streamElement = currentManifest.streamElements[i];
355       if (streamElement.chunkCount > 0) {
356         long elementLiveEdgeTimestampUs =
357             streamElement.getStartTimeUs(streamElement.chunkCount - 1)
358             + streamElement.getChunkDurationUs(streamElement.chunkCount - 1);
359         liveEdgeTimestampUs = Math.max(liveEdgeTimestampUs, elementLiveEdgeTimestampUs);
360       }
361     }
362     return liveEdgeTimestampUs - liveEdgeLatencyUs;
363   }

365   private StreamElement getElement(SmoothStreamingManifest manifest) {
366     return manifest.streamElements[streamElementIndex];
367   }

369   private int getTrackIndex(Format format) {
370     TrackElement[] tracks = currentManifest.streamElements[streamElementIndex].tracks;
371     for (int i = 0; i < tracks.length; i++) {
372       if (tracks[i].format.equals(format)) {
373         return i;
374       }
375     }
376     // Should never happen.
377     throw new IllegalStateException("Invalid format: " + format);
378   }

380   private static MediaFormat getMediaFormat(StreamElement streamElement, int trackIndex) {
381     TrackElement trackElement = streamElement.tracks[trackIndex];
382     Format trackFormat = trackElement.format;
383     String mimeType = trackFormat.mimeType;
384     if (streamElement.type == StreamElement.TYPE_VIDEO) {
385       MediaFormat format = MediaFormat.createVideoFormat(mimeType, MediaFormat.NO_VALUE,
386           trackFormat.width, trackFormat.height, Arrays.asList(trackElement.csd));
387       format.setMaxVideoDimensions(streamElement.maxWidth, streamElement.maxHeight);
388       return format;
389     } else if (streamElement.type == StreamElement.TYPE_AUDIO) {
390       List<byte[]> csd;
391       if (trackElement.csd != null) {
392         csd = Arrays.asList(trackElement.csd);
393       } else {
394         csd = Collections.singletonList(CodecSpecificDataUtil.buildAacAudioSpecificConfig(
395             trackFormat.audioSamplingRate, trackFormat.numChannels));
396       }
397       MediaFormat format = MediaFormat.createAudioFormat(mimeType, MediaFormat.NO_VALUE,
398           trackFormat.numChannels, trackFormat.audioSamplingRate, csd);
399       return format;
400     } else if (streamElement.type == StreamElement.TYPE_TEXT) {
401       return MediaFormat.createTextFormat(trackFormat.mimeType);
402     }
403     return null;
404   }

406   private static MediaChunk newMediaChunk(Format formatInfo, Uri uri, String cacheKey,
407       ChunkExtractorWrapper extractorWrapper, DrmInitData drmInitData, DataSource dataSource,
408       int chunkIndex, boolean isLast, long chunkStartTimeUs, long chunkEndTimeUs,
409       int trigger, MediaFormat mediaFormat) {
410     long offset = 0;
411     DataSpec dataSpec = new DataSpec(uri, offset, -1, cacheKey);
412     // In SmoothStreaming each chunk contains sample timestamps relative to the start of the chunk.
413     // To convert them the absolute timestamps, we need to set sampleOffsetUs to -chunkStartTimeUs.
414     return new ContainerMediaChunk(dataSource, dataSpec, trigger, formatInfo, chunkStartTimeUs,
415         chunkEndTimeUs, chunkIndex, isLast, chunkStartTimeUs, extractorWrapper, mediaFormat,
416         drmInitData, true);
417   }

419   private static byte[] getKeyId(byte[] initData) {
420     StringBuilder initDataStringBuilder = new StringBuilder();
421     for (int i = 0; i < initData.length; i += 2) {
422       initDataStringBuilder.append((char) initData[i]);
423     }
424     String initDataString = initDataStringBuilder.toString();
425     String keyIdString = initDataString.substring(
426         initDataString.indexOf("<KID>") + 5, initDataString.indexOf("</KID>"));
427     byte[] keyId = Base64.decode(keyIdString, Base64.DEFAULT);
428     swap(keyId, 0, 3);
429     swap(keyId, 1, 2);
430     swap(keyId, 4, 5);
431     swap(keyId, 6, 7);
432     return keyId;
433   }
434 
435   private static void swap(byte[] data, int firstPosition, int secondPosition) {
436     byte temp = data[firstPosition];
437     data[firstPosition] = data[secondPosition];
438     data[secondPosition] = temp;
439   }
