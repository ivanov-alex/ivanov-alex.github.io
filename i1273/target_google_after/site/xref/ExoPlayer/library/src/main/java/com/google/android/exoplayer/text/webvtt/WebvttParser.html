36  /**
37   * A simple WebVTT parser.
38   * <p>
39   * @see <a href="http://dev.w3.org/html5/webvtt">WebVTT specification</a>
40   */
41  public class WebvttParser implements SubtitleParser {
42  
43    private static final String TAG = "WebvttParser";
44  
45    private static final long SAMPLING_RATE = 90;
46  
47    private static final String WEBVTT_FILE_HEADER_STRING = "^\uFEFF?WEBVTT((\\u0020|\u0009).*)?$";
48    private static final Pattern WEBVTT_FILE_HEADER =
49        Pattern.compile(WEBVTT_FILE_HEADER_STRING);
50  
51    private static final String WEBVTT_METADATA_HEADER_STRING = "\\S*[:=]\\S*";
52    private static final Pattern WEBVTT_METADATA_HEADER =
53        Pattern.compile(WEBVTT_METADATA_HEADER_STRING);
54  
55    private static final String WEBVTT_CUE_IDENTIFIER_STRING = "^(?!.*(-->)).*$";
56    private static final Pattern WEBVTT_CUE_IDENTIFIER =
57        Pattern.compile(WEBVTT_CUE_IDENTIFIER_STRING);
58  
59    private static final String WEBVTT_TIMESTAMP_STRING = "(\\d+:)?[0-5]\\d:[0-5]\\d\\.\\d{3}";
60    private static final Pattern WEBVTT_TIMESTAMP = Pattern.compile(WEBVTT_TIMESTAMP_STRING);
61  
62    private static final String WEBVTT_CUE_SETTING_STRING = "\\S*:\\S*";
63    private static final Pattern WEBVTT_CUE_SETTING = Pattern.compile(WEBVTT_CUE_SETTING_STRING);
64  
65    private static final Pattern MEDIA_TIMESTAMP_OFFSET =
66        Pattern.compile(C.WEBVTT_EXO_HEADER_OFFSET + "\\-?\\d+");
67    private static final Pattern MEDIA_TIMESTAMP = Pattern.compile("MPEGTS:\\d+");
68  
69    private static final String NON_NUMERIC_STRING = ".*[^0-9].*";
70  
71    private final StringBuilder textBuilder;
72  
73    private final boolean strictParsing;
74  
75    /**
76     * Equivalent to {@code WebvttParser(false)}.
77     */
78    public WebvttParser() {
79      this(false);
80    }
81  
82    /**
83     * @param strictParsing If true, {@link #parse(InputStream, String, long)} will throw a
84     *     {@link ParserException} if the stream contains invalid data. If false, the parser will
85     *     make a best effort to ignore minor errors in the stream. Note however that a
86     *     {@link ParserException} will still be thrown when this is not possible.
87     */
88    public WebvttParser(boolean strictParsing) {
89      this.strictParsing = strictParsing;
90      textBuilder = new StringBuilder();
91    }
92  
93    @Override
94    public final WebvttSubtitle parse(InputStream inputStream, String inputEncoding, long startTimeUs)
95        throws IOException {
96      ArrayList<WebvttCue> subtitles = new ArrayList<>();
97      long mediaTimestampUs = startTimeUs;
98      long mediaTimestampOffsetUs = 0;
99  
100     BufferedReader webvttData = new BufferedReader(new InputStreamReader(inputStream, C.UTF8_NAME));
101     String line;
102 
103     // file should start with "WEBVTT" on the first line or "EXO-HEADER"
104     line = webvttData.readLine();
105     if (line == null) {
106       throw new ParserException("Expected WEBVTT or EXO-HEADER. Got null");
107     }
108 
109     if (line.startsWith(C.WEBVTT_EXO_HEADER)) {
110       // parse the timestamp offset, if present
111       Matcher matcher = MEDIA_TIMESTAMP_OFFSET.matcher(line);
112       if (matcher.find()) {
113         mediaTimestampOffsetUs = Long.parseLong(matcher.group().substring(7));
114       }
115 
116       // read the next line, which should now be WEBVTT
117       line = webvttData.readLine();
118       if (line == null) {
119         throw new ParserException("Expected WEBVTT. Got null");
120       }
121     }
122 
123     if (!WEBVTT_FILE_HEADER.matcher(line).matches()) {
124       throw new ParserException("Expected WEBVTT. Got " + line);
125     }
126 
127     // parse the remainder of the header
128     while (true) {
129       line = webvttData.readLine();
130       if (line == null) {
131         // we reached EOF before finishing the header
132         throw new ParserException("Expected an empty line after webvtt header");
133       } else if (line.isEmpty()) {
134         // we've read the newline that separates the header from the body
135         break;
136       }
137 
138       Matcher matcher = WEBVTT_METADATA_HEADER.matcher(line);
139       if (!matcher.find()) {
140         handleNoncompliantLine(line);
141       }
142 
143       if (line.startsWith("X-TIMESTAMP-MAP")) {
144         // parse the media timestamp
145         Matcher timestampMatcher = MEDIA_TIMESTAMP.matcher(line);
146         if (!timestampMatcher.find()) {
147           throw new ParserException("X-TIMESTAMP-MAP doesn't contain media timestamp: " + line);
148         } else {
149           mediaTimestampUs = (Long.parseLong(timestampMatcher.group().substring(7)) * 1000)
150               / SAMPLING_RATE + mediaTimestampOffsetUs;
151         }
152         mediaTimestampUs = getAdjustedStartTime(mediaTimestampUs);
153       }
154     }
155 
156     // process the cues and text
157     while ((line = webvttData.readLine()) != null) {
158 
159       // parse the cue identifier (if present) {
160       Matcher matcher = WEBVTT_CUE_IDENTIFIER.matcher(line);
161       if (matcher.find()) {
162         // ignore the identifier (we currently don't use it) and read the next line
163         line = webvttData.readLine();
164       }
165 
166       long startTime = Cue.UNSET_VALUE;
167       long endTime = Cue.UNSET_VALUE;
168       CharSequence text = null;
169       int lineNum = Cue.UNSET_VALUE;
170       int position = Cue.UNSET_VALUE;
171       Alignment alignment = null;
172       int size = Cue.UNSET_VALUE;
173 
174       // parse the cue timestamps
175       matcher = WEBVTT_TIMESTAMP.matcher(line);
176 
177       // parse start timestamp
178       if (!matcher.find()) {
179         throw new ParserException("Expected cue start time: " + line);
180       } else {
181         startTime = parseTimestampUs(matcher.group()) + mediaTimestampUs;
182       }
183 
184       // parse end timestamp
185       String endTimeString;
186       if (!matcher.find()) {
187         throw new ParserException("Expected cue end time: " + line);
188       } else {
189         endTimeString = matcher.group();
190         endTime = parseTimestampUs(endTimeString) + mediaTimestampUs;
191       }
192 
193       // parse the (optional) cue setting list
194       line = line.substring(line.indexOf(endTimeString) + endTimeString.length());
195       matcher = WEBVTT_CUE_SETTING.matcher(line);
196       while (matcher.find()) {
197         String match = matcher.group();
198         String[] parts = match.split(":", 2);
199         String name = parts[0];
200         String value = parts[1];
201 
202         try {
203           if ("line".equals(name)) {
204             if (value.endsWith("%")) {
205               lineNum = parseIntPercentage(value);
206             } else if (value.matches(NON_NUMERIC_STRING)) {
207               Log.w(TAG, "Invalid line value: " + value);
208             } else {
209               lineNum = Integer.parseInt(value);
210             }
211           } else if ("align".equals(name)) {
212             // TODO: handle for RTL languages
213             if ("start".equals(value)) {
214               alignment = Alignment.ALIGN_NORMAL;
215             } else if ("middle".equals(value)) {
216               alignment = Alignment.ALIGN_CENTER;
217             } else if ("end".equals(value)) {
218               alignment = Alignment.ALIGN_OPPOSITE;
219             } else if ("left".equals(value)) {
220               alignment = Alignment.ALIGN_NORMAL;
221             } else if ("right".equals(value)) {
222               alignment = Alignment.ALIGN_OPPOSITE;
223             } else {
224               Log.w(TAG, "Invalid align value: " + value);
225             }
226           } else if ("position".equals(name)) {
227             position = parseIntPercentage(value);
228           } else if ("size".equals(name)) {
229             size = parseIntPercentage(value);
230           } else {
231             Log.w(TAG, "Unknown cue setting " + name + ":" + value);
232           }
233         } catch (NumberFormatException e) {
234           Log.w(TAG, name + " contains an invalid value " + value, e);
235         }
236       }
237 
238       // parse text
239       textBuilder.setLength(0);
240       while (((line = webvttData.readLine()) != null) && (!line.isEmpty())) {
241         if (textBuilder.length() > 0) {
242           textBuilder.append("<br>");
243         }
244         textBuilder.append(line.trim());
245       }
246       text = Html.fromHtml(textBuilder.toString());
247 
248       WebvttCue cue = new WebvttCue(startTime, endTime, text, lineNum, position, alignment, size);
249       subtitles.add(cue);
250     }
251 
252     return new WebvttSubtitle(subtitles, mediaTimestampUs);
253   }
254 
255   @Override
256   public final boolean canParse(String mimeType) {
257     return MimeTypes.TEXT_VTT.equals(mimeType);
258   }
259 
260   protected long getAdjustedStartTime(long startTimeUs) {
261     return startTimeUs;
262   }
263 
264   protected void handleNoncompliantLine(String line) throws ParserException {
265     if (strictParsing) {
266       throw new ParserException("Unexpected line: " + line);
267     }
268   }
269 
270   private static int parseIntPercentage(String s) throws NumberFormatException {
271     if (!s.endsWith("%")) {
272       throw new NumberFormatException(s + " doesn't end with '%'");
273     }
274 
275     s = s.substring(0, s.length() - 1);
276     if (s.matches(NON_NUMERIC_STRING)) {
277       throw new NumberFormatException(s + " contains an invalid character");
278     }
279 
280     int value = Integer.parseInt(s);
281     if (value < 0 || value > 100) {
282       throw new NumberFormatException(value + " is out of range [0-100]");
283     }
284     return value;
285   }
286 
287   private static long parseTimestampUs(String s) throws NumberFormatException {
288     if (!s.matches(WEBVTT_TIMESTAMP_STRING)) {
289       throw new NumberFormatException("has invalid format");
290     }
291 
292     String[] parts = s.split("\\.", 2);
293     long value = 0;
294     for (String group : parts[0].split(":")) {
295       value = value * 60 + Long.parseLong(group);
296     }
297     return (value * 1000 + Long.parseLong(parts[1])) * 1000;
298   }
