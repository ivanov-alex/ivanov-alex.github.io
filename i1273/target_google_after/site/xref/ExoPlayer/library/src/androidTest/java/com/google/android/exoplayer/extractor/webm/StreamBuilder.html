178   /**
179    * Serializes the constructed stream to a {@code byte[]} using the specified number of cue points.
180    */
181   public byte[] build(int cuePointCount) {
182     Assertions.checkNotNull(header);
183     Assertions.checkNotNull(info);
184 
185     EbmlElement tracks = element(0x1654AE6B, trackEntries.toArray(new EbmlElement[0]));
186     EbmlElement[] children;
187 
188     if (cuePointCount == 0) {
189       children = new EbmlElement[2 + mediaSegments.size()];
190       System.arraycopy(mediaSegments.toArray(new EbmlElement[0]), 0, children, 2,
191           mediaSegments.size());
192       children[0] = info;
193       children[1] = tracks;
194     } else {
195       // Get the size of the initialization segment.
196       EbmlElement[] cuePointElements = new EbmlElement[cuePointCount];
197       for (int i = 0; i < cuePointCount; i++) {
198         cuePointElements[i] = createCuePointElement(10 * i, 0);
199       }
200       EbmlElement cues = element(0x1C53BB6B, cuePointElements); // Cues
201       long initializationSegmentSize = info.getSize() + tracks.getSize() + cues.getSize();
202 
203       // Recreate the initialization segment using its size as an offset.
204       for (int i = 0; i < cuePointCount; i++) {
205         cuePointElements[i] = createCuePointElement(10 * i, (int) initializationSegmentSize);
206       }
207       cues = element(0x1C53BB6B, cuePointElements); // Cues
208 
209       // Build the top-level segment element.
210       children = new EbmlElement[3 + mediaSegments.size()];
211       System.arraycopy(mediaSegments.toArray(new EbmlElement[0]), 0, children, 3,
212           mediaSegments.size());
213       children[0] = info;
214       children[1] = tracks;
215       children[2] = cues;
216     }
217 
218     EbmlElement segmentElement = element(0x18538067, children); // Segment
219 
220     // Serialize the EBML header and the top-level segment element.
221     return EbmlElement.serialize(header, segmentElement);
222   }

224   private static EbmlElement createCuePointElement(int cueTime, int cueClusterPosition) {
225     byte[] positionBytes = getLongBytes(cueClusterPosition);
226     return element(0xBB, // CuePoint
227         element(0xB3, (byte) (cueTime & 0xFF)), // CueTime
228         element(0xB7, // CueTrackPositions
229             element(0xF1, positionBytes))); // CueClusterPosition
230   }

232   private static EbmlElement createEbmlElement(int ebmlReadVersion, String docType,
233       int docTypeReadVersion) {
234     return element(0x1A45DFA3, // EBML
235         element(0x42F7, (byte) (ebmlReadVersion & 0xFF)), // EBMLReadVersion
236         element(0x4282, docType.getBytes()), // DocType
237         element(0x4285, (byte) (docTypeReadVersion & 0xFF))); // DocTypeReadVersion

240   private EbmlElement createInfoElement(int timecodeScale, long durationTimecode,
241       boolean durationFirst, boolean omitDefaultTimecodeScale) {
242     byte[] timecodeScaleBytes = getIntegerBytes(timecodeScale);
243     byte[] durationBytes = getLongBytes(Double.doubleToLongBits(durationTimecode));
244     EbmlElement durationElement = element(0x4489, durationBytes);
245     EbmlElement timescaleElement = element(0x2AD7B1, timecodeScaleBytes);
246     if (omitDefaultTimecodeScale && timecodeScale == 1000000) {
247       return element(0x1549A966, // Info
248           durationElement);
249     }
250     return element(0x1549A966, // Info
251         durationFirst ? durationElement : timescaleElement,
252         durationFirst ? timescaleElement : durationElement);
253   }

255   private static EbmlElement createVideoTrackEntry(String codecId, int pixelWidth, int pixelHeight,
256       ContentEncodingSettings contentEncodingSettings, byte[] codecPrivate) {
257     byte[] widthBytes = getIntegerBytes(pixelWidth);
258     byte[] heightBytes = getIntegerBytes(pixelHeight);
259     EbmlElement contentEncodingSettingsElement;
260     if (contentEncodingSettings != null) {
261       EbmlElement encryptionOrCompressionElement;
262       if (contentEncodingSettings.type == 0) {
263         encryptionOrCompressionElement = element(0x5034, // ContentCompression
264             element(0x4254, (byte) (contentEncodingSettings.algorithm & 0xFF)), // ContentCompAlgo
265             element(0x4255, contentEncodingSettings.strippedBytes)); // ContentCompSettings
266       } else if (contentEncodingSettings.type == 1) {
267         encryptionOrCompressionElement = element(0x5035, // ContentEncryption
268             // ContentEncAlgo
269             element(0x47E1, (byte) (contentEncodingSettings.algorithm & 0xFF)),
270             element(0x47E2, TEST_ENCRYPTION_KEY_ID), // ContentEncKeyID
271             element(0x47E7, // ContentEncAESSettings
272                 // AESSettingsCipherMode
273                 element(0x47E8, (byte) (contentEncodingSettings.aesCipherMode & 0xFF))));
274       } else {
275         throw new IllegalArgumentException("Unexpected encoding type.");
276       }
277 
278       contentEncodingSettingsElement =
279           element(0x6D80, // ContentEncodings
280             element(0x6240, // ContentEncoding
281                 // ContentEncodingOrder
282                 element(0x5031, (byte) (contentEncodingSettings.order & 0xFF)),
283                 // ContentEncodingScope
284                 element(0x5032, (byte) (contentEncodingSettings.scope & 0xFF)),
285                 // ContentEncodingType
286                 element(0x5033, (byte) (contentEncodingSettings.type & 0xFF)),
287                 encryptionOrCompressionElement));
288     } else {
289       contentEncodingSettingsElement = empty();
290     }
291     EbmlElement codecPrivateElement;
292     if (codecPrivate != null) {
293       codecPrivateElement = element(0x63A2, codecPrivate); // CodecPrivate
294     } else {
295       codecPrivateElement = empty();
296     }
297 
298     return element(0xAE, // TrackEntry
299         element(0x86, codecId.getBytes()), // CodecID
300         element(0xD7, (byte) 0x01), // TrackNumber
301         element(0x83, (byte) 0x01), // TrackType
302         contentEncodingSettingsElement,
303         element(0xE0, // Video
304             element(0xB0, widthBytes[2], widthBytes[3]),
305             element(0xBA, heightBytes[2], heightBytes[3])),
306         codecPrivateElement);
307   }

309   private static EbmlElement createAudioTrackEntry(String codecId, int channelCount, int sampleRate,
310       byte[] codecPrivate, int codecDelay, int seekPreRoll, int defaultDurationNs) {
311     byte channelCountByte = (byte) (channelCount & 0xFF);
312     byte[] sampleRateDoubleBytes = getLongBytes(Double.doubleToLongBits(sampleRate));
313     return element(0xAE, // TrackEntry
314         element(0x86, codecId.getBytes()), // CodecID
315         element(0xD7, (byte) 0x02), // TrackNumber
316         element(0x83, (byte) 0x02), // TrackType
317         // CodecDelay
318         codecDelay != NO_VALUE ? element(0x56AA, getIntegerBytes(codecDelay)) : empty(),
319         // SeekPreRoll
320         seekPreRoll != NO_VALUE ? element(0x56BB, getIntegerBytes(seekPreRoll)) : empty(),
321         element(0xE1, // Audio
322             element(0x9F, channelCountByte), // Channels
323             element(0xB5, sampleRateDoubleBytes)), // SamplingFrequency
324         // DefaultDuration
325         defaultDurationNs != NO_VALUE ? element(0x23E383, getIntegerBytes(defaultDurationNs))
326             : empty(),
327         element(0x63A2, codecPrivate)); // CodecPrivate
328   }

336   private static EbmlElement createSimpleBlock(int trackNumber, int timecode, int flags,
337       boolean encrypted, boolean validSignalByte, int lacingFrameCount, byte[] data) {
338     byte[] trackNumberBytes = getIntegerBytes(trackNumber);
339     byte[] timeBytes = getIntegerBytes(timecode);
340     byte[] simpleBlockBytes;
341     if (lacingFrameCount > 1) {
342       flags |= 0x04; // Fixed-size lacing
343       simpleBlockBytes = TestUtil.createByteArray(
344           0x40, trackNumberBytes[3], // Track number size=2
345           timeBytes[2], timeBytes[3], flags, lacingFrameCount - 1); // Timecode, flags and lacing.
346     } else {
347       simpleBlockBytes = TestUtil.createByteArray(
348           0x40, trackNumberBytes[3], // Track number size=2
349           timeBytes[2], timeBytes[3], flags); // Timecode and flags
350     }
351     if (encrypted) {
352       simpleBlockBytes = TestUtil.joinByteArrays(
353           simpleBlockBytes, TestUtil.createByteArray(validSignalByte ? 0x01 : 0x80),
354           Arrays.copyOfRange(TEST_INITIALIZATION_VECTOR, 0, 8));
355     }
356     return element(0xA3, // SimpleBlock
357         TestUtil.joinByteArrays(simpleBlockBytes, data));
358   }

360   private static EbmlElement createSimpleBlock(int trackNumber, int timecode, int flags,
361       boolean encrypted, boolean validSignalByte, byte[] data, int... xiphLacingSampleSizes) {
362     byte[] trackNumberBytes = getIntegerBytes(trackNumber);
363     byte[] timeBytes = getIntegerBytes(timecode);
364     byte[] simpleBlockBytes;
365     flags |= 0x02; // Xiph lacing
366     simpleBlockBytes = TestUtil.createByteArray(
367         0x40, trackNumberBytes[3], // Track number size=2
368         timeBytes[2], timeBytes[3], // Timecode
369         flags, xiphLacingSampleSizes.length - 1); // Flags and lacing.
370     int lacingBufferSize = 0;
371     for (int sampleIndex = 0; sampleIndex < xiphLacingSampleSizes.length - 1; sampleIndex++) {
372       lacingBufferSize += (xiphLacingSampleSizes[sampleIndex] + 254) / 255;
373     }
374     ByteBuffer lacingBytes = ByteBuffer.allocate(lacingBufferSize);
375     for (int sampleIndex = 0; sampleIndex < xiphLacingSampleSizes.length - 1; sampleIndex++) {
376       int sampleSize = xiphLacingSampleSizes[sampleIndex];
377       while (sampleSize > 255) {
378         sampleSize -= 255;
379         lacingBytes.put((byte) 0xFF);
380       }
381       lacingBytes.put((byte) sampleSize);
382     }
383     simpleBlockBytes = TestUtil.joinByteArrays(simpleBlockBytes, lacingBytes.array());
384 
385     if (encrypted) {
386       simpleBlockBytes = TestUtil.joinByteArrays(
387           simpleBlockBytes, TestUtil.createByteArray(validSignalByte ? 0x01 : 0x80),
388           Arrays.copyOfRange(TEST_INITIALIZATION_VECTOR, 0, 8));
389     }
390     return element(0xA3, // SimpleBlock
391         TestUtil.joinByteArrays(simpleBlockBytes, data));
392   }

394   private static EbmlElement createBlock(int trackNumber, int timecode, boolean keyframe, int flags,
395       byte[] data) {
396     byte[] trackNumberBytes = getIntegerBytes(trackNumber);
397     byte[] timeBytes = getIntegerBytes(timecode);
398     byte[] blockBytes = TestUtil.createByteArray(
399         0x40, trackNumberBytes[3], // Track number size=2
400         timeBytes[2], timeBytes[3], flags); // Timecode and flags
401     EbmlElement block = element(0xA1, // Block
402         TestUtil.joinByteArrays(blockBytes, data));
403     EbmlElement referenceBlock = keyframe ? empty() : element(0xFB, (byte) 0x00); // ReferenceBlock
404     return element(0xA0, // BlockGroup
405         referenceBlock,
406         block);
407   }

409   private static byte[] getIntegerBytes(int value) {
410     return TestUtil.createByteArray(
411         (value & 0xFF000000) >> 24,
412         (value & 0x00FF0000) >> 16,
413         (value & 0x0000FF00) >> 8,
414         (value & 0x000000FF));
415   }

417   private static byte[] getLongBytes(long value) {
418     byte[] result = new byte[8];
419     for (int i = 0; i < 8; i++) {
420       result[7 - i] = (byte) ((value >> (8 * i)) & 0xFF);
421     }
422     return result;
423   }

440   /** Represents a WebM EBML element that can be serialized as a byte array. */
441   private static final class EbmlElement {
442 
443     /** Returns a byte[] containing the concatenation of the data from all {@code elements}. */
444     public static byte[] serialize(EbmlElement... elements) {
445       int size = 0;
446       for (EbmlElement element : elements) {
447         size += element.getSize();
448       }
449       ByteBuffer buffer = ByteBuffer.allocate(size);
450       for (EbmlElement element : elements) {
451         element.getData(buffer);
452       }
453       return buffer.array();
454     }
455 
456     private final int id;
457     private final EbmlElement[] childElements;
458     private final byte[] payload;
459 
460     /** Creates an element containing the specified {@code childElements}. */
461     EbmlElement(int id, EbmlElement... childElements) {
462       this.id = id;
463       this.childElements = childElements;
464       payload = null;
465     }
466 
467     /** Creates an element containing the specified {@code payload}. */
468     EbmlElement(int id, byte... payload) {
469       this.id = id;
470       this.payload = payload;
471       childElements = null;
472     }
473 
474     /** Creates a completely empty element that will contribute no bytes to the output. */
475     EbmlElement() {
476       id = NO_VALUE;
477       payload = null;
478       childElements = null;
479     }
480 
481     private long getSize() {
482       if (id == NO_VALUE) {
483         return 0;
484       }
485 
486       long payloadSize = getPayloadSize();
487       return getIdLength() + getVIntLength(payloadSize) + payloadSize;
488     }
489 
490     private long getPayloadSize() {
491       if (payload != null) {
492         return payload.length;
493       }
494 
495       int payloadSize = 0;
496       for (EbmlElement element : childElements) {
497         payloadSize += element.getSize();
498       }
499       return payloadSize;
500     }
501 
502     private void getData(ByteBuffer byteBuffer) {
503       if (id == NO_VALUE) {
504         return;
505       }
506 
507       putId(byteBuffer);
508       putVInt(byteBuffer, getPayloadSize());
509       if (payload != null) {
510         byteBuffer.put(payload);
511       } else {
512         for (EbmlElement atom : childElements) {
513           atom.getData(byteBuffer);
514         }
515       }
516     }
517 
518     private int getIdLength() {
519       if (id == NO_VALUE) {
520         return 0;
521       }
522 
523       for (int i = 0; i < 4; i++) {
524         if (id < 1 << (7 * i + 8)) {
525           return i + 1;
526         }
527       }
528       throw new IllegalArgumentException();
529     }
530 
531     private static int getVIntLength(long vInt) {
532       for (int i = 1; i < 9; i++) {
533         if (vInt < 1 << (7 * i)) {
534           return i;
535         }
536       }
537       throw new IllegalArgumentException();
538     }
539 
540     private void putId(ByteBuffer byteBuffer) {
541       int length = getIdLength();
542       for (int i = length - 1; i >= 0; i--) {
543         byteBuffer.put((byte) ((id >> (i * 8)) & 0xFF));
544       }
545     }
546 
547     private static void putVInt(ByteBuffer byteBuffer, long vInt) {
548       int vIntLength = getVIntLength(vInt);
549       vInt |= 1 << (vIntLength * 7);
550       for (int i = vIntLength - 1; i >= 0; i--) {
551         byteBuffer.put((byte) ((vInt >> (i * 8)) & 0xFF));
552       }
553     }
554 
555   }
