56  /**
57   * A {@link RendererBuilder} for SmoothStreaming.
58   */
59  public class SmoothStreamingRendererBuilder implements RendererBuilder {
60  
61    private static final int BUFFER_SEGMENT_SIZE = 64 * 1024;
62    private static final int VIDEO_BUFFER_SEGMENTS = 200;
63    private static final int AUDIO_BUFFER_SEGMENTS = 60;
64    private static final int TEXT_BUFFER_SEGMENTS = 2;
65    private static final int LIVE_EDGE_LATENCY_MS = 30000;
66  
67    private final Context context;
68    private final String userAgent;
69    private final String url;
70    private final MediaDrmCallback drmCallback;
71  
72    private AsyncRendererBuilder currentAsyncBuilder;
73  
74    public SmoothStreamingRendererBuilder(Context context, String userAgent, String url,
75        MediaDrmCallback drmCallback) {
76      this.context = context;
77      this.userAgent = userAgent;
78      this.url = Util.toLowerInvariant(url).endsWith("/manifest") ? url : url + "/Manifest";
79      this.drmCallback = drmCallback;
80    }
81  
82    @Override
83    public void buildRenderers(DemoPlayer player) {
84      currentAsyncBuilder = new AsyncRendererBuilder(context, userAgent, url, drmCallback, player);
85      currentAsyncBuilder.init();
86    }
87  
88    @Override
89    public void cancel() {
90      if (currentAsyncBuilder != null) {
91        currentAsyncBuilder.cancel();
92        currentAsyncBuilder = null;
93      }
94    }
95  
96    private static final class AsyncRendererBuilder
97        implements ManifestFetcher.ManifestCallback<SmoothStreamingManifest> {
98  
99      private final Context context;
100     private final String userAgent;
101     private final MediaDrmCallback drmCallback;
102     private final DemoPlayer player;
103     private final ManifestFetcher<SmoothStreamingManifest> manifestFetcher;
104 
105     private boolean canceled;
106 
107     public AsyncRendererBuilder(Context context, String userAgent, String url,
108         MediaDrmCallback drmCallback, DemoPlayer player) {
109       this.context = context;
110       this.userAgent = userAgent;
111       this.drmCallback = drmCallback;
112       this.player = player;
113       SmoothStreamingManifestParser parser = new SmoothStreamingManifestParser();
114       manifestFetcher = new ManifestFetcher<>(url, new DefaultHttpDataSource(userAgent, null),
115           parser);
116     }
117 
118     public void init() {
119       manifestFetcher.singleLoad(player.getMainHandler().getLooper(), this);
120     }
121 
122     public void cancel() {
123       canceled = true;
124     }
125 
126     @Override
127     public void onSingleManifestError(IOException exception) {
128       if (canceled) {
129         return;
130       }
131 
132       player.onRenderersError(exception);
133     }
134 
135     @Override
136     public void onSingleManifest(SmoothStreamingManifest manifest) {
137       if (canceled) {
138         return;
139       }
140 
141       Handler mainHandler = player.getMainHandler();
142       LoadControl loadControl = new DefaultLoadControl(new DefaultAllocator(BUFFER_SEGMENT_SIZE));
143       DefaultBandwidthMeter bandwidthMeter = new DefaultBandwidthMeter(mainHandler, player);
144 
145       // Check drm support if necessary.
146       DrmSessionManager drmSessionManager = null;
147       if (manifest.protectionElement != null) {
148         if (Util.SDK_INT < 18) {
149           player.onRenderersError(
150               new UnsupportedDrmException(UnsupportedDrmException.REASON_UNSUPPORTED_SCHEME));
151           return;
152         }
153         try {
154           drmSessionManager = new StreamingDrmSessionManager(manifest.protectionElement.uuid,
155               player.getPlaybackLooper(), drmCallback, null, player.getMainHandler(), player);
156         } catch (UnsupportedDrmException e) {
157           player.onRenderersError(e);
158           return;
159         }
160       }
161 
162       // Obtain stream elements for playback.
163       int audioStreamElementCount = 0;
164       int textStreamElementCount = 0;
165       int videoStreamElementIndex = -1;
166       for (int i = 0; i < manifest.streamElements.length; i++) {
167         if (manifest.streamElements[i].type == StreamElement.TYPE_AUDIO) {
168           audioStreamElementCount++;
169         } else if (manifest.streamElements[i].type == StreamElement.TYPE_TEXT) {
170           textStreamElementCount++;
171         } else if (videoStreamElementIndex == -1
172             && manifest.streamElements[i].type == StreamElement.TYPE_VIDEO) {
173           videoStreamElementIndex = i;
174         }
175       }
176 
177       // Determine which video tracks we should use for playback.
178       int[] videoTrackIndices = null;
179       if (videoStreamElementIndex != -1) {
180         try {
181           videoTrackIndices = VideoFormatSelectorUtil.selectVideoFormatsForDefaultDisplay(context,
182               Arrays.asList(manifest.streamElements[videoStreamElementIndex].tracks), null, false);
183         } catch (DecoderQueryException e) {
184           player.onRenderersError(e);
185           return;
186         }
187       }
188 
189       // Build the video renderer.
190       final MediaCodecVideoTrackRenderer videoRenderer;
191       if (videoTrackIndices == null || videoTrackIndices.length == 0) {
192         videoRenderer = null;
193       } else {
194         DataSource videoDataSource = new DefaultUriDataSource(context, bandwidthMeter, userAgent);
195         ChunkSource videoChunkSource = new SmoothStreamingChunkSource(manifestFetcher,
196             videoStreamElementIndex, videoTrackIndices, videoDataSource,
197             new AdaptiveEvaluator(bandwidthMeter), LIVE_EDGE_LATENCY_MS);
198         ChunkSampleSource videoSampleSource = new ChunkSampleSource(videoChunkSource, loadControl,
199             VIDEO_BUFFER_SEGMENTS * BUFFER_SEGMENT_SIZE, mainHandler, player,
200             DemoPlayer.TYPE_VIDEO);
201         videoRenderer = new MediaCodecVideoTrackRenderer(videoSampleSource, drmSessionManager, true,
202             MediaCodec.VIDEO_SCALING_MODE_SCALE_TO_FIT, 5000, null, mainHandler, player, 50);
203       }
204 
205       // Build the audio renderer.
206       final String[] audioTrackNames;
207       final MultiTrackChunkSource audioChunkSource;
208       final MediaCodecAudioTrackRenderer audioRenderer;
209       if (audioStreamElementCount == 0) {
210         audioTrackNames = null;
211         audioChunkSource = null;
212         audioRenderer = null;
213       } else {
214         audioTrackNames = new String[audioStreamElementCount];
215         ChunkSource[] audioChunkSources = new ChunkSource[audioStreamElementCount];
216         DataSource audioDataSource = new DefaultUriDataSource(context, bandwidthMeter, userAgent);
217         FormatEvaluator audioFormatEvaluator = new FormatEvaluator.FixedEvaluator();
218         audioStreamElementCount = 0;
219         for (int i = 0; i < manifest.streamElements.length; i++) {
220           if (manifest.streamElements[i].type == StreamElement.TYPE_AUDIO) {
221             audioTrackNames[audioStreamElementCount] = manifest.streamElements[i].name;
222             audioChunkSources[audioStreamElementCount] = new SmoothStreamingChunkSource(
223                 manifestFetcher, i, new int[] {0}, audioDataSource, audioFormatEvaluator,
224                 LIVE_EDGE_LATENCY_MS);
225             audioStreamElementCount++;
226           }
227         }
228         audioChunkSource = new MultiTrackChunkSource(audioChunkSources);
229         ChunkSampleSource audioSampleSource = new ChunkSampleSource(audioChunkSource, loadControl,
230             AUDIO_BUFFER_SEGMENTS * BUFFER_SEGMENT_SIZE, mainHandler, player,
231             DemoPlayer.TYPE_AUDIO);
232         audioRenderer = new MediaCodecAudioTrackRenderer(audioSampleSource, drmSessionManager, true,
233             mainHandler, player);
234       }
235 
236       // Build the text renderer.
237       final String[] textTrackNames;
238       final MultiTrackChunkSource textChunkSource;
239       final TrackRenderer textRenderer;
240       if (textStreamElementCount == 0) {
241         textTrackNames = null;
242         textChunkSource = null;
243         textRenderer = null;
244       } else {
245         textTrackNames = new String[textStreamElementCount];
246         ChunkSource[] textChunkSources = new ChunkSource[textStreamElementCount];
247         DataSource ttmlDataSource = new DefaultUriDataSource(context, bandwidthMeter, userAgent);
248         FormatEvaluator ttmlFormatEvaluator = new FormatEvaluator.FixedEvaluator();
249         textStreamElementCount = 0;
250         for (int i = 0; i < manifest.streamElements.length; i++) {
251           if (manifest.streamElements[i].type == StreamElement.TYPE_TEXT) {
252             textTrackNames[textStreamElementCount] = manifest.streamElements[i].language;
253             textChunkSources[textStreamElementCount] = new SmoothStreamingChunkSource(
254                 manifestFetcher, i, new int[] {0}, ttmlDataSource, ttmlFormatEvaluator,
255                 LIVE_EDGE_LATENCY_MS);
256             textStreamElementCount++;
257           }
258         }
259         textChunkSource = new MultiTrackChunkSource(textChunkSources);
260         ChunkSampleSource ttmlSampleSource = new ChunkSampleSource(textChunkSource, loadControl,
261             TEXT_BUFFER_SEGMENTS * BUFFER_SEGMENT_SIZE, mainHandler, player,
262             DemoPlayer.TYPE_TEXT);
263         textRenderer = new TextTrackRenderer(ttmlSampleSource, player, mainHandler.getLooper(),
264             new TtmlParser());
265       }
266 
267       // Invoke the callback.
268       String[][] trackNames = new String[DemoPlayer.RENDERER_COUNT][];
269       trackNames[DemoPlayer.TYPE_AUDIO] = audioTrackNames;
270       trackNames[DemoPlayer.TYPE_TEXT] = textTrackNames;
271 
272       MultiTrackChunkSource[] multiTrackChunkSources =
273           new MultiTrackChunkSource[DemoPlayer.RENDERER_COUNT];
274       multiTrackChunkSources[DemoPlayer.TYPE_AUDIO] = audioChunkSource;
275       multiTrackChunkSources[DemoPlayer.TYPE_TEXT] = textChunkSource;
276 
277       TrackRenderer[] renderers = new TrackRenderer[DemoPlayer.RENDERER_COUNT];
278       renderers[DemoPlayer.TYPE_VIDEO] = videoRenderer;
279       renderers[DemoPlayer.TYPE_AUDIO] = audioRenderer;
280       renderers[DemoPlayer.TYPE_TEXT] = textRenderer;
281       player.onRenderers(trackNames, multiTrackChunkSources, renderers, bandwidthMeter);
282     }
283 
284   }
