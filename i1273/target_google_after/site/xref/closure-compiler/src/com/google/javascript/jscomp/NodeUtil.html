72    /**
73     * Gets the boolean value of a node that represents a expression. This method
74     * effectively emulates the <code>Boolean()</code> JavaScript cast function.
75     * Note: unlike getPureBooleanValue this function does not return UNKNOWN
76     * for expressions with side-effects.
77     */
78    static TernaryValue getImpureBooleanValue(Node n) {
79      switch (n.getType()) {
80        case Token.ASSIGN:
81        case Token.COMMA:
82          // For ASSIGN and COMMA the value is the value of the RHS.
83          return getImpureBooleanValue(n.getLastChild());
84        case Token.NOT:
85          TernaryValue value = getImpureBooleanValue(n.getLastChild());
86          return value.not();
87        case Token.AND: {
88          TernaryValue lhs = getImpureBooleanValue(n.getFirstChild());
89          TernaryValue rhs = getImpureBooleanValue(n.getLastChild());
90          return lhs.and(rhs);
91        }
92        case Token.OR:  {
93          TernaryValue lhs = getImpureBooleanValue(n.getFirstChild());
94          TernaryValue rhs = getImpureBooleanValue(n.getLastChild());
95          return lhs.or(rhs);
96        }
97        case Token.HOOK:  {
98          TernaryValue trueValue = getImpureBooleanValue(
99              n.getFirstChild().getNext());
100         TernaryValue falseValue = getImpureBooleanValue(n.getLastChild());
101         if (trueValue.equals(falseValue)) {
102           return trueValue;
103         } else {
104           return TernaryValue.UNKNOWN;
105         }
106       }
107       case Token.ARRAYLIT:
108       case Token.OBJECTLIT:
109         // ignoring side-effects
110         return TernaryValue.TRUE;
111 
112       case Token.VOID:
113         return TernaryValue.FALSE;
114 
115       default:
116         return getPureBooleanValue(n);
117     }
118   }

120   /**
121    * Gets the boolean value of a node that represents a literal. This method
122    * effectively emulates the <code>Boolean()</code> JavaScript cast function
123    * except it return UNKNOWN for known values with side-effects, use
124    * getImpureBooleanValue if you don't care about side-effects.
125    */
126   static TernaryValue getPureBooleanValue(Node n) {
127     switch (n.getType()) {
128       case Token.TEMPLATELIT:
129         if (n.hasOneChild()) {
130           return TernaryValue.forBoolean(!n.getFirstChild().getString().isEmpty());
131         }
132         break;
133 
134       case Token.STRING:
135         return TernaryValue.forBoolean(n.getString().length() > 0);
136 
137       case Token.NUMBER:
138         return TernaryValue.forBoolean(n.getDouble() != 0);
139 
140       case Token.NOT:
141         return getPureBooleanValue(n.getLastChild()).not();
142 
143       case Token.NULL:
144       case Token.FALSE:
145         return TernaryValue.FALSE;
146 
147       case Token.VOID:
148         if (!mayHaveSideEffects(n.getFirstChild())) {
149           return TernaryValue.FALSE;
150         }
151         break;
152 
153       case Token.NAME:
154         String name = n.getString();
155         if ("undefined".equals(name)
156             || "NaN".equals(name)) {
157           // We assume here that programs don't change the value of the keyword
158           // undefined to something other than the value undefined.
159           return TernaryValue.FALSE;
160         } else if ("Infinity".equals(name)) {
161           return TernaryValue.TRUE;
162         }
163         break;
164 
165       case Token.TRUE:
166       case Token.REGEXP:
167       case Token.CLASS:
168         return TernaryValue.TRUE;
169 
170       case Token.ARRAYLIT:
171       case Token.OBJECTLIT:
172         if (!mayHaveSideEffects(n)) {
173           return TernaryValue.TRUE;
174         }
175         break;
176     }
177 
178     return TernaryValue.UNKNOWN;
179   }

181   /**
182    * Gets the value of a node as a String, or null if it cannot be converted.
183    * When it returns a non-null String, this method effectively emulates the
184    * <code>String()</code> JavaScript cast function.
185    */
186   static String getStringValue(Node n) {
187     // TODO(user): regex literals as well.
188     switch (n.getType()) {
189       case Token.STRING:
190       case Token.STRING_KEY:
191         return n.getString();
192 
193       case Token.NAME:
194         String name = n.getString();
195         if ("undefined".equals(name)
196             || "Infinity".equals(name)
197             || "NaN".equals(name)) {
198           return name;
199         }
200         break;
201 
202       case Token.NUMBER:
203         return getStringValue(n.getDouble());
204 
205       case Token.FALSE:
206         return "false";
207 
208       case Token.TRUE:
209         return "true";
210 
211       case Token.NULL:
212         return "null";
213 
214       case Token.VOID:
215         return "undefined";
216 
217       case Token.NOT:
218         TernaryValue child = getPureBooleanValue(n.getFirstChild());
219         if (child != TernaryValue.UNKNOWN) {
220           return child.toBoolean(true) ? "false" : "true"; // reversed.
221         }
222         break;
223 
224       case Token.ARRAYLIT:
225         return arrayToString(n);
226 
227       case Token.OBJECTLIT:
228         return "[object Object]";
229     }
230     return null;
231   }

234     long longValue = (long) value;
235 
236     // Return "1" instead of "1.0"
237     if (longValue == value) {
238       return Long.toString(longValue);
239     } else {
240       return Double.toString(value);
241     }
242   }

245    * When converting arrays to string using Array.prototype.toString or
246    * Array.prototype.join, the rules for conversion to String are different
247    * than converting each element individually.  Specifically, "null" and
248    * "undefined" are converted to an empty string.
249    * @param n A node that is a member of an Array.
250    * @return The string representation.

253     return (NodeUtil.isNullOrUndefined(n) || n.isEmpty())

257   static String arrayToString(Node literal) {
258     Node first = literal.getFirstChild();
259     StringBuilder result = new StringBuilder();
260     for (Node n = first; n != null; n = n.getNext()) {
261       String childValue = getArrayElementStringValue(n);
262       if (childValue == null) {
263         return null;
264       }
265       if (n != first) {
266         result.append(',');
267       }
268       result.append(childValue);
269     }
270     return result.toString();
271   }

273   /**
274    * Gets the value of a node as a Number, or null if it cannot be converted.
275    * When it returns a non-null Double, this method effectively emulates the
276    * <code>Number()</code> JavaScript cast function.
277    */
278   static Double getNumberValue(Node n) {
279     switch (n.getType()) {
280       case Token.TRUE:
281         return 1.0;
282 
283       case Token.FALSE:
284       case Token.NULL:
285         return 0.0;
286 
287       case Token.NUMBER:
288         return n.getDouble();
289 
290       case Token.VOID:
291         if (mayHaveSideEffects(n.getFirstChild())) {
292           return null;
293         } else {
294           return Double.NaN;
295         }
296 
297       case Token.NAME:
298         // Check for known constants
299         String name = n.getString();
300         if (name.equals("undefined")) {
301           return Double.NaN;
302         }
303         if (name.equals("NaN")) {
304           return Double.NaN;
305         }
306         if (name.equals("Infinity")) {
307           return Double.POSITIVE_INFINITY;
308         }
309         return null;
310 
311       case Token.NEG:
312         if (n.getChildCount() == 1 && n.getFirstChild().isName()
313             && n.getFirstChild().getString().equals("Infinity")) {
314           return Double.NEGATIVE_INFINITY;
315         }
316         return null;
317 
318       case Token.NOT:
319         TernaryValue child = getPureBooleanValue(n.getFirstChild());
320         if (child != TernaryValue.UNKNOWN) {
321           return child.toBoolean(true) ? 0.0 : 1.0; // reversed.
322         }
323         break;
324 
325       case Token.STRING:
326         return getStringNumberValue(n.getString());
327 
328       case Token.ARRAYLIT:
329       case Token.OBJECTLIT:
330         String value = getStringValue(n);
331         return value != null ? getStringNumberValue(value) : null;
332     }
333 
334     return null;
335   }

337   static Double getStringNumberValue(String rawJsString) {
338     if (rawJsString.contains("\u000b")) {
339       // vertical tab is not always whitespace
340       return null;
341     }
342 
343     String s = trimJsWhiteSpace(rawJsString);
344     // return ScriptRuntime.toNumber(s);
345     if (s.isEmpty()) {
346       return 0.0;
347     }
348 
349     if (s.length() > 2
350         && s.charAt(0) == '0'
351         && (s.charAt(1) == 'x' || s.charAt(1) == 'X')) {
352       // Attempt to convert hex numbers.
353       try {
354         return Double.valueOf(Integer.parseInt(s.substring(2), 16));
355       } catch (NumberFormatException e) {
356         return Double.NaN;
357       }
358     }
359 
360     if (s.length() > 3
361         && (s.charAt(0) == '-' || s.charAt(0) == '+')
362         && s.charAt(1) == '0'
363         && (s.charAt(2) == 'x' || s.charAt(2) == 'X')) {
364       // hex numbers with explicit signs vary between browsers.
365       return null;
366     }
367 
368     // Firefox and IE treat the "Infinity" differently. Firefox is case
369     // insensitive, but IE treats "infinity" as NaN.  So leave it alone.
370     if (s.equals("infinity")
371         || s.equals("-infinity")
372         || s.equals("+infinity")) {
373       return null;
374     }
375 
376     try {
377       return Double.parseDouble(s);
378     } catch (NumberFormatException e) {
379       return Double.NaN;
380     }
381   }

383   static String trimJsWhiteSpace(String s) {
384     int start = 0;
385     int end = s.length();
386     while (end > 0
387         && TokenUtil.isStrWhiteSpaceChar(s.charAt(end - 1)) == TernaryValue.TRUE) {
388       end--;
389     }
390     while (start < end
391         && TokenUtil.isStrWhiteSpaceChar(s.charAt(start)) == TernaryValue.TRUE) {
392       start++;
393     }
394     return s.substring(start, end);
395   }

447   /** Implementation for getFunctionNameNode and getClassNameNode. */
448   private static Node getNameNode(Node n) {
449     Node parent = n.getParent();
450     switch (parent.getType()) {
451       case Token.NAME:
452         // var name = function() ...
453         // var name2 = function name1() ...
454         return parent;
455 
456       case Token.ASSIGN:
457         // qualified.name = function() ...
458         // qualified.name2 = function name1() ...
459         return parent.getFirstChild();
460 
461       default:
462         // function name() ...
463         Node funNameNode = n.getFirstChild();
464         // Don't return the name node for anonymous functions/classes.
465         // TODO(tbreisacher): Currently we do two kinds of "empty" checks because
466         // anonymous classes have an EMPTY name node while anonymous functions
467         // have a STRING node with an empty string. Consider making these the same.
468         return (funNameNode.isEmpty() || funNameNode.getString().isEmpty())
469             ? null : funNameNode;
470     }
471   }

475    * <ul>
476    * <li>{@code &#123;'name': function() ...&#125;}</li>
477    * <li>{@code &#123;name: function() ...&#125;}</li>
478    * <li>{@code function name() ...}</li>
479    * <li>{@code var name = function() ...}</li>
480    * <li>{@code var obj = {name() {} ...}}</li>
481    * <li>{@code qualified.name = function() ...}</li>
482    * <li>{@code var name2 = function name1() ...}</li>
483    * <li>{@code qualified.name2 = function name1() ...}</li>
484    * </ul>
485    *
486    * @param n a node whose type is {@link Token#FUNCTION}
487    * @return the function's name, or {@code null} if it has no name

489   public static String getNearestFunctionName(Node n) {
490     if (!n.isFunction()) {
491       return null;
492     }
493 
494     String name = getFunctionName(n);
495     if (name != null) {
496       return name;
497     }
498 
499     // Check for the form { 'x' : function() { }} and {x() {}}
500     Node parent = n.getParent();
501     switch (parent.getType()) {
502       case Token.MEMBER_FUNCTION_DEF:
503       case Token.SETTER_DEF:
504       case Token.GETTER_DEF:
505       case Token.STRING_KEY:
506         // Return the name of the literal's key.
507         return parent.getString();
508       case Token.NUMBER:
509         return getStringValue(parent);
510     }
511 
512     return null;
513   }

517    * Returns true if this is an immutable value.
518    */
519   static boolean isImmutableValue(Node n) {
520     // TODO(johnlenz): rename this function.  It is currently being used
521     // in two disjoint cases:
522     // 1) We only care about the result of the expression
523     //    (in which case NOT here should return true)
524     // 2) We care that expression is a side-effect free and can't
525     //    be side-effected by other expressions.
526     // This should only be used to say the value is immuable and
527     // hasSideEffects and canBeSideEffected should be used for the other case.
528 
529     switch (n.getType()) {
530       case Token.STRING:
531       case Token.NUMBER:
532       case Token.NULL:
533       case Token.TRUE:
534       case Token.FALSE:
535         return true;
536       case Token.CAST:
537       case Token.NOT:
538       case Token.VOID:
539       case Token.NEG:
540         return isImmutableValue(n.getFirstChild());
541       case Token.NAME:
542         String name = n.getString();
543         // We assume here that programs don't change the value of the keyword
544         // undefined to something other than the value undefined.
545         return "undefined".equals(name) || "Infinity".equals(name) || "NaN".equals(name);
546     }
547 
548     return false;
549   }

614   /**
615    * Returns true if this is a literal value. We define a literal value
616    * as any node that evaluates to the same thing regardless of when or
617    * where it is evaluated. So /xyz/ and [3, 5] are literals, but
618    * the name a is not.
619    *
620    * <p>Function literals do not meet this definition, because they
621    * lexically capture variables. For example, if you have
622    * <code>
623    * function() { return a; }
624    * </code>
625    * If it is evaluated in a different scope, then it
626    * captures a different variable. Even if the function did not read
627    * any captured variables directly, it would still fail this definition,
628    * because it affects the lifecycle of variables in the enclosing scope.
629    *
630    * <p>However, a function literal with respect to a particular scope is
631    * a literal.
632    *
633    * @param includeFunctions If true, all function expressions will be
634    *     treated as literals.
635    */
636   static boolean isLiteralValue(Node n, boolean includeFunctions) {
637     switch (n.getType()) {
638       case Token.CAST:
639         return isLiteralValue(n.getFirstChild(), includeFunctions);
640 
641       case Token.ARRAYLIT:
642         for (Node child = n.getFirstChild(); child != null;
643              child = child.getNext()) {
644           if ((!child.isEmpty()) && !isLiteralValue(child, includeFunctions)) {
645             return false;
646           }
647         }
648         return true;
649 
650       case Token.REGEXP:
651         // Return true only if all children are const.
652         for (Node child = n.getFirstChild(); child != null;
653              child = child.getNext()) {
654           if (!isLiteralValue(child, includeFunctions)) {
655             return false;
656           }
657         }
658         return true;
659 
660       case Token.OBJECTLIT:
661         // Return true only if all values are const.
662         for (Node child = n.getFirstChild(); child != null;
663              child = child.getNext()) {
664           if (!isLiteralValue(child.getFirstChild(), includeFunctions)) {
665             return false;
666           }
667         }
668         return true;
669 
670       case Token.FUNCTION:
671         return includeFunctions && !NodeUtil.isFunctionDeclaration(n);
672 
673       default:
674         return isImmutableValue(n);
675     }
676   }

678   /**
679    * Determines whether the given value may be assigned to a define.
680    *
681    * @param val The value being assigned.
682    * @param defines The list of names of existing defines.
683    */
684   static boolean isValidDefineValue(Node val, Set<String> defines) {
685     switch (val.getType()) {
686       case Token.STRING:
687       case Token.NUMBER:
688       case Token.TRUE:
689       case Token.FALSE:
690         return true;
691 
692       // Binary operators are only valid if both children are valid.
693       case Token.AND:
694       case Token.OR:
695       case Token.ADD:
696       case Token.BITAND:
697       case Token.BITNOT:
698       case Token.BITOR:
699       case Token.BITXOR:
700       case Token.DIV:
701       case Token.EQ:
702       case Token.GE:
703       case Token.GT:
704       case Token.LE:
705       case Token.LSH:
706       case Token.LT:
707       case Token.MOD:
708       case Token.MUL:
709       case Token.NE:
710       case Token.RSH:
711       case Token.SHEQ:
712       case Token.SHNE:
713       case Token.SUB:
714       case Token.URSH:
715         return isValidDefineValue(val.getFirstChild(), defines)
716             && isValidDefineValue(val.getLastChild(), defines);
717 
718       // Unary operators are valid if the child is valid.
719       case Token.NOT:
720       case Token.NEG:
721       case Token.POS:
722         return isValidDefineValue(val.getFirstChild(), defines);
723 
724       // Names are valid if and only if they are defines themselves.
725       case Token.NAME:
726       case Token.GETPROP:
727         if (val.isQualifiedName()) {
728           return defines.contains(val.getQualifiedName());
729         }
730     }
731     return false;
732   }

922   /**
923    * Returns true if some node in n's subtree changes application state.
924    * If {@code checkForNewObjects} is true, we assume that newly created
925    * mutable objects (like object literals) change state. Otherwise, we assume
926    * that they have no side effects.
927    */
928   private static boolean checkForStateChangeHelper(
929       Node n, boolean checkForNewObjects, AbstractCompiler compiler) {
930     // Rather than id which ops may have side effects, id the ones
931     // that we know to be safe
932     switch (n.getType()) {
933       // other side-effect free statements and expressions
934       // Throws are by definition side effects, and yields are similar.
935       case Token.THROW:
936       case Token.YIELD:
937         return true;
938 
939       case Token.OBJECTLIT:
940         if (checkForNewObjects) {
941           return true;
942         }
943         for (Node key = n.getFirstChild(); key != null; key = key.getNext()) {
944           for (Node c = key.getFirstChild(); c != null; c = c.getNext()) {
945             if (checkForStateChangeHelper(c, checkForNewObjects, compiler)) {
946               return true;
947             }
948           }
949         }
950         return false;
951 
952       case Token.ARRAYLIT:
953       case Token.REGEXP:
954         if (checkForNewObjects) {
955           return true;
956         }
957         break;
958 
959       case Token.VAR:    // empty var statement (no declaration)
960       case Token.NAME:   // variable by itself
961         if (n.getFirstChild() != null) {
962           return true;
963         }
964         break;
965 
966       case Token.FUNCTION:
967         // Function expressions don't have side-effects, but function
968         // declarations change the namespace. Either way, we don't need to
969         // check the children, since they aren't executed at declaration time.
970         return checkForNewObjects || !isFunctionExpression(n);
971 
972       case Token.NEW:
973         if (checkForNewObjects) {
974           return true;
975         }
976 
977         if (!constructorCallHasSideEffects(n)) {
978           // loop below will see if the constructor parameters have
979           // side-effects
980           break;
981         }
982         return true;
983 
984       case Token.CALL:
985         // calls to functions that have no side effects have the no
986         // side effect property set.
987         if (!functionCallHasSideEffects(n, compiler)) {
988           // loop below will see if the function parameters have
989           // side-effects
990           break;
991         }
992         return true;
993 
994       case Token.TAGGED_TEMPLATELIT:
995         return functionCallHasSideEffects(n);
996 
997       case Token.CAST:
998       case Token.AND:
999       case Token.BLOCK:
1000       case Token.EXPR_RESULT:
1001       case Token.HOOK:
1002       case Token.IF:
1003       case Token.IN:
1004       case Token.PARAM_LIST:
1005       case Token.NUMBER:
1006       case Token.OR:
1007       case Token.THIS:
1008       case Token.TRUE:
1009       case Token.FALSE:
1010       case Token.NULL:
1011       case Token.STRING:
1012       case Token.STRING_KEY:
1013       case Token.SWITCH:
1014       case Token.TEMPLATELIT_SUB:
1015       case Token.TRY:
1016       case Token.EMPTY:
1017       case Token.TEMPLATELIT:
1018         break;
1019 
1020       default:
1021         if (isSimpleOperator(n)) {
1022           break;
1023         }
1024 
1025         if (isAssignmentOp(n)) {
1026           Node assignTarget = n.getFirstChild();
1027           if (assignTarget.isName()) {
1028             return true;
1029           }
1030 
1031           // Assignments will have side effects if
1032           // a) The RHS has side effects, or
1033           // b) The LHS has side effects, or
1034           // c) A name on the LHS will exist beyond the life of this statement.
1035           if (checkForStateChangeHelper(
1036                   n.getFirstChild(), checkForNewObjects, compiler) ||
1037               checkForStateChangeHelper(
1038                   n.getLastChild(), checkForNewObjects, compiler)) {
1039             return true;
1040           }
1041 
1042           if (isGet(assignTarget)) {
1043             // If the object being assigned to is a local object, don't
1044             // consider this a side-effect as it can't be referenced
1045             // elsewhere.  Don't do this recursively as the property might
1046             // be an alias of another object, unlike a literal below.
1047             Node current = assignTarget.getFirstChild();
1048             if (evaluatesToLocalValue(current)) {
1049               return false;
1050             }
1051 
1052             // A literal value as defined by "isLiteralValue" is guaranteed
1053             // not to be an alias, or any components which are aliases of
1054             // other objects.
1055             // If the root object is a literal don't consider this a
1056             // side-effect.
1057             while (isGet(current)) {
1058               current = current.getFirstChild();
1059             }
1060 
1061             return !isLiteralValue(current, true);
1062           } else {
1063             // TODO(johnlenz): remove this code and make this an exception. This
1064             // is here only for legacy reasons, the AST is not valid but
1065             // preserve existing behavior.
1066             return !isLiteralValue(assignTarget, true);
1067           }
1068         }
1069 
1070         return true;
1071     }
1072 
1073     for (Node c = n.getFirstChild(); c != null; c = c.getNext()) {
1074       if (checkForStateChangeHelper(c, checkForNewObjects, compiler)) {
1075         return true;
1076       }
1077     }
1078 
1079     return false;
1080   }

2247   /** Safely remove children while maintaining a valid node structure. */
2248   static void removeChild(Node parent, Node node) {
2249     if (isTryFinallyNode(parent, node)) {
2250       if (NodeUtil.hasCatchHandler(getCatchBlock(parent))) {
2251         // A finally can only be removed if there is a catch.
2252         parent.removeChild(node);
2253       } else {
2254         // Otherwise, only its children can be removed.
2255         node.detachChildren();
2256       }
2257     } else if (node.isCatch()) {
2258       // The CATCH can can only be removed if there is a finally clause.
2259       Node tryNode = node.getParent().getParent();
2260       Preconditions.checkState(NodeUtil.hasFinally(tryNode));
2261       node.detachFromParent();
2262     } else if (isTryCatchNodeContainer(node)) {
2263       // The container node itself can't be removed, but the contained CATCH
2264       // can if there is a 'finally' clause
2265       Node tryNode = node.getParent();
2266       Preconditions.checkState(NodeUtil.hasFinally(tryNode));
2267       node.detachChildren();
2268     } else if (node.isBlock()) {
2269       // Simply empty the block.  This maintains source location and
2270       // "synthetic"-ness.
2271       node.detachChildren();
2272     } else if (isStatementBlock(parent)
2273         || isSwitchCase(node)) {
2274       // A statement in a block can simply be removed.
2275       parent.removeChild(node);
2276     } else if (parent.isVar()) {
2277       if (parent.hasMoreThanOneChild()) {
2278         parent.removeChild(node);
2279       } else {
2280         // Remove the node from the parent, so it can be reused.
2281         parent.removeChild(node);
2282         // This would leave an empty VAR, remove the VAR itself.
2283         removeChild(parent.getParent(), parent);
2284       }
2285     } else if (parent.isLabel()
2286         && node == parent.getLastChild()) {
2287       // Remove the node from the parent, so it can be reused.
2288       parent.removeChild(node);
2289       // A LABEL without children can not be referred to, remove it.
2290       removeChild(parent.getParent(), parent);
2291     } else if (parent.isFor()
2292         && parent.getChildCount() == 4) {
2293       // Only Token.FOR can have an Token.EMPTY other control structure
2294       // need something for the condition. Others need to be replaced
2295       // or the structure removed.
2296       parent.replaceChild(node, IR.empty());
2297     } else {
2298       throw new IllegalStateException("Invalid attempt to remove node: " + node + " of " + parent);
2299     }
2300   }

2578   public static boolean isLhsByDestructuring(Node n) {
2579     Node parent = n.getParent();
2580 
2581     if (parent.isDestructuringPattern()
2582         || (parent.isStringKey() && parent.getParent().isObjectPattern())) {
2583       if (n.isStringKey() && n.hasChildren()) {
2584         return false;
2585       }
2586       Node childNode = n;
2587       boolean isLastChildOfHighestPattern = false;
2588 
2589       // The AST structure for destructuring patterns are a little bit complicated.
2590       // The last child of a destructuring pattern is NOT the left hand side UNLESS
2591       // the destructuring pattern is:
2592       //   1) the variable for for-of loop;
2593       //   2) within the first child of ASSIGN
2594       //   3) within the first child of DEFAULT_VALUE (default parameter)
2595       // Also, in nested destructuring only the last child of the highest pattern
2596       // and its descendants are NOT lhs.
2597       for (Node currAncestor : n.getAncestors()) {
2598         if (currAncestor.isForOf() || currAncestor.isFor()
2599             || currAncestor.isAssign()
2600             || currAncestor.isDefaultValue()) {
2601           return currAncestor.getFirstChild() == childNode;
2602         }
2603         if (currAncestor.isDestructuringPattern()) {
2604           isLastChildOfHighestPattern = currAncestor.getLastChild() == childNode;
2605         }

2607       }
2608       return !isLastChildOfHighestPattern;

3642   /**
3643    * @param locals A predicate to apply to unknown local values.
3644    * @return Whether the node is known to be a value that is not a reference
3645    *     outside the expression scope.
3646    */
3647   static boolean evaluatesToLocalValue(Node value, Predicate<Node> locals) {
3648     switch (value.getType()) {
3649       case Token.CAST:
3650         return evaluatesToLocalValue(value.getFirstChild(), locals);
3651       case Token.ASSIGN:
3652         // A result that is aliased by a non-local name, is the effectively the
3653         // same as returning a non-local name, but this doesn't matter if the
3654         // value is immutable.
3655         return NodeUtil.isImmutableValue(value.getLastChild())
3656             || (locals.apply(value)
3657                 && evaluatesToLocalValue(value.getLastChild(), locals));
3658       case Token.COMMA:
3659         return evaluatesToLocalValue(value.getLastChild(), locals);
3660       case Token.AND:
3661       case Token.OR:
3662         return evaluatesToLocalValue(value.getFirstChild(), locals)
3663            && evaluatesToLocalValue(value.getLastChild(), locals);
3664       case Token.HOOK:
3665         return evaluatesToLocalValue(value.getFirstChild().getNext(), locals)
3666            && evaluatesToLocalValue(value.getLastChild(), locals);
3667       case Token.INC:
3668       case Token.DEC:
3669         if (value.getBooleanProp(Node.INCRDECR_PROP)) {
3670           return evaluatesToLocalValue(value.getFirstChild(), locals);
3671         } else {
3672           return true;
3673         }
3674       case Token.THIS:
3675         return locals.apply(value);
3676       case Token.NAME:
3677         return isImmutableValue(value) || locals.apply(value);
3678       case Token.GETELEM:
3679       case Token.GETPROP:
3680         // There is no information about the locality of object properties.
3681         return locals.apply(value);
3682       case Token.CALL:
3683         return callHasLocalResult(value)
3684             || isToStringMethodCall(value)
3685             || locals.apply(value);
3686       case Token.NEW:
3687         return newHasLocalResult(value)
3688                || locals.apply(value);
3689       case Token.FUNCTION:
3690       case Token.REGEXP:
3691       case Token.ARRAYLIT:
3692       case Token.OBJECTLIT:
3693         // Literals objects with non-literal children are allowed.
3694         return true;
3695       case Token.DELPROP:
3696       case Token.IN:
3697         // TODO(johnlenz): should IN operator be included in #isSimpleOperator?
3698         return true;
3699       default:
3700         // Other op force a local value:
3701         //  x = '' + g (x is now an local string)
3702         //  x -= g (x is now an local number)
3703         if (isAssignmentOp(value)
3704             || isSimpleOperator(value)
3705             || isImmutableValue(value)) {
3706           return true;
3707         }
3708 
3709         throw new IllegalStateException(
3710             "Unexpected expression node" + value +
3711             "\n parent:" + value.getParent());
3712     }

3877    * @return false iff the result of the expression is not consumed.
3878    */
3879   static boolean isExpressionResultUsed(Node expr) {
3880     // TODO(johnlenz): consider sharing some code with trySimpleUnusedResult.
3881     Node parent = expr.getParent();
3882     switch (parent.getType()) {
3883       case Token.BLOCK:
3884       case Token.EXPR_RESULT:
3885         return false;
3886       case Token.CAST:
3887         return isExpressionResultUsed(parent);
3888       case Token.HOOK:
3889       case Token.AND:
3890       case Token.OR:
3891         return (expr == parent.getFirstChild()) || isExpressionResultUsed(parent);
3892       case Token.COMMA:

3894         if (grandparent.isCall() &&
3895             parent == grandparent.getFirstChild()) {
3896           // Semantically, a direct call to eval is different from an indirect
3897           // call to an eval. See ECMA-262 S15.1.2.1. So it's OK for the first

3901               parent.getChildCount() == 2 &&
3902               expr.getNext().isName() &&
3903               "eval".equals(expr.getNext().getString())) {
3904             return true;
3905           }
3906         }
3907 
3908         return (expr == parent.getFirstChild())
3909             ? false : isExpressionResultUsed(parent);
3910       case Token.FOR:
3911         if (!NodeUtil.isForIn(parent)) {
3912           // Only an expression whose result is in the condition part of the

3922    * @param n The expression to check.
3923    * @return Whether the expression is unconditionally executed only once in the
3924    *     containing execution scope.
3925    */
3926   static boolean isExecutedExactlyOnce(Node n) {
3927     inspect: do {
3928       Node parent = n.getParent();
3929       switch (parent.getType()) {
3930         case Token.IF:
3931         case Token.HOOK:
3932         case Token.AND:
3933         case Token.OR:
3934           if (parent.getFirstChild() != n) {
3935             return false;
3936           }
3937           // other ancestors may be conditional
3938           continue inspect;
3939         case Token.FOR:
3940           if (NodeUtil.isForIn(parent)) {
3941             if (parent.getChildAtIndex(1) != n) {
3942               return false;
3943             }
3944           } else {
3945             if (parent.getFirstChild() != n) {
3946               return false;
3947             }
3948           }
3949           // other ancestors may be conditional
3950           continue inspect;
3951         case Token.WHILE:
3952         case Token.DO:
3953           return false;
3954         case Token.TRY:
3955           // Consider all code under a try/catch to be conditionally executed.
3956           if (!hasFinally(parent) || parent.getLastChild() != n) {
3957             return false;
3958           }
3959           continue inspect;
3960         case Token.CASE:
3961         case Token.DEFAULT_CASE:
3962           return false;
3963         case Token.SCRIPT:
3964         case Token.FUNCTION:
3965           // Done, we've reached the scope root.
3966           break inspect;
3967       }
3968     } while ((n = n.getParent()) != null);
3969     return true;

4029   /** Checks that the scope roots marked as changed have indeed changed */
4030   public static void verifyScopeChanges(Map<Node, Node> map,
4031       Node main, boolean verifyUnchangedNodes,
4032       AbstractCompiler compiler) {
4033     // compiler is passed only to call compiler.toSource during debugging to see
4034     // mismatches in scopes
4035 
4036     // If verifyUnchangedNodes is false, we are comparing the initial AST to the
4037     // final AST. Don't check unmarked nodes b/c they may have been changed by
4038     // non-loopable passes.
4039     // If verifyUnchangedNodes is true, we are comparing the ASTs before & after
4040     // a pass. Check all scope roots.
4041     final Map<Node, Node> mtoc = map;
4042     final boolean checkUnchanged = verifyUnchangedNodes;
4043     Node clone = mtoc.get(main);
4044     if (main.getChangeTime() > clone.getChangeTime()) {
4045       Preconditions.checkState(!isEquivalentToExcludingFunctions(main, clone));
4046     } else if (checkUnchanged) {
4047       Preconditions.checkState(isEquivalentToExcludingFunctions(main, clone));
4048     }
4049     visitPreOrder(main,
4050         new Visitor() {
4051           @Override
4052           public void visit(Node n) {
4053             if (n.isFunction() && mtoc.containsKey(n)) {
4054               Node clone = mtoc.get(n);
4055               if (n.getChangeTime() > clone.getChangeTime()) {
4056                 Preconditions.checkState(
4057                     !isEquivalentToExcludingFunctions(n, clone));
4058               } else if (checkUnchanged) {
4059                 Preconditions.checkState(
4060                     isEquivalentToExcludingFunctions(n, clone));
4061               }

4090    * differences any descendant functions differences.
4091    */

4093       Node thisNode, Node thatNode) {

4095       return thisNode == null && thatNode == null;
4096     }

4098       return false;

4100     if (thisNode.getChildCount() != thatNode.getChildCount()) {
4101       return false;
4102     }
4103     Node thisChild = thisNode.getFirstChild();
4104     Node thatChild = thatNode.getFirstChild();
4105     while (thisChild != null && thatChild != null) {
4106       if (thisChild.isFunction()) {

4108         return thatChild.isFunction();

4111         return false;

4113       thisChild = thisChild.getNext();
4114       thatChild = thatChild.getNext();
4115     }
