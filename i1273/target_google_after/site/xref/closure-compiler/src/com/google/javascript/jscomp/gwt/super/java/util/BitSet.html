22  /**
23   * This implementation uses bit groups of size 32 to keep track of when bits are
24   * set to true or false.  This implementation also uses the sparse nature of
25   * JavaScript arrays to speed up cases when very few bits are set in a large bit
26   * set.
27   *
28   * Since there is no speed advantage to pre-allocating array sizes in JavaScript
29   * the underlying array's length is shrunk to Sun's "logical length" whenever
30   * length() is called.  This length is the index of the highest true bit, plus
31   * one, or zero if there are aren't any.  This may cause the size() method to
32   * return a different size than in a true Java VM.
33   *
34   * TODO(moz): Add this to GWT. Pending changelist at:
35   * https://gwt-review.googlesource.com/#/c/5771/
36   */
37  public class BitSet {
38    // To speed up certain operations this class also uses the index properties
39    // of arrays as described in section 15.4 of "Standard ECMA-262" (June 1997),
40    // which can currently be found here:
41    // http://www.mozilla.org/js/language/E262.pdf
42    //
43    // 15.4 Array Objects
44    // Array objects give special treatment to a certain class of property names.
45    // A property name P (in the form of a string value) is an array index if and
46    // only if ToString(ToUint32(P)) is equal to P and ToUint32(P) is not equal
47    // to (2^32)-1.
48  
49    // checks the index range
50    private static void checkIndex(int bitIndex) {
51      // we only need to test for negatives, as there is no bit index too high.
52      if (bitIndex < 0) {
53        throw new IndexOutOfBoundsException("bitIndex < 0: " + bitIndex);
54      }
55    }
56  
57    // checks to ensure indexes are not negative and not in reverse order
58    private static void checkRange(int fromIndex, int toIndex) {
59      if (fromIndex < 0) {
60        throw new IndexOutOfBoundsException("fromIndex < 0: " + fromIndex);
61      }
62      if (toIndex < 0) {
63        throw new IndexOutOfBoundsException("toIndex < 0: " + toIndex);
64      }
65      if (fromIndex > toIndex) {
66        throw new IndexOutOfBoundsException("fromIndex: " + fromIndex
67            + " > toIndex: " + toIndex);
68      }
69    }

71    // converts from a bit index to a word index
72    private static int wordIndex(int bitIndex) {
73      // 32 bits per index
74      return bitIndex >>> 5;
75    }
76  
77    // converts from a word index to a bit index
78    private static int bitIndex(int wordIndex) {
79      // 1 word index for every 32 bit indexes
80      return wordIndex << 5;
81    }
82  
83    // gives the word offset for a bit index
84    private static int bitOffset(int bitIndex) {
85      return bitIndex & 0x1f;
86    }

92    // clears one bit
93    private static void clear(JsArrayInteger array, int bitIndex) {
94      int index = wordIndex(bitIndex);
95      int word = getWord(array, index);
96      if (word != 0) {
97        // mask the correct bit out
98        setWord(array, index, word & ~(1 << (bitOffset(bitIndex))));
99      }
100   }

107   // flips one bit
108   private static void flip(JsArrayInteger array, int bitIndex) {
109     // calculate index and offset
110     int index = wordIndex(bitIndex);
111     int offset = bitOffset(bitIndex);
112 
113     // figure out if the bit is on or off
114     int word = getWord(array, index);
115     if (((word >>> offset) & 1) == 1) {
116       // if on, turn it off
117       setWord(array, index, word & ~(1 << offset));
118     } else {
119       // if off, turn it on
120       array.set(index, word | (1 << offset));
121     }
122   }

124   // gets one bit
125   private static boolean get(JsArrayInteger array, int bitIndex) {
126     // retrieve the bits for the given index
127     int word = getWord(array, wordIndex(bitIndex));
128 
129     // shift and mask the bit out
130     return ((word >>> (bitOffset(bitIndex))) & 1) == 1;
131   }

133   // sets one bit to true
134   private static void set(JsArrayInteger array, int bitIndex) {
135     int index = wordIndex(bitIndex);
136     array.set(index, getWord(array, index) | (1 << (bitOffset(bitIndex))));
137   }

139   // sets all bits to true within the given range
140   private static void set(JsArrayInteger array, int fromIndex, int toIndex) {
141     int first = wordIndex(fromIndex);
142     int last = wordIndex(toIndex);
143     int startBit = bitOffset(fromIndex);
144     int endBit = bitOffset(toIndex);
145 
146     if (first == last) {
147       // set the bits in between first and last
148       maskInWord(array, first, startBit, endBit);
149 
150     } else {
151       // set the bits from fromIndex to the next 32 bit boundary
152       if (startBit != 0) {
153         maskInWord(array, first++, startBit, 32);
154       }
155 
156       // set the bits from the last 32 bit boundary to the toIndex
157       if (endBit != 0) {
158         maskInWord(array, last, 0, endBit);
159       }
160 
161       //
162       // set everything in between
163       //
164       for (int i = first; i < last; i++) {
165         array.set(i, 0xffffffff);
166       }
167     }
168   }

176   // trims the array to the minimum size it can without losing data
177   // returns index of the last element in the array, or -1 if empty
178   private static native int trimToSize(JsArrayInteger array) /*-{
179     var length = array.length;
180     if (length === 0) {
181       return -1;
182     }
183 
184     // check if the last bit is false
185     var last = length - 1;
186     if (array[last] !== undefined) {
187       return last;
188     }
189 
190     // interleave property checks and linear index checks from the end
191     var biggestSeen = -1;
192     for (var property in array) {
193 
194       // test the index first
195       if (--last === -1) {
196         return -1;
197       }

200       }

205         if (number > biggestSeen) {

207         }
208       }

224   // flips all bits stored at a certain index
225   private static void flipWord(JsArrayInteger array, int index) {
226     int word = getWord(array, index);
227     if (word == 0) {
228       array.set(index, 0xffffffff);
229     } else {
230       word = ~word;
231       setWord(array, index, word);
232     }
233   }

235   // flips all bits stored at a certain index within the given range
236   private static void flipMaskedWord(JsArrayInteger array, int index,
237       int from, int to) {
238     if (from == to) {
239       return;
240     }
241     // get the bits
242     int word = getWord(array, index);
243     // adjust "to" so it will shift out those bits
244     to = 32 - to;
245     // create a mask and XOR it in
246     word ^= (((0xffffffff >>> from) << from) << to) >>> to;
247     setWord(array, index, word);
248   }

250   // returns all bits stored at a certain index
251   private static native int getWord(JsArrayInteger array, int index) /*-{
252     // OR converts an undefined to 0
253     return array[index] | 0;
254   }-*/;

256   // sets all bits to true at a certain index within the given bit range
257   private static void maskInWord(JsArrayInteger array, int index,
258       int from, int to) {
259     // shifting by 32 is the same as shifting by 0, this check prevents that
260     // from happening in addition to the obvious avoidance of extra work
261     if (from != to) {
262       // adjust "to" so it will shift out those bits
263       to = 32 - to;
264       // create a mask and OR it in
265       int value = getWord(array, index);
266       value |= ((0xffffffff >>> from) << (from + to)) >>> to;
267       array.set(index, value);
268     }
269   }
270 
271   // sets all bits to false at a certain index within the given bit range
272   private static void maskOutWord(JsArrayInteger array, int index,
273       int from, int to) {
274     int word = getWord(array, index);
275     // something only happens if word has bits set
276     if (word != 0) {
277       // create a mask
278       int mask;
279       if (from != 0) {
280         mask = 0xffffffff >>> (32 - from);
281       } else {
282         mask = 0;
283       }
284       // shifting by 32 is the same as shifting by 0
285       if (to != 32) {
286         mask |= 0xffffffff << to;
287       }
288 
289       // mask it out
290       word &= mask;
291       setWord(array, index, word);
292     }
293   }

295   private static native int nextSetWord(JsArrayInteger array, int index) /*-{
296     // interleave property checks and linear "index" checks
297     var length = array.length;
298     var localMinimum = @java.lang.Integer::MAX_VALUE;
299     for (var property in array) {
300 
301       // test the index first
302       if (array[index] !== undefined) {
303         return index;
304       }
305       if (++index >= length) {
306         return -1;
307       }
308 
309       // now check the property
310       var number = property >>> 0;
311       if (String(number) == property && number !== 0xffffffff) {
312         if (number >= index && number < localMinimum) {
313           localMinimum = number;
314         }
315       }
316     }
317 
318     // if local minimum is what we started at, we found nothing
319     if (localMinimum === @java.lang.Integer::MAX_VALUE) {
320       return -1;
321     }
322 
323     return localMinimum;
324   }-*/;

327   private static void setWord(JsArrayInteger array, int index, int value) {
328     // keep 0s out of the array
329     if (value == 0) {
330       deleteWord(array, index);
331     } else {
332       array.set(index, value);
333     }
334   }

349   public BitSet(int nbits) {
350     this();
351 
352     // throw an exception to be consistent
353     // but (do we want to be consistent?)
354     if (nbits < 0) {
355       throw new NegativeArraySizeException("nbits < 0: " + nbits);
356     }
357 
358     // even though the array's length is loosely kept to that of Sun's "logical
359     // length," this might help in some cases where code uses size() to fill in
360     // bits after constructing a BitSet, or after having one passed in as a
361     // parameter.
362     setLengthWords(array, wordIndex(nbits + 31));
363   }

369   public void and(BitSet set) {
370     // a & a is just a
371     if (this == set) {
372       return;
373     }
374 
375     // trim the second set to avoid extra work
376     trimToSize(set.array);
377 
378     // check if the length is longer than otherLength
379     int otherLength = set.array.length();
380     if (array.length() > otherLength) {
381       // shrink the array, effectively ANDing those bits to false
382       setLengthWords(array, otherLength);
383     }
384 
385     // truth table
386     //
387     // case | a     | b     | a & b | change?
388     // 1    | false | false | false | a is already false
389     // 2    | false | true  | false | a is already false
390     // 3    | true  | false | false | set a to false
391     // 4    | true  | true  | true  | a is already true
392     //
393     // we only need to change something in case 3, so iterate over set a
394     int index = 0;
395     while ((index = nextSetWord(array, index)) != -1) {
396       setWord(array, index,
397           array.get(index) & getWord(set.array, index));
398       index++;
399     }
400   }

402   public void andNot(BitSet set) {
403     // a & !a is false
404     if (this == set) {
405       // all falses result in an empty BitSet
406       clear();
407       return;
408     }
409 
410     // trim the second set to avoid extra work
411     trimToSize(array);
412     int length = array.length();
413 
414     // truth table
415     //
416     // case | a     | b     | !b    | a & !b | change?
417     // 1    | false | false | true  | false  | a is already false
418     // 2    | false | true  | false | false  | a is already false
419     // 3    | true  | false | true  | true   | a is already true
420     // 4    | true  | true  | false | false  | set a to false
421     //
422     // we only need to change something in case 4
423     // whenever b is true, a should be false, so iterate over set b
424     int index = 0;
425     while ((index = nextSetWord(set.array, index)) != -1) {
426       setWord(array, index,
427           getWord(array, index) & ~set.array.get(index));
428       if (++index >= length) {
429         // nothing further will affect anything
430         break;
431       }
432     }
433   }

457   public void clear(int fromIndex, int toIndex) {
458     checkRange(fromIndex, toIndex);
459 
460     int length = length();
461     if (fromIndex >= length) {
462       // nothing to do
463       return;
464     }
465 
466     // check to see if toIndex is greater than our array length
467     if (toIndex >= length) {
468       // truncate the array by setting it's length
469       int newLength = wordIndex(fromIndex + 31);
470       setLengthWords(array, newLength);
471 
472       // remove the extra bits off the end
473       if ((bitIndex(newLength)) - fromIndex != 0) {
474         maskOutWord(array, newLength - 1, bitOffset(fromIndex), 32);
475       }
476 
477     } else {
478       int first = wordIndex(fromIndex);
479       int last = wordIndex(toIndex);
480       int startBit = bitOffset(fromIndex);
481       int endBit = bitOffset(toIndex);
482 
483       if (first == last) {
484         // clear the bits in between first and last
485         maskOutWord(array, first, startBit, endBit);
486 
487       } else {
488         // clear the bits from fromIndex to the next 32 bit boundary
489         if (startBit != 0) {
490           maskOutWord(array, first++, startBit, 32);
491         }
492 
493         // clear the bits from the last 32 bit boundary to the toIndex
494         if (endBit != 0) {
495           maskOutWord(array, last, 0, endBit);
496         }
497 
498         //
499         // delete everything in between
500         //
501         for (int i = first; i < last; i++) {
502           deleteWord(array, i);
503         }
504       }
505     }
506   }

512   @Override
513   public boolean equals(Object obj) {
514     if (this != obj) {
515 
516       if (!(obj instanceof BitSet)) {
517         return false;
518       }
519 
520       BitSet other = (BitSet) obj;
521 
522       int last = trimToSize(array);
523       if (last != trimToSize(other.array)) {
524         return false;
525       }
526 
527       int index = 0;
528       while ((index = nextSetWord(array, index)) != -1) {
529         if (getWord(array, index) != getWord(other.array, index)) {
530           return false;
531         }
532         index++;
533       }
534     }
535 
536     return true;
537   }

544   public void flip(int fromIndex, int toIndex) {
545     checkRange(fromIndex, toIndex);
546 
547     int length = length();
548 
549     // if we are flipping bits beyond our length, we are setting them to true
550     if (fromIndex >= length) {
551       set(array, fromIndex, toIndex);
552       return;
553     }
554 
555     // check to see if toIndex is greater than our array length
556     if (toIndex >= length) {
557       set(array, length, toIndex);
558       toIndex = length;
559     }
560 
561     int first = wordIndex(fromIndex);
562     int last = wordIndex(toIndex);
563     int startBit = bitOffset(fromIndex);
564     int end = bitOffset(toIndex);
565 
566     if (first == last) {
567       // flip the bits in between first and last
568       flipMaskedWord(array, first, startBit, end);
569 
570     } else {
571       // clear the bits from fromIndex to the next 32 bit boundary
572       if (startBit != 0) {
573         flipMaskedWord(array, first++, startBit, 32);
574       }
575 
576       // clear the bits from the last 32 bit boundary to the toIndex
577       if (end != 0) {
578         flipMaskedWord(array, last, 0, end);
579       }
580 
581       // flip everything in between
582       for (int i = first; i < last; i++) {
583         flipWord(array, i);
584       }
585     }
586   }

593   public BitSet get(int fromIndex, int toIndex) {
594     checkRange(fromIndex, toIndex);
595 
596     // no need to go past our length
597     toIndex = Math.min(toIndex, length());
598 
599     // this is the bit shift offset for each group of bits
600     int rightShift = bitOffset(fromIndex);
601 
602     if (rightShift == 0) {
603       int subFrom = wordIndex(fromIndex);
604       int subTo = wordIndex(toIndex + 31);
605       JsArrayInteger subSet = slice(array, subFrom, subTo);
606       int leftOvers = bitOffset(toIndex);
607       if (leftOvers != 0) {
608         maskOutWord(subSet, subTo - subFrom - 1, leftOvers, 32);
609       }
610       return new BitSet(subSet);
611     }
612 
613     BitSet subSet = new BitSet();
614 
615     int first = wordIndex(fromIndex);
616     int last = wordIndex(toIndex);
617 
618     if (first == last) {
619       // number of bits to cut from the end
620       int end = 32 - (bitOffset(toIndex));
621       // raw bits
622       int word = getWord(array, first);
623       // shift out those bits
624       word = ((word << end) >>> end) >>> rightShift;
625       // set it
626       if (word != 0) {
627         subSet.set(0, word);
628       }
629 
630     } else {
631       // this will hold the newly packed bits
632       int current = 0;
633 
634       // this is the raw index into the sub set
635       int subIndex = 0;
636 
637       // fence post, carry over initial bits
638       int word = getWord(array, first++);
639       current = word >>> rightShift;
640 
641       // a left shift will be used to shift our bits to the top of "current"
642       int leftShift = 32 - rightShift;
643 
644       // loop through everything in the middle
645       for (int i = first; i <= last; i++) {
646         word = getWord(array, i);
647 
648         // shift out the bits from the top, OR them into current bits
649         current |= word << leftShift;
650 
651         // flush it out
652         if (current != 0) {
653           subSet.array.set(subIndex, current);
654         }
655 
656         // keep track of our index
657         subIndex++;
658 
659         // carry over the unused bits
660         current = word >>> rightShift;
661       }
662 
663       // fence post, flush out the extra bits, but don't go past the "end"
664       int end = 32 - (bitOffset(toIndex));
665       current = (current << (rightShift + end)) >>> (rightShift + end);
666       if (current != 0) {
667         subSet.array.set(subIndex, current);
668       }
669     }
670 
671     return subSet;
672   }

674   /**
675    * This hash is different than the one described in Sun's documentation.  The
676    * described hash uses 64 bit integers and that's not practical in JavaScript.
677    */
678   @Override
679   public int hashCode() {
680     // FNV constants
681     final int fnvOffset = 0x811c9dc5;
682     final int fnvPrime = 0x1000193;
683 
684     // initialize
685     final int last = trimToSize(array);
686     int hash = fnvOffset ^ last;
687 
688     // loop over the data
689     for (int i = 0; i <= last; i++) {
690       int value = getWord(array, i);
691       // hash one byte at a time using FNV1
692       hash = (hash * fnvPrime) ^ (value & 0xff);
693       hash = (hash * fnvPrime) ^ ((value >>> 8) & 0xff);
694       hash = (hash * fnvPrime) ^ ((value >>> 16) & 0xff);
695       hash = (hash * fnvPrime) ^ (value >>> 24);
696     }
697 
698     return hash;
699   }

701   public boolean intersects(BitSet set) {
702     int last = trimToSize(array);
703 
704     if (this == set) {
705       // if it has any bits then it intersects itself
706       return last != -1;
707     }
708 
709     int length = set.array.length();
710     int index = 0;
711     while ((index = nextSetWord(array, index)) != -1) {
712       if ((array.get(index) & getWord(set.array, index)) != 0) {
713         return true;
714       }
715       if (++index >= length) {
716         // nothing further can intersect
717         break;
718       }
719     }
720 
721     return false;
722   }

728   public int length() {
729     int last = trimToSize(array);
730     if (last == -1) {
731       return 0;
732     }
733 
734     // compute the position of the leftmost bit's index
735     int offsets[] = { 16, 8, 4, 2, 1 };
736     int bitMasks[] = { 0xffff0000, 0xff00, 0xf0, 0xc, 0x2 };
737     int position = bitIndex(last) + 1;
738     int word = getWord(array, last);
739     for (int i = 0; i < offsets.length; i++) {
740       if ((word & bitMasks[i]) != 0) {
741         word >>>= offsets[i];
742         position += offsets[i];
743       }
744     }
745     return position;
746   }

748   public int nextClearBit(int fromIndex) {
749     checkIndex(fromIndex);
750     int index = wordIndex(fromIndex);
751 
752     // special case for first index
753     int fromBit = fromIndex - (bitIndex(index));
754     int word = getWord(array, index);
755     for (int i = fromBit; i < 32; i++) {
756       if ((word & (1 << i)) == 0) {
757         return (bitIndex(index)) + i;
758       }
759     }
760 
761     // loop through the rest
762     do {
763       index++;
764       word = getWord(array, index);
765     } while (word == 0xffffffff);
766     return bitIndex(index) + Integer.numberOfTrailingZeros(~word);
767   }

769   public int nextSetBit(int fromIndex) {
770     checkIndex(fromIndex);
771 
772     int index = wordIndex(fromIndex);
773 
774     // check the current word
775     int word = getWord(array, index);
776     if (word != 0) {
777       for (int i = bitOffset(fromIndex); i < 32; i++) {
778         if ((word & (1 << i)) != 0) {
779           return (bitIndex(index)) + i;
780         }
781       }
782     }
783     index++;
784 
785     // find the next set word
786     trimToSize(array);
787     index = nextSetWord(array, index);
788     if (index == -1) {
789       return -1;
790     }
791 
792     // return the next set bit
793     return (bitIndex(index)) + Integer.numberOfTrailingZeros(array.get(index));
794   }

796   public void or(BitSet set) {
797     // a | a is just a
798     if (this == set) {
799       return;
800     }
801 
802     // truth table
803     //
804     // case | a     | b     | a | b | change?
805     // 1    | false | false | false | a is already false
806     // 2    | false | true  | true  | set a to true
807     // 3    | true  | false | true  | a is already true
808     // 4    | true  | true  | true  | a is already true
809     //
810     // we only need to change something in case 2
811     // case 2 only happens when b is true, so iterate over set b
812     int index = 0;
813     while ((index = nextSetWord(set.array, index)) != -1) {
814       setWord(array, index,
815           getWord(array, index) | set.array.get(index));
816       index++;
817     }
818   }

851   @Override
852   public String toString() {
853     // possibly faster if done in JavaScript and all numerical properties are
854     // put into an array and sorted
855 
856     int length = length();
857     if (length == 0) {
858       // a "length" of 0 means there are no bits set to true
859       return "{}";
860     }
861 
862     StringBuilder sb = new StringBuilder("{");
863 
864     // at this point, there is at least one true bit, nextSetBit can not fail
865     int next = nextSetBit(0);
866     sb.append(next);
867 
868     // loop until nextSetBit returns -1
869     while ((next = nextSetBit(next + 1)) != -1) {
870       sb.append(", ");
871       sb.append(next);
872     }
873 
874     sb.append("}");
875     return sb.toString();
876   }

878   public void xor(BitSet set) {
879     // a ^ a is false
880     if (this == set) {
881       // this results in an empty BitSet
882       clear();
883       return;
884     }
885 
886     // truth table
887     //
888     // case | a     | b     | a ^ b | change?
889     // 1    | false | false | false | a is already false
890     // 2    | false | true  | true  | set a to true
891     // 3    | true  | false | true  | a is already true
892     // 4    | true  | true  | false | set a to false
893     //
894     // we need to change something in cases 2 and 4
895     // cases 2 and 4 only happen when b is true, so iterate over set b
896     int index = 0;
897     while ((index = nextSetWord(set.array, index)) != -1) {
898       setWord(array, index,
899           getWord(array, index) ^ set.array.get(index));
900       index++;
901     }
902   }
