43  /**
44   * A PerformanceTracker collects statistics about the runtime of each pass, and
45   * how much a pass impacts the size of the compiled output, before and after
46   * gzip.
47   *
48   * TODO(moz): Make this GWT compatible.
49   *
50   * @author dimvar@google.com (Dimitris Vardoulakis)
51   */
52  @GwtIncompatible("java.io.ByteArrayOutputStream")
53  public final class PerformanceTracker {
54  
55    private static final int DEFAULT_WHEN_SIZE_UNTRACKED = -1;
56  
57    private final Node jsRoot;
58    private final boolean trackSize;
59    private final boolean trackGzSize;
60  
61    // Keeps track of AST changes and computes code size estimation
62    // if there is any.
63    private final RecentChange codeChange = new RecentChange();
64  
65    private int initCodeSize = DEFAULT_WHEN_SIZE_UNTRACKED;
66    private int initGzCodeSize = DEFAULT_WHEN_SIZE_UNTRACKED;
67  
68    private int runtime = 0;
69    private int runs = 0;
70    private int changes = 0;
71    private int loopRuns = 0;
72    private int loopChanges = 0;
73  
74    // The following fields for tracking size changes are just estimates.
75    // They do not take into account preserved license blocks, newline padding,
76    // or pretty printing (if enabled), since they don't use CodePrinter.
77    // To get exact sizes, call compiler.toSource() for the final generated code.
78    private int codeSize = DEFAULT_WHEN_SIZE_UNTRACKED;
79    private int gzCodeSize = DEFAULT_WHEN_SIZE_UNTRACKED;
80    private int diff = 0;
81    private int gzDiff = 0;
82  
83    private final Deque<Stats> currentPass = new ArrayDeque<>();
84  
85    /** Summary stats by pass name. */
86    private final Map<String, Stats> summary = new HashMap<>();
87  
88    // To share with the rest of the program
89    private ImmutableMap<String, Stats> summaryCopy;
90  
91    /** Stats for each run of a compiler pass. */
92    private final List<Stats> log = new ArrayList<>();
93  
94    PerformanceTracker(Node jsRoot, TracerMode mode) {
95      this.jsRoot = jsRoot;
96      switch (mode) {
97        case TIMING_ONLY:
98          this.trackSize = false;
99          this.trackGzSize = false;
100         break;
101 
102       case RAW_SIZE:
103         this.trackSize = true;
104         this.trackGzSize = false;
105         break;
106 
107       case ALL:
108         this.trackSize = true;
109         this.trackGzSize = true;
110         break;
111 
112       case OFF:
113       default:
114         throw new IllegalArgumentException(
115             "PerformanceTracker can't work without tracer data.");
116     }
117   }

124     currentPass.push(new Stats(passName, isOneTime));

128   /**
129    * Collects information about a pass P after P finishes running, eg, how much
130    * time P took and what was its impact on code size.
131    *
132    * @param passName short name of the pass
133    * @param runtime execution time in milliseconds
134    */
135   void recordPassStop(String passName, long runtime) {
136     Stats logStats = currentPass.pop();
137     Preconditions.checkState(passName.equals(logStats.pass));
138 
139     // After parsing, initialize codeSize and gzCodeSize
140     if (passName.equals(Compiler.PARSING_PASS_NAME) && trackSize) {
141       CodeSizeEstimatePrinter estimatePrinter = new CodeSizeEstimatePrinter();
142       CodeGenerator.forCostEstimation(estimatePrinter).add(jsRoot);
143       initCodeSize = codeSize = estimatePrinter.calcSize();
144       if (this.trackGzSize) {
145         initGzCodeSize = gzCodeSize = estimatePrinter.calcZippedSize();
146       }
147     }
148 
149     // Populate log and summary
150     log.add(logStats);
151     Stats summaryStats = summary.get(passName);
152     if (summaryStats == null) {
153       summaryStats = new Stats(passName, logStats.isOneTime);
154       summary.put(passName, summaryStats);
155     }
156 
157     // Update fields that aren't related to code size
158     logStats.runtime = runtime;
159     logStats.runs = 1;
160     summaryStats.runtime += runtime;
161     summaryStats.runs += 1;
162     if (codeChange.hasCodeChanged()) {
163       logStats.changes = 1;
164       summaryStats.changes += 1;
165     }
166 
167     // Update fields related to code size
168     if (codeChange.hasCodeChanged() && trackSize) {
169       int newSize = 0;
170       CodeSizeEstimatePrinter estimatePrinter = new CodeSizeEstimatePrinter();
171       CodeGenerator.forCostEstimation(estimatePrinter).add(jsRoot);
172       if (trackSize) {
173         newSize = estimatePrinter.calcSize();
174         logStats.diff = codeSize - newSize;
175         summaryStats.diff += logStats.diff;
176         codeSize = summaryStats.size = logStats.size = newSize;
177       }
178       if (trackGzSize) {
179         newSize = estimatePrinter.calcZippedSize();
180         logStats.gzDiff = gzCodeSize - newSize;
181         summaryStats.gzDiff += logStats.gzDiff;
182         gzCodeSize = summaryStats.gzSize = logStats.gzSize = newSize;
183       }
184     }
185   }

239   private void calcTotalStats() {
240     // This method only does work the first time it's called
241     if (summaryCopy != null) {
242       return;
243     }
244     summaryCopy = ImmutableMap.copyOf(summary);
245     for (Entry<String, Stats> entry : summary.entrySet()) {
246       Stats stats = entry.getValue();
247       runtime += stats.runtime;
248       runs += stats.runs;
249       changes += stats.changes;
250       if (!stats.isOneTime) {
251         loopRuns += stats.runs;
252         loopChanges += stats.changes;
253       }
254       diff += stats.diff;
255       gzDiff += stats.gzDiff;
256     }
257     Preconditions.checkState(!trackSize || initCodeSize == diff + codeSize);
258     Preconditions.checkState(!trackGzSize
259         || initGzCodeSize == gzDiff + gzCodeSize);
260   }

262   /**
263    * Prints a summary, which contains aggregate stats for all runs of each pass
264    * and a log, which contains stats for each individual run.
265    */
266   public void outputTracerReport(PrintStream pstr) {
267     JvmMetrics.maybeWriteJvmMetrics(pstr, "verbose:pretty:all");
268     OutputStreamWriter output = new OutputStreamWriter(pstr, UTF_8);
269     try {
270       calcTotalStats();
271 
272       ArrayList<Entry<String, Stats>> statEntries = new ArrayList<>();
273       statEntries.addAll(summary.entrySet());
274       Collections.sort(statEntries, new Comparator<Entry<String, Stats>>() {
275         @Override
276         public int compare(Entry<String, Stats> e1, Entry<String, Stats> e2) {
277           return (int) (e1.getValue().runtime - e2.getValue().runtime);
278         }
279       });
280 
281       output.write("Summary:\n" +
282           "pass,runtime,runs,changingRuns,reduction,gzReduction\n");
283       for (Entry<String, Stats> entry : statEntries) {
284         String key = entry.getKey();
285         Stats stats = entry.getValue();
286         output.write(String.format("%s,%d,%d,%d,%d,%d\n", key, stats.runtime,
287             stats.runs, stats.changes, stats.diff, stats.gzDiff));
288       }
289       output.write("\nTOTAL:"
290           + "\nRuntime(ms): " + runtime + "\n#Runs: " + runs
291           + "\n#Changing runs: " + changes + "\n#Loopable runs: " + loopRuns
292           + "\n#Changing loopable runs: " + loopChanges + "\nEstimated Reduction(bytes): " + diff
293           + "\nEstimated GzReduction(bytes): " + gzDiff + "\nEstimated Size(bytes): " + codeSize
294           + "\nEstimated GzSize(bytes): " + gzCodeSize + "\n\n");
295 
296       output.write("Log:\n" +
297           "pass,runtime,runs,changingRuns,reduction,gzReduction,size,gzSize\n");
298       for (Stats stats : log) {
299         output.write(String.format("%s,%d,%d,%d,%d,%d,%d,%d\n",
300             stats.pass, stats.runtime, stats.runs, stats.changes,
301             stats.diff, stats.gzDiff, stats.size, stats.gzSize));
302       }
303       output.write("\n");
304       // output can be System.out, so don't close it to not lose subsequent
305       // error messages. Flush to ensure that you will see the tracer report.
306       output.flush();
307     } catch (IOException e) {
308       throw new RuntimeException("Failed to write statistics to output.", e);
309     }
310   }

312   /**
313    * A Stats object contains statistics about a pass run, such as running time,
314    * size changes, etc
315    */
316   public static class Stats {
317     Stats(String pass, boolean iot) {
318       this.pass = pass;
319       this.isOneTime = iot;
320     }
321     public final String pass;
322     public final boolean isOneTime;
323     public long runtime = 0;
324     public int runs = 0;
325     public int changes = 0;
326     public int diff = 0;
327     public int gzDiff = 0;
328     public int size;
329     public int gzSize;
330   }
331 
332   /** An object to get a gzsize estimate; it doesn't generate code. */
333   private final class CodeSizeEstimatePrinter extends CodeConsumer {
334     private int size = 0;
335     private char lastChar = '\0';
336     private final ByteArrayOutputStream output = new ByteArrayOutputStream();
337     private final GZIPOutputStream stream;
338 
339     private CodeSizeEstimatePrinter() {
340       try {
341         stream = new GZIPOutputStream(output);
342       } catch (IOException e) {
343         throw new RuntimeException(e);
344       }
345     }
346 
347     @Override
348     void append(String str) {
349       int len = str.length();
350       if (len > 0) {
351         size += len;
352         lastChar = str.charAt(len - 1);
353         if (trackGzSize) {
354           try {
355             stream.write(str.getBytes(UTF_8));
356           } catch (IOException e) {
357             throw new RuntimeException(e);
358           }
359         }
360       }
361     }
362 
363     @Override
364     char getLastChar() {
365       return lastChar;
366     }
367 
368     private int calcSize() {
369       return size;
370     }
371 
372     // Called iff trackGzSize is true
373     private int calcZippedSize() {
374       try {
375         stream.finish();
376         stream.close();
377         return output.size();
378       } catch (IOException e) {
379         throw new RuntimeException(e);
380       }
381     }
382   }
