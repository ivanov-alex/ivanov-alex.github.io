34  /**
35   * A flow scope that tries to store as little symbol information as possible,
36   * instead delegating to its parents. Optimized for low memory use.
37   *
38   * @author nicksantos@google.com (Nick Santos)
39   */
40  class LinkedFlowScope implements FlowScope {
41    // The closest flow scope cache.
42    private final FlatFlowScopeCache cache;
43  
44    // The parent flow scope.
45    private final LinkedFlowScope parent;
46  
47    // The distance between this flow scope and the closest flat flow scope.
48    private int depth;
49  
50    static final int MAX_DEPTH = 250;
51  
52    // A FlatFlowScopeCache equivalent to this scope.
53    private FlatFlowScopeCache flattened;
54  
55    // Flow scopes assume that all their ancestors are immutable.
56    // So once a child scope is created, this flow scope may not be modified.
57    private boolean frozen = false;
58  
59    // The last slot defined in this flow instruction, and the head of the
60    // linked list of slots.
61    private LinkedFlowSlot lastSlot;
62  
63    private LinkedFlowScope(FlatFlowScopeCache cache,
64        LinkedFlowScope directParent) {
65      this.cache = cache;
66      if (directParent == null) {
67        this.lastSlot = null;
68        this.depth = 0;
69        this.parent = cache.linkedEquivalent;
70      } else {
71        this.lastSlot = directParent.lastSlot;
72        this.depth = directParent.depth + 1;
73        this.parent = directParent;
74      }
75    }
76  
77    LinkedFlowScope(FlatFlowScopeCache cache) {
78      this(cache, null);
79    }
80  
81    LinkedFlowScope(LinkedFlowScope directParent) {
82      this(directParent.cache, directParent);
83    }

85    /** Gets the function scope for this flow scope. */
86    private TypedScope getFunctionScope() {
87      return cache.functionScope;
88    }
89  
90    /** Whether this flows from a bottom scope. */
91    private boolean flowsFromBottom() {
92      return getFunctionScope().isBottom();
93    }

95    /**
96     * Creates an entry lattice for the flow.
97     */
98    public static LinkedFlowScope createEntryLattice(TypedScope scope) {
99      return new LinkedFlowScope(new FlatFlowScopeCache(scope));

102   @Override
103   public void inferSlotType(String symbol, JSType type) {
104     Preconditions.checkState(!frozen);
105     lastSlot = new LinkedFlowSlot(symbol, type, lastSlot);
106     depth++;
107     cache.dirtySymbols.add(symbol);
108   }

110   @Override
111   public void inferQualifiedSlot(Node node, String symbol, JSType bottomType,
112       JSType inferredType, boolean declared) {
113     TypedScope functionScope = getFunctionScope();
114     if (functionScope.isLocal()) {
115       TypedVar v  = functionScope.getVar(symbol);
116       if (v == null && !functionScope.isBottom()) {
117         v = functionScope.declare(symbol, node, bottomType, null, !declared);
118       }
119 
120       if (v != null && !v.isTypeInferred()) {
121         JSType declaredType = v.getType();
122         // Use the inferred type over the declared type only if the
123         // inferred type is a strict subtype of the declared type.
124         if (declaredType != null && inferredType.isSubtype(declaredType)
125             && !declaredType.isSubtype(inferredType)
126             && !inferredType.isEquivalentTo(declaredType)) {
127           inferSlotType(symbol, inferredType);
128         }
129       } else {
130         inferSlotType(symbol, inferredType);
131       }
132     }
133   }

150   /**
151    * Get the slot for the given symbol.
152    */
153   @Override
154   public StaticTypedSlot<JSType> getSlot(String name) {
155     if (cache.dirtySymbols.contains(name)) {
156       for (LinkedFlowSlot slot = lastSlot;
157            slot != null; slot = slot.parent) {
158         if (slot.getName().equals(name)) {
159           return slot;
160         }
161       }
162     }
163     return cache.getSlot(name);
164   }

167   public StaticTypedSlot<JSType> getOwnSlot(String name) {
168     throw new UnsupportedOperationException();
169   }

171   @Override
172   public FlowScope createChildFlowScope() {
173     frozen = true;
174 
175     if (depth > MAX_DEPTH) {
176       if (flattened == null) {
177         flattened = new FlatFlowScopeCache(this);
178       }
179       return new LinkedFlowScope(flattened);
180     }
181 
182     return new LinkedFlowScope(this);
183   }

185   /**
186    * Iterate through all the linked flow scopes before this one.
187    * If there's one and only one slot defined between this scope
188    * and the blind scope, return it.
189    */
190   @Override
191   public StaticTypedSlot<JSType> findUniqueRefinedSlot(FlowScope blindScope) {
192     StaticTypedSlot<JSType> result = null;
193 
194     for (LinkedFlowScope currentScope = this;
195          currentScope != blindScope;
196          currentScope = currentScope.parent) {
197       for (LinkedFlowSlot currentSlot = currentScope.lastSlot;
198            currentSlot != null &&
199            (currentScope.parent == null ||
200             currentScope.parent.lastSlot != currentSlot);
201            currentSlot = currentSlot.parent) {
202         if (result == null) {
203           result = currentSlot;
204         } else if (!currentSlot.getName().equals(result.getName())) {
205           return null;
206         }
207       }
208     }
209 
210     return result;
211   }

213   /**
214    * Look through the given scope, and try to find slots where it doesn't
215    * have enough type information. Then fill in that type information
216    * with stuff that we've inferred in the local flow.
217    */
218   @Override
219   public void completeScope(StaticTypedScope<JSType> staticScope) {
220     TypedScope scope = (TypedScope) staticScope;
221     for (Iterator<TypedVar> it = scope.getVars(); it.hasNext();) {
222       TypedVar var = it.next();
223       if (var.isTypeInferred()) {
224         JSType type = var.getType();
225         if (type == null || type.isUnknownType()) {
226           JSType flowType = getSlot(var.getName()).getType();
227           var.setType(flowType);
228         }
229       }
230     }
231   }

233   /**
234    * Remove flow scopes that add nothing to the flow.
235    */
236   // NOTE(nicksantos): This function breaks findUniqueRefinedSlot, because
237   // findUniqueRefinedSlot assumes that this scope is a direct descendant
238   // of blindScope. This is not necessarily true if this scope has been
239   // optimize()d and blindScope has not. This should be fixed. For now,
240   // we only use optimize() where we know that we won't have to do
241   // a findUniqueRefinedSlot on it.
242   @Override
243   public LinkedFlowScope optimize() {
244     LinkedFlowScope current;
245     for (current = this;
246          current.parent != null &&
247              current.lastSlot == current.parent.lastSlot;
248          current = current.parent) {}
249     return current;
250   }

252   /** Join the two FlowScopes. */
253   static class FlowScopeJoinOp extends JoinOp.BinaryJoinOp<FlowScope> {
254     @SuppressWarnings("unchecked")
255     @Override
256     public FlowScope apply(FlowScope a, FlowScope b) {
257       // To join the two scopes, we have to
258       LinkedFlowScope linkedA = (LinkedFlowScope) a;
259       LinkedFlowScope linkedB = (LinkedFlowScope) b;
260       linkedA.frozen = true;
261       linkedB.frozen = true;
262       if (linkedA.optimize() == linkedB.optimize()) {
263         return linkedA.createChildFlowScope();
264       }
265       return new LinkedFlowScope(new FlatFlowScopeCache(linkedA, linkedB));
266     }
267   }

269   @Override
270   public boolean equals(Object other) {
271     if (other instanceof LinkedFlowScope) {
272       LinkedFlowScope that = (LinkedFlowScope) other;
273       if (this.optimize() == that.optimize()) {
274         return true;
275       }
276 
277       // If two flow scopes are in the same function, then they could have
278       // two possible function scopes: the real one and the BOTTOM scope.
279       // If they have different function scopes, we *should* iterate through all
280       // the variables in each scope and compare. However, 99.9% of the time,
281       // they're not equal. And the other .1% of the time, we can pretend
282       // they're equal--this just means that data flow analysis will have
283       // to propagate the entry lattice a little bit further than it
284       // really needs to. Everything will still come out ok.
285       if (this.getFunctionScope() != that.getFunctionScope()) {
286         return false;
287       }
288 
289       if (cache == that.cache) {
290         // If the two flow scopes have the same cache, then we can check
291         // equality a lot faster: by just looking at the "dirty" elements
292         // in the cache, and comparing them in both scopes.
293         for (String name : cache.dirtySymbols) {
294           if (diffSlots(getSlot(name), that.getSlot(name))) {
295             return false;
296           }
297         }
298 
299         return true;
300       }
301 
302       Map<String, StaticTypedSlot<JSType>> myFlowSlots = allFlowSlots();
303       Map<String, StaticTypedSlot<JSType>> otherFlowSlots = that.allFlowSlots();
304 
305       for (StaticTypedSlot<JSType> slot : myFlowSlots.values()) {
306         if (diffSlots(slot, otherFlowSlots.get(slot.getName()))) {
307           return false;
308         }
309         otherFlowSlots.remove(slot.getName());
310       }
311       for (StaticTypedSlot<JSType> slot : otherFlowSlots.values()) {
312         if (diffSlots(slot, myFlowSlots.get(slot.getName()))) {
313           return false;
314         }
315       }
316       return true;
317     }
318     return false;
319   }

321   /**
322    * Determines whether two slots are meaningfully different for the
323    * purposes of data flow analysis.
324    */
325   private static boolean diffSlots(StaticTypedSlot<JSType> slotA,
326                                    StaticTypedSlot<JSType> slotB) {
327     boolean aIsNull = slotA == null || slotA.getType() == null;
328     boolean bIsNull = slotB == null || slotB.getType() == null;
329     if (aIsNull && bIsNull) {
330       return false;
331     } else if (aIsNull ^ bIsNull) {
332       return true;
333     }
334 
335     // Both slots and types must be non-null.
336     return slotA.getType().differsFrom(slotB.getType());
337   }

339   /**
340    * Gets all the symbols that have been defined before this point
341    * in the current flow. Does not return slots that have not changed during
342    * the flow.
343    *
344    * For example, consider the code:
345    * <code>
346    * var x = 3;
347    * function f() {
348    *   var y = 5;
349    *   y = 6; // FLOW POINT
350    *   var z = y;
351    *   return z;
352    * }
353    * </code>
354    * A FlowScope at FLOW POINT will return a slot for y, but not
355    * a slot for x or z.
356    */
357   private Map<String, StaticTypedSlot<JSType>> allFlowSlots() {
358     Map<String, StaticTypedSlot<JSType>> slots = new HashMap<>();
359     for (LinkedFlowSlot slot = lastSlot;
360          slot != null; slot = slot.parent) {
361       if (!slots.containsKey(slot.getName())) {
362         slots.put(slot.getName(), slot);
363       }
364     }
365 
366     for (Map.Entry<String, StaticTypedSlot<JSType>> symbolEntry : cache.symbols.entrySet()) {
367       if (!slots.containsKey(symbolEntry.getKey())) {
368         slots.put(symbolEntry.getKey(), symbolEntry.getValue());
369       }
370     }
371 
372     return slots;
373   }

387       super(name, type, true);
388       this.parent = parent;
389     }

392   /**
393    * A map that tries to cache as much symbol table information
394    * as possible in a map. Optimized for fast lookup.
395    */
396   private static class FlatFlowScopeCache {
397     // The TypedScope for the entire function or for the global scope.
398     private final TypedScope functionScope;
399 
400     // The linked flow scope that this cache represents.
401     private final LinkedFlowScope linkedEquivalent;
402 
403     // All the symbols defined before this point in the local flow.
404     // May not include lazily declared qualified names.
405     private Map<String, StaticTypedSlot<JSType>> symbols = new HashMap<>();
406 
407     // Used to help make lookup faster for LinkedFlowScopes by recording
408     // symbols that may be redefined "soon", for an arbitrary definition
409     // of "soon". ;)
410     //
411     // More rigorously, if a symbol is redefined in a LinkedFlowScope,
412     // and this is the closest FlatFlowScopeCache, then that symbol is marked
413     // "dirty". In this way, we don't waste time looking in the LinkedFlowScope
414     // list for symbols that aren't defined anywhere nearby.
415     final Set<String> dirtySymbols = new HashSet<>();
416 
417     // The cache at the bottom of the lattice.
418     FlatFlowScopeCache(TypedScope functionScope) {
419       this.functionScope = functionScope;
420       symbols = ImmutableMap.of();
421       linkedEquivalent = null;
422     }
423 
424     // A cache in the middle of a long scope chain.
425     FlatFlowScopeCache(LinkedFlowScope directParent) {
426       FlatFlowScopeCache cache = directParent.cache;
427 
428       functionScope = cache.functionScope;
429       symbols = directParent.allFlowSlots();
430       linkedEquivalent = directParent;
431     }
432 
433     // A cache at the join of two scope chains.
434     FlatFlowScopeCache(LinkedFlowScope joinedScopeA,
435         LinkedFlowScope joinedScopeB) {
436       linkedEquivalent = null;
437 
438       // Always prefer the "real" function scope to the faked-out
439       // bottom scope.
440       functionScope = joinedScopeA.flowsFromBottom() ?
441           joinedScopeB.getFunctionScope() : joinedScopeA.getFunctionScope();
442 
443       Map<String, StaticTypedSlot<JSType>> slotsA = joinedScopeA.allFlowSlots();
444       Map<String, StaticTypedSlot<JSType>> slotsB = joinedScopeB.allFlowSlots();
445 
446       symbols = slotsA;
447 
448       // There are 5 different join cases:
449       // 1) The type is declared in joinedScopeA, not in joinedScopeB,
450       //    and not in functionScope. Just use the one in A.
451       // 2) The type is declared in joinedScopeB, not in joinedScopeA,
452       //    and not in functionScope. Just use the one in B.
453       // 3) The type is declared in functionScope and joinedScopeA, but
454       //    not in joinedScopeB. Join the two types.
455       // 4) The type is declared in functionScope and joinedScopeB, but
456       //    not in joinedScopeA. Join the two types.
457       // 5) The type is declared in joinedScopeA and joinedScopeB. Join
458       //    the two types.
459       Set<String> symbolNames = new HashSet<>(symbols.keySet());
460       symbolNames.addAll(slotsB.keySet());
461 
462       for (String name : symbolNames) {
463         StaticTypedSlot<JSType> slotA = slotsA.get(name);
464         StaticTypedSlot<JSType> slotB = slotsB.get(name);
465 
466         JSType joinedType = null;
467         if (slotB == null || slotB.getType() == null) {
468           StaticTypedSlot<JSType> fnSlot
469               = joinedScopeB.getFunctionScope().getSlot(name);
470           JSType fnSlotType = fnSlot == null ? null : fnSlot.getType();
471           if (fnSlotType == null) {
472             // Case #1 -- already inserted.
473           } else {
474             // Case #3
475             joinedType = slotA.getType().getLeastSupertype(fnSlotType);
476           }
477         } else if (slotA == null || slotA.getType() == null) {
478           StaticTypedSlot<JSType> fnSlot
479               = joinedScopeA.getFunctionScope().getSlot(name);
480           JSType fnSlotType = fnSlot == null ? null : fnSlot.getType();
481           if (fnSlotType == null) {
482             // Case #2
483             symbols.put(name, slotB);
484           } else {
485             // Case #4
486             joinedType = slotB.getType().getLeastSupertype(fnSlotType);
487           }
488         } else {
489           // Case #5
490           joinedType =
491               slotA.getType().getLeastSupertype(slotB.getType());
492         }
493 
494         if (joinedType != null) {
495           symbols.put(name, new SimpleSlot(name, joinedType, true));
496         }
497       }
498     }
499 
500     /**
501      * Get the slot for the given symbol.
502      */
503     public StaticTypedSlot<JSType> getSlot(String name) {
504       if (symbols.containsKey(name)) {
505         return symbols.get(name);
506       } else {
507         return functionScope.getSlot(name);
508       }
509     }
