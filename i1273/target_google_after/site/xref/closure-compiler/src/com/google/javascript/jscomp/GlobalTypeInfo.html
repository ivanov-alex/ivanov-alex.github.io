293   @Override
294   public void process(Node externs, Node root) {
295     Preconditions.checkNotNull(warnings, "Cannot rerun GlobalTypeInfo.process");
296     Preconditions.checkArgument(externs == null || externs.isSyntheticBlock());
297     Preconditions.checkArgument(root.isSyntheticBlock());
298     globalScope = new NTIScope(root, null, ImmutableList.<String>of(), commonTypes);
299     scopes.add(globalScope);
300 
301     // Processing of a scope is split into many separate phases, and it's not
302     // straightforward to remember which phase does what.
303 
304     // (1) Find names of classes, interfaces, typedefs, enums, and namespaces
305     //   defined in the global scope.
306     CollectNamedTypes rootCnt = new CollectNamedTypes(globalScope);
307     if (externs != null) {
308       NodeTraversal.traverseEs6(compiler, externs, rootCnt);
309     }
310     NodeTraversal.traverseEs6(compiler, root, rootCnt);
311     // (2) Determine the type represented by each typedef and each enum
312     globalScope.resolveTypedefs(typeParser);
313     globalScope.resolveEnums(typeParser);
314     // (3) Repeat steps 1-2 for all the other scopes (outer-to-inner)
315     for (int i = 1; i < scopes.size(); i++) {
316       NTIScope s = scopes.get(i);
317       CollectNamedTypes cnt = new CollectNamedTypes(s);
318       NodeTraversal.traverseEs6(compiler, s.getBody(), cnt);
319       s.resolveTypedefs(typeParser);
320       s.resolveEnums(typeParser);
321       if (NewTypeInference.measureMem) {
322         NewTypeInference.updatePeakMem();
323       }
324     }
325 
326     // If the Function constructor isn't defined, we cannot create function
327     // types. Exit early.
328     if (this.commonTypes.getFunctionType() == null) {
329       warnings.add(JSError.make(root, FUNCTION_CONSTRUCTOR_NOT_DEFINED));
330       return;
331     }
332 
333     // (4) The bulk of the global-scope processing happens here:
334     //     - Create scopes for functions
335     //     - Declare properties on types
336     ProcessScope rootPs = new ProcessScope(globalScope);
337     if (externs != null) {
338       NodeTraversal.traverseEs6(compiler, externs, rootPs);
339     }
340     NodeTraversal.traverseEs6(compiler, root, rootPs);
341     // (5) Things that must happen after the traversal of the scope
342     rootPs.finishProcessingScope();
343     for (String name : globalScope.getUnknownTypeNames()) {
344       globalScope.mayDeclareUnknownType(QualifiedName.fromQualifiedString(name));
345     }
346 
347     // (6) Repeat steps 4-5 for all the other scopes (outer-to-inner)
348     for (int i = 1; i < scopes.size(); i++) {
349       NTIScope s = scopes.get(i);
350       ProcessScope ps = new ProcessScope(s);
351       NodeTraversal.traverseEs6(compiler, s.getBody(), ps);
352       ps.finishProcessingScope();
353       if (NewTypeInference.measureMem) {
354         NewTypeInference.updatePeakMem();
355       }
356     }
357 
358     // (7) Adjust types of properties based on inheritance information.
359     //     Report errors in the inheritance chain.
360     for (RawNominalType rawType : nominaltypesByNode.values()) {
361       checkAndFinalizeNominalType(rawType);
362     }
363 
364     nominaltypesByNode = null;
365     propertyDefs = null;
366     for (NTIScope s : scopes) {
367       s.removeTmpData();
368     }
369     Map<Node, String> unknownTypes = typeParser.getUnknownTypesMap();
370     for (Map.Entry<Node, String> unknownTypeEntry : unknownTypes.entrySet()) {
371       warnings.add(JSError.make(unknownTypeEntry.getKey(),
372               UNRECOGNIZED_TYPE_NAME, unknownTypeEntry.getValue()));
373     }
374     // The jsdoc parser doesn't have access to the error functions in the jscomp
375     // package, so we collect its warnings here.
376     for (JSError warning : typeParser.getWarnings()) {
377       warnings.add(warning);
378     }
379     typeParser = null;
380     compiler.setSymbolTable(this);
381     warnings = null;
382 
383     // If a scope s1 contains a scope s2, then s2 must be before s1 in scopes.
384     // The type inference relies on this fact to process deeper scopes
385     // before shallower scopes.
386     Collections.reverse(scopes);
387   }

389   private Collection<PropertyDef> getPropDefsFromInterface(
390       NominalType nominalType, String pname) {
391     Preconditions.checkArgument(nominalType.isFinalized());
392     Preconditions.checkArgument(nominalType.isInterface());
393     if (nominalType.getPropDeclaredType(pname) == null) {
394       return ImmutableSet.of();
395     } else if (propertyDefs.get(nominalType.getId(), pname) != null) {
396       return ImmutableSet.of(propertyDefs.get(nominalType.getId(), pname));
397     }
398     ImmutableSet.Builder<PropertyDef> result = ImmutableSet.builder();
399     for (NominalType interf : nominalType.getInstantiatedInterfaces()) {
400       result.addAll(getPropDefsFromInterface(interf, pname));
401     }
402     return result.build();
403   }

405   private PropertyDef getPropDefFromClass(
406       NominalType nominalType, String pname) {
407     while (nominalType.getPropDeclaredType(pname) != null) {
408       Preconditions.checkArgument(nominalType.isFinalized());
409       Preconditions.checkArgument(nominalType.isClass());
410 
411       if (propertyDefs.get(nominalType.getId(), pname) != null) {
412         return propertyDefs.get(nominalType.getId(), pname);
413       }
414       nominalType = nominalType.getInstantiatedSuperclass();
415     }
416     return null;
417   }

419   private void checkAndFinalizeNominalType(RawNominalType rawType) {
420     if (rawType.isFinalized()) {
421       return;
422     }
423     NominalType superClass = rawType.getSuperClass();
424     Set<String> nonInheritedPropNames = rawType.getAllOwnProps();
425     if (superClass != null && !superClass.isFinalized()) {
426       checkAndFinalizeNominalType(superClass.getRawNominalType());
427     }
428     for (NominalType superInterf : rawType.getInterfaces()) {
429       if (!superInterf.isFinalized()) {
430         checkAndFinalizeNominalType(superInterf.getRawNominalType());
431       }
432     }
433 
434     Multimap<String, DeclaredFunctionType> propMethodTypesToProcess =
435         LinkedHashMultimap.create();
436     Multimap<String, JSType> propTypesToProcess = LinkedHashMultimap.create();
437     // Collect inherited types for extended classes
438     if (superClass != null) {
439       Preconditions.checkState(superClass.isFinalized());
440       // TODO(blickly): Can we optimize this to skip unnecessary iterations?
441       for (String pname : superClass.getAllPropsOfClass()) {
442         nonInheritedPropNames.remove(pname);
443         checkSuperProperty(rawType, superClass, pname,
444             propMethodTypesToProcess, propTypesToProcess);
445       }
446     }
447 
448     // Collect inherited types for extended/implemented interfaces
449     for (NominalType superInterf : rawType.getInterfaces()) {
450       Preconditions.checkState(superInterf.isFinalized());
451       for (String pname : superInterf.getAllPropsOfInterface()) {
452         nonInheritedPropNames.remove(pname);
453         checkSuperProperty(rawType, superInterf, pname,
454             propMethodTypesToProcess, propTypesToProcess);
455       }
456     }
457 
458     // Munge inherited types of methods
459     for (String pname : propMethodTypesToProcess.keySet()) {
460       Collection<DeclaredFunctionType> methodTypes =
461           propMethodTypesToProcess.get(pname);
462       Preconditions.checkState(!methodTypes.isEmpty());
463       PropertyDef localPropDef =
464           propertyDefs.get(rawType, pname);
465       // To find the declared type of a method, we must meet declared types
466       // from all inherited methods.
467       DeclaredFunctionType superMethodType =
468           DeclaredFunctionType.meet(methodTypes);
469       DeclaredFunctionType updatedMethodType =
470           localPropDef.methodType.withTypeInfoFromSuper(
471               superMethodType, getsTypeInfoFromParentMethod(localPropDef));
472       localPropDef.updateMethodType(updatedMethodType);
473       propTypesToProcess.put(pname,
474           commonTypes.fromFunctionType(updatedMethodType.toFunctionType()));
475     }
476 
477     // Check inherited types of all props
478  add_interface_props:
479     for (String pname : propTypesToProcess.keySet()) {
480       Collection<JSType> defs = propTypesToProcess.get(pname);
481       Preconditions.checkState(!defs.isEmpty());
482       JSType resultType = JSType.TOP;
483       for (JSType inheritedType : defs) {
484         resultType = JSType.meet(resultType, inheritedType);
485         if (!resultType.isBottom()) {
486           resultType = inheritedType;
487         } else {
488           Node defSite = rawType.getDefSite();
489           // TODO(blickly): Fix this error message to include supertype names
490           warnings.add(JSError.make(
491               defSite, TypeCheck.INCOMPATIBLE_EXTENDED_PROPERTY_TYPE,
492               NodeUtil.getNearestFunctionName(defSite), pname, "", ""));
493           continue add_interface_props;
494         }
495       }
496       // TODO(dimvar): check if we can have @const props here
497       rawType.addProtoProperty(pname, null, resultType, false);
498     }
499 
500     // Warn for a prop declared with @override that isn't overriding anything.
501     for (String pname : nonInheritedPropNames) {
502       Node propDefsite = propertyDefs.get(rawType, pname).defSite;
503       JSDocInfo jsdoc = NodeUtil.getBestJSDocInfo(propDefsite);
504       if (jsdoc != null && jsdoc.isOverride()) {
505         warnings.add(JSError.make(propDefsite, TypeCheck.UNKNOWN_OVERRIDE,
506                 pname, rawType.getName()));
507       }
508     }
509 
510     // Finalize nominal type once all properties are added.
511     rawType.finalize();
512   }

514   private void checkSuperProperty(
515       RawNominalType current, NominalType superType, String pname,
516       Multimap<String, DeclaredFunctionType> propMethodTypesToProcess,
517       Multimap<String, JSType> propTypesToProcess) {
518     JSType inheritedPropType = superType.getPropDeclaredType(pname);
519     if (inheritedPropType == null) {
520       // No need to go further for undeclared props.
521       return;
522     }
523     Collection<PropertyDef> inheritedPropDefs;
524     if (superType.isInterface()) {
525       inheritedPropDefs = getPropDefsFromInterface(superType, pname);
526     } else {
527       inheritedPropDefs =
528           ImmutableSet.of(getPropDefFromClass(superType, pname));
529     }
530     if (superType.isInterface() && current.isClass()
531         && !current.mayHaveProp(pname)) {
532       warnings.add(JSError.make(
533           inheritedPropDefs.iterator().next().defSite,
534           TypeValidator.INTERFACE_METHOD_NOT_IMPLEMENTED,
535           pname, superType.toString(), current.toString()));
536       return;
537     }
538     PropertyDef localPropDef = propertyDefs.get(current, pname);
539     JSType localPropType = localPropDef == null
540         ? null : current.getInstancePropDeclaredType(pname);
541     if (localPropDef != null && superType.isClass()
542         && localPropType.getFunType() != null
543         && superType.hasConstantProp(pname)) {
544       // TODO(dimvar): This doesn't work for multiple levels in the hierarchy.
545       // Clean up how we process inherited properties and then fix this.
546       warnings.add(JSError.make(
547           localPropDef.defSite, CANNOT_OVERRIDE_FINAL_METHOD, pname));
548       return;
549     }
550     // System.out.println("nominalType: " + current + "'s " + pname +
551     //     " localPropType: " + localPropType +
552     //     " with super: " + superType +
553     //     " inheritedPropType: " + inheritedPropType);
554     if (localPropType == null) {
555       // Add property from interface to class
556       propTypesToProcess.put(pname, inheritedPropType);
557     } else if (!getsTypeInfoFromParentMethod(localPropDef)
558         && !localPropType.isSubtypeOf(inheritedPropType)) {
559       warnings.add(JSError.make(
560           localPropDef.defSite, INVALID_PROP_OVERRIDE, pname,
561           inheritedPropType.toString(), localPropType.toString()));
562     } else if (localPropDef.methodType != null) {
563       // If we are looking at a method definition, munging may be needed
564       for (PropertyDef inheritedPropDef : inheritedPropDefs) {
565         if (inheritedPropDef.methodType != null) {
566           propMethodTypesToProcess.put(pname,
567               inheritedPropDef.methodType.substituteNominalGenerics(superType));
568         }
569       }
570     }
571   }

573   private static boolean getsTypeInfoFromParentMethod(PropertyDef pd) {
574     if (pd == null || pd.methodType == null) {
575       return false;
576     }
577     JSDocInfo jsdoc = NodeUtil.getBestJSDocInfo(pd.defSite);
578     return jsdoc == null
579         || jsdoc.isOverride() && !jsdoc.containsFunctionDeclaration();
580   }

594     @Override
595     public void visit(NodeTraversal t, Node n, Node parent) {
596       switch (n.getType()) {
597         case Token.FUNCTION: {
598           visitFunctionEarly(n);
599           break;
600         }
601         case Token.VAR: {
602           Node nameNode = n.getFirstChild();
603           if (NodeUtil.isNamespaceDecl(nameNode)) {
604             visitObjlitNamespace(nameNode);
605           } else if (NodeUtil.isTypedefDecl(nameNode)) {
606             visitTypedef(nameNode);
607           } else if (NodeUtil.isEnumDecl(nameNode)) {
608             visitEnum(nameNode);
609           } else if (isAliasedNamespaceDefinition(nameNode)) {
610             visitAliasedNamespace(nameNode);
611           }
612           break;
613         }
614         case Token.EXPR_RESULT: {
615           Node expr = n.getFirstChild();
616           switch (expr.getType()) {
617             case Token.ASSIGN:
618               if (!expr.getFirstChild().isGetProp()) {
619                 return;
620               }
621               expr = expr.getFirstChild();
622               // fall through
623             case Token.GETPROP:
624               if (isPrototypeProperty(expr)
625                   || NodeUtil.referencesThis(expr)
626                   || !expr.isQualifiedName()) {
627                 // Class & prototype properties are handled in ProcessScope
628                 return;
629               }
630               processQualifiedDefinition(expr);
631               break;
632             case Token.CALL: {
633               List<String> decls = convention.identifyTypeDeclarationCall(expr);
634               if (decls == null || decls.isEmpty()) {
635                 return;
636               }
637               currentScope.addUnknownTypeNames(decls);
638               break;
639             }
640           }
641           break;
642         }
643       }
644     }

646     private void processQualifiedDefinition(Node qnameNode) {
647       Preconditions.checkArgument(qnameNode.isGetProp());
648       Preconditions.checkArgument(qnameNode.isQualifiedName());
649       Node recv = qnameNode.getFirstChild();
650       if (!currentScope.isNamespace(recv) && !mayCreateFunctionNamespace(recv)) {
651         return;
652       }
653       if (NodeUtil.isNamespaceDecl(qnameNode)) {
654         visitObjlitNamespace(qnameNode);
655       } else if (NodeUtil.isTypedefDecl(qnameNode)) {
656         visitTypedef(qnameNode);
657       } else if (NodeUtil.isEnumDecl(qnameNode)) {
658         visitEnum(qnameNode);
659       } else if (isAliasedNamespaceDefinition(qnameNode)) {
660         visitAliasedNamespace(qnameNode);
661       } else if (isQualifiedFunctionDefinition(qnameNode)) {
662         maybeAddFunctionScopeToNamespace(qnameNode);
663       }
664     }

666     private boolean isAliasedNamespaceDefinition(Node qnameNode) {
667       Node rhs = NodeUtil.getRValueOfLValue(qnameNode);
668       if (rhs == null || !rhs.isQualifiedName()) {
669         return false;
670       }
671       JSDocInfo jsdoc = NodeUtil.getBestJSDocInfo(qnameNode);
672       return jsdoc != null
673           && (jsdoc.isConstructorOrInterface() || jsdoc.hasConstAnnotation());
674     }

676     private boolean isQualifiedFunctionDefinition(Node qnameNode) {
677       Preconditions.checkArgument(qnameNode.isGetProp());
678       Preconditions.checkArgument(qnameNode.isQualifiedName());
679       Node parent = qnameNode.getParent();
680       return parent.isAssign()
681           && parent.getParent().isExprResult()
682           && parent.getLastChild().isFunction();

685     // Returns true iff it creates a new function namespace
686     private boolean mayCreateFunctionNamespace(Node qnameNode) {
687       if (!qnameNode.isQualifiedName()) {
688         return false;
689       }
690       QualifiedName qname = QualifiedName.fromNode(qnameNode);
691       Preconditions.checkState(!currentScope.isNamespace(qname));
692       if (!currentScope.isKnownFunction(qname)) {
693         return false;
694       }
695       if (qnameNode.isGetProp()) {
696         markAssignNodeAsAnalyzed(qnameNode.getParent().getParent());
697       }
698       NTIScope s;
699       if (qnameNode.isName()) {
700         // s is the scope that contains the function
701         s = currentScope.getScope(qnameNode.getString()).getParent();
702       } else {
703         s = currentScope;
704       }
705       s.addNamespaceLit(qnameNode);
706       return true;
707     }

749     private void visitEnum(Node qnameNode) {
750       Preconditions.checkState(qnameNode.isQualifiedName());
751       qnameNode.putBooleanProp(Node.ANALYZED_DURING_GTI, true);
752       // if (qnameNode.isName()
753       //     && currentScope.isDefinedLocally(qnameNode.getString())) {
754       //   String qname = qnameNode.getQualifiedName();
755       //   warnings.add(JSError.make(qnameNode,
756       //           VariableReferenceCheck.REDECLARED_VARIABLE, qname));
757       // }
758       if (currentScope.isDefined(qnameNode)) {
759         return;
760       }
761       Node init = NodeUtil.getRValueOfLValue(qnameNode);
762       // First check if the definition is an alias of a previous enum.
763       if (init != null && init.isQualifiedName()) {
764         EnumType et = currentScope.getEnum(QualifiedName.fromNode(init));
765         if (et != null) {
766           currentScope.addEnum(qnameNode, et);
767           return;
768         }
769       }
770       // Then check if the enum initializer is an object literal.
771       if (init == null || !init.isObjectLit() || init.getFirstChild() == null) {
772         warnings.add(JSError.make(qnameNode, MALFORMED_ENUM));
773         return;
774       }
775       // Last, read the object-literal properties and create the EnumType.
776       JSDocInfo jsdoc = NodeUtil.getBestJSDocInfo(qnameNode);
777       Set<String> propNames = new LinkedHashSet<>();
778       for (Node prop : init.children()) {
779         String pname = NodeUtil.getObjectLitKeyName(prop);
780         if (propNames.contains(pname)) {
781           warnings.add(JSError.make(qnameNode, DUPLICATE_PROP_IN_ENUM, pname));
782         }
783         if (!convention.isValidEnumKey(pname)) {
784           warnings.add(
785               JSError.make(prop, TypeCheck.ENUM_NOT_CONSTANT, pname));
786         }
787         propNames.add(pname);
788       }
789       currentScope.addEnum(qnameNode,
790           EnumType.make(
791               qnameNode.getQualifiedName(),
792               jsdoc.getEnumParameterType(),
793               ImmutableSet.copyOf(propNames)));
794     }

796     private void visitFunctionEarly(Node fn) {
797       JSDocInfo fnDoc = NodeUtil.getBestJSDocInfo(fn);
798       Node nameNode = NodeUtil.getFunctionNameNode(fn);
799       String internalName = createFunctionInternalName(fn, nameNode);
800       boolean isRedeclaration;
801       if (nameNode == null || !nameNode.isQualifiedName()) {
802         isRedeclaration = false;
803       } else if (nameNode.isName()) {
804         isRedeclaration = currentScope.isDefinedLocally(nameNode.getString(), false);
805       } else {
806         isRedeclaration = currentScope.isDefined(nameNode);
807       }
808       NTIScope fnScope = new NTIScope(fn, this.currentScope, collectFormals(fn, fnDoc), null);
809       if (!fn.isFromExterns()) {
810         scopes.add(fnScope);
811       }
812       this.currentScope.addLocalFunDef(internalName, fnScope);
813       maybeRecordNominalType(fn, nameNode, fnDoc, isRedeclaration);
814     }

816     private String createFunctionInternalName(Node fn, Node nameNode) {
817       String internalName = null;
818       if (nameNode == null || !nameNode.isName()
819           || nameNode.getParent().isAssign()) {
820         // Anonymous functions, qualified names, and stray assignments
821         // (eg, f = function(x) { ... }; ) get gensymed names.
822         internalName = ANON_FUN_PREFIX + freshId;
823         anonFunNames.put(fn, internalName);
824         freshId++;
825       } else if (currentScope.isDefinedLocally(nameNode.getString(), false)) {
826         String fnName = nameNode.getString();
827         Preconditions.checkState(!fnName.contains("."));
828         // warnings.add(JSError.make(
829         //     fn, VariableReferenceCheck.REDECLARED_VARIABLE, fnName));
830         // Redeclared variables also need gensymed names
831         internalName = ANON_FUN_PREFIX + freshId;
832         anonFunNames.put(fn, internalName);
833         freshId++;
834       } else {
835         // fnNameNode is undefined simple name
836         internalName = nameNode.getString();
837       }
838       return internalName;
839     }

841     private ArrayList<String> collectFormals(Node fn, JSDocInfo fnDoc) {
842       Preconditions.checkArgument(fn.isFunction());
843       // Collect the names of the formals.
844       // If a formal is a placeholder for variable arity, eg,
845       // /** @param {...?} var_args */ function f(var_args) { ... }
846       // then we don't collect it.
847       // But to decide that we can't just use the jsdoc b/c the type parser
848       // may ignore the jsdoc; the only reliable way is to collect the names of
849       // formals after building the declared function type.
850       ArrayList<String> formals = new ArrayList<>();
851       // tmpRestFormals is used only for error checking
852       ArrayList<String> tmpRestFormals = new ArrayList<>();
853       Node param = NodeUtil.getFunctionParameters(fn).getFirstChild();
854       while (param != null) {
855         if (JSTypeCreatorFromJSDoc.isRestArg(fnDoc, param.getString())
856             && param.getNext() == null) {
857           tmpRestFormals.add(param.getString());
858         } else {
859           formals.add(param.getString());
860         }
861         param = param.getNext();
862       }
863       if (fnDoc != null) {
864         for (String formalInJsdoc : fnDoc.getParameterNames()) {
865           if (!formals.contains(formalInJsdoc)
866               && !tmpRestFormals.contains(formalInJsdoc)) {
867             String functionName = NodeUtil.getNearestFunctionName(fn);
868             warnings.add(JSError.make(fn, INEXISTENT_PARAM, formalInJsdoc, functionName));
869           }
870         }
871       }
872       return formals;
873     }

875     private void maybeRecordNominalType(
876         Node fn, Node nameNode, JSDocInfo fnDoc, boolean isRedeclaration) {
877       if (fnDoc != null && fnDoc.isConstructorOrInterface()) {
878         QualifiedName qname = QualifiedName.fromNode(nameNode);
879         if (qname == null) {
880           warnings.add(JSError.make(fn, ANONYMOUS_NOMINAL_TYPE));
881           return;
882         }
883         ImmutableList<String> typeParameters = fnDoc.getTemplateTypeNames();
884         RawNominalType rawNominalType;
885         if (fnDoc.isInterface()) {
886           rawNominalType = RawNominalType.makeInterface(fn, qname, typeParameters);
887         } else if (fnDoc.makesStructs()) {
888           rawNominalType = RawNominalType.makeStructClass(fn, qname, typeParameters);
889         } else if (fnDoc.makesDicts()) {
890           rawNominalType = RawNominalType.makeDictClass(fn, qname, typeParameters);
891         } else {
892           rawNominalType = RawNominalType.makeUnrestrictedClass(fn, qname, typeParameters);
893         }
894         nominaltypesByNode.put(fn, rawNominalType);
895         if (isRedeclaration) {
896           return;
897         }
898         if (nameNode.isName()
899             || currentScope.isNamespace(nameNode.getFirstChild())
900             || mayCreateFunctionNamespace(nameNode.getFirstChild())) {
901           if (nameNode.isGetProp()) {
902             fn.getParent().getFirstChild().putBooleanProp(Node.ANALYZED_DURING_GTI, true);
903           } else if (currentScope.isTopLevel()) {
904             maybeRecordBuiltinType(nameNode.getString(), rawNominalType);
905           }
906           currentScope.addNominalType(nameNode, rawNominalType);
907         }
908       } else if (fnDoc != null) {
909         if (fnDoc.makesStructs()) {
910           warnings.add(JSError.make(fn, STRUCTDICT_WITHOUT_CTOR, "@struct"));
911         } else if (fnDoc.makesDicts()) {
912           warnings.add(JSError.make(fn, STRUCTDICT_WITHOUT_CTOR, "@dict"));
913         }
914       }
915     }

947     private void visitAliasedNamespace(Node lhs) {
948       Node rhs = NodeUtil.getRValueOfLValue(lhs);
949       QualifiedName rhsQname = QualifiedName.fromNode(rhs);
950       JSDocInfo jsdoc = NodeUtil.getBestJSDocInfo(lhs);
951 
952       if (jsdoc != null && jsdoc.isConstructorOrInterface()) {
953         RawNominalType rawType = this.currentScope.getNominalType(rhsQname);
954         if (jsdoc.isConstructor()
955             && (rawType == null || rawType.isInterface())) {
956           warnings.add(JSError.make(rhs, EXPECTED_CONSTRUCTOR, rhsQname.toString()));
957           return;
958         }
959         if (jsdoc.isInterface()
960             && (rawType == null || rawType.isClass())) {
961           warnings.add(JSError.make(rhs, EXPECTED_INTERFACE, rhsQname.toString()));
962           return;
963         }
964       }
965       Namespace ns = this.currentScope.getNamespace(rhsQname);
966       if (ns != null) {
967         this.currentScope.addNamespace(lhs, ns);
968       }
969     }

1010      * @lends can lend properties to an object X being defined in the same
1011      * statement as the @lends. To make sure that we've seen the definition of
1012      * X, we process @lends annotations after we've traversed the scope.
1013      *
1014      * @lends can only add properties to namespaces, constructors and prototypes

1016     void processLendsNode(Node objlit) {
1017       JSDocInfo jsdoc = objlit.getJSDocInfo();
1018       String lendsName = jsdoc.getLendsName();
1019       Preconditions.checkNotNull(lendsName);
1020       QualifiedName lendsQname = QualifiedName.fromQualifiedString(lendsName);
1021       if (currentScope.isNamespace(lendsQname)) {
1022         processLendsToNamespace(lendsQname, lendsName, objlit);
1023       } else {
1024         RawNominalType rawType = checkValidLendsToPrototypeAndGetClass(
1025             lendsQname, lendsName, objlit);
1026         if (rawType == null) {
1027           return;
1028         }
1029         for (Node prop : objlit.children()) {
1030           String pname =  NodeUtil.getObjectLitKeyName(prop);
1031           mayAddPropToPrototype(rawType, pname, prop, prop.getFirstChild());
1032         }
1033       }

1036     void processLendsToNamespace(
1037         QualifiedName lendsQname, String lendsName, Node objlit) {
1038       RawNominalType rawType = currentScope.getNominalType(lendsQname);
1039       if (rawType != null && rawType.isInterface()) {
1040         warnings.add(JSError.make(objlit, LENDS_ON_BAD_TYPE, lendsName));
1041         return;
1042       }
1043       Namespace borrowerNamespace = currentScope.getNamespace(lendsQname);
1044       for (Node prop : objlit.children()) {
1045         String pname = NodeUtil.getObjectLitKeyName(prop);
1046         JSType propDeclType = declaredObjLitProps.get(prop);
1047         if (propDeclType != null) {
1048           borrowerNamespace.addProperty(pname, prop, propDeclType, false);
1049         } else {
1050           JSType t = simpleInferExprType(prop.getFirstChild());
1051           if (t == null) {
1052             t = JSType.UNKNOWN;
1053           }
1054           borrowerNamespace.addProperty(pname, prop, t, false);
1055         }
1056       }
1057     }

1059     RawNominalType checkValidLendsToPrototypeAndGetClass(
1060         QualifiedName lendsQname, String lendsName, Node objlit) {
1061       if (!lendsQname.getRightmostName().equals("prototype")) {
1062         warnings.add(JSError.make(objlit, LENDS_ON_BAD_TYPE, lendsName));
1063         return null;
1064       }

1067       if (rawType == null || rawType.isInterface()) {
1068         warnings.add(JSError.make(objlit, LENDS_ON_BAD_TYPE, lendsName));
1069       }
1070       return rawType;

1073     @Override
1074     public void visit(NodeTraversal t, Node n, Node parent) {
1075       switch (n.getType()) {
1076         case Token.FUNCTION:
1077           Node grandparent = parent.getParent();
1078           if (grandparent == null
1079               || !isPrototypePropertyDeclaration(grandparent)) {
1080             RawNominalType ownerType = maybeGetOwnerType(n, parent);
1081             visitFunctionLate(n, ownerType);
1082           }
1083           break;
1084         case Token.NAME: {
1085           String name = n.getString();
1086           if (name == null || "undefined".equals(name) || parent.isFunction()) {
1087             return;
1088           }
1089           // TODO(dimvar): Handle local scopes introduced by catch properly,
1090           // after we decide what to do with variables in general, eg, will we
1091           // use unique numeric ids?
1092           if (parent.isVar() || parent.isCatch()) {
1093             visitVar(n, parent);
1094           } else if (currentScope.isOuterVarEarly(name)) {
1095             currentScope.addOuterVar(name);
1096           } else if (// Typedef variables can't be referenced in the source.
1097               currentScope.getTypedef(name) != null
1098               || !name.equals(currentScope.getName())
1099               && !currentScope.isDefinedLocally(name, false)) {
1100             // undeclaredVars.put(name, n);
1101           }
1102           break;
1103         }
1104         case Token.GETPROP:
1105           if (parent.isExprResult() && n.isQualifiedName()) {
1106             visitPropertyDeclaration(n);
1107           }
1108           break;
1109         case Token.ASSIGN: {
1110           Node lvalue = n.getFirstChild();
1111           if (lvalue.isGetProp() && lvalue.isQualifiedName() && parent.isExprResult()) {
1112             visitPropertyDeclaration(lvalue);
1113           }
1114           break;
1115         }
1116         case Token.CAST:
1117           castTypes.put(n,
1118               getDeclaredTypeOfNode(n.getJSDocInfo(), currentScope));
1119           break;
1120         case Token.OBJECTLIT:
1121           visitObjectLit(n, parent);
1122           break;
1123         case Token.CALL:
1124           visitCall(n);
1125           break;
1126       }
1127     }

1129     private void visitVar(Node nameNode, Node parent) {
1130       String name = nameNode.getString();
1131       if (this.currentScope.isNamespace(name)
1132           || NodeUtil.isTypedefDecl(nameNode)
1133           || NodeUtil.isEnumDecl(nameNode)) {
1134         if (!this.currentScope.isDefinedLocally(name, false)) {
1135           // Malformed enum or typedef
1136           this.currentScope.addLocal(
1137               name, JSType.UNKNOWN, false, nameNode.isFromExterns());
1138         }
1139         return;
1140       }
1141       Node initializer = nameNode.getFirstChild();
1142       if (initializer != null && initializer.isFunction()) {
1143         return;
1144       } else if (this.currentScope.isDefinedLocally(name, false)) {
1145         // warnings.add(JSError.make(
1146         //     n, VariableReferenceCheck.REDECLARED_VARIABLE, name));
1147         return;
1148       }
1149       // for (Node useBeforeDeclNode : undeclaredVars.get(name)) {
1150       //   warnings.add(JSError.make(useBeforeDeclNode,
1151       //       VariableReferenceCheck.EARLY_REFERENCE, name));
1152       // }
1153       // undeclaredVars.removeAll(name);
1154       if (parent.isCatch()) {
1155         this.currentScope.addLocal(name, JSType.UNKNOWN, false, false);
1156       } else {
1157         boolean isConst = isConst(nameNode);
1158         JSType declType = getVarTypeFromAnnotation(nameNode);
1159         if (declType == null) {
1160           declType = mayInferFromRhsIfConst(nameNode);
1161         }
1162         this.currentScope.addLocal(name, declType, isConst, nameNode.isFromExterns());
1163       }
1164     }

1209     private void visitPropertyDeclaration(Node getProp) {
1210       // Class property
1211       if (isClassPropAccess(getProp, currentScope)) {
1212         if (isAnnotatedAsConst(getProp) && currentScope.isPrototypeMethod()) {

1232       }

1269       Preconditions.checkArgument(getProp.isGetProp());
1270       Node parent = getProp.getParent();

1329     private void visitConstructorPropertyDeclaration(Node getProp) {
1330       Preconditions.checkArgument(getProp.isGetProp());
1331       // Named types have already been crawled in CollectNamedTypes
1332       if (isNamedType(getProp)) {
1333         return;
1334       }
1335       String ctorName = getProp.getFirstChild().getQualifiedName();
1336       QualifiedName ctorQname = QualifiedName.fromNode(getProp.getFirstChild());
1337       Preconditions.checkState(currentScope.isLocalFunDef(ctorName));
1338       RawNominalType classType = currentScope.getNominalType(ctorQname);
1339       String pname = getProp.getLastChild().getString();
1340       JSDocInfo jsdoc = NodeUtil.getBestJSDocInfo(getProp);
1341       JSType propDeclType = getDeclaredTypeOfNode(jsdoc, currentScope);
1342       boolean isConst = isConst(getProp);

1344         JSType previousPropType = classType.getCtorPropDeclaredType(pname);

1346             && previousPropType != null
1347             && !suppressDupPropWarning(jsdoc, propDeclType, previousPropType)) {
1348           warnings.add(JSError.make(getProp, REDECLARED_PROPERTY,
1349                   pname, classType.toString()));
1350           return;
1351         }
1352         if (propDeclType == null) {
1353           propDeclType = mayInferFromRhsIfConst(getProp);
1354         }
1355         classType.addCtorProperty(pname, getProp, propDeclType, isConst);
1356         getProp.putBooleanProp(Node.ANALYZED_DURING_GTI, true);
1357         if (isConst) {
1358           getProp.putBooleanProp(Node.CONSTANT_PROPERTY_DEF, true);
1359         }
1360       } else {
1361         classType.addUndeclaredCtorProperty(pname, getProp);
1362       }

1376     private void visitNamespacePropertyDeclaration(
1377         Node declNode, Node recv, String pname) {
1378       Preconditions.checkArgument(declNode.isGetProp() || declNode.isStringKey());
1379       Preconditions.checkArgument(currentScope.isNamespace(recv));
1380       EnumType et = currentScope.getEnum(QualifiedName.fromNode(recv));
1381       // If there is a reassignment to one of the enum's members, don't consider
1382       // that a definition of a new property.
1383       if (et != null && et.enumLiteralHasKey(pname)) {
1384         return;
1385       }
1386       Namespace ns = currentScope.getNamespace(QualifiedName.fromNode(recv));
1387       JSDocInfo jsdoc = NodeUtil.getBestJSDocInfo(declNode);
1388       JSType propDeclType = getDeclaredTypeOfNode(jsdoc, currentScope);
1389       boolean isConst = isConst(declNode);
1390       if (propDeclType != null || isConst) {
1391         JSType previousPropType = ns.getPropDeclaredType(pname);
1392         if (ns.hasProp(pname)
1393             && previousPropType != null
1394             && !suppressDupPropWarning(jsdoc, propDeclType, previousPropType)) {
1395           warnings.add(JSError.make(declNode, REDECLARED_PROPERTY,
1396                   pname, ns.toString()));
1397           return;
1398         }
1399         if (propDeclType == null) {
1400           propDeclType = mayInferFromRhsIfConst(declNode);
1401         }
1402         ns.addProperty(pname, declNode, propDeclType, isConst);
1403         declNode.putBooleanProp(Node.ANALYZED_DURING_GTI, true);
1404         if (declNode.isGetProp() && isConst) {
1405           declNode.putBooleanProp(Node.CONSTANT_PROPERTY_DEF, true);
1406         }
1407       } else {
1408         // Try to infer the prop type, but don't say that the prop is declared.
1409         Node initializer = NodeUtil.getRValueOfLValue(declNode);
1410         JSType t = initializer == null
1411             ? null : simpleInferExprType(initializer);
1412         if (t == null) {
1413           t = JSType.UNKNOWN;
1414         }
1415         ns.addUndeclaredProperty(pname, declNode, t, false);
1416       }
1417     }

1419     private void visitClassPropertyDeclaration(Node getProp) {
1420       Preconditions.checkArgument(getProp.isGetProp());
1421       NominalType thisType = currentScope.getDeclaredFunctionType().getThisType();
1422       if (thisType == null) {
1423         // This will get caught in NewTypeInference
1424         return;
1425       }
1426       RawNominalType rawNominalType = thisType.getRawNominalType();
1427       String pname = getProp.getLastChild().getString();
1428       // TODO(blickly): Support @param, @return style fun declarations here.
1429       JSType declType = getDeclaredTypeOfNode(
1430           NodeUtil.getBestJSDocInfo(getProp), currentScope);
1431       boolean isConst = isConst(getProp);
1432       if (declType != null || isConst) {
1433         mayWarnAboutExistingProp(rawNominalType, pname, getProp, declType);
1434         // Intentionally, we keep going even if we warned for redeclared prop.
1435         // The reason is that if a prop is defined on a class and on its proto
1436         // with conflicting types, we prefer the type of the class.
1437         if (declType == null) {
1438           declType = mayInferFromRhsIfConst(getProp);
1439         }
1440         if (mayAddPropToType(getProp, rawNominalType)) {
1441           rawNominalType.addClassProperty(pname, getProp, declType, isConst);
1442         }
1443         if (isConst) {
1444           getProp.putBooleanProp(Node.CONSTANT_PROPERTY_DEF, true);
1445         }
1446       } else if (mayAddPropToType(getProp, rawNominalType)) {
1447         rawNominalType.addUndeclaredClassProperty(pname, getProp);
1448       }
1449       propertyDefs.put(rawNominalType, pname,
1450           new PropertyDef(getProp, null, null));
1451     }

1453     private void visitOtherPropertyDeclaration(Node getProp) {
1454       Preconditions.checkArgument(getProp.isGetProp());
1455       Preconditions.checkArgument(getProp.isQualifiedName());
1456       if (isAnnotatedAsConst(getProp)) {
1457         warnings.add(JSError.make(getProp, MISPLACED_CONST_ANNOTATION));
1458       }
1459       QualifiedName recvQname = QualifiedName.fromNode(getProp.getFirstChild());
1460       Declaration d = this.currentScope.getDeclaration(recvQname, false);
1461       if (d == null) {
1462         return;
1463       }
1464       if (d.getTypedef() != null) {
1465         warnings.add(JSError.make(getProp, CANNOT_ADD_PROPERTIES_TO_TYPEDEF));
1466         getProp.getParent().putBooleanProp(Node.ANALYZED_DURING_GTI, true);
1467         return;
1468       }
1469       JSType recvType = d.getTypeOfSimpleDecl();
1470       if (recvType == null) {
1471         return;
1472       }
1473       NominalType nt = recvType.getNominalTypeIfSingletonObj();
1474       // Don't add stray properties to Object.
1475       if (nt == null || nt.equals(commonTypes.getObjectType())) {
1476         return;
1477       }
1478       RawNominalType rawType = nt.getRawNominalType();
1479       String pname = getProp.getLastChild().getString();
1480       JSType declType = getDeclaredTypeOfNode(
1481           NodeUtil.getBestJSDocInfo(getProp), currentScope);
1482       if (declType != null) {
1483         declType = declType.substituteGenericsWithUnknown();
1484         if (mayWarnAboutExistingProp(rawType, pname, getProp, declType)) {
1485           return;
1486         }
1487         rawType.addPropertyWhichMayNotBeOnAllInstances(pname, declType);
1488       } else if (!rawType.mayHaveProp(pname)) {
1489         rawType.addPropertyWhichMayNotBeOnAllInstances(pname, null);
1490       }
1491     }

1549     private JSType simpleInferExprType(Node n) {
1550       switch (n.getType()) {
1551         case Token.REGEXP:
1552           return commonTypes.getRegexpType();
1553         case Token.ARRAYLIT: {
1554           if (!n.hasChildren()) {
1555             return null;
1556           }
1557           Node child = n.getFirstChild();
1558           JSType arrayType = simpleInferExprType(child);
1559           if (arrayType == null) {
1560             return null;
1561           }
1562           while (null != (child = child.getNext())) {
1563             if (!arrayType.equals(simpleInferExprType(child))) {
1564               return null;
1565             }
1566           }
1567           return commonTypes.getArrayInstance(arrayType);
1568         }
1569         case Token.TRUE:
1570           return JSType.TRUE_TYPE;
1571         case Token.FALSE:
1572           return JSType.FALSE_TYPE;
1573         case Token.THIS:
1574           return this.currentScope.getDeclaredTypeOf("this");
1575         case Token.NAME:
1576           return simpleInferDeclaration(
1577               this.currentScope.getDeclaration(n.getString(), false));
1578         case Token.OBJECTLIT: {
1579           JSType objLitType = JSType.TOP_OBJECT;
1580           for (Node prop : n.children()) {
1581             JSType propType = simpleInferExprType(prop.getFirstChild());
1582             if (propType == null) {
1583               return null;
1584             }
1585             objLitType = objLitType.withProperty(
1586                 new QualifiedName(NodeUtil.getObjectLitKeyName(prop)),
1587                 propType);
1588           }
1589           return objLitType;
1590         }
1591         case Token.GETPROP:
1592           return simpleInferGetpropType(n);
1593         case Token.COMMA:
1594         case Token.ASSIGN:
1595           return simpleInferExprType(n.getLastChild());
1596         case Token.CALL:
1597         case Token.NEW: {
1598           Node callee = n.getFirstChild();
1599           // We special-case the function goog.getMsg, which is used by the
1600           // compiler for i18n.
1601           if (callee.matchesQualifiedName("goog.getMsg")) {
1602             return JSType.STRING;
1603           }
1604           FunctionType funType = simpleInferFunctionType(callee);
1605           if (funType == null) {
1606             return null;
1607           }
1608           if (funType.isGeneric()) {
1609             ImmutableList.Builder<JSType> argTypes = ImmutableList.builder();
1610             for (Node argNode = n.getFirstChild().getNext();
1611                  argNode != null;
1612                  argNode = argNode.getNext()) {
1613               JSType t = simpleInferExprType(argNode);
1614               if (t == null) {
1615                 return null;
1616               }
1617               argTypes.add(t);
1618             }
1619             funType = funType
1620                 .instantiateGenericsFromArgumentTypes(argTypes.build());
1621             if (funType == null) {
1622               return null;
1623             }
1624           }
1625           JSType retType =
1626               n.isNew() ? funType.getThisType() : funType.getReturnType();
1627           return retType;
1628         }
1629         default:
1630           switch (NodeUtil.getKnownValueType(n)) {
1631             case NULL:
1632               return JSType.NULL;
1633             case VOID:
1634               return JSType.UNDEFINED;
1635             case NUMBER:
1636               return JSType.NUMBER;
1637             case STRING:
1638               return JSType.STRING;
1639             case BOOLEAN:
1640               return JSType.BOOLEAN;
1641             case UNDETERMINED:
1642             default:
1643               return null;
1644           }
1645       }
1646     }

1721     // All suppressions happen in SuppressDocWarningsGuard.java, except one.
1722     // At a duplicate property definition annotated with @suppress {duplicate},
1723     // if the type in the jsdoc is the same as the already declared type,
1724     // then don't warn.
1725     // Type info is required to enforce this, so the current type inference
1726     // does it in TypeValidator.java, and we do it here.
1727     // This is a hacky suppression.
1728     // 1) Why is it just specific to "duplicate" and to properties?
1729     // 2) The docs say that it's only allowed in the top level, but the code
1730     //    allows it in all scopes.
1731     //    https://github.com/google/closure-compiler/wiki/Warnings#suppress-tags
1732     // For now, we implement it b/c it exists in the current type inference.
1733     // But I wouldn't mind if we stopped supporting it.
1734     private boolean suppressDupPropWarning(
1735         JSDocInfo propCreationJsdoc, JSType typeInJsdoc, JSType previousType) {
1736       if (propCreationJsdoc == null
1737           || !propCreationJsdoc.getSuppressions().contains("duplicate")) {
1738         return false;
1739       }
1740       return typeInJsdoc != null && previousType != null
1741           && typeInJsdoc.equals(previousType);

1744     private DeclaredFunctionType computeFnDeclaredType(
1745         JSDocInfo fnDoc, String functionName, Node declNode,
1746         RawNominalType ownerType, NTIScope parentScope) {
1747       Preconditions.checkArgument(declNode.isFunction() || declNode.isGetProp());
1748 
1749       // For an unannotated function, check if we can grab a type signature for
1750       // it from the surrounding code where it appears.
1751       if (fnDoc == null && !NodeUtil.functionHasInlineJsdocs(declNode)) {
1752         DeclaredFunctionType t = getDeclaredFunctionTypeFromContext(
1753             functionName, declNode, parentScope);
1754         if (t != null) {
1755           return t;
1756         }
1757       }
1758       // TODO(dimvar): warn if multiple jsdocs for a fun
1759       RawNominalType ctorType =
1760           declNode.isFunction() ? nominaltypesByNode.get(declNode) : null;
1761       FunctionAndSlotType result = typeParser.getFunctionType(
1762           fnDoc, functionName, declNode, ctorType, ownerType, parentScope);
1763       Node qnameNode = declNode.isGetProp() ? declNode : NodeUtil.getFunctionNameNode(declNode);
1764       if (result.slotType != null && qnameNode != null && qnameNode.isName()) {
1765         parentScope.addSimpleType(qnameNode, result.slotType);
1766       }
1767       if (ctorType != null) {
1768         ctorType.setCtorFunction(result.functionType.toFunctionType(), commonTypes);
1769       }
1770       if (declNode.isFunction()) {
1771         maybeWarnFunctionDeclaration(declNode, result.functionType);
1772       }
1773       return result.functionType;
1774     }

1776     private void maybeWarnFunctionDeclaration(Node funNode, DeclaredFunctionType funType) {
1777       if (funNode.getParent().isSetterDef()) {
1778         JSType returnType = funType.getReturnType();
1779         if (returnType != null && !returnType.isUnknown() && !returnType.isUndefined()) {
1780           warnings.add(JSError.make(funNode, SETTER_WITH_RETURN));
1781         }
1782       }
1783       int declaredArity = funType.getOptionalArity();
1784       int parameterCount = funNode.getFirstChild().getNext().getChildCount();
1785       if (!funType.hasRestFormals() && parameterCount != declaredArity) {
1786         warnings.add(JSError.make(funNode, WRONG_PARAMETER_COUNT,
1787             String.valueOf(declaredArity), String.valueOf(parameterCount)));
1788       }
1789     }

1791     // We only return a non-null result if the arity of declNode matches the
1792     // arity we get from declaredTypeAsJSType.
1793     private DeclaredFunctionType computeFnDeclaredTypeFromCallee(
1794         Node declNode, JSType declaredTypeAsJSType) {
1795       Preconditions.checkArgument(declNode.isFunction());
1796       Preconditions.checkArgument(declNode.getParent().isCall());
1797       Preconditions.checkNotNull(declaredTypeAsJSType);
1798 
1799       FunctionType funType = declaredTypeAsJSType.getFunType();
1800       if (funType == null || funType.isConstructor() || funType.isInterfaceDefinition()) {
1801         return null;
1802       }
1803       DeclaredFunctionType declType = funType.toDeclaredFunctionType();
1804       if (declType == null) {
1805         return null;
1806       }
1807       int numFormals = declNode.getChildAtIndex(1).getChildCount();
1808       int reqArity = declType.getRequiredArity();
1809       int optArity = declType.getOptionalArity();
1810       boolean hasRestFormals = declType.hasRestFormals();
1811       if (reqArity == optArity && !hasRestFormals) {
1812         return numFormals == reqArity ? declType : null;
1813       }
1814       if (numFormals == optArity && !hasRestFormals
1815           || numFormals == (optArity + 1) && hasRestFormals) {
1816         return declType;
1817       }
1818       return null;
1819     }

1821     // Returns null if it can't find a suitable type in the context
1822     private DeclaredFunctionType getDeclaredFunctionTypeFromContext(
1823         String functionName, Node declNode, NTIScope parentScope) {
1824       Node parent = declNode.getParent();
1825       Node maybeBind = parent.isCall() ? parent.getFirstChild() : parent;
1826 
1827       // The function literal is used with .bind or goog.bind
1828       if (NodeUtil.isFunctionBind(maybeBind) && !NodeUtil.isGoogPartial(maybeBind)) {
1829         Node call = maybeBind.getParent();
1830         Bind bindComponents = convention.describeFunctionBind(call, true, false);
1831         JSType recvType = simpleInferExprType(bindComponents.thisValue);
1832         if (recvType == null) {
1833           return null;
1834         }
1835         // Use typeParser for the formals, and only add the receiver type here.
1836         DeclaredFunctionType allButRecvType = typeParser.getFunctionType(
1837             null, functionName, declNode, null, null, parentScope).functionType;
1838         return allButRecvType.withReceiverType(recvType.getNominalTypeIfSingletonObj());
1839       }
1840 
1841       // The function literal is an argument at a call
1842       if (parent.isCall() && declNode != parent.getFirstChild()) {
1843         DeclaredFunctionType calleeDeclType = getDeclaredFunctionTypeOfCalleeIfAny(
1844             parent.getFirstChild(), parentScope);
1845         if (calleeDeclType != null && !calleeDeclType.isGeneric()) {
1846           int index = parent.getIndexOfChild(declNode) - 1;
1847           JSType declTypeFromCallee = calleeDeclType.getFormalType(index);
1848           if (declTypeFromCallee != null) {
1849             DeclaredFunctionType t =
1850                 computeFnDeclaredTypeFromCallee(declNode, declTypeFromCallee);
1851             if (t != null) {
1852               return t;
1853             }
1854           }
1855         }
1856       }
1857 
1858       return null;
1859     }

1861     private JSType getVarTypeFromAnnotation(Node nameNode) {
1862       Preconditions.checkArgument(nameNode.getParent().isVar());
1863       Node varNode = nameNode.getParent();
1864       JSType varType =
1865           getDeclaredTypeOfNode(varNode.getJSDocInfo(), currentScope);
1866       if (varNode.getChildCount() > 1 && varType != null) {
1867         warnings.add(JSError.make(varNode, TypeCheck.MULTIPLE_VAR_DEF));
1868       }
1869       String varName = nameNode.getString();
1870       JSType nameNodeType =
1871           getDeclaredTypeOfNode(nameNode.getJSDocInfo(), currentScope);
1872       if (nameNodeType != null) {
1873         if (varType != null) {
1874           warnings.add(JSError.make(nameNode, DUPLICATE_JSDOC, varName));
1875         }
1876         return nameNodeType;
1877       } else {
1878         return varType;
1879       }

1882     /**
1883      * Called for the usual style of prototype-property definitions,
1884      * but also for @lends and for direct assignments of object literals to prototypes.
1885      */
1886     private void mayAddPropToPrototype(
1887         RawNominalType rawType, String pname, Node defSite, Node initializer) {
1888       NTIScope methodScope = null;
1889       DeclaredFunctionType methodType = null;
1890       JSType propDeclType = null;
1891 
1892       JSDocInfo jsdoc = NodeUtil.getBestJSDocInfo(defSite);
1893       if (initializer != null && initializer.isFunction()) {
1894         methodScope = visitFunctionLate(initializer, rawType);
1895         methodType = methodScope.getDeclaredFunctionType();
1896       } else if (jsdoc != null && jsdoc.containsFunctionDeclaration()) {
1897         // We're parsing a function declaration without a function initializer
1898         methodType = computeFnDeclaredType(jsdoc, pname, defSite, rawType, currentScope);
1899       }
1900 
1901       // Find the declared type of the property.
1902       if (jsdoc != null && jsdoc.hasType()) {
1903         propDeclType = typeParser.getDeclaredTypeOfNode(jsdoc, rawType, currentScope);
1904       } else if (methodType != null) {
1905         propDeclType = commonTypes.fromFunctionType(methodType.toFunctionType());
1906       }
1907       propertyDefs.put(rawType, pname, new PropertyDef(defSite, methodType, methodScope));
1908 
1909       // Add the property to the class with the appropriate type.
1910       boolean isConst = isConst(defSite);
1911       if (propDeclType != null || isConst) {
1912         if (mayWarnAboutExistingProp(rawType, pname, defSite, propDeclType)) {
1913           return;
1914         }
1915         if (propDeclType == null) {
1916           propDeclType = mayInferFromRhsIfConst(defSite);
1917         }
1918         rawType.addProtoProperty(pname, defSite, propDeclType, isConst);
1919         if (defSite.isGetProp()) { // Don't bother saving for @lends
1920           defSite.putBooleanProp(Node.ANALYZED_DURING_GTI, true);
1921           if (isConst) {
1922             defSite.putBooleanProp(Node.CONSTANT_PROPERTY_DEF, true);
1923           }
1924         }
1925       } else {
1926         rawType.addUndeclaredProtoProperty(pname, defSite);
1927       }
1928     }

1957   private DeclaredFunctionType getDeclaredFunctionTypeOfCalleeIfAny(
1958       Node fn, NTIScope currentScope) {
1959     Preconditions.checkArgument(fn.getParent().isCall());
1960     if (fn.isThis() || !fn.isFunction() && !fn.isQualifiedName()) {
1961       return null;
1962     }
1963     if (fn.isFunction()) {
1964       return currentScope.getScope(getFunInternalName(fn)).getDeclaredFunctionType();
1965     }
1966     Preconditions.checkState(fn.isQualifiedName());
1967     Declaration decl = currentScope.getDeclaration(QualifiedName.fromNode(fn), false);
1968     if (decl == null) {
1969       return null;
1970     }
1971     if (decl.getFunctionScope() != null) {
1972       return decl.getFunctionScope().getDeclaredFunctionType();
1973     }
1974     if (decl.getTypeOfSimpleDecl() != null) {
1975       FunctionType funType = decl.getTypeOfSimpleDecl().getFunType();
1976       if (funType != null) {
1977         return funType.toDeclaredFunctionType();
1978       }
1979     }
1980     return null;
1981   }
