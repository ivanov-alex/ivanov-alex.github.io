53  /**
54   * The object type represents instances of JavaScript objects such as
55   * {@code Object}, {@code Date}, {@code Function}.<p>
56   *
57   * Objects in JavaScript are unordered collections of properties.
58   * Each property consists of a name, a value and a set of attributes.<p>
59   *
60   * Each instance has an implicit prototype property ({@code [[Prototype]]})
61   * pointing to an object instance, which itself has an implicit property, thus
62   * forming a chain.<p>
63   *
64   * A class begins life with no name.  Later, a name may be provided once it
65   * can be inferred.  Note that the name in this case is strictly for
66   * debugging purposes.  Looking up type name references goes through the
67   * {@link JSTypeRegistry}.<p>
68   */
69  public class PrototypeObjectType extends ObjectType {
70    private static final long serialVersionUID = 1L;
71  
72    private final String className;
73    private final PropertyMap properties;
74    private final boolean nativeType;
75    private final boolean anonymousType;
76  
77    // NOTE(nicksantos): The implicit prototype can change over time.
78    // Modeling this is a bear. Always call getImplicitPrototype(), because
79    // some subclasses override this to do special resolution handling.
80    private ObjectType implicitPrototypeFallback;

82    // If this is a function prototype, then this is the owner.
83    // A PrototypeObjectType can only be the prototype of one function. If we try
84    // to do this for multiple functions, then we'll have to create a new one.
85    private FunctionType ownerFunction = null;

154   /**
155    * Creates an object type, allowing specification of the implicit prototype,
156    * whether the object is native, and any templatized types.
157    */
158   PrototypeObjectType(JSTypeRegistry registry, String className,
159       ObjectType implicitPrototype, boolean nativeType,
160       TemplateTypeMap templateTypeMap, boolean anonymousType) {
161     super(registry, templateTypeMap);
162     this.properties = new PropertyMap();
163     this.properties.setParentSource(this);
164 
165     this.className = className;
166     this.nativeType = nativeType;
167     this.anonymousType = anonymousType;
168     if (nativeType || implicitPrototype != null) {
169       setImplicitPrototype(implicitPrototype);
170     } else {
171       setImplicitPrototype(
172           registry.getNativeObjectType(JSTypeNative.OBJECT_TYPE));
173     }
174   }

181   @Override
182   boolean defineProperty(String name, JSType type, boolean inferred,
183       Node propertyNode) {
184     if (hasOwnDeclaredProperty(name)) {
185       return false;
186     }
187     Property newProp = new Property(
188         name, type, inferred, propertyNode);
189     properties.putProperty(name, newProp);
190     return true;

198   @Override
199   public void setPropertyJSDocInfo(String propertyName, JSDocInfo info) {
200     if (info != null) {
201       if (properties.getOwnProperty(propertyName) == null) {
202         // If docInfo was attached, but the type of the property
203         // was not defined anywhere, then we consider this an explicit
204         // declaration of the property.
205         defineInferredProperty(propertyName, getPropertyType(propertyName),
206             null);
207       }
208 
209       // The prototype property is not represented as a normal Property.
210       // We probably don't want to attach any JSDoc to it anyway.
211       Property property = properties.getOwnProperty(propertyName);
212       if (property != null) {
213         property.setJSDocInfo(info);
214       }
215     }
216   }

239   /**
240    * Given the name of a native object property, checks whether the property is
241    * present on the object and different from the native one.
242    */
243   private boolean hasOverridenNativeProperty(String propertyName) {
244     if (isNativeObjectType()) {
245       return false;
246     }
247 
248     JSType propertyType = getPropertyType(propertyName);
249     ObjectType nativeType =
250         isFunctionType() ?
251         registry.getNativeObjectType(JSTypeNative.FUNCTION_PROTOTYPE) :
252         registry.getNativeObjectType(JSTypeNative.OBJECT_PROTOTYPE);
253     JSType nativePropertyType = nativeType.getPropertyType(propertyName);
254     return propertyType != nativePropertyType;
255   }

280   @Override
281   String toStringHelper(boolean forAnnotations) {
282     if (hasReferenceName()) {
283       return getReferenceName();
284     } else if (prettyPrint) {
285       // Don't pretty print recursively.
286       prettyPrint = false;
287 
288       // Use a tree set so that the properties are sorted.
289       Set<String> propertyNames = new TreeSet<>();
290       for (ObjectType current = this;
291            current != null && !current.isNativeObjectType() &&
292                propertyNames.size() <= MAX_PRETTY_PRINTED_PROPERTIES;
293            current = current.getImplicitPrototype()) {
294         propertyNames.addAll(current.getOwnPropertyNames());
295       }
296 
297       StringBuilder sb = new StringBuilder();
298       sb.append("{");
299 
300       int i = 0;
301       for (String property : propertyNames) {
302         if (i > 0) {
303           sb.append(", ");
304         }
305 
306         sb.append(property);
307         sb.append(": ");
308         sb.append(getPropertyType(property).toStringHelper(forAnnotations));
309 
310         ++i;
311         if (!forAnnotations && i == MAX_PRETTY_PRINTED_PROPERTIES) {
312           sb.append(", ...");
313           break;
314         }
315       }
316 
317       sb.append("}");
318 
319       prettyPrint = true;
320       return sb.toString();
321     } else {
322       return forAnnotations ? "?" : "{...}";
323     }
324   }

380   @Override
381   protected boolean isSubtype(JSType that,
382       ImplCache implicitImplCache) {
383     if (JSType.isSubtypeHelper(this, that, implicitImplCache)) {
384       return true;
385     }
386 
387     // Union types
388     if (that.isUnionType()) {
389       // The static {@code JSType.isSubtype} check already decomposed
390       // union types, so we don't need to check those again.
391       return false;
392     }
393 
394     // record types
395     if (that.isRecordType()) {
396       return RecordType.isSubtype(this, that.toMaybeRecordType(), implicitImplCache);
397     }
398 
399     // Interfaces
400     // Find all the interfaces implemented by this class and compare each one
401     // to the interface instance.
402     ObjectType thatObj = that.toObjectType();
403     FunctionType thatCtor = thatObj == null ? null : thatObj.getConstructor();
404 
405     if (getConstructor() != null && getConstructor().isInterface()) {
406       for (ObjectType thisInterface : getCtorExtendedInterfaces()) {
407         if (thisInterface.isSubtype(that, implicitImplCache)) {
408           return true;
409         }
410       }
411     } else if (thatCtor != null && thatCtor.isInterface()) {
412       Iterable<ObjectType> thisInterfaces = getCtorImplementedInterfaces();
413       for (ObjectType thisInterface : thisInterfaces) {
414         if (thisInterface.isSubtype(that, implicitImplCache)) {
415           return true;
416         }
417       }
418     }
419 
420     // other prototype based objects
421     if (isUnknownType() || implicitPrototypeChainIsUnknown()) {
422       // If unsure, say 'yes', to avoid spurious warnings.
423       // TODO(user): resolve the prototype chain completely in all cases,
424       // to avoid guessing.
425       return true;
426     }
427     return thatObj != null && isImplicitPrototype(thatObj);
428   }
429 
430   private boolean implicitPrototypeChainIsUnknown() {
431     ObjectType p = getImplicitPrototype();
432     while (p != null) {
433       if (p.isUnknownType()) {
434         return true;
435       }
436       p = p.getImplicitPrototype();
437     }
438     return false;
439   }

477   @Override
478   JSType resolveInternal(ErrorReporter t, StaticTypedScope<JSType> scope) {
479     setResolvedTypeInternal(this);
480 
481     ObjectType implicitPrototype = getImplicitPrototype();
482     if (implicitPrototype != null) {
483       implicitPrototypeFallback =
484           (ObjectType) implicitPrototype.resolve(t, scope);
485       FunctionType ctor = getConstructor();
486       if (ctor != null) {
487         FunctionType superCtor = ctor.getSuperClassConstructor();
488         if (superCtor != null) {
489           // If the super ctor of this prototype object was not known before resolution, then the
490           // subTypes would not have been set. Update them.
491           superCtor.addSubTypeIfNotPresent(ctor);
492         }
493       }
494     }
495     for (Property prop : properties.values()) {
496       prop.setType(safeResolve(prop.getType(), t, scope));
497     }
498     return this;
499   }

501   @Override
502   public void matchConstraint(JSType constraint) {
503     // We only want to match constraints on anonymous types.
504     if (hasReferenceName()) {
505       return;
506     }
507 
508     // Handle the case where the constraint object is a record type.
509     //
510     // param constraint {{prop: (number|undefined)}}
511     // function f(constraint) {}
512     // f({});
513     //
514     // We want to modify the object literal to match the constraint, by
515     // taking any each property on the record and trying to match
516     // properties on this object.
517     if (constraint.isRecordType()) {
518       matchRecordTypeConstraint(constraint.toObjectType());
519     } else if (constraint.isUnionType()) {
520       for (JSType alt : constraint.toMaybeUnionType().getAlternates()) {
521         if (alt.isRecordType()) {
522           matchRecordTypeConstraint(alt.toObjectType());
523         }
524       }
525     }
526   }
527 
528   public void matchRecordTypeConstraint(ObjectType constraintObj) {
529     for (String prop : constraintObj.getOwnPropertyNames()) {
530       JSType propType = constraintObj.getPropertyType(prop);
531       if (!isPropertyTypeDeclared(prop)) {
532         JSType typeToInfer = propType;
533         if (!hasProperty(prop)) {
534           typeToInfer = getNativeType(JSTypeNative.VOID_TYPE)
535               .getLeastSupertype(propType);
536         }
537         defineInferredProperty(prop, typeToInfer, null);
538       }
539     }
