83    NTIScope(Node root, NTIScope parent, List<String> formals, JSTypes commonTypes) {
84      if (parent == null) {
85        this.name = null;
86        this.externs = new LinkedHashMap<>();
87      } else {
88        String nameOnAst = root.getFirstChild().getString();
89        this.name = nameOnAst.isEmpty() ? null : nameOnAst;
90        this.externs = ImmutableMap.of();
91      }
92      this.root = root;
93      this.parent = parent;
94      this.formals = formals;
95      this.commonTypes = commonTypes;
96    }

182   boolean isFunctionNamespace(String name) {
183     Preconditions.checkArgument(!name.contains("."));
184     Preconditions.checkState(isFinalized);
185     Declaration d = getDeclaration(name, false);
186     if (d == null || d.getFunctionScope() == null || d.getTypeOfSimpleDecl() == null) {
187       return false;
188     }
189     return d.getTypeOfSimpleDecl().getObjTypeIfSingletonObj() != null;

192   // In other languages, type names and variable names are in distinct
193   // namespaces and don't clash.
194   // But because our typedefs and enums are var declarations, they are in the
195   // same namespace as other variables.
196   boolean isDefinedLocally(String name, boolean includeTypes) {
197     Preconditions.checkNotNull(name);
198     Preconditions.checkState(!name.contains("."));
199     if (locals.containsKey(name)
200         || formals.contains(name)
201         || localClassDefs.containsKey(name)
202         || localFunDefs.containsKey(name)
203         || "this".equals(name)
204         || externs.containsKey(name)
205         || localNamespaces.containsKey(name)
206         || localTypedefs.containsKey(name)
207         || localEnums.containsKey(name)) {
208       return true;
209     }
210     if (includeTypes) {
211       return unknownTypeNames.contains(name)
212           || declaredType != null && declaredType.isTypeVariableDefinedLocally(name);
213     }
214     return false;
215   }

217   boolean isDefined(Node qnameNode) {
218     Preconditions.checkArgument(qnameNode.isQualifiedName());
219     if (qnameNode.isName()) {
220       return isDefinedLocally(qnameNode.getString(), false);
221     } else if (qnameNode.isThis()) {
222       return true;
223     }
224     QualifiedName qname = QualifiedName.fromNode(qnameNode);
225     String leftmost = qname.getLeftmostName();
226     if (isNamespace(leftmost)) {
227       return getNamespace(leftmost).isDefined(qname.getAllButLeftmost());
228     }
229     return parent == null ? false : parent.isDefined(qnameNode);
230   }

315   @Override
316   public JSType getDeclaredTypeOf(String name) {
317     Preconditions.checkArgument(!name.contains("."));
318     if ("this".equals(name)) {
319       if (!hasThis()) {
320         return null;
321       }
322       return getDeclaredFunctionType().getThisType().getInstanceAsJSType();
323     }
324     Declaration decl = getLocalDeclaration(name, false);
325     if (decl != null) {
326       if (decl.getTypeOfSimpleDecl() != null) {
327         Preconditions.checkState(!decl.getTypeOfSimpleDecl().isBottom(),
328             "%s was bottom", name);
329         return decl.getTypeOfSimpleDecl();
330       }
331       NTIScope funScope = (NTIScope) decl.getFunctionScope();
332       if (funScope != null) {
333         return getCommonTypes().fromFunctionType(
334             funScope.getDeclaredFunctionType().toFunctionType());
335       }
336       Preconditions.checkState(decl.getNamespace() == null);
337       return null;
338     }
339     // When a function is a namespace, the parent scope has a better type.
340     if (name.equals(this.name) && !parent.isFunctionNamespace(name)) {
341       return getCommonTypes()
342           .fromFunctionType(getDeclaredFunctionType().toFunctionType());
343     }
344     if (parent != null) {
345       return parent.getDeclaredTypeOf(name);
346     }
347     return null;
348   }

350   boolean hasUndeclaredFormalsOrOuters() {
351     for (String formal : formals) {
352       if (getDeclaredTypeOf(formal) == null) {
353         return true;
354       }
355     }
356     for (String outer : outerVars) {
357       JSType declType = getDeclaredTypeOf(outer);
358       if (declType == null
359           // Undeclared functions have a non-null declared type,
360           //  but they always have a return type of unknown
361           || (declType.getFunType() != null
362               && declType.getFunType().getReturnType().isUnknown())) {
363         return true;
364       }
365     }
366     return false;
367   }

431   private void addNamespaceLit(Node qnameNode, NamespaceLit nslit) {
432     Preconditions.checkArgument(!isNamespace(qnameNode));
433     if (qnameNode.isName()) {
434       String varName = qnameNode.getString();
435       localNamespaces.put(varName, nslit);
436       if (qnameNode.isFromExterns()) {
437         // We don't know the full type of a namespace until after we see all
438         // its properties. But we want to add it to the externs, otherwise it
439         // is treated as a local and initialized to the wrong thing in NTI.
440         externs.put(qnameNode.getString(), null);
441       }
442     } else {
443       QualifiedName qname = QualifiedName.fromNode(qnameNode);
444       Namespace ns = getNamespace(qname.getLeftmostName());
445       ns.addSubnamespace(qname.getAllButLeftmost(), nslit);
446     }
447   }

450     if (isDefinedLocally(name, false)) {
451       locals.put(name, newDeclType);

454     } else {
455       throw new RuntimeException(
456           "Cannot update type of unknown variable: " + name);
457     }

536   private Declaration getLocalDeclaration(String name, boolean includeTypes) {
537     Preconditions.checkArgument(!name.contains("."));
538     if (!isDefinedLocally(name, includeTypes)) {
539       return null;
540     }
541     JSType type = null;
542     boolean isForwardDeclaration = false;
543     boolean isTypeVar = false;
544     if (locals.containsKey(name)) {
545       type = locals.get(name);
546     } else if (formals.contains(name)) {
547       int formalIndex = formals.indexOf(name);
548       if (declaredType != null && formalIndex != -1) {
549         JSType formalType = declaredType.getFormalType(formalIndex);
550         if (formalType != null && !formalType.isBottom()) {
551           type = formalType;
552         }
553       }
554     } else if (localFunDefs.containsKey(name)) {
555       // After finalization, the externs contain the correct type for
556       // external function namespaces, don't rely on localFunDefs
557       if (isFinalized && externs.containsKey(name)) {
558         type = externs.get(name);
559       }
560     } else if (localTypedefs.containsKey(name) || localNamespaces.containsKey(name)
561         || localEnums.containsKey(name) || localClassDefs.containsKey(name)) {
562       // Any further declarations are shadowed
563     } else if (declaredType != null && declaredType.isTypeVariableDefinedLocally(name)) {
564       isTypeVar = true;
565       type = JSType.fromTypeVar(name);
566     } else if (externs.containsKey(name)) {
567       type = externs.get(name);
568     } else if (unknownTypeNames.contains(name)) {
569       isForwardDeclaration = true;
570     }
571     return new Declaration(
572         type,
573         localTypedefs.get(name),
574         localNamespaces.get(name),
575         localEnums.get(name),
576         localFunDefs.get(name),
577         localClassDefs.get(name),
578         isTypeVar,
579         constVars.contains(name),
580         isForwardDeclaration);
581   }

583   public Declaration getDeclaration(QualifiedName qname, boolean includeTypes) {
584     if (qname.isIdentifier()) {
585       return getDeclaration(qname.getLeftmostName(), includeTypes);
586     }
587     Preconditions.checkState(!this.isFinalized,
588         "Namespaces are removed from scopes after finalization");
589     Namespace ns = getNamespace(qname.getLeftmostName());
590     if (ns == null) {
591       return null;
592     }
593     Declaration decl = ns.getDeclaration(qname.getAllButLeftmost());
594     if (decl == null && unknownTypeNames.contains(qname.toString())) {
595       return new Declaration(
596           JSType.UNKNOWN, null, null, null, null, null, false, false, true);
597     }
598     return decl;
599   }

601   public Declaration getDeclaration(String name, boolean includeTypes) {
602     Preconditions.checkArgument(!name.contains("."));
603     Declaration decl = getLocalDeclaration(name, includeTypes);
604     if (decl != null) {
605       return decl;
606     }
607     return parent == null ? null : parent.getDeclaration(name, includeTypes);
608   }

618   void resolveEnums(JSTypeCreatorFromJSDoc typeParser) {
619     for (EnumType e : localEnums.values()) {
620       if (!e.isResolved()) {
621         typeParser.resolveEnum(e, this);
622       }
623     }
624     for (EnumType e : qualifiedEnums) {
625       if (!e.isResolved()) {
626         typeParser.resolveEnum(e, this);
627       }
628     }
629     qualifiedEnums = null;
630   }

632   void mayDeclareUnknownType(QualifiedName qname) {
633     if (qname.isIdentifier() || null == getNamespace(qname.getLeftmostName())) {
634       // TODO(dimvar): If the code before the return is deleted, no unit tests fail.
635       // That's because we consider the unknown type a global variable, which we type ?.
636       // Any reason to keep this code, or just delete it?
637       String name = qname.getLeftmostName();
638       if (!locals.containsKey(name)) {
639         externs.put(name, JSType.UNKNOWN);
640       }
641       return;
642     }
643     Namespace leftmost = getNamespace(qname.getLeftmostName());
644     QualifiedName props = qname.getAllButLeftmost();
645     // The forward declared type may be on an undeclared namespace.
646     // e.g. 'ns.Foo.Bar.Baz' when we don't even have a definition for ns.Foo.
647     // Thus, we need to find the prefix of the qname that is not declared.
648     while (!props.isIdentifier()
649         && !leftmost.hasSubnamespace(props.getAllButRightmost())) {
650       props = props.getAllButRightmost();
651     }
652     Namespace ns = props.isIdentifier()
653         ? leftmost : leftmost.getSubnamespace(props.getAllButRightmost());
654     String pname = props.getRightmostName();
655     ns.addUndeclaredProperty(pname, null, JSType.UNKNOWN, /* isConst */ false);
656   }

658   void removeTmpData() {
659     unknownTypeNames = ImmutableSet.of();
660     JSTypes commonTypes = getCommonTypes();
661     // For now, we put types of namespaces directly into the locals.
662     // Alternatively, we could move this into NewTypeInference.initEdgeEnvs
663     for (Map.Entry<String, NamespaceLit> entry : localNamespaces.entrySet()) {
664       String name = entry.getKey();
665       NamespaceLit nslit = entry.getValue();
666       JSType t = nslit.toJSType(commonTypes);
667       // If it's a function namespace, add the function type to the result
668       if (localFunDefs.containsKey(name)) {
669         t = t.withFunction(
670             localFunDefs.get(name).getDeclaredFunctionType().toFunctionType(),
671             commonTypes.getFunctionType());
672       }
673       if (externs.containsKey(name)) {
674         externs.put(name, t);
675       } else {
676         locals.put(name, t);
677       }
678     }
679     for (Map.Entry<String, EnumType> entry : localEnums.entrySet()) {
680       locals.put(entry.getKey(), entry.getValue().toJSType(commonTypes));
681     }
682     for (String typedefName : localTypedefs.keySet()) {
683       locals.put(typedefName, JSType.UNDEFINED);
684     }
685     for (Map.Entry<String, RawNominalType> entry : localClassDefs.entrySet()) {
686       String name = entry.getKey();
687       // localClassDefs may contain aliased nominal types; these aren't stored
688       // in localFunDefs like the other nominal types.
689       if (!localFunDefs.containsKey(name)) {
690         locals.put(name, entry.getValue().toJSType(commonTypes));
691       }
692     }
693     localNamespaces = ImmutableMap.of();
694     localClassDefs = ImmutableMap.of();
695     localTypedefs = ImmutableMap.of();
696     localEnums = ImmutableMap.of();
697     isFinalized = true;
698   }

700   @Override
701   public String toString() {
702     StringBuilder sb = new StringBuilder();
703     if (isTopLevel()) {
704       sb.append("<TOP SCOPE>");
705     } else {
706       sb.append(getReadableName());
707       sb.append('(');
708       Joiner.on(',').appendTo(sb, formals);
709       sb.append(')');
710     }
711     sb.append(" with root: ");
712     sb.append(root);
713     return sb.toString();
714   }
