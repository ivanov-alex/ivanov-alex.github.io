41  /**
42   * Check to ensure there exists a path to dispose of each eventful object
43   * created.
44   *
45   * <p>An eventful class is any class that derives from goog.events.EventHandler
46   * or (in aggressive mode) is disposable and disposes of an eventful class when
47   * it is disposed (see http://research.google.com/pubs/pub40738.html).
48   *
49   * <p>This pass is heuristic based and should not be used for any check
50   * of pass/fail testing. The pass traverses the AST and marks as errors
51   * cases where an eventful object is allocated but a dispose call is not found.
52   * It only tracks eventful objects that has a easily identifiable static name,
53   * i.e., objects assigned to arrays, returned from functions or captured in
54   * closures are not considered. It simply tries to see if there exists a call to
55   * a dispose method in the AST for every object seen as eventful.
56   *
57   * <p>This compiler pass uses the inferred types and hence either type checking or
58   * type inference needs to be enabled.
59   *
60   *
61   */
62   // TODO(user): Pass needs to be updated for listenable interfaces.
63  public final class CheckEventfulObjectDisposal implements CompilerPass {

179   /**
180    * Add a new call that is used to dispose an JS object.
181    * @param functionOrMethodName The name or suffix of a function or method
182    *  that disposes of/registers an object as disposable
183    * @param argumentsThatAreDisposed An array of integers (ideally sorted) that
184    *   specifies the arguments of the function being disposed
185    */
186   private void addDisposeCall(String functionOrMethodName,
187       List<Integer> argumentsThatAreDisposed) {
188     String potentiallyTypeName, propertyName;
189     JSType objectType = null;
190 
191     int lastPeriod = functionOrMethodName.lastIndexOf('.');
192     // If function call has a period it is potentially a method function.
193     if (lastPeriod >= 0) {
194       potentiallyTypeName = functionOrMethodName.substring(0, lastPeriod).
195         replaceFirst(".prototype$", "");
196       propertyName = functionOrMethodName.substring(lastPeriod);
197       objectType = this.typeRegistry.getType(potentiallyTypeName);
198     } else {
199       propertyName = functionOrMethodName;
200     }
201 
202     // Find or create property map for object type
203     Map<String, List<Integer>> map = this.disposeCalls.get(objectType);
204     if (map == null) {
205       map = new HashMap<>();
206       this.disposeCalls.put(objectType, map);
207     }
208 
209     /*
210      * If this is a static function call store the full function name,
211      * else only the method of the object.
212      */
213     if (objectType == null) {
214       map.put(functionOrMethodName, argumentsThatAreDisposed);
215     } else {
216       map.put(propertyName, argumentsThatAreDisposed);
217     }
218   }

221   /*
222    * Initialize disposeMethods map with calls to dispose calls.
223    */
224   private void initializeDisposeMethodsMap() {
225     this.disposeCalls = new HashMap<>();
226 
227     /*
228      * Initialize dispose calls map. Checks for:
229      *    - Y.registerDisposable(X)
230      *      (Y has to be of type goog.Disposable)
231      *    - X.dispose()
232      *    - goog.dispose(X)
233      *    - goog.disposeAll(X...)
234      *    - X.removeAll() (X is of type goog.events.EventHandler)
235      *    - goog.array.extend(_, X...)
236      *    - Y.add(X...) or Y.push(X)
237      */
238     this.addDisposeCall("goog.array.extend", ImmutableList.of(DISPOSE_ALL));
239     this.addDisposeCall("goog.dispose", ImmutableList.of(0));
240     this.addDisposeCall("goog.Disposable.registerDisposable", ImmutableList.of(0));
241     this.addDisposeCall("goog.disposeAll", ImmutableList.of(DISPOSE_ALL));
242     this.addDisposeCall("goog.events.EventHandler.removeAll", ImmutableList.of(DISPOSE_SELF));
243     this.addDisposeCall(".dispose", ImmutableList.of(DISPOSE_SELF));
244     this.addDisposeCall(".push", ImmutableList.of(0));
245     this.addDisposeCall(".add", ImmutableList.of(DISPOSE_SELF));
246   }

249   private static Node getBase(Node n) {
250     Node base = n;
251     while (base.isGetProp()) {
252       base = base.getFirstChild();
253     }
254 
255     return base;
256   }

273   /**
274    * Determines if thisType is possibly a subtype of thatType.
275    *
276    *  <p>It differs from isSubtype only in that thisType gets expanded
277    *  if it is a union.
278    *
279    *  <p>Common case targeted is a function returning an eventful object
280    *  that may also return a null.
281    *
282    *  @param thisType the JSType being tested
283    *  @param thatType the JSType that is possibly a base of thisType
284    *  @return whether thisType is possibly subtype of thatType
285    */
286   private static boolean isPossiblySubtype(JSType thisType, JSType thatType) {
287     if (thisType == null) {
288       return false;
289     }
290 
291     JSType type = thisType;
292 
293     if (type.isUnionType()) {
294       for (JSType alternate : type.toMaybeUnionType().getAlternates()) {
295         if (alternate.isSubtype(thatType)) {
296           return true;
297         }
298       }
299     } else {
300       if (type.isSubtype(thatType)) {
301         return true;
302       }
303     }
304 
305     return false;
306   }

308   private static JSType dereference(JSType type) {
309     return type == null ? null : type.dereference();
310   }

312   /*
313    * Create a unique identification string for Node n, or null if function
314    * called with invalid argument.
315    *
316    * This function is basically used to distinguish between:
317    *   A.B = function() {
318    *     this.eh = new ...
319    *   }
320    * and
321    *   C.D = function() {
322    *     this.eh = new ...
323    *   }
324    *
325    * As well as
326    *   A.B = function() {
327    *     var eh = new ...
328    *   }
329    * and
330    *   C.D = function() {
331    *     var eh = new ...
332    *   }
333    *
334    * Warning: Inheritance is not currently handled.
335    */
336   private static String generateKey(NodeTraversal t, Node n,
337       boolean noLocalVariables) {
338     if (n == null) {
339       return null;
340     }
341     String key;
342 
343     Node scopeNode = t.getScopeRoot();
344 
345     if (n.isName()) {
346       if (noLocalVariables) {
347         return null;
348       }
349       key = n.getQualifiedName();
350 
351       if (scopeNode.isFunction()) {
352         JSType parentScopeType = t.getTypedScope().getParentScope().getTypeOfThis();
353         /*
354          * If the locally defined variable is defined within a function, use
355          * the function name to create ID.
356          */
357         if (!parentScopeType.isGlobalThisType()) {
358           key = parentScopeType + "~" + key;
359         }
360         key = NodeUtil.getFunctionName(scopeNode) + "=" + key;
361       }
362     } else {
363       /*
364        * Only handle cases such as a.b.c.X and not cases where the
365        * eventful object is stored in an array or uses a function to
366        * determine the index.
367        *
368        * Note: Inheritance changes the name that should be returned here
369        */
370       if (!n.isQualifiedName()) {
371         return null;
372       }
373       key = n.getQualifiedName();
374 
375       /*
376        * If it is not a simple variable and doesn't use this, then we assume
377        * global variable.
378        */
379       Node base = getBase(n);
380       if (base != null && base.isThis()) {
381         if (base.getJSType().isUnknownType()) {
382           // Handle anonymous function created in constructor:
383           //
384           // /**
385           // * @extends {goog.SubDisposable}
386           // * @constructor */
387           // speel.Person = function() {
388           //  this.run = function() {
389           //    this.eh = new goog.events.EventHandler();
390           //  }
391           //};
392           key = t.getTypedScope().getParentScope().getTypeOfThis() + "~" + key;
393         } else {
394           if (n.getFirstChild() == null) {
395             key = base.getJSType() + "=" + key;
396           } else {
397             ObjectType objectType =
398                 ObjectType.cast(dereference(n.getFirstChild().getJSType()));
399             if (objectType == null) {
400               return null;
401             }
402 
403             ObjectType hObjT = objectType;
404             String propertyName = n.getLastChild().getString();
405 
406             while (objectType != null) {
407               hObjT = objectType;
408               objectType = objectType.getImplicitPrototype();
409               if (objectType == null) {
410                 break;
411               }
412               if (objectType.getDisplayName().endsWith("prototype")) {
413                 continue;
414               }
415               if (!objectType.getPropertyNames().contains(propertyName)) {
416                 break;
417               }
418             }
419             key = hObjT + "=" + key;
420           }
421         }
422       }
423     }
424 
425     return key;
426   }

428   @Override
429   public void process(Node externs, Node root) {
430     // This pass should not have gotten added in this case
431     Preconditions.checkArgument(checkingPolicy != DisposalCheckingPolicy.OFF);
432 
433     // Initialize types
434     googDisposableInterfaceType = this.typeRegistry.getType(DISPOSABLE_INTERFACE_TYPE_NAME);
435     googEventsEventHandlerType = this.typeRegistry.getType(EVENT_HANDLER_TYPE_NAME);
436 
437     /*
438      * Required types not found therefore the kind of pattern considered
439      * will not be found.
440      */
441     if (googEventsEventHandlerType == null || googDisposableInterfaceType == null) {
442       return;
443     }
444 
445     // Seed list of disposable stype
446     eventfulTypes = new HashSet<>();
447     eventfulTypes.add(googEventsEventHandlerType);
448 
449     // Construct eventizer graph
450     if (checkingPolicy == DisposalCheckingPolicy.AGGRESSIVE) {
451       NodeTraversal.traverseTyped(compiler, root, new ComputeEventizeTraversal());
452       computeEventful();
453     }
454 
455     /*
456      * eventfulObjectMap maps a eventful object's "name" to its corresponding
457      * EventfulObjectState which tracks the state (allocated, disposed of)
458      * as well as allocation site.
459      */
460     eventfulObjectMap = new HashMap<>();
461 
462     // Traverse tree
463     NodeTraversal.traverseTyped(compiler, root, new Traversal());
464 
465     /*
466      * Scan eventfulObjectMap for allocated eventful objects that
467      * had no dispose calls.
468      */
469     for (EventfulObjectState e : eventfulObjectMap.values()) {
470       Node n = e.allocationSite;
471       if (e.seen == SeenType.ALLOCATED) {
472         compiler.report(JSError.make(n, EVENTFUL_OBJECT_NOT_DISPOSED));
473       } else if (e.seen == SeenType.ALLOCATED_LOCALLY
474           && checkingPolicy == DisposalCheckingPolicy.AGGRESSIVE) {
475         compiler.report(JSError.make(n, EVENTFUL_OBJECT_PURELY_LOCAL));
476       }
477     }
478   }

480   private void computeEventful() {
481     /*
482      * Topological order of Eventize DAG
483      */
484     String[] order = new String[eventizes.size()];
485 
486     /*
487      * Perform topological sort
488      */
489     int white = 0, gray = 1, black = 2;
490     int last = eventizes.size() - 1;
491     Map<String, Integer> color = new HashMap<>();
492     Stack<String> dfsStack = new Stack<>();
493 
494     /*
495      * Initialize color.
496      * Some types are only on one or the other side of the
497      * inference.
498      */
499     for (Map.Entry<String, Set<String>> eventizesEntry : eventizes.entrySet()) {
500       color.put(eventizesEntry.getKey(), white);
501       for (String s : eventizesEntry.getValue()) {
502         color.put(s, white);
503       }
504     }
505 
506     int indx = 0;
507     for (String s : eventizes.keySet()) {
508       dfsStack.push(s);
509       while (!dfsStack.isEmpty()) {
510         String top = dfsStack.pop();
511         if (!color.containsKey(top)) {
512           continue;
513         }
514         if (color.get(top) == white) {
515           color.put(top, gray);
516           dfsStack.push(top);
517           // for v in Adj[s]
518           if (eventizes.containsKey(top)) {
519             for (String v : eventizes.get(top)) {
520               if (color.get(v) == white) {
521                 dfsStack.push(v);
522               }
523             }
524           }
525         } else if (color.get(top) == gray && eventizes.containsKey(top)) {
526           order[last - indx] = top;
527           ++indx;
528           color.put(top, black);
529         }
530       }
531     }
532 
533     /*
534      * Propagate eventfulness by iterating in topological order
535      */
536     for (String s : order) {
537       if (eventfulTypes.contains(typeRegistry.getType(s))) {
538         for (String v : eventizes.get(s)) {
539           eventfulTypes.add((JSType) typeRegistry.getType(v));
540         }
541       }
542     }
543   }

545   private JSType maybeReturnDisposedType(Node n, boolean checkDispose) {
546     /*
547      * Checks for:
548      *    - Y.registerDisposable(X)
549      *      (Y has to be of type goog.Disposable)
550      *    - X.dispose()
551      *    - goog.dispose(X)
552      *    - X.removeAll() (X is of type goog.events.EventHandler)
553      *    - <array>.property(X) or Y.push(X)
554      */
555     Node first = n.getFirstChild();
556 
557     if (first == null || !first.isQualifiedName()) {
558       return null;
559     }
560     String property = first.getQualifiedName();
561 
562     if (property.endsWith(".registerDisposable"))  {
563       /*
564        *  Ensure object is of type disposable
565        */
566       Node base = first.getFirstChild();
567       JSType baseType = base.getJSType();
568 
569       if (baseType == null
570           || !isPossiblySubtype(baseType, googDisposableInterfaceType)) {
571         return null;
572       }
573 
574       return n.getLastChild().getJSType();
575     }
576 
577     if (checkDispose) {
578       if (property.equals("goog.dispose")) {
579         return n.getLastChild().getJSType();
580       }
581       if (property.endsWith(".dispose")) {
582         /*
583          * n -> call
584          *   n.firstChild -> "dispose"
585          *   n.firstChild.firstChild -> object
586          */
587         return n.getFirstChild().getFirstChild().getJSType();
588       }
589     }
590 
591     return null;
592   }

594   /*
595    * Compute eventize relationship graph.
596    */
597   private class ComputeEventizeTraversal extends AbstractPostOrderCallback
598       implements ScopedCallback {
599 
600     /*
601      * Keep track of whether in the constructor or disposal scope.
602      */
603     Stack<Boolean> isConstructorStack;
604     Stack<Boolean> isDisposalStack;
605 
606 
607     public ComputeEventizeTraversal() {
608       isConstructorStack = new Stack<>();
609       isDisposalStack = new Stack<>();
610       eventizes = new HashMap<>();
611     }
612 
613     private Boolean inConstructorScope() {
614       Preconditions.checkNotNull(isConstructorStack);
615       if (!isDisposalStack.isEmpty()) {
616         return isConstructorStack.peek();
617       }
618       return null;
619     }
620 
621     private Boolean inDisposalScope() {
622       Preconditions.checkNotNull(isDisposalStack);
623       if (!isDisposalStack.isEmpty()) {
624         return isDisposalStack.peek();
625       }
626       return null;
627     }
628 
629     /*
630      * Filter types not interested in for eventize graph
631      */
632     private boolean collectorFilterType(JSType type) {
633       if (type == null) {
634         return true;
635       }
636 
637       return type.isEmptyType() || type.isUnknownType()
638           || !isPossiblySubtype(type, googDisposableInterfaceType);
639     }
640 
641     /*
642      * Log that thisType eventizes thatType.
643      */
644     private void addEventize(JSType thisType, JSType thatType) {
645       if (collectorFilterType(thisType)
646           || collectorFilterType(thatType)
647           || thisType.isEquivalentTo(thatType)) {
648         return;
649       }
650 
651       String className = thisType.getDisplayName();
652       if (thatType.isUnionType()) {
653         UnionType ut = thatType.toMaybeUnionType();
654         for (JSType type : ut.getAlternates()) {
655           if (type.isObject()) {
656             addEventizeClass(className, type);
657           }
658         }
659       } else {
660         addEventizeClass(className, thatType);
661       }
662     }
663 
664     private void addEventizeClass(String className, JSType thatType) {
665       String propertyJsTypeName = thatType.getDisplayName();
666 
667       Set<String> eventize = eventizes.get(propertyJsTypeName);
668       if (eventize == null) {
669         eventize = new HashSet<>();
670         eventizes.put(propertyJsTypeName, eventize);
671       }
672       eventize.add(className);
673     }

675     @Override
676     public void enterScope(NodeTraversal t) {
677       Node n = t.getScopeRoot();
678       boolean isConstructor = false;
679       boolean isInDisposal = false;
680       String functionName = null;
681 
682       /*
683        * TypedScope entered is a function definition
684        */
685       if (n.isFunction()) {
686         functionName = NodeUtil.getFunctionName(n);
687 
688         /*
689          *  Skip anonymous functions
690          */
691         if (functionName != null) {
692 
693           JSDocInfo jsDocInfo = NodeUtil.getBestJSDocInfo(n);
694           if (jsDocInfo != null) {
695             /*
696              *  Record constructor of a type
697              */
698             if (jsDocInfo.isConstructor()) {
699               isConstructor = true;
700 
701               /*
702                * Initialize eventizes relationship
703                */
704               if (t.getTypedScope() != null
705                   && t.getTypedScope().getTypeOfThis() != null) {
706                 ObjectType objectType = ObjectType.cast(t.getTypedScope()
707                     .getTypeOfThis().dereference());
708 
709                 /*
710                  * Eventize due to inheritance
711                  */
712 
713                 while (objectType != null) {
714                   objectType = objectType.getImplicitPrototype();
715                   if (objectType == null) {
716                     break;
717                   }
718 
719                   if (objectType.getDisplayName().endsWith("prototype")) {
720                     continue;
721                   }
722 
723                   addEventize((JSType) compiler.getTypeIRegistry().getType(functionName),
724                       objectType);
725 
726                   /*
727                    * Don't add transitive eventize edges here, it will be
728                    * taken care of in computeEventful
729                    */
730                   break;
731                 }
732               }
733             }
734           }
735 
736           /*
737            *  Indicate within a disposeInternal member
738            */
739           if (functionName.endsWith(".disposeInternal")) {
740             isInDisposal = true;
741           }
742         }
743 
744         isConstructorStack.push(isConstructor);
745         isDisposalStack.push(isInDisposal);
746       } else {
747         isConstructorStack.push(inConstructorScope());
748         isDisposalStack.push(inDisposalScope());
749       }
750     }

758     /*
759      * Is the current node a call to goog.events.unlisten
760      */
761     private void isGoogEventsUnlisten(Node n) {
762       Preconditions.checkArgument(n.getChildCount() > 3);
763 
764       Node listener = n.getChildAtIndex(3);
765 
766       Node objectWithListener = n.getChildAtIndex(1);
767       if (!objectWithListener.isQualifiedName()) {
768         return;
769       }
770 
771       if (listener.isFunction()) {
772         /*
773          * Anonymous function
774          */
775         compiler.report(JSError.make(n, UNLISTEN_WITH_ANONBOUND));
776       } else if (listener.isCall()) {
777         if (!listener.getFirstChild().isQualifiedName()) {
778           /*
779            * Anonymous function
780            */
781           compiler.report(JSError.make(n, UNLISTEN_WITH_ANONBOUND));
782         } else if (listener.getFirstChild().matchesQualifiedName("goog.bind")) {
783           /*
784            * Using goog.bind to unlisten
785            */
786           compiler.report(JSError.make(n, UNLISTEN_WITH_ANONBOUND));
787         }
788       }
789     }

792     private void visitCall(NodeTraversal t, Node n) {
793       Node functionCalled = n.getFirstChild();
794       if (functionCalled == null || !functionCalled.isQualifiedName()) {
795           return;
796       }
797       JSType typeOfThis = getTypeOfThisForScope(t);
798       if (typeOfThis == null) {
799         return;
800       }
801 
802       /*
803        * Class considered eventful if there is an unlisten call in the
804        * disposal.
805        */
806       if (functionCalled.matchesQualifiedName("goog.events.unlisten")) {
807 
808         if (inDisposalScope()) {
809           eventfulTypes.add(typeOfThis);
810         }
811         isGoogEventsUnlisten(n);
812       }
813       if (inDisposalScope()
814           && functionCalled.matchesQualifiedName("goog.events.removeAll")) {
815         eventfulTypes.add(typeOfThis);
816       }
817 
818       /*
819        * If member with qualified name gets disposed of when this class
820        * gets disposed, consider the member type as an eventizer of this
821        * class.
822        */
823       JSType disposedType = maybeReturnDisposedType(n, inDisposalScope());
824       if (!collectorFilterType(disposedType)) {
825         addEventize(getTypeOfThisForScope(t), disposedType);
826       }
827     }

836           break;

843     /*
844      * Checks if the input node correspond to the creation of an eventful object
845      */
846     private boolean createsEventfulObject(Node n) {
847       Node first = n.getFirstChild();
848       JSType type = n.getJSType();
849       if (first == null
850           || !first.isQualifiedName()
851           || type.isEmptyType()
852           || type.isUnknownType()) {
853         return false;
854       }
855 
856       boolean isOfTypeNeedingDisposal = false;
857       for (JSType disposableType : eventfulTypes) {
858         if (type.isSubtype(disposableType)) {
859           isOfTypeNeedingDisposal = true;
860           break;
861         }
862       }
863       return isOfTypeNeedingDisposal;
864     }

866     /*
867      * This function traverses the current scope to see if a locally
868      * defined eventful object is assigned to a live-out variable.
869      *
870      * Note: This function could be called multiple times to traverse
871      * the same scope if multiple local eventful objects are created in the
872      * scope.
873      */
874     private Node localEventfulObjectAssign(
875         NodeTraversal t, Node propertyNode) {
876       Node parent;
877       if (!t.getTypedScope().isGlobal()) {
878         /*
879          * In function
880          */
881         parent = NodeUtil.getFunctionBody(t.getScopeRoot());
882       } else {
883         /*
884          * In global scope
885          */
886         parent = t.getScopeRoot().getFirstChild();
887       }
888 
889       /*
890        * Check to see if locally created EventHandler is assigned to field
891        */
892       for (Node sibling : parent.children()) {
893         if (sibling.isExprResult()) {
894           Node assign = sibling.getFirstChild();
895           if (assign.isAssign()) {
896             // assign.getLastChild().isEquivalentTo(propertyNode) did not work
897             if (propertyNode.matchesQualifiedName(assign.getLastChild())) {
898               if (!assign.getFirstChild().isName()) {
899                 return assign.getFirstChild();
900               }
901             }
902           }
903         }
904       }
905 
906       /*
907        * Eventful object created and assigned to a local variable which is not
908        * assigned to another variable in a way to allow disposal.
909        */
910       String key = generateKey(t, propertyNode, false);
911       if (key == null) {
912         return null;
913       }
914 
915       EventfulObjectState e;
916       if (eventfulObjectMap.containsKey(key)) {
917         e = eventfulObjectMap.get(key);
918 
919         if (e.seen == SeenType.ALLOCATED) {
920           e.seen = SeenType.ALLOCATED_LOCALLY;
921         }
922       } else {
923         e = new EventfulObjectState();
924         e.seen = SeenType.ALLOCATED_LOCALLY;
925 
926         eventfulObjectMap.put(key, e);
927       }
928       e.allocationSite = propertyNode;
929 
930 
931       return null;
932     }

934     /*
935      * Record the creation of a new eventful object.
936      */
937     private void visitNew(NodeTraversal t, Node n, Node parent) {
938       if (!createsEventfulObject(n)) {
939         return;
940       }
941 
942       /*
943        * Insert allocation site and construct into eventfulObjectMap
944        */
945       String key;
946       Node propertyNode;
947 
948       /*
949        * Handles (E is an eventful class):
950        *  - object.something = new E();
951        *  - local = new E();
952        *  - var local = new E();
953        */
954       if (parent.isAssign()) {
955         propertyNode = parent.getFirstChild();
956       } else {
957         propertyNode = parent;
958       }
959 
960       key = generateKey(t, propertyNode, false);
961       if (key == null) {
962         return;
963       }
964 
965       EventfulObjectState e;
966       if (eventfulObjectMap.containsKey(key)) {
967         e = eventfulObjectMap.get(key);
968       } else {
969         e = new EventfulObjectState();
970         e.seen = SeenType.ALLOCATED;
971 
972         eventfulObjectMap.put(key, e);
973       }
974       e.allocationSite = propertyNode;
975 
976       /*
977        * Check if locally defined eventful object is assigned to global variable
978        * and create an entry mapping to the previous site.
979        */
980       if (propertyNode.isName()) {
981         Node globalVarNode = localEventfulObjectAssign(t, propertyNode);
982         if (globalVarNode != null) {
983           key = generateKey(t, globalVarNode, false);
984           if (key == null) {
985             /*
986              * Local variable is assigned to an array or in a manner requiring
987              * a function call.
988              */
989             e.seen = SeenType.POSSIBLY_DISPOSED;
990             return;
991           }
992           eventfulObjectMap.put(key, e);
993         }
994       }
995     }

997     private void addDisposeArgumentsMatched(Map<String, List<Integer>> map,
998         Node n, String property, List<Node> foundDisposeCalls) {
999       for (Map.Entry<String, List<Integer>> disposeCallsEntry : map.entrySet()) {
1000         if (property.endsWith(disposeCallsEntry.getKey())) {
1001           List<Integer> disposeArguments = disposeCallsEntry.getValue();
1002 
1003           // Dispose specific arguments only
1004           Node t = n.getNext();
1005           int tsArgument = 0;
1006           for (Integer disposeArgument : disposeArguments) {
1007             switch (disposeArgument) {
1008               // Dispose all arguments
1009               case DISPOSE_ALL:
1010                 for (Node tt = n.getNext(); tt != null; tt = tt.getNext()) {
1011                   foundDisposeCalls.add(tt);
1012                 }
1013                 break;
1014               // Dispose objects called on
1015               case DISPOSE_SELF:
1016                 Node calledOn = n.getFirstChild();
1017 
1018                 foundDisposeCalls.add(calledOn);
1019                 break;
1020               default:
1021                 // The current item pointed to by t is beyond that requested in
1022                 // current array element.
1023                 if (tsArgument > disposeArgument) {
1024                   t = n.getNext();
1025                   tsArgument = 0;
1026                 }
1027                 for (; tsArgument < disposeArgument && t != null;
1028                         ++tsArgument) {
1029                   t = t.getNext();
1030                 }
1031                 if (tsArgument == disposeArgument && t != null) {
1032                   foundDisposeCalls.add(t);
1033                 }
1034                 break;
1035             }
1036           }
1037         }
1038       }
1039     }

1041     private List<Node> maybeGetValueNodesFromCall(Node n) {
1042       List<Node> ret = new ArrayList<>();
1043       Node first = n.getFirstChild();
1044 
1045       if (first == null || !first.isQualifiedName()) {
1046         return ret;
1047       }
1048       String property = first.getQualifiedName();
1049 
1050       Node base = first.getFirstChild();
1051       JSType baseType = null;
1052       if (base != null) {
1053         baseType = base.getJSType();
1054       }
1055 
1056       for (Map.Entry<JSType, Map<String, List<Integer>>> disposeCallEntry :
1057           disposeCalls.entrySet()) {
1058         JSType key = disposeCallEntry.getKey();
1059         if (key == null
1060             || (baseType != null && isPossiblySubtype(baseType, key))) {
1061           addDisposeArgumentsMatched(disposeCallEntry.getValue(), first,
1062               property, ret);
1063         }
1064       }
1065 
1066       return ret;
1067     }

1069     /*
1070      * Look for calls to an eventful object's disposal functions.
1071      * (dispose or removeAll will remove all event listeners from
1072      * an EventHandler).
1073      */
1074     private void visitCall(NodeTraversal t, Node n) {
1075       // Filter the calls to find a "dispose" call
1076       List<Node> variableNodes = maybeGetValueNodesFromCall(n);
1077 
1078       for (Node variableNode : variableNodes) {
1079         Preconditions.checkState(variableNode != null);
1080 
1081         // Only consider removals on eventful object
1082         boolean isTrackedRemoval = false;
1083         JSType vnType = variableNode.getJSType();
1084         for (JSType type : eventfulTypes) {
1085           if (isPossiblySubtype(vnType, type)) {
1086             isTrackedRemoval = true;
1087           }
1088         }
1089         if (!isTrackedRemoval) {
1090           continue;
1091         }
1092 
1093         String key = generateKey(t, variableNode, false);
1094         if (key == null) {
1095           continue;
1096         }
1097 
1098         eventfulObjectDisposed(t, variableNode);
1099       }
1100     }

1103      * Dereference a type, autoboxing it and filtering out null.
1104      * From {@link CheckAccessControls}

1110     /*
1111      * Check function definitions to add custom dispose methods.
1112      */
1113     public void visitFunction(NodeTraversal t, Node n) {
1114       Preconditions.checkArgument(n.isFunction());
1115       JSDocInfo jsDocInfo = NodeUtil.getBestJSDocInfo(n);
1116 
1117       // Function annotated to dispose of
1118       if (jsDocInfo != null && jsDocInfo.isDisposes()) {
1119         JSType type = n.getJSType();
1120         if (type == null || type.isUnknownType()) {
1121           return;
1122         }
1123 
1124         FunctionType funType = type.toMaybeFunctionType();
1125         Node paramNode = NodeUtil.getFunctionParameters(n).getFirstChild();
1126         List<Integer> positionalDisposedParameters = new ArrayList<>();
1127 
1128         if (jsDocInfo.disposesOf("*")) {
1129           positionalDisposedParameters.add(DISPOSE_ALL);
1130         } else {
1131           // Record index of parameters that are disposed.
1132           for (int index = 0; index < funType.getMaxArguments(); ++index) {
1133             // Bail out if the paramNode is not there.
1134             if (paramNode == null) {
1135               break;
1136             }
1137             if (jsDocInfo.disposesOf(paramNode.getString())) {
1138               positionalDisposedParameters.add(index);
1139             }
1140             paramNode = paramNode.getNext();
1141           }
1142         }
1143         addDisposeCall(NodeUtil.getFunctionName(n),
1144             positionalDisposedParameters);
1145       }
1146     }

1148     /*
1149      * Track assignments to see if a private field is being
1150      * overwritten.
1151      *
1152      * Assigning to an array element is taken care of by the generateKey
1153      * returning null on array ("complex") variable names.
1154      */
1155     public void visitAssign(NodeTraversal t, Node n) {
1156       Node assignedTo = n.getFirstChild();
1157       JSType assignedToType = assignedTo.getJSType();
1158       if (assignedToType == null || assignedToType.isEmptyType()) {
1159         return;
1160       }
1161 
1162       if (n.getFirstChild().isGetProp()) {
1163         boolean isTrackedAssign = false;
1164         for (JSType disposalType : eventfulTypes) {
1165           if (assignedToType.isSubtype(disposalType)) {
1166             isTrackedAssign = true;
1167             break;
1168           }
1169         }
1170         if (!isTrackedAssign) {
1171           return;
1172         }
1173 
1174         JSDocInfo di = n.getJSDocInfo();
1175         ObjectType objectType =
1176             ObjectType.cast(dereference(n.getFirstChild().getFirstChild()
1177                 .getJSType()));
1178         String propertyName = n.getFirstChild().getLastChild().getString();
1179 
1180         boolean fieldIsPrivate =
1181             (di != null) && (di.getVisibility() == Visibility.PRIVATE);
1182 
1183         /*
1184          * See if field is defined as private in superclass
1185          */
1186         while (objectType != null) {
1187           di = null;
1188           objectType = objectType.getImplicitPrototype();
1189           if (objectType == null) {
1190             break;
1191           }
1192 
1193           /*
1194            * Skip prototype definitions:
1195            *   Don't flag a field declared private in assignment as well
1196            *   as in prototype declaration
1197            * Assumption: The inheritance hierarchy is similar to
1198            *   class
1199            *   class.prototype
1200            *   superclass
1201            *   superclass.prototype
1202            */
1203           if (objectType.getDisplayName().endsWith("prototype")) {
1204             continue;
1205           }
1206 
1207           di = objectType.getOwnPropertyJSDocInfo(propertyName);
1208           if (di != null) {
1209             if (fieldIsPrivate || di.getVisibility() == Visibility.PRIVATE) {
1210               compiler.report(
1211                   t.makeError(n, OVERWRITE_PRIVATE_EVENTFUL_OBJECT));
1212               break;
1213             }
1214           }
1215         }
1216       }
1217     }

1219     /*
1220      * Filter out any eventful objects returned.
1221      */
1222     private void visitReturn(NodeTraversal t, Node n) {
1223       Node variableNode = n.getFirstChild();
1224       if (variableNode == null) {
1225         return;
1226       }
1227 
1228       if (!variableNode.isArrayLit()) {
1229         eventfulObjectDisposed(t, variableNode);
1230       } else {
1231         for (Node child : variableNode.children()) {
1232           eventfulObjectDisposed(t, child);
1233         }
1234       }
1235     }

1237     /*
1238      * Mark an eventful object as being disposed.
1239      */
1240     private void eventfulObjectDisposed(NodeTraversal t, Node variableNode) {
1241       String key = generateKey(t, variableNode, false);
1242       if (key == null) {
1243         return;
1244       }
1245 
1246       EventfulObjectState e = eventfulObjectMap.get(key);
1247       if (e == null) {
1248         e = new EventfulObjectState();
1249         eventfulObjectMap.put(key, e);
1250       }
1251       e.seen = SeenType.POSSIBLY_DISPOSED;
1252     }

1255     public void enterScope(NodeTraversal t) {
1256       /*
1257        * Local variables captured in scope are filtered at present.
1258        * LiveVariableAnalysis used to filter such variables.
1259        */
1260       ControlFlowGraph<Node> cfg = t.getControlFlowGraph();
1261       LiveVariablesAnalysis liveness =
1262           new LiveVariablesAnalysis(cfg, t.getTypedScope(), compiler);
1263       liveness.analyze();
1264 
1265       for (TypedVar v : ((Set<TypedVar>) liveness.getEscapedLocals())) {
1266         eventfulObjectDisposed(t, v.getNode());
1267       }

1274     @Override
1275     public void visit(NodeTraversal t, Node n, Node parent) {
1276       switch (n.getType()) {
