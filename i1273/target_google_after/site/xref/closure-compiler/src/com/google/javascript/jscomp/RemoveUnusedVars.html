38  /**
39   * Garbage collection for variable and function definitions. Basically performs
40   * a mark-and-sweep type algorithm over the JavaScript parse tree.
41   *
42   * For each scope:
43   * (1) Scan the variable/function declarations at that scope.
44   * (2) Traverse the scope for references, marking all referenced variables.
45   *     Unlike other compiler passes, this is a pre-order traversal, not a
46   *     post-order traversal.
47   * (3) If the traversal encounters an assign without other side-effects,
48   *     create a continuation. Continue the continuation iff the assigned
49   *     variable is referenced.
50   * (4) When the traversal completes, remove all unreferenced variables.
51   *
52   * If it makes it easier, you can think of the continuations of the traversal
53   * as a reference graph. Each continuation represents a set of edges, where the
54   * source node is a known variable, and the destination nodes are lazily
55   * evaluated when the continuation is executed.
56   *
57   * This algorithm is similar to the algorithm used by {@code SmartNameRemoval}.
58   * {@code SmartNameRemoval} maintains an explicit graph of dependencies
59   * between global symbols. However, {@code SmartNameRemoval} cannot handle
60   * non-trivial edges in the reference graph ("A is referenced iff both B and C
61   * are referenced"), or local variables. {@code SmartNameRemoval} is also
62   * substantially more complicated because it tries to handle namespaces
63   * (which is largely unnecessary in the presence of {@code CollapseProperties}.
64   *
65   * This pass also uses a more complex analysis of assignments, where
66   * an assignment to a variable or a property of that variable does not
67   * necessarily count as a reference to that variable, unless we can prove
68   * that it modifies external state. This is similar to
69   * {@code FlowSensitiveInlineVariables}, except that it works for variables
70   * used across scopes.
71   *
72   * @author nicksantos@google.com (Nick Santos)
73   */
74  class RemoveUnusedVars
75      implements CompilerPass, OptimizeCalls.CallGraphCompilerPass {

140   /**
141    * Traverses the root, removing all unused variables. Multiple traversals
142    * may occur to ensure all unused variables are removed.
143    */
144   @Override
145   public void process(Node externs, Node root) {
146     Preconditions.checkState(compiler.getLifeCycleStage().isNormalized());
147     SimpleDefinitionFinder defFinder = null;
148 
149     if (modifyCallSites) {
150       // For testing, allow the SimpleDefinitionFinder to be build now.
151       defFinder = new SimpleDefinitionFinder(compiler);
152       defFinder.process(externs, root);
153     }
154     process(externs, root, defFinder);
155   }
156 
157   @Override
158   public void process(
159       Node externs, Node root, SimpleDefinitionFinder defFinder) {
160     if (modifyCallSites) {
161       Preconditions.checkNotNull(defFinder);
162       callSiteOptimizer = new CallSiteOptimizer(compiler, defFinder);
163     }
164     traverseAndRemoveUnusedReferences(root);
165     if (callSiteOptimizer != null) {
166       callSiteOptimizer.applyChanges();
167     }
168   }

170   /**
171    * Traverses a node recursively. Call this once per pass.
172    */
173   private void traverseAndRemoveUnusedReferences(Node root) {
174     Scope scope = SyntacticScopeCreator.makeUntyped(compiler).createScope(root, null);
175     traverseNode(root, null, scope);
176 
177     if (removeGlobals) {
178       collectMaybeUnreferencedVars(scope);
179     }
180 
181     interpretAssigns();
182     removeUnreferencedVars();
183     for (Scope fnScope : allFunctionScopes) {
184       removeUnreferencedFunctionArgs(fnScope);
185     }
186   }

188   /**
189    * Traverses everything in the current scope and marks variables that
190    * are referenced.
191    *
192    * During traversal, we identify subtrees that will only be
193    * referenced if their enclosing variables are referenced. Instead of
194    * traversing those subtrees, we create a continuation for them,
195    * and traverse them lazily.
196    */
197   private void traverseNode(Node n, Node parent, Scope scope) {
198     int type = n.getType();
199     Var var = null;
200     switch (type) {
201       case Token.FUNCTION:
202         // If this function is a removable var, then create a continuation
203         // for it instead of traversing immediately.
204         if (NodeUtil.isFunctionDeclaration(n)) {
205           var = scope.getVar(n.getFirstChild().getString());
206         }
207 
208         if (var != null && isRemovableVar(var)) {
209           continuations.put(var, new Continuation(n, scope));
210         } else {
211           traverseFunction(n, scope);
212         }
213         return;
214 
215       case Token.ASSIGN:
216         Assign maybeAssign = Assign.maybeCreateAssign(n);
217         if (maybeAssign != null) {
218           // Put this in the assign map. It might count as a reference,
219           // but we won't know that until we have an index of all assigns.
220           var = scope.getVar(maybeAssign.nameNode.getString());
221           if (var != null) {
222             assignsByVar.put(var, maybeAssign);
223             assignsByNode.put(maybeAssign.nameNode, maybeAssign);
224 
225             if (isRemovableVar(var) &&
226                 !maybeAssign.mayHaveSecondarySideEffects) {
227               // If the var is unreferenced and performing this assign has
228               // no secondary side effects, then we can create a continuation
229               // for it instead of traversing immediately.
230               continuations.put(var, new Continuation(n, scope));
231               return;
232             }
233           }
234         }
235         break;
236 
237       case Token.CALL:
238         Var modifiedVar = null;
239 
240         // Look for calls to inheritance-defining calls (such as goog.inherits).
241         SubclassRelationship subclassRelationship =
242             codingConvention.getClassesDefinedByCall(n);
243         if (subclassRelationship != null) {
244           modifiedVar = scope.getVar(subclassRelationship.subclassName);
245         } else {
246           // Look for calls to addSingletonGetter calls.
247           String className = codingConvention.getSingletonGetterClassName(n);
248           if (className != null) {
249             modifiedVar = scope.getVar(className);
250           }
251         }
252 
253         // Don't try to track the inheritance calls for non-globals. It would
254         // be more correct to only not track when the subclass does not
255         // reference a constructor, but checking that it is a global is
256         // easier and mostly the same.
257         if (modifiedVar != null && modifiedVar.isGlobal()
258             && !referenced.contains(modifiedVar)) {
259           // Save a reference to the EXPR node.
260           classDefiningCalls.put(modifiedVar, parent);
261           continuations.put(modifiedVar, new Continuation(n, scope));
262           return;
263         }
264         break;
265 
266       case Token.NAME:
267         var = scope.getVar(n.getString());
268         if (parent.isVar()) {
269           Node value = n.getFirstChild();
270           if (value != null && var != null && isRemovableVar(var) &&
271               !NodeUtil.mayHaveSideEffects(value, compiler)) {
272             // If the var is unreferenced and creating its value has no side
273             // effects, then we can create a continuation for it instead
274             // of traversing immediately.
275             continuations.put(var, new Continuation(n, scope));
276             return;
277           }
278         } else {
279 
280           // If arguments is escaped, we just assume the worst and continue
281           // on all the parameters.
282           if ("arguments".equals(n.getString()) && scope.isLocal()) {
283             Node lp = scope.getRootNode().getFirstChild().getNext();
284             for (Node a = lp.getFirstChild(); a != null; a = a.getNext()) {
285               markReferencedVar(scope.getVar(a.getString()));
286             }
287           }
288 
289           // All name references that aren't declarations or assigns
290           // are references to other vars.
291           if (var != null) {
292             // If that var hasn't already been marked referenced, then
293             // start tracking it.  If this is an assign, do nothing
294             // for now.
295             if (isRemovableVar(var)) {
296               if (!assignsByNode.containsKey(n)) {
297                 markReferencedVar(var);
298               }
299             } else {
300               markReferencedVar(var);
301             }
302           }
303         }
304         break;
305     }
306 
307     for (Node c = n.getFirstChild(); c != null; c = c.getNext()) {
308       traverseNode(c, n, scope);
309     }
310   }
311 
312   private boolean isRemovableVar(Var var) {
313     // Global variables are off-limits if the user might be using them.
314     if (!removeGlobals && var.isGlobal()) {
315       return false;
316     }
317 
318     // Referenced variables are off-limits.
319     if (referenced.contains(var)) {
320       return false;
321     }
322 
323     // Exported variables are off-limits.
324     return !codingConvention.isExported(var.getName());
325   }

327   /**
328    * Traverses a function, which creates a new scope in JavaScript.
329    *
330    * Note that CATCH blocks also create a new scope, but only for the
331    * catch variable. Declarations within the block actually belong to the
332    * enclosing scope. Because we don't remove catch variables, there's
333    * no need to treat CATCH blocks differently like we do functions.
334    */
335   private void traverseFunction(Node n, Scope parentScope) {
336     Preconditions.checkState(n.getChildCount() == 3);
337     Preconditions.checkState(n.isFunction());
338 
339     final Node body = n.getLastChild();
340     Preconditions.checkState(body.getNext() == null &&
341             body.isBlock());
342 
343     Scope fnScope = SyntacticScopeCreator.makeUntyped(compiler).createScope(n, parentScope);
344     traverseNode(body, n, fnScope);
345 
346     collectMaybeUnreferencedVars(fnScope);
347     allFunctionScopes.add(fnScope);
348   }
349 
350   /**
351    * For each variable in this scope that we haven't found a reference
352    * for yet, add it to the list of variables to check later.
353    */
354   private void collectMaybeUnreferencedVars(Scope scope) {
355     for (Iterator<Var> it = scope.getVars(); it.hasNext(); ) {
356       Var var = it.next();
357       if (isRemovableVar(var)) {
358         maybeUnreferenced.add(var);
359       }
360     }
361   }

363   /**
364    * Removes unreferenced arguments from a function declaration and when
365    * possible the function's callSites.
366    *
367    * @param fnScope The scope inside the function
368    */
369   private void removeUnreferencedFunctionArgs(Scope fnScope) {
370     // Notice that removing unreferenced function args breaks
371     // Function.prototype.length. In advanced mode, we don't really care
372     // about this: we consider "length" the equivalent of reflecting on
373     // the function's lexical source.
374     //
375     // Rather than create a new option for this, we assume that if the user
376     // is removing globals, then it's OK to remove unused function args.
377     //
378     // See http://code.google.com/p/closure-compiler/issues/detail?id=253
379     if (!removeGlobals) {
380       return;
381     }
382 
383     Node function = fnScope.getRootNode();
384 
385     Preconditions.checkState(function.isFunction());
386     if (NodeUtil.isGetOrSetKey(function.getParent())) {
387       // The parameters object literal setters can not be removed.
388       return;
389     }
390 
391     Node argList = getFunctionArgList(function);
392     boolean modifyCallers = modifyCallSites
393         && callSiteOptimizer.canModifyCallers(function);
394     if (!modifyCallers) {
395       // Strip unreferenced args off the end of the function declaration.
396       Node lastArg;
397       while ((lastArg = argList.getLastChild()) != null) {
398         Var var = fnScope.getVar(lastArg.getString());
399         if (!referenced.contains(var)) {
400           compiler.reportChangeToEnclosingScope(lastArg);
401           argList.removeChild(lastArg);
402         } else {
403           break;
404         }
405       }
406     } else {
407       callSiteOptimizer.optimize(fnScope, referenced);
408     }
409   }

412   /**

415   private static Node getFunctionArgList(Node function) {
416     return function.getFirstChild().getNext();
417   }

419   private static class CallSiteOptimizer {
420     private final AbstractCompiler compiler;
421     private final SimpleDefinitionFinder defFinder;
422     private final List<Node> toRemove = new ArrayList<>();
423     private final List<Node> toReplaceWithZero = new ArrayList<>();
424 
425     CallSiteOptimizer(
426         AbstractCompiler compiler,
427         SimpleDefinitionFinder defFinder) {
428       this.compiler = compiler;
429       this.defFinder = defFinder;
430     }
431 
432     public void optimize(Scope fnScope, Set<Var> referenced) {
433       Node function = fnScope.getRootNode();
434       Preconditions.checkState(function.isFunction());
435       Node argList = getFunctionArgList(function);
436 
437       // In this path we try to modify all the call sites to remove unused
438       // function parameters.
439       boolean changeCallSignature = canChangeSignature(function);
440       markUnreferencedFunctionArgs(
441           fnScope, function, referenced,
442           argList.getFirstChild(), 0, changeCallSignature);
443     }
444 
445     /**
446      * Applies optimizations to all previously marked nodes.
447      */
448     public void applyChanges() {
449       for (Node n : toRemove) {
450         compiler.reportChangeToEnclosingScope(n);
451         n.getParent().removeChild(n);
452       }
453       for (Node n : toReplaceWithZero) {
454         compiler.reportChangeToEnclosingScope(n);
455         n.getParent().replaceChild(n, IR.number(0).srcref(n));
456       }
457     }
458 
459     /**
460      * For each unused function parameter, determine if it can be removed
461      * from all the call sites, if so, remove it from the function signature
462      * and the call sites otherwise replace the unused value where possible
463      * with a constant (0).
464      *
465      * @param scope The function scope
466      * @param function The function
467      * @param param The current parameter node in the parameter list.
468      * @param paramIndex The index of the current parameter
469      * @param canChangeSignature Whether function signature can be change.
470      * @return Whether there is a following function parameter.
471      */
472     private boolean markUnreferencedFunctionArgs(
473         Scope scope, Node function, Set<Var> referenced,
474         Node param, int paramIndex,
475         boolean canChangeSignature) {
476       if (param != null) {
477         // Take care of the following siblings first.
478         boolean hasFollowing = markUnreferencedFunctionArgs(
479             scope, function, referenced, param.getNext(), paramIndex + 1,
480             canChangeSignature);
481 
482         Var var = scope.getVar(param.getString());
483         if (!referenced.contains(var)) {
484           Preconditions.checkNotNull(var);
485 
486           // Remove call parameter if we can generally change the signature
487           // or if it is the last parameter in the parameter list.
488           boolean modifyAllCallSites = canChangeSignature || !hasFollowing;
489           if (modifyAllCallSites) {
490             modifyAllCallSites = canRemoveArgFromCallSites(
491                 function, paramIndex);
492           }
493 
494           tryRemoveArgFromCallSites(function, paramIndex, modifyAllCallSites);
495 
496           // Remove an unused function parameter if all the call sites can
497           // be modified to remove it, or if it is the last parameter.
498           if (modifyAllCallSites || !hasFollowing) {
499             toRemove.add(param);
500             return hasFollowing;
501           }
502         }
503         return true;
504       } else {
505         // Anything past the last formal parameter can be removed from the call
506         // sites.
507         tryRemoveAllFollowingArgs(function, paramIndex - 1);
508         return false;
509       }
510     }
511 
512     /**
513      * Remove all references to a parameter, otherwise simplify the known
514      * references.
515      * @return Whether all the references were removed.
516      */
517     private boolean canRemoveArgFromCallSites(Node function, int argIndex) {
518       Definition definition = getFunctionDefinition(function);
519 
520       // Check all the call sites.
521       for (UseSite site : defFinder.getUseSites(definition)) {
522         if (isModifiableCallSite(site)) {
523           Node arg = getArgumentForCallOrNewOrDotCall(site, argIndex);
524           // TODO(johnlenz): try to remove parameters with side-effects by
525           // decomposing the call expression.
526           if (arg != null && NodeUtil.mayHaveSideEffects(arg, compiler)) {
527             return false;
528           }
529         } else {
530           return false;
531         }
532       }
533 
534       return true;
535     }
536 
537     /**
538      * Remove all references to a parameter if possible otherwise simplify the
539      * side-effect free parameters.
540      */
541     private void tryRemoveArgFromCallSites(
542         Node function, int argIndex, boolean canModifyAllSites) {
543       Definition definition = getFunctionDefinition(function);
544 
545       for (UseSite site : defFinder.getUseSites(definition)) {
546         if (isModifiableCallSite(site)) {
547           Node arg = getArgumentForCallOrNewOrDotCall(site, argIndex);
548           if (arg != null) {
549             // Even if we can't change the signature in general we can always
550             // remove an unused value off the end of the parameter list.
551             if (canModifyAllSites
552                 || (arg.getNext() == null
553                     && !NodeUtil.mayHaveSideEffects(arg, compiler))) {
554               toRemove.add(arg);
555             } else {
556               // replace the node in the arg with 0
557               if (!NodeUtil.mayHaveSideEffects(arg, compiler)
558                   && (!arg.isNumber() || arg.getDouble() != 0)) {
559                 toReplaceWithZero.add(arg);
560               }
561             }
562           }
563         }
564       }
565     }
566 
567     /**
568      * Remove all the following parameters without side-effects
569      */
570     private void tryRemoveAllFollowingArgs(Node function, final int argIndex) {
571       Definition definition = getFunctionDefinition(function);
572       for (UseSite site : defFinder.getUseSites(definition)) {
573         if (!isModifiableCallSite(site)) {
574           continue;
575         }
576         Node arg = getArgumentForCallOrNewOrDotCall(site, argIndex + 1);
577         while (arg != null) {
578           if (!NodeUtil.mayHaveSideEffects(arg)) {
579             toRemove.add(arg);
580           }
581           arg = arg.getNext();
582         }
583       }
584     }

586     /**
587      * Returns the nth argument node given a usage site for a direct function
588      * call or for a func.call() node.
589      */
590     private static Node getArgumentForCallOrNewOrDotCall(UseSite site,
591         final int argIndex) {
592       int adjustedArgIndex = argIndex;
593       Node parent = site.node.getParent();
594       if (NodeUtil.isFunctionObjectCall(parent)) {
595         adjustedArgIndex++;
596       }
597       return NodeUtil.getArgumentForCallOrNew(parent, adjustedArgIndex);
598     }

600     /**
601      * @param function
602      * @return Whether the callers to this function can be modified in any way.
603      */
604     boolean canModifyCallers(Node function) {
605       if (NodeUtil.isVarArgsFunction(function)) {
606         return false;
607       }
608 
609       DefinitionSite defSite = defFinder.getDefinitionForFunction(function);
610       if (defSite == null) {
611         return false;
612       }
613 
614       Definition definition = defSite.definition;
615 
616       // Be conservative, don't try to optimize any declaration that isn't as
617       // simple function declaration or assignment.
618       if (!SimpleDefinitionFinder.isSimpleFunctionDeclaration(function)) {
619         return false;
620       }
621 
622       return defFinder.canModifyDefinition(definition);
623     }
624 
625     /**
626      * @param site The site to inspect
627      * @return Whether the call site is suitable for modification
628      */
629     private static boolean isModifiableCallSite(UseSite site) {
630       return SimpleDefinitionFinder.isCallOrNewSite(site)
631           && !NodeUtil.isFunctionObjectApply(site.node.getParent());
632     }
633 
634     /**
635      * @return Whether the definitionSite represents a function whose call
636      *      signature can be modified.
637      */
638     private boolean canChangeSignature(Node function) {
639       Definition definition = getFunctionDefinition(function);
640       CodingConvention convention = compiler.getCodingConvention();
641 
642       Preconditions.checkState(!definition.isExtern());
643 
644       Collection<UseSite> useSites = defFinder.getUseSites(definition);
645       for (UseSite site : useSites) {
646         Node parent = site.node.getParent();
647 
648         // This was a use site removed by something else before we run.
649         // 1. By another pass before us which means the definition graph is
650         //    no updated properly.
651         // 2. By the continuations algorithm above.
652         if (parent == null) {
653           continue; // Ignore it.
654         }
655 
656         // Ignore references within goog.inherits calls.
657         if (parent.isCall() &&
658             convention.getClassesDefinedByCall(parent) != null) {
659           continue;
660         }
661 
662         // Accessing the property directly prevents rewrite.
663         if (!SimpleDefinitionFinder.isCallOrNewSite(site)) {
664           if (!(parent.isGetProp() &&
665               NodeUtil.isFunctionObjectCall(parent.getParent()))) {
666             return false;
667           }
668         }
669 
670         if (NodeUtil.isFunctionObjectApply(parent)) {
671           return false;
672         }
673 
674         // TODO(johnlenz): support specialization
675 
676         // Multiple definitions prevent rewrite.
677         // Attempt to validate the state of the simple definition finder.
678         Node nameNode = site.node;
679         Collection<Definition> singleSiteDefinitions =
680             defFinder.getDefinitionsReferencedAt(nameNode);
681         Preconditions.checkState(singleSiteDefinitions.size() == 1);
682         Preconditions.checkState(singleSiteDefinitions.contains(definition));
683       }
684 
685       return true;
686     }
687 
688     /**
689      * @param function
690      * @return the Definition object for the function.
691      */
692     private Definition getFunctionDefinition(Node function) {
693       DefinitionSite definitionSite = defFinder.getDefinitionForFunction(
694           function);
695       Preconditions.checkNotNull(definitionSite);
696       Definition definition = definitionSite.definition;
697       Preconditions.checkState(!definitionSite.inExterns);
698       Preconditions.checkState(definition.getRValue() == function);
699       return definition;
700     }
701   }

704   /**
705    * Look at all the property assigns to all variables.
706    * These may or may not count as references. For example,
707    *
708    * <code>
709    * var x = {};
710    * x.foo = 3; // not a reference.
711    * var y = foo();
712    * y.foo = 3; // is a reference.
713    * </code>
714    *
715    * Interpreting assignments could mark a variable as referenced that
716    * wasn't referenced before, in order to keep it alive. Because we find
717    * references by lazily traversing subtrees, marking a variable as
718    * referenced could trigger new traversals of new subtrees, which could
719    * find new references.
720    *
721    * Therefore, this interpretation needs to be run to a fixed point.
722    */
723   private void interpretAssigns() {
724     boolean changes = false;
725     do {
726       changes = false;
727 
728       // We can't use traditional iterators and iterables for this list,
729       // because our lazily-evaluated continuations will modify it while
730       // we traverse it.
731       for (int current = 0; current < maybeUnreferenced.size(); current++) {
732         Var var = maybeUnreferenced.get(current);
733         if (referenced.contains(var)) {
734           maybeUnreferenced.remove(current);
735           current--;
736         } else {
737           boolean assignedToUnknownValue = false;
738           boolean hasPropertyAssign = false;
739 
740           if (var.getParentNode().isVar() &&
741               !NodeUtil.isForIn(var.getParentNode().getParent())) {
742             Node value = var.getInitialValue();
743             assignedToUnknownValue = value != null &&
744                 !NodeUtil.isLiteralValue(value, true);
745           } else {
746             // This was initialized to a function arg or a catch param
747             // or a for...in variable.
748             assignedToUnknownValue = true;
749           }
750 
751           boolean maybeEscaped = false;
752           for (Assign assign : assignsByVar.get(var)) {
753             if (assign.isPropertyAssign) {
754               hasPropertyAssign = true;
755             } else if (!NodeUtil.isLiteralValue(
756                 assign.assignNode.getLastChild(), true)) {
757               assignedToUnknownValue = true;
758             }
759             if (assign.maybeAliased) {
760               maybeEscaped = true;
761             }
762           }
763 
764           if ((assignedToUnknownValue || maybeEscaped) && hasPropertyAssign) {
765             changes = markReferencedVar(var) || changes;
766             maybeUnreferenced.remove(current);
767             current--;
768           }
769         }
770       }
771     } while (changes);
772   }
773 
774   /**
775    * Remove all assigns to a var.
776    */
777   private void removeAllAssigns(Var var) {
778     for (Assign assign : assignsByVar.get(var)) {
779       compiler.reportChangeToEnclosingScope(assign.assignNode);
780       assign.remove();
781     }
782   }
783 
784   /**
785    * Marks a var as referenced, recursing into any values of this var
786    * that we skipped.
787    * @return True if this variable had not been referenced before.
788    */
789   private boolean markReferencedVar(Var var) {
790     if (referenced.add(var)) {
791       for (Continuation c : continuations.get(var)) {
792         c.apply();
793       }
794       return true;
795     }
796     return false;
797   }
798 
799   /**
800    * Removes any vars in the scope that were not referenced. Removes any
801    * assignments to those variables as well.
802    */
803   private void removeUnreferencedVars() {
804     for (Var var : maybeUnreferenced) {
805       // Remove calls to inheritance-defining functions where the unreferenced
806       // class is the subclass.
807       for (Node exprCallNode : classDefiningCalls.get(var)) {
808         compiler.reportChangeToEnclosingScope(exprCallNode);
809         NodeUtil.removeChild(exprCallNode.getParent(), exprCallNode);
810       }
811 
812       // Regardless of what happens to the original declaration,
813       // we need to remove all assigns, because they may contain references
814       // to other unreferenced variables.
815       removeAllAssigns(var);
816 
817       compiler.addToDebugLog("Unreferenced var: " + var.name);
818       Node nameNode = var.nameNode;
819       Node toRemove = nameNode.getParent();
820       Node parent = toRemove.getParent();
821 
822       Preconditions.checkState(
823           toRemove.isVar() ||
824           toRemove.isFunction() ||
825           toRemove.isParamList() &&
826           parent.isFunction(),
827           "We should only declare vars and functions and function args");
828 
829       if (toRemove.isParamList() &&
830           parent.isFunction()) {
831         // Don't remove function arguments here. That's a special case
832         // that's taken care of in removeUnreferencedFunctionArgs.
833       } else if (NodeUtil.isFunctionExpression(toRemove)) {
834         if (!preserveFunctionExpressionNames) {
835           compiler.reportChangeToEnclosingScope(toRemove);
836           toRemove.getFirstChild().setString("");
837         }
838         // Don't remove bleeding functions.
839       } else if (parent != null &&
840           parent.isFor() &&
841           parent.getChildCount() < 4) {
842         // foreach iterations have 3 children. Leave them alone.
843       } else if (toRemove.isVar() &&
844           nameNode.hasChildren() &&
845           NodeUtil.mayHaveSideEffects(nameNode.getFirstChild(), compiler)) {
846         // If this is a single var declaration, we can at least remove the
847         // declaration itself and just leave the value, e.g.,
848         // var a = foo(); => foo();
849         if (toRemove.getChildCount() == 1) {
850           compiler.reportChangeToEnclosingScope(toRemove);
851           parent.replaceChild(toRemove,
852               IR.exprResult(nameNode.removeFirstChild()));
853         }
854       } else if (toRemove.isVar() &&
855           toRemove.getChildCount() > 1) {
856         // For var declarations with multiple names (i.e. var a, b, c),
857         // only remove the unreferenced name
858         compiler.reportChangeToEnclosingScope(toRemove);
859         toRemove.removeChild(nameNode);
860       } else if (parent != null) {
861         compiler.reportChangeToEnclosingScope(toRemove);
862         NodeUtil.removeChild(parent, toRemove);
863       }
864     }
865   }

867   /**
868    * Our progress in a traversal can be expressed completely as the
869    * current node and scope. The continuation lets us save that
870    * information so that we can continue the traversal later.
871    */
872   private class Continuation {
873     private final Node node;
874     private final Scope scope;
875 
876     Continuation(Node node, Scope scope) {
877       this.node = node;
878       this.scope = scope;
879     }
880 
881     void apply() {
882       if (NodeUtil.isFunctionDeclaration(node)) {
883         traverseFunction(node, scope);
884       } else {
885         for (Node child = node.getFirstChild();
886              child != null; child = child.getNext()) {
887           traverseNode(child, node, scope);
888         }
889       }
890     }
891   }

893   private static class Assign {
894 
895     final Node assignNode;
896 
897     final Node nameNode;
898 
899     // If false, then this is an assign to the normal variable. Otherwise,
900     // this is an assign to a property of that variable.
901     final boolean isPropertyAssign;
902 
903     // Secondary side effects are any side effects in this assign statement
904     // that aren't caused by the assignment operation itself. For example,
905     // a().b = 3;
906     // a = b();
907     // var foo = (a = b);
908     // In the first two cases, the sides of the assignment have side-effects.
909     // In the last one, the result of the assignment is used, so we
910     // are conservative and assume that it may be used in a side-effecting
911     // way.
912     final boolean mayHaveSecondarySideEffects;
913 
914     // If true, the value may have escaped and any modification is a use.
915     final boolean maybeAliased;
916 
917     Assign(Node assignNode, Node nameNode, boolean isPropertyAssign) {
918       Preconditions.checkState(NodeUtil.isAssignmentOp(assignNode));
919       this.assignNode = assignNode;
920       this.nameNode = nameNode;
921       this.isPropertyAssign = isPropertyAssign;
922 
923       this.maybeAliased = NodeUtil.isExpressionResultUsed(assignNode);
924       this.mayHaveSecondarySideEffects =
925           maybeAliased ||
926           NodeUtil.mayHaveSideEffects(assignNode.getFirstChild()) ||
927           NodeUtil.mayHaveSideEffects(assignNode.getLastChild());
928     }
929 
930     /**
931      * If this is an assign to a variable or its property, return it.
932      * Otherwise, return null.
933      */
934     static Assign maybeCreateAssign(Node assignNode) {
935       Preconditions.checkState(NodeUtil.isAssignmentOp(assignNode));
936 
937       // Skip one level of GETPROPs or GETELEMs.
938       //
939       // Don't skip more than one level, because then we get into
940       // situations where assigns to properties of properties will always
941       // trigger side-effects, and the variable they're on cannot be removed.
942       boolean isPropAssign = false;
943       Node current = assignNode.getFirstChild();
944       if (NodeUtil.isGet(current)) {
945         current = current.getFirstChild();
946         isPropAssign = true;
947 
948         if (current.isGetProp() &&
949             current.getLastChild().getString().equals("prototype")) {
950           // Prototype properties sets should be considered like normal
951           // property sets.
952           current = current.getFirstChild();
953         }
954       }
955 
956       if (current.isName()) {
957         return new Assign(assignNode, current, isPropAssign);
958       }
959       return null;
960     }
961 
962     /**
963      * Replace the current assign with its right hand side.
964      */
965     void remove() {
966       Node parent = assignNode.getParent();
967       if (mayHaveSecondarySideEffects) {
968         Node replacement = assignNode.getLastChild().detachFromParent();
969 
970         // Aggregate any expressions in GETELEMs.
971         for (Node current = assignNode.getFirstChild();
972              !current.isName();
973              current = current.getFirstChild()) {
974           if (current.isGetElem()) {
975             replacement = IR.comma(
976                 current.getLastChild().detachFromParent(), replacement);
977             replacement.useSourceInfoIfMissingFrom(current);
978           }
979         }
980 
981         parent.replaceChild(assignNode, replacement);
982       } else {
983         Node grandparent = parent.getParent();
984         if (parent.isExprResult()) {
985           grandparent.removeChild(parent);
986         } else {
987           parent.replaceChild(assignNode,
988               assignNode.getLastChild().detachFromParent());
989         }
990       }
991     }
992   }
