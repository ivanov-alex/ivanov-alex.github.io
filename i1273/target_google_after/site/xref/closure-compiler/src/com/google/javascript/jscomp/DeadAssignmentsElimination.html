32  /**
33   * Removes local variable assignments that are useless based on information from
34   * {@link LiveVariablesAnalysis}. If there is an assignment to variable
35   * {@code x} and {@code x} is dead after this assignment, we know that the
36   * current content of {@code x} will not be read and this assignment is useless.
37   *
38   */
39  class DeadAssignmentsElimination extends AbstractPostOrderCallback implements
40      CompilerPass, ScopedCallback {
41  
42    private final AbstractCompiler compiler;
43    private LiveVariablesAnalysis liveness;
44  
45    // Matches all assignment operators and increment/decrement operators.
46    // Does *not* match VAR initialization, since RemoveUnusedVariables
47    // will already remove variables that are initialized but unused.
48    private static final Predicate<Node> matchRemovableAssigns =
49        new Predicate<Node>() {
50      @Override
51      public boolean apply(Node n) {
52        return (NodeUtil.isAssignmentOp(n) &&
53                n.getFirstChild().isName()) ||
54            n.isInc() || n.isDec();
55      }
56    };
57  
58    public DeadAssignmentsElimination(AbstractCompiler compiler) {
59      this.compiler = compiler;
60    }
61  
62    @Override
63    public void process(Node externs, Node root) {
64      Preconditions.checkNotNull(externs);
65      Preconditions.checkNotNull(root);
66      NodeTraversal.traverseEs6(compiler, root, this);
67    }

73    @Override
74    public void exitScope(NodeTraversal t) {
75      // Do nothing on global scope / global blocks
76      if (t.inGlobalHoistScope()) {
77        return;
78      }
79  
80      Scope scope = t.getScope();
81      // Elevate all variable declarations up till the enclosing function scope
82      // so the liveness analysis has all variables for the process.
83      if (!scope.isFunctionScope()) {
84        for (Var var : scope.getVarIterable()) {
85          Preconditions.checkArgument(
86              !var.isClass() && !var.isLet() && !var.isConst());
87          scope.getParent().declare(
88              var.getName(), var.getNameNode(), var.getInput());
89        }
90      } else {
91        if (LiveVariablesAnalysis.MAX_VARIABLES_TO_ANALYZE <
92            scope.getVarCount()) {
93          return;
94        }
95  
96        // We are not going to do any dead assignment elimination in when there is
97        // at least one inner function because in most browsers, when there is a
98        // closure, ALL the variables are saved (escaped).
99        Node fnBlock = t.getScopeRoot().getLastChild();
100       if (NodeUtil.containsFunction(fnBlock)) {
101         return;
102       }
103 
104       // We don't do any dead assignment elimination if there are no assigns
105       // to eliminate. :)
106       if (!NodeUtil.has(fnBlock, matchRemovableAssigns,
107               Predicates.<Node>alwaysTrue())) {
108         return;
109       }
110 
111       // Computes liveness information first.
112       ControlFlowGraph<Node> cfg = t.getControlFlowGraph();
113       liveness = new LiveVariablesAnalysis(cfg, scope, compiler);
114       liveness.analyze();
115       tryRemoveDeadAssignments(t, cfg);
116     }
117   }

119   @Override

123   /**
124    * Try to remove useless assignments from a control flow graph that has been
125    * annotated with liveness information.
126    *
127    * @param t The node traversal.
128    * @param cfg The control flow graph of the program annotated with liveness
129    *        information.
130    */
131   private void tryRemoveDeadAssignments(NodeTraversal t,
132       ControlFlowGraph<Node> cfg) {
133     Iterable<DiGraphNode<Node, Branch>> nodes = cfg.getDirectedGraphNodes();
134 
135     for (DiGraphNode<Node, Branch> cfgNode : nodes) {
136       FlowState<LiveVariableLattice> state =
137           cfgNode.getAnnotation();
138       Node n = cfgNode.getValue();
139       if (n == null) {
140         continue;
141       }
142       switch (n.getType()) {
143         case Token.IF:
144         case Token.WHILE:
145         case Token.DO:
146           tryRemoveAssignment(t, NodeUtil.getConditionExpression(n), state);
147           continue;
148         case Token.FOR:
149           if (!NodeUtil.isForIn(n)) {
150             tryRemoveAssignment(
151                 t, NodeUtil.getConditionExpression(n), state);
152           }
153           continue;
154         case Token.SWITCH:
155         case Token.CASE:
156         case Token.RETURN:
157           if (n.hasChildren()) {
158             tryRemoveAssignment(t, n.getFirstChild(), state);
159           }
160           continue;
161         // TODO(user): case Token.VAR: Remove var a=1;a=2;.....
162       }
163 
164       tryRemoveAssignment(t, n, state);
165     }
166   }
167 
168   private void tryRemoveAssignment(NodeTraversal t, Node n,
169       FlowState<LiveVariableLattice> state) {
170     tryRemoveAssignment(t, n, n, state);
171   }

173   /**
174    * Determines if any local variables are dead after the instruction {@code n}
175    * and are assigned within the subtree of {@code n}. Removes those assignments
176    * if there are any.
177    *
178    * @param n Target instruction.
179    * @param exprRoot The CFG node where the liveness information in state is
180    *     still correct.
181    * @param state The liveness information at {@code n}.
182    */
183   private void tryRemoveAssignment(NodeTraversal t, Node n, Node exprRoot,
184       FlowState<LiveVariableLattice> state) {
185 
186     Node parent = n.getParent();
187 
188     if (NodeUtil.isAssignmentOp(n) ||
189         n.isInc() || n.isDec()) {
190 
191       Node lhs = n.getFirstChild();
192       Node rhs = lhs.getNext();
193 
194       // Recurse first. Example: dead_x = dead_y = 1; We try to clean up dead_y
195       // first.
196       if (rhs != null) {
197         tryRemoveAssignment(t, rhs, exprRoot, state);
198         rhs = lhs.getNext();
199       }
200 
201       Scope scope = t.getScope();
202       if (!lhs.isName()) {
203         return; // Not a local variable assignment.
204       }
205       String name = lhs.getString();
206       if (!scope.isDeclared(name, false)) {
207         return;
208       }
209       Var var = scope.getVar(name);
210 
211       if (liveness.getEscapedLocals().contains(var)) {
212         return; // Local variable that might be escaped due to closures.
213       }
214 
215       // If we have an identity assignment such as a=a, always remove it
216       // regardless of what the liveness results because it
217       // does not change the result afterward.
218       if (rhs != null &&
219           rhs.isName() &&
220           rhs.getString().equals(var.name) &&
221           n.isAssign()) {
222         n.removeChild(rhs);
223         n.getParent().replaceChild(n, rhs);
224         compiler.reportCodeChange();
225         return;
226       }
227 
228       if (state.getOut().isLive(var)) {
229         return; // Variable not dead.
230       }
231 
232       if (state.getIn().isLive(var) &&
233           isVariableStillLiveWithinExpression(n, exprRoot, var.name)) {
234         // The variable is killed here but it is also live before it.
235         // This is possible if we have say:
236         //    if (X = a && a = C) {..} ; .......; a = S;
237         // In this case we are safe to remove "a = C" because it is dead.
238         // However if we have:
239         //    if (a = C && X = a) {..} ; .......; a = S;
240         // removing "a = C" is NOT correct, although the live set at the node
241         // is exactly the same.
242         // TODO(user): We need more fine grain CFA or we need to keep track
243         // of GEN sets when we recurse here.
244         return;
245       }
246 
247       if (n.isAssign()) {
248         n.removeChild(rhs);
249         n.getParent().replaceChild(n, rhs);
250       } else if (NodeUtil.isAssignmentOp(n)) {
251         n.removeChild(rhs);
252         n.removeChild(lhs);
253         Node op = new Node(NodeUtil.getOpFromAssignmentOp(n), lhs, rhs);
254         parent.replaceChild(n, op);
255       } else if (n.isInc() || n.isDec()) {
256         if (parent.isExprResult()) {
257           parent.replaceChild(n,
258               IR.voidNode(IR.number(0).srcref(n)));
259         } else if (n.isComma() && n != parent.getLastChild()) {
260           parent.removeChild(n);
261         } else if (parent.isFor() && !NodeUtil.isForIn(parent) &&
262             NodeUtil.getConditionExpression(parent) != n) {
263           parent.replaceChild(n, IR.empty());
264         } else {
265           // Cannot replace x = a++ with x = a because that's not valid
266           // when a is not a number.
267           return;
268         }
269       } else {
270         // Not reachable.
271         throw new IllegalStateException("Unknown statement");
272       }
273 
274       compiler.reportCodeChange();
275       return;
276 
277     } else {
278       for (Node c = n.getFirstChild(); c != null;) {
279         Node next = c.getNext();
280         if (!ControlFlowGraph.isEnteringNewCfgNode(c)) {
281           tryRemoveAssignment(t, c, exprRoot, state);
282         }
283         c = next;
284       }
285       return;
286     }
287   }

289   /**
290    * Given a variable, node n in the tree and a sub-tree denoted by exprRoot as
291    * the root, this function returns true if there exists a read of that
292    * variable before a write to that variable that is on the right side of n.
293    *
294    * For example, suppose the node is x = 1:
295    *
296    * y = 1, x = 1; // false, there is no reads at all.
297    * y = 1, x = 1, print(x) // true, there is a read right of n.
298    * y = 1, x = 1, x = 2, print(x) // false, there is a read right of n but
299    *                               // it is after a write.
300    *
301    * @param n The current node we should look at.
302    * @param exprRoot The node
303    */
304   private boolean isVariableStillLiveWithinExpression(
305       Node n, Node exprRoot, String variable) {
306     while (n != exprRoot) {
307       VariableLiveness state = VariableLiveness.MAYBE_LIVE;
308       switch (n.getParent().getType()) {
309         case Token.OR:
310         case Token.AND:
311           // If the currently node is the first child of
312           // AND/OR, be conservative only consider the READs
313           // of the second operand.
314           if (n.getNext() != null) {
315             state = isVariableReadBeforeKill(
316                 n.getNext(), variable);
317             if (state == VariableLiveness.KILL) {
318               state = VariableLiveness.MAYBE_LIVE;
319             }
320           }
321           break;
322 
323         case Token.HOOK:
324           // If current node is the condition, check each following
325           // branch, otherwise it is a conditional branch and the
326           // other branch can be ignored.
327           if (n.getNext() != null && n.getNext().getNext() != null) {
328             state = checkHookBranchReadBeforeKill(
329                 n.getNext(), n.getNext().getNext(), variable);
330           }
331           break;
332 
333         default:
334           for (Node sibling = n.getNext(); sibling != null;
335                sibling = sibling.getNext()) {
336             state = isVariableReadBeforeKill(sibling, variable);
337             if (state != VariableLiveness.MAYBE_LIVE) {
338               break;
339             }
340           }
341       }
342 
343       // If we see a READ or KILL there is no need to continue.
344       if (state == VariableLiveness.READ) {
345         return true;
346       } else if (state == VariableLiveness.KILL) {
347         return false;
348       }
349       n = n.getParent();
350     }
351     return false;
352   }

354   // The current liveness of the variable
355   private enum VariableLiveness {
356     MAYBE_LIVE, // May be still live in the current expression tree.
357     READ, // Known there is a read left of it.
358     KILL, // Known there is a write before any read.
359   }
360 
361   /**
362    * Give an expression and a variable. It returns READ, if the first
363    * reference of that variable is a read. It returns KILL, if the first
364    * reference of that variable is an assignment. It returns MAY_LIVE otherwise.
365    */
366   private VariableLiveness isVariableReadBeforeKill(
367       Node n, String variable) {
368     if (ControlFlowGraph.isEnteringNewCfgNode(n)) { // Not a FUNCTION
369       return VariableLiveness.MAYBE_LIVE;
370     }
371 
372     if (n.isName() && variable.equals(n.getString())) {
373       if (NodeUtil.isVarOrSimpleAssignLhs(n, n.getParent())) {
374         Preconditions.checkState(n.getParent().isAssign());
375         // The expression to which the assignment is made is evaluated before
376         // the RHS is evaluated (normal left to right evaluation) but the KILL
377         // occurs after the RHS is evaluated.
378         Node rhs = n.getNext();
379         VariableLiveness state = isVariableReadBeforeKill(rhs, variable);
380         if (state == VariableLiveness.READ) {
381           return state;
382         }
383         return VariableLiveness.KILL;
384       } else {
385         return VariableLiveness.READ;
386       }
387     }
388 
389     switch (n.getType()) {
390       // Conditionals
391       case Token.OR:
392       case Token.AND:
393         VariableLiveness v1 = isVariableReadBeforeKill(
394           n.getFirstChild(), variable);
395         VariableLiveness v2 = isVariableReadBeforeKill(
396           n.getLastChild(), variable);
397         // With a AND/OR the first branch always runs, but the second is
398         // may not.
399         if (v1 != VariableLiveness.MAYBE_LIVE) {
400           return v1;
401         } else if (v2 == VariableLiveness.READ) {
402           return VariableLiveness.READ;
403         } else {
404           return VariableLiveness.MAYBE_LIVE;
405         }
406       case Token.HOOK:
407         VariableLiveness first = isVariableReadBeforeKill(
408             n.getFirstChild(), variable);
409         if (first != VariableLiveness.MAYBE_LIVE) {
410           return first;
411         }
412         return checkHookBranchReadBeforeKill(
413             n.getFirstChild().getNext(), n.getLastChild(), variable);
414 
415       default:
416         // Expressions are evaluated left-right, depth first.
417         for (Node child = n.getFirstChild();
418             child != null; child = child.getNext()) {
419           VariableLiveness state = isVariableReadBeforeKill(child, variable);
420           if (state != VariableLiveness.MAYBE_LIVE) {
421             return state;
422           }
423         }
424     }
425 
426     return VariableLiveness.MAYBE_LIVE;
427   }
428 
429   private VariableLiveness checkHookBranchReadBeforeKill(
430       Node trueCase, Node falseCase, String variable) {
431     VariableLiveness v1 = isVariableReadBeforeKill(
432       trueCase, variable);
433     VariableLiveness v2 = isVariableReadBeforeKill(
434       falseCase, variable);
435     // With a hook it is unknown which branch will run, so
436     // we must be conservative.  A read by either is a READ, and
437     // a KILL is only considered if both KILL.
438     if (v1 == VariableLiveness.READ || v2 == VariableLiveness.READ) {
439       return VariableLiveness.READ;
440     } else if (v1 == VariableLiveness.KILL && v2 == VariableLiveness.KILL) {
441       return VariableLiveness.KILL;
442     } else {
443       return VariableLiveness.MAYBE_LIVE;
444     }
445   }
