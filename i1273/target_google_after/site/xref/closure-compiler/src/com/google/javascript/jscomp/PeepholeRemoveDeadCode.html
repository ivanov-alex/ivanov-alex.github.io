29  /**
30   * Peephole optimization to remove useless code such as IF's with false
31   * guard conditions, comma operator left hand sides with no side effects, etc.
32   *
33   */
34  class PeepholeRemoveDeadCode extends AbstractPeepholeOptimization {
35  
36    // TODO(dcc): Some (all) of these can probably be better achieved
37    // using the control flow graph (like CheckUnreachableCode).
38    // There is an existing CFG pass (UnreachableCodeElimination) that
39    // could be changed to use code from CheckUnreachableCode to do this.
40  
41    @Override
42    Node optimizeSubtree(Node subtree) {
43      switch(subtree.getType()) {
44        case Token.ASSIGN:
45          return tryFoldAssignment(subtree);
46        case Token.COMMA:
47          return tryFoldComma(subtree);
48        case Token.SCRIPT:
49        case Token.BLOCK:
50          return tryOptimizeBlock(subtree);
51        case Token.EXPR_RESULT:
52          subtree = tryFoldExpr(subtree);
53          return subtree;
54        case Token.HOOK:
55          return tryFoldHook(subtree);
56        case Token.SWITCH:
57          return tryOptimizeSwitch(subtree);
58        case Token.IF:
59          return tryFoldIf(subtree);
60        case Token.WHILE:
61          return tryFoldWhile(subtree);
62        case Token.FOR: {
63          Node condition = NodeUtil.getConditionExpression(subtree);
64          if (condition != null) {
65            tryFoldForCondition(condition);
66          }
67          return tryFoldFor(subtree);
68        }
69        case Token.DO:
70          Node foldedDo = tryFoldDoAway(subtree);
71          if (foldedDo.isDo()) {
72            return tryFoldEmptyDo(foldedDo);
73          }
74          return foldedDo;
75  
76        case Token.TRY:
77          return tryFoldTry(subtree);
78        default:
79            return subtree;
80      }
81    }

83    /**
84     * Remove try blocks without catch blocks and with empty or not
85     * existent finally blocks.
86     * Or, only leave the finally blocks if try body blocks are empty
87     * @return the replacement node, if changed, or the original if not
88     */
89    private Node tryFoldTry(Node n) {
90      Preconditions.checkState(n.isTry());
91      Node body = n.getFirstChild();
92      Node catchBlock = body.getNext();
93      Node finallyBlock = catchBlock.getNext();
94  
95      // Removes TRYs that had its CATCH removed and/or empty FINALLY.
96      if (!catchBlock.hasChildren() &&
97          (finallyBlock == null || !finallyBlock.hasChildren())) {
98        n.removeChild(body);
99        n.getParent().replaceChild(n, body);
100       reportCodeChange();
101       return body;
102     }
103 
104     // Only leave FINALLYs if TRYs are empty
105     if (!body.hasChildren()) {
106       NodeUtil.redeclareVarsInsideBranch(catchBlock);
107       if (finallyBlock != null) {
108         n.removeChild(finallyBlock);
109         n.getParent().replaceChild(n, finallyBlock);
110       } else {
111         n.getParent().removeChild(n);
112       }
113       reportCodeChange();
114       return finallyBlock;
115     }
116 
117     return n;
118   }

120   /**
121    * Try removing identity assignments
122    * @return the replacement node, if changed, or the original if not
123    */
124   private Node tryFoldAssignment(Node subtree) {
125     Preconditions.checkState(subtree.isAssign());
126     Node left = subtree.getFirstChild();
127     Node right = subtree.getLastChild();
128     // Only names
129     if (left.isName()
130         && right.isName()
131         && left.getString().equals(right.getString())) {
132       subtree.getParent().replaceChild(subtree, right.detachFromParent());
133       reportCodeChange();
134       return right;
135     }
136     return subtree;
137   }

139   /**
140    * Try folding EXPR_RESULT nodes by removing useless Ops and expressions.
141    * @return the replacement node, if changed, or the original if not
142    */
143   private Node tryFoldExpr(Node subtree) {
144     Node result = trySimplifyUnusedResult(subtree.getFirstChild());
145     if (result == null) {
146       Node parent = subtree.getParent();
147       // If the EXPR_RESULT no longer has any children, remove it as well.
148       if (parent.isLabel()) {
149         Node replacement = IR.block().srcref(subtree);
150         parent.replaceChild(subtree, replacement);
151         subtree = replacement;
152       } else {
153         subtree.detachFromParent();
154         subtree = null;
155       }
156     }
157     return subtree;
158   }

160   /**
161    * General cascading unused operation node removal.
162    * @param n The root of the expression to simplify.
163    * @return The replacement node, or null if the node was is not useful.
164    */
165   private Node trySimplifyUnusedResult(Node n) {
166     return trySimplifyUnusedResult(n, true);
167   }

169   /**
170    * General cascading unused operation node removal.
171    * @param n The root of the expression to simplify.
172    * @param removeUnused If true, the node is removed from the AST if
173    *     it is not useful, otherwise it replaced with an EMPTY node.
174    * @return The replacement node, or null if the node was is not useful.
175    */
176   private Node trySimplifyUnusedResult(Node n, boolean removeUnused) {
177     Node result = n;
178 
179     // Simplify the results of conditional expressions
180     switch (n.getType()) {
181       case Token.HOOK:
182         Node trueNode = trySimplifyUnusedResult(n.getFirstChild().getNext());
183         Node falseNode = trySimplifyUnusedResult(n.getLastChild());
184         // If one or more of the conditional children were removed,
185         // transform the HOOK to an equivalent operation:
186         //    x() ? foo() : 1 --> x() && foo()
187         //    x() ? 1 : foo() --> x() || foo()
188         //    x() ? 1 : 1 --> x()
189         //    x ? 1 : 1 --> null
190         if (trueNode == null && falseNode != null) {
191           n.setType(Token.OR);
192           Preconditions.checkState(n.getChildCount() == 2);
193         } else if (trueNode != null && falseNode == null) {
194           n.setType(Token.AND);
195           Preconditions.checkState(n.getChildCount() == 2);
196         } else if (trueNode == null && falseNode == null) {
197           result = trySimplifyUnusedResult(n.getFirstChild());
198         } else {
199           // The structure didn't change.
200           result = n;
201         }
202         break;
203       case Token.AND:
204       case Token.OR:
205         // Try to remove the second operand from a AND or OR operations:
206         //    x() || f --> x()
207         //    x() && f --> x()
208         Node conditionalResultNode = trySimplifyUnusedResult(
209             n.getLastChild());
210         if (conditionalResultNode == null) {
211           Preconditions.checkState(n.hasOneChild());
212           // The conditionally executed code was removed, so
213           // replace the AND/OR with its LHS or remove it if it isn't useful.
214           result = trySimplifyUnusedResult(n.getFirstChild());
215         }
216         break;
217       case Token.FUNCTION:
218         // A function expression isn't useful if it isn't used, remove it and
219         // don't bother to look at its children.
220         result = null;
221         break;
222       case Token.COMMA:
223         // We rewrite other operations as COMMA expressions (which will later
224         // get split into individual EXPR_RESULT statement, if possible), so
225         // we special case COMMA (we don't want to rewrite COMMAs as new COMMAs
226         // nodes.
227         Node left = trySimplifyUnusedResult(n.getFirstChild());
228         Node right = trySimplifyUnusedResult(n.getLastChild());
229         if (left == null && right == null) {
230           result = null;
231         } else if (left == null) {
232           result = right;
233         } else if (right == null){
234           result = left;
235         } else {
236           // The structure didn't change.
237           result = n;
238         }
239         break;
240       default:
241         if (!nodeTypeMayHaveSideEffects(n)) {
242           // This is the meat of this function. The node itself doesn't generate
243           // any side-effects but preserve any side-effects in the children.
244           Node resultList = null;
245           for (Node next, c = n.getFirstChild(); c != null; c = next) {
246             next = c.getNext();
247             c = trySimplifyUnusedResult(c);
248             if (c != null) {
249               c.detachFromParent();
250               if (resultList == null)  {
251                 // The first side-effect can be used stand-alone.
252                 resultList = c;
253               } else {
254                 // Leave the side-effects in-place, simplifying it to a COMMA
255                 // expression.
256                 resultList = IR.comma(resultList, c).srcref(c);
257               }
258             }
259           }
260           result = resultList;
261         }
262     }
263 
264     // Fix up the AST, replace or remove the an unused node (if requested).
265     if (n != result) {
266       Node parent = n.getParent();
267       if (result == null) {
268         if (removeUnused) {
269           parent.removeChild(n);
270         } else {
271           result = IR.empty().srcref(n);
272           parent.replaceChild(n, result);
273         }
274       } else {
275         // A new COMMA expression may not have an existing parent.
276         if (result.getParent() != null) {
277           result.detachFromParent();
278         }
279         n.getParent().replaceChild(n, result);
280       }
281       reportCodeChange();
282     }
283 
284     return result;
285   }

299   /**
300    * Remove useless switches and cases.
301    */
302   private Node tryOptimizeSwitch(Node n) {
303     Preconditions.checkState(n.isSwitch());
304 
305     Node defaultCase = tryOptimizeDefaultCase(n);
306 
307     // Removing cases when there exists a default case is not safe.
308     if (defaultCase == null) {
309       Node cond = n.getFirstChild(), prev = null, next = null, cur;
310 
311       for (cur = cond.getNext(); cur != null; cur = next) {
312         next = cur.getNext();
313         if (!mayHaveSideEffects(cur.getFirstChild()) &&
314             isUselessCase(cur, prev)) {
315           removeCase(n, cur);
316         } else {
317           prev = cur;
318         }
319       }
320 
321       // Optimize switches with constant condition
322       if (NodeUtil.isLiteralValue(cond, false)) {
323         Node caseLabel;
324         TernaryValue caseMatches = TernaryValue.TRUE;
325         // Remove cases until you find one that may match
326         for (cur = cond.getNext(); cur != null; cur = next) {
327           next = cur.getNext();
328           caseLabel = cur.getFirstChild();
329           caseMatches = PeepholeFoldConstants.evaluateComparison(
330               Token.SHEQ, cond, caseLabel);
331           if (caseMatches == TernaryValue.TRUE) {
332             break;
333           } else if (caseMatches == TernaryValue.UNKNOWN) {
334             break;
335           } else {
336             removeCase(n, cur);
337           }
338         }
339         if (caseMatches != TernaryValue.UNKNOWN) {
340           Node block, lastStm;
341           // Skip cases until you find one whose last stm is a removable break
342           while (cur != null) {
343             block = cur.getLastChild();
344             lastStm = block.getLastChild();
345             cur = cur.getNext();
346             if (lastStm != null
347                 && lastStm.isBreak()
348                 && !lastStm.hasChildren()) {
349               block.removeChild(lastStm);
350               reportCodeChange();
351               break;
352             }
353           }
354           // Remove any remaining cases
355           for (; cur != null; cur = next) {
356             next = cur.getNext();
357             removeCase(n, cur);
358           }
359           // If there is one case left, we may be able to fold it
360           cur = cond.getNext();
361           if (cur != null && cur.getNext() == null) {
362             block = cur.getLastChild();
363             if (!(NodeUtil.has(block, MATCH_UNNAMED_BREAK,
364                 NodeUtil.MATCH_NOT_FUNCTION))) {
365               cur.removeChild(block);
366               block.setIsSyntheticBlock(false);
367               n.getParent().replaceChild(n, block);
368               reportCodeChange();
369               return block;
370             }
371           }
372         }
373       }
374     }
375 
376     // Remove the switch if there are no remaining cases.
377     if (n.hasOneChild()) {
378       Node condition = n.removeFirstChild();
379       Node replacement = IR.exprResult(condition).srcref(n);
380       n.getParent().replaceChild(n, replacement);
381       reportCodeChange();
382       return replacement;
383     }
384 
385     return null;
386   }

388   /**
389    * @return the default case node or null if there is no default case or
390    *     if the default case is removed.
391    */
392   private Node tryOptimizeDefaultCase(Node n) {
393     Preconditions.checkState(n.isSwitch());
394 
395     Node lastNonRemovable = n.getFirstChild();  // The switch condition
396 
397     // The first child is the switch conditions skip it when looking for cases.
398     for (Node c = n.getFirstChild().getNext(); c != null; c = c.getNext()) {
399       if (c.isDefaultCase()) {
400         // Remove cases that fall-through to the default case
401         Node caseToRemove = lastNonRemovable.getNext();
402         for (Node next; caseToRemove != c; caseToRemove = next) {
403           next = caseToRemove.getNext();
404           removeCase(n, caseToRemove);
405         }
406 
407         // Don't use the switch condition as the previous case.
408         Node prevCase = (lastNonRemovable == n.getFirstChild())
409             ? null : lastNonRemovable;
410 
411         // Remove the default case if we can
412         if (isUselessCase(c, prevCase)) {
413           removeCase(n, c);
414           return null;
415         }
416         return c;
417       } else {
418         Preconditions.checkState(c.isCase());
419         if (c.getLastChild().hasChildren()
420             || mayHaveSideEffects(c.getFirstChild())) {
421           lastNonRemovable = c;
422         }
423       }
424     }
425     return null;
426   }

428   /**
429    * Remove the case from the switch redeclaring any variables declared in it.
430    * @param caseNode The case to remove.
431    */
432   private void removeCase(Node switchNode, Node caseNode) {
433     NodeUtil.redeclareVarsInsideBranch(caseNode);
434     switchNode.removeChild(caseNode);
435     reportCodeChange();
436   }

438   /**
439    * The function assumes that when checking a CASE node there is no
440    * DEFAULT node in the SWITCH.
441    * @return Whether the CASE or DEFAULT block does anything useful.
442    */
443   private boolean isUselessCase(Node caseNode, @Nullable Node previousCase) {
444     Preconditions.checkState(
445         previousCase == null || previousCase.getNext() == caseNode);
446     // A case isn't useless can't be useless if a previous case falls
447     // through to it unless it happens to be the last case in the switch.
448     Node switchNode = caseNode.getParent();
449     if (switchNode.getLastChild() != caseNode
450         && previousCase != null) {
451       Node previousBlock = previousCase.getLastChild();
452       if (!previousBlock.hasChildren()
453           || !isExit(previousBlock.getLastChild())) {
454         return false;
455       }
456     }
457 
458     Node executingCase = caseNode;
459     while (executingCase != null) {
460       Preconditions.checkState(executingCase.isDefaultCase()
461           || executingCase.isCase());
462       // We only expect a DEFAULT case if the case we are checking is the
463       // DEFAULT case.  Otherwise, we assume the DEFAULT case has already
464       // been removed.
465       Preconditions.checkState(caseNode == executingCase
466           || !executingCase.isDefaultCase());
467       Node block = executingCase.getLastChild();
468       Preconditions.checkState(block.isBlock());
469       if (block.hasChildren()) {
470         for (Node blockChild : block.children()) {
471           // If this is a block with a labelless break, it is useless.
472           switch (blockChild.getType()) {
473             case Token.BREAK:
474               // A break to a different control structure isn't useless.
475               return blockChild.getFirstChild() == null;
476             case Token.VAR:
477               if (blockChild.hasOneChild()
478                   && blockChild.getFirstChild().getFirstChild() == null) {
479                 // Variable declarations without initializations are OK.
480                 continue;
481               }
482               return false;
483             default:
484               return false;
485           }
486         }
487       }
488       // Look at the fallthrough case
489       executingCase = executingCase.getNext();
490     }
491     return true;
492   }

494   /**
495    * @return Whether the node is an obvious control flow exit.
496    */
497   private static boolean isExit(Node n) {
498     switch (n.getType()) {
499       case Token.BREAK:
500       case Token.CONTINUE:
501       case Token.RETURN:
502       case Token.THROW:
503         return true;
504       default:
505         return false;
506     }
507   }

509   private Node tryFoldComma(Node n) {
510     // If the left side does nothing replace the comma with the result.
511     Node parent = n.getParent();
512     Node left = n.getFirstChild();
513     Node right = left.getNext();
514 
515     left = trySimplifyUnusedResult(left);
516     if (left == null || !mayHaveSideEffects(left)) {
517       // Fold it!
518       n.removeChild(right);
519       parent.replaceChild(n, right);
520       reportCodeChange();
521       return right;
522     }
523     return n;
524   }

526   /**
527    * Try removing unneeded block nodes and their useless children
528    */
529   Node tryOptimizeBlock(Node n) {
530     // Remove any useless children
531     for (Node c = n.getFirstChild(); c != null; ) {
532       Node next = c.getNext();  // save c.next, since 'c' may be removed
533       if (!isUnremovableNode(c) && !mayHaveSideEffects(c)) {
534         // TODO(johnlenz): determine what this is actually removing. Candidates
535         //    include: EMPTY nodes, control structures without children
536         //    (removing infinite loops), empty try blocks.  What else?
537         n.removeChild(c);  // lazy kids
538         reportCodeChange();
539       } else {
540         tryOptimizeConditionalAfterAssign(c);
541       }
542       c = next;
543     }
544 
545     if (n.isSyntheticBlock() || n.isScript() || n.getParent() == null) {
546       return n;
547     }
548 
549     // Try to remove the block.
550     if (NodeUtil.tryMergeBlock(n)) {
551       reportCodeChange();
552       return null;
553     }
554 
555     return n;
556   }

562     return (n.isBlock() && n.isSyntheticBlock()) || n.isScript();

565   // TODO(johnlenz): Consider moving this to a separate peephole pass.
566   /**
567    * Attempt to replace the condition of if or hook immediately that is a
568    * reference to a name that is assigned immediately before.
569    */
570   private void tryOptimizeConditionalAfterAssign(Node n) {
571     Node next = n.getNext();
572 
573     // Look for patterns like the following and replace the if-condition with
574     // a constant value so it can later be folded:
575     //   var a = /a/;
576     //   if (a) {foo(a)}
577     // or
578     //   a = 0;
579     //   a ? foo(a) : c;
580     // or
581     //   a = 0;
582     //   a || foo(a);
583     // or
584     //   a = 0;
585     //   a && foo(a)
586     //
587     // TODO(johnlenz): This would be better handled by control-flow sensitive
588     // constant propagation. As the other case that I want to handle is:
589     //   i=0; for(;i<0;i++){}
590     // as right now nothing facilitates removing a loop like that.
591     // This is here simply to remove the cruft left behind goog.userAgent and
592     // similar cases.
593 
594     if (isSimpleAssignment(n) && isConditionalStatement(next)) {
595       Node lhsAssign = getSimpleAssignmentName(n);
596 
597       Node condition = getConditionalStatementCondition(next);
598       if (lhsAssign.isName() && condition.isName()
599           && lhsAssign.getString().equals(condition.getString())) {
600         Node rhsAssign = getSimpleAssignmentValue(n);
601         TernaryValue value = NodeUtil.getImpureBooleanValue(rhsAssign);
602         if (value != TernaryValue.UNKNOWN) {
603           Node replacementConditionNode =
604               NodeUtil.booleanNode(value.toBoolean(true));
605           condition.getParent().replaceChild(condition,
606               replacementConditionNode);
607           reportCodeChange();
608         }
609       }
610     }
611   }

613   /**
614    * @return whether the node is a assignment to a simple name, or simple var
615    * declaration with initialization.
616    */
617   private static boolean isSimpleAssignment(Node n) {
618     // For our purposes we define a simple assignment to be a assignment
619     // to a NAME node, or a VAR declaration with one child and a initializer.
620     if (NodeUtil.isExprAssign(n)
621         && n.getFirstChild().getFirstChild().isName()) {
622       return true;
623     } else if (n.isVar() && n.hasOneChild() &&
624         n.getFirstChild().getFirstChild() != null) {
625       return true;
626     }
627 
628     return false;
629   }

631   /**
632    * @return The name being assigned to.
633    */
634   private Node getSimpleAssignmentName(Node n) {
635     Preconditions.checkState(isSimpleAssignment(n));
636     if (NodeUtil.isExprAssign(n)) {
637       return n.getFirstChild().getFirstChild();
638     } else {
639       // A var declaration.
640       return n.getFirstChild();
641     }
642   }

644   /**
645    * @return The value assigned in the simple assignment
646    */
647   private Node getSimpleAssignmentValue(Node n) {
648     Preconditions.checkState(isSimpleAssignment(n));
649     return n.getFirstChild().getLastChild();

652   /**
653    * @return Whether the node is a conditional statement.
654    */
655   private boolean isConditionalStatement(Node n) {
656     // We defined a conditional statement to be a IF or EXPR_RESULT rooted with
657     // a HOOK, AND, or OR node.
658     return n != null && (n.isIf() || isExprConditional(n));
659   }

661   /**
662    * @return Whether the node is a rooted with a HOOK, AND, or OR node.
663    */
664   private static boolean isExprConditional(Node n) {
665     if (n.isExprResult()) {
666       switch (n.getFirstChild().getType()) {
667         case Token.HOOK:
668         case Token.AND:
669         case Token.OR:
670           return true;
671       }
672     }
673     return false;
674   }

676   /**
677    * @return The condition of a conditional statement.
678    */
679   private Node getConditionalStatementCondition(Node n) {
680     if (n.isIf()) {
681       return NodeUtil.getConditionExpression(n);
682     } else {
683       Preconditions.checkState(isExprConditional(n));
684       return n.getFirstChild().getFirstChild();

688   /**
689    * Try folding IF nodes by removing dead branches.
690    * @return the replacement node, if changed, or the original if not
691    */
692   private Node tryFoldIf(Node n) {
693     Preconditions.checkState(n.isIf());
694     Node parent = n.getParent();
695     Preconditions.checkNotNull(parent);
696     int type = n.getType();
697     Node cond = n.getFirstChild();
698     Node thenBody = cond.getNext();
699     Node elseBody = thenBody.getNext();
700 
701     // if (x) { .. } else { } --> if (x) { ... }
702     if (elseBody != null && !mayHaveSideEffects(elseBody)) {
703       n.removeChild(elseBody);
704       elseBody = null;
705       reportCodeChange();
706     }
707 
708     // if (x) { } else { ... } --> if (!x) { ... }
709     if (!mayHaveSideEffects(thenBody) && elseBody != null) {
710       n.removeChild(elseBody);
711       n.replaceChild(thenBody, elseBody);
712       Node notCond = new Node(Token.NOT);
713       n.replaceChild(cond, notCond);
714       notCond.addChildToFront(cond);
715       cond = notCond;
716       thenBody = cond.getNext();
717       elseBody = null;
718       reportCodeChange();
719     }
720 
721     // if (x()) { }
722     if (!mayHaveSideEffects(thenBody) && elseBody == null) {
723       if (mayHaveSideEffects(cond)) {
724         // x() has side effects, just leave the condition on its own.
725         n.removeChild(cond);
726         Node replacement = NodeUtil.newExpr(cond);
727         parent.replaceChild(n, replacement);
728         reportCodeChange();
729         return replacement;
730       } else {
731         // x() has no side effects, the whole tree is useless now.
732         NodeUtil.removeChild(parent, n);
733         reportCodeChange();
734         return null;
735       }
736     }
737 
738     // Try transforms that apply to both IF and HOOK.
739     TernaryValue condValue = NodeUtil.getImpureBooleanValue(cond);
740     if (condValue == TernaryValue.UNKNOWN) {
741       return n;  // We can't remove branches otherwise!
742     }
743 
744     if (mayHaveSideEffects(cond)) {
745       // Transform "if (a = 2) {x =2}" into "if (true) {a=2;x=2}"
746       boolean newConditionValue = condValue == TernaryValue.TRUE;
747       // Add an elseBody if it is needed.
748       if (!newConditionValue && elseBody == null) {
749         elseBody = IR.block().srcref(n);
750         n.addChildToBack(elseBody);
751       }
752       Node newCond = NodeUtil.booleanNode(newConditionValue);
753       n.replaceChild(cond, newCond);
754       Node branchToKeep = newConditionValue ? thenBody : elseBody;
755       branchToKeep.addChildToFront(IR.exprResult(cond).srcref(cond));
756       reportCodeChange();
757       cond = newCond;
758     }
759 
760     boolean condTrue = condValue.toBoolean(true);
761     if (n.getChildCount() == 2) {
762       Preconditions.checkState(type == Token.IF);
763 
764       if (condTrue) {
765         // Replace "if (true) { X }" with "X".
766         Node thenStmt = n.getFirstChild().getNext();
767         n.removeChild(thenStmt);
768         parent.replaceChild(n, thenStmt);
769         reportCodeChange();
770         return thenStmt;
771       } else {
772         // Remove "if (false) { X }" completely.
773         NodeUtil.redeclareVarsInsideBranch(n);
774         NodeUtil.removeChild(parent, n);
775         reportCodeChange();
776         return null;
777       }
778     } else {
779       // Replace "if (true) { X } else { Y }" with X, or
780       // replace "if (false) { X } else { Y }" with Y.
781       Node trueBranch = n.getFirstChild().getNext();
782       Node falseBranch = trueBranch.getNext();
783       Node branchToKeep = condTrue ? trueBranch : falseBranch;
784       Node branchToRemove = condTrue ? falseBranch : trueBranch;
785       NodeUtil.redeclareVarsInsideBranch(branchToRemove);
786       n.removeChild(branchToKeep);
787       parent.replaceChild(n, branchToKeep);
788       reportCodeChange();
789       return branchToKeep;
790     }
791   }

793   /**
794    * Try folding HOOK (?:) if the condition results of the condition is known.
795    * @return the replacement node, if changed, or the original if not
796    */
797   private Node tryFoldHook(Node n) {
798     Preconditions.checkState(n.isHook());
799     Node parent = n.getParent();
800     Preconditions.checkNotNull(parent);
801     Node cond = n.getFirstChild();
802     Node thenBody = cond.getNext();
803     Node elseBody = thenBody.getNext();
804 
805     TernaryValue condValue = NodeUtil.getImpureBooleanValue(cond);
806     if (condValue == TernaryValue.UNKNOWN) {
807       // If the result nodes are equivalent, then one of the nodes can be
808       // removed and it doesn't matter which.
809       if (!areNodesEqualForInlining(thenBody, elseBody)) {
810         return n;  // We can't remove branches otherwise!
811       }
812     }
813 
814     // Transform "(a = 2) ? x =2 : y" into "a=2,x=2"
815     Node branchToKeep = condValue.toBoolean(true) ? thenBody : elseBody;
816     Node replacement;
817     boolean condHasSideEffects = mayHaveSideEffects(cond);
818     // Must detach after checking for side effects, to ensure that the parents
819     // of nodes are set correctly.
820     n.detachChildren();
821 
822     if (condHasSideEffects) {
823       replacement = IR.comma(cond, branchToKeep).srcref(n);
824     } else {
825       replacement = branchToKeep;
826     }
827 
828     parent.replaceChild(n, replacement);
829     reportCodeChange();
830     return replacement;
831   }

833   /**
834    * Removes WHILEs that always evaluate to false.
835    */
836   Node tryFoldWhile(Node n) {
837     Preconditions.checkArgument(n.isWhile());
838     Node cond = NodeUtil.getConditionExpression(n);
839     if (NodeUtil.getPureBooleanValue(cond) != TernaryValue.FALSE) {
840       return n;
841     }
842     NodeUtil.redeclareVarsInsideBranch(n);
843     NodeUtil.removeChild(n.getParent(), n);
844     reportCodeChange();
845 
846     return null;
847   }

849   /**
850    * Removes FORs that always evaluate to false.
851    */
852   Node tryFoldFor(Node n) {
853     Preconditions.checkArgument(n.isFor());
854     // If this is a FOR-IN loop skip it.
855     if (NodeUtil.isForIn(n)) {
856       return n;
857     }
858 
859     Node init = n.getFirstChild();
860     Node cond = init.getNext();
861     Node increment = cond.getNext();
862 
863     if (!init.isEmpty() && !init.isVar()) {
864       init = trySimplifyUnusedResult(init, false);
865     }
866 
867     if (!increment.isEmpty()) {
868       increment = trySimplifyUnusedResult(increment, false);
869     }
870 
871     // There is an initializer skip it
872     if (!n.getFirstChild().isEmpty()) {
873       return n;
874     }
875 
876     if (NodeUtil.getImpureBooleanValue(cond) != TernaryValue.FALSE) {
877       return n;
878     }
879 
880     Node parent = n.getParent();
881     NodeUtil.redeclareVarsInsideBranch(n);
882     if (!mayHaveSideEffects(cond)) {
883       NodeUtil.removeChild(parent, n);
884     } else {
885       Node statement = IR.exprResult(cond.detachFromParent())
886           .useSourceInfoIfMissingFrom(cond);
887       if (parent.isLabel()) {
888         Node block = IR.block();
889         block.useSourceInfoIfMissingFrom(statement);
890         block.addChildToFront(statement);
891         statement = block;
892       }
893       parent.replaceChild(n, statement);
894     }
895     reportCodeChange();
896     return null;
897   }

899   /**
900    * Removes DOs that always evaluate to false. This leaves the
901    * statements that were in the loop in a BLOCK node.
902    * The block will be removed in a later pass, if possible.
903    */
904   Node tryFoldDoAway(Node n) {
905     Preconditions.checkArgument(n.isDo());
906 
907     Node cond = NodeUtil.getConditionExpression(n);
908     if (NodeUtil.getImpureBooleanValue(cond) != TernaryValue.FALSE) {
909       return n;
910     }
911 
912     // TODO(johnlenz): The do-while can be turned into a label with
913     // named breaks and the label optimized away (maybe).
914     if (hasBreakOrContinue(n)) {
915       return n;
916     }
917 
918     Preconditions.checkState(
919         NodeUtil.isControlStructureCodeBlock(n, n.getFirstChild()));
920     Node block = n.removeFirstChild();
921 
922     Node parent =  n.getParent();
923     parent.replaceChild(n, block);
924     if (mayHaveSideEffects(cond)) {
925       Node condStatement = IR.exprResult(cond.detachFromParent())
926           .srcref(cond);
927       parent.addChildAfter(condStatement, block);
928     }
929     reportCodeChange();
930 
931     return block;
932   }

934   /**
935    * Removes DOs that have empty bodies into FORs, which are
936    * much easier for the CFA to analyze.
937    */
938   Node tryFoldEmptyDo(Node n) {
939     Preconditions.checkArgument(n.isDo());
940 
941     Node body = NodeUtil.getLoopCodeBlock(n);
942     if (body.isBlock() && !body.hasChildren()) {
943       Node cond = NodeUtil.getConditionExpression(n);
944       Node whileNode =
945           IR.forNode(IR.empty().srcref(n),
946                      cond.detachFromParent(),
947                      IR.empty().srcref(n),
948                      body.detachFromParent())
949           .srcref(n);
950       n.getParent().replaceChild(n, whileNode);
951       reportCodeChange();
952       return whileNode;
953     }
954     return n;
955   }

957   /**
958    *
959    */
960   static boolean hasBreakOrContinue(Node n) {
961     // TODO(johnlenz): This is overkill as named breaks may refer to outer
962     // loops or labels, and any break my refer to an inner loop.
963     // More generally, this check may be more expensive than we like.
964     return NodeUtil.has(
965         n,
966         Predicates.or(
967             new NodeUtil.MatchNodeType(Token.BREAK),
968             new NodeUtil.MatchNodeType(Token.CONTINUE)),
969         NodeUtil.MATCH_NOT_FUNCTION);
970   }

973    * Remove always true loop conditions.
974    */
975   private void tryFoldForCondition(Node forCondition) {
976     if (NodeUtil.getPureBooleanValue(forCondition) == TernaryValue.TRUE) {
977       forCondition.getParent().replaceChild(forCondition,
978           IR.empty());
979       reportCodeChange();
