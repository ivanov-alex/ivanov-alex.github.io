17  package com.google.javascript.jscomp.parsing;

19  import com.google.common.base.Preconditions;
20  import com.google.javascript.rhino.TokenUtil;

22  /**
23   * This class implements the scanner for JsDoc strings.
24   *
25   * It is heavily based on Rhino's TokenStream.
26   *
27   */
28  class JsDocTokenStream {
29    /*
30     * For chars - because we need something out-of-range
31     * to check.  (And checking EOF by exception is annoying.)
32     * Note distinction from EOF token type!
33     */
34    private static final int EOF_CHAR = -1;
35  
36    JsDocTokenStream(String sourceString) {
37      this(sourceString, 0);
38    }
39  
40    JsDocTokenStream(String sourceString, int lineno) {
41      this(sourceString, lineno, 0);
42    }
43  
44    JsDocTokenStream(String sourceString, int lineno, int initCharno) {
45      Preconditions.checkNotNull(sourceString);
46      this.lineno = lineno;
47      this.sourceString = sourceString;
48      this.sourceEnd = sourceString.length();
49      this.sourceCursor = this.cursor = 0;
50      this.initLineno = lineno;
51      this.initCharno = initCharno;
52    }
53  
54    /**
55     * Tokenizes JSDoc comments.
56     */
57    @SuppressWarnings("fallthrough")
58    final JsDocToken getJsDocToken() {
59      int c;
60      stringBufferTop = 0;
61      for (;;) {
62        // eat white spaces
63        for (;;) {
64          charno = -1;
65          c = getChar();
66          if (c == EOF_CHAR) {
67            return JsDocToken.EOF;
68          } else if (c == '\n') {
69            return JsDocToken.EOL;
70          } else if (!TokenUtil.isJSSpace(c)) {
71            break;
72          }
73        }
74  
75        switch (c) {
76          // annotation, e.g. @type or @constructor
77          case '@':
78            do {
79              c = getChar();
80              if (isAlpha(c)) {
81                addToString(c);
82              } else {
83                ungetChar(c);
84                this.string = getStringFromBuffer();
85                stringBufferTop = 0;
86                return JsDocToken.ANNOTATION;
87              }
88            } while (true);
89  
90          case '*':
91            if (matchChar('/')) {
92              return JsDocToken.EOC;
93            } else {
94              return JsDocToken.STAR;
95            }
96  
97          case ',':
98            return JsDocToken.COMMA;
99  
100         case '>':
101           return JsDocToken.RIGHT_ANGLE;
102 
103         case '(':
104           return JsDocToken.LEFT_PAREN;
105 
106         case ')':
107           return JsDocToken.RIGHT_PAREN;
108 
109         case '{':
110           return JsDocToken.LEFT_CURLY;
111 
112         case '}':
113           return JsDocToken.RIGHT_CURLY;
114 
115         case '[':
116           return JsDocToken.LEFT_SQUARE;
117 
118         case ']':
119           return JsDocToken.RIGHT_SQUARE;
120 
121         case '?':
122           return JsDocToken.QMARK;
123 
124         case '!':
125           return JsDocToken.BANG;
126 
127         case ':':
128           return JsDocToken.COLON;
129 
130         case '=':
131           return JsDocToken.EQUALS;
132 
133         case '|':
134           return JsDocToken.PIPE;
135 
136         case '<':
137           return JsDocToken.LEFT_ANGLE;
138 
139         case '.':
140           c = getChar();
141           if (c == '<') {
142             return JsDocToken.LEFT_ANGLE;
143           } else {
144             if (c == '.') {
145               c = getChar();
146               if (c == '.') {
147                 return JsDocToken.ELLIPSIS;
148               } else {
149                 addToString('.');
150               }
151             }
152             // we may backtrack across line boundary
153             ungetBuffer[ungetCursor++] = c;
154             c = '.';
155           }
156           // fall through
157 
158         default: {
159           // recognize a JsDoc string but discard last . if it is followed by
160           // a non-JsDoc comment char, e.g. Array.<
161           int c1 = c;
162           addToString(c);
163           int c2 = getChar();
164           if (!isJSDocString(c2)) {
165             ungetChar(c2);
166             this.string = getStringFromBuffer();
167             stringBufferTop = 0;
168             return JsDocToken.STRING;
169           } else {
170             do {
171               c1 = c2;
172               c2 = getChar();
173               if (c1 == '.' && c2 == '<') {
174                 ungetChar(c2);
175                 ungetChar(c1);
176                 this.string = getStringFromBuffer();
177                 stringBufferTop = 0;
178                 return JsDocToken.STRING;
179               } else {
180                 if (isJSDocString(c2)) {
181                   addToString(c1);
182                 } else {
183                   ungetChar(c2);
184                   addToString(c1);
185                   this.string = getStringFromBuffer();
186                   stringBufferTop = 0;
187                   return JsDocToken.STRING;
188                 }
189               }
190             } while (true);
191           }
192         }
193       }
194     }
195   }

197   /**
198    * Gets the remaining JSDoc line without the {@link JsDocToken#EOL},
199    * {@link JsDocToken#EOF} or {@link JsDocToken#EOC}.
200    */
201   @SuppressWarnings("fallthrough")
202   String getRemainingJSDocLine() {
203     int c;
204     for (;;) {
205       c = getChar();
206       switch (c) {
207         case '*':
208           if (peekChar() != '/') {
209             addToString(c);
210             break;
211           }
212           // fall through
213         case EOF_CHAR:
214         case '\n':
215           ungetChar(c);
216           this.string = getStringFromBuffer();
217           stringBufferTop = 0;
218           return this.string;
219 
220         default:
221           addToString(c);
222           break;
223       }
224     }
225   }
226 
227   final int getLineno() { return lineno; }
228 
229   final int getCharno() {
230     return lineno == initLineno ? initCharno + charno : charno;
231   }
232 
233   final String getString() { return string; }
234 
235   private String getStringFromBuffer() {
236     String s = new String(stringBuffer, 0, stringBufferTop);
237     return s.intern();
238   }
239 
240   private void addToString(int c) {
241     int n = stringBufferTop;
242     if (n == stringBuffer.length) {
243         char[] tmp = new char[stringBuffer.length * 2];
244         System.arraycopy(stringBuffer, 0, tmp, 0, n);
245         stringBuffer = tmp;
246     }
247     stringBuffer[n] = (char) c;
248     stringBufferTop = n + 1;
249   }
250 
251   void ungetChar(int c) {
252     // can not unread past across line boundary
253     assert(!(ungetCursor != 0 && ungetBuffer[ungetCursor - 1] == '\n'));
254     ungetBuffer[ungetCursor++] = c;
255     cursor--;
256   }
257 
258   private boolean matchChar(int test) {
259     int c = getCharIgnoreLineEnd();
260     if (c == test) {
261       return true;
262     } else {
263       ungetCharIgnoreLineEnd(c);
264       return false;
265     }
266   }
267 
268   private static boolean isAlpha(int c) {
269     // Use 'Z' < 'a'
270     if (c <= 'Z') {
271       return 'A' <= c;
272     } else {
273       return 'a' <= c && c <= 'z';
274     }
275   }
276 
277   private static boolean isJSDocString(int c) {
278     switch (c) {
279       case '@':
280       case '*':
281       case ',':
282       case '<':
283       case '>':
284       case ':':
285       case '(':
286       case ')':
287       case '{':
288       case '}':
289       case '[':
290       case ']':
291       case '?':
292       case '!':
293       case '|':
294       case '=':
295       case EOF_CHAR:
296       case '\n':
297         return false;
298 
299       default:
300         return !TokenUtil.isJSSpace(c);
301     }
302   }

304   /**
305    * Allows the JSDocParser to update the character offset
306    * so that getCharno() returns a valid character position.
307    */
308   void update() {
309     charno = getOffset();
310   }
311 
312   private int peekChar() {
313     int c = getChar();
314     ungetChar(c);
315     return c;
316   }
317 
318   protected int getChar() {
319     if (ungetCursor != 0) {
320       cursor++;
321       --ungetCursor;
322       if (charno == -1) {
323         charno = getOffset();
324       }
325       return ungetBuffer[ungetCursor];
326     }
327 
328     for (;;) {
329       int c;
330       if (sourceCursor == sourceEnd) {
331         if (charno == -1) {
332           charno = getOffset();
333         }
334         return EOF_CHAR;
335       }
336       cursor++;
337       c = sourceString.charAt(sourceCursor++);
338 
339 
340       if (lineEndChar >= 0) {
341         if (lineEndChar == '\r' && c == '\n') {
342           lineEndChar = '\n';
343           continue;
344         }
345         lineEndChar = -1;
346         lineStart = sourceCursor - 1;
347         lineno++;
348       }
349 
350       if (c <= 127) {
351         if (c == '\n' || c == '\r') {
352           lineEndChar = c;
353           c = '\n';
354         }
355       } else {
356         if (TokenUtil.isJSFormatChar(c)) {
357           continue;
358         }
359         if (isJSLineTerminator(c)) {
360           lineEndChar = c;
361           c = '\n';
362         }
363       }
364 
365       if (charno == -1) {
366         charno = getOffset();
367       }
368 
369       return c;
370     }
371   }
372 
373   private int getCharIgnoreLineEnd() {
374     if (ungetCursor != 0) {
375       cursor++;
376       --ungetCursor;
377       if (charno == -1) {
378         charno = getOffset();
379       }
380       return ungetBuffer[ungetCursor];
381     }
382 
383     for (;;) {
384       int c;
385       if (sourceCursor == sourceEnd) {
386         if (charno == -1) {
387           charno = getOffset();
388         }
389         return EOF_CHAR;
390       }
391       cursor++;
392       c = sourceString.charAt(sourceCursor++);
393 
394 
395       if (c <= 127) {
396         if (c == '\n' || c == '\r') {
397           lineEndChar = c;
398           c = '\n';
399         }
400       } else {
401         if (TokenUtil.isJSFormatChar(c)) {
402           continue;
403         }
404         if (isJSLineTerminator(c)) {
405           lineEndChar = c;
406           c = '\n';
407         }
408       }
409 
410       if (charno == -1) {
411         charno = getOffset();
412       }
413 
414       return c;
415     }
416   }
417 
418   private static boolean isJSLineTerminator(int c) {
419     // Optimization for faster check for eol character:
420     // they do not have 0xDFD0 bits set
421     if ((c & 0xDFD0) != 0) {
422       return false;
423     }
424     return c == '\n' || c == '\r' || c == 0x2028 || c == 0x2029;
425   }
426 
427   private void ungetCharIgnoreLineEnd(int c) {
428     ungetBuffer[ungetCursor++] = c;
429     cursor--;
430   }
431 
432   /**
433    * Returns the offset into the current line.
434    */
435   final int getOffset() {
436     return sourceCursor - lineStart - ungetCursor - 1;
437   }

439   // Set this to an initial non-null value so that the Parser has
440   // something to retrieve even if an error has occurred and no
441   // string is found.  Fosters one class of error, but saves lots of
442   // code.
443   private String string = "";
444 
445   private char[] stringBuffer = new char[128];
446   private int stringBufferTop;
447 
448   // Room to backtrace from to < on failed match of the last - in <!--
449   private final int[] ungetBuffer = new int[3];
450   private int ungetCursor;
451 
452   private int lineStart = 0;
453   private int lineEndChar = -1;
454   int lineno;
455   private int charno = -1;
456   private final int initCharno;
457   private final int initLineno;
458 
459   private final String sourceString;
460   private final int sourceEnd;
461 
462   // sourceCursor is an index into a small buffer that keeps a
463   // sliding window of the source stream.
464   int sourceCursor;
465 
466   // cursor is a monotonically increasing index into the original
467   // source stream, tracking exactly how far scanning has progressed.
468   // Its value is the index of the next character to be scanned.
469   int cursor;
