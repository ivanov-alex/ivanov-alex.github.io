39  package com.google.javascript.rhino;
40  
41  /**
42   * A simple implementation of {@code StaticSourceFile} for testing.
43   *
44   * @author nicksantos@google.com (Nick Santos)
45   */
46  public final class SimpleSourceFile implements StaticSourceFile {
47    private final String name;
48    private final boolean extern;
49  
50    public SimpleSourceFile(String name, boolean extern) {
51      this.name = name;
52      this.extern = extern;
53    }
54  
55    @Override
56    public String getName() {
57      return name;
58    }
59  
60    @Override
61    public boolean isExtern() {
62      return extern;
63    }
64  
65    @Override
66    public int getColumnOfOffset(int offset) {
67      return 0;
68    }
69  
70    @Override
71    public int getLineOfOffset(int offset) {
72      return 1;
73    }
74  
75    @Override
76    public int getLineOffset(int line) {
77      if (line < 1) {
78        throw new IllegalStateException(
79            "Should not call getLineOffset with line number " + line);
80      }
81      return Integer.MIN_VALUE;
82    }
83  
84    @Override
85    public String toString() {
86      return name;
87    }
88  }
