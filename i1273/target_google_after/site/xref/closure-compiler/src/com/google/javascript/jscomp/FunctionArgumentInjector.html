31   * A nifty set of functions to deal with the issues of replacing function
32   * parameters with a set of call argument expressions.

38    // A string to use to represent "this".  Anything that is not a valid
39    // identifier can be used, so we use "this".
40    static final String THIS_MARKER = "this";
41  
42    private FunctionArgumentInjector() {
43      // A private constructor to prevent instantiation.
44    }
45  
46    /**
47     * With the map provided, replace the names with expression trees.
48     * @param node The root of the node tree within which to perform the
49     *     substitutions.
50     * @param parent The parent root node.
51     * @param replacements The map of names to template node trees with which
52     *     to replace the name Nodes.
53     * @return The root node or its replacement.
54     */
55    static Node inject(AbstractCompiler compiler, Node node, Node parent,
56        Map<String, Node> replacements) {
57      return inject(compiler, node, parent, replacements, true);
58    }
59  
60    static Node inject(AbstractCompiler compiler, Node node, Node parent,
61        Map<String, Node> replacements, boolean replaceThis) {
62      if (node.isName()) {
63        Node replacementTemplate = replacements.get(node.getString());
64        if (replacementTemplate != null) {
65          // This should not be replacing declared names.
66          Preconditions.checkState(!parent.isFunction()
67              || !parent.isVar()
68              || !parent.isCatch());
69          // The name may need to be replaced more than once,
70          // so we need to clone the node.
71          Node replacement = replacementTemplate.cloneTree();
72          parent.replaceChild(node, replacement);
73          return replacement;
74        }
75      } else if (replaceThis && node.isThis()) {
76        Node replacementTemplate = replacements.get(THIS_MARKER);
77        Preconditions.checkNotNull(replacementTemplate);
78        if (!replacementTemplate.isThis()) {
79          // The name may need to be replaced more than once,
80          // so we need to clone the node.
81          Node replacement = replacementTemplate.cloneTree();
82          parent.replaceChild(node, replacement);
83  
84          // Remove the value.  This isn't required but it ensures that we won't
85          // inject side-effects multiple times as it will trigger the null
86          // check above if we do.
87          if (NodeUtil.mayHaveSideEffects(replacementTemplate, compiler)) {
88            replacements.remove(THIS_MARKER);
89          }
90  
91          return replacement;
92        }
93      } else if (node.isFunction()) {
94        // Once we enter another scope the "this" value changes, don't try
95        // to replace it within an inner scope.
96        replaceThis = false;
97      }
98  
99      for (Node c = node.getFirstChild(); c != null; c = c.getNext()) {
100       // We have to reassign c in case it was replaced, because the removed c's
101       // getNext() would no longer be correct.
102       c = inject(compiler, c, node, replacements, replaceThis);
103     }
104 
105     return node;
106   }

108   /**
109    * Get a mapping for function parameter names to call arguments.
110    */
111   static LinkedHashMap<String, Node> getFunctionCallParameterMap(
112       Node fnNode, Node callNode, Supplier<String> safeNameIdSupplier) {
113     // Create an argName -> expression map
114     // NOTE: A linked map is created here to provide ordering.
115     LinkedHashMap<String, Node> argMap = new LinkedHashMap<>();
116 
117     // CALL NODE: [ NAME, ARG1, ARG2, ... ]
118     Node cArg = callNode.getFirstChild().getNext();
119     if (cArg != null && NodeUtil.isFunctionObjectCall(callNode)) {
120       argMap.put(THIS_MARKER, cArg);
121       cArg = cArg.getNext();
122     } else {
123       // 'apply' isn't supported yet.
124       Preconditions.checkState(!NodeUtil.isFunctionObjectApply(callNode));
125       argMap.put(THIS_MARKER, NodeUtil.newUndefinedNode(callNode));
126     }
127 
128     for (Node fnArg : NodeUtil.getFunctionParameters(fnNode).children()) {
129       if (cArg != null) {
130         argMap.put(fnArg.getString(), cArg);
131         cArg = cArg.getNext();
132       } else {
133         Node srcLocation = callNode;
134         argMap.put(fnArg.getString(), NodeUtil.newUndefinedNode(srcLocation));
135       }
136     }
137 
138     // Add temp names for arguments that don't have named parameters in the
139     // called function.
140     while (cArg != null) {
141       String uniquePlaceholder =
142         getUniqueAnonymousParameterName(safeNameIdSupplier);
143       argMap.put(uniquePlaceholder, cArg);
144       cArg = cArg.getNext();
145     }
146 
147     return argMap;
148   }

152    */

154       Supplier<String> safeNameIdSupplier) {
155     return "JSCompiler_inline_anon_param_" + safeNameIdSupplier.get();
156   }

158   /**
159    * Retrieve a set of names that can not be safely substituted in place.
160    * Example:
161    *   function(a) {
162    *     a = 0;
163    *   }
164    * Inlining this without taking precautions would cause the call site value
165    * to be modified (bad).
166    */
167   static Set<String> findModifiedParameters(Node fnNode) {
168     Set<String> names = getFunctionParameterSet(fnNode);
169     Set<String> unsafeNames = new HashSet<>();
170     return findModifiedParameters(
171         fnNode.getLastChild(), null, names, unsafeNames, false);
172   }

174   /**
175    * Check for uses of the named value that imply a pass-by-value
176    * parameter is expected.  This is used to prevent cases like:
177    *
178    *   function (x) {
179    *     x=2;
180    *     return x;
181    *   }
182    *
183    * We don't want "undefined" to be substituted for "x", and get
184    *   undefined=2
185    *
186    * @param n The node in question.
187    * @param parent The parent of the node.
188    * @param names The set of names to check.
189    * @param unsafe The set of names that require aliases.
190    * @param inInnerFunction Whether the inspection is occurring on a inner
191    *     function.
192    */
193   private static Set<String> findModifiedParameters(
194       Node n, Node parent, Set<String> names, Set<String> unsafe,
195       boolean inInnerFunction) {
196     Preconditions.checkArgument(unsafe != null);
197     if (n.isName()) {
198       if (names.contains(n.getString()) && (inInnerFunction || canNameValueChange(n, parent))) {
199         unsafe.add(n.getString());
200       }
201     } else if (n.isFunction()) {
202       // A function parameter can not be replaced with a direct inlined value
203       // if it is referred to by an inner function. The inner function
204       // can out live the call we are replacing, so inner function must
205       // capture a unique name.  This approach does not work within loop
206       // bodies so those are forbidden elsewhere.
207       inInnerFunction = true;
208     }
209 
210     for (Node c : n.children()) {
211       findModifiedParameters(c, n, names, unsafe, inInnerFunction);
212     }
213 
214     return unsafe;
215   }

217   /**
218    * This is similar to NodeUtil.isLValue except that object properties and
219    * array member modification aren't important ("o" in "o.a = 2" is still "o"
220    * after assignment, where in as "o = x", "o" is now "x").
221    *
222    * This also looks for the redefinition of a name.
223    *   function (x){var x;}
224    *
225    * @param n The NAME node in question.
226    * @param parent The parent of the node.
227    */
228   private static boolean canNameValueChange(Node n, Node parent) {
229     int type = parent.getType();
230     return (type == Token.VAR || type == Token.INC || type == Token.DEC ||
231         (NodeUtil.isAssignmentOp(parent) && parent.getFirstChild() == n) ||
232         (NodeUtil.isForIn(parent)));
233   }

235   /**
236    * Updates the set of parameter names in set unsafe to include any
237    * arguments from the call site that require aliases.
238    * @param fnNode The FUNCTION node to be inlined.
239    * @param argMap The argument list for the call to fnNode.
240    * @param namesNeedingTemps The set of names to update.
241    */
242   static void maybeAddTempsForCallArguments(
243       Node fnNode, Map<String, Node> argMap, Set<String> namesNeedingTemps,
244       CodingConvention convention) {
245     if (argMap.isEmpty()) {
246       // No arguments to check, we are done.
247       return;
248     }
249 
250     Preconditions.checkArgument(fnNode.isFunction());
251     Node block = fnNode.getLastChild();
252 
253     Set<String> parameters = argMap.keySet();
254 
255     // Get the list of parameters that may need temporaries due to
256     // side-effects.
257     Set<String> namesAfterSideEffects = findParametersReferencedAfterSideEffect(
258         parameters, block);
259 
260     // Check for arguments that are evaluated more than once.
261     for (Map.Entry<String, Node> entry : argMap.entrySet()) {
262       String argName = entry.getKey();
263       if (namesNeedingTemps.contains(argName)) {
264         continue;
265       }
266       Node cArg = entry.getValue();
267       boolean safe = true;
268       int references = NodeUtil.getNameReferenceCount(block, argName);
269 
270       if (NodeUtil.mayEffectMutableState(cArg) && references > 0) {
271         // Note: Mutable arguments should be assigned to temps, as the
272         // may be within in a loop:
273         //   function x(a) {
274         //     for(var i=0; i<0; i++) {
275         //       foo(a);
276         //     }
277         //   x( [] );
278         //
279         //   The parameter in the call to foo should not become "[]".
280         safe = false;
281       } else if (NodeUtil.mayHaveSideEffects(cArg)) {
282         // Even if there are no references, we still need to evaluate the
283         // expression if it has side-effects.
284         safe = false;
285       } else if (NodeUtil.canBeSideEffected(cArg)
286           && namesAfterSideEffects.contains(argName)) {
287         safe = false;
288       } else if (references > 1) {
289         // Safe is a misnomer, this is a check for "large".
290         switch (cArg.getType()) {
291           case Token.NAME:
292             String name = cArg.getString();
293             safe = !(convention.isExported(name));
294             break;
295           case Token.THIS:
296             safe = true;
297             break;
298           case Token.STRING:
299             safe = (cArg.getString().length() < 2);
300             break;
301           default:
302             safe = NodeUtil.isImmutableValue(cArg);
303             break;
304         }
305       }
306 
307       if (!safe) {
308         namesNeedingTemps.add(argName);
309       }
310     }
311   }

313   /**
314    * Boot strap a traversal to look for parameters referenced
315    * after a non-local side-effect.
316    * NOTE: This assumes no-inner functions.
317    * @param parameters The set of parameter names.
318    * @param root The function code block.
319    * @return The subset of parameters referenced after the first
320    *     seen non-local side-effect.
321    */
322   private static Set<String> findParametersReferencedAfterSideEffect(
323       Set<String> parameters, Node root) {
324 
325     // TODO(johnlenz): Consider using scope for this.
326     Set<String> locals = new HashSet<>(parameters);
327     gatherLocalNames(root, locals);
328 
329     ReferencedAfterSideEffect collector = new ReferencedAfterSideEffect(
330         parameters, locals);
331     NodeUtil.visitPostOrder(
332         root,
333         collector,
334         collector);
335     return collector.getResults();
336   }

338   /**
339    * Collect parameter names referenced after a non-local side-effect.
340    *
341    * Assumptions:
342    * - We assume parameters are not modified in the function body
343    * (that is checked separately).
344    * - There are no inner functions (also checked separately).
345    *
346    * As we are trying to replace parameters with there passed in values
347    * we are interested in anything that may affect those value.  So, ignoring
348    * changes to local variables, we look for things that may affect anything
349    * outside the local-state.  Once such a side-effect is seen any following
350    * reference to the function parameters are collected.  These will need
351    * to be assigned to temporaries to prevent changes to their value as would
352    * have happened during the function call.
353    *
354    * To properly handle loop structures all references to the function
355    * parameters are recorded and the decision to keep or throw away those
356    * references is deferred until exiting the loop structure.
357    */
358   private static class ReferencedAfterSideEffect
359       implements Visitor, Predicate<Node> {
360     private final Set<String> parameters;
361     private final Set<String> locals;
362     private boolean sideEffectSeen = false;
363     private Set<String> parametersReferenced = new HashSet<>();
364     private int loopsEntered = 0;
365 
366     ReferencedAfterSideEffect(Set<String> parameters, Set<String> locals) {
367       this.parameters = parameters;
368       this.locals = locals;
369     }
370 
371     Set<String> getResults() {
372       return parametersReferenced;
373     }
374 
375     @Override
376     public boolean apply(Node node) {
377       // Keep track of any loop structures entered.
378       if (NodeUtil.isLoopStructure(node)) {
379         loopsEntered++;
380       }
381 
382       // If we have found all the parameters, don't bother looking
383       // at the children.
384       return !(sideEffectSeen
385           && parameters.size() == parametersReferenced.size());
386     }
387 
388     boolean inLoop() {
389       return loopsEntered != 0;
390     }
391 
392     @Override
393     public void visit(Node n) {
394       // If we are exiting a loop.
395       if (NodeUtil.isLoopStructure(n)) {
396         loopsEntered--;
397         if (!inLoop() && !sideEffectSeen) {
398           // Now that the loops has been fully traversed and
399           // no side-effects have been seen, throw away
400           // the references seen in them.
401           parametersReferenced.clear();
402         }
403       }
404 
405       if (!sideEffectSeen) {
406         // Look for side-effects.
407         if (hasNonLocalSideEffect(n)) {
408           sideEffectSeen = true;
409         }
410       }
411 
412       // If traversing the nodes of a loop save any references
413       // that are seen.
414       if (inLoop() || sideEffectSeen) {
415         // Record references to parameters.
416         if (n.isName()) {
417           String name = n.getString();
418           if (parameters.contains(name)) {
419             parametersReferenced.add(name);
420           }
421         } else if (n.isThis()) {
422           parametersReferenced.add(THIS_MARKER);
423         }
424       }
425     }
426 
427     /**
428      * @return Whether the node may have non-local side-effects.
429      */
430     private boolean hasNonLocalSideEffect(Node n) {
431       boolean sideEffect = false;
432       int type = n.getType();
433       // Note: Only care about changes to non-local names, specifically
434       // ignore VAR declaration assignments.
435       if (NodeUtil.isAssignmentOp(n)
436           || type == Token.INC
437           || type == Token.DEC) {
438         Node lhs = n.getFirstChild();
439         // Ignore changes to local names.
440         if (!isLocalName(lhs)) {
441           sideEffect = true;
442         }
443       } else if (type == Token.CALL) {
444         sideEffect = NodeUtil.functionCallHasSideEffects(n);
445       } else if (type == Token.NEW) {
446         sideEffect = NodeUtil.constructorCallHasSideEffects(n);
447       } else if (type == Token.DELPROP) {
448         sideEffect = true;
449       }
450 
451       return sideEffect;
452     }
453 
454     /**
455      * @return Whether node is a reference to locally declared name.
456      */
457     private boolean isLocalName(Node node) {
458       if (node.isName()) {
459         String name = node.getString();
460         return locals.contains(name);
461       }
462       return false;
463     }
464   }

466   /**
467    * Gather any names declared in the local scope.
468    */
469   private static void gatherLocalNames(Node n, Set<String> names) {
470     if (n.isFunction()) {
471       if (NodeUtil.isFunctionDeclaration(n)) {
472         names.add(n.getFirstChild().getString());
473       }
474       // Don't traverse into inner function scopes;
475       return;
476     } else if (n.isName()) {
477       switch (n.getParent().getType()) {
478         case Token.VAR:
479         case Token.CATCH:
480           names.add(n.getString());
481       }
482     }
483 
484     for (Node c = n.getFirstChild(); c != null; c = c.getNext()) {
485       gatherLocalNames(c, names);
486     }
487   }

489   /**
490    * Get a set of function parameter names.
491    */
492   private static Set<String> getFunctionParameterSet(Node fnNode) {
493     Set<String> set = new HashSet<>();
494     for (Node n : NodeUtil.getFunctionParameters(fnNode).children()) {
495       set.add(n.getString());
496     }
497     return set;
498   }
