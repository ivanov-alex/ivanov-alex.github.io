57  /**
58   * Represents JavaScript value types.<p>
59   *
60   * Types are split into two separate families: value types and object types.
61   *
62   * A special {@link UnknownType} exists to represent a wildcard type on which
63   * no information can be gathered. In particular, it can assign to everyone,
64   * is a subtype of everyone (and everyone is a subtype of it).<p>
65   *
66   * If you remove the {@link UnknownType}, the set of types in the type system
67   * forms a lattice with the {@link #isSubtype} relation defining the partial
68   * order of types. All types are united at the top of the lattice by the
69   * {@link AllType} and at the bottom by the {@link NoType}.<p>
70   *
71   */

274   /**
275    * Returns true iff {@code this} can be a {@code struct}.
276    * UnionType overrides the method, assume {@code this} is not a union here.
277    */
278   public boolean isStruct() {
279     if (isObject()) {
280       ObjectType objType = toObjectType();
281       ObjectType iproto = objType.getImplicitPrototype();
282       // For the case when a @struct constructor is assigned to a function's
283       // prototype property
284       if (iproto != null && iproto.isStruct()) {
285         return true;
286       }
287       FunctionType ctor = objType.getConstructor();
288       // This test is true for object literals
289       if (ctor == null) {
290         JSDocInfo info = objType.getJSDocInfo();
291         return info != null && info.makesStructs();
292       } else {
293         return ctor.makesStructs();
294       }
295     }
296     return false;
297   }

593   /**
594    * An equivalence visitor.
595    */
596   boolean checkEquivalenceHelper(
597       final JSType that, EquivalenceMethod eqMethod) {
598     return checkEquivalenceHelper(that, eqMethod, EqCache.create());
599   }

601   boolean checkEquivalenceHelper(final JSType that, EquivalenceMethod eqMethod,
602       EqCache eqCache) {
603     if (this == that) {
604       return true;
605     }
606 
607     boolean thisUnknown = isUnknownType();
608     boolean thatUnknown = that.isUnknownType();
609     if (thisUnknown || thatUnknown) {
610       if (eqMethod == EquivalenceMethod.INVARIANT) {
611         // If we're checking for invariance, the unknown type is invariant
612         // with everyone.
613         return true;
614       } else if (eqMethod == EquivalenceMethod.DATA_FLOW) {
615         // If we're checking data flow, then two types are the same if they're
616         // both unknown.
617         return thisUnknown && thatUnknown;
618       } else if (thisUnknown && thatUnknown &&
619           (isNominalType() ^ that.isNominalType())) {
620         // If they're both unknown, but one is a nominal type and the other
621         // is not, then we should fail out immediately. This ensures that
622         // we won't unbox the unknowns further down.
623         return false;
624       }
625     }
626 
627     if (isUnionType() && that.isUnionType()) {
628       return toMaybeUnionType().checkUnionEquivalenceHelper(
629           that.toMaybeUnionType(), eqMethod, eqCache);
630     }
631 
632     if (isFunctionType() && that.isFunctionType()) {
633       return toMaybeFunctionType().checkFunctionEquivalenceHelper(
634           that.toMaybeFunctionType(), eqMethod, eqCache);
635     }
636 
637     if (isRecordType() && that.isRecordType()) {
638       return toMaybeRecordType().checkRecordEquivalenceHelper(
639           that.toMaybeRecordType(), eqMethod, eqCache);
640     }
641 
642     if (!getTemplateTypeMap().checkEquivalenceHelper(
643         that.getTemplateTypeMap(), eqMethod, eqCache)) {
644       return false;
645     }
646 
647     if (eqCache.isStructuralTyping()
648         && this.isStructuralType() && that.isStructuralType()) {
649       // TODO: handle template type equivalence, see:
650       // testDuplicateVariableDefinition8_7 in TypeCheckTest.java
651       return checkStructuralEquivalenceHelper(that, eqMethod, eqCache);
652     }
653 
654     if (isNominalType() && that.isNominalType()) {
655       // TODO(johnlenz): is this valid across scopes?
656       return getConcreteNominalTypeName(this.toObjectType())
657           .equals(getConcreteNominalTypeName(that.toObjectType()));
658     }
659 
660     if (isTemplateType() && that.isTemplateType()) {
661       // TemplateType are they same only if they are object identical,
662       // which we check at the start of this function.
663       return false;
664     }
665 
666     // Unbox other proxies.
667     if (this instanceof ProxyObjectType) {
668       return ((ProxyObjectType) this)
669           .getReferencedTypeInternal().checkEquivalenceHelper(
670               that, eqMethod, eqCache);
671     }
672 
673     if (that instanceof ProxyObjectType) {
674       return checkEquivalenceHelper(
675           ((ProxyObjectType) that).getReferencedTypeInternal(),
676           eqMethod, eqCache);
677     }
678 
679     // Relies on the fact that for the base {@link JSType}, only one
680     // instance of each sub-type will ever be created in a given registry, so
681     // there is no need to verify members. If the object pointers are not
682     // identical, then the type member must be different.
683     return false;
684   }

686   private boolean checkStructuralEquivalenceHelper(final JSType that,
687       EquivalenceMethod eqMethod, EqCache eqCache) {
688     if (this.isFunctionType() || that.isFunctionType()) {
689       return false;
690     }
691     Preconditions.checkState(eqCache.isStructuralTyping());
692     Preconditions.checkState((isRecordType() || that.isRecordType())
693         || isNominalType() && that.isNominalType());
694 
695     if (isNominalType() && that.isNominalType()) {
696       if (getConcreteNominalTypeName(this.toObjectType())
697           .equals(getConcreteNominalTypeName(that.toObjectType()))) {
698         return true;
699       }
700       FunctionType thatConstructor = that.toObjectType().getConstructor();
701       FunctionType thisConstructor = this.toObjectType().getConstructor();
702       return thisConstructor.checkStructuralInterfaceEquivalenceHelper(
703           thatConstructor, eqMethod, eqCache);
704     }
705     return checkObjectRecordEquivalenceHelper(that, eqMethod, eqCache);
706   }

708   /**
709    * do structural equivalence check for a record type and
710    * an instance of a structural interface type
711    */
712   private boolean checkObjectRecordEquivalenceHelper(final JSType that,
713       EquivalenceMethod eqMethod, EqCache eqCache) {
714     Preconditions.checkState((isInstanceType() && that.isRecordType())
715         || (isRecordType() && that.isInstanceType()));
716 
717     FunctionType constructor = isInstanceType() ?
718         toObjectType().getConstructor()
719         : that.toObjectType().getConstructor();
720     RecordType recordType = isRecordType() ? toMaybeRecordType()
721         : that.toMaybeRecordType();
722     return constructor.checkStructuralInterfaceEquivalenceHelper(
723         recordType, eqMethod, eqCache);
724   }

921   TernaryValue testForEqualityHelper(JSType aType, JSType bType) {
922     if (bType.isAllType() || bType.isUnknownType() ||

956       return bType.testForEquality(aType);

1050   /**
1051    * A generic implementation meant to be used as a helper for common
1052    * getGreatestSubtype implementations.
1053    */
1054   static JSType getGreatestSubtype(JSType thisType, JSType thatType) {
1055     if (thisType.isFunctionType() && thatType.isFunctionType()) {
1056       // The FunctionType sub-lattice is not well-defined. i.e., the
1057       // proposition
1058       // A < B => sup(A, B) == B
1059       // does not hold because of unknown parameters and return types.
1060       // See the comment in supAndInfHelper for more info on this.
1061       return thisType.toMaybeFunctionType().supAndInfHelper(
1062           thatType.toMaybeFunctionType(), false);
1063     } else if (thisType.isEquivalentTo(thatType)) {
1064       return thisType;
1065     } else if (thisType.isUnknownType() || thatType.isUnknownType()) {
1066       // The greatest subtype with any unknown type is the universal
1067       // unknown type, unless the two types are equal.
1068       return thisType.isEquivalentTo(thatType) ? thisType :
1069           thisType.getNativeType(JSTypeNative.UNKNOWN_TYPE);
1070     } else if (thisType.isUnionType()) {
1071       return thisType.toMaybeUnionType().meet(thatType);
1072     } else if (thatType.isUnionType()) {
1073       return thatType.toMaybeUnionType().meet(thisType);
1074     } else if (thisType.isTemplatizedType()) {
1075       return thisType.toMaybeTemplatizedType().getGreatestSubtypeHelper(
1076           thatType);
1077     }  else if (thatType.isTemplatizedType()) {
1078       return thatType.toMaybeTemplatizedType().getGreatestSubtypeHelper(
1079           thisType);
1080     } else if (thisType.isSubtype(thatType)) {
1081       return filterNoResolvedType(thisType);
1082     } else if (thatType.isSubtype(thisType)) {
1083       return filterNoResolvedType(thatType);
1084     } else if (thisType.isRecordType()) {
1085       return thisType.toMaybeRecordType().getGreatestSubtypeHelper(thatType);
1086     } else if (thatType.isRecordType()) {
1087       return thatType.toMaybeRecordType().getGreatestSubtypeHelper(thisType);
1088     }
1089 
1090     if (thisType.isEnumElementType()) {
1091       JSType inf = thisType.toMaybeEnumElementType().meet(thatType);
1092       if (inf != null) {
1093         return inf;
1094       }
1095     } else if (thatType.isEnumElementType()) {
1096       JSType inf = thatType.toMaybeEnumElementType().meet(thisType);
1097       if (inf != null) {
1098         return inf;
1099       }
1100     }
1101 
1102     if (thisType.isObject() && thatType.isObject()) {
1103       return thisType.getNativeType(JSTypeNative.NO_OBJECT_TYPE);
1104     }
1105     return thisType.getNativeType(JSTypeNative.NO_TYPE);
1106   }

1108   /**
1109    * When computing infima, we may get a situation like
1110    * inf(Type1, Type2)
1111    * where both types are unresolved, so they're technically
1112    * subtypes of one another.
1113    *
1114    * If this happens, filter them down to NoResolvedType.
1115    */
1116   static JSType filterNoResolvedType(JSType type) {
1117     if (type.isNoResolvedType()) {
1118       // inf(UnresolvedType1, UnresolvedType2) needs to resolve
1119       // to the base unresolved type, so that the relation is symmetric.
1120       return type.getNativeType(JSTypeNative.NO_RESOLVED_TYPE);
1121     } else if (type.isUnionType()) {
1122       UnionType unionType = type.toMaybeUnionType();
1123       boolean needsFiltering = false;
1124       for (JSType alt : unionType.getAlternates()) {
1125         if (alt.isNoResolvedType()) {
1126           needsFiltering = true;
1127           break;
1128         }
1129       }
1130 
1131       if (needsFiltering) {
1132         UnionTypeBuilder builder = new UnionTypeBuilder(type.registry);
1133         builder.addAlternate(type.getNativeType(JSTypeNative.NO_RESOLVED_TYPE));
1134         for (JSType alt : unionType.getAlternates()) {
1135           if (!alt.isNoResolvedType()) {
1136             builder.addAlternate(alt);
1137           }
1138         }
1139         return builder.build();
1140       }
1141     }
1142     return type;

1365   /**
1366    * if implicitImplCache is null, then there will
1367    * be no structural interface matching
1368    */
1369   static boolean isSubtypeHelper(JSType thisType, JSType thatType,
1370       ImplCache implicitImplCache) {
1371     // unknown
1372     if (thatType.isUnknownType()) {
1373       return true;
1374     }
1375     // all type
1376     if (thatType.isAllType()) {
1377       return true;
1378     }
1379     // equality
1380     if (thisType.isEquivalentTo(thatType, implicitImplCache.isStructuralTyping())) {
1381       return true;

1384     if (thatType.isUnionType()) {
1385       UnionType union = thatType.toMaybeUnionType();
1386       for (JSType element : union.alternatesWithoutStucturalTyping) {
1387         if (thisType.isSubtype(element, implicitImplCache)) {
1388           return true;
1389         }
1390       }
1391       return false;
1392     }
1393 
1394     // TemplateTypeMaps. This check only returns false if the TemplateTypeMaps
1395     // are not equivalent.
1396     TemplateTypeMap thisTypeParams = thisType.getTemplateTypeMap();
1397     TemplateTypeMap thatTypeParams = thatType.getTemplateTypeMap();
1398     boolean templateMatch = true;
1399     if (isExemptFromTemplateTypeInvariance(thatType)) {
1400       // Array and Object are exempt from template type invariance; their
1401       // template types maps are considered a match only if the ObjectElementKey
1402       // values are subtypes/supertypes of one another.
1403       TemplateType key = thisType.registry.getObjectElementKey();
1404       JSType thisElement = thisTypeParams.getTemplateType(key);
1405       JSType thatElement = thatTypeParams.getTemplateType(key);
1406 
1407       templateMatch = thisElement.isSubtype(thatElement, implicitImplCache)
1408           || thatElement.isSubtype(thisElement, implicitImplCache);
1409     } else {
1410       templateMatch = thisTypeParams.checkEquivalenceHelper(
1411           thatTypeParams, EquivalenceMethod.INVARIANT);
1412     }
1413     if (!templateMatch) {
1414       return implicitImplCache.isStructuralTyping()
1415           && implicitMatch(thisType, thatType, implicitImplCache);
1416     }
1417 
1418     // Templatized types. The above check guarantees TemplateTypeMap
1419     // equivalence; check if the base type is a subtype.
1420     if (thisType.isTemplatizedType()) {
1421       return thisType.toMaybeTemplatizedType().getReferencedType().isSubtype(
1422           thatType, implicitImplCache);
1423     }
1424 
1425     // proxy types
1426     if (thatType instanceof ProxyObjectType) {
1427       return thisType.isSubtype(
1428           ((ProxyObjectType) thatType).getReferencedTypeInternal(), implicitImplCache);
1429     }
1430     return implicitImplCache.isStructuralTyping()
1431         && implicitMatch(thisType, thatType, implicitImplCache);
1432   }

1434   /**
1435    * perform structural interface matching to see if the second argument type
1436    * implicitly implements the first argument type. Consider this example:
1437    * varLeft = varRight;
1438    *
1439    * (this is an internal method that uses cache to speedup and
1440    * prevent infinite recursive invocation.
1441    * As an example of possible infinite recursive invocation,
1442    * see testStructuralInterfaceMatching13 in TypeCheckTest.java)
1443    *
1444    * @param rightType the type of varRight
1445    * @param leftType the type of varLeft
1446    * @param implicitImplCache
1447    * @return true if the second argument type structurally matches the
1448    * first argument type
1449    */
1450   protected static boolean implicitMatch(JSType rightType, JSType leftType,
1451       ImplCache implicitImplCache) {
1452 
1453     // union type should be handled by isSubtype already
1454     if (rightType.isUnionType() || leftType.isUnionType()) {
1455       return false;
1456     }
1457     // anything other than two object types was already handled
1458     if (!rightType.isObject() || !leftType.isObject()) {
1459       return false;
1460     }
1461     // currently the structural interface matching does not support
1462     // implicit matching for templatized type
1463     if (leftType.isTemplatizedType()) {
1464       return false;
1465     }
1466     // if both are functionTypes
1467     if (leftType.isStructuralInterface() && rightType.isFunctionType()) {
1468       return checkConstructorImplicitMatch(rightType.toMaybeFunctionType(),
1469           leftType.toMaybeFunctionType(), implicitImplCache);
1470     }
1471     // matches a record type and an interface
1472     if (leftType.isStructuralInterface() && rightType.isRecordType()) {
1473       return checkObjectImplicitMatch(rightType.toMaybeObjectType(),
1474           leftType.toMaybeFunctionType(), implicitImplCache);
1475     }
1476     // then both are object instance types
1477     FunctionType leftConstructor = leftType.toMaybeObjectType().getConstructor();
1478     FunctionType rightConstructor = rightType.toMaybeObjectType().getConstructor();
1479     if (leftConstructor == null || !leftConstructor.isStructuralInterface()) {
1480       return false;
1481     }
1482     if (rightConstructor != null) {
1483       return checkConstructorImplicitMatch(rightConstructor, leftConstructor,
1484           implicitImplCache);
1485     }
1486     return checkObjectImplicitMatch(
1487         rightType.toMaybeObjectType(), leftConstructor, implicitImplCache);
1488   }

1490   /**
1491    * check if the rightType (an object type)
1492    * is compatible with the leftType (must be an interface function type)
1493    *
1494    * @param rightType must be either a function type or a record type
1495    * @param leftType must be an interface function type
1496    * @param implicitImplCache
1497    * @return true if the rightType structurally matches the leftType
1498    */
1499   protected static boolean checkObjectImplicitMatch(
1500       ObjectType rightType, FunctionType leftType,
1501       ImplCache implicitImplCache) {
1502     Preconditions.checkArgument(leftType.isStructuralInterface());
1503     Map<String, JSType> leftPropList = getPropertyTypeMap(leftType);
1504     Map<String, JSType> rightPropList = getPropertyTypeMap(rightType);
1505     // structural interface matching
1506     for (String propName : leftPropList.keySet()) {
1507       JSType leftPropType = leftPropList.get(propName);
1508       JSType rightPropType = rightPropList.get(propName);
1509       if (rightPropType == null) {
1510         return false;
1511       }
1512       // apply structural interface matching on properties
1513       if (!rightPropType.isSubtype(leftPropType, implicitImplCache)) {
1514         return false;
1515       }
1516     }
1517     return true;
1518   }

1520   /**
1521    * check if functionType implicitly implements interfaceType
1522    *
1523    * @param rightType must be a function type
1524    * @param leftType (must be an interface function type)
1525    * @return true if all properties required by interfaceType
1526    * are in functionType
1527    */
1528   protected static boolean checkConstructorImplicitMatch(FunctionType rightType,
1529       FunctionType leftType,
1530       ImplCache implicitImplCache) {
1531     // non-interfaces have been handled by isSubtype
1532     Preconditions.checkArgument(leftType.isStructuralInterface());
1533     // classes that have a explicitly declared relationship
1534     // should be handled by isSubType method.
1535     if (rightType.explicitlyImplOrExtInterface(leftType)) {
1536       return true;
1537     }
1538     MatchStatus result = implicitImplCache.checkCache(rightType, leftType);
1539     if (result != null) {
1540       return result.subtypeValue();
1541     }
1542     // the following case should not type check
1543     // var1 : IArrayLike<string> = var2 : IArrayLike<number>
1544     // the template types mismatch
1545     if (leftType.hasAnyTemplateTypes()) {
1546       implicitImplCache.updateCache(leftType, rightType, MatchStatus.NOT_MATCH);
1547       return false;
1548     }
1549     Map<String, JSType> interfacePropList = getPropertyTypeMap(leftType);
1550     Map<String, JSType> functionPropList = getPropertyTypeMap(rightType);
1551     for (String propName : interfacePropList.keySet()) {
1552       JSType typeInInterface = interfacePropList.get(propName);
1553       JSType typeInFunction = functionPropList.get(propName);
1554       if (typeInFunction == null
1555           || !typeInFunction.isSubtype(typeInInterface, implicitImplCache)) {
1556         implicitImplCache.updateCache(leftType, rightType, MatchStatus.NOT_MATCH);
1557         return false;
1558       }
1559     }
1560     // if all properties required by the interface
1561     // have compatible property in the function,
1562     // consider that the function implements the interface
1563     implicitImplCache.updateCache(leftType, rightType, MatchStatus.MATCH);
1564     return true;
1565   }

1790    * base cache data structure
1791    */
1792   protected abstract static class MatchCache {
1793     protected Table<JSType, JSType, MatchStatus> matchCache;
1794     protected boolean isStructuralTyping;
1795 
1796     protected MatchCache(boolean isStructuralTyping) {
1797       this.isStructuralTyping = isStructuralTyping;
1798       this.matchCache = null;
1799     }
1800 
1801     protected boolean isStructuralTyping() {
1802       return isStructuralTyping;
1803     }
1804 
1805     void updateCache(JSType leftType,
1806         JSType rightType, MatchStatus isMatch) {
1807       this.matchCache.put(leftType, rightType, isMatch);
1808     }
1809 
1810     MatchStatus checkCache(JSType rightType, JSType leftType) {
1811       if (this.matchCache == null) {
1812         this.matchCache = HashBasedTable.create();
1813       }
1814       // check the cache
1815       if (this.matchCache.contains(leftType, rightType)) {
1816         return this.matchCache.get(leftType, rightType);
1817       } else {
1818         this.updateCache(leftType, rightType, MatchStatus.PROCESSING);
1819         return null;
1820       }
1821     }
1822   }
