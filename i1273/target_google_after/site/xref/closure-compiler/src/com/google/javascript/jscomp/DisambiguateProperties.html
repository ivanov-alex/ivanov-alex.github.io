50  /**
51   * DisambiguateProperties renames properties to disambiguate between unrelated
52   * fields with the same name. Two properties are considered related if they
53   * share a definition on their prototype chains, or if they are potentially
54   * referenced together via union types.
55   *
56   * <p> Renamimg only occurs if there are two or more distinct properties with
57   * the same name.
58   *
59   * <p> This pass allows other passes, such as inlining and code removal to take
60   * advantage of type information implicitly.
61   *
62   * <pre>
63   *   Foo.a;
64   *   Bar.a;
65   * </pre>
66   *
67   * <p> will become
68   *
69   * <pre>
70   *   Foo.a$Foo;
71   *   Bar.a$Bar;
72   * </pre>
73   *
74   */
75  class DisambiguateProperties<T> implements CompilerPass {
76    // To prevent the logs from filling up, we cap the number of warnings
77    // that we tell the user to fix per-property.
78    private static final int MAX_INVALDIATION_WARNINGS_PER_PROPERTY = 10;
79  
80    private static final Logger logger = Logger.getLogger(
81        DisambiguateProperties.class.getName());
82    private static final Pattern NONWORD_PATTERN = Pattern.compile("[^\\w$]");

85      // TODO(user): {1} and {2} are not exactly useful for most people.
86      static final DiagnosticType INVALIDATION = DiagnosticType.disabled(

100   /**
101    * Map of a type to all the related errors that invalidated the type
102    * for disambiguation. It has be Object because of the generic nature of
103    * this pass.
104    */
105   private Multimap<Object, JSError> invalidationMap;

107   /**
108    * In practice any large code base will have thousands and thousands of
109    * type invalidations, which makes reporting all of the errors useless.
110    * However, certain properties are worth specifically guarding because of the
111    * large amount of code that can be removed as dead code. This list contains
112    * the properties (eg: "toString") that we care about; if any of these
113    * properties is invalidated it causes an error.
114    */
115   private final Map<String, CheckLevel> propertiesToErrorFor;

117   private class Property {
118     /** The name of the property. */
119     final String name;
120 
121     /** All types on which the field exists, grouped together if related. */
122     private UnionFind<T> types;
123 
124     /**
125      * A set of types for which renaming this field should be skipped. This
126      * list is first filled by fields defined in the externs file.
127      */
128     Set<T> typesToSkip = new HashSet<>();
129 
130     /**
131      * If true, do not rename any instance of this field, as it has been
132      * referenced from an unknown type.
133      */
134     boolean skipRenaming;
135 
136     /** Set of nodes for this field that need renaming. */
137     Set<Node> renameNodes = new HashSet<>();
138 
139     /**
140      * Map from node to the highest type in the prototype chain containing the
141      * field for that node. In the case of a union, the type is the highest type
142      * of one of the types in the union.
143      */
144     final Map<Node, T> rootTypes = new HashMap<>();
145 
146     Property(String name) {
147       this.name = name;
148     }
149 
150     /** Returns the types on which this field is referenced. */
151     UnionFind<T> getTypes() {
152       if (types == null) {
153         types = new StandardUnionFind<>();
154       }
155       return types;
156     }
157 
158     /**
159      * Record that this property is referenced from this type.
160      * @return true if the type was recorded for this property, else false,
161      *     which would happen if the type was invalidating.
162      */
163     boolean addType(T type, T top, T relatedType) {
164       checkState(!skipRenaming, "Attempt to record skipped property: %s", name);
165       if (typeSystem.isInvalidatingType(top)) {
166         invalidate();
167         return false;
168       } else {
169         if (typeSystem.isTypeToSkip(top)) {
170           addTypeToSkip(top);
171         }
172 
173         if (relatedType == null) {
174           getTypes().add(top);
175         } else {
176           getTypes().union(top, relatedType);
177         }
178         typeSystem.recordInterfaces(type, top, this);
179         return true;
180       }
181     }
182 
183     /** Records the given type as one to skip for this property. */
184     void addTypeToSkip(T type) {
185       for (T skipType : typeSystem.getTypesToSkipForType(type)) {
186         typesToSkip.add(skipType);
187         getTypes().union(skipType, type);
188       }
189     }
190 
191     /** Invalidates any types related to invalid types. */
192     void expandTypesToSkip() {
193       // If we are not going to rename any properties, then we do not need to
194       // update the list of invalid types, as they are all invalid.
195       if (shouldRename()) {
196         int count = 0;
197         while (true) {
198           // It should usually only take one time through this do-while.
199           checkState(++count < 10, "Stuck in loop expanding types to skip.");
200 
201           // Make sure that the representative type for each type to skip is
202           // marked as being skipped.
203           Set<T> rootTypesToSkip = new HashSet<>();
204           for (T subType : typesToSkip) {
205             rootTypesToSkip.add(types.find(subType));
206           }
207           typesToSkip.addAll(rootTypesToSkip);
208 
209           Set<T> newTypesToSkip = new HashSet<>();
210           Set<T> allTypes = types.elements();
211           int originalTypesSize = allTypes.size();
212           for (T subType : allTypes) {
213             if (!typesToSkip.contains(subType)
214                 && typesToSkip.contains(types.find(subType))) {
215               newTypesToSkip.add(subType);
216             }
217           }
218 
219           for (T newType : newTypesToSkip) {
220             addTypeToSkip(newType);
221           }
222 
223           // If there were not any new types added, we are done here.
224           if (types.elements().size() == originalTypesSize) {
225             break;
226           }
227         }
228       }
229     }
230 
231     /** Returns true if any instance of this property should be renamed. */
232     boolean shouldRename() {
233       return !skipRenaming && types != null
234           && types.allEquivalenceClasses().size() > 1;
235     }
236 
237     /**
238      * Returns true if this property should be renamed on this type.
239      * expandTypesToSkip() should be called before this, if anything has been
240      * added to the typesToSkip list.
241      */
242     boolean shouldRename(T type) {
243       return !skipRenaming && !typesToSkip.contains(type);
244     }
245 
246     /**
247      * Invalidates a field from renaming.  Used for field references on an
248      * object with unknown type.
249      */
250     boolean invalidate() {
251       boolean changed = !skipRenaming;
252       skipRenaming = true;
253       types = null;
254       return changed;
255     }
256 
257     /**
258      * Schedule the node to potentially be renamed.
259      * @param node the node to rename
260      * @param type the highest type in the prototype chain for which the
261      *     property is defined
262      * @return True if type was accepted without invalidation or if the property
263      *     was already invalidated.  False if this property was invalidated this
264      *     time.
265      */
266     boolean scheduleRenaming(Node node, T type) {
267       if (!skipRenaming) {
268         if (typeSystem.isInvalidatingType(type)) {
269           invalidate();
270           return false;
271         }
272         renameNodes.add(node);
273         rootTypes.put(node, type);
274       }
275       return true;
276     }
277   }

288   /**
289    * This constructor should only be called by one of the helper functions
290    * above for either the JSType system, or the concrete type system.
291    */
292   private DisambiguateProperties(AbstractCompiler compiler,
293       TypeSystem<T> typeSystem, Map<String, CheckLevel> propertiesToErrorFor) {
294     this.compiler = compiler;
295     this.typeSystem = typeSystem;
296     this.propertiesToErrorFor = propertiesToErrorFor;
297     if (!this.propertiesToErrorFor.isEmpty()) {
298       this.invalidationMap = LinkedHashMultimap.create();
299     } else {
300       this.invalidationMap = null;
301     }
302   }

304   @Override
305   public void process(Node externs, Node root) {
306     Preconditions.checkState(
307         compiler.getLifeCycleStage() == LifeCycleStage.NORMALIZED);
308     // TypeValidator records places where a type A is used in a context that
309     // expects a type B.
310     // For each pair (A, B), here we mark both A and B as types whose properties
311     // cannot be renamed.
312     for (TypeMismatch mis : compiler.getTypeMismatches()) {
313       addInvalidatingType(mis.typeA, mis.src);
314       addInvalidatingType(mis.typeB, mis.src);
315     }
316     for (TypeMismatch mis : compiler.getImplicitInterfaceUses()) {
317       addInvalidatingType(mis.typeA, mis.src);
318       addInvalidatingType(mis.typeB, mis.src);
319     }
320     // Gather names of properties in externs; these properties can't be renamed.
321     NodeTraversal.traverseEs6(compiler, externs, new FindExternProperties());
322     // Look at each unquoted property access and decide if that property will
323     // be renamed.
324     NodeTraversal.traverseEs6(compiler, root, new FindRenameableProperties());
325     // Do the actual renaming.
326     renameProperties();
327   }

329   private void recordInvalidationError(JSType t, JSError error) {
330     if (!t.isObject()) {
331       return;
332     }
333     if (invalidationMap != null) {
334       invalidationMap.put(t, error);
335     }
336   }

338   /**
339    * Invalidates the given type, so that no properties on it will be renamed.
340    */
341   private void addInvalidatingType(JSType type, JSError error) {
342     type = type.restrictByNotNullOrUndefined();
343     if (type.isUnionType()) {
344       for (JSType alt : type.toMaybeUnionType().getAlternatesWithoutStructuralTyping()) {
345         addInvalidatingType(alt, error);
346       }
347     } else if (type.isEnumElementType()) {
348       addInvalidatingType(
349           type.toMaybeEnumElementType().getPrimitiveType(), error);
350     } else {
351       typeSystem.addInvalidatingType(type);
352       recordInvalidationError(type, error);
353       ObjectType objType = ObjectType.cast(type);
354       if (objType != null && objType.getImplicitPrototype() != null) {
355         typeSystem.addInvalidatingType(objType.getImplicitPrototype());
356         recordInvalidationError(objType.getImplicitPrototype(), error);
357       }
358     }
359   }

375   /** Tracks the current type system scope while traversing. */
376   private abstract class AbstractScopingCallback implements ScopedCallback {
377     protected final Stack<StaticTypedScope<T>> scopes =
378         new Stack<>();
379 
380     @Override
381     public boolean shouldTraverse(NodeTraversal t, Node n, Node parent) {
382       return true;
383     }
384 
385     @Override
386     public void enterScope(NodeTraversal t) {
387       if (t.inGlobalScope()) {
388         scopes.push(typeSystem.getRootScope());
389       } else {
390         scopes.push(typeSystem.getFunctionScope(t.getScopeRoot()));
391       }
392     }
393 
394     @Override
395     public void exitScope(NodeTraversal t) {
396       scopes.pop();
397     }
398 
399     /** Returns the current scope at this point in the file. */
400     protected StaticTypedScope<T> getScope() {
401       return scopes.peek();
402     }
403   }

405   /**
406    * Finds all properties defined in the externs file and sets them as
407    * ineligible for renaming from the type on which they are defined.
408    */
409   private class FindExternProperties extends AbstractScopingCallback {
410     @Override public void visit(NodeTraversal t, Node n, Node parent) {
411       // TODO(johnlenz): Support object-literal property definitions.
412       if (n.isGetProp()) {
413         String field = n.getLastChild().getString();
414         T type = typeSystem.getType(getScope(), n.getFirstChild(), field);
415         Property prop = getProperty(field);
416         if (typeSystem.isInvalidatingType(type)) {
417           prop.invalidate();
418         } else {
419           prop.addTypeToSkip(type);
420 
421           // If this is a prototype property, then we want to skip assignments
422           // to the instance type as well.  These assignments are not usually
423           // seen in the extern code itself, so we must handle them here.
424           if ((type = typeSystem.getInstanceFromPrototype(type)) != null) {
425             prop.getTypes().add(type);
426             prop.typesToSkip.add(type);
427           }
428         }
429       }
430     }
431   }

433   /**
434    * Traverses the tree, building a map from field names to Nodes for all
435    * fields that can be renamed.
436    */
437   private class FindRenameableProperties extends AbstractScopingCallback {
438     @Override
439     public void visit(NodeTraversal t, Node n, Node parent) {
440       if (n.isGetProp()) {
441         handleGetProp(t, n);
442       } else if (n.isObjectLit()) {
443         handleObjectLit(t, n);
444       }
445     }
446 
447     private void handleGetProp(NodeTraversal t, Node n) {
448       String name = n.getLastChild().getString();
449       T type = typeSystem.getType(getScope(), n.getFirstChild(), name);
450 
451       Property prop = getProperty(name);
452       if (!prop.scheduleRenaming(n.getLastChild(), processProperty(t, prop, type, null))
453           && propertiesToErrorFor.containsKey(name)) {
454         String suggestion = "";
455         if (type instanceof JSType) {
456           JSType jsType = (JSType) type;
457           if (jsType.isAllType() || jsType.isUnknownType()) {
458             if (n.getFirstChild().isThis()) {
459               suggestion = "The \"this\" object is unknown in the function," +
460                     "consider using @this";
461             } else {
462               String qName = n.getFirstChild().getQualifiedName();
463               suggestion = "Consider casting " + qName + " if you know its type.";
464             }
465           } else {
466             List<String> errors = new ArrayList<>();
467             printErrorLocations(errors, jsType);
468             if (!errors.isEmpty()) {
469               suggestion = "Consider fixing errors for the following types:\n";
470               suggestion += Joiner.on("\n").join(errors);
471             }
472           }
473         }
474         compiler.report(JSError.make(n, propertiesToErrorFor.get(name), Warnings.INVALIDATION, name,
475             (String.valueOf(type)), n.toString(), suggestion));
476       }
477     }
478 
479     private void handleObjectLit(NodeTraversal t, Node n) {
480       for (Node child = n.getFirstChild();
481           child != null;
482           child = child.getNext()) {
483         // Maybe STRING, GET, SET
484         if (child.isQuotedString()) {
485           continue;
486         }
487 
488         // We should never see a mix of numbers and strings.
489         String name = child.getString();
490         T type = typeSystem.getType(getScope(), n, name);
491 
492         Property prop = getProperty(name);
493         if (!prop.scheduleRenaming(child,
494                                    processProperty(t, prop, type, null))) {
495           // TODO(user): It doesn't look like the user can do much in this
496           // case right now.
497           if (propertiesToErrorFor.containsKey(name)) {
498             compiler.report(JSError.make(child, propertiesToErrorFor.get(name),
499                 Warnings.INVALIDATION, name, (String.valueOf(type)), n.toString(), ""));
500           }
501         }
502       }
503     }
504 
505     private void printErrorLocations(List<String> errors, JSType t) {
506       if (!t.isObject() || t.isAllType()) {
507         return;
508       }
509 
510       if (t.isUnionType()) {
511         for (JSType alt : t.toMaybeUnionType().getAlternates()) {
512           printErrorLocations(errors, alt);
513         }
514         return;
515       }
516 
517       for (JSError error : invalidationMap.get(t)) {
518         if (errors.size() > MAX_INVALDIATION_WARNINGS_PER_PROPERTY) {
519           return;
520         }
521 
522         errors.add(t + " at " + error.sourceName + ":" + error.lineNumber);
523       }
524     }
525 
526     /**
527      * Processes a property, adding it to the list of properties to rename.
528      * @return a representative type for the property reference, which will be
529      *   the highest type on the prototype chain of the provided type.  In the
530      *   case of a union type, it will be the highest type on the prototype
531      *   chain of one of the members of the union.
532      */
533     private T processProperty(
534         NodeTraversal t, Property prop, T type, T relatedType) {
535       type = typeSystem.restrictByNotNullOrUndefined(type);
536       if (prop.skipRenaming || typeSystem.isInvalidatingType(type)) {
537         return null;
538       }
539 
540       Iterable<T> alternatives = typeSystem.getTypeAlternatives(type);
541       if (alternatives != null) {
542         T firstType = relatedType;
543         for (T subType : alternatives) {
544           T lastType = processProperty(t, prop, subType, firstType);
545           if (lastType != null) {
546             firstType = firstType == null ? lastType : firstType;
547           }
548         }
549         return firstType;
550       } else {
551         T topType = typeSystem.getTypeWithProperty(prop.name, type);
552         if (typeSystem.isInvalidatingType(topType)) {
553           return null;
554         }
555         prop.addType(type, topType, relatedType);
556         return topType;
557       }
558     }
559   }

561   /** Renames all properties with references on more than one type. */
562   void renameProperties() {
563     int propsRenamed = 0, propsSkipped = 0, instancesRenamed = 0,
564         instancesSkipped = 0, singleTypeProps = 0;
565 
566     Set<String> reported = new HashSet<>();
567     for (Property prop : properties.values()) {
568       if (prop.shouldRename()) {
569         Map<T, String> propNames = buildPropNames(prop.getTypes(), prop.name);
570 
571         ++propsRenamed;
572         prop.expandTypesToSkip();
573         // This loop has poor locality, because instead of walking the AST,
574         // we iterate over all accesses of a property, which can be in very
575         // different places in the code.
576         for (Node node : prop.renameNodes) {
577           T rootType = prop.rootTypes.get(node);
578           if (prop.shouldRename(rootType)) {
579             String newName = propNames.get(rootType);
580             node.setString(newName);
581             compiler.reportCodeChange();
582             ++instancesRenamed;
583           } else {
584             ++instancesSkipped;
585 
586             CheckLevel checkLevelForProp = propertiesToErrorFor.get(prop.name);
587             if (checkLevelForProp != null &&
588                 checkLevelForProp != CheckLevel.OFF &&
589                 !reported.contains(prop.name)) {
590               reported.add(prop.name);
591               compiler.report(JSError.make(
592                   node,
593                   checkLevelForProp,
594                   Warnings.INVALIDATION_ON_TYPE, prop.name,
595                   rootType.toString(), ""));
596             }
597           }
598         }
599       } else {
600         if (prop.skipRenaming) {
601           ++propsSkipped;
602         } else {
603           ++singleTypeProps;
604         }
605       }
606     }
607     logger.fine("Renamed " + instancesRenamed + " instances of "
608                 + propsRenamed + " properties.");
609     logger.fine("Skipped renaming " + instancesSkipped + " invalidated "
610                 + "properties, " + propsSkipped + " instances of properties "
611                 + "that were skipped for specific types and " + singleTypeProps
612                 + " properties that were referenced from only one type.");
613   }

615   /**
616    * Chooses a name to use for renaming in each equivalence class and maps
617    * each type in that class to it.
618    */
619   private Map<T, String> buildPropNames(UnionFind<T> types, String name) {
620     Map<T, String> names = new HashMap<>();
621     for (Set<T> set : types.allEquivalenceClasses()) {
622       checkState(!set.isEmpty());
623 
624       String typeName = null;
625       for (T type : set) {
626         if (typeName == null || type.toString().compareTo(typeName) < 0) {
627           typeName = type.toString();
628         }
629       }
630 
631       String newName;
632       if ("{...}".equals(typeName)) {
633         newName = name;
634       } else {
635         newName = NONWORD_PATTERN.matcher(typeName).replaceAll("_") + '$'
636             + name;
637       }
638 
639       for (T type : set) {
640         names.put(type, newName);
641       }
642     }
643     return names;
644   }

646   /** Returns a map from field name to types for which it will be renamed. */
647   Multimap<String, Collection<T>> getRenamedTypesForTesting() {
648     Multimap<String, Collection<T>> ret = HashMultimap.create();
649     for (Map.Entry<String, Property> entry : properties.entrySet()) {
650       Property prop = entry.getValue();
651       if (!prop.skipRenaming) {
652         for (Collection<T> c : prop.getTypes().allEquivalenceClasses()) {
653           if (!c.isEmpty() && !prop.typesToSkip.contains(c.iterator().next())) {
654             ret.put(entry.getKey(), c);
655           }
656         }
657       }
658     }
659     return ret;
660   }

662   /** Interface for providing the type information needed by this pass. */
663   private interface TypeSystem<T> {
664     // TODO(user): add a getUniqueName(T type) method that is guaranteed
665     // to be unique, performant and human-readable.
666 
667     /** Returns the top-most scope used by the type system (if any). */
668     StaticTypedScope<T> getRootScope();
669 
670     /** Returns the new scope started at the given function node. */
671     StaticTypedScope<T> getFunctionScope(Node node);
672 
673     /**
674      * Returns the type of the given node.
675      * @param prop Only types with this property need to be returned. In general
676      *     with type tightening, this will require no special processing, but in
677      *     the case of an unknown JSType, we might need to add in the native
678      *     types since we don't track them, but only if they have the given
679      *     property.
680      */
681     T getType(StaticTypedScope<T> scope, Node node, String prop);
682 
683     /**
684      * Returns true if a field reference on this type will invalidate all
685      * references to that field as candidates for renaming. This is true if the
686      * type is unknown or all-inclusive, as variables with such a type could be
687      * references to any object.
688      */
689     boolean isInvalidatingType(T type);
690 
691     /**
692      * Informs the given type system that a type is invalidating due to a type
693      * mismatch found during type checking.
694      */
695     void addInvalidatingType(JSType type);
696 
697     /**
698      * Returns a set of types that should be skipped given the given type.
699      * This is necessary for interfaces when using JSTypes, as all super
700      * interfaces must also be skipped.
701      */
702     ImmutableSet<T> getTypesToSkipForType(T type);
703 
704     /**
705      * Determines whether the given type is one whose properties should not be
706      * considered for renaming.
707      */
708     boolean isTypeToSkip(T type);
709 
710     /** Remove null and undefined from the options in the given type. */
711     T restrictByNotNullOrUndefined(T type);
712 
713     /**
714      * Returns the alternatives if this is a type that represents multiple
715      * types, and null if not. Union and interface types can correspond to
716      * multiple other types.
717      */
718     Iterable<T> getTypeAlternatives(T type);
719 
720     /**
721      * Returns the type in the chain from the given type that contains the given
722      * field or null if it is not found anywhere.
723      */
724     T getTypeWithProperty(String field, T type);
725 
726     /**
727      * Returns the type of the instance of which this is the prototype or null
728      * if this is not a function prototype.
729      */
730     T getInstanceFromPrototype(T type);
731 
732     /**
733      * Records that this property could be referenced from any interface that
734      * this type, or any type in its superclass chain, implements.
735      */
736     void recordInterfaces(T type, T relatedType,
737                           DisambiguateProperties<T>.Property p);

740   /** Implementation of TypeSystem using JSTypes. */
741   private static class JSTypeSystem implements TypeSystem<JSType> {
742     private final Set<JSType> invalidatingTypes;
743     private final Map<FunctionType, Iterable<ObjectType>> implementedInterfaces;
744     private JSTypeRegistry registry;
745 
746     public JSTypeSystem(AbstractCompiler compiler) {
747       registry = compiler.getTypeRegistry();
748       implementedInterfaces = new HashMap<>();
749       invalidatingTypes = new HashSet<>(ImmutableSet.of(
750           registry.getNativeType(JSTypeNative.ALL_TYPE),
751           registry.getNativeType(JSTypeNative.NO_OBJECT_TYPE),
752           registry.getNativeType(JSTypeNative.NO_TYPE),
753           registry.getNativeType(JSTypeNative.FUNCTION_PROTOTYPE),
754           registry.getNativeType(JSTypeNative.FUNCTION_INSTANCE_TYPE),
755           registry.getNativeType(JSTypeNative.OBJECT_PROTOTYPE),
756           registry.getNativeType(JSTypeNative.TOP_LEVEL_PROTOTYPE),
757           registry.getNativeType(JSTypeNative.UNKNOWN_TYPE)));
758     }
759 
760     @Override public void addInvalidatingType(JSType type) {
761       checkState(!type.isUnionType());
762       invalidatingTypes.add(type);
763     }
764 
765     @Override public StaticTypedScope<JSType> getRootScope() { return null; }
766 
767     @Override public StaticTypedScope<JSType> getFunctionScope(Node node) {
768       return null;
769     }
770 
771     @Override public JSType getType(
772         StaticTypedScope<JSType> scope, Node node, String prop) {
773       if (node.getJSType() == null) {
774         return registry.getNativeType(JSTypeNative.UNKNOWN_TYPE);
775       }
776       return node.getJSType();
777     }
778 
779     @Override public boolean isInvalidatingType(JSType type) {
780       if (type == null || invalidatingTypes.contains(type) ||
781           type.isUnknownType() /* unresolved types */) {
782         return true;
783       }
784 
785       ObjectType objType = ObjectType.cast(type);
786       return objType != null && !objType.hasReferenceName();
787     }
788 
789     @Override public ImmutableSet<JSType> getTypesToSkipForType(JSType type) {
790       type = type.restrictByNotNullOrUndefined();
791       if (type.isUnionType()) {
792         ImmutableSet.Builder<JSType> types = ImmutableSet.builder();
793         types.add(type);
794         for (JSType alt : type.toMaybeUnionType().getAlternates()) {
795           types.addAll(getTypesToSkipForTypeNonUnion(alt));
796         }
797         return types.build();
798       } else if (type.isEnumElementType()) {
799         return getTypesToSkipForType(
800             type.toMaybeEnumElementType().getPrimitiveType());
801       }
802       return ImmutableSet.copyOf(getTypesToSkipForTypeNonUnion(type));
803     }
804 
805     private static Set<JSType> getTypesToSkipForTypeNonUnion(JSType type) {
806       Set<JSType> types = new HashSet<>();
807       JSType skipType = type;
808       while (skipType != null) {
809         types.add(skipType);
810 
811         ObjectType objSkipType = skipType.toObjectType();
812         if (objSkipType != null) {
813           skipType = objSkipType.getImplicitPrototype();
814         } else {
815           break;
816         }
817       }
818       return types;
819     }
820 
821     @Override public boolean isTypeToSkip(JSType type) {
822       return type.isEnumType() || (type.autoboxesTo() != null);
823     }
824 
825     @Override public JSType restrictByNotNullOrUndefined(JSType type) {
826       return type.restrictByNotNullOrUndefined();
827     }
828 
829     @Override public Iterable<JSType> getTypeAlternatives(JSType type) {
830       if (type.isUnionType()) {
831         return type.toMaybeUnionType().getAlternatesWithoutStructuralTyping();
832       } else {
833         ObjectType objType = type.toObjectType();
834         if (objType != null &&
835             objType.getConstructor() != null &&
836             objType.getConstructor().isInterface()) {
837           List<JSType> list = new ArrayList<>();
838           for (FunctionType impl
839                    : registry.getDirectImplementors(objType)) {
840             list.add(impl.getInstanceType());
841           }
842           return list;
843         } else {
844           return null;
845         }
846       }
847     }
848 
849     @Override public ObjectType getTypeWithProperty(String field, JSType type) {
850       if (type == null) {
851         return null;
852       }
853 
854       if (type.isEnumElementType()) {
855         return getTypeWithProperty(
856             field, type.toMaybeEnumElementType().getPrimitiveType());
857       }
858 
859       if (!(type instanceof ObjectType)) {
860         if (type.autoboxesTo() != null) {
861           type = type.autoboxesTo();
862         } else {
863           return null;
864         }
865       }
866 
867       // Ignore the prototype itself at all times.
868       if ("prototype".equals(field)) {
869         return null;
870       }
871 
872       // We look up the prototype chain to find the highest place (if any) that
873       // this appears.  This will make references to overridden properties look
874       // like references to the initial property, so they are renamed alike.
875       ObjectType foundType = null;
876       ObjectType objType = ObjectType.cast(type);
877       if (objType != null && objType.getConstructor() != null
878           && objType.getConstructor().isInterface()) {
879         ObjectType topInterface = FunctionType.getTopDefiningInterface(
880             objType, field);
881         if (topInterface != null && topInterface.getConstructor() != null) {
882           foundType = topInterface.getConstructor().getPrototype();
883         }
884       } else {
885         while (objType != null && objType.getImplicitPrototype() != objType) {
886           if (objType.hasOwnProperty(field)) {
887             foundType = objType;
888           }
889           objType = objType.getImplicitPrototype();
890         }
891       }
892 
893       // If the property does not exist on the referenced type but the original
894       // type is an object type, see if any subtype has the property.
895       if (foundType == null) {
896         ObjectType maybeType = ObjectType.cast(
897             registry.getGreatestSubtypeWithProperty(type, field));
898         // getGreatestSubtypeWithProperty does not guarantee that the property
899         // is defined on the returned type, it just indicates that it might be,
900         // so we have to double check.
901         if (maybeType != null && maybeType.hasOwnProperty(field)) {
902           foundType = maybeType;
903         }
904       }
905 
906       // Unwrap templatized types, they are not unique at runtime.
907       if (foundType != null && foundType.isTemplatizedType()) {
908         foundType = foundType.toMaybeTemplatizedType().getReferencedType();
909       }
910 
911       // Since disambiguation just looks at names, we must return a uniquely named type rather
912       // than an "equivalent" type. In particular, we must manually unwrap named types
913       // so that the returned type has the correct name.
914       if (foundType != null && foundType.isNamedType()) {
915         foundType = foundType.toMaybeNamedType().getReferencedType().toMaybeObjectType();
916       }
917 
918       return foundType;
919     }
920 
921     @Override public JSType getInstanceFromPrototype(JSType type) {
922       if (type.isFunctionPrototypeType()) {
923         ObjectType prototype = (ObjectType) type;
924         FunctionType owner = prototype.getOwnerFunction();
925         if (owner.isConstructor() || owner.isInterface()) {
926           return prototype.getOwnerFunction().getInstanceType();
927         }
928       }
929       return null;
930     }
931 
932     @Override
933     public void recordInterfaces(JSType type, JSType relatedType,
934                                  DisambiguateProperties<JSType>.Property p) {
935       ObjectType objType = ObjectType.cast(type);
936       if (objType == null) {
937         return;
938       }
939       FunctionType constructor;
940       if (objType.isFunctionType()) {
941         constructor = objType.toMaybeFunctionType();
942       } else if (objType.isFunctionPrototypeType()) {
943         constructor = objType.getOwnerFunction();
944       } else {
945         constructor = objType.getConstructor();
946       }
947       if (constructor == null) {
948         return;
949       }
950       Iterable<ObjectType> interfaces = implementedInterfaces.get(constructor);
951       if (interfaces == null) {
952         interfaces = constructor.getImplementedInterfaces();
953         implementedInterfaces.put(constructor, interfaces);
954       }
955       for (ObjectType itype : interfaces) {
956         JSType top = getTypeWithProperty(p.name, itype);
957         if (top != null) {
958           p.addType(itype, top, relatedType);
959         } else {
960           recordInterfaces(itype, relatedType, p);
961         }
962         // If this interface invalidated this property, return now.
963         if (p.skipRenaming) {
964           return;
965         }
966       }
967     }
