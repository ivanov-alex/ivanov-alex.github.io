36  /**
37   * Using the infrastructure provided by VariableReferencePass, identify
38   * variables that are used only once and in a way that is safe to move, and then
39   * inline them.
40   *
41   * This pass has two "modes." One mode only inlines variables declared as
42   * constants, for legacy compiler clients. The second mode inlines any
43   * variable that we can provably inline. Note that the second mode is a
44   * superset of the first mode. We only support the first mode for
45   * backwards-compatibility with compiler clients that don't want
46   * --inline_variables.
47   *
48   * The approach of this pass is similar to {@link CrossModuleCodeMotion}
49   *
50   * @author kushal@google.com (Kushal Dave)
51   * @author nicksantos@google.com (Nick Santos)
52   */
53  class InlineVariables implements CompilerPass {
54  
55    private final AbstractCompiler compiler;
56  
57    enum Mode {
58      // Only inline things explicitly marked as constant.
59      CONSTANTS_ONLY,
60      // Locals only
61      LOCALS_ONLY,
62      ALL
63    }
64  
65    private final Mode mode;
66  
67    // Inlines all strings, even if they increase the size of the gzipped binary.
68    private final boolean inlineAllStrings;
69  
70    private final IdentifyConstants identifyConstants = new IdentifyConstants();
71  
72    InlineVariables(
73        AbstractCompiler compiler,
74        Mode mode,
75        boolean inlineAllStrings) {
76      this.compiler = compiler;
77      this.mode = mode;
78      this.inlineAllStrings = inlineAllStrings;
79    }
80  
81    @Override
82    public void process(Node externs, Node root) {
83      ReferenceCollectingCallback callback = new ReferenceCollectingCallback(
84          compiler, new InliningBehavior(), getFilterForMode());
85      callback.process(externs, root);
86    }
87  
88    private Predicate<Var> getFilterForMode() {
89      switch (mode) {
90        case ALL:
91          return Predicates.alwaysTrue();
92        case LOCALS_ONLY:
93          return new IdentifyLocals();
94        case CONSTANTS_ONLY:
95          return new IdentifyConstants();
96        default:
97          throw new IllegalStateException();
98      }
99    }

101   /**
102    * Filters variables declared as "constant", and declares them in the outer
103    * declaredConstants map.
104    *
105    * In Google coding conventions, this means anything declared with @const
106    * or named in all caps, and initialized to an immutable value.
107    * CheckConsts has already verified that these are truly constants.
108    */
109   private static class IdentifyConstants implements Predicate<Var> {
110     @Override
111     public boolean apply(Var var) {
112       return var.isInferredConst();
113     }
114   }
115 
116   /**
117    * Filters non-global variables.
118    */
119   private static class IdentifyLocals implements Predicate<Var> {
120     @Override
121     public boolean apply(Var var) {
122       return var.scope.isLocal();
123     }
124   }

126   private static class AliasCandidate {
127     private final Var alias;
128     private final ReferenceCollection refInfo;
129 
130     AliasCandidate(Var alias, ReferenceCollection refInfo) {
131       this.alias = alias;
132       this.refInfo = refInfo;
133     }
134   }

136   /**
137    * Builds up information about nodes in each scope. When exiting the
138    * scope, inspects all variables in that scope, and inlines any
139    * that we can.
140    */
141   private class InliningBehavior implements Behavior {
142 
143     /**
144      * A list of variables that should not be inlined, because their
145      * reference information is out of sync with the state of the AST.
146      */
147     private final Set<Var> staleVars = new HashSet<>();
148 
149     /**
150      * Stored possible aliases of variables that never change, with
151      * all the reference info about those variables. Hashed by the NAME
152      * node of the variable being aliased.
153      */
154     final Map<Node, AliasCandidate> aliasCandidates = new HashMap<>();
155 
156     @Override
157     public void afterExitScope(NodeTraversal t, ReferenceMap referenceMap) {
158       collectAliasCandidates(t, referenceMap);
159       doInlinesForScope(t, referenceMap);
160     }
161 
162     /**
163      * If any of the variables are well-defined and alias other variables,
164      * mark them as aliasing candidates.
165      */
166     private void collectAliasCandidates(NodeTraversal t,
167         ReferenceMap referenceMap) {
168       if (mode != Mode.CONSTANTS_ONLY) {
169         for (Iterator<Var> it = t.getScope().getVars(); it.hasNext();) {
170           Var v = it.next();
171           ReferenceCollection referenceInfo = referenceMap.getReferences(v);
172 
173           // NOTE(nicksantos): Don't handle variables that are never used.
174           // The tests are much easier to write if you don't, and there's
175           // another pass that handles unused variables much more elegantly.
176           if (referenceInfo != null && referenceInfo.references.size() >= 2 &&
177               referenceInfo.isWellDefined() &&
178               referenceInfo.isAssignedOnceInLifetime()) {
179             Reference init = referenceInfo.getInitializingReference();
180             Node value = init.getAssignedValue();
181             if (value != null && value.isName()
182                 && !value.getString().equals(v.getName())) {
183               aliasCandidates.put(value, new AliasCandidate(v, referenceInfo));
184             }
185           }
186         }
187       }
188     }
189 
190     /**
191      * For all variables in this scope, see if they are only used once.
192      * If it looks safe to do so, inline them.
193      */
194     private void doInlinesForScope(NodeTraversal t, ReferenceMap referenceMap) {
195 
196       boolean maybeModifiedArguments =
197           maybeEscapedOrModifiedArguments(t.getScope(), referenceMap);
198       for (Iterator<Var> it = t.getScope().getVars(); it.hasNext();) {
199         Var v = it.next();
200 
201         ReferenceCollection referenceInfo = referenceMap.getReferences(v);
202 
203         // referenceInfo will be null if we're in constants-only mode
204         // and the variable is not a constant.
205         if (referenceInfo == null || isVarInlineForbidden(v)) {
206           // Never try to inline exported variables or variables that
207           // were not collected or variables that have already been inlined.
208           continue;
209         } else if (isInlineableDeclaredConstant(v, referenceInfo)) {
210           Reference init = referenceInfo.getInitializingReferenceForConstants();
211           Node value = init.getAssignedValue();
212           inlineDeclaredConstant(v, value, referenceInfo.references);
213           staleVars.add(v);
214         } else if (mode == Mode.CONSTANTS_ONLY) {
215           // If we're in constants-only mode, don't run more aggressive
216           // inlining heuristics. See InlineConstantsTest.
217           continue;
218         } else {
219           inlineNonConstants(v, referenceInfo, maybeModifiedArguments);
220         }
221       }
222     }
223 
224     private boolean maybeEscapedOrModifiedArguments(
225         Scope scope, ReferenceMap referenceMap) {
226       if (scope.isLocal()) {
227         Var arguments = scope.getArgumentsVar();
228         ReferenceCollection refs = referenceMap.getReferences(arguments);
229         if (refs != null && !refs.references.isEmpty()) {
230           for (Reference ref : refs.references) {
231             Node refNode = ref.getNode();
232             Node refParent = ref.getParent();
233             // Any reference that is not a read of the arguments property
234             // consider a escape of the arguments object.
235             if (!(NodeUtil.isGet(refParent)
236                 && refNode == ref.getParent().getFirstChild()
237                 && !isLValue(refParent))) {
238               return true;
239             }
240           }
241         }
242       }
243       return false;
244     }
245 
246     private boolean isLValue(Node n) {
247       Node parent = n.getParent();
248       return (parent.isInc()
249           || parent.isDec()
250           || (NodeUtil.isAssignmentOp(parent)
251           && parent.getFirstChild() == n));
252     }

254     private void inlineNonConstants(
255         Var v, ReferenceCollection referenceInfo,
256         boolean maybeModifiedArguments) {
257       int refCount = referenceInfo.references.size();
258       Reference declaration = referenceInfo.references.get(0);
259       Reference init = referenceInfo.getInitializingReference();
260       int firstRefAfterInit = (declaration == init) ? 2 : 3;
261 
262       if (refCount > 1 &&
263           isImmutableAndWellDefinedVariable(v, referenceInfo)) {
264         // if the variable is referenced more than once, we can only
265         // inline it if it's immutable and never defined before referenced.
266         Node value;
267         if (init != null) {
268           value = init.getAssignedValue();
269         } else {
270           // Create a new node for variable that is never initialized.
271           Node srcLocation = declaration.getNode();
272           value = NodeUtil.newUndefinedNode(srcLocation);
273         }
274         Preconditions.checkNotNull(value);
275         inlineWellDefinedVariable(v, value, referenceInfo.references);
276         staleVars.add(v);
277       } else if (refCount == firstRefAfterInit) {
278         // The variable likely only read once, try some more
279         // complex inlining heuristics.
280         Reference reference = referenceInfo.references.get(
281             firstRefAfterInit - 1);
282         if (canInline(declaration, init, reference)) {
283           inline(v, declaration, init, reference);
284           staleVars.add(v);
285         }
286       } else if (declaration != init && refCount == 2) {
287         if (isValidDeclaration(declaration) && isValidInitialization(init)) {
288           // The only reference is the initialization, remove the assignment and
289           // the variable declaration.
290           Node value = init.getAssignedValue();
291           Preconditions.checkNotNull(value);
292           inlineWellDefinedVariable(v, value, referenceInfo.references);
293           staleVars.add(v);
294         }
295       }
296 
297       // If this variable was not inlined normally, check if we can
298       // inline an alias of it. (If the variable was inlined, then the
299       // reference data is out of sync. We're better off just waiting for
300       // the next pass.)
301       if (!maybeModifiedArguments &&
302           !staleVars.contains(v) && referenceInfo.isWellDefined() &&
303           referenceInfo.isAssignedOnceInLifetime()) {
304         List<Reference> refs = referenceInfo.references;
305         for (int i = 1 /* start from a read */; i < refs.size(); i++) {
306           Node nameNode = refs.get(i).getNode();
307           if (aliasCandidates.containsKey(nameNode)) {
308             AliasCandidate candidate = aliasCandidates.get(nameNode);
309             if (!staleVars.contains(candidate.alias) &&
310                 !isVarInlineForbidden(candidate.alias)) {
311               Reference aliasInit;
312               aliasInit = candidate.refInfo.getInitializingReference();
313               Node value = aliasInit.getAssignedValue();
314               Preconditions.checkNotNull(value);
315               inlineWellDefinedVariable(candidate.alias,
316                   value,
317                   candidate.refInfo.references);
318               staleVars.add(candidate.alias);
319             }
320           }
321         }
322       }
323     }

325     /**
326      * If there are any variable references in the given node tree, blacklist
327      * them to prevent the pass from trying to inline the variable.
328      */
329     private void blacklistVarReferencesInTree(Node root, Scope scope) {
330       for (Node c = root.getFirstChild(); c != null; c = c.getNext()) {
331         blacklistVarReferencesInTree(c, scope);
332       }
333 
334       if (root.isName()) {
335         staleVars.add(scope.getVar(root.getString()));
336       }
337     }

339     /**
340      * Whether the given variable is forbidden from being inlined.
341      */
342     private boolean isVarInlineForbidden(Var var) {
343       // A variable may not be inlined if:
344       // 1) The variable is exported,
345       // 2) A reference to the variable has been inlined. We're downstream
346       //    of the mechanism that creates variable references, so we don't
347       //    have a good way to update the reference. Just punt on it.
348       // 3) Don't inline the special RENAME_PROPERTY_FUNCTION_NAME
349       return var.isExtern()
350           || compiler.getCodingConvention().isExported(var.name)
351           || RenameProperties.RENAME_PROPERTY_FUNCTION_NAME.equals(var.name)
352           || staleVars.contains(var);
353     }
354 
355     /**
356      * Do the actual work of inlining a single declaration into a single
357      * reference.
358      */
359     private void inline(Var v, Reference decl, Reference init, Reference ref) {
360       Node value = init.getAssignedValue();
361       Preconditions.checkState(value != null);
362       // Check for function declarations before the value is moved in the AST.
363       boolean isFunctionDeclaration = NodeUtil.isFunctionDeclaration(value);
364       compiler.reportChangeToEnclosingScope(ref.getNode());
365       inlineValue(v, ref, value.detachFromParent());
366       if (decl != init) {
367         Node expressRoot = init.getGrandparent();
368         Preconditions.checkState(expressRoot.isExprResult());
369         NodeUtil.removeChild(expressRoot.getParent(), expressRoot);
370       }
371       // Function declarations have already been removed.
372       if (!isFunctionDeclaration) {
373         compiler.reportChangeToEnclosingScope(decl.getNode());
374         removeDeclaration(decl);
375       }
376     }
377 
378     /**
379      * Inline an immutable variable into all of its references.
380      */
381     private void inlineWellDefinedVariable(Var v, Node value,
382         List<Reference> refSet) {
383       Reference decl = refSet.get(0);
384       for (int i = 1; i < refSet.size(); i++) {
385         inlineValue(v, refSet.get(i), value.cloneTree());
386       }
387       removeDeclaration(decl);
388     }
389 
390     /**
391      * Inline a declared constant.
392      */
393     private void inlineDeclaredConstant(Var v, Node value,
394         List<Reference> refSet) {
395       // Replace the references with the constant value
396       Reference decl = null;
397 
398       for (Reference r : refSet) {
399         if (r.getNode() == v.getNameNode()) {
400           decl = r;
401         } else {
402           inlineValue(v, r, value.cloneTree());
403         }
404       }
405 
406       removeDeclaration(decl);
407     }
408 
409     /**
410      * Remove the given VAR declaration.
411      */
412     private void removeDeclaration(Reference decl) {
413       Node varNode = decl.getParent();
414       Node grandparent = decl.getGrandparent();
415 
416       compiler.reportChangeToEnclosingScope(decl.getNode());
417       varNode.removeChild(decl.getNode());
418       // Remove var node if empty
419       if (!varNode.hasChildren()) {
420         Preconditions.checkState(varNode.isVar());
421         NodeUtil.removeChild(grandparent, varNode);
422       }
423     }
424 
425     /**
426      * Replace the given reference with the given value node.
427      *
428      * @param v The variable that's referenced.
429      * @param ref The reference to replace.
430      * @param value The node tree to replace it with. This tree should be safe
431      *     to re-parent.
432      */
433     private void inlineValue(Var v, Reference ref, Node value) {
434       compiler.reportChangeToEnclosingScope(ref.getNode());
435       if (ref.isSimpleAssignmentToName()) {
436         // This is the initial assignment.
437         ref.getGrandparent().replaceChild(ref.getParent(), value);
438       } else {
439         ref.getParent().replaceChild(ref.getNode(), value);
440       }
441       blacklistVarReferencesInTree(value, v.scope);
442     }

444     /**
445      * Determines whether the given variable is declared as a constant
446      * and may be inlined.
447      */
448     private boolean isInlineableDeclaredConstant(Var var,
449         ReferenceCollection refInfo) {
450       if (!identifyConstants.apply(var)) {
451         return false;
452       }
453 
454       if (!refInfo.isAssignedOnceInLifetime()) {
455         return false;
456       }
457 
458       Reference init = refInfo.getInitializingReferenceForConstants();
459       if (init == null) {
460         return false;
461       }
462 
463       Node value = init.getAssignedValue();
464       if (value == null) {
465         // This constant is either externally defined or initialized indirectly
466         // (e.g. in an function expression used to hide
467         // temporary variables), so the constant is ineligible for inlining.
468         return false;
469       }
470 
471       // Is the constant's value immutable?
472       if (!NodeUtil.isImmutableValue(value)) {
473         return false;
474       }
475 
476       // Determine if we should really inline a String or not.
477       return !value.isString() ||
478           isStringWorthInlining(var, refInfo.references);
479     }
480 
481     /**
482      * Compute whether the given string is worth inlining.
483      */
484     private boolean isStringWorthInlining(Var var, List<Reference> refs) {
485       if (!inlineAllStrings && !var.isDefine()) {
486         int len = var.getInitialValue().getString().length() + "''".length();
487 
488         // if not inlined: var xx="value"; .. xx .. xx ..
489         // The 4 bytes per reference is just a heuristic:
490         // 2 bytes per var name plus maybe 2 bytes if we don't inline, e.g.
491         // in the case of "foo " + CONST + " bar"
492         int noInlineBytes = "var xx=;".length() + len +
493                             4 * (refs.size() - 1);
494 
495         // if inlined:
496         // I'm going to assume that half of the quotes will be eliminated
497         // thanks to constant folding, therefore I subtract 1 (2/2=1) from
498         // the string length.
499         int inlineBytes = (len - 1) * (refs.size() - 1);
500 
501         // Not inlining if doing so uses more bytes, or this constant is being
502         // defined.
503         return noInlineBytes >= inlineBytes;
504       }
505 
506       return true;
507     }

509     /**
510      * @return true if the provided reference and declaration can be safely
511      *         inlined according to our criteria
512      */
513     private boolean canInline(
514         Reference declaration,
515         Reference initialization,
516         Reference reference) {
517       if (!isValidDeclaration(declaration)
518           || !isValidInitialization(initialization)
519           || !isValidReference(reference)) {
520         return false;
521       }
522 
523       // If the value is read more than once, skip it.
524       // VAR declarations and EXPR_RESULT don't need the value, but other
525       // ASSIGN expressions parents do.
526       if (declaration != initialization &&
527           !initialization.getGrandparent().isExprResult()) {
528         return false;
529       }
530 
531       // Be very conservative and do no cross control structures or
532       // scope boundaries
533       if (declaration.getBasicBlock() != initialization.getBasicBlock()
534           || declaration.getBasicBlock() != reference.getBasicBlock()) {
535         return false;
536       }
537 
538       // Do not inline into a call node. This would change
539       // the context in which it was being called. For example,
540       //   var a = b.c;
541       //   a();
542       // should not be inlined, because it calls a in the context of b
543       // rather than the context of the window.
544       //   var a = b.c;
545       //   f(a)
546       // is OK.
547       Node value = initialization.getAssignedValue();
548       Preconditions.checkState(value != null);
549       if (value.isGetProp()
550           && reference.getParent().isCall()
551           && reference.getParent().getFirstChild() == reference.getNode()) {
552         return false;
553       }
554 
555       if (value.isFunction()) {
556         Node callNode = reference.getParent();
557         if (reference.getParent().isCall()) {
558           CodingConvention convention = compiler.getCodingConvention();
559           // Bug 2388531: Don't inline subclass definitions into class defining
560           // calls as this confused class removing logic.
561           SubclassRelationship relationship =
562               convention.getClassesDefinedByCall(callNode);
563           if (relationship != null) {
564             return false;
565           }
566 
567           // issue 668: Don't inline singleton getter methods
568           // calls as this confused class removing logic.
569           if (convention.getSingletonGetterClassName(callNode) != null) {
570             return false;
571           }
572         }
573       }
574 
575       return canMoveAggressively(value) ||
576           canMoveModerately(initialization, reference);
577     }
578 
579     /**
580      * If the value is a literal, we can cross more boundaries to inline it.
581      */
582     private boolean canMoveAggressively(Node value) {
583       // Function expressions and other mutable objects can move within
584       // the same basic block.
585       return NodeUtil.isLiteralValue(value, true)
586           || value.isFunction();
587     }

589     /**
590      * If the value of a variable is not constant, then it may read or modify
591      * state. Therefore it cannot be moved past anything else that may modify
592      * the value being read or read values that are modified.
593      */
594     private boolean canMoveModerately(
595         Reference initialization,
596         Reference reference) {
597       // Check if declaration can be inlined without passing
598       // any side-effect causing nodes.
599       Iterator<Node> it;
600       if (initialization.getParent().isVar()) {
601         it = NodeIterators.LocalVarMotion.forVar(
602             initialization.getNode(),     // NAME
603             initialization.getParent(),       // VAR
604             initialization.getGrandparent()); // VAR container
605       } else if (initialization.getParent().isAssign()) {
606         Preconditions.checkState(
607             initialization.getGrandparent().isExprResult());
608         it = NodeIterators.LocalVarMotion.forAssign(
609             initialization.getNode(),     // NAME
610             initialization.getParent(),       // ASSIGN
611             initialization.getGrandparent(),  // EXPR_RESULT
612             initialization.getGrandparent().getParent()); // EXPR container
613       } else {
614         throw new IllegalStateException("Unexpected initialization parent " +
615             initialization.getParent().toStringTree());
616       }
617       Node targetName = reference.getNode();
618       while (it.hasNext()) {
619         Node curNode = it.next();
620         if (curNode == targetName) {
621           return true;
622         }
623       }
624 
625       return false;
626     }

628     /**
629      * @return true if the reference is a normal VAR or FUNCTION declaration.
630      */
631     private boolean isValidDeclaration(Reference declaration) {
632       return (declaration.getParent().isVar()
633           && !declaration.getGrandparent().isFor())
634           || NodeUtil.isFunctionDeclaration(declaration.getParent());
635     }

637     /**
638      * @return Whether there is a initial value.
639      */
640     private boolean isValidInitialization(Reference initialization) {
641       if (initialization == null) {
642         return false;
643       } else if (initialization.isDeclaration()) {
644         // The reference is a FUNCTION declaration or normal VAR declaration
645         // with a value.
646         if (!NodeUtil.isFunctionDeclaration(initialization.getParent())
647             && initialization.getNode().getFirstChild() == null) {
648           return false;
649         }
650       } else {
651         Node parent = initialization.getParent();
652         Preconditions.checkState(
653             parent.isAssign()
654             && parent.getFirstChild() == initialization.getNode());
655       }
656 
657       Node n = initialization.getAssignedValue();
658       if (n.isFunction()) {
659         return compiler.getCodingConvention().isInlinableFunction(n);
660       }
661 
662       return true;
663     }

665     /**
666      * @return true if the reference is a candidate for inlining
667      */
668     private boolean isValidReference(Reference reference) {
669       return !reference.isDeclaration() && !reference.isLvalue();
670     }

672     /**
673      * Determines whether the reference collection describes a variable that
674      * is initialized to an immutable value, never modified, and defined before
675      * every reference.
676      */
677     private boolean isImmutableAndWellDefinedVariable(Var v,
678         ReferenceCollection refInfo) {
679       List<Reference> refSet = refInfo.references;
680       int startingReadRef = 1;
681       Reference refDecl = refSet.get(0);
682       if (!isValidDeclaration(refDecl)) {
683         return false;
684       }
685 
686       boolean isNeverAssigned = refInfo.isNeverAssigned();
687       // For values that are never assigned, only the references need to be
688       // checked.
689       if (!isNeverAssigned) {
690         Reference refInit = refInfo.getInitializingReference();
691         if (!isValidInitialization(refInit)) {
692           return false;
693         }
694 
695         if (refDecl != refInit) {
696           Preconditions.checkState(refInit == refSet.get(1));
697           startingReadRef = 2;
698         }
699 
700         if (!refInfo.isWellDefined()) {
701           return false;
702         }
703 
704         Node value = refInit.getAssignedValue();
705         Preconditions.checkNotNull(value);
706 
707         boolean isImmutableValueWorthInlining =
708             NodeUtil.isImmutableValue(value) &&
709             (!value.isString() ||
710                 isStringWorthInlining(v, refInfo.references));
711         boolean isInlinableThisAlias =
712             value.isThis() &&
713             !refInfo.isEscaped();
714         if (!isImmutableValueWorthInlining && !isInlinableThisAlias) {
715           return false;
716         }
717       }
718
