388   private static void addMethod(
389       JSTypeRegistry registry, ObjectType receivingType, String methodName,
390       JSType returnType) {
391     receivingType.defineDeclaredProperty(methodName,
392         new FunctionBuilder(registry).withReturnType(returnType).build(),
393         null);
394   }

423   /**
424    * Asserts that a Node representing a type expression resolves to the
425    * correct {@code JSType}.
426    */
427   protected void assertTypeEquals(JSType expected, Node actual) {
428     assertTypeEquals(expected, new JSTypeExpression(actual, ""));

431   /**
432    * Asserts that a a type expression resolves to the correct {@code JSType}.
433    */
434   protected void assertTypeEquals(JSType expected, JSTypeExpression actual) {
435     assertEquals(expected, resolve(actual));
436   }

438   /**
439    * Resolves a type expression, expecting the given warnings.
440    */
441   protected JSType resolve(JSTypeExpression n, String... warnings) {
442     errorReporter.setWarnings(warnings);
443     return n.evaluate(null, registry);
