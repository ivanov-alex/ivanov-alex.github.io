32  /**
33   * Replaces calls to id generators with ids.
34   *
35   * Use this to get unique and short ids.
36   *

38  class ReplaceIdGenerators implements CompilerPass {
39    static final DiagnosticType NON_GLOBAL_ID_GENERATOR_CALL =
40        DiagnosticType.error(
41            "JSC_NON_GLOBAL_ID_GENERATOR_CALL",
42            "Id generator call must be in the global scope");
43  
44    static final DiagnosticType CONDITIONAL_ID_GENERATOR_CALL =
45        DiagnosticType.error(
46            "JSC_CONDITIONAL_ID_GENERATOR_CALL",
47            "Id generator call must be unconditional");
48  
49    static final DiagnosticType CONFLICTING_GENERATOR_TYPE =
50        DiagnosticType.error(
51            "JSC_CONFLICTING_ID_GENERATOR_TYPE",
52            "Id generator can only be one of " +
53            "consistent, inconsistent, mapped or stable.");
54  
55    static final DiagnosticType INVALID_GENERATOR_ID_MAPPING =
56        DiagnosticType.error(
57            "JSC_INVALID_GENERATOR_ID_MAPPING",
58            "Invalid generator id mapping. {0}");
59  
60    static final DiagnosticType MISSING_NAME_MAP_FOR_GENERATOR =
61        DiagnosticType.warning(
62            "JSC_MISSING_NAME_MAP_FOR_GENERATOR",
63            "The mapped id generator, does not have a renaming map supplied.");
64  
65    static final DiagnosticType INVALID_GENERATOR_PARAMETER =
66        DiagnosticType.warning(
67            "JSC_INVALID_GENERATOR_PARAMETER",
68            "An id generator must be called with a literal.");
69  
70    static final DiagnosticType SHORTHAND_FUNCTION_NOT_SUPPORTED_IN_ID_GEN =
71        DiagnosticType.error(
72            "JSC_SHORTHAND_FUNCTION_NOT_SUPPORTED_IN_ID_GEN",
73            "Object literal shorthand functions is not allowed in the "
74            + "arguments of an id generator");
75  
76    static final DiagnosticType SHORTHAND_ASSIGNMENT_NOT_SUPPORTED_IN_ID_GEN =
77        DiagnosticType.error(
78            "JSC_SHORTHAND_ASSIGNMENT_NOT_SUPPORTED_IN_ID_GEN",
79            "Object literal shorthand assignment is not allowed in the "
80            + "arguments of an id generator");
81  
82    static final DiagnosticType COMPUTED_PROP_NOT_SUPPORTED_IN_ID_GEN =
83        DiagnosticType.error(
84            "JSC_COMPUTED_PROP_NOT_SUPPORTED_IN_ID_GEN",
85            "Object literal computed property name is not allowed in the "
86            + "arguments of an id generator");

89    private final AbstractCompiler compiler;
90    private final Map<String, NameSupplier> nameGenerators;
91    private final Map<String, Map<String, String>> consistNameMap;
92  
93    private final Map<String, Map<String, String>> idGeneratorMaps;
94    private final Map<String, BiMap<String, String>> previousMap;
95  
96    private final boolean generatePseudoNames;

98    public static final RenamingMap UNIQUE = new UniqueRenamingToken();
99  
100   private static class UniqueRenamingToken implements RenamingMap {
101     @Override public String get(String value) { return null; }
102   }
103 
104   public ReplaceIdGenerators(
105       AbstractCompiler compiler, Map<String, RenamingMap> idGens,
106       boolean generatePseudoNames,
107       String previousMapSerialized) {
108     this.compiler = compiler;
109     this.generatePseudoNames = generatePseudoNames;
110     nameGenerators = new LinkedHashMap<>();
111     idGeneratorMaps = new LinkedHashMap<>();
112     consistNameMap = new LinkedHashMap<>();
113 
114     Map<String, BiMap<String, String>> previousMap;
115     previousMap = IdMappingUtil.parseSerializedIdMappings(previousMapSerialized);
116     this.previousMap = previousMap;
117 
118     if (idGens != null) {
119       for (Entry<String, RenamingMap> gen : idGens.entrySet()) {
120         String name = gen.getKey();
121         RenamingMap map = gen.getValue();
122         if (map instanceof UniqueRenamingToken) {
123           nameGenerators.put(name,
124               createNameSupplier(
125                   RenameStrategy.INCONSISTENT, previousMap.get(name)));
126         } else {
127           nameGenerators.put(name,
128               createNameSupplier(
129                   RenameStrategy.MAPPED, map));
130         }
131         idGeneratorMaps.put(name, new LinkedHashMap<String, String>());
132       }
133     }
134   }
135 
136   enum RenameStrategy {
137     CONSISTENT,
138     INCONSISTENT,
139     MAPPED,
140     STABLE
141   }
142 
143   private static interface NameSupplier {
144     String getName(String id, String name);
145     RenameStrategy getRenameStrategy();
146   }
147 
148   private static class ObfuscatedNameSuppier implements NameSupplier {
149     private final NameGenerator generator;
150     private final Map<String, String> previousMappings;
151     private RenameStrategy renameStrategy;
152 
153     public ObfuscatedNameSuppier(
154         RenameStrategy renameStrategy, BiMap<String, String> previousMappings) {
155       this.previousMappings = previousMappings.inverse();
156       this.generator =
157           new NameGenerator(previousMappings.keySet(), "", null);
158       this.renameStrategy = renameStrategy;
159     }
160 
161     @Override
162     public String getName(String id, String name) {
163       String newName = previousMappings.get(id);
164       if (newName == null) {
165         newName = generator.generateNextName();
166       }
167       return newName;
168     }
169 
170     @Override
171     public RenameStrategy getRenameStrategy() {
172       return renameStrategy;
173     }
174   }
175 
176   private static class PseudoNameSuppier implements NameSupplier {
177     private int counter = 0;
178     private RenameStrategy renameStrategy;
179 
180     public PseudoNameSuppier(RenameStrategy renameStrategy) {
181       this.renameStrategy = renameStrategy;
182     }
183 
184     @Override
185     public String getName(String id, String name) {
186       if (renameStrategy == RenameStrategy.INCONSISTENT) {
187         return name + "$" + counter++;
188       }
189       return name + "$0";
190     }
191 
192     @Override
193     public RenameStrategy getRenameStrategy() {
194       return renameStrategy;
195     }
196   }
197 
198   private static class StableNameSupplier implements NameSupplier {
199     @Override
200     public String getName(String id, String name) {
201       return Base64.base64EncodeInt(name.hashCode());
202     }
203     @Override
204     public RenameStrategy getRenameStrategy() {
205       return RenameStrategy.STABLE;
206     }
207   }
208 
209   private static class MappedNameSupplier implements NameSupplier {
210     private final RenamingMap map;
211 
212     MappedNameSupplier(RenamingMap map) {
213       this.map = map;
214     }
215 
216     @Override
217     public String getName(String id, String name) {
218       return map.get(name);
219     }
220 
221     @Override
222     public RenameStrategy getRenameStrategy() {
223       return RenameStrategy.MAPPED;
224     }
225   }

227   private NameSupplier createNameSupplier(
228       RenameStrategy renameStrategy, BiMap<String, String> previousMappings) {
229     previousMappings = previousMappings != null ?
230         previousMappings :
231         ImmutableBiMap.<String, String>of();
232     if (renameStrategy == RenameStrategy.STABLE) {
233       return new StableNameSupplier();
234     } else if (generatePseudoNames) {
235       return new PseudoNameSuppier(renameStrategy);
236     } else {
237       return new ObfuscatedNameSuppier(renameStrategy, previousMappings);
238     }
239   }
240 
241   private static NameSupplier createNameSupplier(
242       RenameStrategy renameStrategy, RenamingMap mappings) {
243     Preconditions.checkState(renameStrategy == RenameStrategy.MAPPED);
244     return new MappedNameSupplier(mappings);
245   }

247   private class GatherGenerators extends AbstractPostOrderCallback {
248 
249     @Override
250     public void visit(NodeTraversal t, Node n, Node parent) {
251       JSDocInfo doc = n.getJSDocInfo();
252       if (doc == null) {
253         return;
254       }
255 
256       int numGeneratorAnnotations =
257           (doc.isConsistentIdGenerator() ? 1 : 0) +
258           (doc.isIdGenerator() ? 1 : 0) +
259           (doc.isStableIdGenerator() ? 1 : 0) +
260           (doc.isMappedIdGenerator() ? 1 : 0);
261       if (numGeneratorAnnotations == 0) {
262         return;
263       } else if (numGeneratorAnnotations > 1) {
264         compiler.report(t.makeError(n, CONFLICTING_GENERATOR_TYPE));
265       }
266 
267       String name = null;
268       if (n.isAssign()) {
269         name = n.getFirstChild().getQualifiedName();
270       } else if (NodeUtil.isNameDeclaration(n)) {
271         name = n.getFirstChild().getString();
272       } else if (n.isFunction()){
273         name = n.getFirstChild().getString();
274         if (name.isEmpty()) {
275           return;
276         }
277       }
278 
279       if (doc.isConsistentIdGenerator()) {
280         consistNameMap.put(name, new LinkedHashMap<String, String>());
281         nameGenerators.put(
282             name, createNameSupplier(
283                 RenameStrategy.CONSISTENT, previousMap.get(name)));
284       } else if (doc.isStableIdGenerator()) {
285         nameGenerators.put(
286             name, createNameSupplier(
287                 RenameStrategy.STABLE, previousMap.get(name)));
288       } else if (doc.isIdGenerator()) {
289         nameGenerators.put(
290             name, createNameSupplier(
291                 RenameStrategy.INCONSISTENT, previousMap.get(name)));
292       } else if (doc.isMappedIdGenerator()) {
293         NameSupplier supplier = nameGenerators.get(name);
294         if (supplier == null
295             || supplier.getRenameStrategy() != RenameStrategy.MAPPED) {
296           compiler.report(t.makeError(n, MISSING_NAME_MAP_FOR_GENERATOR));
297           // skip registering the name in the list of Generators if there no
298           // mapping.
299           return;
300         }
301       } else {
302         throw new IllegalStateException("unexpected");
303       }
304       idGeneratorMaps.put(name, new LinkedHashMap<String, String>());
305     }
306   }
307 
308   @Override
309   public void process(Node externs, Node root) {
310     NodeTraversal.traverseEs6(compiler, root, new GatherGenerators());
311     if (!nameGenerators.isEmpty()) {
312       NodeTraversal.traverseEs6(compiler, root, new ReplaceGenerators());
313     }
314   }

316   private class ReplaceGenerators extends AbstractPostOrderCallback {
317     @Override
318     public void visit(NodeTraversal t, Node n, Node parent) {
319       if (!n.isCall()) {
320         return;
321       }
322 
323       String callName = n.getFirstChild().getQualifiedName();
324       NameSupplier nameGenerator = nameGenerators.get(callName);
325       if (nameGenerator == null) {
326         return;
327       }
328 
329       if (!t.inGlobalHoistScope()
330           && nameGenerator.getRenameStrategy() == RenameStrategy.INCONSISTENT) {
331         // Warn about calls not in the global scope.
332         compiler.report(t.makeError(n, NON_GLOBAL_ID_GENERATOR_CALL));
333         return;
334       }
335 
336       if (nameGenerator.getRenameStrategy() == RenameStrategy.INCONSISTENT) {
337         for (Node ancestor : n.getAncestors()) {
338           if (NodeUtil.isControlStructure(ancestor)) {
339             // Warn about conditional calls.
340             compiler.report(t.makeError(n, CONDITIONAL_ID_GENERATOR_CALL));
341             return;
342           }
343         }
344       }
345 
346       Node arg = n.getFirstChild().getNext();
347       if (arg == null) {
348         compiler.report(t.makeError(n, INVALID_GENERATOR_PARAMETER));
349       } else if (arg.isString()) {
350         String rename = getObfuscatedName(
351             arg, callName, nameGenerator, arg.getString());
352         parent.replaceChild(n, IR.string(rename));
353         compiler.reportCodeChange();
354       } else if (arg.isObjectLit()) {
355         for (Node key : arg.children()) {
356           if (key.isMemberFunctionDef()) {
357             compiler.report(t.makeError(n, SHORTHAND_FUNCTION_NOT_SUPPORTED_IN_ID_GEN));
358             return;
359           }
360           if (key.isStringKey() && !key.hasChildren()) {
361             compiler.report(t.makeError(n, SHORTHAND_ASSIGNMENT_NOT_SUPPORTED_IN_ID_GEN));
362             return;
363           }
364           if (key.isComputedProp()) {
365             compiler.report(t.makeError(n, COMPUTED_PROP_NOT_SUPPORTED_IN_ID_GEN));
366             return;
367           }
368 
369           String rename = getObfuscatedName(
370               key, callName, nameGenerator, key.getString());
371           key.setString(rename);
372           // Prevent standard renaming by marking the key as quoted.
373           key.putBooleanProp(Node.QUOTED_PROP, true);
374         }
375         arg.detachFromParent();
376         parent.replaceChild(n, arg);
377         compiler.reportCodeChange();
378       } else {
379         compiler.report(t.makeError(n, INVALID_GENERATOR_PARAMETER));
380       }
381     }
382 
383     private String getObfuscatedName(
384         Node id, String callName, NameSupplier nameGenerator, String name) {
385       String rename = null;
386       Map<String, String> idGeneratorMap = idGeneratorMaps.get(callName);
387       String instanceId = getIdForGeneratorNode(
388           nameGenerator.getRenameStrategy() != RenameStrategy.INCONSISTENT, id);
389       if (nameGenerator.getRenameStrategy() == RenameStrategy.CONSISTENT) {
390         Map<String, String> entry = consistNameMap.get(callName);
391         rename = entry.get(instanceId);
392         if (rename == null) {
393           rename = nameGenerator.getName(instanceId, name);
394           entry.put(instanceId, rename);
395         }
396       } else {
397         rename = nameGenerator.getName(instanceId, name);
398       }
399       idGeneratorMap.put(rename, instanceId);
400       return rename;
401     }
402   }

409   public String getSerializedIdMappings() {
410     return IdMappingUtil.generateSerializedIdMappings(idGeneratorMaps);

413   static String getIdForGeneratorNode(boolean consistent, Node n) {
414     Preconditions.checkState(n.isString() || n.isStringKey());
415     if (consistent) {
416       return n.getString();
417     } else {
418       return n.getSourceFileName() + ':' + n.getLineno() + ":" + n.getCharno();
419     }
