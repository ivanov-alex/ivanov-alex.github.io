54   * has annotated a variable (or property) with one type, but has assigned
55   * another type to it.

58   * type-based optimizations (like AmbiguateProperties) can be fault-tolerant.
59   *
60   * @author nicksantos@google.com (Nick Santos)
61   */

154   TypeValidator(AbstractCompiler compiler) {
155     this.compiler = compiler;
156     this.typeRegistry = compiler.getTypeRegistry();
157     this.allValueTypes = typeRegistry.createUnionType(
158         STRING_TYPE, NUMBER_TYPE, BOOLEAN_TYPE, NULL_TYPE, VOID_TYPE);
159     this.nullOrUndefined = typeRegistry.createUnionType(
160         NULL_TYPE, VOID_TYPE);
161     this.reportUnnecessaryCasts = ((Compiler) compiler).getOptions().enables(
162         DiagnosticGroups.UNNECESSARY_CASTS);
163   }

165   /**
166    * Utility function for getting a function type from a var.
167    */
168   static FunctionType getFunctionType(@Nullable TypedVar v) {
169     JSType t = v == null ? null : v.getType();
170     ObjectType o = t == null ? null : t.dereference();
171     return JSType.toMaybeFunctionType(o);
172   }

174   /**
175    * Utility function for getting an instance type from a var pointing
176    * to the constructor.
177    */
178   static ObjectType getInstanceOfCtor(@Nullable TypedVar v) {
179     FunctionType ctor = getFunctionType(v);
180     if (ctor != null && ctor.isConstructor()) {
181       return ctor.getInstanceType();
182     }
183     return null;
184   }

220    * Expect the type to be an object, or a type convertible to object. If the
221    * expectation is not met, issue a warning at the provided node's source code
222    * position.
223    * @return True if there was no warning, false if there was a mismatch.
224    */
225   boolean expectObject(NodeTraversal t, Node n, JSType type, String msg) {
226     if (!type.matchesObjectContext()) {
227       mismatch(t, n, msg, type, OBJECT_TYPE);
228       return false;
229     }
230     return true;

299   /**
300    * Expect the type to be anything but the null or void type. If the
301    * expectation is not met, issue a warning at the provided node's
302    * source code position. Note that a union type that includes the
303    * void type and at least one other type meets the expectation.
304    * @return Whether the expectation was met.
305    */
306   boolean expectNotNullOrUndefined(
307       NodeTraversal t, Node n, JSType type, String msg, JSType expectedType) {
308     if (!type.isNoType() && !type.isUnknownType()
309         && type.isSubtype(nullOrUndefined)
310         && !containsForwardDeclaredUnresolvedName(type)) {
311 
312       // There's one edge case right now that we don't handle well, and
313       // that we don't want to warn about.
314       // if (this.x == null) {
315       //   this.initializeX();
316       //   this.x.foo();
317       // }
318       // In this case, we incorrectly type x because of how we
319       // infer properties locally. See issue 109.
320       // http://code.google.com/p/closure-compiler/issues/detail?id=109
321       //
322       // We do not do this inference globally.
323       if (n.isGetProp() &&
324           !t.inGlobalScope() && type.isNullType()) {
325         return true;
326       }
327 
328       mismatch(t, n, msg, type, expectedType);
329       return false;
330     }
331     return true;
332   }

334   private static boolean containsForwardDeclaredUnresolvedName(JSType type) {
335     if (type.isUnionType()) {
336       for (JSType alt : type.toMaybeUnionType().getAlternates()) {
337         if (containsForwardDeclaredUnresolvedName(alt)) {
338           return true;
339         }
340       }
341     }
342     return type.isNoResolvedType();
343   }

345   /**
346    * Expect that the type of a switch condition matches the type of its
347    * case condition.
348    */
349   void expectSwitchMatchesCase(NodeTraversal t, Node n, JSType switchType,
350       JSType caseType) {
351     // ECMA-262, page 68, step 3 of evaluation of CaseBlock,
352     // but allowing extra autoboxing.
353     // TODO(user): remove extra conditions when type annotations
354     // in the code base have adapted to the change in the compiler.
355     if (!switchType.canTestForShallowEqualityWith(caseType) &&
356         (caseType.autoboxesTo() == null ||
357         !caseType.autoboxesTo().isSubtype(switchType))) {
358       mismatch(t, n.getFirstChild(),
359           "case expression doesn't match switch",
360           caseType, switchType);
361     } else if (!switchType.canTestForShallowEqualityWith(caseType)
362         && (caseType.autoboxesTo() == null
363         || !caseType.autoboxesTo()
364         .isSubtypeWithoutStructuralTyping(switchType))) {
365       recordStructuralInterfaceUses(caseType, switchType);
366     }
367   }

369   /**
370    * Expect that the first type can be addressed with GETELEM syntax,
371    * and that the second type is the right type for an index into the
372    * first type.
373    *
374    * @param t The node traversal.
375    * @param n The GETELEM node to issue warnings on.
376    * @param objType The type of the left side of the GETELEM.
377    * @param indexType The type inside the brackets of the GETELEM.
378    */
379   void expectIndexMatch(NodeTraversal t, Node n, JSType objType,
380                         JSType indexType) {
381     Preconditions.checkState(n.isGetElem());
382     Node indexNode = n.getLastChild();
383     if (objType.isStruct()) {
384       report(JSError.make(indexNode,
385                           ILLEGAL_PROPERTY_ACCESS, "'[]'", "struct"));
386     }
387     if (objType.isUnknownType()) {
388       expectStringOrNumber(t, indexNode, indexType, "property access");
389     } else {
390       ObjectType dereferenced = objType.dereference();
391       if (dereferenced != null && dereferenced
392           .getTemplateTypeMap()
393           .hasTemplateKey(typeRegistry.getObjectIndexKey())) {
394         expectCanAssignTo(t, indexNode, indexType, dereferenced
395             .getTemplateTypeMap().getTemplateType(typeRegistry.getObjectIndexKey()),
396             "restricted index type");
397       } else if (dereferenced != null && dereferenced.isArrayType()) {
398         expectNumber(t, indexNode, indexType, "array access");
399       } else if (objType.matchesObjectContext()) {
400         expectString(t, indexNode, indexType, "property access");
401       } else {
402         mismatch(t, n, "only arrays or objects can be accessed",
403             objType,
404             typeRegistry.createUnionType(ARRAY_TYPE, OBJECT_TYPE));
405       }
406     }
407   }

409   /**
410    * Expect that the first type can be assigned to a symbol of the second
411    * type.
412    *
413    * @param t The node traversal.
414    * @param n The node to issue warnings on.
415    * @param rightType The type on the RHS of the assign.
416    * @param leftType The type of the symbol on the LHS of the assign.
417    * @param owner The owner of the property being assigned to.
418    * @param propName The name of the property being assigned to.
419    * @return True if the types matched, false otherwise.
420    */
421   boolean expectCanAssignToPropertyOf(NodeTraversal t, Node n, JSType rightType,
422       JSType leftType, Node owner, String propName) {
423     // The NoType check is a hack to make typedefs work OK.
424     if (!leftType.isNoType() && !rightType.isSubtype(leftType)) {
425       // Do not type-check interface methods, because we expect that
426       // they will have dummy implementations that do not match the type
427       // annotations.
428       JSType ownerType = getJSType(owner);
429       if (ownerType.isFunctionPrototypeType()) {
430         FunctionType ownerFn = ownerType.toObjectType().getOwnerFunction();
431         if (ownerFn.isInterface()
432             && rightType.isFunctionType() && leftType.isFunctionType()) {
433           return true;
434         }
435       }
436 
437       mismatch(t, n,
438           "assignment to property " + propName + " of " +
439           typeRegistry.getReadableTypeName(owner),
440           rightType, leftType);
441       return false;
442     } else if (!leftType.isNoType()
443         && !rightType.isSubtypeWithoutStructuralTyping(leftType)){
444       recordStructuralInterfaceUses(rightType, leftType);
445     }
446     return true;
447   }

495   /**
496    * Expect that the first type is the direct superclass of the second type.
497    *
498    * @param t The node traversal.
499    * @param n The node where warnings should point to.
500    * @param superObject The expected super instance type.
501    * @param subObject The sub instance type.
502    */
503   void expectSuperType(NodeTraversal t, Node n, ObjectType superObject,
504       ObjectType subObject) {
505     FunctionType subCtor = subObject.getConstructor();
506     ObjectType implicitProto = subObject.getImplicitPrototype();
507     ObjectType declaredSuper =
508         implicitProto == null ? null : implicitProto.getImplicitPrototype();
509     if (declaredSuper != null && declaredSuper.isTemplatizedType()) {
510       declaredSuper =
511           declaredSuper.toMaybeTemplatizedType().getReferencedType();
512     }
513     if (declaredSuper != null &&
514         !(superObject instanceof UnknownType) &&
515         !declaredSuper.isEquivalentTo(superObject)) {
516       if (declaredSuper.isEquivalentTo(getNativeType(OBJECT_TYPE))) {
517         registerMismatch(superObject, declaredSuper, report(
518             t.makeError(n, MISSING_EXTENDS_TAG_WARNING, subObject.toString())));
519       } else {
520         mismatch(n, "mismatch in declaration of superclass type",
521             superObject, declaredSuper);
522       }
523 
524       // Correct the super type.
525       if (!subCtor.hasCachedValues()) {
526         subCtor.setPrototypeBasedOn(superObject);
527       }
528     }
529   }

531   /**
532    * Expect that the first type can be cast to the second type. The first type
533    * must have some relationship with the second.
534    *
535    * @param t The node traversal.
536    * @param n The node where warnings should point.
537    * @param type The type being cast from.
538    * @param castType The type being cast to.
539    */
540   void expectCanCast(NodeTraversal t, Node n, JSType castType, JSType type) {
541     if (!type.canCastTo(castType)) {
542       registerMismatch(type, castType, report(t.makeError(n, INVALID_CAST,
543           type.toString(), castType.toString())));
544     } else if (!type.isSubtypeWithoutStructuralTyping(castType)){
545       recordStructuralInterfaceUses(type, castType);
546     }
547   }

549   /**
550    * Expect that casting type to castType is necessary. A cast is considered
551    * unnecessary if type is a subtype of castType, or identical to castType.
552    *
553    * @param t The node traversal.
554    * @param n The node where warnings should point.
555    * @param castType The type being cast to.
556    * @param type The type being cast from.
557    */
558   void expectCastIsNecessary(NodeTraversal t, Node n, JSType castType, JSType type) {
559     if (!reportUnnecessaryCasts) {
560       return;
561     }
562 
563     // If either type is "no resolved type" don't report an error, because we can't
564     // know if the cast is necessary or not.
565     if (type.isNoResolvedType() || castType.isNoResolvedType()) {
566       return;
567     }
568 
569     if (type.isEquivalentTo(castType) ||
570         (type.isSubtype(castType) && !castType.isSubtype(type))) {
571       report(t.makeError(n, UNNECESSARY_CAST,
572           type.toString(), castType.toString()));
573     }
574   }

576   /**
577    * Expect that the given variable has not been declared with a type.
578    *
579    * @param sourceName The name of the source file we're in.
580    * @param n The node where warnings should point to.
581    * @param parent The parent of {@code n}.
582    * @param var The variable that we're checking.
583    * @param variableName The name of the variable.
584    * @param newType The type being applied to the variable. Mostly just here
585    *     for the benefit of the warning.
586    * @return The variable we end up with. Most of the time, this will just
587    *     be {@code var}, but in some rare cases we will need to declare
588    *     a new var with new source info.
589    */
590   TypedVar expectUndeclaredVariable(String sourceName, CompilerInput input,
591       Node n, Node parent, TypedVar var, String variableName, JSType newType) {
592     TypedVar newVar = var;
593     boolean allowDupe = false;
594     if (n.isGetProp() ||
595         NodeUtil.isObjectLitKey(n)) {
596       JSDocInfo info = n.getJSDocInfo();
597       if (info == null) {
598         info = parent.getJSDocInfo();
599       }
600       allowDupe =
601           info != null && info.getSuppressions().contains("duplicate");
602     }
603 
604     JSType varType = var.getType();
605 
606     // Only report duplicate declarations that have types. Other duplicates
607     // will be reported by the syntactic scope creator later in the
608     // compilation process.
609     if (varType != null &&
610         varType != typeRegistry.getNativeType(UNKNOWN_TYPE) &&
611         newType != null &&
612         newType != typeRegistry.getNativeType(UNKNOWN_TYPE)) {
613       // If there are two typed declarations of the same variable, that
614       // is an error and the second declaration is ignored, except in the
615       // case of native types. A null input type means that the declaration
616       // was made in TypedScopeCreator#createInitialScope and is a
617       // native type. We should redeclare it at the new input site.
618       if (var.input == null) {
619         TypedScope s = var.getScope();
620         s.undeclare(var);
621         newVar = s.declare(variableName, n, varType, input, false);
622 
623         n.setJSType(varType);
624         if (parent.isVar()) {
625           if (n.getFirstChild() != null) {
626             n.getFirstChild().setJSType(varType);
627           }
628         } else {
629           Preconditions.checkState(parent.isFunction());
630           parent.setJSType(varType);
631         }
632       } else {
633         // Always warn about duplicates if the overridden type does not
634         // match the original type.
635         //
636         // If the types match, suppress the warning iff there was a @suppress
637         // tag, or if the original declaration was a stub.
638         if (!(allowDupe ||
639               var.getParentNode().isExprResult()) ||
640             !newType.isEquivalentTo(varType)) {
641 
642           if (newType.isEquivalentTo(varType)) {
643             report(JSError.make(n, DUP_VAR_DECLARATION,
644                 variableName, var.getInputName(),
645                 String.valueOf(var.nameNode.getLineno())));
646           } else {
647             report(JSError.make(n, DUP_VAR_DECLARATION_TYPE_MISMATCH,
648                 variableName, newType.toString(), var.getInputName(),
649                 String.valueOf(var.nameNode.getLineno()),
650                 varType.toString()));
651           }
652         }
653       }
654     }
655 
656     return newVar;
657   }

659   /**
660    * Expect that all properties on interfaces that this type implements are
661    * implemented and correctly typed.
662    */
663   void expectAllInterfaceProperties(NodeTraversal t, Node n,
664       FunctionType type) {
665     ObjectType instance = type.getInstanceType();
666     for (ObjectType implemented : type.getAllImplementedInterfaces()) {
667       if (implemented.getImplicitPrototype() != null) {
668         for (String prop :
669              implemented.getImplicitPrototype().getOwnPropertyNames()) {
670           expectInterfaceProperty(t, n, instance, implemented, prop);
671         }
672       }
673     }
674   }

676   /**
677    * Expect that the property in an interface that this type implements is
678    * implemented and correctly typed.
679    */
680   private void expectInterfaceProperty(NodeTraversal t, Node n,
681       ObjectType instance, ObjectType implementedInterface, String prop) {
682     StaticTypedSlot<JSType> propSlot = instance.getSlot(prop);
683     if (propSlot == null) {
684       // Not implemented
685       String sourceName = n.getSourceFileName();
686       sourceName = sourceName == null ? "" : sourceName;
687       registerMismatch(instance, implementedInterface,
688           report(JSError.make(n,
689           INTERFACE_METHOD_NOT_IMPLEMENTED,
690           prop, implementedInterface.toString(), instance.toString())));
691     } else {
692       Node propNode = propSlot.getDeclaration() == null ?
693           null : propSlot.getDeclaration().getNode();
694 
695       // Fall back on the constructor node if we can't find a node for the
696       // property.
697       propNode = propNode == null ? n : propNode;
698 
699       JSType found = propSlot.getType();
700       found = found.restrictByNotNullOrUndefined();
701 
702       JSType required
703           = implementedInterface.getImplicitPrototype().getPropertyType(prop);
704       TemplateTypeMap typeMap = implementedInterface.getTemplateTypeMap();
705       if (!typeMap.isEmpty()) {
706         TemplateTypeMapReplacer replacer = new TemplateTypeMapReplacer(
707             typeRegistry, typeMap);
708         required = required.visit(replacer);
709       }
710       required = required.restrictByNotNullOrUndefined();
711 
712       if (!found.isSubtype(required)) {
713         // Implemented, but not correctly typed
714         FunctionType constructor =
715             implementedInterface.toObjectType().getConstructor();
716         registerMismatch(found, required, report(t.makeError(propNode,
717             HIDDEN_INTERFACE_PROPERTY_MISMATCH, prop,
718             constructor.getTopMostDefiningType(prop).toString(),
719             required.toString(), found.toString())));
720       }
721     }
722   }

743   private void recordStructuralInterfaceUses(JSType found, JSType required) {
744     boolean strictMismatch =
745         !found.isSubtypeWithoutStructuralTyping(required)
746         && !required.isSubtypeWithoutStructuralTyping(found);
747     boolean mismatch = !found.isSubtype(required) && !required.isSubtype(found);
748     if (strictMismatch && !mismatch) {
749       implicitStructuralInterfaceUses.add(new TypeMismatch(found, required, null));
750     }
751   }
752 
753   private void registerMismatch(JSType found, JSType required, JSError error) {
754     // Don't register a mismatch for differences in null or undefined or if the
755     // code didn't downcast.
756     found = found.restrictByNotNullOrUndefined();
757     required = required.restrictByNotNullOrUndefined();
758 
759     if (found.isSubtype(required) || required.isSubtype(found)) {
760       boolean strictMismatch =
761         !found.isSubtypeWithoutStructuralTyping(required)
762         && !required.isSubtypeWithoutStructuralTyping(found);
763       if (strictMismatch) {
764         implicitStructuralInterfaceUses.add(new TypeMismatch(found, required, error));
765       }
766       return;
767     }
768 
769     mismatches.add(new TypeMismatch(found, required, error));
770 
771     if (found.isFunctionType() &&
772         required.isFunctionType()) {
773       FunctionType fnTypeA = found.toMaybeFunctionType();
774       FunctionType fnTypeB = required.toMaybeFunctionType();
775       Iterator<Node> paramItA = fnTypeA.getParameters().iterator();
776       Iterator<Node> paramItB = fnTypeB.getParameters().iterator();
777       while (paramItA.hasNext() && paramItB.hasNext()) {
778         registerIfMismatch(paramItA.next().getJSType(),
779             paramItB.next().getJSType(), error);
780       }
781 
782       registerIfMismatch(
783           fnTypeA.getReturnType(), fnTypeB.getReturnType(), error);
784     }
785   }
786 
787   private void registerIfMismatch(
788       JSType found, JSType required, JSError error) {
789     if (found != null && required != null &&
790         !found.isSubtypeWithoutStructuralTyping(required)) {
791       registerMismatch(found, required, error);
792     }
793   }

795   /**
796    * Formats a found/required error message.
797    */
798   private static String formatFoundRequired(String description, JSType found,
799                                             JSType required) {
800     String foundStr = found.toString();
801     String requiredStr = required.toString();
802     if (foundStr.equals(requiredStr)) {
803       foundStr = found.toAnnotationString();
804       requiredStr = required.toAnnotationString();
805     }
806     return MessageFormat.format(FOUND_REQUIRED, description, foundStr, requiredStr);
807   }

813   private JSType getJSType(Node n) {
814     JSType jsType = n.getJSType();
815     if (jsType == null) {
816       // TODO(user): This branch indicates a compiler bug, not worthy of
817       // halting the compilation but we should log this and analyze to track
818       // down why it happens. This is not critical and will be resolved over
819       // time as the type checker is extended.
820       return getNativeType(UNKNOWN_TYPE);
821     } else {
822       return jsType;
823     }

835   /**
836    * Signals that the first type and the second type have been
837    * used interchangeably.
838    *
839    * Type-based optimizations should take this into account
840    * so that they don't wreck code with type warnings.
841    */
842   static class TypeMismatch {
843     final JSType typeA;
844     final JSType typeB;
845     final JSError src;
846 
847     /**
848      * It's the responsibility of the class that creates the
849      * {@code TypeMismatch} to ensure that {@code a} and {@code b} are
850      * non-matching types.
851      */
852     TypeMismatch(JSType a, JSType b, JSError src) {
853       this.typeA = a;
854       this.typeB = b;
855       this.src = src;
856     }
857 
858     @Override public boolean equals(Object object) {
859       if (object instanceof TypeMismatch) {
860         TypeMismatch that = (TypeMismatch) object;
861         return (that.typeA.isEquivalentTo(this.typeA)
862                 && that.typeB.isEquivalentTo(this.typeB))
863             || (that.typeB.isEquivalentTo(this.typeA)
864                 && that.typeA.isEquivalentTo(this.typeB));
865       }
866       return false;
867     }
868 
869     @Override public int hashCode() {
870       return Objects.hash(typeA, typeB);
871     }
872 
873     @Override public String toString() {
874       return "(" + typeA + ", " + typeB + ")";
875     }
876   }
