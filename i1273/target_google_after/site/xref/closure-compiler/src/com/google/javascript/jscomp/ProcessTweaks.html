34  /**
35   * Process goog.tweak primitives. Checks that:
36   * <ul>
37   * <li>parameters to goog.tweak.register* are literals of the correct type.
38   * <li>the parameter to goog.tweak.get* is a string literal.
39   * <li>parameters to goog.tweak.overrideDefaultValue are literals of the correct
40   *     type.
41   * <li>tweak IDs passed to goog.tweak.get* and goog.tweak.overrideDefaultValue
42   *     correspond to registered tweaks.
43   * <li>all calls to goog.tweak.register* and goog.tweak.overrideDefaultValue are
44   *     within the top-level context.
45   * <li>each tweak is registered only once.
46   * <li>calls to goog.tweak.overrideDefaultValue occur before the call to the
47   *     corresponding goog.tweak.register* function.
48   * </ul>
49   * @author agrieve@google.com (Andrew Grieve)
50   */
51  class ProcessTweaks implements CompilerPass {
52  
53    private final AbstractCompiler compiler;
54    private final boolean stripTweaks;
55    private final SortedMap<String, Node> compilerDefaultValueOverrides;
56  
57    private static final CharMatcher ID_MATCHER = CharMatcher.inRange('a', 'z').
58        or(CharMatcher.inRange('A', 'Z')).or(CharMatcher.anyOf("0123456789_."));

60    // Warnings and Errors.
61    static final DiagnosticType UNKNOWN_TWEAK_WARNING =
62        DiagnosticType.warning(
63            "JSC_UNKNOWN_TWEAK_WARNING",
64            "no tweak registered with ID {0}");
65  
66    static final DiagnosticType TWEAK_MULTIPLY_REGISTERED_ERROR =
67        DiagnosticType.error(
68            "JSC_TWEAK_MULTIPLY_REGISTERED_ERROR",
69            "Tweak {0} has already been registered.");
70  
71    static final DiagnosticType NON_LITERAL_TWEAK_ID_ERROR =
72        DiagnosticType.error(
73            "JSC_NON_LITERAL_TWEAK_ID_ERROR",
74            "tweak ID must be a string literal");
75  
76    static final DiagnosticType INVALID_TWEAK_DEFAULT_VALUE_WARNING =
77        DiagnosticType.warning(
78            "JSC_INVALID_TWEAK_DEFAULT_VALUE_WARNING",
79            "tweak {0} registered with {1} must have a default value that is a " +
80            "literal of type {2}");
81  
82    static final DiagnosticType NON_GLOBAL_TWEAK_INIT_ERROR =
83        DiagnosticType.error(
84            "JSC_NON_GLOBAL_TWEAK_INIT_ERROR",
85            "tweak declaration {0} must occur in the global scope");
86  
87    static final DiagnosticType TWEAK_OVERRIDE_AFTER_REGISTERED_ERROR =
88        DiagnosticType.error(
89            "JSC_TWEAK_OVERRIDE_AFTER_REGISTERED_ERROR",
90            "Cannot override the default value of tweak {0} after it has been " +
91            "registered");
92  
93    static final DiagnosticType TWEAK_WRONG_GETTER_TYPE_WARNING =
94        DiagnosticType.warning(
95            "JSC_TWEAK_WRONG_GETTER_TYPE_WARNING",
96            "tweak getter function {0} used for tweak registered using {1}");
97  
98    static final DiagnosticType INVALID_TWEAK_ID_ERROR =
99        DiagnosticType.error(
100           "JSC_INVALID_TWEAK_ID_ERROR",
101           "tweak ID contains illegal characters. Only letters, numbers, _ " +
102           "and . are allowed");

104   /**
105    * An enum of goog.tweak functions.
106    */
107   private static enum TweakFunction {
108     REGISTER_BOOLEAN("goog.tweak.registerBoolean", "boolean", Token.TRUE,
109         Token.FALSE),
110     REGISTER_NUMBER("goog.tweak.registerNumber", "number", Token.NUMBER),
111     REGISTER_STRING("goog.tweak.registerString", "string", Token.STRING),
112     OVERRIDE_DEFAULT_VALUE("goog.tweak.overrideDefaultValue"),
113     GET_COMPILER_OVERRIDES("goog.tweak.getCompilerOverrides_"),
114     GET_BOOLEAN("goog.tweak.getBoolean", REGISTER_BOOLEAN),
115     GET_NUMBER("goog.tweak.getNumber", REGISTER_NUMBER),
116     GET_STRING("goog.tweak.getString", REGISTER_STRING);
117 
118     final String name;
119     final String expectedTypeName;
120     final int validNodeTypeA;
121     final int validNodeTypeB;
122     final TweakFunction registerFunction;
123 
124     TweakFunction(String name) {
125       this(name, null, Token.ERROR, Token.ERROR, null);
126     }
127 
128     TweakFunction(String name, String expectedTypeName,
129         int validNodeTypeA) {
130       this(name, expectedTypeName, validNodeTypeA, Token.ERROR, null);
131     }
132 
133     TweakFunction(String name, String expectedTypeName,
134         int validNodeTypeA, int validNodeTypeB) {
135       this(name, expectedTypeName, validNodeTypeA, validNodeTypeB, null);
136     }
137 
138     TweakFunction(String name, TweakFunction registerFunction) {
139       this(name, null, Token.ERROR, Token.ERROR, registerFunction);
140     }
141 
142     TweakFunction(String name, String expectedTypeName,
143         int validNodeTypeA, int validNodeTypeB,
144         TweakFunction registerFunction) {
145       this.name = name;
146       this.expectedTypeName = expectedTypeName;
147       this.validNodeTypeA = validNodeTypeA;
148       this.validNodeTypeB = validNodeTypeB;
149       this.registerFunction = registerFunction;
150     }
151 
152     boolean isValidNodeType(int type) {
153       return type == validNodeTypeA || type == validNodeTypeB;
154     }
155 
156     boolean isCorrectRegisterFunction(TweakFunction registerFunction) {
157       Preconditions.checkNotNull(registerFunction);
158       return this.registerFunction == registerFunction;
159     }
160 
161     boolean isGetterFunction() {
162       return registerFunction != null;
163     }
164 
165     String getName() {
166       return name;
167     }
168 
169     String getExpectedTypeName() {
170       return expectedTypeName;
171     }
172 
173     Node createDefaultValueNode() {
174       switch (this) {
175         case REGISTER_BOOLEAN:
176           return IR.falseNode();
177         case REGISTER_NUMBER:
178           return IR.number(0);
179         case REGISTER_STRING:
180           return IR.string("");
181         default:
182           throw new IllegalStateException();
183       }
184     }
185   }

187   // A map of function name -> TweakFunction.
188   private static final Map<String, TweakFunction> TWEAK_FUNCTIONS_MAP;
189   static {
190     TWEAK_FUNCTIONS_MAP = new HashMap<>();
191     for (TweakFunction func : TweakFunction.values()) {
192       TWEAK_FUNCTIONS_MAP.put(func.getName(), func);
193     }
194   }

196   ProcessTweaks(AbstractCompiler compiler, boolean stripTweaks,
197       Map<String, Node> compilerDefaultValueOverrides) {
198     this.compiler = compiler;
199     this.stripTweaks = stripTweaks;
200     // Having the map sorted is required for the unit tests to be deterministic.
201     this.compilerDefaultValueOverrides = new TreeMap<>();
202     this.compilerDefaultValueOverrides.putAll(compilerDefaultValueOverrides);
203   }
204 
205   @Override
206   public void process(Node externs, Node root) {
207     CollectTweaksResult result = collectTweaks(root);
208     applyCompilerDefaultValueOverrides(result.tweakInfos);
209 
210     boolean changed = false;
211 
212     if (stripTweaks) {
213       changed = stripAllCalls(result.tweakInfos);
214     } else if (!compilerDefaultValueOverrides.isEmpty()) {
215       changed = replaceGetCompilerOverridesCalls(result.getOverridesCalls);
216     }
217     if (changed) {
218       compiler.reportCodeChange();
219     }
220   }

222   /**
223    * Passes the compiler default value overrides to the JS by replacing calls
224    * to goog.tweak.getCompilerOverrids_ with a map of tweak ID->default value;
225    */
226   private boolean replaceGetCompilerOverridesCalls(
227       List<TweakFunctionCall> calls) {
228     for (TweakFunctionCall call : calls) {
229       Node callNode = call.callNode;
230       Node objNode = createCompilerDefaultValueOverridesVarNode(callNode);
231       callNode.getParent().replaceChild(callNode, objNode);
232     }
233     return !calls.isEmpty();
234   }
235 
236   /**
237    * Removes all CALL nodes in the given TweakInfos, replacing calls to getter
238    * functions with the tweak's default value.
239    */
240   private boolean stripAllCalls(Map<String, TweakInfo> tweakInfos) {
241     for (TweakInfo tweakInfo : tweakInfos.values()) {
242       boolean isRegistered = tweakInfo.isRegistered();
243       for (TweakFunctionCall functionCall : tweakInfo.functionCalls) {
244         Node callNode = functionCall.callNode;
245         Node parent = callNode.getParent();
246         if (functionCall.tweakFunc.isGetterFunction()) {
247           Node newValue;
248           if (isRegistered) {
249             newValue = tweakInfo.getDefaultValueNode().cloneNode();
250           } else {
251             // When we find a getter of an unregistered tweak, there has
252             // already been a warning about it, so now just use a default
253             // value when stripping.
254             TweakFunction registerFunction =
255                 functionCall.tweakFunc.registerFunction;
256             newValue = registerFunction.createDefaultValueNode();
257           }
258           parent.replaceChild(callNode, newValue);
259         } else {
260           Node voidZeroNode = IR.voidNode(IR.number(0).srcref(callNode))
261               .srcref(callNode);
262           parent.replaceChild(callNode, voidZeroNode);
263         }
264       }
265     }
266     return !tweakInfos.isEmpty();
267   }

269   /**
270    * Creates a JS object that holds a map of tweakId -> default value override.
271    */
272   private Node createCompilerDefaultValueOverridesVarNode(
273       Node sourceInformationNode) {
274     Node objNode = IR.objectlit().srcref(sourceInformationNode);
275     for (Entry<String, Node> entry : compilerDefaultValueOverrides.entrySet()) {
276       Node objKeyNode = IR.stringKey(entry.getKey())
277           .useSourceInfoIfMissingFrom(sourceInformationNode);
278       Node objValueNode = entry.getValue().cloneNode()
279           .useSourceInfoIfMissingFrom(sourceInformationNode);
280       objKeyNode.addChildToBack(objValueNode);
281       objNode.addChildToBack(objKeyNode);
282     }
283     return objNode;
284   }

286   /** Sets the default values of tweaks based on compiler options. */
287   private void applyCompilerDefaultValueOverrides(
288       Map<String, TweakInfo> tweakInfos) {
289     for (Entry<String, Node> entry : compilerDefaultValueOverrides.entrySet()) {
290       String tweakId = entry.getKey();
291       TweakInfo tweakInfo = tweakInfos.get(tweakId);
292       if (tweakInfo == null) {
293         compiler.report(JSError.make(UNKNOWN_TWEAK_WARNING, tweakId));
294       } else {
295         TweakFunction registerFunc = tweakInfo.registerCall.tweakFunc;
296         Node value = entry.getValue();
297         if (!registerFunc.isValidNodeType(value.getType())) {
298           compiler.report(JSError.make(INVALID_TWEAK_DEFAULT_VALUE_WARNING,
299               tweakId, registerFunc.getName(),
300               registerFunc.getExpectedTypeName()));
301         } else {
302           tweakInfo.defaultValueNode = value;
303         }
304       }
305     }
306   }

308   /**
309    * Finds all calls to goog.tweak functions and emits warnings/errors if any
310    * of the calls have issues.
311    * @return A map of {@link TweakInfo} structures, keyed by tweak ID.
312    */
313   private CollectTweaksResult collectTweaks(Node root) {
314     CollectTweaks pass = new CollectTweaks();
315     NodeTraversal.traverseEs6(compiler, root, pass);
316 
317     Map<String, TweakInfo> tweakInfos = pass.allTweaks;
318     for (TweakInfo tweakInfo : tweakInfos.values()) {
319       tweakInfo.emitAllWarnings();
320     }
321     return new CollectTweaksResult(tweakInfos, pass.getOverridesCalls);
322   }

324   private static final class CollectTweaksResult {
325     final Map<String, TweakInfo> tweakInfos;
326     final List<TweakFunctionCall> getOverridesCalls;
327 
328     CollectTweaksResult(Map<String, TweakInfo> tweakInfos,
329         List<TweakFunctionCall> getOverridesCalls) {
330       this.tweakInfos = tweakInfos;
331       this.getOverridesCalls = getOverridesCalls;
332     }
333   }

335   /**
336    * Processes all calls to goog.tweak functions.
337    */
338   private final class CollectTweaks extends AbstractPostOrderCallback {
339     final Map<String, TweakInfo> allTweaks = new HashMap<>();
340     final List<TweakFunctionCall> getOverridesCalls = new ArrayList<>();
341 
342     @SuppressWarnings("incomplete-switch")
343     @Override
344     public void visit(NodeTraversal t, Node n, Node parent) {
345       if (!n.isCall()) {
346         return;
347       }
348 
349       String callName = n.getFirstChild().getQualifiedName();
350       TweakFunction tweakFunc = TWEAK_FUNCTIONS_MAP.get(callName);
351       if (tweakFunc == null) {
352         return;
353       }
354 
355       if (tweakFunc == TweakFunction.GET_COMPILER_OVERRIDES) {
356         getOverridesCalls.add(
357             new TweakFunctionCall(tweakFunc, n));
358         return;
359       }
360 
361       // Ensure the first parameter (the tweak ID) is a string literal.
362       Node tweakIdNode = n.getFirstChild().getNext();
363       if (!tweakIdNode.isString()) {
364         compiler.report(t.makeError(tweakIdNode, NON_LITERAL_TWEAK_ID_ERROR));
365         return;
366       }
367       String tweakId = tweakIdNode.getString();
368 
369       // Make sure there is a TweakInfo structure for it.
370       TweakInfo tweakInfo = allTweaks.get(tweakId);
371       if (tweakInfo == null) {
372         tweakInfo = new TweakInfo(tweakId);
373         allTweaks.put(tweakId, tweakInfo);
374       }
375 
376       switch (tweakFunc) {
377         case REGISTER_BOOLEAN:
378         case REGISTER_NUMBER:
379         case REGISTER_STRING:
380           // Ensure the ID contains only valid characters.
381           if (!ID_MATCHER.matchesAllOf(tweakId)) {
382             compiler.report(t.makeError(tweakIdNode, INVALID_TWEAK_ID_ERROR));
383           }
384 
385           // Ensure tweaks are registered in the global scope.
386           if (!t.inGlobalHoistScope()) {
387             compiler.report(
388                 t.makeError(n, NON_GLOBAL_TWEAK_INIT_ERROR, tweakId));
389             break;
390           }
391 
392           // Ensure tweaks are registered only once.
393           if (tweakInfo.isRegistered()) {
394             compiler.report(
395                 t.makeError(n, TWEAK_MULTIPLY_REGISTERED_ERROR, tweakId));
396             break;
397           }
398 
399           Node tweakDefaultValueNode = tweakIdNode.getNext().getNext();
400           tweakInfo.addRegisterCall(t.getSourceName(), tweakFunc, n,
401               tweakDefaultValueNode);
402           break;
403         case OVERRIDE_DEFAULT_VALUE:
404           // Ensure tweaks overrides occur in the global scope.
405           if (!t.inGlobalScope()) {
406             compiler.report(
407                 t.makeError(n, NON_GLOBAL_TWEAK_INIT_ERROR, tweakId));
408             break;
409           }
410           // Ensure tweak overrides occur before the tweak is registered.
411           if (tweakInfo.isRegistered()) {
412             compiler.report(
413                 t.makeError(n, TWEAK_OVERRIDE_AFTER_REGISTERED_ERROR, tweakId));
414             break;
415           }
416 
417           tweakDefaultValueNode = tweakIdNode.getNext();
418           tweakInfo.addOverrideDefaultValueCall(t.getSourceName(), tweakFunc, n,
419               tweakDefaultValueNode);
420           break;
421         case GET_BOOLEAN:
422         case GET_NUMBER:
423         case GET_STRING:
424           tweakInfo.addGetterCall(t.getSourceName(), tweakFunc, n);
425       }
426     }
427   }

429   /**
430    * Holds information about a call to a goog.tweak function.
431    */
432   private static final class TweakFunctionCall {
433     final TweakFunction tweakFunc;
434     final Node callNode;
435     final Node valueNode;
436 
437     TweakFunctionCall(TweakFunction tweakFunc, Node callNode) {
438       this(tweakFunc, callNode, null);
439     }
440 
441     TweakFunctionCall(TweakFunction tweakFunc, Node callNode, Node valueNode) {
442       this.callNode = callNode;
443       this.tweakFunc = tweakFunc;
444       this.valueNode = valueNode;
445     }
446 
447     Node getIdNode() {
448       return callNode.getFirstChild().getNext();
449     }
450   }

452   /**
453    * Stores information about a single tweak.
454    */
455   private final class TweakInfo {
456     final String tweakId;
457     final List<TweakFunctionCall> functionCalls;
458     TweakFunctionCall registerCall;
459     Node defaultValueNode;
460 
461     TweakInfo(String tweakId) {
462       this.tweakId = tweakId;
463       functionCalls = new ArrayList<>();
464     }
465 
466     /**
467      * If this tweak is registered, then looks for type warnings in default
468      * value parameters and getter functions. If it is not registered, emits an
469      * error for each function call.
470      */
471     void emitAllWarnings() {
472       if (isRegistered()) {
473         emitAllTypeWarnings();
474       } else {
475         emitUnknownTweakErrors();
476       }
477     }
478 
479     /**
480      * Emits a warning for each default value parameter that has the wrong type
481      * and for each getter function that was used for the wrong type of tweak.
482      */
483     void emitAllTypeWarnings() {
484       for (TweakFunctionCall call : functionCalls) {
485         Node valueNode = call.valueNode;
486         TweakFunction tweakFunc = call.tweakFunc;
487         TweakFunction registerFunc = registerCall.tweakFunc;
488         if (valueNode != null) {
489           // For register* and overrideDefaultValue calls, ensure the default
490           // value is a literal of the correct type.
491           if (!registerFunc.isValidNodeType(valueNode.getType())) {
492             compiler.report(JSError.make(
493                 valueNode, INVALID_TWEAK_DEFAULT_VALUE_WARNING,
494                 tweakId, registerFunc.getName(),
495                 registerFunc.getExpectedTypeName()));
496           }
497         } else if (tweakFunc.isGetterFunction()) {
498           // For getter calls, ensure the correct getter was used.
499           if (!tweakFunc.isCorrectRegisterFunction(registerFunc)) {
500             compiler.report(JSError.make(
501                 call.callNode, TWEAK_WRONG_GETTER_TYPE_WARNING,
502                 tweakFunc.getName(), registerFunc.getName()));
503           }
504         }
505       }
506     }
507 
508     /**
509      * Emits an error for each function call that was found.
510      */
511     void emitUnknownTweakErrors() {
512       for (TweakFunctionCall call : functionCalls) {
513         compiler.report(JSError.make(
514             call.getIdNode(), UNKNOWN_TWEAK_WARNING, tweakId));
515       }
516     }
517 
518     void addRegisterCall(String sourceName, TweakFunction tweakFunc,
519         Node callNode, Node defaultValueNode) {
520       registerCall = new TweakFunctionCall(tweakFunc, callNode,
521           defaultValueNode);
522       functionCalls.add(registerCall);
523     }
524 
525     void addOverrideDefaultValueCall(String sourceName,
526         TweakFunction tweakFunc, Node callNode, Node defaultValueNode) {
527       functionCalls.add(new TweakFunctionCall(tweakFunc, callNode,
528           defaultValueNode));
529       this.defaultValueNode = defaultValueNode;
530     }
531 
532     void addGetterCall(String sourceName, TweakFunction tweakFunc,
533         Node callNode) {
534       functionCalls.add(new TweakFunctionCall(tweakFunc, callNode));
535     }
536 
537     boolean isRegistered() {
538       return registerCall != null;
539     }
540 
541     Node getDefaultValueNode() {
542       Preconditions.checkState(isRegistered());
543       // Use calls to goog.tweak.overrideDefaultValue() first.
544       if (defaultValueNode != null) {
545         return defaultValueNode;
546       }
547       // Use the value passed to the register function next.
548       if (registerCall.valueNode != null) {
549         return registerCall.valueNode;
550       }
551       // Otherwise, use the default value for the tweak's type.
552       return registerCall.tweakFunc.createDefaultValueNode();
553     }
