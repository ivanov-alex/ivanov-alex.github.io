42  /**
43   * Object representing the fixes to apply to the source code to create the
44   * refactoring CL. To create a class, use the {@link Builder} class and helper
45   * functions.
46   *
47   * @author mknichel@google.com (Mark Knichel)
48   */
49  public final class SuggestedFix {
50  
51    private final Node originalMatchedNode;
52    // Multimap of filename to a modification to that file.
53    private final SetMultimap<String, CodeReplacement> replacements;
54  
55    // An optional description of the fix, to distinguish between the various possible fixes
56    // for errors that have multiple fixes.
57    @Nullable private final String description;
58  
59    private SuggestedFix(
60        Node originalMatchedNode,
61        SetMultimap<String, CodeReplacement> replacements,
62        @Nullable String description) {
63      this.originalMatchedNode = originalMatchedNode;
64      this.replacements = replacements;
65      this.description = description;
66    }
67  
68    /**
69     * Returns the JS Compiler Node for the original node that caused this SuggestedFix to
70     * be constructed.
71     */
72    public Node getOriginalMatchedNode() {
73      return originalMatchedNode;
74    }

76    /**
77     * Returns a multimap from filename to all the replacements that should be
78     * applied for this given fix.
79     */
80    public SetMultimap<String, CodeReplacement> getReplacements() {
81      return replacements;
82    }
83  
84    @Nullable public String getDescription() {
85      return description;
86    }
87  
88    @Override public String toString() {
89      StringBuilder sb = new StringBuilder();
90      for (Map.Entry<String, Collection<CodeReplacement>> entry : replacements.asMap().entrySet()) {
91        sb.append("Replacements for file: " + entry.getKey() + "\n");
92        Joiner.on("\n").appendTo(sb, entry.getValue());
93      }
94      return sb.toString();
95    }

136     /**
137      * Inserts a string before the provided node. This is useful for inserting
138      * comments into a file since the JS Compiler doesn't currently support
139      * printing comments.
140      */
141     public Builder insertBefore(Node nodeToInsertBefore, String content) {
142       int startPosition = nodeToInsertBefore.getSourceOffset();
143       // TODO(mknichel): This case is not covered by NodeUtil.getBestJSDocInfo
144       JSDocInfo jsDoc = nodeToInsertBefore.isExprResult()
145           ? nodeToInsertBefore.getFirstChild().getJSDocInfo()
146           : nodeToInsertBefore.getJSDocInfo();
147       if (jsDoc != null) {
148         startPosition = jsDoc.getOriginalCommentPosition();
149       }
150       Preconditions.checkNotNull(nodeToInsertBefore.getSourceFileName(),
151           "No source file name for node: %s", nodeToInsertBefore);
152       replacements.put(
153           nodeToInsertBefore.getSourceFileName(),
154           new CodeReplacement(startPosition, 0, content));
155       return this;
156     }

173     /**
174      * Deletes a node and its contents from the source file.
175      */
176     private Builder delete(Node n, boolean deleteSurroundingWhitespace) {
177       int startPosition = n.getSourceOffset();
178       int length = n.getLength();
179       // TODO(mknichel): This case is not covered by NodeUtil.getBestJSDocInfo
180       JSDocInfo jsDoc = n.isExprResult() ? n.getFirstChild().getJSDocInfo() : n.getJSDocInfo();
181       if (jsDoc != null) {
182         length = n.getLength() + (startPosition - jsDoc.getOriginalCommentPosition());
183         startPosition = jsDoc.getOriginalCommentPosition();
184       }
185       // Variable declarations require special handling since the NAME node doesn't contain enough
186       // information if the variable is declared in a multi-variable declaration. The NAME node
187       // in a VAR declaration doesn't include its child in its length if there is an inline
188       // assignment, and the code needs to know how to delete the commas. See SuggestedFixTest for
189       // more information.
190       // TODO(mknichel): Move this logic and the start position logic to a helper function
191       // so that it can be reused in other methods.
192       if (n.isName() && n.getParent().isVar()) {
193         if (n.getNext() != null) {
194           length = n.getNext().getSourceOffset() - startPosition;
195         } else if (n.hasChildren()) {
196           Node child = n.getFirstChild();
197           length = (child.getSourceOffset() + child.getLength()) - startPosition;
198         }
199         if (n.getParent().getLastChild() == n && n != n.getParent().getFirstChild()) {
200           Node previousSibling = n.getParent().getChildBefore(n);
201           if (previousSibling.hasChildren()) {
202             Node child = previousSibling.getFirstChild();
203             int startPositionDiff = startPosition - (child.getSourceOffset() + child.getLength());
204             startPosition -= startPositionDiff;
205             length += startPositionDiff;
206           } else {
207             int startPositionDiff = startPosition - (
208                 previousSibling.getSourceOffset() + previousSibling.getLength());
209             startPosition -= startPositionDiff;
210             length += startPositionDiff;
211           }
212         }
213       }
214       Node parent = n.getParent();
215       if (deleteSurroundingWhitespace
216           && parent != null
217           && (parent.isScript() || parent.isBlock())) {
218         Node previousSibling = parent.getChildBefore(n);
219         if (previousSibling != null) {
220           int previousSiblingEndPosition =
221               previousSibling.getSourceOffset() + previousSibling.getLength();
222           length += (startPosition - previousSiblingEndPosition);
223           startPosition = previousSiblingEndPosition;
224         }
225       }
226       replacements.put(n.getSourceFileName(), new CodeReplacement(startPosition, length, ""));
227       return this;
228     }

239     /**
240      * Renames a given node to the provided name.
241      * @param n The node to rename.
242      * @param name The new name for the node.
243      * @param replaceEntireName True to replace the entire name of the node. The
244      *     default is to replace just the last property in the node with the new
245      *     name. For instance, if {@code replaceEntireName} is false, then
246      *     {@code this.foo()} will be renamed to {@code this.bar()}. However, if
247      *     it is true, it will be renamed to {@code bar()}.
248      */
249     public Builder rename(Node n, String name, boolean replaceEntireName) {
250       Node nodeToRename = null;
251       if (n.isCall() || n.isTaggedTemplateLit()) {
252         Node child = n.getFirstChild();
253         nodeToRename = child;
254         if (!replaceEntireName && child.isGetProp()) {
255           nodeToRename = child.getLastChild();
256         }
257       } else if (n.isGetProp()) {
258         nodeToRename = n.getLastChild();
259         if (replaceEntireName) {
260           // Trace up from the property access to the root.
261           while (nodeToRename.getParent().isGetProp()) {
262             nodeToRename = nodeToRename.getParent();
263           }
264         }
265       } else if (n.isStringKey()) {
266         nodeToRename = n;
267       } else if (n.isString()) {
268         Preconditions.checkState(n.getParent().isGetProp());
269         nodeToRename = n;
270       } else {
271         // TODO(mknichel): Implement the rest of this function.
272         throw new UnsupportedOperationException(
273             "Rename is not implemented for this node type: " + n);
274       }
275       replacements.put(
276           nodeToRename.getSourceFileName(),
277           new CodeReplacement(nodeToRename.getSourceOffset(), nodeToRename.getLength(), name));
278       return this;
279     }

281     /**
282      * Replaces the provided node with new node in the source file.
283      */
284     public Builder replace(Node original, Node newNode, AbstractCompiler compiler) {
285       Node parent = original.getParent();
286       // EXPR_RESULT nodes will contain the trailing semicolons, but the child node
287       // will not. Replace the EXPR_RESULT node to ensure that the semicolons are
288       // correct in the final output.
289       if (original.getParent().isExprResult()) {
290         original = original.getParent();
291       }
292       // TODO(mknichel): Move this logic to CodePrinter.
293       String newCode = generateCode(compiler, newNode);
294       // The generated code may contain a trailing newline but that is never wanted.
295       if (newCode.endsWith("\n")) {
296         newCode = newCode.substring(0, newCode.length() - 1);
297       }
298       // Most replacements don't need the semicolon in the new generated code - however, some
299       // statements that are blocks or expressions will need the semicolon.
300       boolean needsSemicolon = parent.isExprResult() || parent.isBlock() || parent.isScript();
301       if (newCode.endsWith(";") && !needsSemicolon) {
302         newCode = newCode.substring(0, newCode.length() - 1);
303       }
304       replacements.put(
305           original.getSourceFileName(),
306           new CodeReplacement(original.getSourceOffset(), original.getLength(), newCode));
307       return this;
308     }

324     /**
325      * Removes a cast from the given node.
326      */
327     public Builder removeCast(Node n, AbstractCompiler compiler) {
328       Preconditions.checkArgument(n.isCast());
329       JSDocInfo jsDoc = n.getJSDocInfo();
330       replacements.put(
331           n.getSourceFileName(),
332           new CodeReplacement(
333               jsDoc.getOriginalCommentPosition(),
334               n.getFirstChild().getSourceOffset() - jsDoc.getOriginalCommentPosition(),
335               ""));
336       replacements.put(
337           n.getSourceFileName(),
338           new CodeReplacement(
339               n.getSourceOffset() + n.getLength() - 1,
340               1 /* length */,
341               ""));
342       return this;

360     /**
361      * Changes the JS Doc Type of the given node.
362      */
363     public Builder changeJsDocType(Node n, AbstractCompiler compiler, String type) {
364       JSDocInfo info = NodeUtil.getBestJSDocInfo(n);
365       Preconditions.checkNotNull(info, "Node %s does not have JS Doc associated with it.", n);
366       Node typeNode = JsDocInfoParser.parseTypeString(type);
367       Preconditions.checkNotNull(typeNode, "Invalid type: %s", type);
368       JSTypeExpression typeExpr = new JSTypeExpression(typeNode, "jsflume");
369       JSType newJsType = typeExpr.evaluate(null, compiler.getTypeRegistry());
370       if (newJsType == null) {
371         throw new RuntimeException("JS Compiler does not recognize type: " + type);
372       }
373 
374       String originalComment = info.getOriginalCommentString();
375       int originalPosition = info.getOriginalCommentPosition();
376 
377       // TODO(mknichel): Support multiline @type annotations.
378       Pattern typeDocPattern = Pattern.compile(
379           "@(type|private|protected|public|const|return) *\\{?[^\\s}]+\\}?");
380       Matcher m = typeDocPattern.matcher(originalComment);
381       while (m.find()) {
382         replacements.put(
383             n.getSourceFileName(),
384             new CodeReplacement(
385                 originalPosition + m.start(),
386                 m.end() - m.start(),
387                 "@" + m.group(1) + " {" + type + "}"));
388       }
389 
390       return this;
391     }

393     /**
394      * Inserts arguments into an existing function call.
395      */
396     public Builder insertArguments(Node n, int position, String... args) {
397       Preconditions.checkArgument(
398           n.isCall(), "insertArguments is only applicable to function call nodes.");
399       int startPosition;
400       Node argument = n.getFirstChild().getNext();
401       int i = 0;
402       while (argument != null && i < position) {
403         argument = argument.getNext();
404         i++;
405       }
406       if (argument == null) {
407         Preconditions.checkArgument(
408             position == i, "The specified position must be less than the number of arguments.");
409         startPosition = n.getSourceOffset() + n.getLength() - 1;
410       } else {
411         JSDocInfo jsDoc = argument.getJSDocInfo();
412         if (jsDoc != null) {
413           // Remove any cast or associated JS Doc if it exists.
414           startPosition = jsDoc.getOriginalCommentPosition();
415         } else {
416           startPosition = argument.getSourceOffset();
417         }
418       }
419 
420       String newContent = Joiner.on(", ").join(args);
421       if (argument != null) {
422         newContent += ", ";
423       } else if (i > 0) {
424         newContent = ", " + newContent;
425       }
426       replacements.put(n.getSourceFileName(), new CodeReplacement(startPosition, 0, newContent));
427 
428       return this;
429     }

431     /**
432      * Deletes an argument from an existing function call, including any JS doc that precedes it.
433      * WARNING: If jsdoc erroneously follows the argument, it will not be removed as the parser
434      *     considers the comment to belong to the next argument.
435      */
436     public Builder deleteArgument(Node n, int position) {
437       Preconditions.checkArgument(
438           n.isCall(), "deleteArgument is only applicable to function call nodes.");
439 
440       // A CALL node's first child is the name of the function being called, and subsequent children
441       // are the arguments being passed to that function.
442       int numArguments = n.getChildCount() - 1;
443       Preconditions.checkState(numArguments > 0,
444           "deleteArgument() cannot be used on a function call with no arguments");
445       Preconditions.checkArgument(position >= 0 && position < numArguments,
446           "The specified position must be less than the number of arguments.");
447       Node argument = n.getFirstChild().getNext();
448 
449       // Points at the first position in the code we will remove.
450       int startOfArgumentToRemove = -1;
451       // Points one past the last position in the code we will remove.
452       int endOfArgumentToRemove = -1;
453       int i = 0;
454       while (argument != null) {
455         // If we are removing the first argument, we remove from the start of it (including any
456         // jsdoc).  Otherwise, we remove from the end of the previous argument (to remove the comma
457         // and any whitespace).
458 
459         // If we are removing the first argument and it's not the only argument, we remove to the
460         // beginning of the next argument (to remove the comma and any whitespace).  Otherwise we
461         // remove to the end of the argument.
462         if (i < position) {
463           startOfArgumentToRemove = argument.getSourceOffset() + argument.getLength();
464         } else if (i == position) {
465           if (position == 0) {
466             startOfArgumentToRemove = argument.getSourceOffset();
467 
468             // If we have a prefix jsdoc, back up further and remove that too.
469             JSDocInfo jsDoc = argument.getJSDocInfo();
470             if (jsDoc != null) {
471               int jsDocPosition = jsDoc.getOriginalCommentPosition();
472               if (jsDocPosition < startOfArgumentToRemove) {
473                 startOfArgumentToRemove = jsDocPosition;
474               }
475             }
476           }
477 
478           endOfArgumentToRemove = argument.getSourceOffset() + argument.getLength();
479         } else if (i > position) {
480           if (position == 0) {
481             endOfArgumentToRemove = argument.getSourceOffset();
482           }
483           // We have all the information we need to remove the argument, break early.
484           break;
485         }
486 
487         argument = argument.getNext();
488         i++;
489       }
490 
491       // Remove the argument by replacing it with an empty string.
492       int lengthOfArgumentToRemove = endOfArgumentToRemove - startOfArgumentToRemove;
493       replacements.put(n.getSourceFileName(),
494           new CodeReplacement(startOfArgumentToRemove, lengthOfArgumentToRemove, ""));
495       return this;
496     }

498     /**
499      * Adds a goog.require for the given namespace to the file if it does not
500      * already exist.
501      */
502     public Builder addGoogRequire(Match m, String namespace) {
503       Node node = m.getNode();
504       NodeMetadata metadata = m.getMetadata();
505       Node existingNode = findGoogRequireNode(m.getNode(), metadata, namespace);
506       if (existingNode != null) {
507         return this;
508       }
509       Node googRequireNode = IR.exprResult(IR.call(
510           IR.getprop(IR.name("goog"), IR.string("require")),
511           IR.string(namespace)));
512 
513       // Find the right goog.require node to insert this after.
514       Node script = node.getParent();
515       while (script != null && !script.isScript()) {
516         script = script.getParent();
517       }
518       if (script == null) {
519         return this;
520       }
521       Node lastGoogProvideNode = null;
522       Node lastGoogRequireNode = null;
523       Node nodeToInsertBefore = null;
524       Node child = script.getFirstChild();
525       while (child != null) {
526         if (child.isExprResult() && child.getFirstChild().isCall()) {
527           // TODO(mknichel): Replace this logic with a function argument
528           // Matcher when it exists.
529           Node grandchild = child.getFirstChild();
530           if (Matchers.functionCall("goog.provide").matches(grandchild, metadata)) {
531             lastGoogProvideNode = grandchild;
532           } else if (Matchers.functionCall("goog.require").matches(grandchild, metadata)) {
533             lastGoogRequireNode = grandchild;
534             if (grandchild.getLastChild().isString()
535                 && namespace.compareTo(grandchild.getLastChild().getString()) < 0) {
536               nodeToInsertBefore = child;
537               break;
538             }
539           }
540         }
541         child = child.getNext();
542       }
543       if (nodeToInsertBefore == null) {
544         // The file has goog.provide or goog.require nodes but they come before
545         // the new goog.require node alphabetically.
546         if (lastGoogProvideNode != null || lastGoogRequireNode != null) {
547           Node nodeToInsertAfter =
548               lastGoogRequireNode != null ? lastGoogRequireNode : lastGoogProvideNode;
549           int startPosition =
550               nodeToInsertAfter.getSourceOffset() + nodeToInsertAfter.getLength() + 2;
551           replacements.put(nodeToInsertAfter.getSourceFileName(), new CodeReplacement(
552               startPosition,
553               0,
554               generateCode(m.getMetadata().getCompiler(), googRequireNode)));
555           return this;
556         } else {
557           // The file has no goog.provide or goog.require nodes.
558           if (script.getFirstChild() != null) {
559             nodeToInsertBefore = script.getFirstChild();
560           } else {
561             replacements.put(script.getSourceFileName(), new CodeReplacement(
562                 0, 0, generateCode(m.getMetadata().getCompiler(), googRequireNode)));
563             return this;
564           }
565         }
566       }
567 
568       return insertBefore(nodeToInsertBefore, googRequireNode, m.getMetadata().getCompiler());
569     }

571     /**
572      * Removes a goog.require for the given namespace to the file if it
573      * already exists.
574      */
575     public Builder removeGoogRequire(Match m, String namespace) {
576       Node googRequireNode = findGoogRequireNode(m.getNode(), m.getMetadata(), namespace);
577       if (googRequireNode != null) {
578         return deleteWithoutRemovingSurroundWhitespace(googRequireNode);
579       }
580       return this;
581     }

583     private Node findGoogRequireNode(Node n, NodeMetadata metadata, String namespace) {
584       Node script = n.getParent();
585       while (script != null && !script.isScript()) {
586         script = script.getParent();
587       }
588 
589       if (script != null) {
590         Node child = script.getFirstChild();
591         while (child != null) {
592           if (child.isExprResult() && child.getFirstChild().isCall()) {
593             // TODO(mknichel): Replace this logic with a function argument
594             // Matcher when it exists.
595             Node grandchild = child.getFirstChild();
596             if (Matchers.functionCall("goog.require").matches(child.getFirstChild(), metadata)
597                 && grandchild.getLastChild().isString()
598                 && namespace.equals(grandchild.getLastChild().getString())) {
599               return child;
600             }
601           }
602           child = child.getNext();
603         }
604       }
605       return null;
606     }

608     public String generateCode(AbstractCompiler compiler, Node node) {
609       // TODO(mknichel): Fix all the formatting problems with this code.
610       // How does this play with goog.scope?
611       CompilerOptions compilerOptions = new CompilerOptions();
612       compilerOptions.setPreferSingleQuotes(true);
613       compilerOptions.setLineLengthThreshold(80);
614       return new CodePrinter.Builder(node)
615           .setCompilerOptions(compilerOptions)
616           .setTypeRegistry(compiler.getTypeRegistry())
617           .setPrettyPrint(true)
618           .setLineBreak(true)
619           .setOutputTypes(true)
620           .build();
621     }
