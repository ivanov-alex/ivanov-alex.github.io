42    static final DiagnosticType UNUSED =
43        DiagnosticType.warning("JSC_UNUSED", "{0}");
44  
45    public static final Set<String> wildcardExcludedGroups =
46        ImmutableSet.of("reportUnknownTypes");
47  
48    public DiagnosticGroups() {}
49  
50    private static final Map<String, DiagnosticGroup> groupsByName =
51         new HashMap<>();
52  
53    static DiagnosticGroup registerDeprecatedGroup(String name) {
54      return registerGroup(name, new DiagnosticGroup(name, UNUSED));
55    }
56  
57    static DiagnosticGroup registerGroup(String name,
58        DiagnosticGroup group) {
59      groupsByName.put(name, group);
60      return group;
61    }
62  
63    static DiagnosticGroup registerGroup(String name,
64        DiagnosticType ... types) {
65      DiagnosticGroup group = new DiagnosticGroup(name, types);
66      groupsByName.put(name, group);
67      return group;
68    }
69  
70    static DiagnosticGroup registerGroup(String name,
71        DiagnosticGroup ... groups) {
72      DiagnosticGroup group = new DiagnosticGroup(name, groups);
73      groupsByName.put(name, group);
74      return group;
75    }
76  
77    /** Get the registered diagnostic groups, indexed by name. */
78    public Map<String, DiagnosticGroup> getRegisteredGroups() {
79      return ImmutableMap.copyOf(groupsByName);
80    }
81  
82    /** Find the diagnostic group registered under the given name. */
83    public DiagnosticGroup forName(String name) {
84      return groupsByName.get(name);
85    }

497   /**
498    * Adds warning levels by name.
499    */
500   void setWarningLevel(CompilerOptions options,
501       String name, CheckLevel level) {
502     DiagnosticGroup group = forName(name);
503     Preconditions.checkNotNull(group, "No warning class for name: %s", name);
504     options.setWarningLevel(group, level);
505   }
