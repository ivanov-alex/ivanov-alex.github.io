45  /**
46   * Generates deps.js files by scanning JavaScript files for
47   * calls to goog.provide(), goog.require() and goog.addDependency().
48   *
49   * @author agrieve@google.com (Andrew Grieve)

53    public static enum InclusionStrategy {
54      ALWAYS,
55      WHEN_IN_SRCS,
56      DO_NOT_DUPLICATE

69    static final DiagnosticType SAME_FILE_WARNING = DiagnosticType.warning(
70        "DEPS_SAME_FILE",
71        "Namespace \"{0}\" is both required and provided in the same file.");
72  
73    static final DiagnosticType NEVER_PROVIDED_ERROR = DiagnosticType.error(
74        "DEPS_NEVER_PROVIDED",
75        "Namespace \"{0}\" is required but never provided.");
76  
77    static final DiagnosticType DUPE_PROVIDES_WARNING = DiagnosticType.warning(
78        "DEPS_DUPE_PROVIDES",
79        "Multiple calls to goog.provide(\"{0}\")");
80  
81    static final DiagnosticType MULTIPLE_PROVIDES_ERROR = DiagnosticType.error(
82        "DEPS_DUPE_PROVIDES",
83        "Namespace \"{0}\" is already provided in other file {1}");
84  
85    static final DiagnosticType DUPE_REQUIRE_WARNING = DiagnosticType.warning(
86        "DEPS_DUPE_REQUIRES",
87        "Namespace \"{0}\" is required multiple times");
88  
89    static final DiagnosticType NO_DEPS_WARNING = DiagnosticType.warning(
90        "DEPS_NO_DEPS",
91        "No dependencies found in file");

96    public DepsGenerator(
97        Collection<SourceFile> deps,
98        Collection<SourceFile> srcs,
99        InclusionStrategy mergeStrategy,
100       String closurePathAbs,
101       ErrorManager errorManager) {
102     this.deps = deps;
103     this.srcs = srcs;
104     this.mergeStrategy = mergeStrategy;
105     this.closurePathAbs = closurePathAbs;
106     this.errorManager = errorManager;
107   }

109   /**
110    * Performs the parsing inputs and writing of outputs.
111    * @throws IOException Occurs upon an IO error.
112    * @return Returns a String of goog.addDependency calls that will build
113    *     the dependency graph. Returns null if there was an error.
114    */
115   public String computeDependencyCalls() throws IOException {
116     // Build a map of closure-relative path -> DepInfo.
117     Map<String, DependencyInfo> depsFiles = parseDepsFiles();
118     logger.fine("preparsedFiles: " + depsFiles);
119 
120     // Find all goog.provides & goog.requires in src files
121     Map<String, DependencyInfo> jsFiles = parseSources(depsFiles.keySet());
122 
123     // Check if there were any parse errors.
124     if (errorManager.getErrorCount() > 0) {
125       return null;
126     }
127 
128     cleanUpDuplicatedFiles(depsFiles, jsFiles);
129 
130     // Check for missing provides or other semantic inconsistencies.
131     validateDependencies(depsFiles.values(), jsFiles.values());
132 
133     if (errorManager.getErrorCount() > 0) {
134       return null;
135     }
136 
137     ByteArrayOutputStream output = new ByteArrayOutputStream();
138     writeDepsContent(depsFiles, jsFiles, new PrintStream(output));
139     return new String(output.toByteArray(), UTF_8);
140   }

142   /**
143    * Removes duplicated depsInfo from jsFiles if this info already present in
144    * some of the parsed deps.js
145    *
146    * @param depsFiles DepsInfo from deps.js dependencies
147    * @param jsFiles DepsInfo from some of jsSources
148    */
149   protected void cleanUpDuplicatedFiles(Map<String, DependencyInfo> depsFiles,
150       Map<String, DependencyInfo> jsFiles) {
151     Set<String> depsPathsCopy = new HashSet<>(depsFiles.keySet());
152     for (String path : depsPathsCopy) {
153       if (mergeStrategy != InclusionStrategy.WHEN_IN_SRCS) {
154         jsFiles.remove(path);
155       }
156     }
157 
158     for (String path : jsFiles.keySet()) {
159       // If a generated file appears in both the jsFiles and in depsFiles, then
160       // remove it from depsFiles in order to get the full path the generated
161       // file.
162       depsFiles.remove(path);
163     }
164   }

166   /**
167    * Reports if there are any dependency problems with the given dependency
168    * information. Reported problems include:
169    *     - A namespace being provided more than once
170    *     - A namespace being required multiple times from within one file
171    *     - A namespace being provided and required in the same file
172    *     - A namespace being required that is never provided
173    * @param preparsedFileDepedencies Dependency information from existing
174    *     deps.js files.
175    * @param parsedFileDependencies Dependency information from parsed .js files.
176    */
177   private void validateDependencies(Iterable<DependencyInfo> preparsedFileDepedencies,
178       Iterable<DependencyInfo> parsedFileDependencies) {
179     // Create a map of namespace -> file providing it.
180     // Also report any duplicate provides.
181     Map<String, DependencyInfo> providesMap = new HashMap<>();
182     addToProvideMap(preparsedFileDepedencies, providesMap);
183     addToProvideMap(parsedFileDependencies, providesMap);
184     // For each require in the parsed sources:
185     for (DependencyInfo depInfo : parsedFileDependencies) {
186       List<String> requires = new ArrayList<>(depInfo.getRequires());
187       for (int i = 0, l = requires.size(); i < l; ++i) {
188         String namespace = requires.get(i);
189         // Check for multiple requires.
190         if (requires.subList(i + 1, l).contains(namespace)) {
191           reportDuplicateRequire(namespace, depInfo);
192         }
193         // Check for missing provides.
194         DependencyInfo provider = providesMap.get(namespace);
195         if (provider == null) {
196           reportUndefinedNamespace(namespace, depInfo);
197         } else if (provider == depInfo) {
198           reportSameFile(namespace, depInfo);
199         }
200       }
201     }
202   }
203 
204   private void reportSameFile(String namespace, DependencyInfo depInfo) {
205     errorManager.report(CheckLevel.WARNING,
206         JSError.make(depInfo.getName(), -1, -1,
207             SAME_FILE_WARNING, namespace));
208   }
209 
210   private void reportUndefinedNamespace(
211       String namespace, DependencyInfo depInfo) {
212     errorManager.report(CheckLevel.ERROR,
213         JSError.make(depInfo.getName(), -1, -1,
214             NEVER_PROVIDED_ERROR, namespace));
215   }
216 
217   private void reportDuplicateProvide(String namespace, DependencyInfo firstDep,
218       DependencyInfo secondDep) {
219     if (firstDep == secondDep) {
220       errorManager.report(CheckLevel.WARNING,
221           JSError.make(firstDep.getName(), -1, -1,
222               DUPE_PROVIDES_WARNING, namespace));
223     } else {
224       errorManager.report(CheckLevel.ERROR,
225           JSError.make(secondDep.getName(), -1, -1,
226               MULTIPLE_PROVIDES_ERROR, namespace, firstDep.getName()));
227     }
228   }

230   private void reportDuplicateRequire(
231       String namespace, DependencyInfo depInfo) {
232     errorManager.report(CheckLevel.WARNING,
233         JSError.make(depInfo.getName(), -1, -1,
234             DUPE_REQUIRE_WARNING, namespace));
235   }

237   private void reportNoDepsInDepsFile(String filePath) {
238     errorManager.report(CheckLevel.WARNING,
239         JSError.make(filePath, -1, -1, NO_DEPS_WARNING));
240   }

242   /**
243    * Adds the given DependencyInfos to the given providesMap. Also checks for
244    * and reports duplicate provides.
245    */
246   private void addToProvideMap(Iterable<DependencyInfo> depInfos,
247       Map<String, DependencyInfo> providesMap) {
248     for (DependencyInfo depInfo : depInfos) {
249       for (String provide : depInfo.getProvides()) {
250         DependencyInfo prevValue = providesMap.put(provide, depInfo);
251         // Check for duplicate provides.
252         if (prevValue != null) {
253           reportDuplicateProvide(provide, prevValue, depInfo);
254         }
255       }
256     }
257   }

259   protected DepsFileParser createDepsFileParser() {
260     DepsFileParser depsParser = new DepsFileParser(errorManager);
261     depsParser.setShortcutMode(true);
262     return depsParser;

265   /**
266    * Returns whether we should ignore dependency info in the given deps file.
267    */
268   protected boolean shouldSkipDepsFile(SourceFile file) {
269     return false;
270   }

272   /**
273    * Parses all deps.js files in the deps list and creates a map of
274    * closure-relative path -> DependencyInfo.
275    */
276   private Map<String, DependencyInfo> parseDepsFiles() throws IOException {
277     DepsFileParser depsParser = createDepsFileParser();
278     Map<String, DependencyInfo> depsFiles = new HashMap<>();
279     for (SourceFile file : deps) {
280       if (!shouldSkipDepsFile(file)) {
281         List<DependencyInfo>
282             depInfos = depsParser.parseFileReader(
283                 file.getName(), file.getCodeReader());
284         if (depInfos.isEmpty()) {
285           reportNoDepsInDepsFile(file.getName());
286         } else {
287           for (DependencyInfo info : depInfos) {
288             depsFiles.put(info.getPathRelativeToClosureBase(), info);
289           }
290         }
291       }
292     }
293 
294     // If a deps file also appears in srcs, our build tools will move it
295     // into srcs.  So we need to scan all the src files for addDependency
296     // calls as well.
297     for (SourceFile src : srcs) {
298       if ((new File(src.getName())).exists() &&
299           !shouldSkipDepsFile(src)) {
300         List<DependencyInfo> srcInfos =
301             depsParser.parseFileReader(src.getName(), src.getCodeReader());
302         for (DependencyInfo info : srcInfos) {
303           depsFiles.put(info.getPathRelativeToClosureBase(), info);
304         }
305       }
306     }
307 
308     return depsFiles;
309   }

311   /**
312    * Parses all source files for dependency information.
313    * @param preparsedFiles A set of closure-relative paths.
314    *     Files in this set are not parsed if they are encountered in srcs.
315    * @return Returns a map of closure-relative paths -> DependencyInfo for the
316    *     newly parsed files.
317    * @throws IOException Occurs upon an IO error.
318    */
319   private Map<String, DependencyInfo> parseSources(
320       Set<String> preparsedFiles) throws IOException {
321     Map<String, DependencyInfo> parsedFiles = new HashMap<>();
322     JsFileParser jsParser = new JsFileParser(errorManager);
323 
324     for (SourceFile file : srcs) {
325       String closureRelativePath =
326           PathUtil.makeRelative(
327               closurePathAbs, PathUtil.makeAbsolute(file.getName()));
328       logger.fine("Closure-relative path: " + closureRelativePath);
329 
330       if (InclusionStrategy.WHEN_IN_SRCS == mergeStrategy ||
331           !preparsedFiles.contains(closureRelativePath)) {
332         DependencyInfo depInfo =
333             jsParser.parseFile(
334                 file.getName(), closureRelativePath,
335                 file.getCode());
336 
337         // Kick the source out of memory.
338         file.clearCachedSource();
339         parsedFiles.put(closureRelativePath, depInfo);
340       }
341     }
342 
343     return parsedFiles;
344   }

346   /**
347    * Creates the content to put into the output deps.js file. If mergeDeps is
348    * true, then all of the dependency information in the providedDeps will be
349    * included in the output.
350    * @throws IOException Occurs upon an IO error.
351    */
352   private void writeDepsContent(Map<String, DependencyInfo> depsFiles,
353       Map<String, DependencyInfo> jsFiles, PrintStream out)
354       throws IOException {
355     // Print all dependencies extracted from srcs.
356     writeDepInfos(out, jsFiles.values());
357 
358     // Print all dependencies extracted from deps.
359     if (mergeStrategy == InclusionStrategy.ALWAYS) {
360       // This multimap is just for splitting DepsInfo objects by
361       // it's definition deps.js file
362       Multimap<String, DependencyInfo> infosIndex = Multimaps.index(
363           depsFiles.values(),
364           new Function<DependencyInfo, String>() {
365             @Override
366             public String apply(DependencyInfo from) {
367               return from.getName();
368             }
369           });
370 
371       for (String depsPath : infosIndex.keySet()) {
372         String path = formatPathToDepsFile(depsPath);
373         out.println("\n// Included from: " + path);
374         writeDepInfos(out, infosIndex.get(depsPath));
375       }
376     }
377   }

386   /**
387    * Writes goog.addDependency() lines for each DependencyInfo in depInfos.
388    */
389   private void writeDepInfos(PrintStream out, Collection<DependencyInfo> depInfos) {
390     // Print dependencies.
391     // Lines look like this:
392     // goog.addDependency('../../path/to/file.js', ['goog.Delay'],
393     //     ['goog.Disposable', 'goog.Timer']);
394     for (DependencyInfo depInfo : depInfos) {
395       Collection<String> provides = depInfo.getProvides();
396       Collection<String> requires = depInfo.getRequires();
397 
398       out.print("goog.addDependency('" +
399           depInfo.getPathRelativeToClosureBase() + "', ");
400       writeJsArray(out, provides);
401       out.print(", ");
402       writeJsArray(out, requires);
403       // While transitioning, only write "module" for goog.module 
404       if (depInfo.isModule()) {
405         out.print(", ");
406         writeJsBoolean(out, depInfo.isModule());
407       }
408       out.println(");");
409     }
410   }
411 
412   private void writeJsBoolean(PrintStream out, boolean value) {
413     out.print(value ? "true" : "false");
414   }
415 
416   /**
417    * Prints a list of strings formatted as a JavaScript array of string
418    * literals.
419    */
420   private static void writeJsArray(PrintStream out, Collection<String> values) {
421     if (values.isEmpty()) {
422       out.print("[]");
423     } else {
424       out.print("['");
425       out.print(Joiner.on("', '").join(values));
426       out.print("']");
427     }
428   }

430   static List<SourceFile> createSourceFilesFromPaths(
431       Collection<String> paths) {
432     List<SourceFile> files = new ArrayList<>();
433     for (String path : paths) {
434       files.add(SourceFile.fromFile(path));
435     }
436     return files;
437   }

439   static List<SourceFile> createSourceFilesFromZipPaths(
440       Collection<String> paths) throws IOException {
441     List<SourceFile> zipSourceFiles = new ArrayList<>();
442     for (String path : paths) {
443       zipSourceFiles.addAll(SourceFile.fromZipFile(path, UTF_8));
444     }
445     return zipSourceFiles;
446   }
447 
448   static List<SourceFile> createSourceFilesFromPaths(
449       String ... paths) {
450     return createSourceFilesFromPaths(Arrays.asList(paths));
451   }
