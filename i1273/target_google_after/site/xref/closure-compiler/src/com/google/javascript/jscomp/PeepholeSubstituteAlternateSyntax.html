29  /**
30   * A peephole optimization that minimizes code by simplifying conditional
31   * expressions, replacing IFs with HOOKs, replacing object constructors
32   * with literals, and simplifying returns.
33   *
34   */

51    /**
52     * @param late When late is false, this mean we are currently running before
53     * most of the other optimizations. In this case we would avoid optimizations
54     * that would make the code harder to analyze (such as using string splitting,
55     * merging statements with commas, etc). When this is true, we would
56     * do anything to minimize for size.
57     */
58    PeepholeSubstituteAlternateSyntax(boolean late) {
59      this.late = late;
60    }

62    /**
63     * Tries apply our various peephole minimizations on the passed in node.
64     */
65    @Override
66    @SuppressWarnings("fallthrough")
67    public Node optimizeSubtree(Node node) {
68      switch(node.getType()) {
69        case Token.ASSIGN_SUB:
70          return reduceSubstractionAssignment(node);
71  
72        case Token.TRUE:
73        case Token.FALSE:
74          return reduceTrueFalse(node);
75  
76        case Token.NEW:
77          node = tryFoldStandardConstructors(node);
78          if (!node.isCall()) {
79            return node;
80          }
81          // Fall through on purpose because tryFoldStandardConstructors() may
82          // convert a NEW node into a CALL node
83        case Token.CALL:
84          Node result =  tryFoldLiteralConstructor(node);
85          if (result == node) {
86            result = tryFoldSimpleFunctionCall(node);
87            if (result == node) {
88              result = tryFoldImmediateCallToBoundFunction(node);
89            }
90          }
91          return result;
92  
93        case Token.RETURN:
94          return tryReduceReturn(node);
95  
96        case Token.COMMA:
97          return trySplitComma(node);
98  
99        case Token.NAME:
100         return tryReplaceUndefined(node);
101 
102       case Token.ARRAYLIT:
103         return tryMinimizeArrayLiteral(node);
104 
105       case Token.MUL:
106       case Token.AND:
107       case Token.OR:
108       case Token.BITOR:
109       case Token.BITXOR:
110       case Token.BITAND:
111         return tryRotateAssociativeOperator(node);
112 
113       default:
114         return node; //Nothing changed
115     }
116   }

118   private Node tryRotateAssociativeOperator(Node n) {
119     if (!late) {
120       return n;
121     }
122     // All commutative operators are also associative
123     Preconditions.checkArgument(NodeUtil.isAssociative(n.getType()));
124     Node rhs = n.getLastChild();
125     if (n.getType() == rhs.getType()) {
126       // Transform a * (b * c) to a * b * c
127       Node first = n.getFirstChild().detachFromParent();
128       Node second = rhs.getFirstChild().detachFromParent();
129       Node third = rhs.getLastChild().detachFromParent();
130       Node newLhs = new Node(n.getType(), first, second)
131           .useSourceInfoIfMissingFrom(n);
132       Node newRoot = new Node(rhs.getType(), newLhs, third)
133           .useSourceInfoIfMissingFrom(rhs);
134       n.getParent().replaceChild(n, newRoot);
135       reportCodeChange();
136       return newRoot;
137     } else if (NodeUtil.isCommutative(n.getType()) &&
138                !NodeUtil.mayHaveSideEffects(n)) {
139       // Transform a * (b / c) to b / c * a
140       Node lhs = n.getFirstChild();
141       while (lhs.getType() == n.getType()) {
142         lhs = lhs.getFirstChild();
143       }
144       int precedence = NodeUtil.precedence(n.getType());
145       int lhsPrecedence = NodeUtil.precedence(lhs.getType());
146       int rhsPrecedence = NodeUtil.precedence(rhs.getType());
147       if (rhsPrecedence == precedence && lhsPrecedence != precedence) {
148         n.removeChild(rhs);
149         lhs.getParent().replaceChild(lhs, rhs);
150         n.addChildToBack(lhs);
151         reportCodeChange();
152         return n;
153       }
154     }
155     return n;
156   }

158   private Node tryFoldSimpleFunctionCall(Node n) {
159     Preconditions.checkState(n.isCall());
160     Node callTarget = n.getFirstChild();
161     if (callTarget != null && callTarget.isName() &&
162           callTarget.getString().equals("String")) {
163       // Fold String(a) to '' + (a) on immutable literals,
164       // which allows further optimizations
165       //
166       // We can't do this in the general case, because String(a) has
167       // slightly different semantics than '' + (a). See
168       // http://code.google.com/p/closure-compiler/issues/detail?id=759
169       Node value = callTarget.getNext();
170       if (value != null && value.getNext() == null &&
171           NodeUtil.isImmutableValue(value)) {
172         Node addition = IR.add(
173             IR.string("").srcref(callTarget),
174             value.detachFromParent());
175         n.getParent().replaceChild(n, addition);
176         reportCodeChange();
177         return addition;
178       }
179     }
180     return n;
181   }

183   private Node tryFoldImmediateCallToBoundFunction(Node n) {
184     // Rewriting "(fn.bind(a,b))()" to "fn.call(a,b)" makes it inlinable
185     Preconditions.checkState(n.isCall());
186     Node callTarget = n.getFirstChild();
187     Bind bind = getCodingConvention()
188         .describeFunctionBind(callTarget, false, false);
189     if (bind != null) {
190       // replace the call target
191       bind.target.detachFromParent();
192       n.replaceChild(callTarget, bind.target);
193       callTarget = bind.target;
194 
195       // push the parameters
196       addParameterAfter(bind.parameters, callTarget);
197 
198       // add the this value before the parameters if necessary
199       if (bind.thisValue != null && !NodeUtil.isUndefined(bind.thisValue)) {
200         // rewrite from "fn(a, b)" to "fn.call(thisValue, a, b)"
201         Node newCallTarget = IR.getprop(
202             callTarget.cloneTree(),
203             IR.string("call").srcref(callTarget));
204         n.replaceChild(callTarget, newCallTarget);
205         n.addChildAfter(bind.thisValue.cloneTree(), newCallTarget);
206         n.putBooleanProp(Node.FREE_CALL, false);
207       } else {
208         n.putBooleanProp(Node.FREE_CALL, true);
209       }
210       reportCodeChange();
211     }
212     return n;
213   }

215   private static void addParameterAfter(Node parameterList, Node after) {
216     if (parameterList != null) {
217       // push the last parameter to the head of the list first.
218       addParameterAfter(parameterList.getNext(), after);
219       after.getParent().addChildAfter(parameterList.cloneTree(), after);
220     }
221   }

223   private Node trySplitComma(Node n) {
224     if (late) {
225       return n;
226     }
227     Node parent = n.getParent();
228     Node left = n.getFirstChild();
229     Node right = n.getLastChild();
230 
231     if (parent.isExprResult()
232         && !parent.getParent().isLabel()) {
233       // split comma
234       n.detachChildren();
235       // Replace the original expression with the left operand.
236       parent.replaceChild(n, left);
237       // Add the right expression afterward.
238       Node newStatement = IR.exprResult(right);
239       newStatement.useSourceInfoIfMissingFrom(n);
240 
241       //This modifies outside the subtree, which is not
242       //desirable in a peephole optimization.
243       parent.getParent().addChildAfter(newStatement, parent);
244       reportCodeChange();
245       return left;
246     } else {
247       return n;
248     }
249   }

251   /**
252    * Use "void 0" in place of "undefined"
253    */
254   private Node tryReplaceUndefined(Node n) {
255     // TODO(johnlenz): consider doing this as a normalization.
256     if (isASTNormalized()
257         && NodeUtil.isUndefined(n)
258         && !NodeUtil.isLValue(n)) {
259       Node replacement = NodeUtil.newUndefinedNode(n);
260       n.getParent().replaceChild(n, replacement);
261       reportCodeChange();
262       return replacement;
263     }
264     return n;
265   }

267   /**
268    * Reduce "return undefined" or "return void 0" to simply "return".
269    *
270    * @return The original node, maybe simplified.
271    */
272   private Node tryReduceReturn(Node n) {
273     Node result = n.getFirstChild();
274 
275     if (result != null) {
276       switch (result.getType()) {
277         case Token.VOID:
278           Node operand = result.getFirstChild();
279           if (!mayHaveSideEffects(operand)) {
280             n.removeFirstChild();
281             reportCodeChange();
282           }
283           break;
284         case Token.NAME:
285           String name = result.getString();
286           if (name.equals("undefined")) {
287             n.removeFirstChild();
288             reportCodeChange();
289           }
290           break;
291       }
292     }
293 
294     return n;
295   }

297   private static final ImmutableSet<String> STANDARD_OBJECT_CONSTRUCTORS =
298     // String, Number, and Boolean functions return non-object types, whereas
299     // new String, new Number, and new Boolean return object types, so don't
300     // include them here.
301     ImmutableSet.of(
302       "Object",
303       "Array",
304       "Error"
305       );

307   /**
308    * Fold "new Object()" to "Object()".
309    */
310   private Node tryFoldStandardConstructors(Node n) {
311     Preconditions.checkState(n.isNew());
312 
313     if (canFoldStandardConstructors(n)) {
314       n.setType(Token.CALL);
315       n.putBooleanProp(Node.FREE_CALL, true);
316       reportCodeChange();
317     }
318 
319     return n;
320   }
321 
322   /**
323    * @return Whether "new Object()" can be folded to "Object()" on {@code n}.
324    */
325   private boolean canFoldStandardConstructors(Node n) {
326     // If name normalization has been run then we know that
327     // new Object() does in fact refer to what we think it is
328     // and not some custom-defined Object().
329     if (isASTNormalized() && n.getFirstChild().isName()) {
330       String className = n.getFirstChild().getString();
331       if (STANDARD_OBJECT_CONSTRUCTORS.contains(className)) {
332         return true;
333       }
334       if ("RegExp".equals(className)) {
335         // Fold "new RegExp()" to "RegExp()", but only if the argument is a string.
336         // See issue 1260.
337         if (n.getChildAtIndex(1) == null || n.getChildAtIndex(1).isString()) {
338           return true;
339         }
340       }
341     }
342 
343     return false;
344   }

346   /**
347    * Replaces a new Array, Object, or RegExp node with a literal, unless the
348    * call is to a local constructor function with the same name.
349    */
350   private Node tryFoldLiteralConstructor(Node n) {
351     Preconditions.checkArgument(n.isCall()
352         || n.isNew());
353 
354     Node constructorNameNode = n.getFirstChild();
355 
356     Node newLiteralNode = null;
357 
358     // We require the AST to be normalized to ensure that, say,
359     // Object() really refers to the built-in Object constructor
360     // and not a user-defined constructor with the same name.
361 
362     if (isASTNormalized() && Token.NAME == constructorNameNode.getType()) {
363 
364       String className = constructorNameNode.getString();
365 
366       if ("RegExp".equals(className)) {
367         // "RegExp("boo", "g")" --> /boo/g
368         return tryFoldRegularExpressionConstructor(n);
369       } else {
370         boolean constructorHasArgs = constructorNameNode.getNext() != null;
371 
372         if ("Object".equals(className) && !constructorHasArgs) {
373           // "Object()" --> "{}"
374           newLiteralNode = IR.objectlit();
375         } else if ("Array".equals(className)) {
376           // "Array(arg0, arg1, ...)" --> "[arg0, arg1, ...]"
377           Node arg0 = constructorNameNode.getNext();
378           FoldArrayAction action = isSafeToFoldArrayConstructor(arg0);
379 
380           if (action == FoldArrayAction.SAFE_TO_FOLD_WITH_ARGS ||
381               action == FoldArrayAction.SAFE_TO_FOLD_WITHOUT_ARGS) {
382             newLiteralNode = IR.arraylit();
383             n.removeChildren();
384             if (action == FoldArrayAction.SAFE_TO_FOLD_WITH_ARGS) {
385               newLiteralNode.addChildrenToFront(arg0);
386             }
387           }
388         }
389 
390         if (newLiteralNode != null) {
391           n.getParent().replaceChild(n, newLiteralNode);
392           reportCodeChange();
393           return newLiteralNode;
394         }
395       }
396     }
397     return n;
398   }

400   private static enum FoldArrayAction {
401     NOT_SAFE_TO_FOLD, SAFE_TO_FOLD_WITH_ARGS, SAFE_TO_FOLD_WITHOUT_ARGS}
402 
403   /**
404    * Checks if it is safe to fold Array() constructor into []. It can be
405    * obviously done, if the initial constructor has either no arguments or
406    * at least two. The remaining case may be unsafe since Array(number)
407    * actually reserves memory for an empty array which contains number elements.
408    */
409   private static FoldArrayAction isSafeToFoldArrayConstructor(Node arg) {
410     FoldArrayAction action = FoldArrayAction.NOT_SAFE_TO_FOLD;
411 
412     if (arg == null) {
413       action = FoldArrayAction.SAFE_TO_FOLD_WITHOUT_ARGS;
414     } else if (arg.getNext() != null) {
415       action = FoldArrayAction.SAFE_TO_FOLD_WITH_ARGS;
416     } else {
417       switch (arg.getType()) {
418         case Token.STRING:
419           // "Array('a')" --> "['a']"
420           action = FoldArrayAction.SAFE_TO_FOLD_WITH_ARGS;
421           break;
422         case Token.NUMBER:
423           // "Array(0)" --> "[]"
424           if (arg.getDouble() == 0) {
425             action = FoldArrayAction.SAFE_TO_FOLD_WITHOUT_ARGS;
426           }
427           break;
428         case Token.ARRAYLIT:
429           // "Array([args])" --> "[[args]]"
430           action = FoldArrayAction.SAFE_TO_FOLD_WITH_ARGS;
431           break;
432         default:
433       }
434     }
435     return action;
436   }

438   private Node tryFoldRegularExpressionConstructor(Node n) {
439     Node parent = n.getParent();
440     Node constructor = n.getFirstChild();
441     Node pattern = constructor.getNext();  // e.g.  ^foobar$
442     Node flags = null != pattern ? pattern.getNext() : null;  // e.g. gi
443 
444     if (null == pattern || (null != flags && null != flags.getNext())) {
445       // too few or too many arguments
446       return n;
447     }
448 
449     if (// is pattern folded
450         pattern.isString()
451         // make sure empty pattern doesn't fold to //
452         && !"".equals(pattern.getString())
453 
454         && (null == flags || flags.isString())
455         // don't escape patterns with Unicode escapes since Safari behaves badly
456         // (read can't parse or crashes) on regex literals with Unicode escapes
457         && (isEcmaScript5OrGreater()
458             || !containsUnicodeEscape(pattern.getString()))) {
459 
460       // Make sure that / is escaped, so that it will fit safely in /brackets/
461       // and make sure that no LineTerminatorCharacters appear literally inside
462       // the pattern.
463       // pattern is a string value with \\ and similar already escaped
464       pattern = makeForwardSlashBracketSafe(pattern);
465 
466       Node regexLiteral;
467       if (null == flags || "".equals(flags.getString())) {
468         // fold to /foobar/
469         regexLiteral = IR.regexp(pattern);
470       } else {
471         // fold to /foobar/gi
472         if (!areValidRegexpFlags(flags.getString())) {
473           report(INVALID_REGULAR_EXPRESSION_FLAGS, flags);
474           return n;
475         }
476         if (!areSafeFlagsToFold(flags.getString())) {
477           return n;
478         }
479         n.removeChild(flags);
480         regexLiteral = IR.regexp(pattern, flags);
481       }
482 
483       parent.replaceChild(n, regexLiteral);
484       reportCodeChange();
485       return regexLiteral;
486     }
487 
488     return n;
489   }

491   private Node reduceSubstractionAssignment(Node n) {
492     Node right = n.getLastChild();
493     if (right.isNumber()) {
494       if (right.getDouble() == 1) {
495         Node newNode = IR.dec(n.removeFirstChild(), false);
496         n.getParent().replaceChild(n, newNode);
497         reportCodeChange();
498         return newNode;
499       } else if (right.getDouble() == -1) {
500         Node newNode = IR.inc(n.removeFirstChild(), false);
501         n.getParent().replaceChild(n, newNode);
502         reportCodeChange();
503         return newNode;
504       }
505     }
506     return n;
507   }

509   private Node reduceTrueFalse(Node n) {
510     if (late) {
511       switch (n.getParent().getType()) {
512         case Token.EQ:
513         case Token.GT:
514         case Token.GE:
515         case Token.LE:
516         case Token.LT:
517         case Token.NE:
518           Node number = IR.number(n.isTrue() ? 1 : 0);
519           n.getParent().replaceChild(n, number);
520           reportCodeChange();
521           return number;
522       }
523 
524       Node not = IR.not(IR.number(n.isTrue() ? 0 : 1));
525       not.useSourceInfoIfMissingFromForTree(n);
526       n.getParent().replaceChild(n, not);
527       reportCodeChange();
528       return not;
529     }
530     return n;
531   }

533   private Node tryMinimizeArrayLiteral(Node n) {
534     boolean allStrings = true;
535     for (Node cur = n.getFirstChild(); cur != null; cur = cur.getNext()) {
536       if (!cur.isString()) {
537         allStrings = false;
538       }
539     }
540 
541     if (allStrings) {
542       return tryMinimizeStringArrayLiteral(n);
543     } else {
544       return n;
545     }
546   }

548   private Node tryMinimizeStringArrayLiteral(Node n) {
549     if (!late) {
550       return n;
551     }
552 
553     int numElements = n.getChildCount();
554     // We save two bytes per element.
555     int saving = numElements * 2 - STRING_SPLIT_OVERHEAD;
556     if (saving <= 0) {
557       return n;
558     }
559 
560     String[] strings = new String[n.getChildCount()];
561     int idx = 0;
562     for (Node cur = n.getFirstChild(); cur != null; cur = cur.getNext()) {
563       strings[idx++] = cur.getString();
564     }
565 
566     // These delimiters are chars that appears a lot in the program therefore
567     // probably have a small Huffman encoding.
568     String delimiter = pickDelimiter(strings);
569     if (delimiter != null) {
570       String template = Joiner.on(delimiter).join(strings);
571       Node call = IR.call(
572           IR.getprop(
573               IR.string(template),
574               IR.string("split")),
575           IR.string("" + delimiter));
576       call.useSourceInfoIfMissingFromForTree(n);
577       n.getParent().replaceChild(n, call);
578       reportCodeChange();
579       return call;
580     }
581     return n;
582   }

584   /**
585    * Find a delimiter that does not occur in the given strings
586    * @param strings The strings that must be separated.
587    * @return a delimiter string or null
588    */
589   private static String pickDelimiter(String[] strings) {
590     boolean allLength1 = true;
591     for (String s : strings) {
592       if (s.length() != 1) {
593         allLength1 = false;
594         break;
595       }
596     }
597 
598     if (allLength1) {
599       return "";
600     }
601 
602     String[] delimiters = new String[]{" ", ";", ",", "{", "}", null};
603     int i = 0;
604     NEXT_DELIMITER: for (; delimiters[i] != null; i++) {
605       for (String cur : strings) {
606         if (cur.contains(delimiters[i])) {
607           continue NEXT_DELIMITER;
608         }
609       }
610       break;
611     }
612     return delimiters[i];
613   }

615   private static final Pattern REGEXP_FLAGS_RE = Pattern.compile("^[gmi]*$");
616 
617   /**
618    * are the given flags valid regular expression flags?
619    * JavaScript recognizes several suffix flags for regular expressions,
620    * 'g' - global replace, 'i' - case insensitive, 'm' - multi-line.
621    * They are case insensitive, and JavaScript does not recognize the extended
622    * syntax mode, single-line mode, or expression replacement mode from Perl 5.
623    */
624   private static boolean areValidRegexpFlags(String flags) {
625     return REGEXP_FLAGS_RE.matcher(flags).matches();
626   }
627 
628   /**
629    * are the given flags safe to fold?
630    * We don't fold the regular expression if global ('g') flag is on,
631    * because in this case it isn't really a constant: its 'lastIndex'
632    * property contains the state of last execution, so replacing
633    * 'new RegExp('foobar','g')' with '/foobar/g' may change the behavior of
634    * the program if the RegExp is used inside a loop, for example.
635    * <p>
636    * ECMAScript 5 explicitly disallows pooling of regular expression literals so
637    * in ECMAScript 5, {@code /foo/g} and {@code new RegExp('foo', 'g')} are
638    * equivalent.
639    * From section 7.8.5:
640    * "Then each time the literal is evaluated, a new object is created as if by
641    * the expression new RegExp(Pattern, Flags) where RegExp is the standard
642    * built-in constructor with that name."
643    */
644   private boolean areSafeFlagsToFold(String flags) {
645     return isEcmaScript5OrGreater() || flags.indexOf('g') < 0;
646   }

648   /**
649    * returns a string node that can safely be rendered inside /brackets/.
650    */
651   private static Node makeForwardSlashBracketSafe(Node n) {
652     String s = n.getString();
653     // sb contains everything in s[0:pos]
654     StringBuilder sb = null;
655     int pos = 0;
656     boolean isEscaped = false, inCharset = false;
657     for (int i = 0; i < s.length(); ++i) {
658       char ch = s.charAt(i);
659       switch (ch) {
660         case '\\':
661           isEscaped = !isEscaped;
662           continue;
663         case '/':
664           // Escape a literal forward slash if it is not already escaped and is
665           // not inside a character set.
666           //     new RegExp('/') -> /\//
667           // but the following do not need extra escaping
668           //     new RegExp('\\/') -> /\//
669           //     new RegExp('[/]') -> /[/]/
670           if (!isEscaped && !inCharset) {
671             if (null == sb) { sb = new StringBuilder(s.length() + 16); }
672             sb.append(s, pos, i).append('\\');
673             pos = i;
674           }
675           break;
676         case '[':
677           if (!isEscaped) {
678             inCharset = true;
679           }
680           break;
681         case ']':
682           if (!isEscaped) {
683             inCharset = false;
684           }
685           break;
686         case '\r': case '\n': case '\u2028': case '\u2029':
687           // LineTerminators cannot appear raw inside a regular
688           // expression literal.
689           // They can't appear legally in a quoted string, but when
690           // the quoted string from
691           //     new RegExp('\n')
692           // reaches here, the quoting has been removed.
693           // Requote just these code-points.
694           if (null == sb) { sb = new StringBuilder(s.length() + 16); }
695           if (isEscaped) {
696             sb.append(s, pos, i - 1);
697           } else {
698             sb.append(s, pos, i);
699           }
700           switch (ch) {
701             case '\r': sb.append("\\r"); break;
702             case '\n': sb.append("\\n"); break;
703             case '\u2028': sb.append("\\u2028"); break;
704             case '\u2029': sb.append("\\u2029"); break;
705           }
706           pos = i + 1;
707           break;
708       }
709       isEscaped = false;
710     }
711 
712     if (null == sb) { return n.cloneTree(); }
713 
714     sb.append(s, pos, s.length());
715     return IR.string(sb.toString()).srcref(n);
716   }

718   /**
719    * true if the JavaScript string would contain a Unicode escape when written
720    * out as the body of a regular expression literal.
721    */
722   static boolean containsUnicodeEscape(String s) {
723     String esc = REGEXP_ESCAPER.regexpEscape(s);
724     for (int i = -1; (i = esc.indexOf("\\u", i + 1)) >= 0;) {
725       int nSlashes = 0;
726       while (i - nSlashes > 0 && '\\' == esc.charAt(i - nSlashes - 1)) {
727         ++nSlashes;
728       }
729       // if there are an even number of slashes before the \ u then it is a
730       // Unicode literal.
731       if (0 == (nSlashes & 1)) { return true; }
732     }
733     return false;
734   }
