27  /**
28   * A peephole optimization that minimizes conditional expressions
29   * according to De Morgan's laws.
30   * Also rewrites conditional statements as expressions by replacing them
31   * with HOOKs and short-circuit binary operators.
32   *
33   * Based on PeepholeSubstituteAlternateSyntax
34   */
35  class PeepholeMinimizeConditions
36    extends AbstractPeepholeOptimization {
37  
38    private static final int AND_PRECEDENCE = NodeUtil.precedence(Token.AND);
39  
40    private final boolean late;
41  
42    /**
43     * @param late When late is false, this mean we are currently running before
44     * most of the other optimizations. In this case we would avoid optimizations
45     * that would make the code harder to analyze (such as using string splitting,
46     * merging statements with commas, etc). When this is true, we would
47     * do anything to minimize for size.
48     */
49    PeepholeMinimizeConditions(boolean late) {
50      this.late = late;
51    }
52  
53    /**
54     * Tries to apply our various peephole minimizations on the passed in node.
55     */
56    @Override
57    @SuppressWarnings("fallthrough")
58    public Node optimizeSubtree(Node node) {
59      switch(node.getType()) {
60        case Token.THROW:
61        case Token.RETURN: {
62          Node result = tryRemoveRedundantExit(node);
63          if (result != node) {
64            return result;
65          }
66          return tryReplaceExitWithBreak(node);
67        }
68  
69        // TODO(johnlenz): Maybe remove redundant BREAK and CONTINUE. Overlaps
70        // with MinimizeExitPoints.
71  
72        case Token.NOT:
73          tryMinimizeCondition(node.getFirstChild());
74          return tryMinimizeNot(node);
75  
76        case Token.IF:
77          performConditionSubstitutions(node.getFirstChild());
78          return tryMinimizeIf(node);
79  
80        case Token.EXPR_RESULT:
81          performConditionSubstitutions(node.getFirstChild());
82          return tryMinimizeExprResult(node);
83  
84        case Token.HOOK:
85          performConditionSubstitutions(node.getFirstChild());
86          return tryMinimizeHook(node);
87  
88        case Token.WHILE:
89        case Token.DO:
90          tryMinimizeCondition(NodeUtil.getConditionExpression(node));
91          return node;
92  
93        case Token.FOR:
94          if (!NodeUtil.isForIn(node)) {
95            tryJoinForCondition(node);
96            tryMinimizeCondition(NodeUtil.getConditionExpression(node));
97          }
98          return node;
99  
100       case Token.BLOCK:
101         return tryReplaceIf(node);
102 
103       default:
104         return node; //Nothing changed
105     }
106   }
107 
108   private void tryJoinForCondition(Node n) {
109     if (!late) {
110       return;
111     }
112 
113     Node block = n.getLastChild();
114     Node maybeIf = block.getFirstChild();
115     if (maybeIf != null && maybeIf.isIf()) {
116       Node thenBlock = maybeIf.getChildAtIndex(1);
117       Node maybeBreak = thenBlock.getFirstChild();
118       if (maybeBreak != null && maybeBreak.isBreak()
119           && !maybeBreak.hasChildren()) {
120 
121         // Preserve the IF ELSE expression is there is one.
122         if (maybeIf.getChildCount() == 3) {
123           block.replaceChild(maybeIf,
124               maybeIf.getLastChild().detachFromParent());
125         } else {
126           NodeUtil.redeclareVarsInsideBranch(thenBlock);
127           block.removeFirstChild();
128         }
129 
130         Node ifCondition = maybeIf.removeFirstChild();
131         Node fixedIfCondition = IR.not(ifCondition)
132             .srcref(ifCondition);
133 
134         // OK, join the IF expression with the FOR expression
135         Node forCondition = NodeUtil.getConditionExpression(n);
136         if (forCondition.isEmpty()) {
137           n.replaceChild(forCondition, fixedIfCondition);
138         } else {
139           Node replacement = new Node(Token.AND);
140           n.replaceChild(forCondition, replacement);
141           replacement.addChildToBack(forCondition);
142           replacement.addChildToBack(fixedIfCondition);
143         }
144 
145         reportCodeChange();
146       }
147     }
148   }

150   /**
151    * Use "return x?1:2;" in place of "if(x)return 1;return 2;"
152    */
153   private Node tryReplaceIf(Node n) {
154 
155     for (Node child = n.getFirstChild();
156          child != null; child = child.getNext()){
157       if (child.isIf()){
158         Node cond = child.getFirstChild();
159         Node thenBranch = cond.getNext();
160         Node elseBranch = thenBranch.getNext();
161         Node nextNode = child.getNext();
162 
163         if (nextNode != null && elseBranch == null
164             && isReturnBlock(thenBranch)
165             && nextNode.isIf()) {
166           Node nextCond = nextNode.getFirstChild();
167           Node nextThen = nextCond.getNext();
168           Node nextElse = nextThen.getNext();
169           if (thenBranch.isEquivalentToTyped(nextThen)) {
170             // Transform
171             //   if (x) return 1; if (y) return 1;
172             // to
173             //   if (x||y) return 1;
174             child.detachFromParent();
175             child.detachChildren();
176             Node newCond = new Node(Token.OR, cond);
177             nextNode.replaceChild(nextCond, newCond);
178             newCond.addChildToBack(nextCond);
179             reportCodeChange();
180           } else if (nextElse != null
181               && thenBranch.isEquivalentToTyped(nextElse)) {
182             // Transform
183             //   if (x) return 1; if (y) foo() else return 1;
184             // to
185             //   if (!x&&y) foo() else return 1;
186             child.detachFromParent();
187             child.detachChildren();
188             Node newCond = new Node(Token.AND,
189                 IR.not(cond).srcref(cond));
190             nextNode.replaceChild(nextCond, newCond);
191             newCond.addChildToBack(nextCond);
192             reportCodeChange();
193           }
194         } else if (nextNode != null && elseBranch == null &&
195             isReturnBlock(thenBranch) && isReturnExpression(nextNode)) {
196           Node thenExpr = null;
197           // if(x)return; return 1 -> return x?void 0:1
198           if (isReturnExpressBlock(thenBranch)) {
199             thenExpr = getBlockReturnExpression(thenBranch);
200             thenExpr.detachFromParent();
201           } else {
202             thenExpr = NodeUtil.newUndefinedNode(child);
203           }
204 
205           Node elseExpr = nextNode.getFirstChild();
206 
207           cond.detachFromParent();
208           elseExpr.detachFromParent();
209 
210           Node returnNode = IR.returnNode(
211                                 IR.hook(cond, thenExpr, elseExpr)
212                                     .srcref(child));
213           n.replaceChild(child, returnNode);
214           n.removeChild(nextNode);
215           reportCodeChange();
216         } else if (elseBranch != null && statementMustExitParent(thenBranch)) {
217           child.removeChild(elseBranch);
218           n.addChildAfter(elseBranch, child);
219           reportCodeChange();
220         }
221       }
222     }
223     return n;
224   }

226   private static boolean statementMustExitParent(Node n) {
227     switch (n.getType()) {
228       case Token.THROW:
229       case Token.RETURN:
230         return true;
231       case Token.BLOCK:
232         if (n.hasChildren()) {
233           Node child = n.getLastChild();
234           return statementMustExitParent(child);
235         }
236         return false;
237       // TODO(johnlenz): handle TRY/FINALLY
238       case Token.FUNCTION:
239       default:
240         return false;
241     }
242   }

244   /**
245    * Replace duplicate exits in control structures.  If the node following
246    * the exit node expression has the same effect as exit node, the node can
247    * be replaced or removed.
248    * For example:
249    *   "while (a) {return f()} return f();" ==> "while (a) {break} return f();"
250    *   "while (a) {throw 'ow'} throw 'ow';" ==> "while (a) {break} throw 'ow';"
251    *
252    * @param n An follow control exit expression (a THROW or RETURN node)
253    * @return The replacement for n, or the original if no change was made.
254    */
255   private Node tryReplaceExitWithBreak(Node n) {
256     Node result = n.getFirstChild();
257 
258     // Find the enclosing control structure, if any, that a "break" would exit
259     // from.
260     Node breakTarget = n;
261     for (; !ControlFlowAnalysis.isBreakTarget(breakTarget, null /* no label */);
262         breakTarget = breakTarget.getParent()) {
263       if (breakTarget.isFunction() || breakTarget.isScript()) {
264         // No break target.
265         return n;
266       }
267     }
268 
269     Node follow = ControlFlowAnalysis.computeFollowNode(breakTarget);
270 
271     // Skip pass all the finally blocks because both the break and return will
272     // also trigger all the finally blocks. However, the order of execution is
273     // slightly changed. Consider:
274     //
275     // return a() -> finally { b() } -> return a()
276     //
277     // which would call a() first. However, changing the first return to a
278     // break will result in calling b().
279 
280     Node prefinallyFollows = follow;
281     follow = skipFinallyNodes(follow);
282 
283     if (prefinallyFollows != follow) {
284       // There were finally clauses
285       if (!isPure(result)) {
286         // Can't defer the exit
287         return n;
288       }
289     }
290 
291     if (follow == null && (n.isThrow() || result != null)) {
292       // Can't complete remove a throw here or a return with a result.
293       return n;
294     }
295 
296     // When follow is null, this mean the follow of a break target is the
297     // end of a function. This means a break is same as return.
298     if (follow == null || areMatchingExits(n, follow)) {
299       Node replacement = IR.breakNode();
300       n.getParent().replaceChild(n, replacement);
301       this.reportCodeChange();
302       return replacement;
303     }
304 
305     return n;
306   }

308   /**
309    * Remove duplicate exits.  If the node following the exit node expression
310    * has the same effect as exit node, the node can be removed.
311    * For example:
312    *   "if (a) {return f()} return f();" ==> "if (a) {} return f();"
313    *   "if (a) {throw 'ow'} throw 'ow';" ==> "if (a) {} throw 'ow';"
314    *
315    * @param n An follow control exit expression (a THROW or RETURN node)
316    * @return The replacement for n, or the original if no change was made.
317    */
318   private Node tryRemoveRedundantExit(Node n) {
319     Node exitExpr = n.getFirstChild();
320 
321     Node follow = ControlFlowAnalysis.computeFollowNode(n);
322 
323     // Skip pass all the finally blocks because both the fall through and return
324     // will also trigger all the finally blocks.
325     Node prefinallyFollows = follow;
326     follow = skipFinallyNodes(follow);
327     if (prefinallyFollows != follow) {
328       // There were finally clauses
329       if (!isPure(exitExpr)) {
330         // Can't replace the return
331         return n;
332       }
333     }
334 
335     if (follow == null && (n.isThrow() || exitExpr != null)) {
336       // Can't complete remove a throw here or a return with a result.
337       return n;
338     }
339 
340     // When follow is null, this mean the follow of a break target is the
341     // end of a function. This means a break is same as return.
342     if (follow == null || areMatchingExits(n, follow)) {
343       n.detachFromParent();
344       reportCodeChange();
345       return null;
346     }
347 
348     return n;
349   }

351   /**
352    * @return Whether the expression does not produces and can not be affected
353    * by side-effects.
354    */
355   boolean isPure(Node n) {
356     return n == null
357         || (!NodeUtil.canBeSideEffected(n)
358             && !mayHaveSideEffects(n));
359   }

361   /**
362    * @return n or the node following any following finally nodes.
363    */
364   static Node skipFinallyNodes(Node n) {
365     while (n != null && NodeUtil.isTryFinallyNode(n.getParent(), n)) {
366       n = ControlFlowAnalysis.computeFollowNode(n);
367     }
368     return n;
369   }

371   /**
372    * Check whether one exit can be replaced with another. Verify:
373    * 1) They are identical expressions
374    * 2) If an exception is possible that the statements, the original
375    * and the potential replacement are in the same exception handler.
376    */
377   boolean areMatchingExits(Node nodeThis, Node nodeThat) {
378     return nodeThis.isEquivalentTo(nodeThat)
379         && (!isExceptionPossible(nodeThis)
380             || getExceptionHandler(nodeThis) == getExceptionHandler(nodeThat));
381   }
382 
383   static boolean isExceptionPossible(Node n) {
384     // TODO(johnlenz): maybe use ControlFlowAnalysis.mayThrowException?
385     Preconditions.checkState(n.isReturn()
386         || n.isThrow());
387     return n.isThrow()
388         || (n.hasChildren()
389             && !NodeUtil.isLiteralValue(n.getLastChild(), true));
390   }
391 
392   static Node getExceptionHandler(Node n) {
393     return ControlFlowAnalysis.getExceptionHandler(n);
394   }

396   /**
397    * Try to minimize NOT nodes such as !(x==y).
398    *
399    * Returns the replacement for n or the original if no change was made
400    */
401   private Node tryMinimizeNot(Node n) {
402     Preconditions.checkArgument(n.isNot());
403     Node parent = n.getParent();
404 
405     Node notChild = n.getFirstChild();
406     // negative operator of the current one : == -> != for instance.
407     int complementOperator;
408     switch (notChild.getType()) {
409       case Token.EQ:
410         complementOperator = Token.NE;
411         break;
412       case Token.NE:
413         complementOperator = Token.EQ;
414         break;
415       case Token.SHEQ:
416         complementOperator = Token.SHNE;
417         break;
418       case Token.SHNE:
419         complementOperator = Token.SHEQ;
420         break;
421       // GT, GE, LT, LE are not handled in this because !(x<NaN) != x>=NaN.
422       default:
423         return n;
424     }
425     Node newOperator = n.removeFirstChild();
426     newOperator.setType(complementOperator);
427     parent.replaceChild(n, newOperator);
428     reportCodeChange();
429     return newOperator;
430   }

433   /**
434    * Try to remove leading NOTs from EXPR_RESULTS.
435    *
436    * Returns the replacement for n or the original if no replacement was
437    * necessary.
438    */
439   private Node tryMinimizeExprResult(Node n) {
440     MinimizedCondition minCond =
441         MinimizedCondition.fromConditionNode(n.getFirstChild());
442     MinimizedCondition.MeasuredNode mNode =
443         minCond.getMinimized(MinimizationStyle.ALLOW_LEADING_NOT);
444     Node placeholder = minCond.getPlaceholder();
445     if (mNode.getNode().isNot()) {
446       // Remove the leading NOT in the EXPR_RESULT.
447       n.replaceChild(placeholder, mNode.getNode().removeFirstChild());
448       reportCodeChange();
449     } else {
450       replaceNode(placeholder, mNode);
451     }
452     return n;
453   }

455   /**
456    * Try flipping HOOKs that have negated conditions.
457    *
458    * Returns the replacement for n or the original if no replacement was
459    * necessary.
460    */
461   private Node tryMinimizeHook(Node n) {
462     MinimizedCondition minCond =
463       MinimizedCondition.fromConditionNode(n.getFirstChild());
464     MinimizedCondition.MeasuredNode mNode =
465         minCond.getMinimized(MinimizationStyle.ALLOW_LEADING_NOT);
466     Node placeholder = minCond.getPlaceholder();
467     if (mNode.getNode().isNot()) {
468       // Swap the HOOK
469       Node thenBranch = n.getFirstChild().getNext();
470       n.replaceChild(placeholder, mNode.getNode().removeFirstChild());
471       n.removeChild(thenBranch);
472       n.addChildToBack(thenBranch);
473       reportCodeChange();
474     } else {
475       replaceNode(placeholder, mNode);
476     }
477     return n;
478   }

480   /**
481    * Try turning IF nodes into smaller HOOKs
482    *
483    * Returns the replacement for n or the original if no replacement was
484    * necessary.
485    */
486   private Node tryMinimizeIf(Node n) {
487 
488     Node parent = n.getParent();
489 
490     Node originalCond = n.getFirstChild();
491 
492     /* If the condition is a literal, we'll let other
493      * optimizations try to remove useless code.
494      */
495     if (NodeUtil.isLiteralValue(originalCond, true)) {
496       return n;
497     }
498 
499     Node thenBranch = originalCond.getNext();
500     Node elseBranch = thenBranch.getNext();
501 
502     MinimizedCondition minCond =
503         MinimizedCondition.fromConditionNode(originalCond);
504     originalCond = null;  // originalCond was mutated and should not be used.
505 
506     Node placeholder = minCond.getPlaceholder();
507 
508     MinimizedCondition.MeasuredNode unnegatedCond;
509     MinimizedCondition.MeasuredNode shortCond;
510     // Compute two minimized representations. The first representation counts
511     // a leading NOT node, and the second ignores a leading NOT node.
512     // If we can fold the if statement into a HOOK or boolean operation,
513     // then the NOT node does not matter, and we prefer the second condition.
514     // If we cannot fold the if statement, then we prefer the first condition.
515     unnegatedCond = minCond.getMinimized(MinimizationStyle.PREFER_UNNEGATED);
516     shortCond = minCond.getMinimized(MinimizationStyle.ALLOW_LEADING_NOT);
517 
518     if (elseBranch == null) {
519       if (isFoldableExpressBlock(thenBranch)) {
520         Node expr = getBlockExpression(thenBranch);
521         if (!late && isPropertyAssignmentInExpression(expr)) {
522           // Keep opportunities for CollapseProperties such as
523           // a.longIdentifier || a.longIdentifier = ... -> var a = ...;
524           // until CollapseProperties has been run.
525           replaceNode(placeholder, unnegatedCond);
526           return n;
527         }
528 
529         if (shortCond.getNode().isNot()) {
530           // if(!x)bar(); -> x||bar();
531           Node or = IR.or(
532               shortCond.getNode().removeFirstChild(),
533               expr.removeFirstChild()).srcref(n);
534           Node newExpr = NodeUtil.newExpr(or);
535           parent.replaceChild(n, newExpr);
536           reportCodeChange();
537 
538           return newExpr;
539         }
540         // True, but removed for performance reasons.
541         // Preconditions.checkState(shortCond.isEquivalentTo(unnegatedCond));
542 
543         // if(x)foo(); -> x&&foo();
544         if (isLowerPrecedence(shortCond.getNode(), AND_PRECEDENCE) &&
545             isLowerPrecedence(expr.getFirstChild(),
546                 AND_PRECEDENCE)) {
547           // One additional set of parentheses is worth the change even if
548           // there is no immediate code size win. However, two extra pair of
549           // {}, we would have to think twice. (unless we know for sure the
550           // we can further optimize its parent.
551           replaceNode(placeholder, shortCond);
552           return n;
553         }
554 
555         n.removeChild(placeholder);
556         Node and = IR.and(shortCond.getNode(), expr.removeFirstChild()).srcref(n);
557         Node newExpr = NodeUtil.newExpr(and);
558         parent.replaceChild(n, newExpr);
559         reportCodeChange();
560 
561         return newExpr;
562       } else {
563 
564         // Try to combine two IF-ELSE
565         if (NodeUtil.isStatementBlock(thenBranch) &&
566             thenBranch.hasOneChild()) {
567           Node innerIf = thenBranch.getFirstChild();
568 
569           if (innerIf.isIf()) {
570             Node innerCond = innerIf.getFirstChild();
571             Node innerThenBranch = innerCond.getNext();
572             Node innerElseBranch = innerThenBranch.getNext();
573 
574             if (innerElseBranch == null &&
575                  !(isLowerPrecedence(unnegatedCond.getNode(), AND_PRECEDENCE) &&
576                    isLowerPrecedence(innerCond, AND_PRECEDENCE))) {
577               n.detachChildren();
578               n.addChildToBack(
579                   IR.and(
580                       unnegatedCond.getNode(),
581                       innerCond.detachFromParent())
582                       .srcref(placeholder));
583               n.addChildrenToBack(innerThenBranch.detachFromParent());
584               reportCodeChange();
585               // Not worth trying to fold the current IF-ELSE into && because
586               // the inner IF-ELSE wasn't able to be folded into && anyways.
587               return n;
588             }
589           }
590         }
591       }
592       replaceNode(placeholder, unnegatedCond);
593       return n;
594     }
595 
596     /* TODO(dcc) This modifies the siblings of n, which is undesirable for a
597      * peephole optimization. This should probably get moved to another pass.
598      */
599     tryRemoveRepeatedStatements(n);
600 
601     // if(!x)foo();else bar(); -> if(x)bar();else foo();
602     // An additional set of curly braces isn't worth it.
603     if (shortCond.getNode().isNot() && !consumesDanglingElse(elseBranch)) {
604       n.replaceChild(placeholder, shortCond.getNode().removeFirstChild());
605       n.removeChild(thenBranch);
606       n.addChildToBack(thenBranch);
607       reportCodeChange();
608       return n;
609     }
610 
611     // if(x)return 1;else return 2; -> return x?1:2;
612     if (isReturnExpressBlock(thenBranch) && isReturnExpressBlock(elseBranch)) {
613       Node thenExpr = getBlockReturnExpression(thenBranch);
614       Node elseExpr = getBlockReturnExpression(elseBranch);
615       n.removeChild(placeholder);
616       thenExpr.detachFromParent();
617       elseExpr.detachFromParent();
618 
619       // note - we ignore any cases with "return;", technically this
620       // can be converted to "return undefined;" or some variant, but
621       // that does not help code size.
622       Node returnNode = IR.returnNode(
623                             IR.hook(shortCond.getNode(), thenExpr, elseExpr)
624                                 .srcref(n));
625       parent.replaceChild(n, returnNode);
626       reportCodeChange();
627       return returnNode;
628     }
629 
630     boolean thenBranchIsExpressionBlock = isFoldableExpressBlock(thenBranch);
631     boolean elseBranchIsExpressionBlock = isFoldableExpressBlock(elseBranch);
632 
633     if (thenBranchIsExpressionBlock && elseBranchIsExpressionBlock) {
634       Node thenOp = getBlockExpression(thenBranch).getFirstChild();
635       Node elseOp = getBlockExpression(elseBranch).getFirstChild();
636       if (thenOp.getType() == elseOp.getType()) {
637         // if(x)a=1;else a=2; -> a=x?1:2;
638         if (NodeUtil.isAssignmentOp(thenOp)) {
639           Node lhs = thenOp.getFirstChild();
640           if (areNodesEqualForInlining(lhs, elseOp.getFirstChild()) &&
641               // if LHS has side effects, don't proceed [since the optimization
642               // evaluates LHS before cond]
643               // NOTE - there are some circumstances where we can
644               // proceed even if there are side effects...
645               !mayEffectMutableState(lhs) &&
646               (!mayHaveSideEffects(unnegatedCond.getNode()) ||
647                   (thenOp.isAssign() && thenOp.getFirstChild().isName()))) {
648 
649             n.removeChild(placeholder);
650             Node assignName = thenOp.removeFirstChild();
651             Node thenExpr = thenOp.removeFirstChild();
652             Node elseExpr = elseOp.getLastChild();
653             elseOp.removeChild(elseExpr);
654 
655             Node hookNode = IR.hook(shortCond.getNode(), thenExpr, elseExpr)
656                 .srcref(n);
657             Node assign = new Node(thenOp.getType(), assignName, hookNode)
658                               .srcref(thenOp);
659             Node expr = NodeUtil.newExpr(assign);
660             parent.replaceChild(n, expr);
661             reportCodeChange();
662 
663             return expr;
664           }
665         }
666       }
667       // if(x)foo();else bar(); -> x?foo():bar()
668       n.removeChild(placeholder);
669       thenOp.detachFromParent();
670       elseOp.detachFromParent();
671       Node expr = IR.exprResult(
672           IR.hook(shortCond.getNode(), thenOp, elseOp).srcref(n));
673       parent.replaceChild(n, expr);
674       reportCodeChange();
675       return expr;
676     }
677 
678     boolean thenBranchIsVar = isVarBlock(thenBranch);
679     boolean elseBranchIsVar = isVarBlock(elseBranch);
680 
681     // if(x)var y=1;else y=2  ->  var y=x?1:2
682     if (thenBranchIsVar && elseBranchIsExpressionBlock &&
683         getBlockExpression(elseBranch).getFirstChild().isAssign()) {
684 
685       Node var = getBlockVar(thenBranch);
686       Node elseAssign = getBlockExpression(elseBranch).getFirstChild();
687 
688       Node name1 = var.getFirstChild();
689       Node maybeName2 = elseAssign.getFirstChild();
690 
691       if (name1.hasChildren()
692           && maybeName2.isName()
693           && name1.getString().equals(maybeName2.getString())) {
694         Node thenExpr = name1.removeChildren();
695         Node elseExpr = elseAssign.getLastChild().detachFromParent();
696         placeholder.detachFromParent();
697         Node hookNode = IR.hook(shortCond.getNode(), thenExpr, elseExpr)
698                             .srcref(n);
699         var.detachFromParent();
700         name1.addChildrenToBack(hookNode);
701         parent.replaceChild(n, var);
702         reportCodeChange();
703         return var;
704       }
705 
706     // if(x)y=1;else var y=2  ->  var y=x?1:2
707     } else if (elseBranchIsVar && thenBranchIsExpressionBlock &&
708         getBlockExpression(thenBranch).getFirstChild().isAssign()) {
709 
710       Node var = getBlockVar(elseBranch);
711       Node thenAssign = getBlockExpression(thenBranch).getFirstChild();
712 
713       Node maybeName1 = thenAssign.getFirstChild();
714       Node name2 = var.getFirstChild();
715 
716       if (name2.hasChildren()
717           && maybeName1.isName()
718           && maybeName1.getString().equals(name2.getString())) {
719         Node thenExpr = thenAssign.getLastChild().detachFromParent();
720         Node elseExpr = name2.removeChildren();
721         placeholder.detachFromParent();
722         Node hookNode = IR.hook(shortCond.getNode(), thenExpr, elseExpr)
723                             .srcref(n);
724         var.detachFromParent();
725         name2.addChildrenToBack(hookNode);
726         parent.replaceChild(n, var);
727         reportCodeChange();
728 
729         return var;
730       }
731     }
732 
733     replaceNode(placeholder, unnegatedCond);
734     return n;
735   }

737   /**
738    * Try to remove duplicate statements from IF blocks. For example:
739    *
740    * if (a) {
741    *   x = 1;
742    *   return true;
743    * } else {
744    *   x = 2;
745    *   return true;
746    * }
747    *
748    * becomes:
749    *
750    * if (a) {
751    *   x = 1;
752    * } else {
753    *   x = 2;
754    * }
755    * return true;
756    *
757    * @param n The IF node to examine.
758    */
759   private void tryRemoveRepeatedStatements(Node n) {
760     Preconditions.checkState(n.isIf());
761 
762     Node parent = n.getParent();
763     if (!NodeUtil.isStatementBlock(parent)) {
764       // If the immediate parent is something like a label, we
765       // can't move the statement, so bail.
766       return;
767     }
768 
769     Node cond = n.getFirstChild();
770     Node trueBranch = cond.getNext();
771     Node falseBranch = trueBranch.getNext();
772     Preconditions.checkNotNull(trueBranch);
773     Preconditions.checkNotNull(falseBranch);
774 
775     while (true) {
776       Node lastTrue = trueBranch.getLastChild();
777       Node lastFalse = falseBranch.getLastChild();
778       if (lastTrue == null || lastFalse == null
779           || !areNodesEqualForInlining(lastTrue, lastFalse)) {
780         break;
781       }
782       lastTrue.detachFromParent();
783       lastFalse.detachFromParent();
784       parent.addChildAfter(lastTrue, n);
785       reportCodeChange();
786     }
787   }

789   /**
790    * @return Whether the node is a block with a single statement that is
791    *     an expression.
792    */
793   private static boolean isFoldableExpressBlock(Node n) {
794     if (n.isBlock()) {
795       if (n.hasOneChild()) {
796         Node maybeExpr = n.getFirstChild();
797         if (maybeExpr.isExprResult()) {
798           // IE has a bug where event handlers behave differently when
799           // their return value is used vs. when their return value is in
800           // an EXPR_RESULT. It's pretty freaking weird. See:
801           // http://code.google.com/p/closure-compiler/issues/detail?id=291
802           // We try to detect this case, and not fold EXPR_RESULTs
803           // into other expressions.
804           if (maybeExpr.getFirstChild().isCall()) {
805             Node calledFn = maybeExpr.getFirstChild().getFirstChild();
806 
807             // We only have to worry about methods with an implicit 'this'
808             // param, or this doesn't happen.
809             if (calledFn.isGetElem()) {
810               return false;
811             } else if (calledFn.isGetProp() &&
812                        calledFn.getLastChild().getString().startsWith("on")) {
813               return false;
814             }
815           }
816 
817           return true;
818         }
819         return false;
820       }
821     }
822 
823     return false;
824   }

909   /**
910    * Does a statement consume a 'dangling else'? A statement consumes
911    * a 'dangling else' if an 'else' token following the statement
912    * would be considered by the parser to be part of the statement.
913    */
914   private static boolean consumesDanglingElse(Node n) {
915     while (true) {
916       switch (n.getType()) {
917         case Token.IF:
918           if (n.getChildCount() < 3) {
919             return true;
920           }
921           // This IF node has no else clause.
922           n = n.getLastChild();
923           continue;
924         case Token.BLOCK:
925           if (n.getChildCount() != 1) {
926             return false;
927           }
928           // This BLOCK has no curly braces.
929           n = n.getLastChild();
930           continue;
931         case Token.WITH:
932         case Token.WHILE:
933         case Token.FOR:
934           n = n.getLastChild();
935           continue;
936         default:
937           return false;
938       }
939     }
940   }

990   /**
991    * Try to minimize the given condition by applying local substitutions.
992    *
993    * The following types of transformations are performed:
994    *   x || true        --> true
995    *   x && true        --> x
996    *   x ? false : true --> !x
997    *   x ? true : y     --> x || y
998    *
999    *   Returns the replacement for n, or the original if no change was made
1000    */
1001   private Node performConditionSubstitutions(Node n) {
1002     Node parent = n.getParent();
1003 
1004     switch (n.getType()) {
1005       case Token.OR:
1006       case Token.AND: {
1007         Node left = n.getFirstChild();
1008         Node right = n.getLastChild();
1009 
1010         // Because the expression is in a boolean context minimize
1011         // the children, this can't be done in the general case.
1012         left = performConditionSubstitutions(left);
1013         right = performConditionSubstitutions(right);
1014 
1015         // Remove useless conditionals
1016         // Handle the following cases:
1017         //   x || false --> x
1018         //   x && true --> x
1019         // This works regardless of whether x has side effects.
1020         //
1021         // If x does not have side effects:
1022         //   x || true  --> true
1023         //   x && false  --> false
1024         //
1025         // If x may have side effects:
1026         //   x || true  --> x,true
1027         //   x && false  --> x,false
1028         //
1029         // In the last two cases, code size may increase slightly (adding
1030         // some parens because the comma operator has a low precedence) but
1031         // the new AST is easier for other passes to handle.
1032         TernaryValue rightVal = NodeUtil.getPureBooleanValue(right);
1033         if (NodeUtil.getPureBooleanValue(right) != TernaryValue.UNKNOWN) {
1034           int type = n.getType();
1035           Node replacement = null;
1036           boolean rval = rightVal.toBoolean(true);
1037 
1038           // (x || FALSE) => x
1039           // (x && TRUE) => x
1040           if (type == Token.OR && !rval ||
1041               type == Token.AND && rval) {
1042             replacement = left;
1043           } else if (!mayHaveSideEffects(left)) {
1044             replacement = right;
1045           } else {
1046             // expr_with_sideeffects || true  =>  expr_with_sideeffects, true
1047             // expr_with_sideeffects && false  =>  expr_with_sideeffects, false
1048             n.detachChildren();
1049             replacement = IR.comma(left, right);
1050           }
1051 
1052           if (replacement != null) {
1053             n.detachChildren();
1054             parent.replaceChild(n, replacement);
1055             reportCodeChange();
1056             return replacement;
1057           }
1058         }
1059         return n;
1060       }
1061 
1062       case Token.HOOK: {
1063         Node condition = n.getFirstChild();
1064         Node trueNode = n.getFirstChild().getNext();
1065         Node falseNode = n.getLastChild();
1066 
1067         // Because the expression is in a boolean context minimize
1068         // the result children, this can't be done in the general case.
1069         // The condition is handled in the general case in #optimizeSubtree
1070         trueNode = performConditionSubstitutions(trueNode);
1071         falseNode = performConditionSubstitutions(falseNode);
1072 
1073         // Handle four cases:
1074         //   x ? true : false --> x
1075         //   x ? false : true --> !x
1076         //   x ? true : y     --> x || y
1077         //   x ? y : false    --> x && y
1078         Node replacement = null;
1079         TernaryValue trueNodeVal = NodeUtil.getPureBooleanValue(trueNode);
1080         TernaryValue falseNodeVal = NodeUtil.getPureBooleanValue(falseNode);
1081         if (trueNodeVal == TernaryValue.TRUE
1082             && falseNodeVal == TernaryValue.FALSE) {
1083           // Remove useless conditionals, keep the condition
1084           condition.detachFromParent();
1085           replacement = condition;
1086         } else if (trueNodeVal == TernaryValue.FALSE
1087             && falseNodeVal == TernaryValue.TRUE) {
1088           // Remove useless conditionals, keep the condition
1089           condition.detachFromParent();
1090           replacement = IR.not(condition);
1091         } else if (trueNodeVal == TernaryValue.TRUE) {
1092           // Remove useless true case.
1093           n.detachChildren();
1094           replacement = IR.or(condition, falseNode);
1095         } else if (falseNodeVal == TernaryValue.FALSE) {
1096           // Remove useless false case
1097           n.detachChildren();
1098           replacement = IR.and(condition, trueNode);
1099         }
1100 
1101         if (replacement != null) {
1102           parent.replaceChild(n, replacement);
1103           n = replacement;
1104           reportCodeChange();
1105         }
1106 
1107         return n;
1108       }
1109 
1110       default:
1111         // while(true) --> while(1)
1112         TernaryValue nVal = NodeUtil.getPureBooleanValue(n);
1113         if (nVal != TernaryValue.UNKNOWN) {
1114           boolean result = nVal.toBoolean(true);
1115           int equivalentResult = result ? 1 : 0;
1116           return maybeReplaceChildWithNumber(n, parent, equivalentResult);
1117         }
1118         // We can't do anything else currently.
1119         return n;
1120     }
1121   }

1123   /**
1124    * Replaces a node with a number node if the new number node is not equivalent
1125    * to the current node.

1127    * Returns the replacement for n if it was replaced, otherwise returns n.
