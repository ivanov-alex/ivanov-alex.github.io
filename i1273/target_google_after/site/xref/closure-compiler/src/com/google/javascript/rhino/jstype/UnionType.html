54  /**
55   * The {@code UnionType} implements a common JavaScript idiom in which the
56   * code is specifically designed to work with multiple input types.  Because
57   * JavaScript always knows the run-time type of an object value, this is safer
58   * than a C union.<p>
59   *
60   * For instance, values of the union type {@code (String,boolean)} can be of
61   * type {@code String} or of type {@code boolean}. The commutativity of the
62   * statement is captured by making {@code (String,boolean)} and
63   * {@code (boolean,String)} equal.<p>
64   *
65   * The implementation of this class prevents the creation of nested
66   * unions.<p>
67   */
68  public class UnionType extends JSType {
69    private static final long serialVersionUID = 1L;
70  
71    // alternates without merging structural interfaces and their subtypes
72    Collection<JSType> alternatesWithoutStucturalTyping;
73    // alternates under structural typing
74    Collection<JSType> alternates;
75    private int hashcode;
76  
77    /**
78     * Creates a union type.
79     *
80     * @param alternatesWithoutStructuralTyping the alternates of the union without
81     * structural typing subtype
82     */
83    UnionType(JSTypeRegistry registry,
84        Collection<JSType> alternatesWithoutStructuralTyping) {
85      super(registry);
86      this.alternatesWithoutStucturalTyping = alternatesWithoutStructuralTyping;
87  
88      UnionTypeBuilder builder = new UnionTypeBuilder(registry);
89      for (JSType alternate : alternatesWithoutStructuralTyping) {
90        builder.addAlternate(alternate, true);
91      }
92      this.alternates = builder.getAlternates();
93      this.hashcode = this.alternatesWithoutStucturalTyping.hashCode();
94    }

96    /**
97     * Gets the alternate types of this union type.
98     * @return The alternate types of this union type. The returned set is
99     *     immutable.
100    */
101   public Collection<JSType> getAlternates() {
102     for (JSType t : alternatesWithoutStucturalTyping) {
103       if (t.isUnionType()) {
104         rebuildAlternates();
105         break;
106       }
107     }
108     return alternates;
109   }

111   /**
112    * Gets the alternate types of this union type, including structural interfaces
113    *  and implicit implementations as are distinct alternates.
114    * @return The alternate types of this union type. The returned set is
115    *     immutable.
116    */
117   public Collection<JSType> getAlternatesWithoutStructuralTyping() {
118     for (JSType t : alternatesWithoutStucturalTyping) {
119       if (t.isUnionType()) {
120         rebuildAlternates();
121         break;
122       }
123     }
124     return alternatesWithoutStucturalTyping;
125   }

127   /**
128    * Use UnionTypeBuilder to rebuild the list of alternates and hashcode
129    * of the current UnionType.
130    */
131   private void rebuildAlternates() {
132     UnionTypeBuilder builder = new UnionTypeBuilder(registry);
133     for (JSType alternate : alternatesWithoutStucturalTyping) {
134       builder.addAlternate(alternate);
135     }
136     alternatesWithoutStucturalTyping = builder.getAlternates();
137     builder = new UnionTypeBuilder(registry);
138     for (JSType alternate : alternatesWithoutStucturalTyping) {
139       builder.addAlternate(alternate, true);
140     }
141     alternates = builder.getAlternates();
142     hashcode = alternatesWithoutStucturalTyping.hashCode();
143   }

211   @Override
212   public JSType findPropertyType(String propertyName) {
213     JSType propertyType = null;
214 
215     for (JSType alternate : getAlternates()) {
216       // Filter out the null/undefined type.
217       if (alternate.isNullType() || alternate.isVoidType()) {
218         continue;
219       }
220 
221       JSType altPropertyType = alternate.findPropertyType(propertyName);
222       if (altPropertyType == null) {
223         continue;
224       }
225 
226       if (propertyType == null) {
227         propertyType = altPropertyType;
228       } else {
229         propertyType = propertyType.getLeastSupertype(altPropertyType);
230       }
231     }
232 
233     return propertyType;
234   }

339   @Override
340   public JSType getLeastSupertype(JSType that) {
341     if (!that.isUnknownType() && !that.isUnionType()) {
342       for (JSType alternate : alternatesWithoutStucturalTyping) {
343         if (!alternate.isUnknownType() && that.isSubtype(alternate)) {
344           return this;
345         }
346       }
347     }
348 
349     return getLeastSupertype(this, that);
350   }

352   JSType meet(JSType that) {
353     UnionTypeBuilder builder = new UnionTypeBuilder(registry);
354     for (JSType alternate : alternatesWithoutStucturalTyping) {
355       if (alternate.isSubtype(that)) {
356         builder.addAlternate(alternate);
357       }
358     }
359 
360     if (that.isUnionType()) {
361       for (JSType otherAlternate : that.toMaybeUnionType().alternatesWithoutStucturalTyping) {
362         if (otherAlternate.isSubtype(this)) {
363           builder.addAlternate(otherAlternate);
364         }
365       }
366     } else if (that.isSubtype(this)) {
367       builder.addAlternate(that);
368     }
369     JSType result = builder.build();
370     if (!result.isNoType()) {
371       return result;
372     } else if (this.isObject() && (that.isObject() && !that.isNoType())) {
373       return getNativeType(JSTypeNative.NO_OBJECT_TYPE);
374     } else {
375       return getNativeType(JSTypeNative.NO_TYPE);
376     }
377   }

379   /**
380    * Two union types are equal if, after flattening nested union types,
381    * they have the same number of alternates and all alternates are equal.
382    */
383   boolean checkUnionEquivalenceHelper(
384       UnionType that, EquivalenceMethod eqMethod, EqCache eqCache) {
385     Collection<JSType> thatAlternates = that.getAlternatesWithoutStructuralTyping();
386     if (eqMethod == EquivalenceMethod.IDENTITY
387         && getAlternatesWithoutStructuralTyping().size() != thatAlternates.size()) {
388       return false;
389     }
390     for (JSType alternate : thatAlternates) {
391       if (!hasAlternate(alternate, eqMethod, eqCache)) {
392         return false;
393       }
394     }
395     return true;
396   }
397 
398   private boolean hasAlternate(JSType type, EquivalenceMethod eqMethod,
399       EqCache eqCache) {
400     for (JSType alternate : getAlternatesWithoutStructuralTyping()) {
401       if (alternate.checkEquivalenceHelper(type, eqMethod, eqCache)) {
402         return true;
403       }
404     }
405     return false;
406   }

455   /**
456    * Returns a more restricted union type than {@code this} one, in which all
457    * subtypes of {@code type} have been removed.<p>
458    *
459    * Examples:
460    * <ul>
461    * <li>{@code (number,string)} restricted by {@code number} is
462    *     {@code string}</li>
463    * <li>{@code (null, EvalError, URIError)} restricted by
464    *     {@code Error} is {@code null}</li>
465    * </ul>
466    *
467    * @param type the supertype of the types to remove from this union type
468    */
469   public JSType getRestrictedUnion(JSType type) {
470     UnionTypeBuilder restricted = new UnionTypeBuilder(registry);
471     for (JSType t : alternatesWithoutStucturalTyping) {
472       // Keep all unknown/unresolved types.
473       if (t.isUnknownType() || t.isNoResolvedType() || !t.isSubtype(type)) {
474         restricted.addAlternate(t);
475       }
476     }
477     return restricted.build();
478   }

480   @Override String toStringHelper(boolean forAnnotations) {
481     StringBuilder result = new StringBuilder();
482     boolean firstAlternate = true;
483 
484     result.append("(");
485     SortedSet<JSType> sorted = new TreeSet<>(ALPHA);
486     sorted.addAll(alternatesWithoutStucturalTyping);
487     for (JSType t : sorted) {
488       if (!firstAlternate) {
489         result.append("|");
490       }
491       result.append(t.toStringHelper(forAnnotations));
492       firstAlternate = false;
493     }
494     result.append(")");
495     return result.toString();
496   }

503   @Override
504   protected boolean isSubtype(JSType that,
505       ImplCache implicitImplCache) {
506     // unknown
507     if (that.isUnknownType()) {
508       return true;
509     }
510     // all type
511     if (that.isAllType()) {
512       return true;
513     }
514     for (JSType element : alternatesWithoutStucturalTyping) {
515       if (!element.isSubtype(that, implicitImplCache)) {
516         return false;
517       }
518     }
519     return true;
520   }

608   @Override
609   JSType resolveInternal(ErrorReporter t, StaticTypedScope<JSType> scope) {
610     setResolvedTypeInternal(this); // for circularly defined types.
611 
612     // Just resolve the alternates, but do not update as that breaks some error
613     // reporting cases.
614     for (JSType alternate : alternatesWithoutStucturalTyping) {
615       alternate.resolve(t, scope);
616     }
617     // Ensure the union is in a normalized state.
618     rebuildAlternates();
619     return this;
620   }

639   @Override
640   public JSType collapseUnion() {
641     JSType currentValue = null;
642     ObjectType currentCommonSuper = null;
643     for (JSType a : alternatesWithoutStucturalTyping) {
644       if (a.isUnknownType()) {
645         return getNativeType(JSTypeNative.UNKNOWN_TYPE);
646       }
647 
648       ObjectType obj = a.toObjectType();
649       if (obj == null) {
650         if (currentValue == null && currentCommonSuper == null) {
651           // If obj is not an object, then it must be a value.
652           currentValue = a;
653         } else {
654           // Multiple values and objects will always collapse to the ALL_TYPE.
655           return getNativeType(JSTypeNative.ALL_TYPE);
656         }
657       } else if (currentValue != null) {
658         // Values and objects will always collapse to the ALL_TYPE.
659         return getNativeType(JSTypeNative.ALL_TYPE);
660       } else if (currentCommonSuper == null) {
661         currentCommonSuper = obj;
662       } else {
663         currentCommonSuper =
664             registry.findCommonSuperObject(currentCommonSuper, obj);
665       }
666     }
667     return currentCommonSuper;
668   }
