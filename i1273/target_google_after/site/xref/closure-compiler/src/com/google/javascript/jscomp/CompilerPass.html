17  package com.google.javascript.jscomp;
18  
19  import com.google.javascript.rhino.Node;
20  
21  /**
22   * <p>Interface for classes that can compile JS.</p>
23   *
24   * <p>Class has single function "process", which is passed
25   * the root node of the parsed JS tree, as well as the
26   * root node of the external JS tree (used to provide a public API
27   * and prevent renaming of system functions).</p>
28   *
29   * <p>Use this class to support testing with BaseCompilerTest</p>
30   *
31   */
32  public interface CompilerPass {
33  
34    /**
35     * Process the JS with root node root.
36     * Can modify the contents of each Node tree
37     * @param externs Top of external JS tree
38     * @param root Top of JS tree
39     */
40    void process(Node externs, Node root);
41  }
