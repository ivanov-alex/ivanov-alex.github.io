52  /**
53   * A {@link JSModule} dependency graph that assigns a depth to each module and
54   * can answer depth-related queries about them. For the purposes of this class,
55   * a module's depth is defined as the number of hops in the longest (non cyclic)
56   * path from the module to a module with no dependencies.
57   *
58   */
59  public final class JSModuleGraph {
60  
61    private List<JSModule> modules;
62  
63    /**
64     * Lists of modules at each depth. <code>modulesByDepth.get(3)</code> is a
65     * list of the modules at depth 3, for example.
66     */
67    private List<List<JSModule>> modulesByDepth;
68  
69    /**
70     * dependencyMap is a cache of dependencies that makes the dependsOn
71     * function faster.  Each map entry associates a starting
72     * JSModule with the set of JSModules that are transitively dependent on the
73     * starting module.
74     *
75     * If the cache returns null, then the entry hasn't been filled in for that
76     * module.
77     *
78     * dependencyMap should be filled from leaf to root so that
79     * getTransitiveDepsDeepestFirst can use its results directly.
80     */
81    private Map<JSModule, Set<JSModule>> dependencyMap = new HashMap<>();

84     * Creates a module graph from a list of modules in dependency order.

86    public JSModuleGraph(JSModule[] modulesInDepOrder) {
87      this(ImmutableList.copyOf(modulesInDepOrder));

90    /**
91     * Creates a module graph from a list of modules in dependency order.
92     */
93    public JSModuleGraph(List<JSModule> modulesInDepOrder) {
94      Preconditions.checkState(
95          modulesInDepOrder.size() == new HashSet<>(modulesInDepOrder).size(),
96          "Found duplicate modules");
97      modules = ImmutableList.copyOf(modulesInDepOrder);
98      modulesByDepth = new ArrayList<>();
99  
100     for (JSModule module : modulesInDepOrder) {
101       int depth = 0;
102       for (JSModule dep : module.getDependencies()) {
103         int depDepth = dep.getDepth();
104         if (depDepth < 0) {
105           throw new ModuleDependenceException(SimpleFormat.format(
106               "Modules not in dependency order: %s preceded %s",
107               module.getName(), dep.getName()),
108               module, dep);
109         }
110         depth = Math.max(depth, depDepth + 1);
111       }
112 
113       module.setDepth(depth);
114       if (depth == modulesByDepth.size()) {
115         modulesByDepth.add(new ArrayList<JSModule>());
116       }
117       modulesByDepth.get(depth).add(module);
118     }
119   }

153   /**
154    * Returns a JSON representation of the JSModuleGraph. Specifically a
155    * JsonArray of "Modules" where each module has a
156    * - "name"
157    * - "dependencies" (list of module names)
158    * - "transitive-dependencies" (list of module names, deepest first)
159    * - "inputs" (list of file names)
160    * @return List of module JSONObjects.
161    */
162   @GwtIncompatible("com.google.gson")
163   JsonArray toJson() {
164     JsonArray modules = new JsonArray();
165     for (JSModule module : getAllModules()) {
166       JsonObject node = new JsonObject();
167       try {
168         node.add("name", new JsonPrimitive(module.getName()));
169         JsonArray deps = new JsonArray();
170         node.add("dependencies", deps);
171         for (JSModule m : module.getDependencies()) {
172           deps.add(new JsonPrimitive(m.getName()));
173         }
174         JsonArray transitiveDeps = new JsonArray();
175         node.add("transitive-dependencies", transitiveDeps);
176         for (JSModule m : getTransitiveDepsDeepestFirst(module)) {
177           transitiveDeps.add(new JsonPrimitive(m.getName()));
178         }
179         JsonArray inputs = new JsonArray();
180         node.add("inputs", inputs);
181         for (CompilerInput input : module.getInputs()) {
182           inputs.add(new JsonPrimitive(
183               input.getSourceFile().getOriginalPath()));
184         }
185         modules.add(node);
186       } catch (JsonParseException e) {
187         Throwables.propagate(e);
188       }
189     }
190     return modules;
191   }

193   /**
194    * Determines whether this module depends on a given module. Note that a
195    * module never depends on itself, as that dependency would be cyclic.
196    */
197   public boolean dependsOn(JSModule src, JSModule m) {
198     Set<JSModule> deps = dependencyMap.get(src);
199     if (deps == null) {
200       deps = getTransitiveDepsDeepestFirst(src);
201       dependencyMap.put(src, deps);
202     }
203 
204     return deps.contains(m);
205   }

207   /**
208    * Finds the deepest common dependency of two modules, not including the two
209    * modules themselves.
210    *
211    * @param m1 A module in this graph
212    * @param m2 A module in this graph
213    * @return The deepest common dep of {@code m1} and {@code m2}, or null if
214    *     they have no common dependencies
215    */
216   JSModule getDeepestCommonDependency(JSModule m1, JSModule m2) {
217     int m1Depth = m1.getDepth();
218     int m2Depth = m2.getDepth();
219     // According our definition of depth, the result must have a strictly
220     // smaller depth than either m1 or m2.
221     for (int depth = Math.min(m1Depth, m2Depth) - 1; depth >= 0; depth--) {
222       List<JSModule> modulesAtDepth = modulesByDepth.get(depth);
223       // Look at the modules at this depth in reverse order, so that we use the
224       // original ordering of the modules to break ties (later meaning deeper).
225       for (int i = modulesAtDepth.size() - 1; i >= 0; i--) {
226         JSModule m = modulesAtDepth.get(i);
227         if (dependsOn(m1, m) && dependsOn(m2, m)) {
228           return m;
229         }
230       }
231     }
232     return null;
233   }

235   /**
236    * Finds the deepest common dependency of two modules, including the
237    * modules themselves.
238    *
239    * @param m1 A module in this graph
240    * @param m2 A module in this graph
241    * @return The deepest common dep of {@code m1} and {@code m2}, or null if
242    *     they have no common dependencies
243    */
244   public JSModule getDeepestCommonDependencyInclusive(
245       JSModule m1, JSModule m2) {
246     if (m2 == m1 || dependsOn(m2, m1)) {
247       return m1;
248     } else if (dependsOn(m1, m2)) {
249       return m2;
250     }
251 
252     return getDeepestCommonDependency(m1, m2);
253   }
254 
255   /** Returns the deepest common dependency of the given modules. */
256   public JSModule getDeepestCommonDependencyInclusive(
257       Collection<JSModule> modules) {
258     Iterator<JSModule> iter = modules.iterator();
259     JSModule dep = iter.next();
260     while (iter.hasNext()) {
261       dep = getDeepestCommonDependencyInclusive(dep, iter.next());
262     }
263     return dep;
264   }

266   /**
267    * Creates an iterable over the transitive dependencies of module {@code m}
268    * in a non-increasing depth ordering. The result does not include the module
269    * {@code m}.
270    *
271    * @param m A module in this graph
272    * @return The transitive dependencies of module {@code m}
273    */
274   Set<JSModule> getTransitiveDepsDeepestFirst(JSModule m) {
275     Set<JSModule> deps = dependencyMap.get(m);
276     if (deps != null) {
277       return deps;
278     }
279     deps = new TreeSet<>(new InverseDepthComparator());
280     addDeps(deps, m);
281     dependencyMap.put(m, deps);
282     return deps;
283   }

285   /**
286    * Adds a module's transitive dependencies to a set.
287    */
288   private static void addDeps(Set<JSModule> deps, JSModule m) {
289     for (JSModule dep : m.getDependencies()) {
290       deps.add(dep);
291       addDeps(deps, dep);
292     }
293   }

295   /**
296    * Replaces any files that are found multiple times with a single instance in
297    * the closest parent module that is common to all modules where it appears.
298    *
299    * JSCompiler normally errors if you attempt to compile modules containing the
300    * same file.  This method can be used to remove duplicates before compiling
301    * to avoid such an error.
302    */
303   public void coalesceDuplicateFiles() {
304     Multimap<String, JSModule> fileRefs = LinkedHashMultimap.create();
305     for (JSModule module : modules) {
306       for (CompilerInput jsFile : module.getInputs()) {
307         fileRefs.put(jsFile.getName(), module);
308       }
309     }
310 
311     for (String path : fileRefs.keySet()) {
312       Collection<JSModule> refModules = fileRefs.get(path);
313       if (refModules.size() > 1) {
314         JSModule depModule = getDeepestCommonDependencyInclusive(refModules);
315         CompilerInput file = refModules.iterator().next().getByName(path);
316         for (JSModule module : refModules) {
317           if (module != depModule) {
318             module.removeByName(path);
319           }
320         }
321         if (!refModules.contains(depModule)) {
322           depModule.add(file);
323         }
324       }
325     }
326   }

328   /**
329    * Applies a DependencyOptions in "dependency sorting" and "dependency pruning"
330    * mode to the given list of inputs. Returns a new list with the files sorted
331    * and removed. This module graph will be updated to reflect the new list.
332    *
333    * If you need more fine-grained dependency management, you should create your
334    * own DependencyOptions and call
335    * {@code manageDependencies(DependencyOptions, List<CompilerInput>)}.
336    *
337    * @param entryPoints The entry points into the program.
338    *     Expressed as JS symbols.
339    * @param inputs The original list of sources. Used to ensure that the sort
340    *     is stable.
341    * @throws CircularDependencyException if there is a circular dependency
342    *     between the provides and requires.
343    * @throws MissingProvideException if an entry point was not provided
344    *     by any of the inputs.
345    * @see DependencyOptions for more info on how this works.
346    */
347   public List<CompilerInput> manageDependencies(
348       List<String> entryPoints,
349       List<CompilerInput> inputs)
350       throws CircularDependencyException,
351           MissingModuleException,
352           MissingProvideException {
353     DependencyOptions depOptions = new DependencyOptions();
354     depOptions.setDependencySorting(true);
355     depOptions.setDependencyPruning(true);
356     depOptions.setEntryPoints(entryPoints);
357     return manageDependencies(depOptions, inputs);
358   }

360   /**
361    * Apply the dependency options to the list of sources, returning a new
362    * source list re-ordering and dropping files as necessary.
363    * This module graph will be updated to reflect the new list.
364    *
365    * @param inputs The original list of sources. Used to ensure that the sort
366    *     is stable.
367    * @throws CircularDependencyException if there is a circular dependency
368    *     between the provides and requires.
369    * @throws MissingProvideException if an entry point was not provided
370    *     by any of the inputs.
371    * @see DependencyOptions for more info on how this works.
372    */
373   public List<CompilerInput> manageDependencies(
374       DependencyOptions depOptions,
375       List<CompilerInput> inputs)
376       throws CircularDependencyException, MissingProvideException,
377           MissingModuleException {
378 
379     SortedDependencies<CompilerInput> sorter =
380         depOptions.isEs6ModuleOrder()
381             ? new Es6SortedDependencies<>(inputs) : new ClosureSortedDependencies<>(inputs);
382 
383     Iterable<CompilerInput> entryPointInputs = createEntryPointInputs(
384         depOptions, inputs, sorter);
385 
386     // The order of inputs, sorted independently of modules.
387     List<CompilerInput> absoluteOrder =
388         sorter.getDependenciesOf(inputs, depOptions.shouldSortDependencies());
389 
390     // Figure out which sources *must* be in each module.
391     ListMultimap<JSModule, CompilerInput> entryPointInputsPerModule =
392         LinkedListMultimap.create();
393     for (CompilerInput input : entryPointInputs) {
394       JSModule module = input.getModule();
395       Preconditions.checkNotNull(module);
396       entryPointInputsPerModule.put(module, input);
397     }
398 
399     // Clear the modules of their inputs. This also nulls out
400     // the input's reference to its module.
401     for (JSModule module : getAllModules()) {
402       module.removeAll();
403     }
404 
405     // Figure out which sources *must* be in each module, or in one
406     // of that module's dependencies.
407     for (JSModule module : entryPointInputsPerModule.keySet()) {
408       List<CompilerInput> transitiveClosure =
409           sorter.getDependenciesOf(
410               entryPointInputsPerModule.get(module),
411               depOptions.shouldSortDependencies());
412       for (CompilerInput input : transitiveClosure) {
413         JSModule oldModule = input.getModule();
414         if (oldModule == null) {
415           input.setModule(module);
416         } else {
417           input.setModule(null);
418           input.setModule(
419               getDeepestCommonDependencyInclusive(oldModule, module));
420         }
421       }
422     }
423 
424     // All the inputs are pointing to the modules that own them. Yeah!
425     // Update the modules to reflect this.
426     for (CompilerInput input : absoluteOrder) {
427       JSModule module = input.getModule();
428       if (module != null) {
429         module.add(input);
430       }
431     }
432 
433     // Now, generate the sorted result.
434     ImmutableList.Builder<CompilerInput> result = ImmutableList.builder();
435     for (JSModule module : getAllModules()) {
436       result.addAll(module.getInputs());
437     }
438 
439     return result.build();
440   }

442   private Collection<CompilerInput> createEntryPointInputs(
443       DependencyOptions depOptions,
444       List<CompilerInput> inputs,
445       SortedDependencies<CompilerInput> sorter)
446       throws MissingModuleException, MissingProvideException {
447     Set<CompilerInput> entryPointInputs = new LinkedHashSet<>();
448     Map<String, JSModule> modulesByName = getModulesByName();
449 
450     if (depOptions.shouldPruneDependencies()) {
451       if (!depOptions.shouldDropMoochers()) {
452         entryPointInputs.addAll(sorter.getInputsWithoutProvides());
453       }
454 
455       for (String entryPoint : depOptions.getEntryPoints()) {
456         // An entry point is either formatted as:
457         // 'foo.bar' - peg foo.bar to its current module
458         // 'modC:foo.bar' - peg foo.bar to modC
459         String inputName = entryPoint;
460         int splitPoint = entryPoint.indexOf(':');
461         CompilerInput entryPointInput = null;
462         if (splitPoint != -1) {
463           String moduleName = entryPoint.substring(0, splitPoint);
464           inputName = entryPoint.substring(
465               Math.min(splitPoint + 1, entryPoint.length() - 1));
466           JSModule module = modulesByName.get(moduleName);
467           if (module == null) {
468             throw new MissingModuleException(moduleName);
469           } else {
470             entryPointInput = sorter.getInputProviding(inputName);
471             entryPointInput.overrideModule(module);
472           }
473         } else {
474           entryPointInput = sorter.getInputProviding(inputName);
475         }
476 
477         entryPointInputs.add(entryPointInput);
478       }
479 
480       CompilerInput baseJs = sorter.maybeGetInputProviding("goog");
481       if (baseJs != null) {
482         entryPointInputs.add(baseJs);
483       }
484     } else {
485       entryPointInputs.addAll(inputs);
486     }
487     return entryPointInputs;
488   }

490   @SuppressWarnings("unused")
491   LinkedDirectedGraph<JSModule, String> toGraphvizGraph() {
492     LinkedDirectedGraph<JSModule, String> graphViz =
493         LinkedDirectedGraph.create();
494     for (JSModule module : getAllModules()) {
495       graphViz.createNode(module);
496       for (JSModule dep : module.getDependencies()) {
497         graphViz.createNode(dep);
498         graphViz.connect(module, "->", dep);
499       }
500     }
501     return graphViz;
502   }

504   /**
505    * A module depth comparator that considers a deeper module to be "less than"
506    * a shallower module. Uses module names to consistently break ties.
507    */
508   private static class InverseDepthComparator implements Comparator<JSModule> {
509     @Override
510     public int compare(JSModule m1, JSModule m2) {
511       return depthCompare(m2, m1);
512     }
513   }
514 
515   private static int depthCompare(JSModule m1, JSModule m2) {
516     if (m1 == m2) {
517       return 0;
518     }
519     int d1 = m1.getDepth();
520     int d2 = m2.getDepth();
521     return d1 < d2 ? -1 : d2 == d1 ? m1.getName().compareTo(m2.getName()) : 1;
522   }

524   /**
525    * Exception class for declaring when the modules being fed into a
526    * JSModuleGraph as input aren't in dependence order, and so can't be
527    * processed for caching of various dependency-related queries.
528    */
529   protected static class ModuleDependenceException
530       extends IllegalArgumentException {
531     private static final long serialVersionUID = 1;
532 
533     private final JSModule module;
534     private final JSModule dependentModule;
535 
536     protected ModuleDependenceException(String message,
537         JSModule module, JSModule dependentModule) {
538       super(message);
539       this.module = module;
540       this.dependentModule = dependentModule;
541     }
542 
543     public JSModule getModule() {
544       return module;
545     }
546 
547     public JSModule getDependentModule() {
548       return dependentModule;
549     }
550   }
