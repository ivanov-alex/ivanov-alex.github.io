39  package com.google.javascript.rhino;
40  
41  import com.google.common.collect.ImmutableList;
42  import junit.framework.TestCase;
43  
44  /**
45   * @author johnlenz@google.com (John Lenz)
46   */
47  public class IRTest extends TestCase {
48  
49    public void testEmpty() {
50      testIR(IR.empty(), "EMPTY\n");
51    }
52  
53    public void testFunction() {
54      testIR(IR.function(IR.name("hi"), IR.paramList(), IR.block()),
55          "FUNCTION hi\n" +
56          "    NAME hi\n" +
57          "    PARAM_LIST\n" +
58          "    BLOCK\n");
59    }
60  
61    public void testParamList() {
62      testIR(IR.paramList(),
63          "PARAM_LIST\n");
64  
65      testIR(IR.paramList(IR.name("a"), IR.name("b")),
66          "PARAM_LIST\n" +
67          "    NAME a\n" +
68          "    NAME b\n");
69  
70      testIR(IR.paramList(ImmutableList.of(IR.name("a"), IR.name("b"))),
71          "PARAM_LIST\n" +
72          "    NAME a\n" +
73          "    NAME b\n");
74    }
75  
76    public void testBlock() {
77      testIR(IR.block(),
78          "BLOCK\n");
79  
80      testIR(IR.block(IR.empty(), IR.empty()),
81          "BLOCK\n" +
82          "    EMPTY\n" +
83          "    EMPTY\n");
84  
85      testIR(IR.block(ImmutableList.of(IR.empty(), IR.empty())),
86          "BLOCK\n" +
87          "    EMPTY\n" +
88          "    EMPTY\n");
89    }
90  
91    public void testScript() {
92      testIR(IR.script(),
93          "SCRIPT\n");
94  
95      testIR(IR.script(IR.empty(), IR.empty()),
96          "SCRIPT\n" +
97          "    EMPTY\n" +
98          "    EMPTY\n");
99  
100     testIR(IR.script(ImmutableList.of(IR.empty(), IR.empty())),
101         "SCRIPT\n" +
102         "    EMPTY\n" +
103         "    EMPTY\n");
104   }
105 
106   public void testScriptThrows() {
107     boolean caught = false;
108     try {
109       IR.script(IR.returnNode());
110     } catch(IllegalStateException e) {
111       caught = true;
112     }
113     assertTrue("expected exception was not seen", caught);
114   }
115 
116   public void testVar() {
117     testIR(IR.var(IR.name("a")),
118         "VAR\n" +
119         "    NAME a\n");
120 
121     testIR(IR.var(IR.name("a"), IR.trueNode()),
122         "VAR\n" +
123         "    NAME a\n" +
124         "        TRUE\n");
125   }
126 
127   public void testReturn() {
128     testIR(IR.returnNode(),
129         "RETURN\n");
130 
131     testIR(IR.returnNode(IR.name("a")),
132         "RETURN\n" +
133         "    NAME a\n");
134   }
135 
136   public void testThrow() {
137     testIR(IR.throwNode(IR.name("a")),
138         "THROW\n" +
139         "    NAME a\n");
140   }
141 
142   public void testExprResult() {
143     testIR(IR.exprResult(IR.name("a")),
144         "EXPR_RESULT\n" +
145         "    NAME a\n");
146   }
147 
148   public void testIf() {
149     testIR(IR.ifNode(IR.name("a"), IR.block()),
150         "IF\n" +
151         "    NAME a\n" +
152         "    BLOCK\n");
153 
154     testIR(IR.ifNode(IR.name("a"), IR.block(), IR.block()),
155         "IF\n" +
156         "    NAME a\n" +
157         "    BLOCK\n" +
158         "    BLOCK\n");
159   }
160 
161   public void testIssue727_1() {
162     testIR(
163         IR.tryFinally(
164             IR.block(),
165             IR.block()),
166         "TRY\n" +
167         "    BLOCK\n" +
168         "    BLOCK\n" +
169         "    BLOCK\n");
170   }
171 
172   public void testIssue727_2() {
173     testIR(
174         IR.tryCatch(
175             IR.block(),
176             IR.catchNode(
177                 IR.name("e"),
178                 IR.block())),
179         "TRY\n" +
180         "    BLOCK\n" +
181         "    BLOCK\n" +
182         "        CATCH\n" +
183         "            NAME e\n" +
184         "            BLOCK\n");
185   }
186 
187   public void testIssue727_3() {
188     testIR(
189         IR.tryCatchFinally(
190             IR.block(),
191             IR.catchNode(IR.name("e"), IR.block()),
192             IR.block()),
193         "TRY\n" +
194         "    BLOCK\n" +
195         "    BLOCK\n" +
196         "        CATCH\n" +
197         "            NAME e\n" +
198         "            BLOCK\n" +
199         "    BLOCK\n");
200   }
201 
202   public void testAdd() {
203     testIR(
204         IR.add(
205             IR.cast(IR.number(1)),
206             IR.number(2)),
207         "ADD\n" +
208         "    CAST\n" +
209         "        NUMBER 1.0\n" +
210         "    NUMBER 2.0\n");
211 
212   }
213 
214   private void testIR(Node node, String expectedStructure) {
215     assertEquals(expectedStructure, node.toStringTree());
216   }
217 }
