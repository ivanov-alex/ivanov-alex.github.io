17  package com.google.javascript.jscomp;
18  
19  /**
20   * Tests for variable declaration collapsing.
21   *
22   */
23  public final class CollapseVariableDeclarationsTest extends CompilerTestCase {
24    @Override
25    public void setUp() {
26      compareJsDoc = false;
27    }
28  
29    public void testCollapsing() throws Exception {
30      // Basic collapsing
31      test("var a;var b;",
32           "var a,b;");
33      // With initial values
34      test("var a = 1;var b = 1;",
35           "var a=1,b=1;");
36      // Already collapsed
37      test("var a, b;",
38           "var a,b;");
39      // Already collapsed with values
40      test("var a = 1, b = 1;",
41           "var a=1,b=1;");
42      // Some already collapsed
43      test("var a;var b, c;var d;",
44           "var a,b,c,d;");
45      // Some already collapsed with values
46      test("var a = 1;var b = 2, c = 3;var d = 4;",
47           "var a=1,b=2,c=3,d=4;");
48    }
49  
50    public void testIssue820() throws Exception {
51      // Don't redeclare function parameters, this is incompatible with
52      // strict mode.
53      testSame("function f(a){ var b=1; a=2; var c; }");
54    }
55  
56    public void testIfElseVarDeclarations() throws Exception {
57      testSame("if (x) var a = 1; else var b = 2;");
58    }
59  
60    public void testAggressiveRedeclaration() {
61      test("var x = 2; foo(x);     x = 3; var y = 2;",
62           "var x = 2; foo(x); var x = 3,     y = 2;");
63  
64      test("var x = 2; foo(x);     x = 3; x = 1; var y = 2;",
65           "var x = 2; foo(x); var x = 3, x = 1,     y = 2;");
66  
67      test("var x = 2; foo(x);     x = 3; x = 1; var y = 2; var z = 4",
68           "var x = 2; foo(x); var x = 3, x = 1,     y = 2,     z = 4");
69  
70      test("var x = 2; foo(x);     x = 3; x = 1; var y = 2; var z = 4; x = 5",
71           "var x = 2; foo(x); var x = 3, x = 1,     y = 2,     z = 4, x = 5");
72    }
73  
74    public void testAggressiveRedeclarationInFor() {
75      testSame("for(var x = 1; x = 2; x = 3) {x = 4}");
76      testSame("for(var x = 1; y = 2; z = 3) {var a = 4}");
77      testSame("var x; for(x = 1; x = 2; z = 3) {x = 4}");
78    }
79  
80    public void testIssue397() {
81      test("var x; var y = 3; x = 5;",
82           "var x, y = 3; x = 5;");
83      testSame("var x; x = 5; var z = 7;");
84      test("var x; var y = 3; x = 5; var z = 7;",
85           "var x, y = 3; x = 5; var z = 7;");
86      test("var a = 1; var x; var y = 3; x = 5;",
87           "var a = 1, x, y = 3; x = 5;");
88    }
89  
90    @Override
91    protected CompilerPass getProcessor(Compiler compiler) {
92      return new CollapseVariableDeclarations(compiler);
93    }
94  }
