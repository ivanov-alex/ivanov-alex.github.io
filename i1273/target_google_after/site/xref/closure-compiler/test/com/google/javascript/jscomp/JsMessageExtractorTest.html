37    private JsMessage.Style mode;

39    @Override
40    protected void setUp() throws Exception {
41      super.setUp();
42      mode = JsMessage.Style.LEGACY;
43    }
44  
45    private Collection<JsMessage> extractMessages(String... js) {
46      try {
47        String sourceCode = Joiner.on("\n").join(js);
48        return new JsMessageExtractor(null, mode)
49            .extractMessages(SourceFile.fromCode("testcode", sourceCode));
50      } catch (IOException e) {
51        fail(e.getMessage());
52        return null;
53      }
54    }
55  
56    private JsMessage extractMessage(String... js) {
57      Collection<JsMessage> messages = extractMessages(js);
58      assertThat(messages).hasSize(1);
59      return messages.iterator().next();
60    }

62    public void testSyntaxError1() {
63      try {
64        extractMessage("if (true) {}}");
65        fail("Expected exception");
66      } catch (RuntimeException e) {
67        assertThat(e.getMessage()).contains("JSCompiler errors\n");
68        assertThat(e.getMessage()).contains("testcode:1: ERROR - Parse error");
69        assertThat(e.getMessage()).contains("if (true) {}}\n");
70      }
71    }

73    public void testSyntaxError2() {
74      try {
75        extractMessage("", "if (true) {}}");
76        fail("Expected exception");
77      } catch (RuntimeException e) {
78        assertTrue(
79            e.getMessage(),
80            e.getMessage().contains("JSCompiler errors\n"));
81        assertTrue(
82            e.getMessage(),
83            e.getMessage().contains(
84                "testcode:2: ERROR - Parse error"));
85        assertTrue(
86            e.getMessage(),
87            e.getMessage().contains("if (true) {}}\n"));
88      }
89    }

163   public void testExtractMixedMessages() {
164     // Several mixed-style messages in succession, one containing newlines.
165     Iterator<JsMessage> msgs = extractMessages(
166         "var MSG_MONEY = function(amount) {",
167         "  return 'You owe $' + amount +",
168         "         ' to the credit card company.';",
169         "};",
170         "var MSG_TIME = goog.getMsg('You need to finish your work in ' +",
171         "                           '{$duration} hours.', {'duration': d});",
172         "var MSG_NAG = 'Clean your room.\\n\\nWash your clothes.';",
173         "var MSG_NAG_HELP = 'Just some ' +",
174         "                   'nags.';").iterator();
175 
176     assertEquals(
177         new JsMessage.Builder("MSG_MONEY")
178             .appendStringPart("You owe $")
179             .appendPlaceholderReference("amount")
180             .appendStringPart(" to the credit card company.")
181             .build(),
182         msgs.next());
183     assertEquals(
184         new JsMessage.Builder("MSG_TIME")
185             .appendStringPart("You need to finish your work in ")
186             .appendPlaceholderReference("duration")
187             .appendStringPart(" hours.")
188             .build(),
189         msgs.next());
190     assertEquals(
191         new JsMessage.Builder("MSG_NAG")
192             .appendStringPart("Clean your room.\n\nWash your clothes.")
193             .setDesc("Just some nags.")
194             .build(),
195         msgs.next());
196   }

198   public void testDuplicateUnnamedVariables() {
199     // Make sure that duplicate unnamed variables don't get swallowed when using
200     // a Google-specific ID generator.
201     Collection<JsMessage> msgs = extractMessages(
202         "function a() {",
203         "  var MSG_UNNAMED_2 = goog.getMsg('foo');",
204         "}",
205         "function b() {",
206         "  var MSG_UNNAMED_2 = goog.getMsg('bar');",
207         "}");
208 
209     assertThat(msgs).hasSize(2);
210     final Iterator<JsMessage> iter = msgs.iterator();
211     assertEquals("foo", iter.next().toString());
212     assertEquals("bar", iter.next().toString());
213   }

215   public void testMeaningAnnotation() {
216     List<JsMessage> msgs = new ArrayList<>(
217         extractMessages(
218             "var MSG_UNNAMED_1 = goog.getMsg('foo');",
219             "var MSG_UNNAMED_2 = goog.getMsg('foo');"));
220     assertThat(msgs).hasSize(2);
221     assertEquals(msgs.get(1).getId(), msgs.get(0).getId());
222     assertEquals(msgs.get(0), msgs.get(1));
223 
224     msgs = new ArrayList<>(
225         extractMessages(
226             "var MSG_UNNAMED_1 = goog.getMsg('foo');",
227             "/** @meaning bar */ var MSG_UNNAMED_2 = goog.getMsg('foo');"));
228     assertThat(msgs).hasSize(2);
229     assertFalse(msgs.get(0).getId().equals(msgs.get(1).getId()));
230   }

232   private void assertEquals(JsMessage expected, JsMessage actual) {
233     assertEquals(expected.getId(), actual.getId());
234     assertEquals(expected.getKey(), actual.getKey());
235     assertEquals(expected.parts(), actual.parts());
236     assertEquals(expected.placeholders(), actual.placeholders());
237     assertEquals(expected.getDesc(), actual.getDesc());
238     assertEquals(expected.isHidden(), actual.isHidden());
239     assertEquals(expected.getMeaning(), actual.getMeaning());
240   }
