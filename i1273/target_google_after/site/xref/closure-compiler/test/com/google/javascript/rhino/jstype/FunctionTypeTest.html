59    public void testSupAndInfOfReturnTypes() {
60      FunctionType retString = new FunctionBuilder(registry)
61          .withParamsNode(registry.createParameters())
62          .withInferredReturnType(STRING_TYPE).build();
63      FunctionType retNumber = new FunctionBuilder(registry)
64          .withParamsNode(registry.createParameters())
65          .withReturnType(NUMBER_TYPE).build();
66  
67      assertLeastSupertype(
68          "function (): (number|string)", retString, retNumber);
69      assertGreatestSubtype(
70          "function (): None", retString, retNumber);
71  
72      assertTrue(retString.isReturnTypeInferred());
73      assertFalse(retNumber.isReturnTypeInferred());
74      assertTrue(
75          ((FunctionType) retString.getLeastSupertype(retNumber))
76          .isReturnTypeInferred());
77      assertTrue(
78          ((FunctionType) retString.getGreatestSubtype(retString))
79          .isReturnTypeInferred());
80    }

178     FunctionType ctor = registry.createConstructorType(
179         "Foo", null, null, null, null);
180     assertFalse(ctor.getInstanceType().isUnknownType());
181

183     ctor.defineDeclaredProperty("prototype", UNKNOWN_TYPE, node);
184     assertTrue(ctor.getInstanceType().isUnknownType());
185

187         ctor.getOwnPropertyNames());

210   public void testInterfacePrototypeChain2() {
211     FunctionType iface = registry.createInterfaceType("I", null,
212         ImmutableList.<TemplateType>of());
213     iface.getPrototype().defineDeclaredProperty(
214         "numberProp", NUMBER_TYPE, null);
215 
216     FunctionType subIface = registry.createInterfaceType("SubI", null,
217         ImmutableList.<TemplateType>of());
218     subIface.setExtendedInterfaces(
219         Lists.<ObjectType>newArrayList(iface.getInstanceType()));
220     assertTypeEquals(
221         subIface.getPrototype(),
222         subIface.getInstanceType().getImplicitPrototype());
223     assertTypeEquals(
224         OBJECT_TYPE,
225         subIface.getPrototype().getImplicitPrototype());
226 
227     ObjectType subIfaceInst = subIface.getInstanceType();
228     assertTrue(subIfaceInst.hasProperty("numberProp"));
229     assertTrue(subIfaceInst.isPropertyTypeDeclared("numberProp"));
230     assertFalse(subIfaceInst.isPropertyTypeInferred("numberProp"));
231   }

274   public void testIsEquivalentToParams() {
275     FunctionType oneNum = new FunctionBuilder(registry)
276         .withParamsNode(registry.createParameters(NUMBER_TYPE))
277         .build();
278     FunctionType optNum = new FunctionBuilder(registry)
279         .withParamsNode(registry.createOptionalParameters(NUMBER_TYPE))
280         .build();
281     FunctionType varNum = new FunctionBuilder(registry)
282         .withParamsNode(registry.createParametersWithVarArgs(NUMBER_TYPE))
283         .build();
284     Asserts.assertEquivalenceOperations(oneNum, oneNum);
285     Asserts.assertEquivalenceOperations(optNum, optNum);
286     Asserts.assertEquivalenceOperations(varNum, varNum);
287     assertFalse(oneNum.isEquivalentTo(optNum));
288     assertFalse(oneNum.isEquivalentTo(varNum));
289     assertFalse(optNum.isEquivalentTo(varNum));
290   }

292   public void testIsEquivalentOptAndVarArgs() {
293     FunctionType varNum = new FunctionBuilder(registry)
294         .withParamsNode(registry.createParametersWithVarArgs(NUMBER_TYPE))
295         .build();
296 
297     FunctionParamBuilder builder = new FunctionParamBuilder(registry);
298     builder.addOptionalParams(NUMBER_TYPE);
299     builder.addVarArgs(NUMBER_TYPE);
300     FunctionType optAndVarNum = new FunctionBuilder(registry)
301         .withParamsNode(builder.build())
302         .build();
303 
304     // We currently do not consider function(T=, ...T) and function(...T)
305     // equivalent. This may change.
306     assertFalse(varNum.isEquivalentTo(optAndVarNum));
307     assertFalse(optAndVarNum.isEquivalentTo(varNum));
308   }

313         .withParamsNode(registry.createParameters(loop))
314         .withReturnType(loop).build();
