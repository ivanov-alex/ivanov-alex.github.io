54    /**
55     * Assert that a type can assign to itself.
56     */
57    private void assertTypeCanAssignToItself(JSType type) {
58      assertTrue(type.isSubtype(type));
59    }

61    /**
62     * Tests the behavior of variants type.
63     */
64    @SuppressWarnings("checked")
65    public void testUnionType() throws Exception {
66      UnionType nullOrString =
67          (UnionType) createUnionType(NULL_TYPE, STRING_OBJECT_TYPE);
68      UnionType stringOrNull =
69          (UnionType) createUnionType(STRING_OBJECT_TYPE, NULL_TYPE);
70  
71      Asserts.assertTypeEquals(nullOrString, stringOrNull);
72      Asserts.assertTypeEquals(stringOrNull, nullOrString);
73  
74      assertTypeCanAssignToItself(createUnionType(VOID_TYPE, NUMBER_TYPE));
75      assertTypeCanAssignToItself(
76          createUnionType(NUMBER_TYPE, STRING_TYPE, OBJECT_TYPE));
77      assertTypeCanAssignToItself(createUnionType(NUMBER_TYPE, BOOLEAN_TYPE));
78      assertTypeCanAssignToItself(createUnionType(VOID_TYPE));
79  
80      UnionType nullOrUnknown =
81          (UnionType) createUnionType(NULL_TYPE, unresolvedNamedType);
82      assertTrue(nullOrUnknown.isUnknownType());
83      Asserts.assertTypeEquals(nullOrUnknown, NULL_TYPE.getLeastSupertype(nullOrUnknown));
84      Asserts.assertTypeEquals(nullOrUnknown, nullOrUnknown.getLeastSupertype(NULL_TYPE));
85      Asserts.assertTypeEquals(UNKNOWN_TYPE,
86          NULL_TYPE.getGreatestSubtype(nullOrUnknown));
87      Asserts.assertTypeEquals(UNKNOWN_TYPE,
88          nullOrUnknown.getGreatestSubtype(NULL_TYPE));
89  
90      assertTrue(NULL_TYPE.differsFrom(nullOrUnknown));
91      assertTrue(nullOrUnknown.differsFrom(NULL_TYPE));
92      assertFalse(nullOrUnknown.differsFrom(unresolvedNamedType));
93  
94      assertTrue(NULL_TYPE.isSubtype(nullOrUnknown));
95      assertTrue(unresolvedNamedType.isSubtype(nullOrUnknown));
96      assertTrue(nullOrUnknown.isSubtype(NULL_TYPE));
97  
98      Asserts.assertTypeEquals(unresolvedNamedType,
99          nullOrUnknown.restrictByNotNullOrUndefined());
100 
101     // findPropertyType
102     Asserts.assertTypeEquals(NUMBER_TYPE, nullOrString.findPropertyType("length"));
103     assertEquals(null, nullOrString.findPropertyType("lengthx"));
104 
105     Asserts.assertResolvesToSame(nullOrString);
106   }

163   /**
164    * Tests subtyping of union types.
165    */
166   public void testSubtypingUnionTypes() throws Exception {
167     // subtypes
168     assertTrue(BOOLEAN_TYPE.
169         isSubtype(createUnionType(BOOLEAN_TYPE, STRING_TYPE)));
170     assertTrue(createUnionType(BOOLEAN_TYPE, STRING_TYPE).
171         isSubtype(createUnionType(BOOLEAN_TYPE, STRING_TYPE)));
172     assertTrue(createUnionType(BOOLEAN_TYPE, STRING_TYPE).
173         isSubtype(createUnionType(BOOLEAN_TYPE, STRING_TYPE, NULL_TYPE)));
174     assertTrue(createUnionType(BOOLEAN_TYPE, STRING_TYPE).
175         isSubtype(createUnionType(BOOLEAN_TYPE, STRING_TYPE, NULL_TYPE)));
176     assertTrue(createUnionType(BOOLEAN_TYPE).
177         isSubtype(createUnionType(BOOLEAN_TYPE, STRING_TYPE, NULL_TYPE)));
178     assertTrue(createUnionType(STRING_TYPE).
179         isSubtype(createUnionType(BOOLEAN_TYPE, STRING_TYPE, NULL_TYPE)));
180     assertTrue(createUnionType(STRING_TYPE, NULL_TYPE).isSubtype(ALL_TYPE));
181     assertTrue(createUnionType(DATE_TYPE, REGEXP_TYPE).isSubtype(OBJECT_TYPE));
182     assertTrue(createUnionType(URI_ERROR_TYPE, EVAL_ERROR_TYPE).
183         isSubtype(ERROR_TYPE));
184     assertTrue(createUnionType(URI_ERROR_TYPE, EVAL_ERROR_TYPE).
185         isSubtype(OBJECT_TYPE));
186 
187     // not subtypes
188     assertFalse(createUnionType(STRING_TYPE, NULL_TYPE).isSubtype(NO_TYPE));
189     assertFalse(createUnionType(STRING_TYPE, NULL_TYPE).
190         isSubtype(NO_OBJECT_TYPE));
191     assertFalse(createUnionType(NO_OBJECT_TYPE, NULL_TYPE).
192         isSubtype(OBJECT_TYPE));
193 
194     // defined unions
195     assertTrue(NUMBER_TYPE.isSubtype(OBJECT_NUMBER_STRING));
196     assertTrue(OBJECT_TYPE.isSubtype(OBJECT_NUMBER_STRING));
197     assertTrue(STRING_TYPE.isSubtype(OBJECT_NUMBER_STRING));
198     assertTrue(NO_OBJECT_TYPE.isSubtype(OBJECT_NUMBER_STRING));
199 
200     assertTrue(NUMBER_TYPE.isSubtype(NUMBER_STRING_BOOLEAN));
201     assertTrue(BOOLEAN_TYPE.isSubtype(NUMBER_STRING_BOOLEAN));
202     assertTrue(STRING_TYPE.isSubtype(NUMBER_STRING_BOOLEAN));
203 
204     assertTrue(NUMBER_TYPE.isSubtype(OBJECT_NUMBER_STRING_BOOLEAN));
205     assertTrue(OBJECT_TYPE.isSubtype(OBJECT_NUMBER_STRING_BOOLEAN));
206     assertTrue(STRING_TYPE.isSubtype(OBJECT_NUMBER_STRING_BOOLEAN));
207     assertTrue(BOOLEAN_TYPE.isSubtype(OBJECT_NUMBER_STRING_BOOLEAN));
208     assertTrue(NO_OBJECT_TYPE.isSubtype(OBJECT_NUMBER_STRING_BOOLEAN));
209   }

211   /**
212    * Tests that special union types can assign to other types.  Unions
213    * containing the unknown type should be able to assign to any other
214    * type.
215    */
216   @SuppressWarnings("checked")
217   public void testSpecialUnionCanAssignTo() throws Exception {
218     // autoboxing quirks
219     UnionType numbers =
220         (UnionType) createUnionType(NUMBER_TYPE, NUMBER_OBJECT_TYPE);
221     assertFalse(numbers.isSubtype(NUMBER_TYPE));
222     assertFalse(numbers.isSubtype(NUMBER_OBJECT_TYPE));
223     assertFalse(numbers.isSubtype(EVAL_ERROR_TYPE));
224 
225     UnionType strings =
226         (UnionType) createUnionType(STRING_OBJECT_TYPE, STRING_TYPE);
227     assertFalse(strings.isSubtype(STRING_TYPE));
228     assertFalse(strings.isSubtype(STRING_OBJECT_TYPE));
229     assertFalse(strings.isSubtype(DATE_TYPE));
230 
231     UnionType booleans =
232         (UnionType) createUnionType(BOOLEAN_OBJECT_TYPE, BOOLEAN_TYPE);
233     assertFalse(booleans.isSubtype(BOOLEAN_TYPE));
234     assertFalse(booleans.isSubtype(BOOLEAN_OBJECT_TYPE));
235     assertFalse(booleans.isSubtype(REGEXP_TYPE));
236 
237     // unknown quirks
238     JSType unknown = createUnionType(UNKNOWN_TYPE, DATE_TYPE);
239     assertTrue(unknown.isSubtype(STRING_TYPE));
240 
241     // all members need to be assignable to
242     UnionType stringDate =
243         (UnionType) createUnionType(STRING_OBJECT_TYPE, DATE_TYPE);
244     assertTrue(stringDate.isSubtype(OBJECT_TYPE));
245     assertFalse(stringDate.isSubtype(STRING_OBJECT_TYPE));
246     assertFalse(stringDate.isSubtype(DATE_TYPE));
247   }

292   public void testProxyUnionType() throws Exception {
293     UnionType stringOrNumber =
294         (UnionType) createUnionType(NUMBER_TYPE, STRING_TYPE);
295     UnionType stringOrBoolean =
296         (UnionType) createUnionType(BOOLEAN_TYPE, STRING_TYPE);
297 
298     assertEquals(
299         "(boolean|number|string)",
300         stringOrNumber.getLeastSupertype(stringOrBoolean).toString());
301     assertEquals(
302         "string",
303         stringOrNumber.getGreatestSubtype(stringOrBoolean).toString());
304     assertEquals(
305         TernaryValue.UNKNOWN,
306         stringOrNumber.testForEquality(stringOrBoolean));
307     assertEquals(
308         "(number|string)",
309         stringOrNumber.getTypesUnderEquality(
310             stringOrBoolean).typeA.toString());
311     assertEquals(
312         "string",
313         stringOrNumber.getTypesUnderShallowEquality(
314             stringOrBoolean).typeA.toString());
315     assertEquals(
316         "(number|string)",
317         stringOrNumber.getTypesUnderInequality(
318             stringOrBoolean).typeA.toString());
319     assertEquals(
320         "(number|string)",
321         stringOrNumber.getTypesUnderShallowInequality(
322             stringOrBoolean).typeA.toString());
323 
324     ObjectType stringOrNumberProxy =
325         new ProxyObjectType(registry, stringOrNumber);
326     ObjectType stringOrBooleanProxy =
327         new ProxyObjectType(registry, stringOrBoolean);
328     assertEquals(
329         "(boolean|number|string)",
330         stringOrNumberProxy.getLeastSupertype(
331             stringOrBooleanProxy).toString());
332     assertEquals(
333         "string",
334         stringOrNumberProxy.getGreatestSubtype(
335             stringOrBooleanProxy).toString());
336     assertEquals(
337         TernaryValue.UNKNOWN,
338         stringOrNumberProxy.testForEquality(stringOrBooleanProxy));
339     assertEquals(
340         "(number|string)",
341         stringOrNumberProxy.getTypesUnderEquality(
342             stringOrBooleanProxy).typeA.toString());
343     assertEquals(
344         "string",
345         stringOrNumberProxy.getTypesUnderShallowEquality(
346             stringOrBooleanProxy).typeA.toString());
347     assertEquals(
348         "(number|string)",
349         stringOrNumberProxy.getTypesUnderInequality(
350             stringOrBooleanProxy).typeA.toString());
351     assertEquals(
352         "(number|string)",
353         stringOrNumberProxy.getTypesUnderShallowInequality(
354             stringOrBooleanProxy).typeA.toString());
355   }
