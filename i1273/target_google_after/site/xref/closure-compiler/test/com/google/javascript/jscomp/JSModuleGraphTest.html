286   private void testManageDependencies5Impl(boolean es6ModuleOrder) throws Exception {
287     A.add(code("a2", provides("a2"), requires("a1")));
288     A.add(code("a1", provides("a1"), requires()));
289     A.add(code("base.js", BASEJS, provides(), requires()));
290 
291     for (CompilerInput input : A.getInputs()) {
292       input.setCompiler(compiler);
293     }
294 
295     DependencyOptions depOptions = new DependencyOptions();
296     depOptions.setDependencySorting(true);
297     depOptions.setEs6ModuleOrder(es6ModuleOrder);
298 
299     List<CompilerInput> inputs = new ArrayList<>();
300     inputs.addAll(A.getInputs());
301     List<CompilerInput> results = graph.manageDependencies(
302         depOptions, inputs);
303 
304     assertInputs(A, "base.js", "a1", "a2");
305 
306     assertThat(sourceNames(results)).containsExactly("base.js", "a1", "a2").inOrder();
307   }

337   private List<CompilerInput> setUpManageDependenciesTest() {
338     List<CompilerInput> inputs = new ArrayList<>();
339 
340     A.add(code("a1", provides("a1"), requires()));
341     A.add(code("a2", provides("a2"), requires("a1")));
342     A.add(code("a3", provides(), requires("a1")));

378   private SourceFile code(
379       String sourceName, List<String> provides, List<String> requires) {
380     return code(sourceName, "", provides, requires);
381   }
382 
383   private SourceFile code(
384       String sourceName, String source,
385       List<String> provides, List<String> requires) {
386     String text = "";
387     for (String p : provides) {
388       text += "goog.provide('" + p + "');\n";
389     }
390     for (String r : requires) {
391       text += "goog.require('" + r + "');\n";
392     }
393     return SourceFile.fromCode(sourceName, text + source);
394   }

404   private void assertDeepestCommonDepInclusive(
405       JSModule expected, JSModule m1, JSModule m2) {
406     assertDeepestCommonDepOneWay(expected, m1, m2, true);
407     assertDeepestCommonDepOneWay(expected, m2, m1, true);
408   }

410   private void assertDeepestCommonDep(
411       JSModule expected, JSModule m1, JSModule m2) {
412     assertDeepestCommonDepOneWay(expected, m1, m2, false);
413     assertDeepestCommonDepOneWay(expected, m2, m1, false);
414   }
415 
416   private void assertDeepestCommonDepOneWay(
417       JSModule expected, JSModule m1, JSModule m2, boolean inclusive) {
418     JSModule actual = inclusive ?
419         graph.getDeepestCommonDependencyInclusive(m1, m2) :
420         graph.getDeepestCommonDependency(m1, m2);
421     if (actual != expected) {
422       fail(String.format(
423           "Deepest common dep of %s and %s should be %s but was %s",
424           m1.getName(), m2.getName(),
425           expected == null ? "null" : expected.getName(),
426           actual ==  null ? "null" : actual.getName()));
427     }
428   }

430   private void assertTransitiveDepsDeepestFirst(JSModule m, JSModule... deps) {
431     Iterable<JSModule> actual = graph.getTransitiveDepsDeepestFirst(m);
432     assertEquals(Arrays.toString(deps),
433                  Arrays.toString(Iterables.toArray(actual, JSModule.class)));
434   }
