35    public void testStraightLine() {
36      // A sample of simple straight line of code with different liveness changes.
37      assertNotLiveBeforeX("X:var a;", "a");
38      assertNotLiveAfterX("X:var a;", "a");
39      assertNotLiveAfterX("X:var a=1;", "a");
40      assertLiveAfterX("X:var a=1; a()", "a");
41      assertNotLiveBeforeX("X:var a=1; a()", "a");
42      assertLiveBeforeX("var a;X:a;", "a");
43      assertLiveBeforeX("var a;X:a=a+1;", "a");
44      assertLiveBeforeX("var a;X:a+=1;", "a");
45      assertLiveBeforeX("var a;X:a++;", "a");
46      assertNotLiveAfterX("var a,b;X:b();", "a");
47      assertNotLiveBeforeX("var a,b;X:b();", "a");
48      assertLiveBeforeX("var a,b;X:b(a);", "a");
49      assertLiveBeforeX("var a,b;X:b(1,2,3,b(a + 1));", "a");
50      assertNotLiveBeforeX("var a,b;X:a=1;b(a)", "a");
51      assertNotLiveAfterX("var a,b;X:b(a);b()", "a");
52      assertLiveBeforeX("var a,b;X:b();b=1;a()", "b");
53      assertLiveAfterX("X:a();var a;a()", "a");
54      assertNotLiveAfterX("X:a();var a=1;a()", "a");
55      assertLiveBeforeX("var a,b;X:a,b=1", "a");
56    }

58    public void testProperties() {
59      // Reading property of a local variable makes that variable live.
60      assertLiveBeforeX("var a,b;X:a.P;", "a");
61  
62      // Assigning to a property doesn't kill "a". It makes it live instead.
63      assertLiveBeforeX("var a,b;X:a.P=1;b()", "a");
64      assertLiveBeforeX("var a,b;X:a.P.Q=1;b()", "a");
65  
66      // An "a" in a different context.
67      assertNotLiveAfterX("var a,b;X:b.P.Q.a=1;", "a");
68  
69      assertLiveBeforeX("var a,b;X:b.P.Q=a;", "a");
70    }

72    public void testConditions() {
73      // Reading the condition makes the variable live.
74      assertLiveBeforeX("var a,b;X:if(a){}", "a");
75      assertLiveBeforeX("var a,b;X:if(a||b) {}", "a");
76      assertLiveBeforeX("var a,b;X:if(b||a) {}", "a");
77      assertLiveBeforeX("var a,b;X:if(b||b(a)) {}", "a");
78      assertNotLiveAfterX("var a,b;X:b();if(a) {}", "b");
79  
80      // We can kill within a condition as well.
81      assertNotLiveAfterX("var a,b;X:a();if(a=b){}a()", "a");
82      assertNotLiveAfterX("var a,b;X:a();while(a=b){}a()", "a");
83  
84      // The kill can be "conditional" due to short circuit.
85      assertNotLiveAfterX("var a,b;X:a();if((a=b)&&b){}a()", "a");
86      assertNotLiveAfterX("var a,b;X:a();while((a=b)&&b){}a()", "a");
87      assertLiveBeforeX("var a,b;a();X:if(b&&(a=b)){}a()", "a"); // Assumed live.
88      assertLiveBeforeX("var a,b;a();X:if(a&&(a=b)){}a()", "a");
89      assertLiveBeforeX("var a,b;a();X:while(b&&(a=b)){}a()", "a");
90      assertLiveBeforeX("var a,b;a();X:while(a&&(a=b)){}a()", "a");
91    }

104   public void testTwoPaths() {
105     // Both Paths.
106     assertLiveBeforeX("var a,b;X:if(b){b(a)}else{b(a)};", "a");
107 
108     // Only one path.
109     assertLiveBeforeX("var a,b;X:if(b){b(b)}else{b(a)};", "a");
110     assertLiveBeforeX("var a,b;X:if(b){b(a)}else{b(b)};", "a");
111 
112     // None of the paths.
113     assertNotLiveAfterX("var a,b;X:if(b){b(b)}else{b(b)};", "a");
114 
115     // At the very end.
116     assertLiveBeforeX("var a,b;X:if(b){b(b)}else{b(b)}a();", "a");
117 
118     // The loop might or might not be executed.
119     assertLiveBeforeX("var a;X:while(param1){a()};", "a");
120     assertLiveBeforeX("var a;X:while(param1){a=1};a()", "a");
121 
122     // Same idea with if.
123     assertLiveBeforeX("var a;X:if(param1){a()};", "a");
124     assertLiveBeforeX("var a;X:if(param1){a=1};a()", "a");
125 
126     // This is different in DO. We know for sure at least one iteration is
127     // executed.
128     assertNotLiveAfterX("X:var a;do{a=1}while(param1);a()", "a");
129   }

147   public void testForLoops() {
148     // Induction variable should not be live after the loop.
149     assertNotLiveBeforeX("var a,b;for(a=0;a<9;a++){b(a)};X:b", "a");
150     assertNotLiveBeforeX("var a,b;for(a in b){a()};X:b", "a");
151     assertNotLiveBeforeX("var a,b;for(a in b){a()};X:a", "b");
152     assertLiveBeforeX("var b;for(var a in b){X:a()};", "a");
153 
154     // It should be live within the loop even if it is not used.
155     assertLiveBeforeX("var a,b;for(a=0;a<9;a++){X:1}", "a");
156     assertLiveAfterX("var a,b;for(a in b){X:b};", "a");
157     // For-In should serve as a gen as well.
158     assertLiveBeforeX("var a,b; X:for(a in b){ }", "a");
159 
160     // "a in b" should kill "a" before it.
161     // Can't prove this unless we have branched backward DFA.
162     //assertNotLiveAfterX("var a,b;X:b;for(a in b){a()};", "a");
163 
164     // Unless it is used before.
165     assertLiveBeforeX("var a,b;X:a();b();for(a in b){a()};", "a");
166 
167     // Initializer
168     assertLiveBeforeX("var a,b;X:b;for(b=a;;){};", "a");
169     assertNotLiveBeforeX("var a,b;X:a;for(b=a;;){b()};b();", "b");
170   }

285   private void assertLiveBeforeX(String src, String var) {
286     FlowState<LiveVariablesAnalysis.LiveVariableLattice> state =
287         getFlowStateAtX(src);
288     assertNotNull(src + " should contain a label 'X:'", state);
289     assertTrue("Variable" + var + " should be live before X",
290         state.getIn().isLive(liveness.getVarIndex(var)));
291   }
292 
293   private void assertLiveAfterX(String src, String var) {
294     FlowState<LiveVariablesAnalysis.LiveVariableLattice> state =
295         getFlowStateAtX(src);
296     assertNotNull("Label X should be in the input program.", state);
297     assertTrue("Variable" + var + " should be live after X",
298         state.getOut().isLive(liveness.getVarIndex(var)));
299   }
300 
301   private void assertNotLiveAfterX(String src, String var) {
302     FlowState<LiveVariablesAnalysis.LiveVariableLattice> state =
303         getFlowStateAtX(src);
304     assertNotNull("Label X should be in the input program.", state);
305     assertFalse("Variable" + var + " should not be live after X",
306         state.getOut().isLive(liveness.getVarIndex(var)));
307   }
308 
309   private void assertNotLiveBeforeX(String src, String var) {
310     FlowState<LiveVariablesAnalysis.LiveVariableLattice> state =
311         getFlowStateAtX(src);
312     assertNotNull("Label X should be in the input program.", state);
313     assertFalse("Variable" + var + " should not be live before X",
314         state.getIn().isLive(liveness.getVarIndex(var)));
315   }
316 
317   private FlowState<LiveVariablesAnalysis.LiveVariableLattice> getFlowStateAtX(
318       String src) {
319     liveness = computeLiveness(src);
320     return getFlowStateAtX(liveness.getCfg().getEntry().getValue(), liveness
321         .getCfg());
322   }
323 
324   private FlowState<LiveVariablesAnalysis.LiveVariableLattice> getFlowStateAtX(
325       Node node, ControlFlowGraph<Node> cfg) {
326     if (node.isLabel()) {
327       if (node.getFirstChild().getString().equals("X")) {
328         return cfg.getNode(node.getLastChild()).getAnnotation();
329       }
330     }
331     for (Node c = node.getFirstChild(); c != null; c = c.getNext()) {
332       FlowState<LiveVariablesAnalysis.LiveVariableLattice> state =
333           getFlowStateAtX(c, cfg);
334       if (state != null) {
335         return state;
336       }
337     }
338     return null;
339   }

341   private static void assertEscaped(String src, String name) {
342     for (Var var : computeLiveness(src).getEscapedLocals()) {
343       if (var.name.equals(name)) {
344         return;
345       }
346     }
347     fail("Variable " + name + " should be in the escaped local list.");
348   }
349 
350   private static void assertNotEscaped(String src, String name) {
351     for (Var var : computeLiveness(src).getEscapedLocals()) {
352       assertFalse(var.name.equals(name));
353     }
354   }

356   private static LiveVariablesAnalysis computeLiveness(String src) {
357     Compiler compiler = new Compiler();
358     CompilerOptions options = new CompilerOptions();
359     options.setCodingConvention(new GoogleCodingConvention());
360     compiler.initOptions(options);
361     src = "function _FUNCTION(param1, param2){" + src + "}";
362     Node n = compiler.parseTestCode(src).removeFirstChild();
363     Node script = new Node(Token.SCRIPT, n);
364     script.setInputId(new InputId("test"));
365     assertEquals(0, compiler.getErrorCount());
366     Scope scope = SyntacticScopeCreator.makeUntyped(compiler)
367         .createScope(n, Scope.createGlobalScope(script));
368     ControlFlowAnalysis cfa = new ControlFlowAnalysis(compiler, false, true);
369     cfa.process(null, n);
370     ControlFlowGraph<Node> cfg = cfa.getCfg();
371     LiveVariablesAnalysis analysis =
372         new LiveVariablesAnalysis(cfg, scope, compiler);
373     analysis.analyze();
374     return analysis;
375   }
