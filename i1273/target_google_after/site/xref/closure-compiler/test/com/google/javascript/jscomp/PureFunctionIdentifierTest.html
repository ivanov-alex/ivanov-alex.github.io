536   public void testNoSideEffectsSimple() throws Exception {
537     String prefix = "function f(){";
538     String suffix = "} f()";
539     List<String> expected = ImmutableList.of("f");
540 
541     checkMarkedCalls(
542         prefix + "" + suffix, expected);
543     checkMarkedCalls(
544         prefix + "return 1" + suffix, expected);
545     checkMarkedCalls(
546         prefix + "return 1 + 2" + suffix, expected);
547 
548     // local var
549     checkMarkedCalls(
550         prefix + "var a = 1; return a" + suffix, expected);
551 
552     // mutate local var
553     checkMarkedCalls(
554         prefix + "var a = 1; a = 2; return a" + suffix, expected);
555     checkMarkedCalls(
556         prefix + "var a = 1; a = 2; return a + 1" + suffix, expected);
557 
558     // read from obj literal
559     checkMarkedCalls(
560         prefix + "var a = {foo : 1}; return a.foo" + suffix, expected);
561     checkMarkedCalls(
562         prefix + "var a = {foo : 1}; return a.foo + 1" + suffix, expected);
563 
564     // read from extern
565     checkMarkedCalls(
566         prefix + "return externObj" + suffix, expected);
567     checkMarkedCalls(
568         "function g(x) { x.foo = 3; }" /* to suppress missing property */ +
569         prefix + "return externObj.foo" + suffix, expected);
570   }

572   public void testResultLocalitySimple() throws Exception {
573     String prefix = "var g; function f(){";
574     String suffix = "} f()";
575     List<String> expected = ImmutableList.of("f");
576     List<String> notExpected = ImmutableList.of();
577 
578     // no return
579     checkLocalityOfMarkedCalls(
580         prefix + "" + suffix, expected);
581     // simple return expressions
582     checkLocalityOfMarkedCalls(
583         prefix + "return 1" + suffix, expected);
584     checkLocalityOfMarkedCalls(
585         prefix + "return 1 + 2" + suffix, expected);
586 
587     // global result
588     checkLocalityOfMarkedCalls(
589         prefix + "return g" + suffix, notExpected);
590 
591     // multiple returns
592     checkLocalityOfMarkedCalls(
593         prefix + "return 1; return 2" + suffix, expected);
594     checkLocalityOfMarkedCalls(
595         prefix + "return 1; return g" + suffix, notExpected);
596 
597     // local var, not yet.
598     checkLocalityOfMarkedCalls(
599         prefix + "var a = 1; return a" + suffix, notExpected);
600 
601     // mutate local var, not yet.
602     checkLocalityOfMarkedCalls(
603         prefix + "var a = 1; a = 2; return a" + suffix, notExpected);
604     checkLocalityOfMarkedCalls(
605         prefix + "var a = 1; a = 2; return a + 1" + suffix, expected);
606 
607     // read from obj literal
608     checkLocalityOfMarkedCalls(
609         prefix + "return {foo : 1}.foo" + suffix,
610         notExpected);
611     checkLocalityOfMarkedCalls(
612         prefix + "var a = {foo : 1}; return a.foo" + suffix,
613         notExpected);
614 
615     // read from extern
616     checkLocalityOfMarkedCalls(
617         prefix + "return externObj" + suffix, notExpected);
618     checkLocalityOfMarkedCalls(
619         "function inner(x) { x.foo = 3; }" /* to suppress missing property */ +
620         prefix + "return externObj.foo" + suffix, notExpected);
621   }

1308   /**
1309    * Run PureFunctionIdentifier, then gather a list of calls that are
1310    * marked as having no side effects.
1311    */
1312   private class NoSideEffectCallEnumerator
1313       extends AbstractPostOrderCallback implements CompilerPass {
1314     private final Compiler compiler;
1315 
1316     NoSideEffectCallEnumerator(Compiler compiler) {
1317       this.compiler = compiler;
1318     }
1319 
1320     @Override
1321     public void process(Node externs, Node root) {
1322       compiler.setHasRegExpGlobalReferences(regExpHaveSideEffects);
1323       SimpleDefinitionFinder defFinder = new SimpleDefinitionFinder(compiler);
1324       defFinder.process(externs, root);
1325       PureFunctionIdentifier passUnderTest =
1326           new PureFunctionIdentifier(compiler, defFinder);
1327       passUnderTest.process(externs, root);
1328 
1329       // Ensure that debug report computation doesn't crash.
1330       passUnderTest.getDebugReport();
1331 
1332       NodeTraversal.traverseEs6(compiler, externs, this);
1333       NodeTraversal.traverseEs6(compiler, root, this);
1334     }
1335 
1336     @Override
1337     public void visit(NodeTraversal t, Node n, Node parent) {
1338       if (n.isNew()) {
1339         if (!NodeUtil.constructorCallHasSideEffects(n)) {
1340           noSideEffectCalls.add(generateNameString(n.getFirstChild()));
1341         }
1342       } else if (n.isCall()) {
1343         if (!NodeUtil.functionCallHasSideEffects(n)) {
1344           noSideEffectCalls.add(generateNameString(n.getFirstChild()));
1345         }
1346         if (NodeUtil.callHasLocalResult(n)) {
1347           localResultCalls.add(generateNameString(n.getFirstChild()));
1348         }
1349       }
1350     }
1351 
1352     private String generateNameString(Node node) {
1353       if (node.isOr()) {
1354         return "(" + generateNameString(node.getFirstChild()) +
1355             " || " + generateNameString(node.getLastChild()) + ")";
1356       } else if (node.isHook()) {
1357         return "(" + generateNameString(node.getFirstChild().getNext()) +
1358             " : " + generateNameString(node.getLastChild()) + ")";
1359       } else {
1360         String result = node.getQualifiedName();
1361         if (result == null) {
1362           if (node.isFunction()) {
1363             result = node.toString(false, false, false).trim();
1364           } else {
1365             result = node.getFirstChild().toString(false, false, false);
1366             result += " " + node.getLastChild().toString(false, false, false);
1367           }
1368         }
1369         return result;
1370       }
1371     }
