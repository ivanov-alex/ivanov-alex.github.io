50    public void testVarDeclarations() {
51      test("var a = 1;", "_.a = 1;");
52      test("var a = 1, b = 2, c = 3;", "_.a = 1; _.b = 2; _.c = 3;");
53      test(
54          "var a = 'str', b = 1, c = { foo: 'bar' }, d = function() {};",
55          "_.a = 'str'; _.b = 1; _.c = { foo: 'bar' }; _.d = function() {};");
56      test("if(1){var x = 1;}", "if(1){_.x = 1;}");
57      test("var x;", "");
58      test("var a, b = 1;", "_.b = 1");
59    }

77    public void testVarDeclarations_acrossModules() {
78      assumeCrossModuleNames = false;
79      test(createModules(
80          "var a = 1;", "a"),
81          new String[] {"_.a = 1", "_.a"});
82      test(createModules(
83          "var a = 1, b = 2, c = 3;", "a;c;"),
84          new String[] {"var b;_.a = 1; b = 2; _.c = 3;", "_.a;_.c"});
85      test(createModules(
86          "var a = 1, b = 2, c = 3;", "b;c;"),
87          new String[] {"var a;a = 1; _.b = 2; _.c = 3;", "_.b;_.c"});
88      test(createModules(
89          "var a = 1, b = 2, c = 3;b;c;", "a;c;"),
90          new String[] {"var b;_.a = 1; b = 2; _.c = 3;b;_.c", "_.a;_.c"});
91      test(createModules(
92          "var a, b = 1;", "b"),
93          new String[] {"var a;_.b = 1;", "_.b"});
94      test(createModules(
95          "var a, b = 1, c = 2;", "b"),
96          new String[] {"var a, c;_.b = 1;c = 2", "_.b"});
97      test(createModules(
98          "var a, b = 1, c = 2;", "a"),
99          new String[] {"var b, c;b = 1;c = 2", "_.a"});
100     test(createModules(
101         "var a=1; var b=2,c=3;", "a;c;"),
102         new String[] {"var b;_.a=1;b=2;_.c=3", "_.a;_.c"});
103     test(createModules(
104         "1;var a, b = 1, c = 2;", "b"),
105         new String[] {"var a, c;1;_.b = 1;c = 2", "_.b"});

177   public void testShadow() {
178     test(
179         "var _ = 1; (function () { _ = 2 })()",
180         "_._ = 1; (function () { _._ = 2 })()");
181     test(
182         "function foo() { var _ = {}; _.foo = foo; _.bar = 1; }",
183         "_.foo = function () { var _$ = {}; _$.foo = _.foo; _$.bar = 1}");
184     test(
185         "function foo() { var _ = {}; _.foo = foo; _.bar = 1; "
186         + "(function() { var _ = 0;})() }",
187         "_.foo = function () { var _$ = {}; _$.foo = _.foo; _$.bar = 1; "
188         + "(function() { var _$ = 0;})() }");
189     test(
190         "function foo() { var _ = {}; _.foo = foo; _.bar = 1; "
191         + "var _$ = 1; }",
192         "_.foo = function () { var _$ = {}; _$.foo = _.foo; _$.bar = 1; "
193         + "var _$$ = 1; }");
194     test(
195         "function foo() { var _ = {}; _.foo = foo; _.bar = 1; "
196         + "var _$ = 1; (function() { _ = _$ })() }",
197         "_.foo = function () { var _$ = {}; _$.foo = _.foo; _$.bar = 1; "
198         + "var _$$ = 1; (function() { _$ = _$$ })() }");
199     test(
200         "function foo() { var _ = {}; _.foo = foo; _.bar = 1; "
201         + "var _$ = 1, _$$ = 2 (function() { _ = _$ = _$$; " +
202         "var _$, _$$$ })() }",
203         "_.foo = function () { var _$ = {}; _$.foo = _.foo; _$.bar = 1; "
204         + "var _$$ = 1, _$$$ = 2 (function() { _$ = _$$ = _$$$; "
205         + "var _$$, _$$$$ })() }");
206     test(
207         "function foo() { var _a = 1;}",
208         "_.foo = function () { var _a = 1;}");
209     // We accept this unnecessary renaming as acceptable to simplify pattern
210     // matching in the traversal.
211     test(
212         "function foo() { var _$a = 1;}",
213         "_.foo = function () { var _$a$ = 1;}");
214   }

217     test(
218         "var document;",
219         "document",
220         "window.document", null, null);
221     test(
222         "var document;",
223         "document.getElementsByTagName('test')",
224         "window.document.getElementsByTagName('test')", null, null);
225     test(
226         "var document;",
227         "window.document.getElementsByTagName('test')",
228         "window.document.getElementsByTagName('test')", null, null);
229     test(
230         "var document;document.getElementsByTagName",
231         "document.getElementsByTagName('test')",
232         "window.document.getElementsByTagName('test')", null, null);
233     test(
234         "var document,navigator",
235         "document.navigator;navigator",
236         "window.document.navigator;window.navigator", null, null);
237     test(
238         "var iframes",
239         "function test() { iframes.resize(); }",
240         "_.test = function() { window.iframes.resize(); }", null, null);
241     test(
242         "var iframes",
243         "var foo = iframes;",
244         "_.foo = window.iframes;", null, null);
245     // Special names.
246     test(
247         "var arguments, window, eval;",
248         "arguments;window;eval;",
249         "arguments;window;eval;", null, null);
250     // Actually not an extern.
251     test(
252         "",
253         "document",
254         "window.document", null, null);
255     // Javascript builtin objects
256     test(
257         "Object;Function;Array;String;Boolean;Number;Math;"
258         + "Date;RegExp;JSON;Error;EvalError;ReferenceError;"
259         + "SyntaxError;TypeError;URIError;",
260         "Object;Function;Array;String;Boolean;Number;Math;"
261         + "Date;RegExp;JSON;Error;EvalError;ReferenceError;"
262         + "SyntaxError;TypeError;URIError;");

279     public void testFreeCallSemantics() {
280       // This triggers free call.
281       test(
282           "function x(){this};var y=function(){var val=x()||{}}",
283           "_.x=function(){this};_.y=function(){var val=(0,_.x)()||{}}");
284       test(
285           "function x(){this;x()}",
286           "_.x=function(){this;(0,_.x)()}");
287       test(
288           "var a=function(){this};a()",
289           "_.a=function(){this};(0,_.a)()");
290       // Cases where free call forcing through (0, foo)() is not necessary.
291       test(
292           "var a=function(){};a()",
293           "_.a=function(){};_.a()");
294       test(
295           "function a(){};a()",
296           "_.a=function(){};_.a()");
297       test(
298           "var a;a=function(){};a()",
299           "_.a=function(){};_.a()");
300       // Ambigious cases.
301       test(
302           "var a=1;a=function(){};a()",
303           "_.a=1;_.a=function(){};(0,_.a)()");
304       test(
305           "var b;var a=b;a()",
306           "_.a=_.b;(0,_.a)()");
307     }
