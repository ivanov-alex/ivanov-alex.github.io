22    @Override
23    protected void setUp() throws Exception {
24      super.setUp();

28    @Override
29    protected CompilerOptions getOptions() {
30      CompilerOptions options = super.getOptions();
31      options.setLanguageIn(LanguageMode.ECMASCRIPT6_TYPED);
32      options.setLanguageOut(LanguageMode.ECMASCRIPT6);
33      return options;
34    }
35  
36    @Override
37    protected CompilerPass getProcessor(Compiler compiler) {
38      PhaseOptimizer optimizer = new PhaseOptimizer(compiler, null, null);
39      optimizer.addOneTimePass(new PassFactory("convertDeclaredTypesToJSDoc", true) {
40        // To make sure types copied.
41        @Override CompilerPass create(AbstractCompiler compiler) {
42          return new Es6TypedToEs6Converter(compiler);
43        }
44      });
45      return optimizer;
46    }

50      return 1;

55          "class C {",
56          "  mv: number;",
57          "  constructor() {",
58          "    this.f = 1;",
59          "  }",
60          "}"),
61          LINE_JOINER.join(
62          "class C {",

69      test(LINE_JOINER.join(
70          "class C {",
71          "  on: {",

73          "  }",
74          "}"),

76          "class C {}",
77          "/** @type {{p: string}} */ C.prototype.on;"));

79      test(
80          "export class C { foo: number; }",
81          "export class C {} /** @type {number} */ C.prototype.foo;");
82  
83      testExternChanges(
84          "declare class C { foo: number; }",
85          "class C {} /** @type {number} */ C.prototype.foo;");
86  
87      testExternChanges(
88          "export declare class C { foo: number; }",
89          "export class C {} /** @type {number} */ C.prototype.foo;");
