94    public void testFoldRegExpConstructor() {
95      enableNormalize();
96  
97      // Cannot fold all the way to a literal because there are too few arguments.
98      fold("x = new RegExp",                    "x = RegExp()");
99      // Empty regexp should not fold to // since that is a line comment in JS
100     fold("x = new RegExp(\"\")",              "x = RegExp(\"\")");
101     fold("x = new RegExp(\"\", \"i\")",       "x = RegExp(\"\",\"i\")");
102     // Bogus flags should not fold
103     testSame("x = RegExp(\"foobar\", \"bogus\")",
104          PeepholeSubstituteAlternateSyntax.INVALID_REGULAR_EXPRESSION_FLAGS);
105     // Don't fold if the argument is not a string. See issue 1260.
106     foldSame("x = new RegExp(y)");
107     // Can Fold
108     fold("x = new RegExp(\"foobar\")",        "x = /foobar/");
109     fold("x = RegExp(\"foobar\")",            "x = /foobar/");
110     fold("x = new RegExp(\"foobar\", \"i\")", "x = /foobar/i");
111     // Make sure that escaping works
112     fold("x = new RegExp(\"\\\\.\", \"i\")",  "x = /\\./i");
113     fold("x = new RegExp(\"/\", \"\")",       "x = /\\//");
114     fold("x = new RegExp(\"[/]\", \"\")",     "x = /[/]/");
115     fold("x = new RegExp(\"///\", \"\")",     "x = /\\/\\/\\//");
116     fold("x = new RegExp(\"\\\\\\/\", \"\")", "x = /\\//");
117     fold("x = new RegExp(\"\\n\")",           "x = /\\n/");
118     fold("x = new RegExp('\\\\\\r')",         "x = /\\r/");
119 
120     // Shouldn't fold RegExp unnormalized because
121     // we can't be sure that RegExp hasn't been redefined
122     disableNormalize();
123 
124     foldSame("x = new RegExp(\"foobar\")");
125   }

127   public void testVersionSpecificRegExpQuirks() {
128     enableNormalize();
129 
130     // Don't fold if the flags contain 'g'
131     setAcceptedLanguage(LanguageMode.ECMASCRIPT3);
132     fold("x = new RegExp(\"foobar\", \"g\")",
133          "x = RegExp(\"foobar\",\"g\")");
134     fold("x = new RegExp(\"foobar\", \"ig\")",
135          "x = RegExp(\"foobar\",\"ig\")");
136     // ... unless in ECMAScript 5 mode per section 7.8.5 of ECMAScript 5.
137     setAcceptedLanguage(LanguageMode.ECMASCRIPT5);
138     fold("x = new RegExp(\"foobar\", \"ig\")",
139          "x = /foobar/ig");
140     // Don't fold things that crash older versions of Safari and that don't work
141     // as regex literals on other old versions of Safari
142     setAcceptedLanguage(LanguageMode.ECMASCRIPT3);
143     fold("x = new RegExp(\"\\u2028\")", "x = RegExp(\"\\u2028\")");
144     fold("x = new RegExp(\"\\\\\\\\u2028\")", "x = /\\\\u2028/");
145     // Sunset Safari exclusions for ECMAScript 5 and later.
146     setAcceptedLanguage(LanguageMode.ECMASCRIPT5);
147     fold("x = new RegExp(\"\\u2028\\u2029\")", "x = /\\u2028\\u2029/");
148     fold("x = new RegExp(\"\\\\u2028\")", "x = /\\u2028/");
149     fold("x = new RegExp(\"\\\\\\\\u2028\")", "x = /\\\\u2028/");
150   }

170         "foo\\\\u2028bar\\u2028"));

195     enableNormalize();

198     fold("x = new Array", "x = []");
199     fold("x = new Array()", "x = []");
200     fold("x = Array()", "x = []");

203     fold("x = new Array(0)", "x = []");
204     fold("x = Array(0)", "x = []");
205     fold("x = new Array(\"a\")", "x = [\"a\"]");
206     fold("x = Array(\"a\")", "x = [\"a\"]");
207 
208     // One argument - cannot be fold when normalized
209     fold("x = new Array(7)", "x = Array(7)");
210     fold("x = Array(7)", "x = Array(7)");
211     fold("x = new Array(y)", "x = Array(y)");
212     fold("x = Array(y)", "x = Array(y)");
213     fold("x = new Array(foo())", "x = Array(foo())");
214     fold("x = Array(foo())", "x = Array(foo())");
215 
216     // More than one argument - can be fold when normalized
217     fold("x = new Array(1, 2, 3, 4)", "x = [1, 2, 3, 4]");
218     fold("x = Array(1, 2, 3, 4)", "x = [1, 2, 3, 4]");

223     fold("x = new Array(Array(1, '2', 3, '4'))", "x = [[1, '2', 3, '4']]");
224     fold("x = Array(Array(1, '2', 3, '4'))", "x = [[1, '2', 3, '4']]");
225     fold("x = new Array(Object(), Array(\"abc\", Object(), Array(Array())))",
226          "x = [{}, [\"abc\", {}, [[]]]]");
227     fold("x = new Array(Object(), Array(\"abc\", Object(), Array(Array())))",
228          "x = [{}, [\"abc\", {}, [[]]]]");
229 
230     disableNormalize();
231     // Cannot fold above when not normalized
232     foldSame("x = new Array");
233     foldSame("x = new Array()");
234     foldSame("x = Array()");
235 
236     foldSame("x = new Array(0)");
237     foldSame("x = Array(0)");
238     foldSame("x = new Array(\"a\")");
239     foldSame("x = Array(\"a\")");
240     foldSame("x = new Array(7)");
241     foldSame("x = Array(7)");
242     foldSame("x = new Array(foo())");
243     foldSame("x = Array(foo())");
244 
245     foldSame("x = new Array(1, 2, 3, 4)");
246     foldSame("x = Array(1, 2, 3, 4)");
247     foldSame("x = new Array('a', 1, 2, 'bc', 3, {}, 'abc')");
248     foldSame("x = Array('a', 1, 2, 'bc', 3, {}, 'abc')");
249     foldSame("x = new Array(Array(1, '2', 3, '4'))");
250     foldSame("x = Array(Array(1, '2', 3, '4'))");
251     foldSame("x = new Array(" +
252         "Object(), Array(\"abc\", Object(), Array(Array())))");
253     foldSame("x = new Array(" +
254         "Object(), Array(\"abc\", Object(), Array(Array())))");

396   public void testBindToCall1() {
397     test("(goog.bind(f))()", "f()");
398     test("(goog.bind(f,a))()", "f.call(a)");
399     test("(goog.bind(f,a,b))()", "f.call(a,b)");
400 
401     test("(goog.bind(f))(a)", "f(a)");
402     test("(goog.bind(f,a))(b)", "f.call(a,b)");
403     test("(goog.bind(f,a,b))(c)", "f.call(a,b,c)");
404 
405     test("(goog.partial(f))()", "f()");
406     test("(goog.partial(f,a))()", "f(a)");
407     test("(goog.partial(f,a,b))()", "f(a,b)");
408 
409     test("(goog.partial(f))(a)", "f(a)");
410     test("(goog.partial(f,a))(b)", "f(a,b)");
411     test("(goog.partial(f,a,b))(c)", "f(a,b,c)");
412 
413     test("((function(){}).bind())()", "((function(){}))()");
414     test("((function(){}).bind(a))()", "((function(){})).call(a)");
415     test("((function(){}).bind(a,b))()", "((function(){})).call(a,b)");
416 
417     test("((function(){}).bind())(a)", "((function(){}))(a)");
418     test("((function(){}).bind(a))(b)", "((function(){})).call(a,b)");
419     test("((function(){}).bind(a,b))(c)", "((function(){})).call(a,b,c)");
420 
421     // Without using type information we don't know "f" is a function.
422     testSame("(f.bind())()");
423     testSame("(f.bind(a))()");
424     testSame("(f.bind())(a)");
425     testSame("(f.bind(a))(b)");
426 
427     // Don't rewrite if the bind isn't the immediate call target
428     testSame("(goog.bind(f)).call(g)");

468   public void testRotateAssociativeOperators() {
469     test("a || (b || c); a * (b * c); a | (b | c)",
470         "(a || b) || c; (a * b) * c; (a | b) | c");
471     testSame("a % (b % c); a / (b / c); a - (b - c);");
472     test("a * (b % c);", "b % c * a");
473     test("a * b * (c / d)", "c / d * b * a");
474     test("(a + b) * (c % d)", "c % d * (a + b)");
475     testSame("(a / b) * (c % d)");
476     testSame("(c = 5) * (c % d)");
477     test("(a + b) * c * (d % e)", "d % e * c * (a + b)");
478     test("!a * c * (d % e)", "d % e * c * !a");
479   }
