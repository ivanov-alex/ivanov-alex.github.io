19  /**
20   * Verifies that valid candidates for object literals are inlined as
21   * expected, and invalid candidates are not touched.
22   *
23   */
24  public final class InlineObjectLiteralsTest extends CompilerTestCase {
25  
26    public InlineObjectLiteralsTest() {
27      enableNormalize();
28    }
29  
30    @Override
31    protected CompilerPass getProcessor(final Compiler compiler) {
32      return new InlineObjectLiterals(
33          compiler,
34          compiler.getUniqueNameIdSupplier());
35    }

39    public void testObject0() {
40      // Don't mess with global variables, that is the job of CollapseProperties.
41      testSame("var a = {x:1}; f(a.x);");
42    }

44    public void testObject1() {
45      testLocal("var a = {x:x(), y:y()}; f(a.x, a.y);",
46           "var JSCompiler_object_inline_x_0=x();" +
47           "var JSCompiler_object_inline_y_1=y();" +
48           "f(JSCompiler_object_inline_x_0, JSCompiler_object_inline_y_1);");
49    }

53           "var JSCompiler_object_inline_x_0;" +

70      // target function.

73    }

130   public void testObject9() {
131     // There is a call, so no inlining
132     testSameLocal("function f(a,b) {" +
133              "  var x = {a:a,b:b}; x.a(); return x.b;" +
134              "}");
135 
136     testLocal("function f(a,b) {" +
137          "  var x = {a:a,b:b}; g(x.a); x = {a:a,b:2}; return x.b;" +
138          "}",
139          "function f(a,b) {" +
140          "  var JSCompiler_object_inline_a_0 = a;" +
141          "  var JSCompiler_object_inline_b_1 = b;" +
142          "  g(JSCompiler_object_inline_a_0);" +
143          "  JSCompiler_object_inline_a_0 = a," +
144          "  JSCompiler_object_inline_b_1=2," +
145          "  true;" +
146          "  return JSCompiler_object_inline_b_1" +
147          "}");
148 
149     testLocal("function f(a,b) { " +
150          "  var x = {a:a,b:b}; g(x.a); x.b = x.c = 2; return x.b; " +
151          "}",
152          "function f(a,b) { " +
153          "  var JSCompiler_object_inline_a_0=a;" +
154          "  var JSCompiler_object_inline_b_1=b; " +
155          "  var JSCompiler_object_inline_c_2;" +
156          "  g(JSCompiler_object_inline_a_0);" +
157          "  JSCompiler_object_inline_b_1=JSCompiler_object_inline_c_2=2;" +
158          "  return JSCompiler_object_inline_b_1" +
159          "}");

353   private final String LOCAL_PREFIX = "function local(){";
354   private final String LOCAL_POSTFIX = "}";
355 
356   private void testLocal(String code, String result) {
357     test(LOCAL_PREFIX + code + LOCAL_POSTFIX,
358          LOCAL_PREFIX + result + LOCAL_POSTFIX);
359   }
360 
361   private void testSameLocal(String code) {
362     testLocal(code, code);
363   }
