40    public void testStringIndexOf() {
41      fold("x = 'abcdef'.indexOf('b')", "x = 1");
42      fold("x = 'abcdefbe'.indexOf('b', 2)", "x = 6");
43      fold("x = 'abcdef'.indexOf('bcd')", "x = 1");
44      fold("x = 'abcdefsdfasdfbcdassd'.indexOf('bcd', 4)", "x = 13");
45  
46      fold("x = 'abcdef'.lastIndexOf('b')", "x = 1");
47      fold("x = 'abcdefbe'.lastIndexOf('b')", "x = 6");
48      fold("x = 'abcdefbe'.lastIndexOf('b', 5)", "x = 1");
49  
50      // Both elements must be strings. Don't do anything if either one is not
51      // string.
52      fold("x = 'abc1def'.indexOf(1)", "x = 3");
53      fold("x = 'abcNaNdef'.indexOf(NaN)", "x = 3");
54      fold("x = 'abcundefineddef'.indexOf(undefined)", "x = 3");
55      fold("x = 'abcnulldef'.indexOf(null)", "x = 3");
56      fold("x = 'abctruedef'.indexOf(true)", "x = 3");
57  
58      // The following test case fails with JSC_PARSE_ERROR. Hence omitted.
59      // foldSame("x = 1.indexOf('bcd');");
60      foldSame("x = NaN.indexOf('bcd')");
61      foldSame("x = undefined.indexOf('bcd')");
62      foldSame("x = null.indexOf('bcd')");
63      foldSame("x = true.indexOf('bcd')");
64      foldSame("x = false.indexOf('bcd')");
65  
66      // Avoid dealing with regex or other types.
67      foldSame("x = 'abcdef'.indexOf(/b./)");
68      foldSame("x = 'abcdef'.indexOf({a:2})");
69      foldSame("x = 'abcdef'.indexOf([1,2])");
70    }

81    public void testStringJoinAdd() {
82      fold("x = ['a', 'b', 'c'].join('')", "x = \"abc\"");
83      fold("x = [].join(',')", "x = \"\"");
84      fold("x = ['a'].join(',')", "x = \"a\"");
85      fold("x = ['a', 'b', 'c'].join(',')", "x = \"a,b,c\"");
86      fold("x = ['a', foo, 'b', 'c'].join(',')",
87          "x = [\"a\",foo,\"b,c\"].join()");
88      fold("x = [foo, 'a', 'b', 'c'].join(',')",
89          "x = [foo,\"a,b,c\"].join()");
90      fold("x = ['a', 'b', 'c', foo].join(',')",
91          "x = [\"a,b,c\",foo].join()");
92  
93      // Works with numbers
94      fold("x = ['a=', 5].join('')", "x = \"a=5\"");
95      fold("x = ['a', '5'].join(7)", "x = \"a75\"");
96  
97      // Works on boolean
98      fold("x = ['a=', false].join('')", "x = \"a=false\"");
99      fold("x = ['a', '5'].join(true)", "x = \"atrue5\"");
100     fold("x = ['a', '5'].join(false)", "x = \"afalse5\"");
101 
102     // Only optimize if it's a size win.
103     fold("x = ['a', '5', 'c'].join('a very very very long chain')",
104          "x = [\"a\",\"5\",\"c\"].join(\"a very very very long chain\")");
105 
106     // TODO(user): Its possible to fold this better.
107     foldSame("x = ['', foo].join('-')");
108     foldSame("x = ['', foo, ''].join()");
109 
110     fold("x = ['', '', foo, ''].join(',')",
111          "x = [',', foo, ''].join()");
112     fold("x = ['', '', foo, '', ''].join(',')",
113          "x = [',', foo, ','].join()");
114 
115     fold("x = ['', '', foo, '', '', bar].join(',')",
116          "x = [',', foo, ',', bar].join()");
117 
118     fold("x = [1,2,3].join('abcdef')",
119          "x = '1abcdef2abcdef3'");
120 
121     fold("x = [1,2].join()", "x = '1,2'");
122     fold("x = [null,undefined,''].join(',')", "x = ',,'");
123     fold("x = [null,undefined,0].join(',')", "x = ',,0'");
124     // This can be folded but we don't currently.
125     foldSame("x = [[1,2],[3,4]].join()"); // would like: "x = '1,2,3,4'"
126   }

162     fold("x = 'abcde'.charAt(1)", "x = 'b'");

194   public void testFoldStringSplit() {
195     late = false;
196     fold("x = 'abcde'.split('foo')", "x = ['abcde']");
197     fold("x = 'abcde'.split()", "x = ['abcde']");
198     fold("x = 'abcde'.split(null)", "x = ['abcde']");
199     fold("x = 'a b c d e'.split(' ')", "x = ['a','b','c','d','e']");
200     fold("x = 'a b c d e'.split(' ', 0)", "x = []");
201     fold("x = 'abcde'.split('cd')", "x = ['ab','e']");
202     fold("x = 'a b c d e'.split(' ', 1)", "x = ['a']");
203     fold("x = 'a b c d e'.split(' ', 3)", "x = ['a','b','c']");
204     fold("x = 'a b c d e'.split(null, 1)", "x = ['a b c d e']");
205     fold("x = 'aaaaa'.split('a')", "x = ['', '', '', '', '', '']");
206     fold("x = 'xyx'.split('x')", "x = ['', 'y', '']");
207 
208     // Empty separator
209     fold("x = 'abcde'.split('')", "x = ['a','b','c','d','e']");
210     fold("x = 'abcde'.split('', 3)", "x = ['a','b','c']");
211 
212     // Empty separator AND empty string
213     fold("x = ''.split('')", "x = []");
214 
215     // Separator equals string
216     fold("x = 'aaa'.split('aaa')", "x = ['','']");
217     fold("x = ' '.split(' ')", "x = ['','']");
218 
219     foldSame("x = 'abcde'.split(/ /)");
220     foldSame("x = 'abcde'.split(' ', -1)");
221 
222     late = true;
223     foldSame("x = 'a b c d e'.split(' ')");
224   }

254   public void testFoldParseNumbers() {
255     enableNormalize();
256     setAcceptedLanguage(LanguageMode.ECMASCRIPT5);
257 
258     fold("x = parseInt('123')", "x = 123");
259     fold("x = parseInt(' 123')", "x = 123");
260     fold("x = parseInt('123', 10)", "x = 123");
261     fold("x = parseInt('0xA')", "x = 10");
262     fold("x = parseInt('0xA', 16)", "x = 10");
263     fold("x = parseInt('07', 8)", "x = 7");
264     fold("x = parseInt('08')", "x = 8");
265     fold("x = parseInt('0')", "x = 0");
266     fold("x = parseFloat('0')", "x = 0");
267     fold("x = parseFloat('1.23')", "x = 1.23");
268     fold("x = parseFloat('1.2300')", "x = 1.23");
269     fold("x = parseFloat(' 0.3333')", "x = 0.3333");
270     fold("x = parseFloat('0100')", "x = 100");
271     fold("x = parseFloat('0100.000')", "x = 100");
272 
273     //Mozilla Dev Center test cases
274     fold("x = parseInt(' 0xF', 16)", "x = 15");
275     fold("x = parseInt(' F', 16)", "x = 15");
276     fold("x = parseInt('17', 8)", "x = 15");
277     fold("x = parseInt('015', 10)", "x = 15");
278     fold("x = parseInt('1111', 2)", "x = 15");
279     fold("x = parseInt('12', 13)", "x = 15");
280     fold("x = parseInt(15.99, 10)", "x = 15");
281     fold("x = parseFloat('3.14')", "x = 3.14");
282     fold("x = parseFloat(3.14)", "x = 3.14");
283 
284     //Valid calls - unable to fold
285     foldSame("x = parseInt('FXX123', 16)");
286     foldSame("x = parseInt('15*3', 10)");
287     foldSame("x = parseInt('15e2', 10)");
288     foldSame("x = parseInt('15px', 10)");
289     foldSame("x = parseInt('-0x08')");
290     foldSame("x = parseInt('1', -1)");
291     foldSame("x = parseFloat('3.14more non-digit characters')");
292     foldSame("x = parseFloat('314e-2')");
293     foldSame("x = parseFloat('0.0314E+2')");
294     foldSame("x = parseFloat('3.333333333333333333333333')");
295 
296     //Invalid calls
297     foldSame("x = parseInt('0xa', 10)");
298     foldSame("x = parseInt('')");
299 
300     setAcceptedLanguage(LanguageMode.ECMASCRIPT3);
301     foldSame("x = parseInt('08')");
302   }
