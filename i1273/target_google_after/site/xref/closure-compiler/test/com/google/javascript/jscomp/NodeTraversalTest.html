39    public void testPruningCallbackShouldTraverse1() {
40      PruningCallback include =
41        new PruningCallback(ImmutableSet.of(Token.SCRIPT, Token.VAR), true);
42  
43      Node script = new Node(Token.SCRIPT);
44      assertTrue(include.shouldTraverse(null, script, null));
45      assertTrue(include.shouldTraverse(null, new Node(Token.VAR), null));
46      assertFalse(include.shouldTraverse(null, new Node(Token.NAME), null));
47      assertFalse(include.shouldTraverse(null, new Node(Token.ADD), null));
48    }

52        new PruningCallback(ImmutableSet.of(Token.SCRIPT, Token.VAR), false);

55      assertFalse(include.shouldTraverse(null, script, null));
56      assertFalse(include.shouldTraverse(null, new Node(Token.VAR), null));
57      assertTrue(include.shouldTraverse(null, new Node(Token.NAME), null));
58      assertTrue(include.shouldTraverse(null, new Node(Token.ADD), null));

76    public void testReport() {
77      final List<JSError> errors = new ArrayList<>();
78  
79      Compiler compiler = new Compiler(new BasicErrorManager() {
80  
81        @Override public void report(CheckLevel level, JSError error) {
82          errors.add(error);
83        }
84  
85        @Override public void println(CheckLevel level, JSError error) {
86        }
87  
88        @Override protected void printSummary() {
89        }
90      });
91      compiler.initCompilerOptionsIfTesting();
92  
93      NodeTraversal t = new NodeTraversal(compiler, null);
94      DiagnosticType dt = DiagnosticType.warning("FOO", "{0}, {1} - {2}");
95  
96      t.report(new Node(Token.EMPTY), dt, "Foo", "Bar", "Hello");
97      assertThat(errors).hasSize(1);
98      assertEquals("Foo, Bar - Hello", errors.get(0).description);
99    }

103   public void testUnexpectedException() {
104 
105     NodeTraversal.Callback cb = new NodeTraversal.AbstractPostOrderCallback() {
106       @Override
107       public void visit(NodeTraversal t, Node n, Node parent) {
108         throw new RuntimeException(TEST_EXCEPTION);
109       }
110     };
111 
112     Compiler compiler = new Compiler();
113

115       String code = "function foo() {}";

120       assertThat(e.getMessage())

123               + "test me");

162   public void testGetLineNoAndGetCharno() {
163     Compiler compiler = new Compiler();
164     String code = ""
165         + "var a; \n"
166         + "function foo() {\n"
167         + "  var b;\n"
168         + "  if (a) { var c;}\n"
169         + "}";
170     Node tree = parse(compiler, code);
171     final StringBuilder builder = new StringBuilder();
172     NodeTraversal.traverseEs6(compiler, tree,
173         new NodeTraversal.ScopedCallback() {
174 
175           @Override
176           public void enterScope(NodeTraversal t) {
177           }
178 
179           @Override
180           public void exitScope(NodeTraversal t) {
181           }
182 
183           @Override
184           public boolean shouldTraverse(NodeTraversal t, Node n, Node parent) {
185             return true;
186           }
187 
188           @Override
189           public void visit(NodeTraversal t, Node n, Node parent) {
190             builder.append("visit ");
191             builder.append(t.getCurrentNode().toString(false, true, true));
192             builder.append(" @");
193             builder.append(t.getLineNumber());
194             builder.append(":");
195             builder.append(t.getCharno());
196             builder.append("\n");
197           }
198         }
199     );
200 
201     // Note the char numbers are 0-indexed but the line numbers are 1-indexed.
202     String expectedResult = ""
203         + "visit NAME a [source_file: [testcode]] @1:4\n"
204         + "visit VAR [source_file: [testcode]] @1:0\n"
205         + "visit NAME foo [source_file: [testcode]] @2:9\n"
206         + "visit PARAM_LIST [source_file: [testcode]] @2:12\n"
207         + "visit NAME b [source_file: [testcode]] @3:6\n"
208         + "visit VAR [source_file: [testcode]] @3:2\n"
209         + "visit NAME a [source_file: [testcode]] @4:6\n"
210         + "visit NAME c [source_file: [testcode]] @4:15\n"
211         + "visit VAR [source_file: [testcode]] @4:11\n"
212         + "visit BLOCK [source_file: [testcode]] @4:9\n"
213         + "visit IF [source_file: [testcode]] @4:2\n"
214         + "visit BLOCK [source_file: [testcode]] @2:15\n"
215         + "visit FUNCTION foo [source_file: [testcode]] @2:0\n"
216         + "visit SCRIPT [synthetic: 1] [source_file: [testcode]] "
217         + "[input_id: InputId: [testcode]] @1:0\n";
218 
219     assertEquals(expectedResult, builder.toString());
220   }

222   public void testGetCurrentNode() {
223     Compiler compiler = new Compiler();
224     ScopeCreator creator = SyntacticScopeCreator.makeUntyped(compiler);
225     ExpectNodeOnEnterScope callback = new ExpectNodeOnEnterScope();
226     NodeTraversal t = new NodeTraversal(compiler, callback, creator);
227 
228     String code = Joiner.on('\n').join(
229         "var a;",
230         "function foo() {",
231         "  var b;",
232         "}");
233 
234     Node tree = parse(compiler, code);
235     Scope topScope = creator.createScope(tree, null);
236 
237     // Calling #traverseWithScope uses the given scope but starts traversal at
238     // the given node.
239     callback.expect(tree.getFirstChild(), tree);
240     t.traverseWithScope(tree.getFirstChild(), topScope);
241     callback.assertEntered();
242 
243     // Calling #traverse creates a new scope with the given node as the root.
244     callback.expect(tree.getFirstChild(), tree.getFirstChild());
245     t.traverse(tree.getFirstChild());
246     callback.assertEntered();
247 
248     // Calling #traverseAtScope starts traversal from the scope's root.
249     Node fn = tree.getFirstChild().getNext();
250     Scope fnScope = creator.createScope(fn, topScope);
251     callback.expect(fn, fn);
252     t.traverseAtScope(fnScope);
253     callback.assertEntered();
254   }

256   public void testTraverseAtScopeWithBlockScope() {
257     Compiler compiler = new Compiler();
258     CompilerOptions options = new CompilerOptions();
259     options.setLanguageIn(LanguageMode.ECMASCRIPT6);
260     compiler.initOptions(options);
261     ScopeCreator creator = new Es6SyntacticScopeCreator(compiler);
262     ExpectNodeOnEnterScope callback = new ExpectNodeOnEnterScope();
263     NodeTraversal t = new NodeTraversal(compiler, callback, creator);
264 
265     String code = Joiner.on('\n').join(
266         "function foo() {",
267         "  if (bar) {",
268         "    let x;",
269         "  }",
270         "}");
271 
272     Node tree = parse(compiler, code);
273     Scope topScope = creator.createScope(tree, null);
274 
275     Node innerBlock = tree  // script
276         .getFirstChild()    // function
277         .getLastChild()     // function body
278         .getFirstChild()    // if
279         .getLastChild();    // block
280 
281     Scope blockScope = creator.createScope(innerBlock, topScope);
282     callback.expect(innerBlock, innerBlock);
283     t.traverseAtScope(blockScope);
284     callback.assertEntered();
285   }

287   // Helper class used to test getCurrentNode
288   private static class ExpectNodeOnEnterScope extends NodeTraversal.AbstractPreOrderCallback
289       implements NodeTraversal.ScopedCallback {
290     private Node node;
291     private Node scopeRoot;
292     private boolean entered = false;
293 
294     private void expect(Node node, Node scopeRoot) {
295       this.node = node;
296       this.scopeRoot = scopeRoot;
297       entered = false;
298     }
299 
300     private void assertEntered() {
301       assertTrue(entered);
302     }
303 
304     @Override
305     public void enterScope(NodeTraversal t) {
306       assertNode(t.getCurrentNode()).isEqualTo(node);
307       assertNode(t.getScopeRoot()).isEqualTo(scopeRoot);
308       entered = true;
309     }
310 
311     @Override
312     public void exitScope(NodeTraversal t) {
313     }
314 
315     @Override
316     public boolean shouldTraverse(NodeTraversal t, Node n, Node parent) {
317       return true;
318     }
319   }

321   private static Node parse(Compiler compiler, String js) {
322     Node n = compiler.parseTestCode(js);
323     assertThat(compiler.getErrors()).isEmpty();
324     return n;
325   }
