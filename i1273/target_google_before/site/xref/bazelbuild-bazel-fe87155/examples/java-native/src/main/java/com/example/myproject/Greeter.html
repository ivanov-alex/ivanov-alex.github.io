1   package com.example.myproject;
2   
3   import java.io.InputStream;
4   import java.io.PrintStream;
5   import java.util.Scanner;
6   
7   /**
8    * Prints a greeting which can be customized by building with resources and/or passing in command-
9    * line arguments.
10   *
11   * <p>Building and running this file will print "Hello world". Build and run the
12   * //examples/java-native/src/main/java/com/example/myproject:hello-world target to demonstrate
13   * this.</p>
14   *
15   * <p>If this is built with a greeting.txt resource included, it will replace "Hello" with the
16   * contents of greeting.txt. The
17   * //examples/java-native/src/main/java/com/example/myproject:hello-resources target demonstrates
18   * this.</p>
19   *
20   * <p>If arguments are passed to the binary on the command line, the first argument will replace
21   * "world" in the output. See //examples/java-native/src/test/java/com/example/myproject:hello's
22   * argument test.</p>
23   */
24  public class Greeter {
25    static PrintStream out = System.out;
26  
27    public static String convertStreamToString(InputStream is) throws Exception {
28      try (Scanner s = new Scanner(is)) {
29        s.useDelimiter("\n");
30        return s.hasNext() ? s.next() : "";
31      }
32    }
33  
34    public void hello(String obj) throws Exception {
35      String greeting = "Hello";
36      InputStream stream  = Greeter.class.getResourceAsStream("/greeting.txt");
37      if (stream != null) {
38        greeting = convertStreamToString(stream);
39      }
40      out.println(greeting + " " + obj);
41    }
42  
43    public static void main(String... args) throws Exception {
44      Greeter g = new Greeter();
45      String obj = args.length > 0 ? args[0] : "world";
46      g.hello(obj);
47    }
48  }
