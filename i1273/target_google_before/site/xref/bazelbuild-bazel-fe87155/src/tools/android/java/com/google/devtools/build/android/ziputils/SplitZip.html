46  /**
47   * Extracts entries from a set of input archives, and copies them to N output archive of
48   * approximately equal size, while attempting to split archives on package (directory) boundaries.
49   * Optionally, accept a list of entries to be added to the first output archive, splitting
50   * remaining entries by package boundaries.
51   */
52  public class SplitZip implements EntryHandler {

54    private final List<ZipIn> inputs;
55    private final List<ZipOut> outputs;
56    private String filterFile;
57    private InputStream filterInputStream;
58    private String resourceFile;
59    private Date date;
60    private DosTime dosTime;
61    // Internal state variables:
62    private boolean finished = false;
63    private Set<String> filter;
64    private ZipOut[] zipOuts;
65    private ZipOut resourceOut;
66    private final Map<String, ZipOut> assignments = new HashMap<>();
67    private final Map<String, CentralDirectory> centralDirectories;
68    private final Set<String> classes = new TreeSet<>();

248   /**
249    * Executes this {@code SplitZip}, reading content from the configured input locations, creating
250    * the specified number of archives, in the configured output directory.
251    *
252    * @return this object
253    * @throws java.io.IOException
254    */
255   public SplitZip run() throws IOException {
256     verbose("SplitZip: Splitting in: " + outputs.size());
257     verbose("SplitZip: with filter: " + filterFile);
258     checkConfig();
259     // Prepare output files
260     zipOuts = outputs.toArray(new ZipOut[outputs.size()]);
261     if (resourceFile != null) {
262       resourceOut = new ZipOut(new FileOutputStream(resourceFile, false).getChannel(),
263           resourceFile);
264     } else if (resourceOut == null) { // may have been set for testing
265       resourceOut = zipOuts[0];
266     }
267 
268     // Read directories of input files
269     for (ZipIn zip : inputs) {
270       zip.endOfCentralDirectory();
271       centralDirectories.put(zip.getFilename(), zip.centralDirectory());
272       zip.centralDirectory();
273     }
274     // Assign input entries to output files
275     split();
276     // Copy entries to the assigned output files
277     for (ZipIn zip : inputs) {
278       zip.scanEntries(this);
279     }
280     return this;
281   }

283   /**
284    * Copies an entry to the assigned output files. Called for each entry in the input files. 
285    * @param in
286    * @param header
287    * @param dirEntry
288    * @param data
289    * @throws IOException 
290    */
291   @Override
292   public void handle(ZipIn in, LocalFileHeader header, DirectoryEntry dirEntry,
293       ByteBuffer data) throws IOException {
294     String localFilename = header.getFilename();
295     ZipOut out = assignments.remove(localFilename);
296     if (out == null) {
297       // Skip unassigned file;
298       return;
299     }
300     if (dirEntry == null) {
301       // Shouldn't get here, as there should be no assignment.
302       System.out.println("Warning: no directory entry");
303       return;
304     }
305     // Clone directory entry
306     DirectoryEntry entryOut = out.nextEntry(dirEntry);
307     if (dosTime != null) {
308       // Overwrite time stamp
309       header.set(LOCTIM, dosTime.time);
310       entryOut.set(CENTIM, dosTime.time);
311     }
312     out.write(header);
313     out.write(data);
314     if ((header.get(LOCFLG) & LocalFileHeader.SIZE_MASKED_FLAG) != 0) {
315       // Instead of this, we could fix the header with the size information
316       // from the directory entry. For now, keep the entry encoded as-is.
317       DataDescriptor desc = DataDescriptor.allocate()
318           .set(EXTCRC, dirEntry.get(CENCRC))
319           .set(EXTSIZ, dirEntry.get(CENSIZ))
320           .set(EXTLEN, dirEntry.get(CENLEN));
321       out.write(desc);
322     }
323   }

325   /**
326    * Writes any remaining output data to the output stream.
327    *
328    * @throws IOException if the output stream or the filter throws an IOException
329    * @throws IllegalStateException if this method was already called earlier
330    */
331   public void finish() throws IOException {
332     checkNotFinished();
333     finished = true;
334     if (resourceOut != null) {
335       resourceOut.finish();
336     }
337     for (ZipOut zo : zipOuts) {
338       zo.finish();
339     }
340   }
341 
342   /**
343    * Writes any remaining output data to the output stream and closes it.
344    *
345    * @throws IOException if the output stream or the filter throws an IOException
346    */
347   public void close() throws IOException {
348     if (!finished) {
349       finish();
350     }
351     if (resourceOut != null) {
352       resourceOut.close();
353     }
354     for (ZipOut zo : zipOuts) {
355       zo.close();
356     }
357   }
358 
359   private void checkNotFinished() {
360     if (finished) {
361       throw new IllegalStateException();
362     }
363   }

365   /**
366    * Validates configuration before execution.
367    */
368   private void checkConfig() throws IOException {
369     if (outputs.size() < 1) {
370       throw new IllegalStateException("Require at least one output file");
371     }
372     filter = filterFile == null && filterInputStream == null ? null : readPaths(filterFile);
373   }
374 
375   /**
376    * Parses the entries and assign each entry to an output file.
377    */
378   private void split() {
379     for (ZipIn in : inputs) {
380       CentralDirectory cdir = centralDirectories.get(in.getFilename());
381       for (DirectoryEntry entry : cdir.list()) {
382         String filename = entry.getFilename();
383         if (filename.endsWith(".class")) {
384           // Only pass classes to the splitter, so that it can do the best job
385           // possible distributing them across output files.
386           classes.add(filename);
387         } else if (!filename.endsWith("/")) {
388           // Non class files (resources) are either assigned to the first
389           // output file, or to a specified resource output file.
390           assignments.put(filename, resourceOut);
391         }
392       }
393     }
394     Splitter entryFilter = new Splitter(outputs.size(), classes.size());
395     if (filter != null) {
396       // Assign files in the filter to the first output file.
397       entryFilter.assign(filter);
398       entryFilter.nextShard(); // minimal initial shard
399     }
400     for (String path : classes) {
401       int assignment = entryFilter.assign(path);
402       Preconditions.checkState(assignment >= 0 && assignment < zipOuts.length);
403       assignments.put(path, zipOuts[assignment]);
404     }
405   }

407   /**
408    * Reads paths of classes required in first shard. For testing purposes, this relies
409    * on the file system configured for the {@code Zip} library class.
410    */
411   private Set<String> readPaths(String fileName) throws IOException {
412     Set<String> paths = new HashSet<>();
413     BufferedReader reader = null;
414     try {
415       if (filterInputStream == null) {
416         filterInputStream = new FileInputStream(fileName);
417       }
418       reader = new BufferedReader(new InputStreamReader(filterInputStream, UTF_8));
419       String line;
420       while (null != (line = reader.readLine())) {
421         paths.add(fixPath(line));
422       }
423       return paths;
424     } finally {
425       if (reader != null) {
426         reader.close();
427       }
428     }
429   }

431   // TODO(bazel-team): Got this from 'dx'. I'm not sure we need this part. Keep it for now,
432   // to make sure we read the main dex list the exact same way that dx would.
433   private String fixPath(String path) {
434     if (File.separatorChar == '\\') {
435       path = path.replace('\\', '/');
436     }
437     int index = path.lastIndexOf("/./");
438     if (index != -1) {
439       return path.substring(index + 3);
440     }
441     if (path.startsWith("./")) {
442       return path.substring(2);
443     }
444     return path;
445   }
