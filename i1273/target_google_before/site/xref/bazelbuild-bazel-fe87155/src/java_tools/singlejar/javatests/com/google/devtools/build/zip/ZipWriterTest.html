61      cal.set(Calendar.HOUR_OF_DAY, rand.nextInt(24));
62      cal.set(Calendar.MINUTE, rand.nextInt(60));
63      cal.set(Calendar.SECOND, rand.nextInt(30) * 2); // Zip files have 2 second resolution.

66      deflater = new Deflater(Deflater.DEFAULT_COMPRESSION, true);

104   @Test public void testSingleEntry() throws IOException {
105     ZipWriter writer = new ZipWriter(new FileOutputStream(test), UTF_8);
106     byte[] content = "content".getBytes(UTF_8);
107     crc.update(content);
108     ZipFileEntry entry = new ZipFileEntry("foo");
109     entry.setSize(content.length);
110     entry.setCompressedSize(content.length);
111     entry.setCrc(crc.getValue());
112     entry.setTime(cal.getTimeInMillis());
113 
114     writer.putNextEntry(entry);
115     writer.write(content);
116     writer.closeEntry();
117     writer.close();
118 
119     byte[] buf = new byte[128];
120     try (ZipFile zipFile = new ZipFile(test)) {
121       ZipEntry foo = zipFile.getEntry("foo");
122       assertThat(foo.getMethod()).isEqualTo(ZipEntry.STORED);
123       assertThat(foo.getSize()).isEqualTo(content.length);
124       assertThat(foo.getCompressedSize()).isEqualTo(content.length);
125       assertThat(foo.getCrc()).isEqualTo(crc.getValue());
126       assertThat(foo.getTime()).isEqualTo(cal.getTimeInMillis());
127       zipFile.getInputStream(foo).read(buf);
128       assertThat(Bytes.indexOf(buf, content)).isEqualTo(0);
129     }
130   }

132   @Test public void testMultipleEntry() throws IOException {
133     ZipWriter writer = new ZipWriter(new FileOutputStream(test), UTF_8);
134     writer.setComment("file comment");
135 
136     byte[] fooContent = "content".getBytes(UTF_8);
137     crc.update(fooContent);
138     long fooCrc = crc.getValue();
139     ZipFileEntry rawFoo = new ZipFileEntry("foo");
140     rawFoo.setMethod(Compression.STORED);
141     rawFoo.setSize(fooContent.length);
142     rawFoo.setCompressedSize(fooContent.length);
143     rawFoo.setCrc(crc.getValue());
144     rawFoo.setTime(cal.getTimeInMillis());
145     rawFoo.setComment("foo comment");
146 
147     writer.putNextEntry(rawFoo);
148     writer.write(fooContent);
149     writer.closeEntry();
150 
151     byte[] barContent = "stuff".getBytes(UTF_8);
152     byte[] deflatedBarContent = new byte[128];
153     crc.reset();
154     crc.update(barContent);
155     long barCrc = crc.getValue();
156     deflater.setInput(barContent);
157     deflater.finish();
158     int deflatedSize = deflater.deflate(deflatedBarContent);
159     ZipFileEntry rawBar = new ZipFileEntry("bar");
160     rawBar.setMethod(Compression.DEFLATED);
161     rawBar.setSize(barContent.length);
162     rawBar.setCompressedSize(deflatedSize);
163     rawBar.setCrc(barCrc);
164     rawBar.setTime(cal.getTimeInMillis());
165 
166     writer.putNextEntry(rawBar);
167     writer.write(deflatedBarContent, 0, deflatedSize);
168     writer.closeEntry();
169 
170     writer.close();
171 
172     byte[] buf = new byte[128];
173     try (ZipFile zipFile = new ZipFile(test)) {
174       assertThat(zipFile.getComment()).isEqualTo("file comment");
175 
176       ZipEntry foo = zipFile.getEntry("foo");
177       assertThat(foo.getMethod()).isEqualTo(ZipEntry.STORED);
178       assertThat(foo.getSize()).isEqualTo(fooContent.length);
179       assertThat(foo.getCompressedSize()).isEqualTo(fooContent.length);
180       assertThat(foo.getCrc()).isEqualTo(fooCrc);
181       assertThat(foo.getTime()).isEqualTo(cal.getTimeInMillis());
182       assertThat(foo.getComment()).isEqualTo("foo comment");
183       zipFile.getInputStream(foo).read(buf);
184       assertThat(Bytes.indexOf(buf, fooContent)).isEqualTo(0);
185 
186       ZipEntry bar = zipFile.getEntry("bar");
187       assertThat(bar.getMethod()).isEqualTo(ZipEntry.DEFLATED);
188       assertThat(bar.getSize()).isEqualTo(barContent.length);
189       assertThat(bar.getCompressedSize()).isEqualTo(deflatedSize);
190       assertThat(bar.getCrc()).isEqualTo(barCrc);
191       assertThat(bar.getTime()).isEqualTo(cal.getTimeInMillis());
192       zipFile.getInputStream(bar).read(buf);
193       assertThat(Bytes.indexOf(buf, barContent)).isEqualTo(0);
194     }
195   }

215   @Test public void testRawZipEntry() throws IOException {
216     ZipWriter writer = new ZipWriter(new FileOutputStream(test), UTF_8);
217     byte[] content = "content".getBytes(UTF_8);
218     crc.update(content);
219     ZipFileEntry entry = new ZipFileEntry("foo");
220     entry.setVersion((short) 1);
221     entry.setVersionNeeded((short) 2);
222     entry.setSize(content.length);
223     entry.setCompressedSize(content.length);
224     entry.setCrc(crc.getValue());
225     entry.setTime(cal.getTimeInMillis());
226     entry.setFlags(ZipUtil.get16(new byte[]{ 0x08, 0x00 }, 0));
227     entry.setInternalAttributes(ZipUtil.get16(new byte[]{ 0x34, 0x12 }, 0));
228     entry.setExternalAttributes(ZipUtil.get32(new byte[]{ 0x0a,  0x09, 0x78, 0x56 }, 0));
229     entry.setLocalHeaderOffset(rand.nextInt(Integer.MAX_VALUE));
230 
231     writer.putNextEntry(entry);
232     writer.write(content);
233     writer.closeEntry();
234     writer.close();
235 
236     byte[] buf = new byte[128];
237     try (ZipFile zipFile = new ZipFile(test)) {
238       ZipEntry foo = zipFile.getEntry("foo");
239       assertThat(foo.getMethod()).isEqualTo(ZipEntry.STORED);
240       assertThat(foo.getSize()).isEqualTo(content.length);
241       assertThat(foo.getCompressedSize()).isEqualTo(content.length);
242       assertThat(foo.getCrc()).isEqualTo(crc.getValue());
243       assertThat(foo.getTime()).isEqualTo(cal.getTimeInMillis());
244       zipFile.getInputStream(foo).read(buf);
245       assertThat(Bytes.indexOf(buf, content)).isEqualTo(0);
246     }
247 
248     try (ZipReader zipFile = new ZipReader(test)) {
249       ZipFileEntry foo = zipFile.getEntry("foo");
250       // Versions should be increased to minimum required for STORED compression.
251       assertThat(foo.getVersion()).isEqualTo((short) 0xa);
252       assertThat(foo.getVersionNeeded()).isEqualTo((short) 0xa);
253       assertThat(foo.getFlags()).isEqualTo((short) 0); // Data descriptor bit should be cleared.
254       assertThat(foo.getInternalAttributes()).isEqualTo((short) 4660);
255       assertThat(foo.getExternalAttributes()).isEqualTo(1450707210);
256     }
257   }

264     writer.endPrefixFile();
265 
266     byte[] content = "content".getBytes(UTF_8);
267     crc.update(content);
268     ZipFileEntry entry = new ZipFileEntry("foo");
269     entry.setSize(content.length);
270     entry.setCompressedSize(content.length);
271     entry.setCrc(crc.getValue());
272     entry.setTime(cal.getTimeInMillis());
273 
274     writer.putNextEntry(entry);
275     writer.write(content);
276     writer.closeEntry();
277     writer.close();
278 
279     byte[] buf = new byte[128];
280     try (ZipFile zipFile = new ZipFile(test)) {
281       ZipEntry foo = zipFile.getEntry("foo");
282       assertThat(foo.getMethod()).isEqualTo(ZipEntry.STORED);
283       assertThat(foo.getSize()).isEqualTo(content.length);
284       assertThat(foo.getCompressedSize()).isEqualTo(content.length);

286       assertThat(foo.getTime()).isEqualTo(cal.getTimeInMillis());
287       zipFile.getInputStream(foo).read(buf);
288       assertThat(Bytes.indexOf(buf, content)).isEqualTo(0);
289     }
290 
291     Process pr = new ProcessBuilder("chmod", "750", test.getAbsolutePath()).start();
292     pr.waitFor();
293     pr = new ProcessBuilder(test.getAbsolutePath()).start();
294     pr.getInputStream().read(buf);
295     pr.waitFor();
296     assertThat(Bytes.indexOf(buf, "hello world".getBytes(UTF_8))).isEqualTo(0);
297   }
