40    @Test public void testConstructFromList() {
41      ExtraData[] extras = new ExtraData[] {
42          new ExtraData((short) 0xcafe, new byte[] { 0x00, 0x11, 0x22 }),
43          new ExtraData((short) 0xbeef, new byte[] { 0x33, 0x44, 0x55 })
44      };
45  
46      ExtraDataList extra = new ExtraDataList(extras);
47      // Expect 0xcafe 0x0003 0x00 0x11 0x22, 0xbeef 0x0003 0x33 0x44 0x55 in little endian
48      assertThat(extra.getBytes()).isEqualTo(new byte[] { (byte) 0xfe, (byte) 0xca, 0x03, 0x00,
49          0x00, 0x11, 0x22, (byte) 0xef, (byte) 0xbe, 0x03, 0x00, 0x33, 0x44, 0x55 });
50      assertThat(extra.contains((short) 0xcafe)).isTrue();
51      assertThat(extra.contains((short) 0xbeef)).isTrue();
52  
53      ExtraData cafe = extra.remove((short) 0xcafe);
54      // Expect 0xbeef 0x0003 0x33 0x44 0x55 in little endian
55      assertThat(extra.getBytes()).isEqualTo(new byte[] { (byte) 0xef, (byte) 0xbe, 0x03, 0x00, 0x33,
56          0x44, 0x55 });
57      assertThat(extra.contains((short) 0xcafe)).isFalse();
58      assertThat(extra.contains((short) 0xbeef)).isTrue();
59  
60      extra.add(cafe);
61      // Expect 0xbeef 0x0003 0x33 0x44 0x55, 0xcafe 0x0003 0x00 0x11 0x22 in little endian
62      assertThat(extra.getBytes()).isEqualTo(new byte[] { (byte) 0xef, (byte) 0xbe, 0x03, 0x00, 0x33,
63          0x44, 0x55, (byte) 0xfe, (byte) 0xca, 0x03, 0x00, 0x00, 0x11, 0x22 });
64      assertThat(extra.contains((short) 0xcafe)).isTrue();
65      assertThat(extra.contains((short) 0xbeef)).isTrue();
66  
67      ExtraData beef = extra.get((short) 0xbeef);
68      assertThat(beef.getId()).isEqualTo((short) 0xbeef);
69    }

100   @Test public void testByteStream() throws IOException {
101     byte[] buffer = new byte[] { (byte) 0xfe, (byte) 0xca, 0x03, 0x00, 0x00, 0x11, 0x22,
102         (byte) 0xef, (byte) 0xbe, 0x03, 0x00, 0x33, 0x44, 0x55 };
103 
104     ExtraDataList extra = new ExtraDataList(buffer);
105     byte[] bytes = new byte[7];
106     InputStream in = extra.getByteStream();
107     in.read(bytes);
108     // Expect 0xcafe 0x0003 0x00 0x11 0x22 in little endian
109     assertThat(bytes).isEqualTo(new byte[] { (byte) 0xfe, (byte) 0xca, 0x03, 0x00, 0x00, 0x11,
110         0x22 });
111     in.read(bytes);
112     // Expect 0xbeef 0x0003 0x33 0x44 0x55 in little endian
113     assertThat(bytes).isEqualTo(new byte[] { (byte) 0xef, (byte) 0xbe, 0x03, 0x00, 0x33, 0x44,
114         0x55 });
115     assertThat(in.read(bytes)).isEqualTo(-1);
116   }
