41  /**
42   * A ZIP file reader.
43   *
44   * <p>This class provides entry data in the form of {@link ZipFileEntry}, which provides more detail
45   * about the entry than the JDK equivalent {@link ZipEntry}. In addition to providing
46   * {@link InputStream}s for entries, similar to JDK {@link ZipFile#getInputStream(ZipEntry)}, it
47   * also provides access to the raw byte entry data via {@link #getRawInputStream(ZipFileEntry)}.
48   *
49   * <p>Using the raw access capabilities allows for more efficient ZIP file processing, such as
50   * merging, by not requiring each entry's data to be decompressed when read.
51   *
52   * <p><em>NOTE:</em> The entries are read from the central directory. If the entry is not listed
53   * there, it will not be returned from {@link #entries()} or {@link #getEntry(String)}.
54   */
55  public class ZipReader implements Closeable, AutoCloseable {
56  
57    /** An input stream for reading the file data of a ZIP file entry. */
58    private class ZipEntryInputStream extends InputStream {
59      private InputStream stream;
60      private long rem;
61  
62      /**
63       * Opens an input stream for reading at the beginning of the ZIP file entry's content.
64       *
65       * @param zipEntry the ZIP file entry to open the input stream for
66       * @throws ZipException if a ZIP format error has occurred
67       * @throws IOException if an I/O error has occurred
68       */
69      private ZipEntryInputStream(ZipFileEntry zipEntry) throws IOException {
70        stream = getStreamAt(zipEntry.getLocalHeaderOffset());
71  
72        byte[] fileHeader = new byte[LocalFileHeader.FIXED_DATA_SIZE];
73        ZipUtil.readFully(stream, fileHeader);
74  
75        if (!ZipUtil.arrayStartsWith(fileHeader,
76            ZipUtil.intToLittleEndian(LocalFileHeader.SIGNATURE))) {
77          throw new ZipException(String.format("The file '%s' is not a correctly formatted zip file: "
78              + "Expected a File Header at file offset %d, but was not present.",
79              file.getName(), zipEntry.getLocalHeaderOffset()));
80        }
81  
82        int nameLength = ZipUtil.getUnsignedShort(fileHeader,
83            LocalFileHeader.FILENAME_LENGTH_OFFSET);
84        int extraFieldLength = ZipUtil.getUnsignedShort(fileHeader,
85            LocalFileHeader.EXTRA_FIELD_LENGTH_OFFSET);
86        ZipUtil.readFully(stream, new byte[nameLength + extraFieldLength]);
87        rem = zipEntry.getSize();
88        if (zipEntry.getMethod() == Compression.DEFLATED) {
89          stream = new InflaterInputStream(stream, new Inflater(true));
90        }
91      }
92  
93      @Override public int available() throws IOException {
94        return rem > Integer.MAX_VALUE ? Integer.MAX_VALUE : (int) rem;
95      }
96  
97      @Override public void close() throws IOException {
98      }
99  
100     @Override public void mark(int readlimit) {
101     }
102 
103     @Override public boolean markSupported() {
104       return false;
105     }
106 
107     @Override public int read() throws IOException {
108       byte[] b = new byte[1];
109       if (read(b, 0, 1) == 1) {
110         return b[0] & 0xff;
111       } else {
112         return -1;
113       }
114     }
115 
116     @Override public int read(byte[] b) throws IOException {
117       return read(b, 0, b.length);
118     }
119 
120     @Override public int read(byte[] b, int off, int len) throws IOException {
121       if (rem == 0) {
122         return -1;
123       }
124       if (len > rem) {
125         len = available();
126       }
127       len = stream.read(b, off, len);
128       rem -= len;
129       return len;
130     }
131 
132     @Override public long skip(long n) throws IOException {
133       if (n > rem) {
134         n = rem;
135       }
136       n = stream.skip(n);
137       rem -= n;
138       return n;
139     }
140 
141     @Override public void reset() throws IOException {
142       throw new IOException("Reset is not supported on this type of stream.");
143     }
144   }

159       stream = getStreamAt(zipEntry.getLocalHeaderOffset());

261   /**
262    * Opens a zip file for raw acceess.
263    *
264    * @param file the zip file
265    * @param charset the charset to use to decode the entry names and comments
266    * @param strictEntries force parsing to use the number of entries recorded in the end of
267    *     central directory as the correct value, not as an estimate
268    * @throws ZipException if a ZIP format error has occurred
269    * @throws IOException if an I/O error has occurred
270    */
271   public ZipReader(File file, Charset charset, boolean strictEntries) throws IOException {
272     if (file == null || charset == null) {
273       throw new NullPointerException();
274     }
275     this.file = file;
276     this.in = new RandomAccessFile(file, "r");
277     this.zipData = new ZipFileData(charset);
278     readCentralDirectory(strictEntries);
279   }

309   /**
310    * Returns an input stream for reading the contents of the specified ZIP file entry.
311    *
312    * <p>Closing this ZIP file will, in turn, close all input streams that have been returned by
313    * invocations of this method.
314    *
315    * @param entry the ZIP file entry
316    * @return the input stream for reading the contents of the specified zip file entry
317    * @throws ZipException if a ZIP format error has occurred
318    * @throws IOException if an I/O error has occurred
319    */
320   public InputStream getInputStream(ZipFileEntry entry) throws IOException {
321     if (!zipData.getEntry(entry.getName()).equals(entry)) {
322       throw new ZipException(String.format(
323           "Zip file '%s' does not contain the requested entry '%s'.", file.getName(),
324           entry.getName()));
325     }
326     return new ZipEntryInputStream(entry);
327   }

329   /**
330    * Returns an input stream for reading the raw contents of the specified ZIP file entry.
331    *
332    * <p><em>NOTE:</em> No inflating will take place; The data read from the input stream will be
333    * the exact byte content of the ZIP file entry on disk.
334    *
335    * <p>Closing this ZIP file will, in turn, close all input streams that have been returned by
336    * invocations of this method.
337    *
338    * @param entry the ZIP file entry
339    * @return the input stream for reading the contents of the specified zip file entry
340    * @throws ZipException if a ZIP format error has occurred
341    * @throws IOException if an I/O error has occurred
342    */
343   public InputStream getRawInputStream(ZipFileEntry entry) throws IOException {
344     if (!zipData.getEntry(entry.getName()).equals(entry)) {
345       throw new ZipException(String.format(
346           "Zip file '%s' does not contain the requested entry '%s'.", file.getName(),
347           entry.getName()));
348     }
349     return new RawZipEntryInputStream(entry);
350   }

362   /**
363    * Finds, reads and parses ZIP file entries from the central directory.
364    *
365    * @param strictEntries force parsing to use the number of entries recorded in the end of
366    *     central directory as the correct value, not as an estimate
367    * @throws ZipException if a ZIP format error has occurred
368    * @throws IOException if an I/O error has occurred
369    */
370   private void readCentralDirectory(boolean strictEntries) throws IOException {
371     long eocdLocation = findEndOfCentralDirectoryRecord();
372     InputStream stream = getStreamAt(eocdLocation);
373     EndOfCentralDirectoryRecord.read(stream, zipData);
374 
375     if (zipData.isMaybeZip64()) {
376       try {
377         stream = getStreamAt(eocdLocation - Zip64EndOfCentralDirectoryLocator.FIXED_DATA_SIZE);
378         Zip64EndOfCentralDirectoryLocator.read(stream, zipData);
379 
380         stream = getStreamAt(zipData.getZip64EndOfCentralDirectoryOffset());
381         Zip64EndOfCentralDirectory.read(stream, zipData);
382       } catch (ZipException e) {
383         // expected if not in Zip64 format
384       }
385     }
386 
387     if (zipData.isZip64() || strictEntries) {
388       // If in Zip64 format or using strict entry numbers, use the parsed information as is to read
389       // the central directory file headers.
390       readCentralDirectoryFileHeaders(zipData.getExpectedEntries(),
391           zipData.getCentralDirectoryOffset());
392     } else {
393       // If not in Zip64 format, compute central directory offset by end of central directory record
394       // offset and central directory size to allow reading large non-compliant Zip32 directories.
395       long centralDirectoryOffset = eocdLocation - zipData.getCentralDirectorySize();
396       // If the lower 4 bytes match, the above calculation is correct; otherwise fallback to
397       // reported offset.
398       if ((int) centralDirectoryOffset == (int) zipData.getCentralDirectoryOffset()) {
399         readCentralDirectoryFileHeaders(centralDirectoryOffset);
400       } else {
401         readCentralDirectoryFileHeaders(zipData.getExpectedEntries(),
402             zipData.getCentralDirectoryOffset());
403       }
404     }
405   }

407   /**
408    * Looks for the target sub array in the buffer scanning backwards starting at offset. Returns the
409    * index where the target is found or -1 if not found.
410    *
411    * @param target the sub array to find
412    * @param buffer the array to scan
413    * @param offset the index of where to begin scanning
414    * @return the index of target within buffer or -1 if not found
415    */
416   private int scanBackwards(byte[] target, byte[] buffer, int offset) {
417     int start = Math.min(offset, buffer.length - target.length);
418     for (int i = start; i >= 0; i--) {
419       for (int j = 0; j < target.length; j++) {
420         if (buffer[i + j] != target[j]) {
421           break;
422         } else if (j == target.length - 1) {
423           return i;
424         }
425       }
426     }
427     return -1;
428   }

430   /**
431    * Finds the file offset of the end of central directory record.
432    *
433    * @return the file offset of the end of central directory record
434    * @throws ZipException if a ZIP format error has occurred
435    * @throws IOException if an I/O error has occurred
436    */
437   private long findEndOfCentralDirectoryRecord() throws IOException {
438     byte[] signature = ZipUtil.intToLittleEndian(EndOfCentralDirectoryRecord.SIGNATURE);
439     byte[] buffer = new byte[(int) Math.min(64, in.length())];
440     int readLength = buffer.length;
441     if (readLength < EndOfCentralDirectoryRecord.FIXED_DATA_SIZE) {
442       throw new ZipException(String.format("Zip file '%s' is malformed. It does not contain an end"
443           + " of central directory record.", file.getName()));
444     }
445 
446     long offset = in.length() - buffer.length;
447     while (offset >= 0) {
448       in.seek(offset);
449       in.readFully(buffer, 0, readLength);
450       int signatureLocation = scanBackwards(signature, buffer, buffer.length);
451       while (signatureLocation != -1) {
452         long eocdSize = in.length() - offset - signatureLocation;
453         if (eocdSize >= EndOfCentralDirectoryRecord.FIXED_DATA_SIZE) {
454           int commentLength = ZipUtil.getUnsignedShort(buffer, signatureLocation
455               + EndOfCentralDirectoryRecord.COMMENT_LENGTH_OFFSET);
456           long readCommentLength = eocdSize - EndOfCentralDirectoryRecord.FIXED_DATA_SIZE;
457           if (commentLength == readCommentLength) {
458             return offset + signatureLocation;
459           }
460         }
461         signatureLocation = scanBackwards(signature, buffer, signatureLocation - 1);
462       }
463       readLength = buffer.length - 3;
464       buffer[buffer.length - 3] = buffer[0];
465       buffer[buffer.length - 2] = buffer[1];
466       buffer[buffer.length - 1] = buffer[2];
467       offset -= readLength;
468     }
469     throw new ZipException(String.format("Zip file '%s' is malformed. It does not contain an end"
470         + " of central directory record.", file.getName()));
471   }

473   /**
474    * Reads and parses ZIP file entries from the central directory.
475    *
476    * @param count the number of entries in the central directory
477    * @param fileOffset the file offset of the start of the central directory
478    * @throws ZipException if a ZIP format error has occurred
479    * @throws IOException if an I/O error has occurred
480    */
481   private void readCentralDirectoryFileHeaders(long count, long fileOffset) throws IOException {
482     InputStream centralDirectory = getStreamAt(fileOffset);
483     for (long i = 0; i < count; i++) {
484       ZipFileEntry entry = CentralDirectoryFileHeader.read(centralDirectory, zipData.getCharset());
485       zipData.addEntry(entry);
486     }
487   }
488 
489   /**
490    * Reads and parses ZIP file entries from the central directory.
491    *
492    * @param fileOffset the file offset of the start of the central directory
493    * @throws ZipException if a ZIP format error has occurred
494    * @throws IOException if an I/O error has occurred
495    */
496   private void readCentralDirectoryFileHeaders(long fileOffset) throws IOException {
497     CountingInputStream centralDirectory = new CountingInputStream(getStreamAt(fileOffset));
498     while (centralDirectory.getCount() < zipData.getCentralDirectorySize()) {
499       ZipFileEntry entry = CentralDirectoryFileHeader.read(centralDirectory, zipData.getCharset());
500       zipData.addEntry(entry);
501     }
502   }

504   /**
505    * Returns a new {@link InputStream} positioned at fileOffset.
506    *
507    * @throws IOException if an I/O error has occurred
508    */
509   protected InputStream getStreamAt(long fileOffset) throws IOException {
510     return new BufferedInputStream(Channels.newInputStream(in.getChannel().position(fileOffset)));
