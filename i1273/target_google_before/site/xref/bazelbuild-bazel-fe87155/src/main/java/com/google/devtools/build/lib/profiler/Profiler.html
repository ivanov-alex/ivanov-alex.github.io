41  /**
42   * Blaze internal profiler. Provides facility to report various Blaze tasks and
43   * store them (asynchronously) in the file for future analysis.
44   * <p>
45   * Implemented as singleton so any caller should use Profiler.instance() to
46   * obtain reference.
47   * <p>
48   * Internally, profiler uses two data structures - ThreadLocal task stack to track
49   * nested tasks and single ConcurrentLinkedQueue to gather all completed tasks.
50   * <p>
51   * Also, due to the nature of the provided functionality (instrumentation of all
52   * Blaze components), build.lib.profiler package will be used by almost every
53   * other Blaze package, so special attention should be paid to avoid any
54   * dependencies on the rest of the Blaze code, including build.lib.util and
55   * build.lib.vfs. This is important because build.lib.util and build.lib.vfs
56   * contain Profiler invocations and any dependency on those two packages would
57   * create circular relationship.
58   * <p>
59   * All gathered instrumentation data will be stored in the file. Please, note,
60   * that while file format is described here it is considered internal and can
61   * change at any time. For scripting, using blaze analyze-profile --dump=raw
62   * would be more robust and stable solution.
63   * <p>
64   * <pre>
65   * Profiler file consists of the deflated stream with following overall structure:
66   *   HEADER
67   *   TASK_TYPE_TABLE
68   *   TASK_RECORD...
69   *   EOF_MARKER
70   *
71   * HEADER:
72   *   int32: magic token (Profiler.MAGIC)
73   *   int32: version format (Profiler.VERSION)
74   *   string: file comment
75   *
76   * TASK_TYPE_TABLE:
77   *   int32: number of type names below
78   *   string... : type names. Each of the type names is assigned id according to
79   *               their position in this table starting from 0.
80   *
81   * TASK_RECORD:
82   *   int32 size: size of the encoded task record
83   *   byte[size] encoded_task_record:
84   *     varint64: thread id - as was returned by Thread.getId()
85   *     varint32: task id - starting from 1.
86   *     varint32: parent task id for subtasks or 0 for root tasks
87   *     varint64: start time in ns, relative to the Profiler.start() invocation
88   *     varint64: task duration in ns
89   *     byte:     task type id (see TASK_TYPE_TABLE)
90   *     varint32: description string index incremented by 1 (>0) or 0 this is
91   *               a first occurrence of the description string
92   *     AGGREGATED_STAT...: remainder of the field (if present) represents
93   *                         aggregated stats for that task
94   *   string: *optional* description string, will appear only if description
95   *           string index above was 0. In that case this string will be
96   *           assigned next sequential id so every unique description string
97   *           will appear in the file only once - after that it will be
98   *           referenced by id.
99   *
100  * AGGREGATE_STAT:
101  *   byte:     stat type
102  *   varint32: total number of subtask invocations
103  *   varint64: cumulative duration of subtask invocations in ns.
104  *
105  * EOF_MARKER:
106  *   int64: -1 - please note that this corresponds to the thread id in the
107  *               TASK_RECORD which is always > 0
108  * </pre>
109  *
110  * @see ProfilerTask enum for recognized task types.
111  */
112 //@ThreadSafe - commented out to avoid cyclic dependency with lib.util package

133   /**
134    * A task that was very slow.
135    */
136   public final class SlowTask implements Comparable<SlowTask> {
137     final long durationNanos;
138     final Object object;
139     ProfilerTask type;
140 
141     private SlowTask(TaskData taskData) {
142       this.durationNanos = taskData.duration;
143       this.object = taskData.object;
144       this.type = taskData.type;
145     }
146 
147     @Override
148     public int compareTo(SlowTask other) {
149       long delta = durationNanos - other.durationNanos;
150       if (delta < 0) {  // Very clumsy
151         return -1;
152       } else if (delta > 0) {
153         return 1;
154       } else {
155         return 0;
156       }
157     }
158 
159     public long getDurationNanos() {
160       return durationNanos;
161     }
162 
163     public String getDescription() {
164       return toDescription(object);
165     }
166 
167     public ProfilerTask getType() {
168       return type;
169     }
170   }

172   /**
173    * Container for the single task record.
174    * Should never be instantiated directly - use TaskStack.create() instead.
175    *
176    * Class itself is not thread safe, but all access to it from Profiler
177    * methods is.
178    */
179   //@ThreadCompatible - commented out to avoid cyclic dependency with lib.util.
180   private final class TaskData {
181     final long threadId;
182     final long startTime;
183     long duration = 0L;
184     final int id;
185     final int parentId;
186     int[] counts; // number of invocations per ProfilerTask type
187     long[] durations; // time spend in the task per ProfilerTask type
188     final ProfilerTask type;
189     final Object object;
190 
191     TaskData(long startTime, TaskData parent,
192              ProfilerTask eventType, Object object) {
193       threadId = Thread.currentThread().getId();
194       counts = null;
195       durations = null;
196       id = taskId.incrementAndGet();
197       parentId = (parent == null  ? 0 : parent.id);
198       this.startTime = startTime;
199       this.type = eventType;
200       this.object = Preconditions.checkNotNull(object);
201     }
202 
203     /**
204      * Aggregates information about an *immediate* subtask.
205      */
206     public void aggregateChild(ProfilerTask type, long duration) {
207       int index = type.ordinal();
208       if (counts == null) {
209         // one entry for each ProfilerTask type
210         counts = new int[TASK_COUNT];
211         durations = new long[TASK_COUNT];
212       }
213       counts[index]++;
214       durations[index] += duration;
215     }
216 
217     @Override
218     public String toString() {
219       return "Thread " + threadId + ", task " + id + ", type " + type + ", " + object;
220     }
221   }

223   /**
224    * Tracks nested tasks for each thread.
225    *
226    * java.util.ArrayDeque is the most efficient stack implementation in the
227    * Java Collections Framework (java.util.Stack class is older synchronized
228    * alternative). It is, however, used here strictly for LIFO operations.
229    * However, ArrayDeque is 1.6 only. For 1.5 best approach would be to utilize
230    * ArrayList and emulate stack using it.
231    */
232   //@ThreadSafe - commented out to avoid cyclic dependency with lib.util.
233   private final class TaskStack extends ThreadLocal<List<TaskData>> {
234 
235     @Override
236     public List<TaskData> initialValue() {
237       return new ArrayList<>();
238     }
239 
240     public TaskData peek() {
241       List<TaskData> list = get();
242       if (list.isEmpty()) {
243         return null;
244       }
245       return list.get(list.size() - 1);
246     }
247 
248     public TaskData pop() {
249       List<TaskData> list = get();
250       return list.remove(list.size() - 1);
251     }
252 
253     public boolean isEmpty() {
254       return get().isEmpty();
255     }
256 
257     public void push(ProfilerTask eventType, Object object) {
258       get().add(create(clock.nanoTime(), eventType, object));
259     }
260 
261     public TaskData create(long startTime, ProfilerTask eventType, Object object) {
262       return new TaskData(startTime, peek(), eventType, object);
263     }
264 
265     @Override
266     public String toString() {
267       StringBuilder builder = new StringBuilder(
268           "Current task stack for thread " + Thread.currentThread().getName() + ":\n");
269       List<TaskData> list = get();
270       for (int i = list.size() - 1; i >= 0; i--) {
271         builder.append(list.get(i));
272         builder.append("\n");
273       }
274       return builder.toString();
275     }
276   }

284   /**
285    * Implements datastore for object description indices. Intended to be used
286    * only by the Profiler.save() method.
287    */
288   //@ThreadCompatible - commented out to avoid cyclic dependency with lib.util.
289   private final class ObjectDescriber {
290     private Map<Object, Integer> descMap = new IdentityHashMap<>(2000);
291     private int indexCounter = 0;
292 
293     ObjectDescriber() { }
294 
295     int getDescriptionIndex(Object object) {
296       Integer index = descMap.get(object);
297       return (index != null) ? index : -1;
298     }
299 
300     String getDescription(Object object) {
301       String description = toDescription(object);
302 
303       Integer oldIndex = descMap.put(object, indexCounter++);
304       // Do not use Preconditions class below due to the rather expensive
305       // toString() calls used in the message.
306       if (oldIndex != null) {
307         throw new IllegalStateException(" Object '" + description + "' @ "
308             + System.identityHashCode(object) + " already had description index "
309             + oldIndex + " while assigning index " + descMap.get(object));
310       } else if (description.length() > 20000) {
311         // Note size 64k byte limitation in DataOutputStream#writeUTF().
312         description = description.substring(0, 20000);
313       }
314       return description;
315     }
316 
317     boolean isUnassigned(int index) {
318       return (index < 0);
319     }
320   }

322   /**
323    * Aggregator class that keeps track of the slowest tasks of the specified type.
324    *
325    * <p><code>priorityQueues</p> is sharded so that all threads need not compete for the same
326    * lock if they do the same operation at the same time. Access to the individual queues is
327    * synchronized on the queue objects themselves.
328    */
329   private final class SlowestTaskAggregator {
330     private static final int SHARDS = 16;
331     private final int size;
332 
333     @SuppressWarnings({"unchecked", "rawtypes"})
334     private final PriorityQueue<SlowTask>[] priorityQueues = new PriorityQueue[SHARDS];
335 
336     SlowestTaskAggregator(int size) {
337       this.size = size;
338 
339       for (int i = 0; i < SHARDS; i++) {
340         priorityQueues[i] = new PriorityQueue<>(size + 1);
341       }
342     }
343 
344     // @ThreadSafe
345     void add(TaskData taskData) {
346       PriorityQueue<SlowTask> queue =
347           priorityQueues[(int) (Thread.currentThread().getId() % SHARDS)];
348       synchronized (queue) {
349         if (queue.size() == size) {
350           // Optimization: check if we are faster than the fastest element. If we are, we would
351           // be the ones to fall off the end of the queue, therefore, we can safely return early.
352           if (queue.peek().getDurationNanos() > taskData.duration) {
353             return;
354           }
355 
356           queue.add(new SlowTask(taskData));
357           queue.remove();
358         } else {
359           queue.add(new SlowTask(taskData));
360         }
361       }
362     }
363 
364     // @ThreadSafe
365     void clear() {
366       for (int i = 0; i < SHARDS; i++) {
367         PriorityQueue<SlowTask> queue = priorityQueues[i];
368         synchronized (queue) {
369           queue.clear();
370         }
371       }
372     }
373 
374     // @ThreadSafe
375     Iterable<SlowTask> getSlowestTasks() {
376       // This is slow, but since it only happens during the end of the invocation, it's OK
377       PriorityQueue<SlowTask> merged = new PriorityQueue<>(size * SHARDS);
378       for (int i = 0; i < SHARDS; i++) {
379         PriorityQueue<SlowTask> queue = priorityQueues[i];
380         synchronized (queue) {
381           merged.addAll(queue);
382         }
383       }
384 
385       while (merged.size() > size) {
386         merged.remove();
387       }
388 
389       return merged;
390     }
391   }

393   /**
394    * Which {@link ProfilerTask}s are profiled.
395    */
396   public enum ProfiledTaskKinds {
397     /**
398      * Do not profile anything.
399      *
400      * <p>Performance is best with this case, but we lose critical path analysis and slowest
401      * operation tracking.
402      */
403     NONE {
404       @Override
405       boolean isProfiling(ProfilerTask type) {
406         return false;
407       }
408     },
409 
410     /**
411      * Profile on a few, known-to-be-slow tasks.
412      *
413      * <p>Performance is somewhat decreased in comparison to {@link #NONE}, but we still track the
414      * slowest operations (VFS).
415      */
416     SLOWEST {
417       @Override
418       boolean isProfiling(ProfilerTask type) {
419         return type.collectsSlowestInstances();
420       }
421     },
422 
423     /**
424      * Profile all tasks.
425      *
426      * <p>This is in use when {@code --profile} is specified.
427      */
428     ALL {
429       @Override
430       boolean isProfiling(ProfilerTask type) {
431         return true;
432       }
433     };
434 
435     /** Whether the Profiler collects data for the given task type. */
436     abstract boolean isProfiling(ProfilerTask type);
437   }

478   /**
479    * Enable profiling.
480    *
481    * <p>Subsequent calls to beginTask/endTask will be recorded
482    * in the provided output stream. Please note that stream performance is
483    * extremely important and buffered streams should be utilized.
484    *
485    * @param profiledTaskKinds which kinds of {@link ProfilerTask}s to track
486    * @param stream output stream to store profile data. Note: passing unbuffered stream object
487    *     reference may result in significant performance penalties
488    * @param comment a comment to insert in the profile data
489    * @param recordAllDurations iff true, record all tasks regardless of their duration; otherwise
490    *     some tasks may get aggregated if they finished quick enough
491    * @param clock a {@code BlazeClock.instance()}
492    * @param execStartTimeNanos execution start time in nanos obtained from {@code clock.nanoTime()}
493    */
494   public synchronized void start(ProfiledTaskKinds profiledTaskKinds, OutputStream stream,
495       String comment, boolean recordAllDurations, Clock clock, long execStartTimeNanos)
496       throws IOException {
497     Preconditions.checkState(!isActive(), "Profiler already active");
498     taskStack = new TaskStack();
499     taskQueue = new ConcurrentLinkedQueue<>();
500     describer = new ObjectDescriber();
501 
502     this.profiledTaskKinds = profiledTaskKinds;
503     this.clock = clock;
504 
505     // sanity check for current limitation on the number of supported types due
506     // to using enum.ordinal() to store them instead of EnumSet for performance reasons.
507     Preconditions.checkState(TASK_COUNT < 256,
508         "The profiler implementation supports only up to 255 different ProfilerTask values.");
509 
510     // reset state for the new profiling session
511     taskId.set(0);
512     this.recordAllDurations = recordAllDurations;
513     this.saveException = null;
514     if (stream != null) {
515       this.timer = new Timer("ProfilerTimer", true);
516       // Wrapping deflater stream in the buffered stream proved to reduce CPU consumption caused by
517       // the save() method. Values for buffer sizes were chosen by running small amount of tests
518       // and identifying point of diminishing returns - but I have not really tried to optimize
519       // them.
520       this.out = new DataOutputStream(new BufferedOutputStream(new DeflaterOutputStream(
521           stream, new Deflater(Deflater.BEST_SPEED, false), 65536), 262144));
522 
523       this.out.writeInt(MAGIC); // magic
524       this.out.writeInt(VERSION); // protocol_version
525       this.out.writeUTF(comment);
526       // ProfileTask.values() method sorts enums using their ordinal() value, so
527       // there there is no need to store ordinal() value for each entry.
528       this.out.writeInt(TASK_COUNT);
529       for (ProfilerTask type : ProfilerTask.values()) {
530         this.out.writeUTF(type.toString());
531       }
532 
533       // Start save thread
534       timer.schedule(new TimerTask() {
535         @Override public void run() { save(); }
536       }, SAVE_DELAY, SAVE_DELAY);
537     } else {
538       this.out = null;
539     }
540 
541     // activate profiler
542     profileStartTime = execStartTimeNanos;
543   }

545   public synchronized Iterable<SlowTask> getSlowestTasks() {
546     List<Iterable<SlowTask>> slowestTasksByType = new ArrayList<>();
547 
548     for (SlowestTaskAggregator aggregator : slowestTasks) {
549       if (aggregator != null) {
550         slowestTasksByType.add(aggregator.getSlowestTasks());
551       }
552     }
553 
554     return Iterables.concat(slowestTasksByType);
555   }

557   /**
558    * Disable profiling and complete profile file creation.
559    * Subsequent calls to beginTask/endTask will no longer
560    * be recorded in the profile.
561    */
562   public synchronized void stop() throws IOException {
563     if (saveException != null) {
564       throw saveException;
565     }
566     if (!isActive()) {
567       return;
568     }
569     // Log a final event to update the duration of ProfilePhase.FINISH.
570     logEvent(ProfilerTask.INFO, "Finishing");
571     save();
572     clear();
573 
574     for (SlowestTaskAggregator aggregator : slowestTasks) {
575       if (aggregator != null) {
576         aggregator.clear();
577       }
578     }
579 
580     if (saveException != null) {
581       throw saveException;
582     }
583     if (out != null) {
584       out.writeInt(EOF_MARKER);
585       out.close();
586       out = null;
587     }
588   }

601   /**
602    * Saves all gathered information from taskQueue queue to the file.
603    * Method is invoked internally by the Timer-based thread and at the end of
604    * profiling session.
605    */
606   private synchronized void save() {
607     if (out == null) {
608       return;
609     }
610     try {
611       // Allocate the sink once to avoid GC
612       ByteBuffer sink = ByteBuffer.allocate(1024);
613       while (!taskQueue.isEmpty()) {
614         sink.clear();
615         TaskData data = taskQueue.poll();
616 
617         VarInt.putVarLong(data.threadId, sink);
618         VarInt.putVarInt(data.id, sink);
619         VarInt.putVarInt(data.parentId, sink);
620         VarInt.putVarLong(data.startTime - profileStartTime, sink);
621         VarInt.putVarLong(data.duration, sink);
622 
623         // To save space (and improve performance), convert all description
624         // strings to the canonical object and use IdentityHashMap to assign
625         // unique numbers for each string.
626         int descIndex = describer.getDescriptionIndex(data.object);
627         VarInt.putVarInt(descIndex + 1, sink); // Add 1 to avoid encoding negative values.
628 
629         // Save types using their ordinal() value
630         sink.put((byte) data.type.ordinal());
631 
632         // Save aggregated data stats.
633         if (data.counts != null) {
634           for (int i = 0; i < TASK_COUNT; i++) {
635             if (data.counts[i] > 0) {
636               sink.put((byte) i); // aggregated type ordinal value
637               VarInt.putVarInt(data.counts[i], sink);
638               VarInt.putVarLong(data.durations[i], sink);
639             }
640           }
641         }
642 
643         this.out.writeInt(sink.position());
644         this.out.write(sink.array(), 0, sink.position());
645         if (describer.isUnassigned(descIndex)) {
646           this.out.writeUTF(describer.getDescription(data.object));
647         }
648       }
649       this.out.flush();
650     } catch (IOException e) {
651       saveException = e;
652       clear();
653       try {
654         out.close();
655       } catch (IOException e2) {
656         // ignore it
657       }
658     }
659   }

661   private synchronized void clear() {
662     profileStartTime = 0L;
663     if (timer != null) {
664       timer.cancel();
665       timer = null;
666     }
667     taskStack = null;
668     taskQueue = null;
669     describer = null;
670 
671     // Note that slowest task aggregator are not cleared here because clearing happens
672     // periodically over the course of a command invocation.
673   }

675   /**
676    * Unless --record_full_profiler_data is given we drop small tasks and add their time to the
677    * parents duration.

679   private boolean wasTaskSlowEnoughToRecord(ProfilerTask type, long duration) {

683   /**
684    * Adds task directly to the main queue bypassing task stack. Used for simple
685    * tasks that are known to not have any subtasks.
686    *
687    * @param startTime task start time (obtained through {@link Profiler#nanoTimeMaybe()})
688    * @param duration task duration
689    * @param type task type
690    * @param object object associated with that task. Can be String object that
691    *               describes it.
692    */
693   private void logTask(long startTime, long duration, ProfilerTask type, Object object) {
694     Preconditions.checkNotNull(object);
695     Preconditions.checkState(startTime > 0, "startTime was %s", startTime);
696     if (duration < 0) {
697       // See note in Clock#nanoTime, which is used by Profiler#nanoTimeMaybe.
698       duration = 0;
699     }
700 
701     TaskData parent = taskStack.peek();
702     if (parent != null) {
703       parent.aggregateChild(type, duration);
704     }
705     if (wasTaskSlowEnoughToRecord(type, duration)) {
706       TaskData data = taskStack.create(startTime, type, object);
707       data.duration = duration;
708       if (out != null) {
709         taskQueue.add(data);
710       }
711 
712       SlowestTaskAggregator aggregator = slowestTasks[type.ordinal()];
713 
714       if (aggregator != null) {

785   /**
786    * Records the beginning of the task specified by the parameters. This method
787    * should always be followed by completeTask() invocation to mark the end of
788    * task execution (usually ensured by try {} finally {} block). Failure to do
789    * so will result in task stack corruption.
790    *
791    * Use of this method allows to support nested task monitoring. For tasks that
792    * are known to not have any subtasks, logSimpleTask() should be used instead.
793    *
794    * @param type predefined task type - see ProfilerTask for available types.
795    * @param object object associated with that task. Can be String object that
796    *               describes it.
797    */
798   public void startTask(ProfilerTask type, Object object) {
799     // ProfilerInfo.allTasksById is supposed to be an id -> Task map, but it is in fact a List,
800     // which means that we cannot drop tasks to which we had already assigned ids. Therefore,
801     // non-leaf tasks must not have a minimum duration. However, we don't quite consistently
802     // enforce this, and Blaze only works because we happen not to add child tasks to those parent
803     // tasks that have a minimum duration.
804     Preconditions.checkNotNull(object);
805     if (isActive() && isProfiling(type)) {
806       taskStack.push(type, object);
807     }
808   }
809 
810   /**
811    * Records the end of the task and moves tasks from the thread-local stack to
812    * the main queue. Will validate that given task type matches task at the top
813    * of the stack.
814    *
815    * @param type task type.
816    */
817   public void completeTask(ProfilerTask type) {
818     if (isActive() && isProfiling(type)) {
819       long endTime = clock.nanoTime();
820       TaskData data = taskStack.pop();
821       // Do not use Preconditions class below due to the very expensive
822       // toString() calls used in the message.
823       if (data.type != type) {
824         throw new IllegalStateException("Inconsistent Profiler.completeTask() call for the "
825             + type + " task.\n " + taskStack);
826       }
827       data.duration = endTime - data.startTime;
828       if (data.parentId > 0) {
829         taskStack.peek().aggregateChild(data.type, data.duration);
830       }
831       boolean shouldRecordTask = wasTaskSlowEnoughToRecord(type, data.duration);
832       if (out != null && (shouldRecordTask || data.counts != null)) {
833         taskQueue.add(data);
834       }
835 
836       if (shouldRecordTask) {
837         SlowestTaskAggregator aggregator = slowestTasks[type.ordinal()];
838 
839         if (aggregator != null) {
840           aggregator.add(data);
841         }
842       }
843     }
844   }
