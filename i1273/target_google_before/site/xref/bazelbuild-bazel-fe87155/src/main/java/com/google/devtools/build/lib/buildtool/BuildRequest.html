52  /**
53   * A BuildRequest represents a single invocation of the build tool by a user.
54   * A request specifies a list of targets to be built for a single
55   * configuration, a pair of output/error streams, and additional options such
56   * as --keep_going, --jobs, etc.
57   */
58  public class BuildRequest implements OptionsClassProvider {
59    /**
60     * A converter for symlink prefixes that defaults to {@code Constants.PRODUCT_NAME} and a
61     * minus sign if the option is not given.
62     *
63     * <p>Required because you cannot specify a non-constant value in annotation attributes.
64     */
65    public static class SymlinkPrefixConverter implements Converter<String> {
66      @Override
67      public String convert(String input) throws OptionsParsingException {
68        return input.isEmpty() ? Constants.PRODUCT_NAME + "-" : input;
69      }
70  
71      @Override
72      public String getTypeDescription() {
73        return "a string";
74      }
75    }

320   private BuildRequest(String commandName,
321                        final OptionsProvider options,
322                        final OptionsProvider startupOptions,
323                        List<String> targets,
324                        OutErr outErr,
325                        UUID id,
326                        long startTimeMillis) {
327     this.commandName = commandName;
328     this.optionsDescription = OptionsUtils.asShellEscapedString(options);
329     this.outErr = outErr;
330     this.targets = targets;
331     this.id = id;
332     this.startTimeMillis = startTimeMillis;
333     this.optionsCache = CacheBuilder.newBuilder()
334         .build(new CacheLoader<Class<? extends OptionsBase>, Optional<OptionsBase>>() {
335           @Override
336           public Optional<OptionsBase> load(Class<? extends OptionsBase> key) throws Exception {
337             OptionsBase result = options.getOptions(key);
338             if (result == null && startupOptions != null) {
339               result = startupOptions.getOptions(key);
340             }
341 
342             return Optional.fromNullable(result);
343           }
344         });
345 
346     for (Class<? extends OptionsBase> optionsClass : MANDATORY_OPTIONS) {
347       Preconditions.checkNotNull(getOptions(optionsClass));
348     }
349   }

469   /**
470    * Validates the options for this BuildRequest.
471    *
472    * <p>Issues warnings or throws {@code InvalidConfigurationException} for option settings that
473    * conflict.
474    *
475    * @return list of warnings
476    */
477   public List<String> validateOptions() throws InvalidConfigurationException {
478     List<String> warnings = new ArrayList<>();
479     // Validate "jobs".
480     int jobs = getBuildOptions().jobs;
481     if (jobs < 0 || jobs > MAX_JOBS) {
482       throw new InvalidConfigurationException(String.format(
483           "Invalid parameter for --jobs: %d. Only values 0 <= jobs <= %d are allowed.", jobs,
484           MAX_JOBS));
485     }
486     if (jobs > JOBS_TOO_HIGH_WARNING) {
487       warnings.add(
488           String.format("High value for --jobs: %d. You may run into memory issues", jobs));
489     }
490 
491     int localTestJobs = getExecutionOptions().localTestJobs;
492     if (localTestJobs < 0) {
493       throw new InvalidConfigurationException(String.format(
494           "Invalid parameter for --local_test_jobs: %d. Only values 0 or greater are "
495               + "allowed.", localTestJobs));
496     }
497     if (localTestJobs > jobs) {
498       warnings.add(
499           String.format("High value for --local_test_jobs: %d. This exceeds the value for --jobs: "
500               + "%d. Only up to %d local tests will run concurrently.", localTestJobs, jobs, jobs));
501     }
502 
503     // Validate other BuildRequest options.
504     if (getBuildOptions().verboseExplanations && getBuildOptions().explanationPath == null) {
505       warnings.add("--verbose_explanations has no effect when --explain=<file> is not enabled");
506     }
507 
508     return warnings;
509   }

511   /** Creates a new TopLevelArtifactContext from this build request. */
512   public TopLevelArtifactContext getTopLevelArtifactContext() {
513     return new TopLevelArtifactContext(
514         getOptions(ExecutionOptions.class).testStrategy.equals("exclusive"),
515         determineOutputGroups());
516   }

518   private ImmutableSortedSet<String> determineOutputGroups() {
519     Set<String> current = new HashSet<>(OutputGroupProvider.DEFAULT_GROUPS);
520 
521     for (String outputGroup : getBuildOptions().outputGroups) {
522       if (outputGroup.startsWith("-")) {
523         current.remove(outputGroup.substring(1));
524       } else {
525         current.add(outputGroup);
526       }
527     }
528 
529     return ImmutableSortedSet.copyOf(current);
530   }

544   public static BuildRequest create(String commandName, OptionsProvider options,
545       OptionsProvider startupOptions,
546       List<String> targets, OutErr outErr, UUID commandId, long commandStartTime) {
547 
548     BuildRequest request = new BuildRequest(commandName, options, startupOptions, targets, outErr,
549         commandId, commandStartTime);
550 
551     // All this, just to pass a global boolean from the client to the server. :(
552     if (options.getOptions(BlazeCommandEventHandler.Options.class).runningInEmacs) {
553       request.setRunningInEmacs();
554     }
555 
556     return request;
557   }
