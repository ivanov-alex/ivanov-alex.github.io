85    public void initCommon(PythonVersion defaultVersion) {
86      this.sourcesVersion = getPythonVersionAttr(
87          ruleContext, "srcs_version", PythonVersion.getAllValues());
88  
89      this.version = ruleContext.getFragment(PythonConfiguration.class)
90          .getPythonVersion(defaultVersion);
91  
92      transitivePythonSources = collectTransitivePythonSources();
93  
94      checkSourceIsCompatible(this.version, this.sourcesVersion, ruleContext.getLabel());
95    }

101   public void initBinary(List<Artifact> srcs) {
102     Preconditions.checkNotNull(version);
103 
104     validatePackageName();
105     executable = ruleContext.createOutputArtifact();
106     if (this.version == PythonVersion.PY2AND3) {
107       // TODO(bazel-team): we need to create two actions
108       ruleContext.ruleError("PY2AND3 is not yet implemented");
109     }
110 
111     filesToBuild = NestedSetBuilder.<Artifact>stableOrder()
112         .addAll(srcs)
113         .add(executable)
114         .build();
115 
116     if (ruleContext.hasErrors()) {
117       return;
118     }
119 
120     addPyExtraActionPseudoAction();
121   }

123   public void addCommonTransitiveInfoProviders(RuleConfiguredTargetBuilder builder,
124       PythonSemantics semantics, NestedSet<Artifact> filesToBuild) {
125     PythonSourcesProvider sourcesProvider =
126         new PythonSourcesProvider(transitivePythonSources, usesSharedLibraries());
127     builder
128         .add(InstrumentedFilesProvider.class, InstrumentedFilesCollector.collect(ruleContext,
129             semantics.getCoverageInstrumentationSpec(), METADATA_COLLECTOR, filesToBuild))
130         .add(PythonSourcesProvider.class, sourcesProvider)
131         .addSkylarkTransitiveInfo(PythonSourcesProvider.SKYLARK_NAME, sourcesProvider)
132         // Python targets are not really compilable. The best we can do is make sure that all
133         // generated source files are ready.
134         .addOutputGroup(OutputGroupProvider.FILES_TO_COMPILE, transitivePythonSources)
135         .addOutputGroup(OutputGroupProvider.COMPILATION_PREREQUISITES, transitivePythonSources);
136   }

146   public static PythonVersion getPythonVersionAttr(RuleContext ruleContext,
147       String attrName, PythonVersion... allowed) {
148     String stringAttr = ruleContext.attributes().get(attrName, Type.STRING);
149     PythonVersion version = PythonVersion.parse(stringAttr, allowed);
150     if (version != null) {
151       return version;
152     }
153     ruleContext.attributeError(attrName,
154         "'" + stringAttr + "' is not a valid value. Expected one of: " + Joiner.on(", ")
155             .join(allowed));
156     return PythonVersion.defaultValue();
157   }

159   /**
160    * Returns a mutable List of the source Artifacts.
161    */
162   public List<Artifact> validateSrcs() {
163     List<Artifact> sourceFiles = new ArrayList<>();
164     // TODO(bazel-team): Need to get the transitive deps closure, not just the
165     //                 sources of the rule.
166     for (FileProvider src : ruleContext
167         .getPrerequisites("srcs", Mode.TARGET, FileProvider.class)) {
168       // Make sure that none of the sources contain hyphens.
169       if (Util.containsHyphen(src.getLabel().getPackageFragment())) {
170         ruleContext.attributeError("srcs", src.getLabel() + ": package name may not contain '-'");
171       }
172       Iterable<Artifact> pySrcs = FileType.filter(src.getFilesToBuild(),
173           PyRuleClasses.PYTHON_SOURCE);
174       Iterables.addAll(sourceFiles, pySrcs);
175       if (Iterables.isEmpty(pySrcs)) {
176         ruleContext.attributeWarning("srcs",
177             "rule '" + src.getLabel() + "' does not produce any Python source files");
178       }
179     }
180 
181     LanguageDependentFragment.Checker.depsSupportsLanguage(ruleContext, PyRuleClasses.LANGUAGE);
182     return convertedFiles != null
183         ? ImmutableList.copyOf(convertedFiles.values())
184         : sourceFiles;
185   }

187   /**
188    * Checks that the package name of this Python rule does not contain a '-'.
189    */
190   void validatePackageName() {
191     if (Util.containsHyphen(ruleContext.getLabel().getPackageFragment())) {
192       ruleContext.ruleError("package name may not contain '-'");
193     }

196   /**
197    * Adds a {@link PseudoAction} to the build graph that is only used
198    * for providing information to the blaze extra_action feature.
199    */
200   void addPyExtraActionPseudoAction() {
201     if (ruleContext.getConfiguration().getActionListeners().isEmpty()) {
202       return;
203     }
204 
205     // Has to be unfiltered sources as filtered will give an error for
206     // unsupported file types where as certain tests only expect a warning.
207     Collection<Artifact> sources = ruleContext.getPrerequisiteArtifacts("srcs", Mode.TARGET).list();
208 
209     // We need to do it in this convoluted way because we must not add the files declared in the
210     // srcs of this rule. Note that it is not enough to remove the direct members from the nested
211     // set of the current rule, because the same files may have been declared in a dependency, too.
212     NestedSetBuilder<Artifact> depBuilder = NestedSetBuilder.compileOrder();
213     collectTransitivePythonSourcesFromDeps(depBuilder);
214     NestedSet<Artifact> dependencies = depBuilder.build();
215 
216     PythonInfo info = PythonInfo.newBuilder()
217         .addAllSourceFile(Artifact.toExecPaths(sources))
218         .addAllDepFile(Artifact.toExecPaths(dependencies))
219         .build();
220 
221     ruleContext.getAnalysisEnvironment()
222         .registerAction(new PyPseudoAction(ruleContext.getActionOwner(),
223             NestedSetBuilder.wrap(Order.STABLE_ORDER, Iterables.concat(sources, dependencies)),
224             ImmutableList.of(PseudoAction.getDummyOutput(ruleContext)), "Python",
225             PythonInfo.pythonInfo, info));
226   }

228   private void addSourceFiles(NestedSetBuilder<Artifact> builder, Iterable<Artifact> artifacts) {
229     Preconditions.checkState(convertedFiles == null);
230     if (sourcesVersion == PythonVersion.PY2 && version == PythonVersion.PY3) {
231       convertedFiles = PythonUtils.generate2to3Actions(ruleContext, artifacts);
232     }
233     builder.addAll(artifacts);
234   }
235 
236   private void collectTransitivePythonSourcesFromDeps(NestedSetBuilder<Artifact> builder) {
237     for (TransitiveInfoCollection dep : ruleContext.getPrerequisites("deps", Mode.TARGET)) {
238       if (dep.getProvider(PythonSourcesProvider.class) != null) {
239         PythonSourcesProvider provider = dep.getProvider(PythonSourcesProvider.class);
240         builder.addTransitive(provider.getTransitivePythonSources());
241       } else {
242         // TODO(bazel-team): We also collect .py source files from deps (e.g. for proto_library
243         // rules). Rules should implement PythonSourcesProvider instead.
244         FileProvider provider = dep.getProvider(FileProvider.class);
245         builder.addAll(FileType.filter(provider.getFilesToBuild(), PyRuleClasses.PYTHON_SOURCE));
246       }
247     }
248   }
249 
250   private NestedSet<Artifact> collectTransitivePythonSources() {
251     NestedSetBuilder<Artifact> builder =
252         NestedSetBuilder.compileOrder();
253     collectTransitivePythonSourcesFromDeps(builder);
254     addSourceFiles(builder, ruleContext
255         .getPrerequisiteArtifacts("srcs", Mode.TARGET).filter(PyRuleClasses.PYTHON_SOURCE).list());
256     return builder.build();
257   }

259   /**
260    * Checks that the source file version is compatible with the Python interpreter.
261    */
262   private void checkSourceIsCompatible(PythonVersion targetVersion, PythonVersion sourceVersion,
263                                           Label source) {
264     if (targetVersion == PythonVersion.PY2 || targetVersion == PythonVersion.PY2AND3) {
265       if (sourceVersion == PythonVersion.PY3ONLY) {
266         ruleContext.ruleError("Rule '" + source
267                   + "' can only be used with Python 3, and cannot be converted to Python 2");
268       } else if (sourceVersion == PythonVersion.PY3) {
269         ruleContext.ruleError("Rule '" + source
270                   + "' need to be converted to Python 2 (not yet implemented)");
271       }
272     }
273     if ((targetVersion == PythonVersion.PY3 || targetVersion == PythonVersion.PY2AND3)
274         && sourceVersion == PythonVersion.PY2ONLY) {
275       ruleContext.ruleError(
276           "Rule '"
277               + source
278               + "' can only be used with Python 2, and cannot be converted to Python 3");
279     }
280   }

282   /**
283    * @return A String that is the full path to the main python entry point.
284    */
285   public String determineMainExecutableSource() {
286     String mainSourceName;
287     Rule target = ruleContext.getRule();
288     boolean explicitMain = target.isAttributeValueExplicitlySpecified("main");
289     if (explicitMain) {
290       mainSourceName = ruleContext.attributes().get("main", Type.LABEL).getName();
291       if (!mainSourceName.endsWith(".py")) {
292         ruleContext.attributeError("main", "main must end in '.py'");
293       }
294     } else {
295       String ruleName = target.getName();
296       if (ruleName.endsWith(".py")) {
297         ruleContext.attributeError("name", "name must not end in '.py'");
298       }
299       mainSourceName = ruleName + ".py";
300     }
301     PathFragment mainSourcePath = new PathFragment(mainSourceName);
302 
303     Artifact mainArtifact = null;
304     for (Artifact outItem : ruleContext.getPrerequisiteArtifacts("srcs", Mode.TARGET).list()) {
305       if (outItem.getRootRelativePath().endsWith(mainSourcePath)) {
306         if (mainArtifact == null) {
307           mainArtifact = outItem;
308         } else {
309           ruleContext.attributeError("srcs",
310               buildMultipleMainMatchesErrorText(explicitMain, mainSourceName,
311                   mainArtifact.getRootRelativePath().toString(),
312                   outItem.getRootRelativePath().toString()));
313         }
314       }
315     }
316 
317     if (mainArtifact == null) {
318       ruleContext.attributeError("srcs", buildNoMainMatchesErrorText(explicitMain, mainSourceName));
319       return null;
320     }
321 
322     PathFragment workspaceName = new PathFragment(ruleContext.getRule().getWorkspaceName());
323     return workspaceName.getRelative(mainArtifact.getRootRelativePath()).getPathString();
324   }

345       ResourceSet.createWithRamCpuIo(10 /* MB */, 1 /* CPU */, 0.0 /* IO */);

347   /**
348    * Utility function to compile multiple .py files to .pyc files.
349    */
350   public Collection<Artifact> createPycFiles(
351       Iterable<Artifact> sources, PathFragment pythonBinary) {
352     List<Artifact> pycFiles = new ArrayList<>();
353     for (Artifact source : sources) {
354       Artifact pycFile = createPycFile(source, pythonBinary);
355       pycFiles.add(pycFile);
356     }
357     return ImmutableList.copyOf(pycFiles);
358   }

360   /**
361    * Given a single .py source artifact generate a .pyc file.
362    */
363   public Artifact createPycFile(
364       Artifact source, PathFragment pythonBinary) {
365     Artifact output =
366         ruleContext.getRelatedArtifact(source.getRootRelativePath(), ".pyc");
367 
368     // TODO(nnorwitz): Consider adding PYTHONHASHSEED=0 to the environment.
369     // This will make the .pyc more stable, though it will still be non-deterministic.
370     // The timestamp is zeroed out above.
371     SpawnAction.Builder builder = new SpawnAction.Builder()
372         .setResources(PY_COMPILE_RESOURCE_SET)
373         .setExecutable(pythonBinary)
374         .setProgressMessage("Compiling Python")
375         .addInputArgument(
376             ruleContext.getPrerequisiteArtifact("$python_precompile", Mode.HOST))
377         .setMnemonic("PyCompile");
378 
379     TransitiveInfoCollection pythonTarget =
380         ruleContext.getPrerequisite(":host_python2_runtime", Mode.HOST);
381     if (pythonTarget != null) {
382       builder.addInputs(pythonTarget
383           .getProvider(FileProvider.class)
384           .getFilesToBuild());
385     }
386 
387     builder.addInputArgument(source);
388     builder.addOutputArgument(output);
389     ruleContext.registerAction(builder.build(ruleContext));
390     return output;
391   }

394   /**
395    * Returns true if this target has an .so file in its transitive dependency closure.
396    */
397   public static boolean checkForSharedLibraries(Iterable<TransitiveInfoCollection> deps) {
398     for (TransitiveInfoCollection dep : deps) {
399       PythonSourcesProvider provider = dep.getProvider(PythonSourcesProvider.class);
400       if (provider != null) {
401         if (provider.usesSharedLibraries()) {
402           return true;
403         }
404       } else if (FileType.contains(
405           dep.getProvider(FileProvider.class).getFilesToBuild(), CppFileTypes.SHARED_LIBRARY)) {
406         return true;
407       }
408     }
409 
410     return false;
411   }

413   private static String buildMultipleMainMatchesErrorText(boolean explicit, String proposedMainName,
414       String match1, String match2) {
415     String errorText;
416     if (explicit) {
417       errorText = "file name '" + proposedMainName
418           + "' specified by 'main' attribute matches multiple files: e.g., '" + match1
419           + "' and '" + match2 + "'";
420     } else {
421       errorText = "default main file name '" + proposedMainName
422           + "' matches multiple files.  Perhaps specify an explicit file with 'main' attribute?  "
423           + "Matches were: '" + match1 + "' and '" + match2 + "'";
424     }
425     return errorText;
426   }
427 
428   private static String buildNoMainMatchesErrorText(boolean explicit, String proposedMainName) {
429     String errorText;
430     if (explicit) {
431       errorText = "could not find '" + proposedMainName
432           + "' as specified by 'main' attribute";
433     } else {
434       errorText = "corresponding default '" + proposedMainName + "' does not appear in srcs. Add it"
435           + " or override default file name with a 'main' attribute";
436     }
437     return errorText;
438   }
