74  public abstract class AndroidBinary implements RuleConfiguredTargetFactory {
75    static final String PROGUARD_SPECS = "proguard_specs";

77    protected abstract JavaSemantics createJavaSemantics();
78    protected abstract AndroidSemantics createAndroidSemantics();

80    @Override
81    public ConfiguredTarget create(RuleContext ruleContext) throws InterruptedException {
82      JavaSemantics javaSemantics = createJavaSemantics();
83      AndroidSemantics androidSemantics = createAndroidSemantics();
84      if (!AndroidSdkProvider.verifyPresence(ruleContext)) {
85        return null;
86      }
87  
88      NestedSetBuilder<Artifact> filesBuilder = NestedSetBuilder.stableOrder();
89      ImmutableList<TransitiveInfoCollection> deps = ImmutableList.<TransitiveInfoCollection>copyOf(
90          ruleContext.getPrerequisites("deps", Mode.TARGET));
91      JavaCommon javaCommon = new JavaCommon(
92          ruleContext, javaSemantics, deps, deps, deps);
93  
94      AndroidCommon androidCommon = new AndroidCommon(
95          ruleContext, javaCommon, true /* asNeverLink */, true /* exportDeps */);
96      try {
97        RuleConfiguredTargetBuilder builder =
98            init(ruleContext, filesBuilder,
99                AndroidCommon.getTransitiveResourceContainers(ruleContext, true),
100               javaCommon, androidCommon, javaSemantics, androidSemantics,
101               ImmutableList.<String>of("deps"));
102       if (builder == null) {
103         return null;
104       }
105       return builder.build();
106     } catch (RuleConfigurationException e) {
107       Preconditions.checkArgument(ruleContext.hasErrors(),
108           "Exception caught but no errors reported:\n %s", Joiner.on("\n").join(e.getStackTrace()));
109       return null;
110     }
111   }

113   private static RuleConfiguredTargetBuilder init(
114       RuleContext ruleContext,
115       NestedSetBuilder<Artifact> filesBuilder,
116       NestedSet<ResourceContainer> resourceContainers,
117       JavaCommon javaCommon,
118       AndroidCommon androidCommon,
119       JavaSemantics javaSemantics,
120       AndroidSemantics androidSemantics,
121       List<String> depsAttributes) throws InterruptedException {
122     // TODO(bazel-team): Find a way to simplify this code.
123     // treeKeys() means that the resulting map sorts the entries by key, which is necessary to
124     // ensure determinism.
125     Multimap<String, TransitiveInfoCollection> depsByArchitecture =
126         MultimapBuilder.treeKeys().arrayListValues().build();
127     AndroidConfiguration config = ruleContext.getFragment(AndroidConfiguration.class);
128     if (config.isFatApk()) {
129       for (String depsAttribute : depsAttributes) {
130         for (Map.Entry<String, ? extends List<? extends TransitiveInfoCollection>> entry :
131             ruleContext.getSplitPrerequisites(depsAttribute).entrySet()) {
132           depsByArchitecture.putAll(entry.getKey(), entry.getValue());
133         }
134       }
135     } else {
136       for (String depsAttribute : depsAttributes) {
137         depsByArchitecture.putAll(
138             config.getCpu(), ruleContext.getPrerequisites(depsAttribute, Mode.TARGET));
139       }
140     }
141     Map<String, BuildConfiguration> configurationMap = new LinkedHashMap<>();
142     Map<String, CcToolchainProvider> toolchainMap = new LinkedHashMap<>();
143     if (config.isFatApk()) {
144       for (Map.Entry<String, ? extends List<? extends TransitiveInfoCollection>> entry :
145           ruleContext.getSplitPrerequisites(":cc_toolchain_split").entrySet()) {
146         TransitiveInfoCollection dep = Iterables.getOnlyElement(entry.getValue());
147         CcToolchainProvider toolchain = CppHelper.getToolchain(ruleContext, dep);
148         configurationMap.put(entry.getKey(), dep.getConfiguration());
149         toolchainMap.put(entry.getKey(), toolchain);
150       }
151     } else {
152       configurationMap.put(config.getCpu(), ruleContext.getConfiguration());
153       toolchainMap.put(config.getCpu(), CppHelper.getToolchain(ruleContext));
154     }
155 
156     NativeLibs nativeLibs = shouldLinkNativeDeps(ruleContext)
157         ? NativeLibs.fromLinkedNativeDeps(ruleContext, androidSemantics.getNativeDepsFileName(),
158             depsByArchitecture, toolchainMap, configurationMap)
159         : NativeLibs.fromPrecompiledObjects(ruleContext, depsByArchitecture);
160 
161     // TODO(bazel-team): Resolve all the different cases of resource handling so this conditional
162     // can go away: recompile from android_resources, and recompile from
163     // android_binary attributes.
164     ApplicationManifest applicationManifest;
165     ResourceApk splitResourceApk;
166     ResourceApk incrementalResourceApk;
167     ResourceApk resourceApk;
168     if (LocalResourceContainer.definesAndroidResources(ruleContext.attributes())) {
169       // Retrieve and compile the resources defined on the android_binary rule.
170       if (!LocalResourceContainer.validateRuleContext(ruleContext)) {
171         throw new RuleConfigurationException();
172       }
173       applicationManifest = androidSemantics.getManifestForRule(ruleContext)
174           .mergeWith(ruleContext, resourceContainers);
175       resourceApk = applicationManifest.packWithDataAndResources(
176           ruleContext.getImplicitOutputArtifact(AndroidRuleClasses.ANDROID_RESOURCES_APK),
177           ruleContext,
178           resourceContainers,
179           null, /* Artifact rTxt */
180           null, /* Artifact symbolsTxt */
181           ruleContext.getTokenizedStringListAttr("resource_configuration_filters"),
182           ruleContext.getTokenizedStringListAttr("nocompress_extensions"),
183           ruleContext.getTokenizedStringListAttr("densities"),
184           ruleContext.attributes().get("application_id", Type.STRING),
185           getExpandedMakeVarsForAttr(ruleContext, "version_code"),
186           getExpandedMakeVarsForAttr(ruleContext, "version_name"),
187           false, getProguardConfigArtifact(ruleContext, ""));
188       incrementalResourceApk = applicationManifest.addStubApplication(ruleContext)
189           .packWithDataAndResources(ruleContext
190                   .getImplicitOutputArtifact(AndroidRuleClasses.ANDROID_INCREMENTAL_RESOURCES_APK),
191               ruleContext,
192               resourceContainers,
193               null, /* Artifact rTxt */
194               null, /* Artifact symbolsTxt */
195               ruleContext.getTokenizedStringListAttr("resource_configuration_filters"),
196               ruleContext.getTokenizedStringListAttr("nocompress_extensions"),
197               ruleContext.getTokenizedStringListAttr("densities"),
198               ruleContext.attributes().get("application_id", Type.STRING),
199               getExpandedMakeVarsForAttr(ruleContext, "version_code"),
200               getExpandedMakeVarsForAttr(ruleContext, "version_name"),
201               true, getProguardConfigArtifact(ruleContext, "incremental"));
202       splitResourceApk = applicationManifest
203           .createSplitManifest(ruleContext, "android_resources", false)
204           .packWithDataAndResources(getDxArtifact(ruleContext, "android_resources.ap_"),
205               ruleContext,
206               resourceContainers,
207               null, /* Artifact rTxt */
208               null, /* Artifact symbolsTxt */
209               ruleContext.getTokenizedStringListAttr("resource_configuration_filters"),
210               ruleContext.getTokenizedStringListAttr("nocompress_extensions"),
211               ruleContext.getTokenizedStringListAttr("densities"),
212               ruleContext.attributes().get("application_id", Type.STRING),
213               getExpandedMakeVarsForAttr(ruleContext, "version_code"),
214               getExpandedMakeVarsForAttr(ruleContext, "version_name"),
215               true, getProguardConfigArtifact(ruleContext, "incremental_split"));
216     } else {
217       // Retrieve the resources from the resources attribute on the android_binary rule
218       // and recompile them if necessary.
219       applicationManifest = ApplicationManifest.fromResourcesRule(ruleContext).mergeWith(
220           ruleContext, resourceContainers);
221       // Always recompiling resources causes AndroidTest to fail in certain circumstances.
222       if (shouldRegenerate(ruleContext, resourceContainers)) {
223         resourceApk = applicationManifest.packWithResources(
224             ruleContext.getImplicitOutputArtifact(AndroidRuleClasses.ANDROID_RESOURCES_APK),
225             ruleContext,
226             resourceContainers,
227             true,
228             getProguardConfigArtifact(ruleContext, ""));
229       } else {
230         resourceApk = applicationManifest.useCurrentResources(ruleContext,
231             getProguardConfigArtifact(ruleContext, ""));
232       }
233       incrementalResourceApk = applicationManifest
234           .addStubApplication(ruleContext)
235           .packWithResources(
236               ruleContext.getImplicitOutputArtifact(
237                   AndroidRuleClasses.ANDROID_INCREMENTAL_RESOURCES_APK),
238               ruleContext,
239               resourceContainers,
240               false,
241               getProguardConfigArtifact(ruleContext, "incremental"));
242 
243       splitResourceApk = applicationManifest
244           .createSplitManifest(ruleContext, "android_resources", false)
245           .packWithResources(getDxArtifact(ruleContext, "android_resources.ap_"),
246             ruleContext,
247             resourceContainers,
248             false,
249             getProguardConfigArtifact(ruleContext, "incremental_split"));
250     }
251 
252     JavaTargetAttributes resourceClasses = androidCommon.init(
253         javaSemantics,
254         androidSemantics,
255         resourceApk,
256         AndroidIdlProvider.EMPTY,
257         ruleContext.getConfiguration().isCodeCoverageEnabled(),
258         true /* collectJavaCompilationArgs */,
259         AndroidRuleClasses.ANDROID_BINARY_GEN_JAR);
260     if (resourceClasses == null) {
261       return null;
262     }
263 
264     Artifact deployJar = createDeployJar(ruleContext, javaSemantics, androidCommon, resourceClasses,
265         ruleContext.getImplicitOutputArtifact(AndroidRuleClasses.ANDROID_BINARY_DEPLOY_JAR));
266 
267     return createAndroidBinary(ruleContext,
268         filesBuilder,
269         deployJar,
270         javaCommon,
271         androidCommon,
272         javaSemantics,
273         androidSemantics,
274         nativeLibs,
275         applicationManifest,
276         resourceApk,
277         incrementalResourceApk,
278         splitResourceApk,
279         resourceClasses,
280         ImmutableList.<Artifact>of(),
281         null);
282   }

285   public static RuleConfiguredTargetBuilder createAndroidBinary(
286       RuleContext ruleContext,
287       NestedSetBuilder<Artifact> filesBuilder,
288       Artifact deployJar,
289       JavaCommon javaCommon,
290       AndroidCommon androidCommon,
291       JavaSemantics javaSemantics,
292       AndroidSemantics androidSemantics,
293       NativeLibs nativeLibs,
294       ApplicationManifest applicationManifest,
295       ResourceApk resourceApk,
296       ResourceApk incrementalResourceApk,
297       ResourceApk splitResourceApk,
298       JavaTargetAttributes resourceClasses,
299       ImmutableList<Artifact> apksUnderTest,
300       Artifact proguardMapping) throws InterruptedException {
301 
302     ImmutableList<Artifact> proguardSpecs =
303         getTransitiveProguardSpecs(
304             ruleContext,
305             resourceApk,
306             ruleContext.getPrerequisiteArtifacts(PROGUARD_SPECS, Mode.TARGET).list());
307 
308     ProguardOutput proguardOutput =
309         applyProguard(
310             ruleContext,
311             androidCommon,
312             deployJar,
313             filesBuilder,
314             proguardSpecs,
315             proguardMapping);
316     Artifact jarToDex = proguardOutput.outputJar;
317     DexingOutput dexingOutput =
318         shouldDexWithJack(ruleContext)
319             ? dexWithJack(ruleContext, androidCommon, proguardSpecs)
320             : dex(
321                 ruleContext,
322                 getMultidexMode(ruleContext),
323                 ruleContext.getTokenizedStringListAttr("dexopts"),
324                 deployJar,
325                 jarToDex,
326                 androidCommon,
327                 resourceClasses);
328 
329     Artifact unsignedApk =
330         ruleContext.getImplicitOutputArtifact(AndroidRuleClasses.ANDROID_BINARY_UNSIGNED_APK);
331     Artifact signedApk =
332         ruleContext.getImplicitOutputArtifact(AndroidRuleClasses.ANDROID_BINARY_SIGNED_APK);
333 
334     ApkActionBuilder apkBuilder = new ApkActionBuilder(ruleContext, androidSemantics)
335         .classesDex(dexingOutput.classesDexZip)
336         .resourceApk(resourceApk.getArtifact())
337         .javaResourceZip(dexingOutput.javaResourceJar)
338         .nativeLibs(nativeLibs);
339 
340     ruleContext.registerAction(apkBuilder
341         .message("Generating unsigned apk")
342         .build(unsignedApk));
343 
344     ruleContext.registerAction(apkBuilder
345         .message("Generating signed apk")
346         .sign(true)
347         .build(signedApk));
348 
349     Artifact zipAlignedApk = zipalignApk(
350         ruleContext,
351         signedApk,
352         ruleContext.getImplicitOutputArtifact(AndroidRuleClasses.ANDROID_BINARY_APK));
353 
354     // Don't add blacklistedApk, so it's only built if explicitly requested.
355     filesBuilder.add(deployJar);
356     filesBuilder.add(unsignedApk);
357     filesBuilder.add(zipAlignedApk);
358 
359     NestedSet<Artifact> filesToBuild = filesBuilder.build();
360 
361     NestedSet<Artifact> coverageMetadata = (androidCommon.getInstrumentedJar() != null)
362         ? NestedSetBuilder.create(Order.STABLE_ORDER, androidCommon.getInstrumentedJar())
363         : NestedSetBuilder.<Artifact>emptySet(Order.STABLE_ORDER);
364 
365     RuleConfiguredTargetBuilder builder =
366         new RuleConfiguredTargetBuilder(ruleContext);
367 
368     Artifact incrementalApk =
369         ruleContext.getImplicitOutputArtifact(AndroidRuleClasses.ANDROID_BINARY_INCREMENTAL_APK);
370 
371     Artifact fullDeployMarker =
372         ruleContext.getImplicitOutputArtifact(AndroidRuleClasses.FULL_DEPLOY_MARKER);
373     Artifact incrementalDeployMarker =
374         ruleContext.getImplicitOutputArtifact(AndroidRuleClasses.INCREMENTAL_DEPLOY_MARKER);
375     Artifact splitDeployMarker =
376         ruleContext.getImplicitOutputArtifact(AndroidRuleClasses.SPLIT_DEPLOY_MARKER);
377 
378     Artifact incrementalDexManifest =
379         ruleContext.getImplicitOutputArtifact(AndroidRuleClasses.DEX_MANIFEST);
380     ruleContext.registerAction(new SpawnAction.Builder()
381         .setMnemonic("AndroidDexManifest")
382         .setProgressMessage("Generating incremental installation manifest for "
383             + ruleContext.getLabel())
384         .setExecutable(
385             ruleContext.getExecutablePrerequisite("$build_incremental_dexmanifest", Mode.HOST))
386         .addOutputArgument(incrementalDexManifest)
387         .addInputArguments(dexingOutput.shardDexZips)
388         .useParameterFile(ParameterFileType.UNQUOTED).build(ruleContext));
389 
390     Artifact stubData = ruleContext.getImplicitOutputArtifact(
391         AndroidRuleClasses.STUB_APPLICATION_DATA);
392     Artifact stubDex = getStubDex(ruleContext, javaSemantics, false);
393     if (ruleContext.hasErrors()) {
394       return null;
395     }
396 
397     ApkActionBuilder incrementalActionBuilder = new ApkActionBuilder(ruleContext, androidSemantics)
398         .classesDex(stubDex)
399         .resourceApk(incrementalResourceApk.getArtifact())
400         .javaResourceZip(dexingOutput.javaResourceJar)
401         .sign(true)
402         .javaResourceFile(stubData)
403         .message("Generating incremental apk");
404 
405     if (!ruleContext.getFragment(AndroidConfiguration.class).useIncrementalNativeLibs()) {
406       incrementalActionBuilder.nativeLibs(nativeLibs);
407     }
408 
409     ruleContext.registerAction(incrementalActionBuilder.build(incrementalApk));
410 
411     Artifact argsArtifact =
412         ruleContext.getImplicitOutputArtifact(AndroidRuleClasses.MOBILE_INSTALL_ARGS);
413     ruleContext.registerAction(
414         new WriteAdbArgsAction(ruleContext.getActionOwner(), argsArtifact));
415 
416     createInstallAction(ruleContext, false, fullDeployMarker, argsArtifact,
417         incrementalDexManifest, incrementalResourceApk.getArtifact(), incrementalApk, nativeLibs,
418         stubData);
419 
420     createInstallAction(ruleContext, true, incrementalDeployMarker,
421         argsArtifact,
422         incrementalDexManifest,
423         incrementalResourceApk.getArtifact(),
424         incrementalApk,
425         nativeLibs,
426         stubData);
427 
428     NestedSetBuilder<Artifact> splitApkSetBuilder = NestedSetBuilder.stableOrder();
429 
430     for (int i = 0; i < dexingOutput.shardDexZips.size(); i++) {
431       String splitName = "dex" + (i + 1);
432       Artifact splitApkResources = createSplitApkResources(
433           ruleContext, applicationManifest, splitName, true);
434       Artifact splitApk = getDxArtifact(ruleContext, splitName + ".apk");
435       ruleContext.registerAction(new ApkActionBuilder(ruleContext, androidSemantics)
436           .classesDex(dexingOutput.shardDexZips.get(i))
437           .resourceApk(splitApkResources)
438           .sign(true)
439           .message("Generating split dex apk " + (i + 1))
440           .build(splitApk));
441       splitApkSetBuilder.add(splitApk);
442     }
443 
444     Artifact nativeSplitApkResources = createSplitApkResources(
445         ruleContext, applicationManifest, "native", false);
446     Artifact nativeSplitApk = getDxArtifact(ruleContext, "native.apk");
447     ruleContext.registerAction(new ApkActionBuilder(ruleContext, androidSemantics)
448         .resourceApk(nativeSplitApkResources)
449         .sign(true)
450         .message("Generating split native apk")
451         .nativeLibs(nativeLibs)
452         .build(nativeSplitApk));
453     splitApkSetBuilder.add(nativeSplitApk);
454 
455     Artifact javaSplitApkResources = createSplitApkResources(
456         ruleContext, applicationManifest, "java_resources", false);
457     Artifact javaSplitApk = getDxArtifact(ruleContext, "java_resources.apk");
458     ruleContext.registerAction(new ApkActionBuilder(ruleContext, androidSemantics)
459         .resourceApk(javaSplitApkResources)
460         .javaResourceZip(dexingOutput.javaResourceJar)
461         .sign(true)
462         .message("Generating split Java resource apk")
463         .build(javaSplitApk));
464     splitApkSetBuilder.add(javaSplitApk);
465 
466     Artifact resourceSplitApk = getDxArtifact(ruleContext, "android_resources.apk");
467     ruleContext.registerAction(new ApkActionBuilder(ruleContext, androidSemantics)
468         .resourceApk(splitResourceApk.getArtifact())
469         .sign(true)
470         .message("Generating split Android resource apk")
471         .build(resourceSplitApk));
472     splitApkSetBuilder.add(resourceSplitApk);
473 
474     Artifact splitMainApkResources = getDxArtifact(ruleContext, "split_main.ap_");
475     ruleContext.registerAction(new SpawnAction.Builder()
476         .setMnemonic("AndroidStripResources")
477         .setProgressMessage("Stripping resources from split main apk")
478         .setExecutable(ruleContext.getExecutablePrerequisite("$strip_resources", Mode.HOST))
479         .addArgument("--input_resource_apk")
480         .addInputArgument(resourceApk.getArtifact())
481         .addArgument("--output_resource_apk")
482         .addOutputArgument(splitMainApkResources)
483         .build(ruleContext));
484 
485     NestedSet<Artifact> splitApks = splitApkSetBuilder.build();
486     Artifact splitMainApk = getDxArtifact(ruleContext, "split_main.apk");
487     Artifact splitStubDex = getStubDex(ruleContext, javaSemantics, true);
488     if (ruleContext.hasErrors()) {
489       return null;
490     }
491     ruleContext.registerAction(new ApkActionBuilder(ruleContext, androidSemantics)
492         .resourceApk(splitMainApkResources)
493         .classesDex(splitStubDex)
494         .sign(true)
495         .message("Generating split main apk")
496         .build(splitMainApk));
497     splitApkSetBuilder.add(splitMainApk);
498     NestedSet<Artifact> allSplitApks = splitApkSetBuilder.build();
499 
500     createSplitInstallAction(ruleContext, splitDeployMarker, argsArtifact, splitMainApk,
501         splitApks, stubData);
502 
503     NestedSet<Artifact> splitOutputGroup = NestedSetBuilder.<Artifact>stableOrder()
504         .addTransitive(allSplitApks)
505         .add(splitDeployMarker)
506         .build();
507 
508     androidCommon.addTransitiveInfoProviders(builder);
509     androidSemantics.addTransitiveInfoProviders(builder, ruleContext, javaCommon, androidCommon,
510         jarToDex, resourceApk, zipAlignedApk, apksUnderTest);
511 
512     if (proguardOutput.mapping != null) {
513       builder.add(ProguardMappingProvider.class,
514           new ProguardMappingProvider(proguardOutput.mapping));
515     }
516 
517     return builder
518         .setFilesToBuild(filesToBuild)
519         .add(RunfilesProvider.class, RunfilesProvider.simple(
520             new Runfiles.Builder(ruleContext.getWorkspaceName())
521             .addRunfiles(ruleContext, RunfilesProvider.DEFAULT_RUNFILES)
522             .addTransitiveArtifacts(filesToBuild)
523             .build()))
524         .add(ApkProvider.class,
525             new ApkProvider(NestedSetBuilder.create(Order.STABLE_ORDER, zipAlignedApk),
526                 coverageMetadata))
527         .add(AndroidPreDexJarProvider.class, new AndroidPreDexJarProvider(jarToDex))
528         .addOutputGroup("mobile_install_full", fullDeployMarker)
529         .addOutputGroup("mobile_install_incremental", incrementalDeployMarker)
530         .addOutputGroup("mobile_install_split", splitOutputGroup);
531   }

533   private static void createSplitInstallAction(RuleContext ruleContext,
534       Artifact marker, Artifact argsArtifact, Artifact splitMainApk, NestedSet<Artifact> splitApks,
535       Artifact stubDataFile) {
536     FilesToRunProvider adb = AndroidSdkProvider.fromRuleContext(ruleContext).getAdb();
537     SpawnAction.Builder builder = new SpawnAction.Builder()
538         .setExecutable(ruleContext.getExecutablePrerequisite("$incremental_install", Mode.HOST))
539         .addTool(adb)
540         .executeUnconditionally()
541         .setMnemonic("AndroidInstall")
542         .setProgressMessage("Installing " + ruleContext.getLabel() + " using split apks")
543         .setExecutionInfo(ImmutableMap.of("local", ""))
544         .addArgument("--output_marker")
545         .addOutputArgument(marker)
546         .addArgument("--stub_datafile")
547         .addInputArgument(stubDataFile)
548         .addArgument("--adb")
549         .addArgument(adb.getExecutable().getExecPathString())
550         .addTool(adb)
551         .addArgument("--flagfile")
552         .addInputArgument(argsArtifact)
553         .addArgument("--split_main_apk")
554         .addInputArgument(splitMainApk);
555 
556     for (Artifact splitApk : splitApks) {
557       builder
558           .addArgument("--split_apk")
559           .addInputArgument(splitApk);
560     }
561 
562     ruleContext.registerAction(builder.build(ruleContext));
563   }

565   private static void createInstallAction(RuleContext ruleContext,
566       boolean incremental, Artifact marker, Artifact argsArtifact,
567       Artifact dexmanifest, Artifact resourceApk, Artifact apk, NativeLibs nativeLibs,
568       Artifact stubDataFile) {
569     FilesToRunProvider adb = AndroidSdkProvider.fromRuleContext(ruleContext).getAdb();
570     SpawnAction.Builder builder = new SpawnAction.Builder()
571         .setExecutable(ruleContext.getExecutablePrerequisite("$incremental_install", Mode.HOST))
572         // We cannot know if the user connected a new device, uninstalled the app from the device
573         // or did anything strange to it, so we always run this action.
574         .executeUnconditionally()
575         .setMnemonic("AndroidInstall")
576         .setProgressMessage(
577             "Installing " + ruleContext.getLabel() + (incremental ? " incrementally" : ""))
578         .setExecutionInfo(ImmutableMap.of("local", ""))
579         .addArgument("--output_marker")
580         .addOutputArgument(marker)
581         .addArgument("--dexmanifest")
582         .addInputArgument(dexmanifest)
583         .addArgument("--resource_apk")
584         .addInputArgument(resourceApk)
585         .addArgument("--stub_datafile")
586         .addInputArgument(stubDataFile)
587         .addArgument("--adb")
588         .addArgument(adb.getExecutable().getExecPathString())
589         .addTool(adb)
590         .addArgument("--flagfile")
591         .addInputArgument(argsArtifact);
592 
593     if (!incremental) {
594       builder
595           .addArgument("--apk")
596           .addInputArgument(apk);
597     }
598 
599     if (ruleContext.getFragment(AndroidConfiguration.class).useIncrementalNativeLibs()) {
600       for (Map.Entry<String, Iterable<Artifact>> arch : nativeLibs.getMap().entrySet()) {
601         for (Artifact lib : arch.getValue()) {
602           builder
603               .addArgument("--native_lib")
604               .addArgument(arch.getKey() + ":" + lib.getExecPathString())
605               .addInput(lib);
606         }
607       }
608     }
609 
610     ruleContext.registerAction(builder.build(ruleContext));
611   }

685   /**
686    * Retrieves the full set of proguard specs that should be applied to this binary.
687    *
688    * <p>If an empty list is passed (i.e., there are no proguardSpecs on this rule), an empty list
689    * will be returned, regardless of any specs from dependencies or the resourceApk.
690    */
691   private static ImmutableList<Artifact> getTransitiveProguardSpecs(
692       RuleContext ruleContext, ResourceApk resourceApk, ImmutableList<Artifact> proguardSpecs) {
693     if (proguardSpecs.isEmpty()) {
694       return proguardSpecs;
695     }
696 
697     ImmutableSortedSet.Builder<Artifact> builder =
698         ImmutableSortedSet.<Artifact>orderedBy(Artifact.EXEC_PATH_COMPARATOR).addAll(proguardSpecs);
699     for (ProguardSpecProvider dep :
700         ruleContext.getPrerequisites("deps", Mode.TARGET, ProguardSpecProvider.class)) {
701       builder.addAll(dep.getTransitiveProguardSpecs());
702     }
703     Artifact output = resourceApk.getResourceProguardConfig();
704     builder.add(output);
705     return builder.build().asList();
706   }

708   /** Applies the proguard specifications, and creates a ProguardedJar. */
709   private static ProguardOutput applyProguard(
710       RuleContext ruleContext,
711       AndroidCommon common,
712       Artifact deployJarArtifact,
713       NestedSetBuilder<Artifact> filesBuilder,
714       ImmutableList<Artifact> proguardSpecs,
715       Artifact proguardMapping) throws InterruptedException {
716     Artifact proguardOutputJar =
717         ruleContext.getImplicitOutputArtifact(AndroidRuleClasses.ANDROID_BINARY_PROGUARD_JAR);
718 
719     // Proguard will be only used for binaries which specify a proguard_spec
720     if (proguardSpecs.isEmpty()) {
721       // Although normally the Proguard jar artifact is not needed for binaries which do not specify
722       // proguard_specs, targets which use a select to provide an empty list to proguard_specs will
723       // still have a Proguard jar implicit output, as it is impossible to tell what a select will
724       // produce at the time of implicit output determination. As a result, this artifact must
725       // always be created.
726       return createEmptyProguardAction(ruleContext, proguardOutputJar, deployJarArtifact);
727     }
728 
729     AndroidSdkProvider sdk = AndroidSdkProvider.fromRuleContext(ruleContext);
730     return createProguardAction(ruleContext, common, sdk.getProguard(), deployJarArtifact,
731         proguardSpecs, proguardMapping, sdk.getAndroidJar(), proguardOutputJar, filesBuilder);
732   }

734   private static ProguardOutput createEmptyProguardAction(RuleContext ruleContext,
735       Artifact proguardOutputJar, Artifact deployJarArtifact) throws InterruptedException {
736     ImmutableList.Builder<Artifact> failures =
737         ImmutableList.<Artifact>builder().add(proguardOutputJar);
738     if (ruleContext.attributes().get("proguard_generate_mapping", Type.BOOLEAN)) {
739       failures.add(
740           ruleContext.getImplicitOutputArtifact(AndroidRuleClasses.ANDROID_BINARY_PROGUARD_MAP));
741     }
742     ruleContext.registerAction(
743         new FailAction(
744             ruleContext.getActionOwner(),
745             failures.build(),
746             "Can't generate Proguard jar or mapping without proguard_specs."));
747     return new ProguardOutput(deployJarArtifact, null);
748   }

750   private static ProguardOutput createProguardAction(RuleContext ruleContext, AndroidCommon common,
751       FilesToRunProvider proguard,
752       Artifact jar, ImmutableList<Artifact> proguardSpecs,
753       Artifact proguardMapping, Artifact androidJar, Artifact proguardOutputJar,
754       NestedSetBuilder<Artifact> filesBuilder) throws InterruptedException {
755     Iterable<Artifact> libraryJars = NestedSetBuilder.<Artifact>naiveLinkOrder()
756         .add(androidJar)
757         .addTransitive(common.getTransitiveNeverLinkLibraries())
758         .build();
759 
760     Builder builder = new SpawnAction.Builder()
761         .addInput(jar)
762         .addInputs(libraryJars)
763         .addInputs(proguardSpecs)
764         .addOutput(proguardOutputJar)
765         .setExecutable(proguard)
766         .setProgressMessage("Trimming binary with proguard")
767         .setMnemonic("Proguard")
768         .addArgument("-injars")
769         .addArgument(jar.getExecPathString());
770 
771     for (Artifact libraryJar : libraryJars) {
772       builder.addArgument("-libraryjars")
773           .addArgument(libraryJar.getExecPathString());
774     }
775 
776     filesBuilder.add(proguardOutputJar);
777 
778     if (proguardMapping != null) {
779       builder.addInput(proguardMapping)
780           .addArgument("-applymapping")
781           .addArgument(proguardMapping.getExecPathString());
782     }
783 
784     Artifact proguardOutputMap = null;
785     if (ruleContext.attributes().get("proguard_generate_mapping", Type.BOOLEAN)) {
786        proguardOutputMap = ruleContext.getImplicitOutputArtifact(
787           AndroidRuleClasses.ANDROID_BINARY_PROGUARD_MAP);
788 
789       builder.addOutput(proguardOutputMap)
790           .addArgument("-printmapping")
791           .addArgument(proguardOutputMap.getExecPathString());
792       filesBuilder.add(proguardOutputMap);
793     }
794 
795     builder.addArgument("-outjars")
796         .addArgument(proguardOutputJar.getExecPathString());
797 
798     for (Artifact proguardSpec : proguardSpecs) {
799       builder.addArgument("@" + proguardSpec.getExecPathString());
800     }
801 
802     ruleContext.registerAction(builder.build(ruleContext));
803     return new ProguardOutput(proguardOutputJar, proguardOutputMap);
804   }

837   /** Dexes the ProguardedJar to generate ClassesDex that has a reference classes.dex. */
838   static DexingOutput dex(RuleContext ruleContext, MultidexMode multidexMode, List<String> dexopts,
839       Artifact deployJar,  Artifact proguardedJar, AndroidCommon common,
840       JavaTargetAttributes attributes) throws InterruptedException {
841     String classesDexFileName = getMultidexMode(ruleContext).getOutputDexFilename();
842     Artifact classesDex = AndroidBinary.getDxArtifact(ruleContext, classesDexFileName);
843     if (!AndroidBinary.supportsMultidexMode(ruleContext, multidexMode)) {
844       ruleContext.ruleError("Multidex mode \"" + multidexMode.getAttributeValue()
845           + "\" not supported by this version of the Android SDK");
846       throw new RuleConfigurationException();
847     }
848 
849     int dexShards = ruleContext.attributes().get("dex_shards", Type.INTEGER);
850     if (dexShards > 1 && multidexMode == MultidexMode.OFF) {
851       ruleContext.ruleError(".dex sharding is only available in multidex mode");
852       throw new RuleConfigurationException();
853     }
854 
855     Artifact mainDexList = ruleContext.getPrerequisiteArtifact("main_dex_list", Mode.TARGET);
856     if ((mainDexList != null && multidexMode != MultidexMode.MANUAL_MAIN_DEX)
857         || (mainDexList == null && multidexMode == MultidexMode.MANUAL_MAIN_DEX)) {
858       ruleContext.ruleError(
859           "Both \"main_dex_list\" and \"multidex='manual_main_dex'\" must be specified.");
860       throw new RuleConfigurationException();
861     }
862 
863     if (multidexMode == MultidexMode.OFF) {
864       // Single dex mode: generate classes.dex directly from the input jar.
865       AndroidCommon.createDexAction(
866           ruleContext, proguardedJar, classesDex, dexopts, false, null);
867       return new DexingOutput(classesDex, deployJar, ImmutableList.of(classesDex));
868     } else {
869       // Multidex mode: generate classes.dex.zip, where the zip contains [classes.dex,
870       // classes2.dex, ... classesN.dex]. Because the dexer also places resources into this zip,
871       // we also need to create a cleanup action that removes all non-.dex files before staging
872       // for apk building.
873       if (multidexMode == MultidexMode.LEGACY) {
874         // For legacy multidex, we need to generate a list for the dexer's --main-dex-list flag.
875         mainDexList = createMainDexListAction(ruleContext, proguardedJar);
876       }
877 
878       if (dexShards > 1) {
879         List<Artifact> shardJars = new ArrayList<>(dexShards);
880         for (int i = 1; i <= dexShards; i++) {
881           shardJars.add(getDxArtifact(ruleContext, "shard" + i + ".jar"));
882         }
883 
884         CustomCommandLine.Builder shardCommandLine = CustomCommandLine.builder()
885             .addBeforeEachExecPath("--output_jar", shardJars);
886 
887         Artifact javaResourceJar =
888             ruleContext.getImplicitOutputArtifact(AndroidRuleClasses.JAVA_RESOURCES_JAR);
889 
890         if (mainDexList != null) {
891           shardCommandLine.addExecPath("--main_dex_filter", mainDexList);
892         }
893 
894         // If we need to run Proguard, all the class files will be in the Proguarded jar and the
895         // deploy jar will already have been built (since it's the input of Proguard) and it will
896         // contain all the Java resources. Otherwise, we don't want to have deploy jar creation on
897         // the critical path, so we put all the jar files that constitute it on the inputs of the
898         // jar shuffler.
899         if (proguardedJar != deployJar) {
900           shardCommandLine.addExecPath("--input_jar", proguardedJar);
901         } else {
902           shardCommandLine
903               .addBeforeEachExecPath("--input_jar", common.getRuntimeJars())
904               .addBeforeEachExecPath("--input_jar", attributes.getRuntimeClassPathForArchive());
905         }
906 
907         shardCommandLine.addExecPath("--output_resources", javaResourceJar);
908 
909         SpawnAction.Builder shardAction = new SpawnAction.Builder()
910             .setMnemonic("ShardClassesToDex")
911             .setProgressMessage("Sharding classes for dexing for " + ruleContext.getLabel())
912             .setExecutable(ruleContext.getExecutablePrerequisite("$shuffle_jars", Mode.HOST))
913             .addOutputs(shardJars)
914             .addOutput(javaResourceJar)
915             .setCommandLine(shardCommandLine.build());
916 
917         if (mainDexList != null) {
918           shardAction.addInput(mainDexList);
919         }
920         if (proguardedJar != deployJar) {
921           shardAction.addInput(proguardedJar);
922         } else {
923           shardAction
924               .addInputs(common.getRuntimeJars())
925               .addInputs(attributes.getRuntimeClassPathForArchive());
926         }
927 
928         ruleContext.registerAction(shardAction.build(ruleContext));
929 
930         List<Artifact> shardDexes = new ArrayList<>(dexShards);
931         for (int i = 1; i <= dexShards; i++) {
932           Artifact shardJar = shardJars.get(i - 1);
933           Artifact shard = getDxArtifact(ruleContext, "shard" + i + ".dex.zip");
934           shardDexes.add(shard);
935           AndroidCommon.createDexAction(
936               ruleContext, shardJar, shard, dexopts, true, null);
937         }
938 
939         CommandLine mergeCommandLine = CustomCommandLine.builder()
940             .addBeforeEachExecPath("--input_zip", shardDexes)
941             .addExecPath("--output_zip", classesDex)
942             .build();
943         ruleContext.registerAction(new SpawnAction.Builder()
944             .setMnemonic("MergeDexZips")
945             .setProgressMessage("Merging dex shards for " + ruleContext.getLabel())
946             .setExecutable(ruleContext.getExecutablePrerequisite("$merge_dexzips", Mode.HOST))
947             .addInputs(shardDexes)
948             .addOutput(classesDex)
949             .setCommandLine(mergeCommandLine)
950             .build(ruleContext));
951         return new DexingOutput(classesDex, javaResourceJar, shardDexes);
952       } else {
953         // Create an artifact for the intermediate zip output that includes non-.dex files.
954         Artifact classesDexIntermediate = AndroidBinary.getDxArtifact(
955             ruleContext,
956             "intermediate_" + classesDexFileName);
957 
958         // Have the dexer generate the intermediate file and the "cleaner" action consume this to
959         // generate the final archive with only .dex files.
960         AndroidCommon.createDexAction(ruleContext, proguardedJar,
961             classesDexIntermediate, dexopts, true, mainDexList);
962         createCleanDexZipAction(ruleContext, classesDexIntermediate, classesDex);
963         return new DexingOutput(classesDex, deployJar, ImmutableList.of(classesDex));
964       }
965     }
966   }

1148      * Creates a generating action for {@code outApk} that builds the APK specified.
1149      */
1150     public Action[] build(Artifact outApk) {
1151       Builder actionBuilder = new SpawnAction.Builder()
1152           .setExecutable(AndroidSdkProvider.fromRuleContext(ruleContext).getApkBuilder())
1153           .setProgressMessage(message)
1154           .setMnemonic("AndroidApkBuilder")
1155           .addOutputArgument(outApk);
1156 
1157       if (javaResourceZip != null) {
1158         actionBuilder
1159             .addArgument("-rj")
1160             .addInputArgument(javaResourceZip);
1161       }
1162 
1163       Artifact nativeSymlinks = nativeLibs.createApkBuilderSymlinks(ruleContext);
1164       if (nativeSymlinks != null) {
1165         PathFragment nativeSymlinksDir = nativeSymlinks.getExecPath().getParentDirectory();
1166         actionBuilder
1167             .addInputManifest(nativeSymlinks, nativeSymlinksDir)
1168             .addInput(nativeSymlinks)
1169             .addInputs(nativeLibs.getAllNativeLibs())
1170             .addArgument("-nf")
1171             // If the native libs are "foo/bar/x86/foo.so", we need to pass "foo/bar" here
1172             .addArgument(nativeSymlinksDir.getPathString());
1173       }
1174 
1175       if (nativeLibs.getName() != null) {
1176         actionBuilder
1177             .addArgument("-rf")
1178             .addArgument(nativeLibs.getName().getExecPath().getParentDirectory().getPathString())
1179             .addInput(nativeLibs.getName());
1180       }
1181 
1182       if (javaResourceFile != null) {
1183         actionBuilder
1184             .addArgument("-rf")
1185             .addArgument((javaResourceFile.getExecPath().getParentDirectory().getPathString()))
1186             .addInput(javaResourceFile);
1187       }
1188 
1189       semantics.addSigningArguments(ruleContext, sign, actionBuilder);
1190 
1191       actionBuilder
1192           .addArgument("-z")
1193           .addInputArgument(resourceApk);
1194 
1195       if (classesDex != null) {
1196         actionBuilder
1197             .addArgument(classesDex.getFilename().endsWith(".dex") ? "-f" : "-z")
1198             .addInputArgument(classesDex);
1199       }
1200 
1201       return actionBuilder.build(ruleContext);
1202     }
1203   }

1205   /** Last step in buildings an apk: align the zip boundaries by 4 bytes. */
1206   static Artifact zipalignApk(RuleContext ruleContext,
1207       Artifact signedApk, Artifact zipAlignedApk) {
1208     List<String> args = new ArrayList<>();
1209     // "4" is the only valid value for zipalign, according to:
1210     // http://developer.android.com/guide/developing/tools/zipalign.html
1211     args.add("4");
1212     args.add(signedApk.getExecPathString());
1213     args.add(zipAlignedApk.getExecPathString());
1214 
1215     ruleContext.registerAction(new SpawnAction.Builder()
1216         .addInput(signedApk)
1217         .addOutput(zipAlignedApk)
1218         .setExecutable(AndroidSdkProvider.fromRuleContext(ruleContext).getZipalign())
1219         .addArguments(args)
1220         .setProgressMessage("Zipaligning apk")
1221         .setMnemonic("AndroidZipAlign")
1222         .build(ruleContext));
1223     args.add(signedApk.getExecPathString());
1224     args.add(zipAlignedApk.getExecPathString());
1225     return zipAlignedApk;
1226   }

1284   /**
1285    * Returns true if the runtime contained in the Android SDK used to build this rule supports the
1286    * given version of multidex mode specified, false otherwise.
1287    */
1288   public static boolean supportsMultidexMode(RuleContext ruleContext, MultidexMode mode) {
1289     if (mode == MultidexMode.NATIVE) {
1290       // Native mode is not supported by Android devices running Android before v21.
1291       String runtime =
1292           AndroidSdkProvider.fromRuleContext(ruleContext).getAndroidJar().getExecPathString();
1293       for (String blacklistedRuntime : RUNTIMES_THAT_DONT_SUPPORT_NATIVE_MULTIDEXING) {
1294         if (runtime.contains(blacklistedRuntime)) {
1295           return false;
1296         }
1297       }
1298     }
1299     return true;
