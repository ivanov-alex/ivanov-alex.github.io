40  /**
41   * A class representing types available in Skylark.
42   *
43   * <p>A SkylarkType can be one of:
44   * <ul>
45   * <li>a Simple type that contains exactly the objects in a given class,
46   *  (including the special TOP and BOTTOM types that respectively contain
47   *  all the objects (Simple type for Object.class) and no object at all
48   *  (Simple type for EmptyType.class, isomorphic to Void.class).
49   * <li>a Combination of a generic class (one of LIST, MAP, SET)
50   *  and an argument type (that itself need not be Simple).
51   * <li>a Union of a finite set of types
52   * <li>a FunctionType associated with a name and a returnType
53   * </ul>
54   *
55   * <p>In a style reminiscent of Java's null, Skylark's None is in all the types
56   *  as far as type inference goes, yet actually no type .contains(it).
57   *
58   * <p>The current implementation fails to distinguish between TOP and ANY,
59   * between BOTTOM and EMPTY (VOID, ZERO, FALSE):
60   * <ul>
61   * <li>In type analysis, we often distinguish a notion of "the type of this object"
62   *  from the notion of "what I know about the type of this object".
63   *  Some languages have a Universal Base Class that contains all objects, and would be the ANY type.
64   *  The Skylark runtime, written in Java, has this ANY type, Java's Object.class.
65   *  But the Skylark validation engine doesn't really have a concept of an ANY class;
66   *  however, it does have a concept of a yet-undermined class, the TOP class
67   *  (called UNKOWN in previous code). In the future, we may have to distinguish between the two,
68   *  at which point type constructor classes would have to be generic in
69   *  "actual type" vs "partial knowledge of type".
70   * <li>Similarly, and EMPTY type (also known as VOID, ZERO or FALSE, in other contexts)
71   *  is a type that has no instance, whereas the BOTTOM type is the type analysis that says
72   *  that there is no possible runtime type for the given object, which may imply that
73   *  the point in the program at which the object is evaluated cannot be reached, etc.
74   * </ul>
75   * So for now, we have puns between TOP and ANY, BOTTOM and EMPTY, between runtime (eval) and
76   * validation-time (validate). Yet in the future, we may need to make a clear distinction,
77   * especially if we are to have types such List(Any) vs List(Top), which contains the former,
78   * but also plenty of other quite distinct types. And yet in a future future, the TOP type
79   * would not be represented explicitly, instead a new type variable would be inserted everywhere
80   * a type is unknown, to be unified with further type information as it becomes available.
81   */
82  // TODO(bazel-team): move the FunctionType side-effect out of the type object
83  // and into the validation environment.

94     * intersectWith() is the internal method from which function intersection(t1, t2) is computed.
95     * OVERRIDE this method in your classes, but DO NOT TO CALL it: only call intersection().
96     * When computing intersection(t1, t2), whichever type defined before the other
97     * knows nothing about the other and about their intersection, and returns BOTTOM;
98     * the other knows about the former, and returns their intersection (which may be BOTTOM).
99     * intersection() will call in one order then the other, and return whichever answer
100    * isn't BOTTOM, if any. By default, types are disjoint and their intersection is BOTTOM.
101    */

103   protected SkylarkType intersectWith(SkylarkType other) {
104     return BOTTOM;
105   }

121   public static SkylarkType intersection(SkylarkType t1, SkylarkType t2) {
122     if (t1.equals(t2)) {
123       return t1;
124     }
125     SkylarkType t = t1.intersectWith(t2);
126     if (t == BOTTOM) {
127       return t2.intersectWith(t1);
128     } else {
129       return t;
130     }
131   }
132 
133   public boolean includes(SkylarkType other) {
134     return intersection(this, other).equals(other);
135   }

223   /** a Simple type contains the instance of a Java class */
224   public static class Simple extends SkylarkType {
225     private final Class<?> type;
226 
227     private Simple(Class<?> type) {
228       this.type = type;
229     }
230 
231     @Override public boolean contains(Object value) {
232       return value != null && type.isInstance(value);
233     }
234     @Override public Class<?> getType() {
235       return type;
236     }
237     @Override public boolean equals(Object other) {
238       return this == other
239           || (this.getClass() == other.getClass() && this.type.equals(((Simple) other).getType()));
240     }
241     @Override public int hashCode() {
242       return 0x513973 + type.hashCode() * 503; // equal underlying types yield the same hashCode
243     }
244     @Override public String toString() {
245       return EvalUtils.getDataTypeNameFromClass(type);
246     }
247     @Override public boolean canBeCastTo(Class<?> type) {
248       return this.type == type || super.canBeCastTo(type);
249     }
250 
251     private static LoadingCache<Class<?>, Simple> simpleCache = CacheBuilder.newBuilder()
252       .build(new CacheLoader<Class<?>, Simple>() {
253           @Override
254           public Simple load(Class<?> type) {
255             return create(type);
256           }
257         });
258 
259     private static Simple create(Class<?> type) {
260       Simple simple;
261       if (type == Object.class) {
262         // Note that this is a bad encoding for "anything", not for "everything", i.e.
263         // for skylark there isn't a type that contains everything, but there's a Top type
264         // that corresponds to not knowing yet which more special type it will be.
265         simple = TOP;
266       } else if (type == Empty.class) {
267         simple = BOTTOM;
268       } else {
269         // Consider all classes that have the same EvalUtils.getSkylarkType() as equivalent,
270         // as a substitute to handling inheritance.
271         Class<?> skylarkType = EvalUtils.getSkylarkType(type);
272         if (skylarkType != type) {
273           simple = Simple.of(skylarkType);
274         } else {
275           simple = new Simple(type);
276         }
277       }
278       return simple;
279     }
280 
281     /**
282      * The public way to create a Simple type
283      * @param type a Class
284      * @return the Simple type that contains exactly the instances of that Class
285      */
286     public static Simple of(Class<?> type) {
287       return simpleCache.getUnchecked(type);
288     }
289   }

291   /** Combination of a generic type and an argument type */
292   public static class Combination extends SkylarkType {
293     // For the moment, we can only combine a Simple type with a Simple type,
294     // and the first one has to be a Java generic class,
295     // and in practice actually one of SkylarkList or SkylarkNestedSet
296     private final SkylarkType genericType; // actually always a Simple, for now.
297     private final SkylarkType argType; // not always Simple
298     private Combination(SkylarkType genericType, SkylarkType argType) {
299       this.genericType = genericType;
300       this.argType = argType;
301     }
302 
303     public boolean contains(Object value) {
304       // The empty collection is member of compatible types
305       if (value == null || !genericType.contains(value)) {
306         return false;
307       } else {
308         SkylarkType valueArgType = getGenericArgType(value);
309         return valueArgType == TOP // empty objects are universal
310             || argType.includes(valueArgType);
311       }
312     }
313     @Override public SkylarkType intersectWith(SkylarkType other) {
314       // For now, we only accept generics with a single covariant parameter
315       if (genericType.equals(other)) {
316         return this;
317       } else if (other instanceof Combination
318           && genericType.equals(((Combination) other).getGenericType())
319           && argType.includes(((Combination) other).getArgType())) {
320         return other;
321       } else if ((LIST.equals(other) || SET.equals(other)) && genericType.equals(other)) {
322         return this;
323       } else {
324         return BOTTOM;
325       }
326     }
327 
328     @Override public boolean equals(Object other) {
329       if (this == other) {
330         return true;
331       } else if (this.getClass() == other.getClass()) {
332         Combination o = (Combination) other;
333         return genericType.equals(o.getGenericType())
334             && argType.equals(o.getArgType());
335       } else {
336         return false;
337       }
338     }
339     @Override public int hashCode() {
340       // equal underlying types yield the same hashCode
341       return 0x20B14A71 + genericType.hashCode() * 1009 + argType.hashCode() * 1013;
342     }
343     @Override public Class<?> getType() {
344       return genericType.getType();
345     }
346     SkylarkType getGenericType() {
347       return genericType;
348     }
349     @Override
350     public SkylarkType getArgType() {
351       return argType;
352     }
353     @Override public String toString() {
354       return genericType + " of " + argType + "s";
355     }
356 
357     private static Interner<Combination> combinationInterner =
358         Interners.<Combination>newWeakInterner();
359 
360     public static SkylarkType of(SkylarkType generic, SkylarkType argument) {
361       // assume all combinations with TOP are the same as the simple type, and canonicalize.
362       Preconditions.checkArgument(generic instanceof Simple);
363       if (argument == TOP) {
364         return generic;
365       } else {
366         return combinationInterner.intern(new Combination(generic, argument));
367       }
368     }
369     public static SkylarkType of(Class<?> generic, Class<?> argument) {
370       return of(Simple.of(generic), Simple.of(argument));
371     }
372   }

374   /** Union types, used a lot in "dynamic" languages such as Python or Skylark */
375   public static class Union extends SkylarkType {
376     private final ImmutableList<SkylarkType> types;
377     private Union(ImmutableList<SkylarkType> types) {
378       this.types = types;
379     }
380 
381     public boolean contains(Object value) {
382       for (SkylarkType type : types) {
383         if (type.contains(value)) {
384           return true;
385         }
386       }
387       return false;
388     }
389     @Override public boolean equals(Object other) {
390       if (this.getClass() == other.getClass()) {
391         Union o = (Union) other;
392         if (types.containsAll(o.types) && o.types.containsAll(types)) {
393           return true;
394         }
395       }
396       return false;
397     }
398     @Override public int hashCode() {
399       // equal underlying types yield the same hashCode
400       int h = 0x4104;
401       for (SkylarkType type : types) {
402         // Important: addition is commutative, like Union
403         h += type.hashCode();
404       }
405       return h;
406     }
407     @Override public String toString() {
408       return Joiner.on(" or ").join(types);
409     }
410     public static List<SkylarkType> addElements(List<SkylarkType> list, SkylarkType type) {
411       if (type instanceof Union) {
412         list.addAll(((Union) type).types);
413       } else if (type != BOTTOM) {
414         list.add(type);
415       }
416       return list;
417     }
418     @Override public SkylarkType intersectWith(SkylarkType other) {
419       List<SkylarkType> otherTypes = addElements(new ArrayList<SkylarkType>(), other);
420       List<SkylarkType> results = new ArrayList<>();
421       for (SkylarkType element : types) {
422         for (SkylarkType otherElement : otherTypes) {
423           addElements(results, intersection(element, otherElement));
424         }
425       }
426       return Union.of(results);
427     }
428     public static SkylarkType of(List<SkylarkType> types) {
429       // When making the union of many types,
430       // canonicalize them into elementary (non-Union) types,
431       // and then eliminate trivially redundant types from the list.
432 
433       // list of all types in the input
434       ArrayList<SkylarkType> elements = new ArrayList<>();
435       for (SkylarkType type : types) {
436         addElements(elements, type);
437       }
438 
439       // canonicalized list of types
440       ArrayList<SkylarkType> canonical = new ArrayList<>();
441 
442       for (SkylarkType newType : elements) {
443         boolean done = false; // done with this element?
444         int i = 0;
445         for (SkylarkType existingType : canonical) {
446           SkylarkType both = intersection(newType, existingType);
447           if (newType.equals(both)) { // newType already included
448             done = true;
449             break;
450           } else if (existingType.equals(both)) { // newType supertype of existingType
451             canonical.set(i, newType);
452             done = true;
453             break;
454           }
455         }
456         if (!done) {
457           canonical.add(newType);
458         }
459       }
460       if (canonical.isEmpty()) {
461         return BOTTOM;
462       } else if (canonical.size() == 1) {
463         return canonical.get(0);
464       } else {
465         return new Union(ImmutableList.<SkylarkType>copyOf(canonical));
466       }
467     }
468     public static SkylarkType of(SkylarkType... types) {
469       return of(Arrays.asList(types));
470     }
471     public static SkylarkType of(SkylarkType t1, SkylarkType t2) {
472       return of(ImmutableList.<SkylarkType>of(t1, t2));
473     }
474     public static SkylarkType of(Class<?> t1, Class<?> t2) {
475       return of(Simple.of(t1), Simple.of(t2));
476     }
477   }

499   /**
500    * A class representing the type of a Skylark function.
501    */
502   public static final class SkylarkFunctionType extends SkylarkType {
503     private final String name;
504     @Nullable private SkylarkType returnType;
505 
506     @Override public SkylarkType intersectWith(SkylarkType other) {
507       // This gives the wrong result if both return types are incompatibly updated later!
508       if (other instanceof SkylarkFunctionType) {
509         SkylarkFunctionType fun = (SkylarkFunctionType) other;
510         SkylarkType type1 = returnType == null ? TOP : returnType;
511         SkylarkType type2 = fun.returnType == null ? TOP : fun.returnType;
512         SkylarkType bothReturnType = intersection(returnType, fun.returnType);
513         if (type1.equals(bothReturnType)) {
514           return this;
515         } else if (type2.equals(bothReturnType)) {
516           return fun;
517         } else {
518           return new SkylarkFunctionType(name, bothReturnType);
519         }
520       } else {
521         return BOTTOM;
522       }
523     }
524     @Override public Class<?> getType() {
525       return BaseFunction.class;
526     }
527     @Override public String toString() {
528       return (returnType == TOP || returnType == null ? "" : returnType + "-returning ")
529           + "function";
530     }
531 
532     public boolean contains(Object value) {
533       // This returns true a bit too much, not looking at the result type.
534       return value instanceof BaseFunction;
535     }
536 
537     public static SkylarkFunctionType of(String name, SkylarkType returnType) {
538       return new SkylarkFunctionType(name, returnType);
539     }

542       this.name = name;
543       this.returnType = returnType;
544     }

551     if (value == null) {
552       return BOTTOM;
553     } else if (value instanceof SkylarkList) {

555     } else if (value instanceof SkylarkNestedSet) {
556       return of(SET, ((SkylarkNestedSet) value).getContentType());
557     } else {
558       return Simple.of(value.getClass());
559     }

565     } else if (value instanceof SkylarkNestedSet) {
566       return ((SkylarkNestedSet) value).getContentType();
567     } else {
568       return TOP;

572   private static boolean isTypeAllowedInSkylark(Object object) {
573     if (object instanceof NestedSet<?>) {
574       return false;

611    * General purpose type-casting facility.
612    *
613    * @param value - the actual value of the parameter
614    * @param genericType - a generic class of one argument for the value
615    * @param argType - a covariant argument for the generic class
616    * @param loc - the location info used in the EvalException
617    * @param format - a format String
618    * @param args - arguments to format, in case there's an exception

622       Location loc, String format, Object... args) throws EvalException {
623     if (of(genericType, argType).contains(value)) {
624       return (T) value;
625     } else {
626       throw new EvalException(loc, String.format(format, args));
627     }

638   /** Cast a List or SkylarkList object into an Iterable of the given type. null means empty List */
639   public static <TYPE> Iterable<TYPE> castList(
640       Object obj, final Class<TYPE> type, final String what) throws EvalException {
641     if (obj == null) {
642       return ImmutableList.of();
643     }
644     List<TYPE> results = new ArrayList<>();
645     for (Object object : com.google.devtools.build.lib.packages.Type.LIST.convert(obj, what)) {
646       try {
647         results.add(type.cast(object));
648       } catch (ClassCastException e) {
649         throw new EvalException(null, String.format(
650             "Illegal argument: expected %s type for '%s' but got %s instead",
651             EvalUtils.getDataTypeNameFromClass(type), what,
652             EvalUtils.getDataTypeName(object)));
653       }
654     }
655     return results;
656   }

658   /**
659    * Cast a Map object into an Iterable of Map entries of the given key, value types.
660    * @param obj the Map object, where null designates an empty map
661    * @param keyType the class of map keys
662    * @param valueType the class of map values
663    * @param what a string indicating what this is about, to include in case of error
664    */
665   @SuppressWarnings("unchecked")
666   public static <KEY_TYPE, VALUE_TYPE> Map<KEY_TYPE, VALUE_TYPE> castMap(Object obj,
667       Class<KEY_TYPE> keyType, Class<VALUE_TYPE> valueType, String what)
668       throws EvalException {
669     if (obj == null) {
670       return ImmutableMap.of();
671     }
672     if (!(obj instanceof Map<?, ?>)) {
673       throw new EvalException(null, String.format(
674           "Illegal argument: expected a dictionary for %s but got %s instead",
675           what, EvalUtils.getDataTypeName(obj)));
676     }
677 
678     for (Map.Entry<?, ?> input : ((Map<?, ?>) obj).entrySet()) {
679       if (!keyType.isAssignableFrom(input.getKey().getClass())
680           || !valueType.isAssignableFrom(input.getValue().getClass())) {
681         throw new EvalException(null, String.format(
682             "Illegal argument: expected <%s, %s> type for '%s' but got <%s, %s> instead",
683             keyType.getSimpleName(), valueType.getSimpleName(), what,
684             EvalUtils.getDataTypeName(input.getKey()),
685             EvalUtils.getDataTypeName(input.getValue())));
686       }
687     }
688 
689     return (Map<KEY_TYPE, VALUE_TYPE>) obj;
690   }

692   private static Class<?> getGenericTypeFromMethod(Method method) {
693     // This is where we can infer generic type information, so SkylarkNestedSets can be
694     // created in a safe way. Eventually we should probably do something with Lists and Maps too.
695     ParameterizedType t = (ParameterizedType) method.getGenericReturnType();
696     Type type = t.getActualTypeArguments()[0];
697     if (type instanceof Class) {
698       return (Class<?>) type;
699     }
700     if (type instanceof WildcardType) {
701       WildcardType wildcard = (WildcardType) type;
702       Type upperBound = wildcard.getUpperBounds()[0];
703       if (upperBound instanceof Class) {
704         // i.e. List<? extends SuperClass>
705         return (Class<?>) upperBound;
706       }
707     }
708     // It means someone annotated a method with @SkylarkCallable with no specific generic type info.
709     // We shouldn't annotate methods which return List<?> or List<T>.
710     throw new IllegalStateException("Cannot infer type from method signature " + method);
711   }

713   /**
714    * Converts an object retrieved from a Java method to a Skylark-compatible type.
715    */
716   static Object convertToSkylark(Object object, Method method) {
717     if (object instanceof NestedSet<?>) {
718       return new SkylarkNestedSet(getGenericTypeFromMethod(method), (NestedSet<?>) object);
719     } else if (object instanceof List<?>) {
720       return SkylarkList.list((List<?>) object, getGenericTypeFromMethod(method));
721     }
722     return object;
723   }
