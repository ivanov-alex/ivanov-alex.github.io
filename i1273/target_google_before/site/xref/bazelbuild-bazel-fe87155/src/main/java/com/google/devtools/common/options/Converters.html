33    /**
34     * Join a list of words as in English.  Examples:
35     * "nothing"
36     * "one"
37     * "one or two"
38     * "one and two"
39     * "one, two or three".
40     * "one, two and three".
41     * The toString method of each element is used.
42     */
43    static String joinEnglishList(Iterable<?> choices) {
44      StringBuilder buf = new StringBuilder();
45      for (Iterator<?> ii = choices.iterator(); ii.hasNext(); ) {
46        Object choice = ii.next();
47        if (buf.length() > 0) {
48          buf.append(ii.hasNext() ? ", " : " or ");
49        }
50        buf.append(choice);
51      }
52      return buf.length() == 0 ? "nothing" : buf.toString();
53    }

55    public static class SeparatedOptionListConverter
56        implements Converter<List<String>> {
57  
58      private final String separatorDescription;
59      private final Splitter splitter;
60  
61      protected SeparatedOptionListConverter(char separator,
62                                             String separatorDescription) {
63        this.separatorDescription = separatorDescription;
64        this.splitter = Splitter.on(separator);
65      }
66  
67      @Override
68      public List<String> convert(String input) {
69        return input.equals("")
70            ? ImmutableList.<String>of()
71            : ImmutableList.copyOf(splitter.split(input));
72      }
73  
74      @Override
75      public String getTypeDescription() {
76        return separatorDescription + "-separated list of options";
77      }
78    }

93    public static class LogLevelConverter implements Converter<Level> {
94  
95      public static Level[] LEVELS = new Level[] {
96        Level.OFF, Level.SEVERE, Level.WARNING, Level.INFO, Level.FINE,
97        Level.FINER, Level.FINEST
98      };
99  
100     @Override
101     public Level convert(String input) throws OptionsParsingException {
102       try {
103         int level = Integer.parseInt(input);
104         return LEVELS[level];
105       } catch (NumberFormatException e) {
106         throw new OptionsParsingException("Not a log level: " + input);
107       } catch (ArrayIndexOutOfBoundsException e) {
108         throw new OptionsParsingException("Not a log level: " + input);
109       }
110     }
111 
112     @Override
113     public String getTypeDescription() {
114       return "0 <= an integer <= " + (LEVELS.length - 1);
115     }

117   }

119   /**
120    * Checks whether a string is part of a set of strings.
121    */
122   public static class StringSetConverter implements Converter<String> {
123 
124     // TODO(bazel-team): if this class never actually contains duplicates, we could s/List/Set/
125     // here.
126     private final List<String> values;
127 
128     public StringSetConverter(String... values) {
129       this.values = ImmutableList.copyOf(values);
130     }
131 
132     @Override
133     public String convert(String input) throws OptionsParsingException {
134       if (values.contains(input)) {
135         return input;
136       }
137 
138       throw new OptionsParsingException("Not one of " + values);
139     }
140 
141     @Override
142     public String getTypeDescription() {
143       return joinEnglishList(values);
144     }
145   }

152     @Override
153     public Pattern convert(String input) throws OptionsParsingException {
154       try {
155         return Pattern.compile(input);

157         throw new OptionsParsingException("Not a valid regular expression: " + e.getMessage());

159     }
160 
161     @Override
162     public String getTypeDescription() {
163       return "a valid Java regular expression";
164     }

191   /**
192    * Checks whether an integer is in the given range.
193    */
194   public static class RangeConverter implements Converter<Integer> {
195     final int minValue;
196     final int maxValue;
197 
198     public RangeConverter(int minValue, int maxValue) {
199       this.minValue = minValue;
200       this.maxValue = maxValue;
201     }
202 
203     @Override
204     public Integer convert(String input) throws OptionsParsingException {
205       try {
206         Integer value = Integer.parseInt(input);
207         if (value < minValue) {
208           throw new OptionsParsingException("'" + input + "' should be >= " + minValue);
209         } else if (value < minValue || value > maxValue) {
210           throw new OptionsParsingException("'" + input + "' should be <= " + maxValue);
211         }
212         return value;
213       } catch (NumberFormatException e) {
214         throw new OptionsParsingException("'" + input + "' is not an int");
215       }
216     }
217 
218     @Override
219     public String getTypeDescription() {
220       if (minValue == Integer.MIN_VALUE) {
221         if (maxValue == Integer.MAX_VALUE) {
222           return "an integer";
223         } else {
224           return "an integer, <= " + maxValue;
225         }
226       } else if (maxValue == Integer.MAX_VALUE) {
227         return "an integer, >= " + minValue;
228       } else {
229         return "an integer in "
230             + (minValue < 0 ? "(" + minValue + ")" : minValue) + "-" + maxValue + " range";
231       }
232     }
233   }

235   /**
236    * A converter for variable assignments from the parameter list of a blaze
237    * command invocation. Assignments are expected to have the form "name=value",
238    * where names and values are defined to be as permissive as possible.
239    */
240   public static class AssignmentConverter implements Converter<Map.Entry<String, String>> {
241 
242     @Override
243     public Map.Entry<String, String> convert(String input)
244         throws OptionsParsingException {
245       int pos = input.indexOf("=");
246       if (pos <= 0) {
247         throw new OptionsParsingException("Variable definitions must be in the form of a "
248             + "'name=value' assignment");
249       }
250       String name = input.substring(0, pos);
251       String value = input.substring(pos + 1);
252       return Maps.immutableEntry(name, value);
253     }
254 
255     @Override
256     public String getTypeDescription() {
257       return "a 'name=value' assignment";
258     }
259 
260   }

262   /**
263    * A converter for variable assignments from the parameter list of a blaze
264    * command invocation. Assignments are expected to have the form "name[=value]",
265    * where names and values are defined to be as permissive as possible and value
266    * part can be optional (in which case it is considered to be null).
267    */
268   public static class OptionalAssignmentConverter implements Converter<Map.Entry<String, String>> {
269 
270     @Override
271     public Map.Entry<String, String> convert(String input)
272         throws OptionsParsingException {
273       int pos = input.indexOf("=");
274       if (pos == 0 || input.length() == 0) {
275         throw new OptionsParsingException("Variable definitions must be in the form of a "
276             + "'name=value' or 'name' assignment");
277       } else if (pos < 0) {
278         return Maps.immutableEntry(input, null);
279       }
280       String name = input.substring(0, pos);
281       String value = input.substring(pos + 1);
282       return Maps.immutableEntry(name, value);
283     }
284 
285     @Override
286     public String getTypeDescription() {
287       return "a 'name=value' assignment with an optional value part";
288     }

298   /**
299    * A converter for boolean values. This is already one of the defaults, so clients
300    * should not typically need to add this.
301    */
302   public static class BooleanConverter implements Converter<Boolean> {
303     @Override
304     public Boolean convert(String input) throws OptionsParsingException {
305       if (input == null) {
306         return false;
307       }
308       input = input.toLowerCase();
309       if (input.equals("true") || input.equals("1") || input.equals("yes") ||
310           input.equals("t") || input.equals("y")) {
311         return true;
312       }
313       if (input.equals("false") || input.equals("0") || input.equals("no") ||
314           input.equals("f") || input.equals("n")) {
315         return false;
316       }
317       throw new OptionsParsingException("'" + input + "' is not a boolean");
318     }
319 
320     @Override
321     public String getTypeDescription() {
322       return "a boolean";
323     }
324   }
