43  /**
44   * Implementation of the semantics of Bazel's constraint specification and enforcement system.
45   *
46   * <p>This is how the system works:
47   *
48   * <p>All build rules can declare which "environments" they can be built for, where an "environment"
49   * is a label instance of an {@link EnvironmentRule} rule declared in a BUILD file. There are
50   * various ways to do this:
51   *
52   * <ul>
53   *   <li>Through a "restricted to" attribute setting
54   *   ({@link RuleClass#RESTRICTED_ENVIRONMENT_ATTR}). This is the most direct form of
55   *   specification - it declares the exact set of environments the rule supports (for its group -
56   *   see precise details below).
57   *   <li>Through a "compatible with" attribute setting
58   *   ({@link RuleClass#COMPATIBLE_ENVIRONMENT_ATTR}. This declares <b>additional</b>
59   *   environments a rule supports in addition to "standard" environments that are supported by
60   *   default (see below).
61   *   <li>Through "default" specifications in {@link EnvironmentGroup} rules. Every environment
62   *   belongs to a group of thematically related peers (e.g. "target architectures", "JDK versions",
63   *   or "mobile devices"). An environment group's definition includes which of these
64   *   environments should be supported "by default" if not otherwise specified by one of the above
65   *   mechanisms. In particular, a rule with no environment-related attributes automatically
66   *   inherits all defaults.
67   *   <li>Through a rule class default ({@link RuleClass.Builder#restrictedTo} and
68   *   {@link RuleClass.Builder#compatibleWith}). This overrides global defaults for all instances
69   *   of the given rule class. This can be used, for example, to make all *_test rules "testable"
70   *   without each instance having to explicitly declare this capability.
71   * </ul>
72   *
73   * <p>Groups exist to model the idea that some environments are related while others have nothing
74   * to do with each other. Say, for example, we want to say a rule works for PowerPC platforms but
75   * not x86. We can do so by setting its "restricted to" attribute to
76   * {@code ['//sample/path:powerpc']}. Because both PowerPC and x86 are in the same
77   * "target architectures" group, this setting removes x86 from the set of supported environments.
78   * But since JDK support belongs to its own group ("JDK versions") it says nothing about which JDK
79   * the rule supports.
80   *
81   * <p>More precisely, if a rule has a "restricted to" value of [A, B, C], this removes support
82   * for all default environments D such that group(D) is in [group(A), group(B), group(C)] AND
83   * D is not in [A, B, C] (in other words, D isn't explicitly opted back in). The rule's full
84   * set of supported environments thus becomes [A, B, C] + all defaults that belong to unrelated
85   * groups.
86   *
87   * <p>If the rule has a "compatible with" value of [E, F, G], these are unconditionally
88   * added to its set of supported environments (in addition to the results from above).
89   *
90   * <p>An environment may not appear in both a rule's "restricted to" and "compatible with" values.
91   * If two environments belong to the same group, they must either both be in "restricted to",
92   * both be in "compatible with", or not explicitly specified.
93   *
94   * <p>Given all the above, constraint enforcement is this: rule A can depend on rule B if, for
95   * every environment A supports, B also supports that environment.
96   */
97  public class ConstraintSemantics {
98    private ConstraintSemantics() {
99    }

101   /**
102    * Provides a set of default environments for a given environment group.
103    */
104   private interface DefaultsProvider {
105     Collection<Label> getDefaults(EnvironmentGroup group);
106   }
107 
108   /**
109    * Provides a group's defaults as specified in the environment group's BUILD declaration.
110    */
111   private static class GroupDefaultsProvider implements DefaultsProvider {
112     @Override
113     public Collection<Label> getDefaults(EnvironmentGroup group) {
114       return group.getDefaults();
115     }
116   }
117 
118   /**
119    * Provides a group's defaults, factoring in rule class defaults as specified by
120    * {@link com.google.devtools.build.lib.packages.RuleClass.Builder#compatibleWith}
121    * and {@link com.google.devtools.build.lib.packages.RuleClass.Builder#restrictedTo}.
122    */
123   private static class RuleClassDefaultsProvider implements DefaultsProvider {
124     private final EnvironmentCollection ruleClassDefaults;
125     private final GroupDefaultsProvider groupDefaults;
126 
127     RuleClassDefaultsProvider(EnvironmentCollection ruleClassDefaults) {
128       this.ruleClassDefaults = ruleClassDefaults;
129       this.groupDefaults = new GroupDefaultsProvider();
130     }
131 
132     @Override
133     public Collection<Label> getDefaults(EnvironmentGroup group) {
134       if (ruleClassDefaults.getGroups().contains(group)) {
135         return ruleClassDefaults.getEnvironments(group);
136       } else {
137         // If there are no rule class defaults for this group, just inherit global defaults.
138         return groupDefaults.getDefaults(group);
139       }
140     }
141   }

143   /**
144    * Collects the set of supported environments for a given rule by merging its
145    * restriction-style and compatibility-style environment declarations as specified by
146    * the given attributes. Only includes environments from "known" groups, i.e. the groups
147    * owning the environments explicitly referenced from these attributes.
148    */
149   private static class EnvironmentCollector {
150     private final RuleContext ruleContext;
151     private final String restrictionAttr;
152     private final String compatibilityAttr;
153     private final DefaultsProvider defaultsProvider;
154 
155     private final EnvironmentCollection restrictionEnvironments;
156     private final EnvironmentCollection compatibilityEnvironments;
157     private final EnvironmentCollection supportedEnvironments;
158 
159     /**
160      * Constructs a new collector on the given attributes.
161      *
162      * @param ruleContext analysis context for the rule
163      * @param restrictionAttr the name of the attribute that declares "restricted to"-style
164      *     environments. If the rule doesn't have this attribute, this is considered an
165      *     empty declaration.
166      * @param compatibilityAttr the name of the attribute that declares "compatible with"-style
167      *     environments. If the rule doesn't have this attribute, this is considered an
168      *     empty declaration.
169      * @param defaultsProvider provider for the default environments within a group if not
170      *     otherwise overriden by the above attributes
171      */
172     EnvironmentCollector(RuleContext ruleContext, String restrictionAttr, String compatibilityAttr,
173         DefaultsProvider defaultsProvider) {
174       this.ruleContext = ruleContext;
175       this.restrictionAttr = restrictionAttr;
176       this.compatibilityAttr = compatibilityAttr;
177       this.defaultsProvider = defaultsProvider;
178 
179       EnvironmentCollection.Builder environmentsBuilder = new EnvironmentCollection.Builder();
180       restrictionEnvironments = collectRestrictionEnvironments(environmentsBuilder);
181       compatibilityEnvironments = collectCompatibilityEnvironments(environmentsBuilder);
182       supportedEnvironments = environmentsBuilder.build();
183     }
184 
185     /**
186      * Returns the set of environments supported by this rule, as determined by the
187      * restriction-style attribute, compatibility-style attribute, and group defaults
188      * provider instantiated with this class.
189      */
190     EnvironmentCollection getEnvironments() {
191       return supportedEnvironments;
192     }
193 
194     /**
195      * Validity-checks that no group has its environment referenced in both the "compatible with"
196      * and restricted to" attributes. Returns true if all is good, returns false and reports
197      * appropriate errors if there are any problems.
198      */
199     boolean validateEnvironmentSpecifications() {
200       ImmutableCollection<EnvironmentGroup> restrictionGroups = restrictionEnvironments.getGroups();
201       boolean hasErrors = false;
202 
203       for (EnvironmentGroup group : compatibilityEnvironments.getGroups()) {
204         if (restrictionGroups.contains(group)) {
205           // To avoid error-spamming the user, when we find a conflict we only report one example
206           // environment from each attribute for that group.
207           Label compatibilityEnv =
208               compatibilityEnvironments.getEnvironments(group).iterator().next();
209           Label restrictionEnv = restrictionEnvironments.getEnvironments(group).iterator().next();
210 
211           if (compatibilityEnv.equals(restrictionEnv)) {
212             ruleContext.attributeError(compatibilityAttr, compatibilityEnv
213                 + " cannot appear both here and in " + restrictionAttr);
214           } else {
215             ruleContext.attributeError(compatibilityAttr, compatibilityEnv + " and "
216                 + restrictionEnv + " belong to the same environment group. They should be declared "
217                 + "together either here or in " + restrictionAttr);
218           }
219           hasErrors = true;
220         }
221       }
222 
223       return !hasErrors;
224     }
225 
226     /**
227      * Adds environments specified in the "restricted to" attribute to the set of supported
228      * environments and returns the environments added.
229      */
230     private EnvironmentCollection collectRestrictionEnvironments(
231         EnvironmentCollection.Builder supportedEnvironments) {
232       return collectEnvironments(restrictionAttr, supportedEnvironments);
233     }
234 
235     /**
236      * Adds environments specified in the "compatible with" attribute to the set of supported
237      * environments, along with all defaults from the groups they belong to. Returns these
238      * environments, not including the defaults.
239      */
240     private EnvironmentCollection collectCompatibilityEnvironments(
241         EnvironmentCollection.Builder supportedEnvironments) {
242       EnvironmentCollection compatibilityEnvironments =
243           collectEnvironments(compatibilityAttr, supportedEnvironments);
244       for (EnvironmentGroup group : compatibilityEnvironments.getGroups()) {
245         supportedEnvironments.putAll(group, defaultsProvider.getDefaults(group));
246       }
247       return compatibilityEnvironments;
248     }
249 
250     /**
251      * Adds environments specified by the given attribute to the set of supported environments
252      * and returns the environments added.
253      *
254      * <p>If this rule doesn't have the given attributes, returns an empty set.
255      */
256     private EnvironmentCollection collectEnvironments(String attrName,
257         EnvironmentCollection.Builder supportedEnvironments) {
258       if (!ruleContext.getRule().isAttrDefined(attrName,  Type.LABEL_LIST)) {
259         return EnvironmentCollection.EMPTY;
260       }
261       EnvironmentCollection.Builder environments = new EnvironmentCollection.Builder();
262       for (TransitiveInfoCollection envTarget :
263           ruleContext.getPrerequisites(attrName, RuleConfiguredTarget.Mode.DONT_CHECK)) {
264         EnvironmentWithGroup envInfo = resolveEnvironment(envTarget);
265         environments.put(envInfo.group(), envInfo.environment());
266         supportedEnvironments.put(envInfo.group(), envInfo.environment());
267       }
268       return environments.build();
269     }
270 
271     /**
272      * Returns the environment and its group. An {@link Environment} rule only "supports" one
273      * environment: itself. Extract that from its more generic provider interface and sanity
274      * check that that's in fact what we see.
275      */
276     private static EnvironmentWithGroup resolveEnvironment(TransitiveInfoCollection envRule) {
277       SupportedEnvironmentsProvider prereq =
278           Preconditions.checkNotNull(envRule.getProvider(SupportedEnvironmentsProvider.class));
279       return Iterables.getOnlyElement(prereq.getEnvironments().getGroupedEnvironments());
280     }
281   }

292   /**
293    * Returns the environment group that owns the given environment. Both must belong to
294    * the same package.
295    *
296    * @throws EnvironmentLookupException if the input is not an {@link EnvironmentRule} or no
297    *     matching group is found
298    */
299   public static EnvironmentGroup getEnvironmentGroup(Target envTarget)
300       throws EnvironmentLookupException {
301     if (!(envTarget instanceof Rule)
302         || !((Rule) envTarget).getRuleClass().equals(EnvironmentRule.RULE_NAME)) {
303       throw new EnvironmentLookupException(
304           envTarget.getLabel() + " is not a valid environment definition");
305     }
306     for (EnvironmentGroup group : envTarget.getPackage().getTargets(EnvironmentGroup.class)) {
307       if (group.getEnvironments().contains(envTarget.getLabel())) {
308         return group;
309       }
310     }
311     throw new EnvironmentLookupException(
312         "cannot find the group for environment " + envTarget.getLabel());
313   }

315   /**
316    * Returns the set of environments this rule supports, applying the logic described in
317    * {@link ConstraintSemantics}.
318    *
319    * <p>Note this set is <b>not complete</b> - it doesn't include environments from groups we don't
320    * "know about". Environments and groups can be declared in any package. If the rule includes
321    * no references to that package, then it simply doesn't know anything about them. But the
322    * constraint semantics say the rule should support the defaults for that group. We encode this
323    * implicitly: given the returned set, for any group that's not in the set the rule is also
324    * considered to support that group's defaults.
325    *
326    * @param ruleContext analysis context for the rule. A rule error is triggered here if
327    *     invalid constraint settings are discovered.
328    * @return the environments this rule supports, not counting defaults "unknown" to this rule
329    *     as described above. Returns null if any errors are encountered.
330    */
331   @Nullable
332   public static EnvironmentCollection getSupportedEnvironments(RuleContext ruleContext) {
333     if (!validateAttributes(ruleContext)) {
334       return null;
335     }
336 
337     // This rule's rule class defaults (or null if the rule class has no defaults).
338     EnvironmentCollector ruleClassCollector = maybeGetRuleClassDefaults(ruleContext);
339     // Default environments for this rule. If the rule has rule class defaults, this is
340     // those defaults. Otherwise it's the global defaults specified by environment_group
341     // declarations.
342     DefaultsProvider ruleDefaults;
343 
344     if (ruleClassCollector != null) {
345       if (!ruleClassCollector.validateEnvironmentSpecifications()) {
346         return null;
347       }
348       ruleDefaults = new RuleClassDefaultsProvider(ruleClassCollector.getEnvironments());
349     } else {
350       ruleDefaults = new GroupDefaultsProvider();
351     }
352 
353     EnvironmentCollector ruleCollector = new EnvironmentCollector(ruleContext,
354         RuleClass.RESTRICTED_ENVIRONMENT_ATTR, RuleClass.COMPATIBLE_ENVIRONMENT_ATTR, ruleDefaults);
355     if (!ruleCollector.validateEnvironmentSpecifications()) {
356       return null;
357     }
358 
359     EnvironmentCollection supportedEnvironments = ruleCollector.getEnvironments();
360     if (ruleClassCollector != null) {
361       // If we have rule class defaults from groups that aren't referenced from the rule itself,
362       // we need to add them in too to override the global defaults.
363       supportedEnvironments =
364           addUnknownGroupsToCollection(supportedEnvironments, ruleClassCollector.getEnvironments());
365     }
366     return supportedEnvironments;
367   }

369   /**
370    * Returns the rule class defaults specified for this rule, or null if there are
371    * no such defaults.
372    */
373   @Nullable
374   private static EnvironmentCollector maybeGetRuleClassDefaults(RuleContext ruleContext) {
375     Rule rule = ruleContext.getRule();
376     String restrictionAttr = RuleClass.DEFAULT_RESTRICTED_ENVIRONMENT_ATTR;
377     String compatibilityAttr = RuleClass.DEFAULT_COMPATIBLE_ENVIRONMENT_ATTR;
378 
379     if (rule.isAttrDefined(restrictionAttr, Type.LABEL_LIST)
380       || rule.isAttrDefined(compatibilityAttr, Type.LABEL_LIST)) {
381       return new EnvironmentCollector(ruleContext, restrictionAttr, compatibilityAttr,
382           new GroupDefaultsProvider());
383     } else {
384       return null;
385     }
386   }

388   /**
389    * Adds environments to an {@link EnvironmentCollection} from groups that aren't already
390    * a part of that collection.
391    *
392    * @param environments the collection to add to
393    * @param toAdd the collection to add. All environments in this collection in groups
394    *     that aren't represented in {@code environments} are added to {@code environments}.
395    * @return the expanded collection.
396    */
397   private static EnvironmentCollection addUnknownGroupsToCollection(
398       EnvironmentCollection environments, EnvironmentCollection toAdd) {
399     EnvironmentCollection.Builder builder = new EnvironmentCollection.Builder();
400     builder.putAll(environments);
401     for (EnvironmentGroup candidateGroup : toAdd.getGroups()) {
402       if (!environments.getGroups().contains(candidateGroup)) {
403         builder.putAll(candidateGroup, toAdd.getEnvironments(candidateGroup));
404       }
405     }
406     return builder.build();
407   }

409   /**
410    * Validity-checks this rule's constraint-related attributes. Returns true if all is good,
411    * returns false and reports appropriate errors if there are any problems.
412    */
413   private static boolean validateAttributes(RuleContext ruleContext) {
414     AttributeMap attributes = ruleContext.attributes();
415 
416     // Report an error if "restricted to" is explicitly set to nothing. Even if this made
417     // conceptual sense, we don't know which groups we should apply that to.
418     String restrictionAttr = RuleClass.RESTRICTED_ENVIRONMENT_ATTR;
419     List<? extends TransitiveInfoCollection> restrictionEnvironments = ruleContext
420         .getPrerequisites(restrictionAttr, RuleConfiguredTarget.Mode.DONT_CHECK);
421     if (restrictionEnvironments.isEmpty()
422         && attributes.isAttributeValueExplicitlySpecified(restrictionAttr)) {
423       ruleContext.attributeError(restrictionAttr, "attribute cannot be empty");
424       return false;
425     }
426 
427     return true;
428   }

430   /**
431    * Performs constraint checking on the given rule's dependencies and reports any errors.
432    *
433    * @param ruleContext the rule to analyze
434    * @param ruleEnvironments the rule's supported environments, as defined by the return
435    *     value of {@link #getSupportedEnvironments}. In particular, for any environment group that's
436    *     not in this collection, the rule is assumed to support the defaults for that group.
437    */
438   public static void checkConstraints(RuleContext ruleContext,
439       EnvironmentCollection ruleEnvironments) {
440     for (TransitiveInfoCollection dependency : getConstraintCheckedDependencies(ruleContext)) {
441       SupportedEnvironmentsProvider depProvider =
442           dependency.getProvider(SupportedEnvironmentsProvider.class);
443       Collection<Label> unsupportedEnvironments =
444           getUnsupportedEnvironments(depProvider.getEnvironments(), ruleEnvironments);
445 
446       if (!unsupportedEnvironments.isEmpty()) {
447         ruleContext.ruleError("dependency " + dependency.getLabel()
448             + " doesn't support expected environment"
449             + (unsupportedEnvironments.size() == 1 ? "" : "s")
450             + ": " + Joiner.on(", ").join(unsupportedEnvironments));
451       }
452     }
453   }

455   /**
456    * Given a collection of environments and a collection of expected environments, returns the
457    * missing environments that would cause constraint expectations to be violated. Includes
458    * the effects of environment group defaults.
459    */
460   public static Collection<Label> getUnsupportedEnvironments(
461       EnvironmentCollection actualEnvironments, EnvironmentCollection expectedEnvironments) {
462     Set<Label> missingEnvironments = new LinkedHashSet<>();
463     Collection<Label> actualEnvironmentLabels = actualEnvironments.getEnvironments();
464 
465     // Check if each explicitly expected environment is satisfied.
466     for (EnvironmentWithGroup expectedEnv : expectedEnvironments.getGroupedEnvironments()) {
467       EnvironmentGroup group = expectedEnv.group();
468       Label environment = expectedEnv.environment();
469       boolean isSatisfied = false;
470       if (actualEnvironments.getGroups().contains(group)) {
471         // If the actual environments include members from the expected environment's group, we
472         // need to either find the environment itself or another one that transitively fulfills it.
473         if (actualEnvironmentLabels.contains(environment)
474             || intersect(actualEnvironmentLabels, group.getFulfillers(environment))) {
475           isSatisfied = true;
476         }
477       } else {
478         // If the actual environments don't reference the expected environment's group at all,
479         // the group's defaults are implicitly included. So we need to check those defaults for
480         // either the expected environment or another environment that transitively fulfills it.
481         if (group.isDefault(environment)
482             || intersect(group.getFulfillers(environment), group.getDefaults())) {
483           isSatisfied = true;
484         }
485       }
486       if (!isSatisfied) {
487         missingEnvironments.add(environment);
488       }
489     }
490 
491     // For any environment group not referenced by the expected environments, its defaults are
492     // implicitly expected. We can ignore this if the actual environments also don't reference the
493     // group (since in that case the same defaults apply), otherwise we have to check.
494     for (EnvironmentGroup group : actualEnvironments.getGroups()) {
495       if (!expectedEnvironments.getGroups().contains(group)) {
496         for (Label expectedDefault : group.getDefaults()) {
497           if (!actualEnvironmentLabels.contains(expectedDefault)
498               && !intersect(actualEnvironmentLabels, group.getFulfillers(expectedDefault))) {
499             missingEnvironments.add(expectedDefault);
500           }
501         }
502       }
503     }
504 
505     return missingEnvironments;
506   }

508   private static boolean intersect(Iterable<Label> labels1, Iterable<Label> labels2) {
509     return !Sets.intersection(Sets.newHashSet(labels1), Sets.newHashSet(labels2)).isEmpty();
510   }

512   /**
513    * Returns all dependencies that should be constraint-checked against the current rule.
514    */
515   private static Iterable<TransitiveInfoCollection> getConstraintCheckedDependencies(
516       RuleContext ruleContext) {
517     Set<TransitiveInfoCollection> depsToCheck = new LinkedHashSet<>();
518     AttributeMap attributes = ruleContext.attributes();
519 
520     for (String attr : attributes.getAttributeNames()) {
521       Attribute attrDef = attributes.getAttributeDefinition(attr);
522       Type<?> attrType = attributes.getAttributeType(attr);
523 
524       // TODO(bazel-team): support a user-definable API for choosing which attributes are checked
525       if ((attrType != Type.LABEL && attrType != Type.LABEL_LIST)
526           || RuleClass.isConstraintAttribute(attr)
527           || attr.equals("visibility")
528           // Use the same implicit deps check that query uses. This facilitates running queries to
529           // determine exactly which rules need to be constraint-annotated for depot migrations.
530           || !Rule.NO_IMPLICIT_DEPS.apply(ruleContext.getRule(), attrDef)
531           // We can't identify host deps by calling BuildConfiguration.isHostConfiguration()
532           // because --nodistinct_host_configuration subverts that call.
533           || attrDef.getConfigurationTransition() == Attribute.ConfigurationTransition.HOST) {
534         continue;
535       }
536 
537       for (TransitiveInfoCollection dep :
538           ruleContext.getPrerequisites(attr, RuleConfiguredTarget.Mode.DONT_CHECK)) {
539         // Output files inherit the environment spec of their generating rule.
540         if (dep instanceof OutputFileConfiguredTarget) {
541           // Note this reassignment means constraint violation errors reference the generating
542           // rule, not the file. This makes the source of the environmental mismatch more clear.
543           dep = ((OutputFileConfiguredTarget) dep).getGeneratingRule();
544         }
545         // Input files don't support environments. We may subsequently opt them into constraint
546         // checking, but for now just pass them by.
547         if (dep.getProvider(SupportedEnvironmentsProvider.class) != null) {
548           depsToCheck.add(dep);
549         }
550       }
551     }
552 
553     return depsToCheck;
554   }
