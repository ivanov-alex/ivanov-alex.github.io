179     private Builder addDependencies(Iterable<XcodeProvider> dependencies, boolean doPropagate) {
180       for (XcodeProvider dependency : dependencies) {
181         // TODO(bazel-team): This is messy. Maybe we should make XcodeProvider be able to specify
182         // how to depend on it rather than require this method to choose based on the dependency's
183         // type.
184         if (dependency.productType == XcodeProductType.EXTENSION) {
185           this.extensions.add(dependency);
186           this.inputsToXcodegen.addTransitive(dependency.inputsToXcodegen);
187         } else {
188           if (doPropagate) {
189             this.propagatedDependencies.add(dependency);
190             this.propagatedDependencies.addTransitive(dependency.propagatedDependencies);
191             this.addTransitiveSets(dependency);
192           } else {
193             this.nonPropagatedDependencies.add(dependency);
194             this.nonPropagatedDependencies.addTransitive(dependency.propagatedDependencies);
195             this.nonPropagatedUserHeaderSearchPaths
196                 .addTransitive(dependency.propagatedUserHeaderSearchPaths);
197             this.nonPropagatedHeaderSearchPaths
198                 .addTransitive(dependency.propagatedHeaderSearchPaths);
199             this.inputsToXcodegen.addTransitive(dependency.inputsToXcodegen);
200           }
201         }
202       }
203       return this;
204     }

358     public static Project fromTopLevelTargets(Iterable<XcodeProvider> topLevelTargets) {
359       NestedSetBuilder<Artifact> inputsToXcodegen = NestedSetBuilder.stableOrder();
360       NestedSetBuilder<Artifact> additionalSources = NestedSetBuilder.stableOrder();
361       for (XcodeProvider target : topLevelTargets) {
362         inputsToXcodegen.addTransitive(target.inputsToXcodegen);
363         additionalSources.addTransitive(target.additionalSources);
364       }
365       return new Project(inputsToXcodegen.build(), additionalSources.build(),
366           ImmutableList.copyOf(topLevelTargets));
367     }

384     /**
385      * Returns all the target controls that must be added to the xcodegen control. No other target
386      * controls are needed to generate a functional project file. This method creates a new list
387      * whenever it is called.
388      */
389     public ImmutableList<TargetControl> targets() {
390       // Collect all the dependencies of all the providers, filtering out duplicates.
391       Set<XcodeProvider> providerSet = new LinkedHashSet<>();
392       for (XcodeProvider target : topLevelTargets) {
393         target.collectProviders(providerSet);
394       }
395 
396       ImmutableList.Builder<TargetControl> controls = new ImmutableList.Builder<>();
397       Map<Label, XcodeProvider> labelToProvider = new HashMap<>();
398       for (XcodeProvider provider : providerSet) {
399         XcodeProvider oldProvider = labelToProvider.put(provider.label, provider);
400         if (oldProvider != null) {
401           if (!oldProvider.architecture.equals(provider.architecture)
402               || oldProvider.configurationDistinguisher != provider.configurationDistinguisher) {
403             // Do not include duplicate dependencies whose architecture or configuration
404             // distinguisher does not match this project's. This check avoids having multiple
405             // conflicting Xcode targets for the same BUILD target that are only distinguished by
406             // these fields (which Xcode does not care about).
407             continue;
408           }
409 
410           throw new IllegalStateException("Depending on multiple versions of the same xcode target "
411               + "is not allowed but occurred for: " + provider.label);
412         }
413         controls.addAll(provider.targetControls());
414       }
415       return controls.build();
416     }
417   }

471   private void collectProviders(Set<XcodeProvider> allProviders) {
472     if (allProviders.add(this)) {
473       for (XcodeProvider dependency : Iterables.concat(propagatedDependencies,
474           nonPropagatedDependencies)) {
475         dependency.collectProviders(allProviders);
476       }
477       for (XcodeProvider justTestHost : testHost.asSet()) {
478         justTestHost.collectProviders(allProviders);
479       }
480       for (XcodeProvider extension : extensions) {
481         extension.collectProviders(allProviders);
482       }
483     }

491   /**
492    * Returns the name of the Xcode target that corresponds to a build target with the given name.
493    * This changes the label to make it a legal Xcode target name (which means removing slashes and
494    * the colon). It also makes the label more readable in the Xcode UI by putting the target name
495    * first and the package elements in reverse. This means the "important" part is visible even if
496    * the project navigator is too narrow to show the entire name.
497    */
498   static String xcodeTargetName(Label label) {
499     return xcodeTargetName(label, /*labelSuffix=*/"");
500   }

502   /**
503    * Returns the name of the companion Xcode library target that corresponds to a build target with
504    * the given name. See {@link XcodeSupport#generateCompanionLibXcodeTarget} for the rationale of
505    * the companion library target and {@link #xcodeTargetName(Label)} for naming details.
506    */
507   static String xcodeCompanionLibTargetName(Label label) {
508     return xcodeTargetName(label, COMPANION_LIB_TARGET_LABEL_SUFFIX);
509   }

511   private static String xcodeTargetName(Label label, String labelSuffix) {
512     String pathFromWorkspaceRoot = (label + labelSuffix).replace("//", "")
513         .replace(':', '/');
514     List<String> components = Splitter.on('/').splitToList(pathFromWorkspaceRoot);
515     return Joiner.on('_').join(Lists.reverse(components));
516   }

522     return generateCompanionLibTarget ? xcodeCompanionLibTargetName(label) : xcodeTargetName(label);
523   }

534   private TargetControl targetControl() {
535     String buildFilePath = label.getPackageFragment().getSafePathString() + "/BUILD";
536     NestedSet<String> userHeaderSearchPaths = NestedSetBuilder.<String>linkOrder()
537         .addTransitive(propagatedUserHeaderSearchPaths)
538         .addTransitive(nonPropagatedUserHeaderSearchPaths)
539         .build();
540     NestedSet<String> headerSearchPaths = NestedSetBuilder.<String>linkOrder()
541         .addTransitive(propagatedHeaderSearchPaths)
542         .addTransitive(nonPropagatedHeaderSearchPaths)
543         .build();
544 
545     // TODO(bazel-team): Add provisioning profile information when Xcodegen supports it.
546     TargetControl.Builder targetControl =
547         TargetControl.newBuilder()
548             .setName(label.getName())
549             .setLabel(xcodeTargetName(label))
550             .setProductType(productType.getIdentifier())
551             .addSupportFile(buildFilePath)
552             .addAllImportedLibrary(Artifact.toExecPaths(objcProvider.get(IMPORTED_LIBRARY)))
553             .addAllUserHeaderSearchPath(userHeaderSearchPaths)
554             .addAllHeaderSearchPath(headerSearchPaths)
555             .addAllSupportFile(Artifact.toExecPaths(headers))
556             .addAllCopt(compilationModeCopts)
557             .addAllCopt(IosSdkCommands.DEFAULT_COMPILER_FLAGS)
558             .addAllCopt(Interspersing.prependEach("-D", objcProvider.get(DEFINE)))
559             .addAllCopt(copts)
560             .addAllLinkopt(
561                 Interspersing.beforeEach("-force_load", objcProvider.get(FORCE_LOAD_FOR_XCODEGEN)))
562             .addAllLinkopt(IosSdkCommands.DEFAULT_LINKER_FLAGS)
563             .addAllLinkopt(
564                 Interspersing.beforeEach(
565                     "-weak_framework", SdkFramework.names(objcProvider.get(WEAK_SDK_FRAMEWORK))))
566             .addAllBuildSetting(xcodeprojBuildSettings)
567             .addAllBuildSetting(IosSdkCommands.defaultWarningsForXcode())
568             .addAllSdkFramework(SdkFramework.names(objcProvider.get(SDK_FRAMEWORK)))
569             .addAllFramework(PathFragment.safePathStrings(objcProvider.get(FRAMEWORK_DIR)))
570             .addAllXcassetsDir(PathFragment.safePathStrings(objcProvider.get(XCASSETS_DIR)))
571             .addAllXcdatamodel(PathFragment.safePathStrings(
572                 Xcdatamodels.datamodelDirs(objcProvider.get(XCDATAMODEL))))
573             .addAllBundleImport(PathFragment.safePathStrings(objcProvider.get(BUNDLE_IMPORT_DIR)))
574             .addAllSdkDylib(objcProvider.get(SDK_DYLIB))
575             .addAllGeneralResourceFile(
576                 Artifact.toExecPaths(objcProvider.get(GENERAL_RESOURCE_FILE)))
577             .addAllGeneralResourceFile(
578                 PathFragment.safePathStrings(objcProvider.get(GENERAL_RESOURCE_DIR)));
579 
580     if (CAN_LINK_PRODUCT_TYPES.contains(productType)) {
581       // For builds with --ios_multi_cpus set, we may have several copies of some XCodeProviders
582       // in the dependencies (one per cpu architecture). We deduplicate the corresponding
583       // xcode target names with a LinkedHashSet before adding to the TargetControl.
584       Set<DependencyControl> dependencySet = new LinkedHashSet<>();
585       for (XcodeProvider dependency : propagatedDependencies) {
586         // Only add a library target to a binary's dependencies if it has source files to compile
587         // and it is not from the "non_propagated_deps" attribute. Xcode cannot build targets
588         // without a source file in the PBXSourceFilesBuildPhase, so if such a target is present in
589         // the control file, it is only to get Xcodegen to put headers and resources not used by the
590         // final binary in the Project Navigator.
591         //
592         // The exceptions to this rule are objc_bundle_library and ios_extension targets. Bundles
593         // are generally used for resources and can lack a PBXSourceFilesBuildPhase in the project
594         // file and still be considered valid by Xcode.
595         //
596         // ios_extension targets are an exception because they have no CompilationArtifact object
597         // but do have a dummy source file to make Xcode happy.
598         boolean hasSources = dependency.compilationArtifacts.isPresent()
599             && dependency.compilationArtifacts.get().getArchive().isPresent();
600         if (hasSources || (dependency.productType == XcodeProductType.BUNDLE)) {
601           String dependencyXcodeTargetName = dependency.dependencyXcodeTargetName();
602           dependencySet.add(DependencyControl.newBuilder()
603                 .setTargetLabel(dependencyXcodeTargetName)
604                 .build());
605         }
606       }
607 
608       for (DependencyControl dependencyControl : dependencySet) {
609         targetControl.addDependency(dependencyControl);
610       }
611     }
612     for (XcodeProvider justTestHost : testHost.asSet()) {
613       targetControl.addDependency(DependencyControl.newBuilder()
614           .setTargetLabel(xcodeTargetName(justTestHost.label))
615           .setTestHost(true)
616           .build());
617     }
618     for (XcodeProvider extension : extensions) {
619       targetControl.addDependency(DependencyControl.newBuilder()
620           .setTargetLabel(xcodeTargetName(extension.label))
621           .build());
622     }
623 
624     if (bundleInfoplist.isPresent()) {
625       targetControl.setInfoplist(bundleInfoplist.get().getExecPathString());
626     }
627     for (CompilationArtifacts artifacts : compilationArtifacts.asSet()) {
628       targetControl
629           .addAllSourceFile(Artifact.toExecPaths(artifacts.getSrcs()))
630           .addAllSupportFile(Artifact.toExecPaths(artifacts.getAdditionalHdrs()))
631           .addAllSupportFile(Artifact.toExecPaths(artifacts.getPrivateHdrs()))
632           .addAllNonArcSourceFile(Artifact.toExecPaths(artifacts.getNonArcSrcs()));
633 
634       for (Artifact pchFile : artifacts.getPchFile().asSet()) {
635         targetControl
636             .setPchPath(pchFile.getExecPathString())
637             .addSupportFile(pchFile.getExecPathString());
638       }
639     }
640 
641     for (Artifact artifact : additionalSources) {
642       targetControl.addSourceFile(artifact.getExecPathString());
643     }
644 
645     if (objcProvider.is(Flag.USES_CPP)) {
646       targetControl.addSdkDylib("libc++");
647     }
648 
649     return targetControl.build();
650   }

652   private TargetControl companionLibTargetControl(TargetControl mainTargetControl) {
653     return TargetControl.newBuilder()
654         .mergeFrom(mainTargetControl)
655         .setName(label.getName() + COMPANION_LIB_TARGET_LABEL_SUFFIX)
656         .setLabel(xcodeCompanionLibTargetName(label))
657         .setProductType(LIBRARY_STATIC.getIdentifier())
658         .clearInfoplist()
659         .clearDependency()
660         .clearBuildSetting()
661         .addAllBuildSetting(companionTargetXcodeprojBuildSettings)
662         .addAllBuildSetting(IosSdkCommands.defaultWarningsForXcode())
663         .build();
664   }

666   /**
667    * Prepends the given path to each path in {@code paths}. Empty paths are
668    * transformed to the value of {@code variable} rather than {@code variable + "/."}. Absolute
669    * paths are returned without modifications.
670    */
671   @VisibleForTesting
672   static Iterable<String> rootEach(final String prefix, Iterable<PathFragment> paths) {
673     Preconditions.checkArgument(prefix.startsWith("$"),
674         "prefix should start with a build setting variable like '$(NAME)': %s", prefix);
675     Preconditions.checkArgument(!prefix.endsWith("/"),
676         "prefix should not end with '/': %s", prefix);
677     return Iterables.transform(paths, new Function<PathFragment, String>() {
678       @Override
679       public String apply(PathFragment input) {
680         if (input.getSafePathString().equals(".")) {
681           return prefix;
682         } else if (input.isAbsolute()) {
683           return input.getSafePathString();
684         } else {
685           return prefix + "/" + input.getSafePathString();
686         }
687       }
688     });
