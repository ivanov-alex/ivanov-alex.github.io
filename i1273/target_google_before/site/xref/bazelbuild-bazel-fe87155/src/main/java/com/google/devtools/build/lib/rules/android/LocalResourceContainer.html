31  /**
32   * The collected resources and assets artifacts and roots.
33   *
34   * <p>This is used to encapsulate the logic and the data associated with the resources and assets
35   * derived from an appropriate android rule in a reusable instance.
36   */
37  @Immutable
38  public final class LocalResourceContainer {
39    public static final String[] RESOURCES_ATTRIBUTES = new String[] {
40        "manifest",
41        "resource_files",
42        "assets",
43        "assets_dir",
44        "inline_constants",
45        "exports_manifest",
46        "application_id",
47        "version_name",
48        "version_code"
49    };
50  
51    /**
52     * Determines if the attributes contain resource and asset attributes.
53     */
54    public static boolean definesAndroidResources(AttributeMap attributes) {
55      for (String attribute : RESOURCES_ATTRIBUTES) {
56        if (attributes.isAttributeValueExplicitlySpecified(attribute)) {
57          return true;
58        }
59      }
60      return false;
61    }
62  
63    /**
64     * Checks validity of a RuleContext to produce an AndroidData.
65     */
66    public static boolean validateRuleContext(RuleContext ruleContext) {
67      boolean valid = validateAssetsAndAssetsDir(ruleContext);
68      valid = valid && validateNoResourcesAttribute(ruleContext);
69      valid = valid && validateNoAndroidResourcesInSources(ruleContext);
70      valid = valid && validateManifest(ruleContext);
71      return valid;
72    }
73  
74    private static boolean validateAssetsAndAssetsDir(RuleContext ruleContext) {
75      if (ruleContext.attributes().isAttributeValueExplicitlySpecified("assets")
76          ^ ruleContext.attributes().isAttributeValueExplicitlySpecified("assets_dir")) {
77        ruleContext.ruleError(
78            "'assets' and 'assets_dir' should be either both empty or both non-empty");
79        return false;
80      }
81      return true;
82    }
83  
84    /**
85     * Validates that there are no resources defined if there are resource attribute defined.
86     */
87    private static boolean validateNoResourcesAttribute(RuleContext ruleContext) {
88      if (ruleContext.attributes().isAttributeValueExplicitlySpecified("resources")) {
89        ruleContext.attributeError("resources",
90            String.format("resources cannot be set when any of %s are defined.",
91                Joiner.on(", ").join(RESOURCES_ATTRIBUTES)));
92        return false;
93      }
94      return true;
95    }
96  
97    /**
98     * Validates that there are no android_resources srcjars in the srcs, as android_resource rules
99     * should not be used with the Android data logic.
100    */
101   private static boolean validateNoAndroidResourcesInSources(RuleContext ruleContext) {
102     Iterable<AndroidResourcesProvider> resources =
103         ruleContext.getPrerequisites("srcs", Mode.TARGET, AndroidResourcesProvider.class);
104     for (AndroidResourcesProvider provider : resources) {
105       ruleContext.attributeError("srcs",
106           String.format("srcs should not contain android_resource label %s", provider.getLabel()));
107       return false;
108     }
109     return true;
110   }
111 
112   private static boolean validateManifest(RuleContext ruleContext) {
113     if (ruleContext.getPrerequisiteArtifact("manifest", Mode.TARGET) == null) {
114       ruleContext.attributeError("manifest",
115           "manifest is required when resource_files or assets are defined.");
116       return false;
117     }
118     return true;
119   }

121   public static class Builder {
122     public static final class InvalidAssetPath extends RuntimeException {
123       InvalidAssetPath(String message) {
124         super(message);
125       }
126     }
127 
128     public static final class InvalidResourcePath extends RuntimeException {
129       InvalidResourcePath(String message) {
130         super(message);
131       }
132     }
133 
134     /**
135      * Set of allowable android directories prefixes.
136      */
137     // TODO(bazel-team): Pull from AOSP constant?
138     public static final ImmutableSet<String> RESOURCE_DIRECTORY_TYPES = ImmutableSet.of(
139         "animator",
140         "anim",
141         "color",
142         "drawable",
143         "interpolator",
144         "layout",
145         "menu",
146         "mipmap",
147         "raw",
148         "transition",
149         "values",
150         "xml");
151 
152     public static final String INCORRECT_RESOURCE_LAYOUT_MESSAGE = String.format(
153         "'%%s' is not in the expected resource directory structure of "
154         + "<resource directory>/{%s}/<file>", Joiner.on(',').join(RESOURCE_DIRECTORY_TYPES));
155 
156     private Collection<Artifact> assets = new LinkedHashSet<>();
157     private Collection<Artifact> resources = new LinkedHashSet<>();
158     private Collection<PathFragment> resourceRoots = new LinkedHashSet<>();
159     private Collection<PathFragment> assetRoots = new LinkedHashSet<>();
160 
161     /**
162      * Retrieves the asset {@link Artifact} and asset root {@link PathFragment}.
163      * @param assetsDir The common directory for the assets, used to get the directory roots and
164      *   verify the artifacts are located beneath the assetsDir
165      * @param targets {@link FileProvider}s for a given set of assets.
166      * @return The Builder.
167      * @throws InvalidAssetPath when a path does not reside under the correct directory.
168      */
169     public LocalResourceContainer.Builder withAssets(
170         PathFragment assetsDir, Iterable<FileProvider> targets) {
171       for (FileProvider target : targets) {
172         for (Artifact file : target.getFilesToBuild()) {
173           PathFragment packageFragment = file.getArtifactOwner().getLabel().getPackageFragment();
174           PathFragment packageRelativePath =
175               file.getRootRelativePath().relativeTo(packageFragment);
176           if (packageRelativePath.startsWith(assetsDir)) {
177             PathFragment relativePath = packageRelativePath.relativeTo(assetsDir);
178             assetRoots.add(trimTail(file.getExecPath(), relativePath));
179           } else {
180             throw new InvalidAssetPath(String.format(
181                 "'%s' (generated by '%s') is not beneath '%s'",
182                 file.getRootRelativePath(), target.getLabel(), assetsDir));
183           }
184           assets.add(file);
185         }
186       }
187       return this;
188     }
189 
190     /**
191      * Retrieves the resource {@link Artifact} and resource root {@link PathFragment}.
192      * @param targets {@link FileProvider}s for a given set of resource.
193      * @return The Builder.
194      * @throws InvalidResourcePath when a path does not reside under the correct directory.
195      */
196     public LocalResourceContainer.Builder withResources(Iterable<FileProvider> targets) {
197       PathFragment lastResourceDir = null;
198       Artifact lastFile = null;
199       for (FileProvider target : targets) {
200         for (Artifact file : target.getFilesToBuild()) {
201           PathFragment packageFragment = file.getArtifactOwner().getLabel().getPackageFragment();
202           PathFragment packageRelativePath =
203               file.getRootRelativePath().relativeTo(packageFragment);
204           PathFragment resourceDir = findResourceDir(file);
205           if (lastResourceDir == null || resourceDir.equals(lastResourceDir)) {
206             resourceRoots.add(
207                 trimTail(file.getExecPath(), makeRelativeTo(resourceDir, packageRelativePath)));
208           } else {
209             throw new InvalidResourcePath(String.format(
210                 "'%s' (generated by '%s') is not in the same directory '%s' (derived from %s)."
211                 + " All resources must share a common directory.",
212                 file.getRootRelativePath(), file.getArtifactOwner().getLabel(), lastResourceDir,
213                 lastFile.getRootRelativePath()));
214           }
215           resources.add(file);
216           lastFile = file;
217           lastResourceDir = resourceDir;
218         }
219       }
220       return this;
221     }
222 
223     /**
224      * Finds and validates the resource directory PathFragment from the artifact Path.
225      *
226      * <p>If the artifact is not a Fileset, the resource directory is presumed to be the second
227      * directory from the end. Filesets are expect to have the last directory as the resource
228      * directory.
229      *
230      */
231     public static PathFragment findResourceDir(Artifact artifact) {
232       PathFragment fragment = artifact.getPath().asFragment();
233       int segmentCount = fragment.segmentCount();
234       if (segmentCount < 3) {
235         throw new InvalidResourcePath(String.format(
236             INCORRECT_RESOURCE_LAYOUT_MESSAGE, artifact.getRootRelativePath()));
237       }
238       // TODO(bazel-team): Expand Fileset to verify, or remove Fileset as an option for resources.
239       if (artifact.isFileset()) {
240         return fragment.subFragment(segmentCount - 1, segmentCount);
241       }
242 
243       // Check the resource folder type layout.
244       // get the prefix of the parent folder of the fragment.
245       String parentDirectory = fragment.getSegment(segmentCount - 2);
246       int dashIndex = parentDirectory.indexOf('-');
247       String androidFolder =
248           dashIndex == -1 ? parentDirectory : parentDirectory.substring(0, dashIndex);
249       if (!RESOURCE_DIRECTORY_TYPES.contains(androidFolder)) {
250         throw new InvalidResourcePath(String.format(
251             INCORRECT_RESOURCE_LAYOUT_MESSAGE, artifact.getRootRelativePath()));
252       }
253 
254       return fragment.subFragment(segmentCount - 3, segmentCount - 2);
255     }
256 
257     /**
258      * Returns the root-part of a given path by trimming off the end specified by
259      * a given tail. Assumes that the tail is known to match, and simply relies on
260      * the segment lengths.
261      */
262     private static PathFragment trimTail(PathFragment path, PathFragment tail) {
263       return path.subFragment(0, path.segmentCount() - tail.segmentCount());
264     }
265 
266     private static PathFragment makeRelativeTo(PathFragment ancestor, PathFragment path) {
267       String cutAtSegment = ancestor.getSegment(ancestor.segmentCount() - 1);
268       int totalPathSegments = path.segmentCount() - 1;
269       for (int i = totalPathSegments; i >= 0; i--) {
270         if (path.getSegment(i).equals(cutAtSegment)) {
271           return path.subFragment(i, totalPathSegments);
272         }
273       }
274       throw new IllegalArgumentException("PathFragment " + path
275           + " is not beneath " + ancestor);
276     }

278     public LocalResourceContainer build() {
279       return new LocalResourceContainer(
280           ImmutableList.copyOf(resources),
281           ImmutableList.copyOf(resourceRoots),
282           ImmutableList.copyOf(assets),
283           ImmutableList.copyOf(assetRoots));
284     }
285   }

287   private final ImmutableList<Artifact> resources;
288   private final ImmutableList<Artifact> assets;
289   private final ImmutableList<PathFragment> assetRoots;
290   private final ImmutableList<PathFragment> resourceRoots;

292   @VisibleForTesting
293   public LocalResourceContainer(
294       ImmutableList<Artifact> resources,
295       ImmutableList<PathFragment> resourceRoots,
296       ImmutableList<Artifact> assets,
297       ImmutableList<PathFragment> assetRoots) {
298         this.resources = resources;
299         this.resourceRoots = resourceRoots;
300         this.assets = assets;
301         this.assetRoots = assetRoots;
302   }
