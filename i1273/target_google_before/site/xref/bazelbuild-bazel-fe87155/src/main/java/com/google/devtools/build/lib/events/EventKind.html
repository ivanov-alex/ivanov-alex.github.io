14  package com.google.devtools.build.lib.events;
15  
16  import java.util.EnumSet;
17  import java.util.Set;
18  
19  /**
20   * Indicates the kind of an {@link Event}.
21   */
22  public enum EventKind {
23  
24    /**
25     * For errors that will prevent a successful, correct build.  In general, the
26     * build tool will not attempt to start or continue a build if an error is
27     * encountered (though the behaviour specified by --keep-going flag is a
28     * counterexample).
29     *
30     * Errors of a more severe nature in the input, such as those which might
31     * cause later passes of the analysis to fail catastrophically, should be
32     * handled by throwing an exception.
33     */
34    ERROR,
35  
36    /**
37     * For warnings of minor problems that do not affect the integrity of a
38     * build.
39     */
40    WARNING,
41  
42    /**
43     * For atemporal information that is true throughout the entire duration
44     * of a build. (e.g. the number of targets found)
45     */
46    INFO,
47  
48    /**
49     * For temporal information that changes during the duration of a build.
50     * (e.g. what action is executing now)
51     */
52    PROGRESS,
53  
54    /**
55     * For progress messages (temporal information) relating to the start
56     * and end of particular tasks.
57     * (e.g. "Loading package foo", "Compiling bar", etc.)
58     */
59    START,
60    FINISH,
61  
62    /**
63     * For command lines of subcommands executed by the build tool (like make-dbg
64     * "-v").
65     */
66    SUBCOMMAND,
67  
68    /**
69     * Output to stdout/stderr from subprocesses.
70     */
71    STDOUT,
72    STDERR,
73  
74    /**
75     * Test result messages (similar to the INFO and ERROR, but test-specific).
76     */
77    PASS,
78    FAIL,
79    TIMEOUT,
80  
81    /**
82     * For the reasoning of the dependency checker (like GNU Make "-d").
83     */
84    DEPCHECKER;
85  
86    // Convenient predefined EnumSets.  Clients should not mutate them!
87  
88    public static final Set<EventKind> ALL_EVENTS =
89        EnumSet.allOf(EventKind.class);
90  
91    public static final Set<EventKind> OUTPUT = EnumSet.of(
92        EventKind.STDOUT,
93        EventKind.STDERR
94        );
95  
96    public static final Set<EventKind> ERRORS = EnumSet.of(
97        EventKind.ERROR,
98        EventKind.FAIL,
99        EventKind.TIMEOUT
100       );
101 
102   public static final Set<EventKind> ERRORS_AND_WARNINGS = EnumSet.of(
103       EventKind.ERROR,
104       EventKind.WARNING,
105       EventKind.FAIL,
106       EventKind.TIMEOUT
107       );
108 
109   public static final Set<EventKind> ERRORS_WARNINGS_AND_INFO = EnumSet.of(
110       EventKind.ERROR,
111       EventKind.WARNING,
112       EventKind.PASS,
113       EventKind.FAIL,
114       EventKind.TIMEOUT,
115       EventKind.INFO
116       );
117 
118   public static final Set<EventKind> ERRORS_AND_OUTPUT = EnumSet.of(
119       EventKind.ERROR,
120       EventKind.FAIL,
121       EventKind.TIMEOUT,
122       EventKind.STDOUT,
123       EventKind.STDERR
124       );
125 
126   public static final Set<EventKind> ERRORS_AND_WARNINGS_AND_OUTPUT = EnumSet.of(
127       EventKind.ERROR,
128       EventKind.WARNING,
129       EventKind.FAIL,
130       EventKind.TIMEOUT,
131       EventKind.STDOUT,
132       EventKind.STDERR
133       );
134 
135   public static final Set<EventKind> ERRORS_WARNINGS_AND_INFO_AND_OUTPUT = EnumSet.of(
136       EventKind.ERROR,
137       EventKind.WARNING,
138       EventKind.PASS,
139       EventKind.FAIL,
140       EventKind.TIMEOUT,
141       EventKind.INFO,
142       EventKind.STDOUT,
143       EventKind.STDERR
144       );
145 
146 }
