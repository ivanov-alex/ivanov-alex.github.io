131   /**
132    * Collects the transitive neverlink dependencies.
133    *
134    * @param ruleContext the context of the rule neverlink deps are to be computed for
135    * @param deps the targets to be treated as dependencies
136    * @param runtimeJars the runtime jars produced by the rule (non-transitive)
137    *
138    * @return a nested set of the neverlink deps.
139    */
140   public static NestedSet<Artifact> collectTransitiveNeverlinkLibraries(
141       RuleContext ruleContext, Iterable<? extends TransitiveInfoCollection> deps,
142       ImmutableList<Artifact> runtimeJars) {
143     NestedSetBuilder<Artifact> builder = NestedSetBuilder.naiveLinkOrder();
144 
145     for (AndroidNeverLinkLibrariesProvider provider : AnalysisUtils.getProviders(deps,
146         AndroidNeverLinkLibrariesProvider.class)) {
147       builder.addTransitive(provider.getTransitiveNeverLinkLibraries());
148     }
149 
150     if (JavaCommon.isNeverLink(ruleContext)) {
151       builder.addAll(runtimeJars);
152       for (JavaCompilationArgsProvider provider : AnalysisUtils.getProviders(
153           deps, JavaCompilationArgsProvider.class)) {
154         builder.addTransitive(provider.getRecursiveJavaCompilationArgs().getRuntimeJars());
155       }
156     }
157 
158     return builder.build();
159   }

161   /**
162    * Creates an action that converts {@code jarToDex} to a dex file. The output will be stored in
163    * the {@link com.google.devtools.build.lib.actions.Artifact} {@code dxJar}.
164    */
165   public static void createDexAction(
166       RuleContext ruleContext,
167       Artifact jarToDex, Artifact classesDex, List<String> dexOptions, boolean multidex,
168       Artifact mainDexList) {
169     List<String> args = new ArrayList<>();
170     args.add("--dex");
171     // Add --no-locals to coverage builds. Otherwise local variable debug information is not
172     // preserved, which leads to runtime errors.
173     if (ruleContext.getConfiguration().isCodeCoverageEnabled()) {
174       args.add("--no-locals");
175     }
176 
177     // Multithreaded dex does not work when using --multi-dex.
178     if (!multidex) {
179       // Multithreaded dex tends to run faster, but only up to about 5 threads (at which point the
180       // law of diminishing returns kicks in). This was determined experimentally, with 5-thread dex
181       // performing about 25% faster than 1-thread dex.
182       args.add("--num-threads=5");
183     }
184 
185     args.addAll(dexOptions);
186     if (multidex) {
187       args.add("--multi-dex");
188       if (mainDexList != null) {
189         args.add("--main-dex-list=" + mainDexList.getExecPathString());
190       }
191     }
192     args.add("--output=" + classesDex.getExecPathString());
193     args.add(jarToDex.getExecPathString());
194 
195     SpawnAction.Builder builder = new SpawnAction.Builder()
196         .setExecutable(AndroidSdkProvider.fromRuleContext(ruleContext).getDx())
197         .addInput(jarToDex)
198         .addOutput(classesDex)
199         .addArguments(args)
200         .setProgressMessage("Converting " + jarToDex.getExecPathString() + " to dex format")
201         .setMnemonic("AndroidDexer")
202         .setResources(ResourceSet.createWithRamCpuIo(4096.0, 5.0, 0.0));
203     if (mainDexList != null) {
204       builder.addInput(mainDexList);
205     }
206     ruleContext.registerAction(builder.build(ruleContext));
207   }

214   public static NestedSet<ResourceContainer> getTransitiveResourceContainers(
215       RuleContext ruleContext, boolean withDeps) {
216     // Traverse through all android_library targets looking for resources
217     NestedSetBuilder<ResourceContainer> resourcesBuilder = NestedSetBuilder.naiveLinkOrder();
218     List<String> attributes = new ArrayList<>();
219     attributes.add("resources");
220     if (withDeps) {
221       attributes.add("deps");
222     }
223 
224     for (String attribute : attributes) {
225       if (!ruleContext.attributes().has(attribute, Type.LABEL)
226           && !ruleContext.attributes().has(attribute, Type.LABEL_LIST)) {
227         continue;
228       }
229 
230       for (AndroidResourcesProvider resources :
231           ruleContext.getPrerequisites(attribute, Mode.TARGET, AndroidResourcesProvider.class)) {
232         resourcesBuilder.addTransitive(resources.getTransitiveAndroidResources());
233       }
234     }
235 
236     return resourcesBuilder.build();
237   }

239   private void compileResources(
240       JavaSemantics javaSemantics,
241       JavaCompilationArtifacts.Builder artifactsBuilder,
242       JavaTargetAttributes.Builder attributes,
243       NestedSet<ResourceContainer> resourceContainers,
244       ResourceContainer updatedResources) throws InterruptedException {
245       Artifact binaryResourcesJar =
246           ruleContext.getImplicitOutputArtifact(JavaSemantics.JAVA_BINARY_CLASS_JAR);
247       compileResourceJar(javaSemantics, binaryResourcesJar, updatedResources.getJavaSourceJar());
248       // combined resource constants needs to come even before own classes that may contain
249       // local resource constants
250       artifactsBuilder.addRuntimeJar(binaryResourcesJar);
251       // Repackages the R.java for each dependency package and places the resultant jars
252       // before the dependency libraries to ensure that the generated resource ids are
253       // correct.
254       createJarJarActions(attributes, resourceContainers,
255           updatedResources.getJavaPackage(),
256           binaryResourcesJar);
257   }

259   private void compileResourceJar(
260       JavaSemantics javaSemantics, Artifact binaryResourcesJar, Artifact javaSourceJar) {
261       JavaCompilationArtifacts.Builder javaArtifactsBuilder =
262           new JavaCompilationArtifacts.Builder();
263       JavaTargetAttributes.Builder javacJarAttributes =
264           new JavaTargetAttributes.Builder(javaSemantics);
265       javacJarAttributes.addSourceJar(javaSourceJar);
266       JavaCompilationHelper javacHelper = new JavaCompilationHelper(
267           ruleContext, javaSemantics, getJavacOpts(), javacJarAttributes);
268       Artifact outputDepsProto =
269           javacHelper.createOutputDepsProtoArtifact(binaryResourcesJar, javaArtifactsBuilder);
270 
271       javacHelper.createCompileActionWithInstrumentation(
272           binaryResourcesJar,
273           null /* manifestProtoOutput */,
274           null /* gensrcJar */,
275           outputDepsProto,
276           javaArtifactsBuilder);
277   }

279   private void createJarJarActions(
280       JavaTargetAttributes.Builder attributes,
281       Iterable<ResourceContainer> resourceContainers,
282       String originalPackage,
283       Artifact binaryResourcesJar) {
284     // Now use jarjar for the rest of the resources. We need to make a copy
285     // of the final generated resources for each of the targets included in
286     // the transitive closure of this binary.
287     for (ResourceContainer otherContainer : resourceContainers) {
288       if (otherContainer.getLabel().equals(ruleContext.getLabel())) {
289         continue;
290       }
291 
292       Artifact resourcesJar = createResourceJarArtifact(ruleContext, otherContainer, ".jar");
293       // combined resource constants copy needs to come before library classes that may contain
294       // their local resource constants
295       attributes.addRuntimeClassPathEntry(resourcesJar);
296 
297       Artifact jarJarRuleFile = createResourceJarArtifact(
298           ruleContext, otherContainer, ".jar_jarjar_rules.txt");
299 
300       String jarJarRule = String.format("rule %s.* %s.@1",
301           originalPackage, otherContainer.getJavaPackage());
302       ruleContext.registerAction(new FileWriteAction(
303           ruleContext.getActionOwner(), jarJarRuleFile, jarJarRule, false));
304 
305       FilesToRunProvider jarjar =
306           ruleContext.getExecutablePrerequisite("$jarjar_bin", Mode.HOST);
307 
308       ruleContext.registerAction(new SpawnAction.Builder()
309           .setExecutable(jarjar)
310           .addArgument("process")
311           .addInputArgument(jarJarRuleFile)
312           .addInputArgument(binaryResourcesJar)
313           .addOutputArgument(resourcesJar)
314           .setProgressMessage("Repackaging jar")
315           .setMnemonic("AndroidRepackageJar")
316           .build(ruleContext));
317     }
318   }

320   private static Artifact createResourceJarArtifact(RuleContext ruleContext,
321       ResourceContainer container, String fileNameSuffix) {
322 
323     String artifactName = container.getLabel().getName() + fileNameSuffix;
324 
325     // Since the Java sources are generated by combining all resources with the
326     // ones included in the binary, the path of the artifact has to be unique
327     // per binary and per library (not only per library).
328     Artifact artifact = ruleContext.getUniqueDirectoryArtifact("resource_jars",
329         container.getLabel().getPackageIdentifier().getPathFragment().getRelative(artifactName),
330         ruleContext.getBinOrGenfilesDirectory());
331     return artifact;
332   }

334   public JavaTargetAttributes init(
335       JavaSemantics javaSemantics, AndroidSemantics androidSemantics,
336       ResourceApk resourceApk, AndroidIdlProvider transitiveIdlImportData,
337       boolean addCoverageSupport, boolean collectJavaCompilationArgs,
338       SafeImplicitOutputsFunction genClassJarImplicitOutput) throws InterruptedException {
339     ImmutableList<Artifact> extraSources =
340         resourceApk.isLegacy() || resourceApk.getResourceJavaSrcJar() == null
341             ? ImmutableList.<Artifact>of()
342             : ImmutableList.of(resourceApk.getResourceJavaSrcJar());
343     JavaTargetAttributes.Builder attributes = init(
344         androidSemantics,
345         transitiveIdlImportData,
346         resourceApk.getTransitiveResources(),
347         extraSources);
348     JavaCompilationArtifacts.Builder artifactsBuilder = new JavaCompilationArtifacts.Builder();
349     if (resourceApk.isLegacy()) {
350       compileResources(javaSemantics, artifactsBuilder, attributes,
351           resourceApk.getTransitiveResources(), resourceApk.getPrimaryResource());
352     }
353 
354     JavaCompilationHelper helper = initAttributes(attributes, javaSemantics);
355     if (ruleContext.hasErrors()) {
356       return null;
357     }
358 
359     if (addCoverageSupport) {
360       androidSemantics.addCoverageSupport(ruleContext, this, javaSemantics, true,
361           attributes, artifactsBuilder);
362       if (ruleContext.hasErrors()) {
363         return null;
364       }
365     }
366 
367     jackCompilationHelper = initJack(helper.getAttributes(), javaSemantics);
368     if (ruleContext.hasErrors()) {
369       return null;
370     }
371 
372     initJava(
373         helper, artifactsBuilder, collectJavaCompilationArgs, resourceApk.getResourceJavaSrcJar(),
374         genClassJarImplicitOutput);
375     if (ruleContext.hasErrors()) {
376       return null;
377     }
378     return helper.getAttributes();
379   }

381   private JavaTargetAttributes.Builder init(
382       AndroidSemantics androidSemantics,
383       AndroidIdlProvider transitiveIdlImportData,
384       NestedSet<AndroidResourcesProvider.ResourceContainer> transitiveResources,
385       Collection<Artifact> extraArtifacts) {
386     this.transitiveIdlImportData = transitiveIdlImportData;
387     this.transitiveResources = transitiveResources;
388 
389     ImmutableList.Builder<Artifact> extraSrcsBuilder =
390         new ImmutableList.Builder<Artifact>().addAll(extraArtifacts);
391 
392     idls = getIdlSrcs(ruleContext);
393     if (!idls.isEmpty() && !ruleContext.hasErrors()) {
394       translatedIdlSources = generateTranslatedIdlArtifacts(ruleContext, idls);
395     }
396 
397     javaCommon.initializeJavacOpts(androidSemantics.getJavacArguments());
398     JavaTargetAttributes.Builder attributes = javaCommon.initCommon(
399         extraSrcsBuilder.addAll(translatedIdlSources.values()).build());
400 
401     attributes.setBootClassPath(ImmutableList.of(
402         AndroidSdkProvider.fromRuleContext(ruleContext).getAndroidJar()));
403     return attributes;
404   }

406   private JavaCompilationHelper initAttributes(
407       JavaTargetAttributes.Builder attributes, JavaSemantics semantics) {
408     JavaCompilationHelper helper = new JavaCompilationHelper(
409         ruleContext, semantics, javaCommon.getJavacOpts(), attributes);
410     Iterable<? extends TransitiveInfoCollection> deps =
411         javaCommon.targetsTreatedAsDeps(ClasspathType.BOTH);
412     helper.addLibrariesToAttributes(deps);
413     helper.addProvidersToAttributes(javaCommon.compilationArgsFromSources(), asNeverLink);
414     attributes.setStrictJavaDeps(getStrictAndroidDeps());
415     attributes.setRuleKind(ruleContext.getRule().getRuleClass());
416     attributes.setTargetLabel(ruleContext.getLabel());
417 
418     JavaCommon.validateConstraint(ruleContext, "android", deps);
419     ruleContext.checkSrcsSamePackage(true);
420     return helper;
421   }

423   private StrictDepsMode getStrictAndroidDeps() {
424     // Get command line strict_android_deps option
425     StrictDepsMode strict = ruleContext.getFragment(AndroidConfiguration.class).getStrictDeps();
426     // Use option if anything but DEFAULT, which is now equivalent to ERROR.
427     return (strict != DEFAULT && strict != STRICT) ? strict : ERROR;
428   }

430   JackCompilationHelper initJack(JavaTargetAttributes attributes, JavaSemantics javaSemantics)
431       throws InterruptedException {
432     Map<PathFragment, Artifact> resourcesMap = new LinkedHashMap<>();
433     for (Artifact resource : attributes.getResources()) {
434       resourcesMap.put(javaSemantics.getJavaResourcePath(resource.getRootRelativePath()), resource);
435     }
436     AndroidSdkProvider sdk = AndroidSdkProvider.fromRuleContext(ruleContext);
437     return new JackCompilationHelper.Builder()
438         // blaze infrastructure
439         .setRuleContext(ruleContext)
440         // configuration
441         .setOutputArtifact(
442             ruleContext.getImplicitOutputArtifact(AndroidRuleClasses.ANDROID_LIBRARY_JACK_FILE))
443         // tools
444         .setAndroidSdk(sdk)
445         // sources
446         .addJavaSources(attributes.getSourceFiles())
447         .addSourceJars(attributes.getSourceJars())
448         .addCompiledJars(attributes.getJarFiles())
449         .addResources(ImmutableMap.copyOf(resourcesMap))
450         .addProcessorNames(attributes.getProcessorNames())
451         .addProcessorClasspathJars(attributes.getProcessorPath())
452         .addExports(JavaCommon.getExports(ruleContext))
453         .addClasspathDeps(javaCommon.targetsTreatedAsDeps(ClasspathType.COMPILE_ONLY))
454         .addRuntimeDeps(javaCommon.targetsTreatedAsDeps(ClasspathType.RUNTIME_ONLY))
455         .build();
456   }

458   private void initJava(
459       JavaCompilationHelper helper,
460       JavaCompilationArtifacts.Builder javaArtifactsBuilder,
461       boolean collectJavaCompilationArgs,
462       @Nullable Artifact additionalSourceJar,
463       SafeImplicitOutputsFunction genClassJarImplicitOutput) throws InterruptedException {
464     NestedSetBuilder<Artifact> filesBuilder = NestedSetBuilder.<Artifact>stableOrder();
465     if (additionalSourceJar != null) {
466       filesBuilder.add(additionalSourceJar);
467     }
468 
469     JavaTargetAttributes attributes = helper.getAttributes();
470     if (ruleContext.hasErrors()) {
471       // Avoid leaving filesToBuild set to null, otherwise we'll get a NullPointerException masking
472       // the real error.
473       filesToBuild = filesBuilder.build();
474       return;
475     }
476 
477     if (attributes.hasJarFiles()) {
478       // This rule is repackaging some source jars as a java library
479 
480       javaArtifactsBuilder.addRuntimeJars(attributes.getJarFiles());
481       javaArtifactsBuilder.addCompileTimeJars(attributes.getCompileTimeJarFiles());
482 
483       filesBuilder.addAll(attributes.getJarFiles());
484     }
485 
486     Artifact jar = null;
487     classJar = ruleContext.getImplicitOutputArtifact(AndroidRuleClasses.ANDROID_LIBRARY_CLASS_JAR);
488     if (attributes.hasSourceFiles() || attributes.hasSourceJars() || attributes.hasResources()) {
489       // We only want to add a jar to the classpath of a dependent rule if it has content.
490       javaArtifactsBuilder.addRuntimeJar(classJar);
491       jar = classJar;
492     }
493 
494     filesBuilder.add(classJar);
495 
496     // The gensrc jar is created only if the target uses annotation processing. Otherwise,
497     // it is null, and the source jar action will not depend on the compile action.
498     gensrcJar = helper.createGensrcJar(classJar);
499     Artifact manifestProtoOutput = helper.createManifestProtoOutput(classJar);
500 
501     // AndroidBinary will pass its -gen.jar output, and AndroidLibrary will pass its own.
502     genJar = ruleContext.getImplicitOutputArtifact(genClassJarImplicitOutput);
503     helper.createGenJarAction(classJar, manifestProtoOutput, genJar);
504 
505     srcJar = ruleContext.getImplicitOutputArtifact(AndroidRuleClasses.ANDROID_LIBRARY_SOURCE_JAR);
506     helper.createSourceJarAction(srcJar, gensrcJar);
507 
508     NestedSetBuilder<Artifact> compileTimeDependenciesBuilder = NestedSetBuilder.stableOrder();
509     Artifact outputDepsProto = helper.createOutputDepsProtoArtifact(classJar, javaArtifactsBuilder);
510     if (outputDepsProto != null) {
511       compileTimeDependenciesBuilder.add(outputDepsProto);
512     }
513     helper.createCompileActionWithInstrumentation(classJar, manifestProtoOutput, gensrcJar,
514         outputDepsProto, javaArtifactsBuilder);
515 
516     compileTimeDependencyArtifacts = compileTimeDependenciesBuilder.build();
517     filesToBuild = filesBuilder.build();
518 
519     if ((attributes.hasSourceFiles() || attributes.hasSourceJars()) && jar != null) {
520       helper.createCompileTimeJarAction(jar, outputDepsProto, javaArtifactsBuilder);
521     }
522     javaCommon.setJavaCompilationArtifacts(javaArtifactsBuilder.build());
523 
524     javaCommon.setClassPathFragment(
525         new ClasspathConfiguredFragment(
526             javaCommon.getJavaCompilationArtifacts(), attributes, asNeverLink));
527 
528     transitiveNeverlinkLibraries = collectTransitiveNeverlinkLibraries(
529         ruleContext,
530         javaCommon.getDependencies(),
531         javaCommon.getJavaCompilationArtifacts().getRuntimeJars());
532     topLevelSourceJars = ImmutableList.of(srcJar);
533     transitiveSourceJars = javaCommon.collectTransitiveSourceJars(srcJar);
534 
535     if (collectJavaCompilationArgs) {
536       boolean includeSrcjars =
537           ruleContext.getFragment(AndroidConfiguration.class).treatSrcjarsAsSrcsForStrictDeps();
538       boolean hasSources =
539           attributes.hasSourceFiles() || (includeSrcjars && attributes.hasSourceJars());
540       this.javaCompilationArgs =
541           collectJavaCompilationArgs(ruleContext, exportDeps, asNeverLink, hasSources);
542       this.recursiveJavaCompilationArgs = collectJavaCompilationArgs(
543           ruleContext, true, asNeverLink, /* hasSources */ true);
544     }
545   }

547   public RuleConfiguredTargetBuilder addTransitiveInfoProviders(
548       RuleConfiguredTargetBuilder builder) {
549     if (!idls.isEmpty()) {
550       generateAndroidIdlActions(
551           ruleContext, idls, transitiveIdlImportData, translatedIdlSources);
552     }
553 
554     Runfiles runfiles = new Runfiles.Builder(ruleContext.getWorkspaceName())
555         .addRunfiles(ruleContext, RunfilesProvider.DEFAULT_RUNFILES)
556         .build();
557 
558     javaCommon.addTransitiveInfoProviders(builder, filesToBuild, classJar);
559 
560     return builder
561         .setFilesToBuild(filesToBuild)
562         .add(
563             JavaRuntimeJarProvider.class,
564             new JavaRuntimeJarProvider(javaCommon.getJavaCompilationArtifacts().getRuntimeJars()))
565         .add(RunfilesProvider.class, RunfilesProvider.simple(runfiles))
566         .add(
567             AndroidResourcesProvider.class,
568             new AndroidResourcesProvider(ruleContext.getLabel(), transitiveResources))
569         .add(AndroidIdlProvider.class, transitiveIdlImportData)
570         .add(
571             JavaCompilationArgsProvider.class,
572             new JavaCompilationArgsProvider(
573                 javaCompilationArgs,
574                 recursiveJavaCompilationArgs,
575                 compileTimeDependencyArtifacts,
576                 NestedSetBuilder.<Artifact>emptySet(Order.STABLE_ORDER)))
577         .add(
578             JackLibraryProvider.class,
579             asNeverLink
580                 ? jackCompilationHelper.compileAsNeverlinkLibrary()
581                 : jackCompilationHelper.compileAsLibrary())
582         .addOutputGroup(
583             OutputGroupProvider.HIDDEN_TOP_LEVEL, collectHiddenTopLevelArtifacts(ruleContext))
584         .addOutputGroup(JavaSemantics.SOURCE_JARS_OUTPUT_GROUP, transitiveSourceJars)
585         .addOutputGroup(JavaSemantics.GENERATED_JARS_OUTPUT_GROUP, genJar);
586   }

610   public static void checkIdlSrcsSamePackage(RuleContext ruleContext) {
611     PathFragment packageName = ruleContext.getLabel().getPackageFragment();
612     Collection<Artifact> idls = ruleContext
613         .getPrerequisiteArtifacts("idl_srcs", Mode.TARGET)
614         .filter(AndroidRuleClasses.ANDROID_IDL)
615         .list();
616     for (Artifact idl : idls) {
617       Label idlLabel = idl.getOwner();
618       if (!packageName.equals(idlLabel.getPackageFragment())) {
619         ruleContext.attributeError("idl_srcs", "do not import '" + idlLabel + "' directly. "
620             + "You should either move the file to this package or depend on "
621             + "an appropriate rule there");
622       }
623     }

626   public static NestedSet<LinkerInput> collectTransitiveNativeLibraries(
627       Iterable<? extends TransitiveInfoCollection> deps) {
628     NestedSetBuilder<LinkerInput> builder = NestedSetBuilder.stableOrder();
629     for (TransitiveInfoCollection dep : deps) {
630       AndroidNativeLibraryProvider android = dep.getProvider(AndroidNativeLibraryProvider.class);
631       if (android != null) {
632         builder.addTransitive(android.getTransitiveAndroidNativeLibraries());
633         continue;
634       }
635 
636       JavaNativeLibraryProvider java = dep.getProvider(JavaNativeLibraryProvider.class);
637       if (java != null) {
638         builder.addTransitive(java.getTransitiveJavaNativeLibraries());
639         continue;
640       }
641 
642       CcNativeLibraryProvider cc = dep.getProvider(CcNativeLibraryProvider.class);
643       if (cc != null) {
644         for (LinkerInput input : cc.getTransitiveCcNativeLibraries()) {
645           Artifact library = input.getOriginalLibraryArtifact();
646           String name = library.getFilename();
647           if (CppFileTypes.SHARED_LIBRARY.matches(name)
648               || CppFileTypes.VERSIONED_SHARED_LIBRARY.matches(name)) {
649             builder.add(input);
650           }
651         }
652         continue;
653       }
654     }
655 
656     return builder.build();
657   }

680   private ImmutableMap<Artifact, Artifact> generateTranslatedIdlArtifacts(
681       RuleContext ruleContext, Collection<Artifact> idls) {
682     ImmutableMap.Builder<Artifact, Artifact> outputJavaSources = ImmutableMap.builder();
683     String ruleName = ruleContext.getRule().getName();
684     // for each aidl file use aggregated preprocessed files to generate Java code
685     for (Artifact idl : idls) {
686       // Reconstruct the package tree under <rule>_aidl to avoid a name conflict
687       // if the same AIDL files are used in multiple targets.
688       PathFragment javaOutputPath = FileSystemUtils.replaceExtension(
689           new PathFragment(ruleName + "_aidl").getRelative(idl.getRootRelativePath()),
690           ".java");
691       Artifact output = ruleContext.getPackageRelativeArtifact(
692           javaOutputPath, ruleContext.getConfiguration().getGenfilesDirectory());
693       outputJavaSources.put(idl, output);
694     }
695     return outputJavaSources.build();
696   }

709   private void generateAndroidIdlActions(RuleContext ruleContext,
710       Collection<Artifact> idls, AndroidIdlProvider transitiveIdlImportData,
711       Map<Artifact, Artifact> translatedIdlSources) {
712     AndroidSdkProvider sdk = AndroidSdkProvider.fromRuleContext(ruleContext);
713     Set<Artifact> preprocessedIdls = new LinkedHashSet<>();
714     List<String> preprocessedArgs = new ArrayList<>();
715 
716     // add imports
717     for (String idlImport : transitiveIdlImportData.getTransitiveIdlImportRoots()) {
718       preprocessedArgs.add("-I" + idlImport);
719     }
720 
721     // preprocess each aidl file
722     preprocessedArgs.add("-p" + sdk.getFrameworkAidl().getExecPathString());
723     String ruleName = ruleContext.getRule().getName();
724     for (Artifact idl : idls) {
725       // Reconstruct the package tree under <rule>_aidl to avoid a name conflict
726       // if the source AIDL files are also generated.
727       PathFragment preprocessedPath = new PathFragment(ruleName + "_aidl")
728           .getRelative(idl.getRootRelativePath());
729       Artifact preprocessed = ruleContext.getPackageRelativeArtifact(
730           preprocessedPath, ruleContext.getConfiguration().getGenfilesDirectory());
731       preprocessedIdls.add(preprocessed);
732       preprocessedArgs.add("-p" + preprocessed.getExecPathString());
733 
734       createAndroidIdlPreprocessAction(ruleContext, idl, preprocessed);
735     }
736 
737     // aggregate all preprocessed aidl files
738     MiddlemanFactory middlemanFactory = ruleContext.getAnalysisEnvironment().getMiddlemanFactory();
739     Artifact preprocessedIdlsMiddleman = middlemanFactory.createAggregatingMiddleman(
740         ruleContext.getActionOwner(), "AndroidIDLMiddleman", preprocessedIdls,
741         ruleContext.getConfiguration().getMiddlemanDirectory());
742 
743     for (Artifact idl : translatedIdlSources.keySet()) {
744       createAndroidIdlAction(ruleContext, idl,
745           transitiveIdlImportData.getTransitiveIdlImports(),
746           preprocessedIdlsMiddleman, translatedIdlSources.get(idl), preprocessedArgs);
747     }
748   }

750   private void createAndroidIdlPreprocessAction(RuleContext ruleContext,
751       Artifact idl, Artifact preprocessed) {
752     AndroidSdkProvider sdk = AndroidSdkProvider.fromRuleContext(ruleContext);
753     ruleContext.registerAction(new SpawnAction.Builder()
754         .setExecutable(sdk.getAidl())
755         // Note the below may be an overapproximation of the actual runfiles, due to "conditional
756         // artifacts" (see Runfiles.PruningManifest).
757         // TODO(bazel-team): When using getFilesToRun(), the middleman is
758         // not expanded. Fix by providing code to expand and use getFilesToRun here.
759         .addInput(idl)
760         .addOutput(preprocessed)
761         .addArgument("--preprocess")
762         .addArgument(preprocessed.getExecPathString())
763         .addArgument(idl.getExecPathString())
764         .setProgressMessage("Android IDL preprocessing")
765         .setMnemonic("AndroidIDLPreprocess")
766         .build(ruleContext));
767   }

769   private void createAndroidIdlAction(RuleContext ruleContext,
770       Artifact idl, Iterable<Artifact> idlImports, Artifact preprocessedIdls,
771       Artifact output, List<String> preprocessedArgs) {
772     AndroidSdkProvider sdk = AndroidSdkProvider.fromRuleContext(ruleContext);
773     ruleContext.registerAction(new SpawnAction.Builder()
774         .setExecutable(sdk.getAidl())
775         .addInput(idl)
776         .addInputs(idlImports)
777         .addInput(preprocessedIdls)
778         .addInput(sdk.getFrameworkAidl())
779         .addOutput(output)
780         .addArgument("-b") // Fail if trying to compile a parcelable.
781         .addArguments(preprocessedArgs)
782         .addArgument(idl.getExecPathString())
783         .addArgument(output.getExecPathString())
784         .setProgressMessage("Android IDL generation")
785         .setMnemonic("AndroidIDLGnerate")
786         .build(ruleContext));
787   }
