31   * Computes various statistics for Skylark and built-in function usage and prints it to a given

36    /**
37     * How many characters from the end of the location of a Skylark function to display.
38     */
39    private static final int NUM_LOCATION_CHARS_UNABBREVIATED = 40;
40    private final ListMultimap<String, Task> userFunctionTasks;
41    private final ListMultimap<String, Task> builtinFunctionTasks;
42    private final List<TasksStatistics> userFunctionStats;
43    private final List<TasksStatistics> builtinFunctionStats;
44    private long userTotalNanos;
45    private long builtinTotalNanos;
46  
47    private final PrintStream out;
48  
49    public SkylarkStatistics(PrintStream out, ProfileInfo info) {
50      this.out = out;
51      userFunctionTasks = info.getSkylarkUserFunctionTasks();
52      builtinFunctionTasks = info.getSkylarkBuiltinFunctionTasks();
53      userFunctionStats = new ArrayList<>();
54      builtinFunctionStats = new ArrayList<>();
55      computeStatistics();
56    }
57  
58    /**
59     * For each Skylark function compute a {@link TasksStatistics} object from the execution times of
60     * all corresponding {@link Task}s from either {@link #userFunctionTasks} or
61     * {@link #builtinFunctionTasks}. Fills fields {@link #userFunctionStats} and
62     * {@link #builtinFunctionStats}.
63     */
64    private void computeStatistics() {
65      userTotalNanos = computeStatistics(userFunctionTasks, userFunctionStats);
66      builtinTotalNanos = computeStatistics(builtinFunctionTasks, builtinFunctionStats);
67    }
68  
69    /**
70     * For each Skylark function compute a {@link TasksStatistics} object from the execution times of
71     * all corresponding {@link Task}s and add it to the list.
72     * @param tasks Map from function name to all corresponding tasks.
73     * @param stats The list to which {@link TasksStatistics} are to be added.
74     * @return The sum of the execution times of all {@link Task} values in the map.
75     */
76    private static long computeStatistics(
77        ListMultimap<String, Task> tasks, List<TasksStatistics> stats) {
78      long total = 0L;
79      for (Entry<String, List<Task>> entry : Multimaps.asMap(tasks).entrySet()) {
80        TasksStatistics functionStats = TasksStatistics.create(entry.getKey(), entry.getValue());
81        stats.add(functionStats);
82        total += functionStats.totalNanos;
83      }
84      return total;
85    }

165   private void printHistogramData(ListMultimap<String, Task> tasks, String category) {
166     out.printf("    '%s': {\n", category);
167     for (String function : tasks.keySet()) {
168       out.printf("      '%s': google.visualization.arrayToDataTable(\n", function);
169       out.print("        [['duration']");
170       for (Task task : tasks.get(function)) {
171         out.printf(",[%f]", task.duration / 1000000.);
172       }
173       out.println("],\n        false),");
174     }
175     out.println("    },");
176   }

178   private void printStatsJs(
179       List<TasksStatistics> statsList,
180       String category,
181       String dataVar,
182       String tableVar,
183       long totalNanos) {
184     String tmpVar = category + dataVar;
185     out.printf("  var statsDiv = document.getElementById('%s_function_stats');\n", category);
186     if (statsList.isEmpty()) {
187       out.println("  statsDiv.innerHTML = '<i>No relevant function calls to display. Some minor"
188           + " builtin functions may have been ignored because their names could not be used as"
189           + " variables in JavaScript.</i>'");
190     } else {
191       out.printf("  var %s = new google.visualization.DataTable();\n", tmpVar);
192       out.printf("  %s.addColumn('string', 'Location');\n", tmpVar);
193       out.printf("  %s.addColumn('string', 'Function');\n", tmpVar);
194       out.printf("  %s.addColumn('number', 'count');\n", tmpVar);
195       out.printf("  %s.addColumn('number', 'min (ms)');\n", tmpVar);
196       out.printf("  %s.addColumn('number', 'mean (ms)');\n", tmpVar);
197       out.printf("  %s.addColumn('number', 'median (ms)');\n", tmpVar);
198       out.printf("  %s.addColumn('number', 'max (ms)');\n", tmpVar);
199       out.printf("  %s.addColumn('number', 'std dev (ms)');\n", tmpVar);
200       out.printf("  %s.addColumn('number', 'mean self (ms)');\n", tmpVar);
201       out.printf("  %s.addColumn('number', 'self (ms)');\n", tmpVar);
202       out.printf("  %s.addColumn('number', 'self (%%)');\n", tmpVar);
203       out.printf("  %s.addColumn('number', 'total (ms)');\n", tmpVar);
204       out.printf("  %s.addColumn('number', 'relative (%%)');\n", tmpVar);
205       out.printf("  %s.addRows([\n", tmpVar);
206       for (TasksStatistics stats : statsList) {
207         double relativeTotal = (double) stats.totalNanos / totalNanos;
208         double relativeSelf = (double) stats.selfNanos / stats.totalNanos;
209         String[] split = stats.name.split("#");
210         String location = split[0];
211         String name = split[1];
212         out.printf("    [{v:'%s', f:'%s'}, ", location, abbreviatePath(location));
213         out.printf("'%s', ", name);
214         out.printf("%d, ", stats.count);
215         out.printf("%.3f, ", stats.minimumMillis());
216         out.printf("%.3f, ", stats.meanMillis());
217         out.printf("%.3f, ", stats.medianMillis());
218         out.printf("%.3f, ", stats.maximumMillis());
219         out.printf("%.3f, ", stats.standardDeviationMillis);
220         out.printf("%.3f, ", stats.selfMeanMillis());
221         out.printf("%.3f, ", stats.selfMillis());
222         out.printf("{v:%.4f, f:'%.3f %%'}, ", relativeSelf, relativeSelf * 100);
223         out.printf("%.3f, ", stats.totalMillis());
224         out.printf("{v:%.4f, f:'%.3f %%'}],\n", relativeTotal, relativeTotal * 100);
225       }
226       out.println("  ]);");
227       out.printf("  %s.%s = %s;\n", dataVar, category, tmpVar);
228       out.printf("  %s.%s = new google.visualization.Table(statsDiv);\n", tableVar, category);
229       out.printf(
230           "  google.visualization.events.addListener(%s.%s, 'select', selectHandler('%s'));\n",
231           tableVar,
232           category,
233           category);
234       out.printf(
235           "  %s.%s.draw(%s.%s, {showRowNumber: true, width: '100%%', height: '100%%'});\n",
236           tableVar,
237           category,
238           dataVar,
239           category);
240     }
241   }

265   /**
266    * Computes a string keeping the structure of the input but reducing the amount of characters on
267    * elements at the front if necessary.
268    *
269    * <p>Reduces the length of function location strings by keeping at least the last element fully
270    * intact and at most {@link SkylarkStatistics#NUM_LOCATION_CHARS_UNABBREVIATED} from other
271    * elements from the end. Elements before are abbreviated with their first two characters.
272    *
273    * <p>Example:
274    * "//source/tree/with/very/descriptive/and/long/hierarchy/of/directories/longfilename.bzl:42"
275    * becomes: "//so/tr/wi/ve/de/an/lo/hierarch/of/directories/longfilename.bzl:42"
276    *
277    * <p>There is no fixed length to the result as the last element is kept and the location may
278    * have many elements.
279    *
280    * @param location Either a sequence of path elements separated by
281    *     {@link StandardSystemProperty#FILE_SEPARATOR} and preceded by some root element
282    *     (e.g. "/", "C:\") or path elements separated by "." and having no root element.
283    */
284   private String abbreviatePath(String location) {
285     String[] elements;
286     int lowestAbbreviateIndex;
287     String root;
288     String separator = StandardSystemProperty.FILE_SEPARATOR.value();
289     if (location.contains(separator)) {
290       elements = location.split(separator);
291       // must take care to preserve file system roots (e.g. "/", "C:\"), keep separate
292       lowestAbbreviateIndex = 1;
293       root = location.substring(0, location.indexOf(separator) + 1);
294     } else {
295       // must be java class name for a builtin function
296       elements = location.split("\\.");
297       lowestAbbreviateIndex = 0;
298       root = "";
299       separator = ".";
300     }
301 
302     String last = elements[elements.length - 1];
303     int remaining = NUM_LOCATION_CHARS_UNABBREVIATED - last.length();
304     // start from the next to last element of the location and add until "remaining" many
305     // chars added, abbreviate rest with first 2 characters
306     for (int index = elements.length - 2; index >= lowestAbbreviateIndex; index--) {
307       String element = elements[index];
308       if (remaining > 0) {
309         int length = Math.min(remaining, element.length());
310         element = element.substring(0, length);
311         remaining -= length;
312       } else {
313         element = element.substring(0, Math.min(2, element.length()));
314       }
315       elements[index] = element;
316     }
317     return root + Joiner.on(separator).join(Arrays.asList(elements).subList(1, elements.length));
318   }
