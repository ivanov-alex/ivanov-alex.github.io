35  /**
36   * A base class for Skylark functions, whether builtin or user-defined.
37   *
38   * <p>Nomenclature:
39   * We call "Parameters" the formal parameters of a function definition.
40   * We call "Arguments" the actual values supplied at the call site.
41   *
42   * <p>The outer calling convention is like that of python3,
43   * with named parameters that can be mandatory or optional, and also be positional or named-only,
44   * and rest parameters for extra positional and keyword arguments.
45   * Callers supply a {@code List<Object>} args for positional arguments
46   * and a {@code Map<String, Object>} for keyword arguments,
47   * where positional arguments will be resolved first, then keyword arguments,
48   * with errors for a clash between the two, for missing mandatory parameter,
49   * or for unexpected extra positional or keyword argument in absence of rest parameter.
50   *
51   * <p>The inner calling convention is to pass the underlying method
52   * an {@code Object[]} of the type-checked argument values, one per expected parameter,
53   * parameters being sorted as documented in {@link FunctionSignature}.
54   *
55   * <p>The function may provide default values for optional parameters not provided by the caller.
56   * These default values can be null if there are no optional parameters or for builtin functions,
57   * but not for user-defined functions that have optional parameters.
58   */
59  // TODO(bazel-team):
60  // Provide optimized argument frobbing depending of FunctionSignature and CallerSignature
61  // (that FuncallExpression must supply), optimizing for the all-positional and all-keyword cases.
62  // Also, use better pure maps to minimize map O(n) re-creation events when processing keyword maps.
63  public abstract class BaseFunction implements Serializable {
64

66    private final String name;

68    // A function signature, including defaults and types
69    // never null after it is configured
70    @Nullable protected FunctionSignature.WithValues<Object, SkylarkType> signature;

73    @Nullable protected Location location;
74  
75    // Some functions are also Namespaces or other Skylark entities.
76    @Nullable protected Class<?> objectType;
77  
78    // Documentation for variables, if any
79    @Nullable protected List<String> paramDoc;
80  
81    // The types actually enforced by the Skylark runtime, as opposed to those enforced by the JVM,
82    // or those displayed to the user in the documentation.
83    @Nullable protected List<SkylarkType> enforcedArgumentTypes;
84  
85    // Defaults to be used when configure(annotation) is called (after the function is constructed).
86    @Nullable private Iterable<Object> unconfiguredDefaultValues;
87    // The configure(annotation) function will include these defaults in the function signature.
88    // We need to supply these defaultValues to the constructor, that will store them here, because
89    // they can't be supplied via Java annotations, due to the limitations in the annotation facility.
90    // (For extra brownies, we could supply them as Skylark expression strings, to be evaluated by our
91    // evaluator without the help of any unconfigured functions, or to be processed at compile-time;
92    // but we resolve annotations at runtime for now.)
93    // Limitations in Java annotations mean we can't express them in the SkylarkSignature annotation.
94    // (In the future, we could parse and evaluate simple Skylark expression strings, but then
95    // we'd have to be very careful of circularities during initialization).
96    // Note that though we want this list to be immutable, we don't use ImmutableList,
97    // because that can't store nulls and nulls are essential for some BuiltinFunction-s.
98    // We trust the user not to modify the list behind our back.

198   /**
199    * Process the caller-provided arguments into an array suitable for the callee (this function).
200    */
201   public Object[] processArguments(List<Object> args,
202       @Nullable Map<String, Object> kwargs,
203       @Nullable Location loc)
204       throws EvalException {
205 
206     Object[] arguments = new Object[getArgArraySize()];
207 
208     // extract function signature
209     FunctionSignature sig = signature.getSignature();
210     FunctionSignature.Shape shape = sig.getShape();
211     ImmutableList<String> names = sig.getNames();
212     List<Object> defaultValues = signature.getDefaultValues();
213 
214     // Note that this variable will be adjusted down if there are extra positionals,
215     // after these extra positionals are dumped into starParam.
216     int numPositionalArgs = args.size();
217 
218     int numMandatoryPositionalParams = shape.getMandatoryPositionals();
219     int numOptionalPositionalParams = shape.getOptionalPositionals();
220     int numMandatoryNamedOnlyParams = shape.getMandatoryNamedOnly();
221     int numOptionalNamedOnlyParams = shape.getOptionalNamedOnly();
222     boolean hasStarParam = shape.hasStarArg();
223     boolean hasKwParam = shape.hasKwArg();
224     int numPositionalParams = numMandatoryPositionalParams + numOptionalPositionalParams;
225     int numNamedOnlyParams = numMandatoryNamedOnlyParams + numOptionalNamedOnlyParams;
226     int numNamedParams = numPositionalParams + numNamedOnlyParams;
227     int kwParamIndex = names.size() - 1; // only valid if hasKwParam
228 
229     // (1) handle positional arguments
230     if (hasStarParam) {
231       // Nota Bene: we collect extra positional arguments in a (tuple,) rather than a [list],
232       // and this is actually the same as in Python.
233       int starParamIndex = numNamedParams;
234       if (numPositionalArgs > numPositionalParams) {
235         arguments[starParamIndex] = SkylarkList.tuple(
236             args.subList(numPositionalParams, numPositionalArgs));
237         numPositionalArgs = numPositionalParams; // clip numPositionalArgs
238       } else {
239         arguments[starParamIndex] = SkylarkList.EMPTY_TUPLE;
240       }
241     } else if (numPositionalArgs > numPositionalParams) {
242       throw new EvalException(loc,
243           numPositionalParams > 0
244           ? "too many (" + numPositionalArgs + ") positional arguments in call to " + this
245           : this + " does not accept positional arguments, but got " + numPositionalArgs);
246     }
247 
248     for (int i = 0; i < numPositionalArgs; i++) {
249       arguments[i] = args.get(i);
250     }
251 
252     // (2) handle keyword arguments
253     if (kwargs == null || kwargs.isEmpty()) {
254       // Easy case (2a): there are no keyword arguments.
255       // All arguments were positional, so check we had enough to fill all mandatory positionals.
256       if (numPositionalArgs < numMandatoryPositionalParams) {
257         throw new EvalException(loc, String.format(
258             "insufficient arguments received by %s (got %s, expected at least %s)",
259             this, numPositionalArgs, numMandatoryPositionalParams));
260       }
261       // We had no named argument, so fail if there were mandatory named-only parameters
262       if (numMandatoryNamedOnlyParams > 0) {
263         throw new EvalException(loc, String.format(
264             "missing mandatory keyword arguments in call to %s", this));
265       }
266       // Fill in defaults for missing optional parameters, that were conveniently grouped together,
267       // thanks to the absence of mandatory named-only parameters as checked above.
268       if (defaultValues != null) {
269         int j = numPositionalArgs - numMandatoryPositionalParams;
270         int endOptionalParams = numPositionalParams + numOptionalNamedOnlyParams;
271         for (int i = numPositionalArgs; i < endOptionalParams; i++) {
272           arguments[i] = defaultValues.get(j++);
273         }
274       }
275       // If there's a kwParam, it's empty.
276       if (hasKwParam) {
277         // TODO(bazel-team): create a fresh mutable dict, like Python does
278         arguments[kwParamIndex] = ImmutableMap.<String, Object>of();
279       }
280     } else if (hasKwParam && numNamedParams == 0) {
281       // Easy case (2b): there are no named parameters, but there is a **kwParam.
282       // Therefore all keyword arguments go directly to the kwParam.
283       // Note that *starParam and **kwParam themselves don't count as named.
284       // Also note that no named parameters means no mandatory parameters that weren't passed,
285       // and no missing optional parameters for which to use a default. Thus, no loops.
286       // TODO(bazel-team): create a fresh mutable dict, like Python does
287       arguments[kwParamIndex] = kwargs; // NB: not 2a means kwarg isn't null
288     } else {
289       // Hard general case (2c): some keyword arguments may correspond to named parameters
290       HashMap<String, Object> kwArg = hasKwParam ? new HashMap<String, Object>() : null;
291 
292       // For nicer stabler error messages, start by checking against
293       // an argument being provided both as positional argument and as keyword argument.
294       ArrayList<String> bothPosKey = new ArrayList<>();
295       for (int i = 0; i < numPositionalArgs; i++) {
296         String name = names.get(i);
297         if (kwargs.containsKey(name)) { bothPosKey.add(name); }
298       }
299       if (!bothPosKey.isEmpty()) {
300         throw new EvalException(loc,
301             String.format("argument%s '%s' passed both by position and by name in call to %s",
302                 (bothPosKey.size() > 1 ? "s" : ""), Joiner.on("', '").join(bothPosKey), this));
303       }
304 
305       // Accept the arguments that were passed.
306       for (Map.Entry<String, Object> entry : kwargs.entrySet()) {
307         String keyword = entry.getKey();
308         Object value = entry.getValue();
309         int pos = names.indexOf(keyword); // the list should be short, so linear scan is OK.
310         if (0 <= pos && pos < numNamedParams) {
311           arguments[pos] = value;
312         } else {
313           if (!hasKwParam) {
314             List<String> unexpected = Ordering.natural().sortedCopy(Sets.difference(
315                 kwargs.keySet(), ImmutableSet.copyOf(names.subList(0, numNamedParams))));
316             throw new EvalException(loc, String.format("unexpected keyword%s '%s' in call to %s",
317                     unexpected.size() > 1 ? "s" : "", Joiner.on("', '").join(unexpected), this));
318           }
319           if (kwArg.containsKey(keyword)) {
320             throw new EvalException(loc, String.format(
321                 "%s got multiple values for keyword argument '%s'", this, keyword));
322           }
323           kwArg.put(keyword, value);
324         }
325       }
326       if (hasKwParam) {
327         // TODO(bazel-team): create a fresh mutable dict, like Python does
328         arguments[kwParamIndex] = ImmutableMap.copyOf(kwArg);
329       }
330 
331       // Check that all mandatory parameters were filled in general case 2c.
332       // Note: it's possible that numPositionalArgs > numMandatoryPositionalParams but that's OK.
333       for (int i = numPositionalArgs; i < numMandatoryPositionalParams; i++) {
334         if (arguments[i] == null) {
335           throw new EvalException(loc, String.format(
336               "missing mandatory positional argument '%s' while calling %s",
337               names.get(i), this));
338         }
339       }
340 
341       int endMandatoryNamedOnlyParams = numPositionalParams + numMandatoryNamedOnlyParams;
342       for (int i = numPositionalParams; i < endMandatoryNamedOnlyParams; i++) {
343         if (arguments[i] == null) {
344           throw new EvalException(loc, String.format(
345               "missing mandatory named-only argument '%s' while calling %s",
346               names.get(i), this));
347         }
348       }
349 
350       // Get defaults for those parameters that weren't passed.
351       if (defaultValues != null) {
352         for (int i = Math.max(numPositionalArgs, numMandatoryPositionalParams);
353              i < numPositionalParams; i++) {
354           if (arguments[i] == null) {
355             arguments[i] = defaultValues.get(i - numMandatoryPositionalParams);
356           }
357         }
358         int numMandatoryParams = numMandatoryPositionalParams + numMandatoryNamedOnlyParams;
359         for (int i = numMandatoryParams + numOptionalPositionalParams; i < numNamedParams; i++) {
360           if (arguments[i] == null) {
361             arguments[i] = defaultValues.get(i - numMandatoryParams);
362           }
363         }
364       }
365     } // End of general case 2c for argument passing.
366 
367     return arguments;
368   }

370   /** check types and convert as required */
371   protected void canonicalizeArguments(Object[] arguments, Location loc) throws EvalException {
372     // TODO(bazel-team): maybe link syntax.SkylarkType and package.Type,
373     // so we can simultaneously typecheck and convert?
374     // Note that a BuiltinFunction already does typechecking of simple types.
375 
376     List<SkylarkType> types = getEnforcedArgumentTypes();
377 
378     // Check types, if supplied
379     if (types == null) {
380       return;
381     }
382     List<String> names = signature.getSignature().getNames();
383     int length = types.size();
384     for (int i = 0; i < length; i++) {
385       Object value = arguments[i];
386       SkylarkType type = types.get(i);
387       if (value != null && type != null && !type.contains(value)) {
388         throw new EvalException(loc,
389             String.format("expected %s for '%s' while calling %s but got %s instead: %s",
390                 type, names.get(i), getName(), EvalUtils.getDataTypeName(value, true), value));
391       }
392     }
393   }

405   /**
406    * The outer calling convention to a BaseFunction.
407    *
408    * @param args a list of all positional arguments (as in *starArg)
409    * @param kwargs a map for key arguments (as in **kwArgs)
410    * @param ast the expression for this function's definition
411    * @param env the Environment in the function is called
412    * @return the value resulting from evaluating the function with the given arguments
413    * @throws EvalException-s containing source information.
414    */
415   public Object call(List<Object> args,
416       @Nullable Map<String, Object> kwargs,
417       @Nullable FuncallExpression ast,
418       Environment env)
419       throws EvalException, InterruptedException {
420     Preconditions.checkState(isConfigured(), "Function %s was not configured", getName());
421 
422     // ast is null when called from Java (as there's no Skylark call site).
423     Location loc = ast == null ? Location.BUILTIN : ast.getLocation();
424 
425     Object[] arguments = processArguments(args, kwargs, loc);
426     canonicalizeArguments(arguments, loc);
427 
428     return call(arguments, ast, env);
429   }

437    * @param env the lexical environment of the function call
438    * @throws InterruptedException may be thrown in the function implementations.

440   // Don't make it abstract, so that subclasses may be defined that @Override the outer call() only.

444     throw new EvalException(

446         String.format("function %s not implemented", getName()));
447   }

452   @Override
453   public String toString() {
454     StringBuilder sb = new StringBuilder();
455     sb.append(getName());
456     if (signature != null) {

464   /** Configure a BaseFunction from a @SkylarkSignature annotation */
465   public void configure(SkylarkSignature annotation) {
466     Preconditions.checkState(!isConfigured()); // must not be configured yet
467 
468     this.paramDoc = new ArrayList<>();
469     this.signature = SkylarkSignatureProcessor.getSignature(
470         getName(), annotation, unconfiguredDefaultValues, paramDoc, getEnforcedArgumentTypes());
471     this.objectType = annotation.objectType().equals(Object.class)
472         ? null : annotation.objectType();
473     configure();
474   }

476   /** Configure a function based on its signature */
477   protected void configure() {
478     // this function is called after the signature was initialized
479     Preconditions.checkState(signature != null);
480     enforcedArgumentTypes = signature.getTypes();
481   }

483   protected boolean hasSelfArgument() {
484     Class<?> clazz = getObjectType();
485     if (clazz == null) {
486       return false;
487     }
488     List<SkylarkType> types = signature.getTypes();
489     ImmutableList<String> names = signature.getSignature().getNames();
490 
491     return (!types.isEmpty() && types.get(0).canBeCastTo(clazz))
492         || (!names.isEmpty() && names.get(0).equals("self"));
493   }

495   protected String getObjectTypeString() {
496     Class<?> clazz = getObjectType();
497     if (clazz == null) {
498       return "";
499     }
500     return EvalUtils.getDataTypeNameFromClass(clazz, false) + ".";
501   }

510   /**
511    * Returns the signature as "[className.]methodName(name1: paramType1, name2: paramType2, ...)"
512    * or "[className.]methodName(paramType1, paramType2, ...)", depending on the value of showNames.
513    */
514   public String getShortSignature(boolean showNames) {
515     StringBuilder builder = new StringBuilder();
516     boolean hasSelf = hasSelfArgument();
517 
518     builder.append(getFullName()).append("(");
519     signature.toStringBuilder(builder, showNames, false, false, hasSelf);
520     builder.append(")");
521 
522     return builder.toString();
523   }

525   /**
526    * Prints the types of the first {@code howManyArgsToPrint} given arguments as
527    * "(type1, type2, ...)"
528    */
529   protected String printTypeString(Object[] args, int howManyArgsToPrint) {
530     StringBuilder builder = new StringBuilder();
531     builder.append("(");
532 
533     int start = hasSelfArgument() ? 1 : 0;
534     for (int pos = start; pos < howManyArgsToPrint; ++pos) {
535       builder.append(EvalUtils.getDataTypeName(args[pos]));
536 
537       if (pos < howManyArgsToPrint - 1) {
538         builder.append(", ");
539       }
540     }
541     builder.append(")");
542     return builder.toString();
543   }

545   @Override
546   public boolean equals(@Nullable Object other) {
547     if (other instanceof BaseFunction) {
548       BaseFunction that = (BaseFunction) other;
549       // In theory, the location alone unambiguously identifies a given function. However, in
550       // some test cases the location might not have a valid value, thus we also check the name.
551       return Objects.equals(this.name, that.name) && Objects.equals(this.location, that.location);
552     }
553     return false;
554   }
