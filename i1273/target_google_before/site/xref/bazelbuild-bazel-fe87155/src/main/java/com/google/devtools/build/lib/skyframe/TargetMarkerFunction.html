29  /**
30   * A SkyFunction for {@link TargetMarkerValue}s. Returns a {@link
31   * TargetMarkerValue#TARGET_MARKER_INSTANCE} if the {@link Label} in the {@link SkyKey}
32   * specifies a {@link Package} that exists and a {@link Target} that exists in that package. The
33   * package may have errors.
34   */
35  public final class TargetMarkerFunction implements SkyFunction {
36  
37    public TargetMarkerFunction() {
38    }
39  
40    @Override
41    public SkyValue compute(SkyKey key, Environment env) throws TargetMarkerFunctionException {
42      Label label = (Label) key.argument();
43      PathFragment pkgForLabel = label.getPackageFragment();
44  
45      if (label.getName().contains("/")) {
46        // This target is in a subdirectory, therefore it could potentially be invalidated by
47        // a new BUILD file appearing in the hierarchy.
48        PathFragment containingDirectory = label.toPathFragment().getParentDirectory();
49        ContainingPackageLookupValue containingPackageLookupValue = null;
50        try {
51          PackageIdentifier newPkgId = new PackageIdentifier(
52              label.getPackageIdentifier().getRepository(), containingDirectory);
53          containingPackageLookupValue = (ContainingPackageLookupValue) env.getValueOrThrow(
54              ContainingPackageLookupValue.key(newPkgId),
55              BuildFileNotFoundException.class, InconsistentFilesystemException.class);
56        } catch (BuildFileNotFoundException e) {
57          // Thrown when there are IO errors looking for BUILD files.
58          throw new TargetMarkerFunctionException(e);
59        } catch (InconsistentFilesystemException e) {
60          throw new TargetMarkerFunctionException(new NoSuchTargetException(label,
61              e.getMessage()));
62        }
63        if (containingPackageLookupValue == null) {
64          return null;
65        }
66        if (!containingPackageLookupValue.hasContainingPackage()) {
67          // This means the label's package doesn't exist. E.g. there is no package 'a' and we are
68          // trying to build the target for label 'a:b/foo'.
69          throw new TargetMarkerFunctionException(new BuildFileNotFoundException(
70              label.getPackageIdentifier(), "BUILD file not found on package path for '"
71                  + pkgForLabel.getPathString() + "'"));
72        }
73        if (!containingPackageLookupValue.getContainingPackageName().equals(
74                label.getPackageIdentifier())) {
75          throw new TargetMarkerFunctionException(new NoSuchTargetException(label,
76              String.format("Label '%s' crosses boundary of subpackage '%s'", label,
77                  containingPackageLookupValue.getContainingPackageName())));
78        }
79      }
80  
81      SkyKey pkgSkyKey = PackageValue.key(label.getPackageIdentifier());
82      NoSuchPackageException nspe = null;
83      Package pkg;
84      try {
85        PackageValue value = (PackageValue)
86            env.getValueOrThrow(pkgSkyKey, NoSuchPackageException.class);
87        if (value == null) {
88          return null;
89        }
90        pkg = value.getPackage();
91      } catch (NoSuchPackageException e) {
92        // For consistency with pre-Skyframe Blaze, we can return a valid Target from a Package
93        // containing errors.
94        pkg = e.getPackage();
95        if (pkg == null) {
96          // Re-throw this exception with our key because root causes should be targets, not packages.
97          throw new TargetMarkerFunctionException(e);
98        }
99        nspe = e;
100     }
101 
102     Target target;
103     try {
104       target = pkg.getTarget(label.getName());
105     } catch (NoSuchTargetException e) {
106       throw new TargetMarkerFunctionException(e);
107     }
108 
109     if (nspe != null) {
110       // There is a target, but its package is in error. We rethrow so that the root cause is the
111       // target, not the package. Note that targets are only in error when their package is
112       // "in error" (because a package is in error if there was an error evaluating the package, or
113       // if one of its targets was in error).
114       throw new TargetMarkerFunctionException(new NoSuchTargetException(target, nspe));
115     }
116     return TargetMarkerValue.TARGET_MARKER_INSTANCE;
117   }
118 
119   @Override
120   public String extractTag(SkyKey skyKey) {
121     return Label.print((Label) skyKey.argument());
122   }
123 
124   /**
125    * Used to declare all the exception types that can be wrapped in the exception thrown by
126    * {@link TargetMarkerFunction#compute}.
127    */
128   private static final class TargetMarkerFunctionException extends SkyFunctionException {
129     public TargetMarkerFunctionException(NoSuchTargetException e) {
130       super(e, Transience.PERSISTENT);
131     }
132 
133     public TargetMarkerFunctionException(NoSuchPackageException e) {
134       super(e, Transience.PERSISTENT);
135     }
136   }
137 }
