144   /**
145    * For every value, this class keeps track of its priority, its free-form source
146    * description, whether it was set as an implicit dependency, and the value.
147    */
148   private static final class ParsedOptionEntry {
149     private final Object value;
150     private final OptionPriority priority;
151     private final String source;
152     private final String implicitDependant;
153     private final String expandedFrom;
154     private final boolean allowMultiple;
155 
156     ParsedOptionEntry(Object value,
157         OptionPriority priority, String source, String implicitDependant, String expandedFrom,
158         boolean allowMultiple) {
159       this.value = value;
160       this.priority = priority;
161       this.source = source;
162       this.implicitDependant = implicitDependant;
163       this.expandedFrom = expandedFrom;
164       this.allowMultiple = allowMultiple;
165     }
166 
167     // Need to suppress unchecked warnings, because the "multiple occurrence"
168     // options use unchecked ListMultimaps due to limitations of Java generics.
169     @SuppressWarnings({"unchecked", "rawtypes"})
170     Object getValue() {
171       if (allowMultiple) {
172         // Sort the results by option priority and return them in a new list.
173         // The generic type of the list is not known at runtime, so we can't
174         // use it here. It was already checked in the constructor, so this is
175         // type-safe.
176         List result = Lists.newArrayList();
177         ListMultimap realValue = (ListMultimap) value;
178         for (OptionPriority priority : OptionPriority.values()) {
179           // If there is no mapping for this key, this check avoids object creation (because
180           // ListMultimap has to return a new object on get) and also an unnecessary addAll call.
181           if (realValue.containsKey(priority)) {
182             result.addAll(realValue.get(priority));
183           }
184         }
185         return result;
186       }
187       return value;
188     }
189 
190     // Need to suppress unchecked warnings, because the "multiple occurrence"
191     // options use unchecked ListMultimaps due to limitations of Java generics.
192     @SuppressWarnings({"unchecked", "rawtypes"})
193     void addValue(OptionPriority addedPriority, Object addedValue) {
194       Preconditions.checkState(allowMultiple);
195       ListMultimap optionValueList = (ListMultimap) value;
196       if (addedValue instanceof List<?>) {
197         for (Object element : (List<?>) addedValue) {
198           optionValueList.put(addedPriority, element);
199         }
200       } else {
201         optionValueList.put(addedPriority, addedValue);
202       }
203     }
204 
205     OptionValueDescription asOptionValueDescription(String fieldName) {
206       return new OptionValueDescription(fieldName, getValue(), priority,
207           source, implicitDependant, expandedFrom);
208     }
209   }

254   static Map<String, Object> optionsAsMap(OptionsBase optionsInstance) {
255     Map<String, Object> map = Maps.newHashMap();
256     for (Field field : OptionsParser.getAllAnnotatedFields(optionsInstance.getClass())) {
257       try {
258         String name = field.getAnnotation(Option.class).name();
259         Object value = field.get(optionsInstance);
260         map.put(name, value);
261       } catch (IllegalAccessException e) {
262         throw new IllegalStateException(e); // unreachable
263       }
264     }
265     return map;
266   }

272   /**
273    * Implements {@link OptionsParser#asListOfUnparsedOptions()}.
274    */
275   List<UnparsedOptionValueDescription> asListOfUnparsedOptions() {
276     List<UnparsedOptionValueDescription> result = Lists.newArrayList(unparsedValues);
277     // It is vital that this sort is stable so that options on the same priority are not reordered.
278     Collections.sort(result, new Comparator<UnparsedOptionValueDescription>() {
279       @Override
280       public int compare(UnparsedOptionValueDescription o1,
281           UnparsedOptionValueDescription o2) {
282         return o1.getPriority().compareTo(o2.getPriority());
283       }
284     });
285     return result;
286   }

288   /**
289    * Implements {@link OptionsParser#asListOfExplicitOptions()}.
290    */
291   List<UnparsedOptionValueDescription> asListOfExplicitOptions() {
292     List<UnparsedOptionValueDescription> result = Lists.newArrayList(Iterables.filter(
293       unparsedValues,
294       new Predicate<UnparsedOptionValueDescription>() {
295         @Override
296         public boolean apply(UnparsedOptionValueDescription input) {
297           return input.isExplicit();
298         }
299     }));
300     // It is vital that this sort is stable so that options on the same priority are not reordered.
301     Collections.sort(result, new Comparator<UnparsedOptionValueDescription>() {
302       @Override
303       public int compare(UnparsedOptionValueDescription o1,
304           UnparsedOptionValueDescription o2) {
305         return o1.getPriority().compareTo(o2.getPriority());
306       }
307     });
308     return result;
309   }

311   /**
312    * Implements {@link OptionsParser#canonicalize}.
313    */
314   List<String> asCanonicalizedList() {
315     List<UnparsedOptionValueDescription> processed = Lists.newArrayList(unparsedValues);
316     Collections.sort(processed, new Comparator<UnparsedOptionValueDescription>() {
317       // This Comparator sorts implicit requirement options to the end, keeping their existing
318       // order, and sorts the other options alphabetically.
319       @Override
320       public int compare(UnparsedOptionValueDescription o1,
321           UnparsedOptionValueDescription o2) {
322         if (o1.isImplicitRequirement()) {
323           return o2.isImplicitRequirement() ? 0 : 1;
324         }
325         if (o2.isImplicitRequirement()) {
326           return -1;
327         }
328         return o1.getName().compareTo(o2.getName());
329       }
330     });
331 
332     List<String> result = Lists.newArrayList();
333     for (int i = 0; i < processed.size(); i++) {
334       UnparsedOptionValueDescription value = processed.get(i);
335       // Skip an option if the next option is the same, but only if the option does not allow
336       // multiple values.
337       if (!value.allowMultiple()) {
338         if ((i < processed.size() - 1) && value.getName().equals(processed.get(i + 1).getName())) {
339           continue;
340         }
341       }
342 
343       // Ignore expansion options.
344       if (value.isExpansion()) {
345         continue;
346       }
347 
348       result.add("--" + value.getName() + "=" + value.getUnparsedValue());
349     }
350     return result;
351   }

354    * Implements {@link OptionsParser#asListOfEffectiveOptions()}.
355    */
356   List<OptionValueDescription> asListOfEffectiveOptions() {
357     List<OptionValueDescription> result = Lists.newArrayList();
358     for (Map.Entry<String,Field> mapEntry : optionsData.getAllNamedFields()) {
359       String fieldName = mapEntry.getKey();
360       Field field = mapEntry.getValue();
361       ParsedOptionEntry entry = parsedValues.get(field);
362       if (entry == null) {
363         Object value = optionsData.getDefaultValue(field);
364         result.add(new OptionValueDescription(fieldName, value, OptionPriority.DEFAULT,
365             null, null, null));
366       } else {
367         result.add(entry.asOptionValueDescription(fieldName));
368       }
369     }
370     return result;
371   }

377   private void maybeAddDeprecationWarning(Field field) {
378     Option option = field.getAnnotation(Option.class);
379     // Continue to support the old behavior for @Deprecated options.
380     String warning = option.deprecationWarning();
381     if (!warning.equals("") || (field.getAnnotation(Deprecated.class) != null)) {
382       warnings.add("Option '" + option.name() + "' is deprecated"
383           + (warning.equals("") ? "" : ": " + warning));
384     }
385   }

387   // Warnings should not end with a '.' because the internal reporter adds one automatically.
388   private void setValue(Field field, String name, Object value,
389       OptionPriority priority, String source, String implicitDependant, String expandedFrom) {
390     ParsedOptionEntry entry = parsedValues.get(field);
391     if (entry != null) {
392       // Override existing option if the new value has higher or equal priority.
393       if (priority.compareTo(entry.priority) >= 0) {
394         // Output warnings:
395         if ((implicitDependant != null) && (entry.implicitDependant != null)) {
396           if (!implicitDependant.equals(entry.implicitDependant)) {
397             warnings.add("Option '" + name + "' is implicitly defined by both option '" +
398                 entry.implicitDependant + "' and option '" + implicitDependant + "'");
399           }
400         } else if ((implicitDependant != null) && priority.equals(entry.priority)) {
401           warnings.add("Option '" + name + "' is implicitly defined by option '" +
402               implicitDependant + "'; the implicitly set value overrides the previous one");
403         } else if (entry.implicitDependant != null) {
404           warnings.add("A new value for option '" + name + "' overrides a previous " +
405               "implicit setting of that option by option '" + entry.implicitDependant + "'");
406         } else if ((priority == entry.priority) &&
407             ((entry.expandedFrom == null) && (expandedFrom != null))) {
408           // Create a warning if an expansion option overrides an explicit option:
409           warnings.add("The option '" + expandedFrom + "' was expanded and now overrides a "
410               + "previous explicitly specified option '" + name + "'");
411         }
412 
413         // Record the new value:
414         parsedValues.put(field,
415             new ParsedOptionEntry(value, priority, source, implicitDependant, expandedFrom, false));
416       }
417     } else {
418       parsedValues.put(field,
419           new ParsedOptionEntry(value, priority, source, implicitDependant, expandedFrom, false));
420       maybeAddDeprecationWarning(field);
421     }
422   }
423 
424   private void addListValue(Field field, Object value, OptionPriority priority, String source,
425       String implicitDependant, String expandedFrom) {
426     ParsedOptionEntry entry = parsedValues.get(field);
427     if (entry == null) {
428       entry = new ParsedOptionEntry(ArrayListMultimap.create(), priority, source,
429           implicitDependant, expandedFrom, true);
430       parsedValues.put(field, entry);
431       maybeAddDeprecationWarning(field);
432     }
433     entry.addValue(priority, value);
434   }

469     return parse(priority, sourceFunction, null, null, args);
470   }

472   /**
473    * Parses the args, and returns what it doesn't parse. May be called multiple
474    * times, and may be called recursively. Calls may contain intersecting sets
475    * of options; in that case, the arg seen last takes precedence.
476    *
477    * <p>The method uses the invariant that if an option has neither an implicit
478    * dependant nor an expanded from value, then it must have been explicitly
479    * set.
480    */
481   private List<String> parse(OptionPriority priority,
482       final Function<? super String, String> sourceFunction, String implicitDependant,
483       String expandedFrom, List<String> args) throws OptionsParsingException {
484     List<String> unparsedArgs = Lists.newArrayList();
485     LinkedHashMap<String,List<String>> implicitRequirements = Maps.newLinkedHashMap();
486     for (int pos = 0; pos < args.size(); pos++) {
487       String arg = args.get(pos);
488       if (!arg.startsWith("-")) {
489         unparsedArgs.add(arg);
490         continue;  // not an option arg
491       }
492       if (arg.equals("--")) {  // "--" means all remaining args aren't options
493         while (++pos < args.size()) {
494           unparsedArgs.add(args.get(pos));
495         }
496         break;
497       }
498 
499       String value = null;
500       Field field;
501       boolean booleanValue = true;
502 
503       if (arg.length() == 2) { // -l  (may be nullary or unary)
504         field = optionsData.getFieldForAbbrev(arg.charAt(1));
505         booleanValue = true;
506 
507       } else if (arg.length() == 3 && arg.charAt(2) == '-') { // -l-  (boolean)
508         field = optionsData.getFieldForAbbrev(arg.charAt(1));
509         booleanValue = false;
510 
511       } else if (allowSingleDashLongOptions // -long_option
512           || arg.startsWith("--")) { // or --long_option
513         int equalsAt = arg.indexOf('=');
514         int nameStartsAt = arg.startsWith("--") ? 2 : 1;
515         String name =
516             equalsAt == -1 ? arg.substring(nameStartsAt) : arg.substring(nameStartsAt, equalsAt);
517         if (name.trim().equals("")) {
518           throw new OptionsParsingException("Invalid options syntax: " + arg, arg);
519         }
520         value = equalsAt == -1 ? null : arg.substring(equalsAt + 1);
521         field = optionsData.getFieldFromName(name);
522 
523         // look for a "no"-prefixed option name: "no<optionname>";
524         // (Undocumented: we also allow --no_foo.  We're generous like that.)
525         if (field == null && name.startsWith("no")) {
526           String realname = name.substring(name.startsWith("no_") ? 3 : 2);
527           field = optionsData.getFieldFromName(realname);
528           booleanValue = false;
529           if (field != null) {
530             // TODO(bazel-team): Add tests for these cases.
531             if (!OptionsParserImpl.isBooleanField(field)) {
532               throw new OptionsParsingException(
533                   "Illegal use of 'no' prefix on non-boolean option: " + arg, arg);
534             }
535             if (value != null) {
536               throw new OptionsParsingException(
537                   "Unexpected value after boolean option: " + arg, arg);
538             }
539             // "no<optionname>" signifies a boolean option w/ false value
540             value = "0";
541           }
542         }
543 
544       } else {
545         throw new OptionsParsingException("Invalid options syntax: " + arg, arg);
546       }
547 
548       if (field == null) {
549         throw new OptionsParsingException("Unrecognized option: " + arg, arg);
550       }
551       
552       if (value == null) {
553         // special case boolean to supply value based on presence of "no" prefix
554         if (OptionsParserImpl.isBooleanField(field)) {
555           value = booleanValue ? "1" : "0";
556         } else if (field.getType().equals(Void.class)) {
557           // this is expected, Void type options have no args
558         } else if (pos != args.size() - 1) {
559           value = args.get(++pos);  // "--flag value" form
560         } else {
561           throw new OptionsParsingException("Expected value after " + arg);
562         }
563       }
564 
565       Option option = field.getAnnotation(Option.class);
566       final String originalName = option.name();
567       if (implicitDependant == null) {
568         // Log explicit options and expanded options in the order they are parsed (can be sorted
569         // later). Also remember whether they were expanded or not. This information is needed to
570         // correctly canonicalize flags.
571         unparsedValues.add(new UnparsedOptionValueDescription(originalName, field, value,
572             priority, sourceFunction.apply(originalName), expandedFrom == null));
573       }
574 
575       // Handle expansion options.
576       if (option.expansion().length > 0) {
577         Function<Object, String> expansionSourceFunction = Functions.<String>constant(
578             "expanded from option --" + originalName + " from " +
579             sourceFunction.apply(originalName));
580         maybeAddDeprecationWarning(field);
581         List<String> unparsed = parse(priority, expansionSourceFunction, null, originalName,
582             ImmutableList.copyOf(option.expansion()));
583         if (!unparsed.isEmpty()) {
584           // Throw an assertion, because this indicates an error in the code that specified the
585           // expansion for the current option.
586           throw new AssertionError("Unparsed options remain after parsing expansion of " +
587             arg + ":" + Joiner.on(' ').join(unparsed));
588         }
589       } else {
590         Converter<?> converter = optionsData.getConverter(field);
591         Object convertedValue;
592         try {
593           convertedValue = converter.convert(value);
594         } catch (OptionsParsingException e) {
595           // The converter doesn't know the option name, so we supply it here by
596           // re-throwing:
597           throw new OptionsParsingException("While parsing option " + arg
598                                             + ": " + e.getMessage(), e);
599         }
600 
601         // ...but allow duplicates of single-use options across separate calls to
602         // parse(); latest wins:
603         if (!option.allowMultiple()) {
604           setValue(field, originalName, convertedValue,
605               priority, sourceFunction.apply(originalName), implicitDependant, expandedFrom);
606         } else {
607           // But if it's a multiple-use option, then just accumulate the
608           // values, in the order in which they were seen.
609           // Note: The type of the list member is not known; Java introspection
610           // only makes it available in String form via the signature string
611           // for the field declaration.
612           addListValue(field, convertedValue, priority, sourceFunction.apply(originalName),
613               implicitDependant, expandedFrom);
614         }
615       }
616 
617       // Collect any implicit requirements.
618       if (option.implicitRequirements().length > 0) {
619         implicitRequirements.put(option.name(), Arrays.asList(option.implicitRequirements()));
620       }
621     }
622 
623     // Now parse any implicit requirements that were collected.
624     // TODO(bazel-team): this should happen when the option is encountered.
625     if (!implicitRequirements.isEmpty()) {
626       for (Map.Entry<String,List<String>> entry : implicitRequirements.entrySet()) {
627         Function<Object, String> requirementSourceFunction = Functions.<String>constant(
628             "implicit requirement of option --" + entry.getKey() + " from " +
629             sourceFunction.apply(entry.getKey()));
630 
631         List<String> unparsed = parse(priority, requirementSourceFunction, entry.getKey(), null,
632             entry.getValue());
633         if (!unparsed.isEmpty()) {
634           // Throw an assertion, because this indicates an error in the code that specified in the
635           // implicit requirements for the option(s).
636           throw new AssertionError("Unparsed options remain after parsing implicit options:"
637               + Joiner.on(' ').join(unparsed));
638         }
639       }
640     }
641 
642     return unparsedArgs;
643   }

646    * Gets the result of parsing the options.
647    */
648   <O extends OptionsBase> O getParsedOptions(Class<O> optionsClass) {
649     // Create the instance:
650     O optionsInstance;
651     try {
652       Constructor<O> constructor = optionsData.getConstructor(optionsClass);
653       if (constructor == null) {
654         return null;
655       }
656       optionsInstance = constructor.newInstance(new Object[0]);
657     } catch (Exception e) {
658       throw new IllegalStateException(e);
659     }
660 
661     // Set the fields
662     for (Field field : optionsData.getFieldsForClass(optionsClass)) {
663       Object value = getValue(field);
664       if (value == null) {
665         value = optionsData.getDefaultValue(field);
666       }
667       try {
668         field.set(optionsInstance, value);
669       } catch (IllegalAccessException e) {
670         throw new IllegalStateException(e);
671       }
672     }
673     return optionsInstance;
674   }

695   static Converter<?> findConverter(Field optionField) {
696     Option annotation = optionField.getAnnotation(Option.class);
697     if (annotation.converter() == Converter.class) {
698       Type type;
699       if (annotation.allowMultiple()) {
700         // The OptionParserImpl already checked that the type is List<T> for some T;
701         // here we extract the type T.
702         type = ((ParameterizedType) optionField.getGenericType()).getActualTypeArguments()[0];
703       } else {
704         type = optionField.getType();
705       }
706       Converter<?> converter = DEFAULT_CONVERTERS.get(type);
707       if (converter == null) {
708         throw new AssertionError("No converter found for "
709             + type + "; possible fix: add "
710             + "converter=... to @Option annotation for "
711             + optionField.getName());
712       }
713       return converter;
714     }
715     try {
716       Class<?> converter = annotation.converter();
717       Constructor<?> constructor = converter.getConstructor(new Class<?>[0]);
718       return (Converter<?>) constructor.newInstance(new Object[0]);
719     } catch (Exception e) {
720       throw new AssertionError(e);
721     }
722   }
