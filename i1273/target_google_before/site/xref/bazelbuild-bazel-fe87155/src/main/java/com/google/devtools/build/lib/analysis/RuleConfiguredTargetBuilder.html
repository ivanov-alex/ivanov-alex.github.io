79    public RuleConfiguredTargetBuilder(RuleContext ruleContext) {
80      this.ruleContext = ruleContext;

85    /**
86     * Constructs the RuleConfiguredTarget instance based on the values set for this Builder.
87     */
88    public ConfiguredTarget build() {
89      if (ruleContext.getConfiguration().enforceConstraints()) {
90        checkConstraints();
91      }
92      if (ruleContext.hasErrors()) {
93        return null;
94      }
95  
96      FilesToRunProvider filesToRunProvider = new FilesToRunProvider(ruleContext.getLabel(),
97          RuleContext.getFilesToRun(runfilesSupport, filesToBuild), runfilesSupport, executable);
98      add(FileProvider.class, new FileProvider(ruleContext.getLabel(), filesToBuild));
99      add(FilesToRunProvider.class, filesToRunProvider);
100     addSkylarkTransitiveInfo(FilesToRunProvider.SKYLARK_NAME, filesToRunProvider);
101 
102     if (runfilesSupport != null) {
103       // If a binary is built, build its runfiles, too
104       addOutputGroup(
105           OutputGroupProvider.HIDDEN_TOP_LEVEL, runfilesSupport.getRunfilesMiddleman());
106     } else if (providers.get(RunfilesProvider.class) != null) {
107       // If we don't have a RunfilesSupport (probably because this is not a binary rule), we still
108       // want to build the files this rule contributes to runfiles of dependent rules so that we
109       // report an error if one of these is broken.
110       //
111       // Note that this is a best-effort thing: there is .getDataRunfiles() and all the language-
112       // specific *RunfilesProvider classes, which we don't add here for reasons that are lost in
113       // the mists of time.
114       addOutputGroup(OutputGroupProvider.HIDDEN_TOP_LEVEL,
115           ((RunfilesProvider) providers.get(RunfilesProvider.class))
116               .getDefaultRunfiles().getAllArtifacts());
117     }
118 
119     // Create test action and artifacts if target was successfully initialized
120     // and is a test.
121     if (TargetUtils.isTestRule(ruleContext.getTarget())) {
122       Preconditions.checkState(runfilesSupport != null);
123       add(TestProvider.class, initializeTestProvider(filesToRunProvider));
124     }
125     add(ExtraActionArtifactsProvider.class, initializeExtraActions());
126     if (!outputGroupBuilders.isEmpty()) {
127       ImmutableMap.Builder<String, NestedSet<Artifact>> outputGroups = ImmutableMap.builder();
128       for (Map.Entry<String, NestedSetBuilder<Artifact>> entry : outputGroupBuilders.entrySet()) {
129         outputGroups.put(entry.getKey(), entry.getValue().build());
130       }
131 
132       add(OutputGroupProvider.class, new OutputGroupProvider(outputGroups.build()));
133     }
134 
135 
136     return new RuleConfiguredTarget(
137         ruleContext, mandatoryStampFiles, skylarkProviders.build(), providers);
138   }

140   /**
141    * Invokes Blaze's constraint enforcement system: checks that this rule's dependencies
142    * support its environments and reports appropriate errors if violations are found. Also
143    * publishes this rule's supported environments for the rules that depend on it.
144    */
145   private void checkConstraints() {
146     if (!ruleContext.getRule().getRuleClassObject().supportsConstraintChecking()) {
147       return;
148     }
149     EnvironmentCollection supportedEnvironments =
150         ConstraintSemantics.getSupportedEnvironments(ruleContext);
151     if (supportedEnvironments != null) {
152       add(SupportedEnvironmentsProvider.class, new SupportedEnvironments(supportedEnvironments));
153       ConstraintSemantics.checkConstraints(ruleContext, supportedEnvironments);
154     }
155   }

157   private TestProvider initializeTestProvider(FilesToRunProvider filesToRunProvider) {
158     int explicitShardCount = ruleContext.attributes().get("shard_count", Type.INTEGER);
159     if (explicitShardCount < 0
160         && ruleContext.getRule().isAttributeValueExplicitlySpecified("shard_count")) {
161       ruleContext.attributeError("shard_count", "Must not be negative.");
162     }
163     if (explicitShardCount > 50) {
164       ruleContext.attributeError("shard_count",
165           "Having more than 50 shards is indicative of poor test organization. "
166           + "Please reduce the number of shards.");
167     }
168     TestActionBuilder testActionBuilder = new TestActionBuilder(ruleContext);
169     InstrumentedFilesProvider instrumentedFilesProvider =
170         findProvider(InstrumentedFilesProvider.class);
171     if (instrumentedFilesProvider != null) {
172       testActionBuilder
173           .setInstrumentedFiles(instrumentedFilesProvider)
174           .setExtraEnv(instrumentedFilesProvider.getExtraEnv());
175     }
176     final TestParams testParams = testActionBuilder
177         .setFilesToRunProvider(filesToRunProvider)
178         .setExecutionRequirements(findProvider(ExecutionInfoProvider.class))
179         .setShardCount(explicitShardCount)
180         .build();
181     final ImmutableList<String> testTags =
182         ImmutableList.copyOf(ruleContext.getRule().getRuleTags());
183     return new TestProvider(testParams, testTags);
184   }

186   private LicensesProvider initializeLicensesProvider() {
187     if (!ruleContext.getConfiguration().checkLicenses()) {
188       return LicensesProviderImpl.EMPTY;
189     }
190 
191     NestedSetBuilder<TargetLicense> builder = NestedSetBuilder.linkOrder();
192     BuildConfiguration configuration = ruleContext.getConfiguration();
193     Rule rule = ruleContext.getRule();
194     License toolOutputLicense = rule.getToolOutputLicense(ruleContext.attributes());
195     if (configuration.isHostConfiguration() && toolOutputLicense != null) {
196       if (toolOutputLicense != License.NO_LICENSE) {
197         builder.add(new TargetLicense(rule.getLabel(), toolOutputLicense));
198       }
199     } else {
200       if (rule.getLicense() != License.NO_LICENSE) {
201         builder.add(new TargetLicense(rule.getLabel(), rule.getLicense()));
202       }
203 
204       for (TransitiveInfoCollection dep : ruleContext.getConfiguredTargetMap().values()) {
205         LicensesProvider provider = dep.getProvider(LicensesProvider.class);
206         if (provider != null) {
207           builder.addTransitive(provider.getTransitiveLicenses());
208         }
209       }
210     }
211 
212     return new LicensesProviderImpl(builder.build());
213   }

215   /**
216    * Scans {@code action_listeners} associated with this build to see if any
217    * {@code extra_actions} should be added to this configured target. If any
218    * action_listeners are present, a partial visit of the artifact/action graph
219    * is performed (for as long as actions found are owned by this {@link
220    * ConfiguredTarget}). Any actions that match the {@code action_listener}
221    * get an {@code extra_action} associated. The output artifacts of the
222    * extra_action are reported to the {@link AnalysisEnvironment} for
223    * bookkeeping.
224    */
225   private ExtraActionArtifactsProvider initializeExtraActions() {
226     BuildConfiguration configuration = ruleContext.getConfiguration();
227     if (configuration.isHostConfiguration()) {
228       return ExtraActionArtifactsProvider.EMPTY;
229     }
230 
231     ImmutableList<Artifact> extraActionArtifacts = ImmutableList.of();
232     NestedSetBuilder<ExtraArtifactSet> builder = NestedSetBuilder.stableOrder();
233 
234     List<Label> actionListenerLabels = configuration.getActionListeners();
235     if (!actionListenerLabels.isEmpty()
236         && ruleContext.getRule().getAttributeDefinition(":action_listener") != null) {
237       ExtraActionsVisitor visitor = new ExtraActionsVisitor(ruleContext,
238           computeMnemonicsToExtraActionMap());
239 
240       // The action list is modified within the body of the loop by the addExtraAction() call,
241       // thus the copy
242       for (Action action : ImmutableList.copyOf(
243           ruleContext.getAnalysisEnvironment().getRegisteredActions())) {
244         if (!actionsWithoutExtraAction.contains(action)) {
245           visitor.addExtraAction(action);
246         }
247       }
248 
249       extraActionArtifacts = visitor.getAndResetExtraArtifacts();
250       if (!extraActionArtifacts.isEmpty()) {
251         builder.add(ExtraArtifactSet.of(ruleContext.getLabel(), extraActionArtifacts));
252       }
253     }
254 
255     // Add extra action artifacts from dependencies
256     for (TransitiveInfoCollection dep : ruleContext.getConfiguredTargetMap().values()) {
257       ExtraActionArtifactsProvider provider =
258           dep.getProvider(ExtraActionArtifactsProvider.class);
259       if (provider != null) {
260         builder.addTransitive(provider.getTransitiveExtraActionArtifacts());
261       }
262     }
263 
264     if (mandatoryStampFiles != null && !mandatoryStampFiles.isEmpty()) {
265       builder.add(ExtraArtifactSet.of(ruleContext.getLabel(), mandatoryStampFiles));
266     }
267 
268     if (extraActionArtifacts.isEmpty() && builder.isEmpty()) {
269       return ExtraActionArtifactsProvider.EMPTY;
270     }
271     return new ExtraActionArtifactsProvider(extraActionArtifacts, builder.build());
272   }

274   /**
275    * Populates the configuration specific mnemonicToExtraActionMap
276    * based on all action_listers selected by the user (via the blaze option
277    * {@code --experimental_action_listener=<target>}).
278    */
279   private Multimap<String, ExtraActionSpec> computeMnemonicsToExtraActionMap() {
280     // We copy the multimap here every time. This could be expensive.
281     Multimap<String, ExtraActionSpec> mnemonicToExtraActionMap = HashMultimap.create();
282     for (TransitiveInfoCollection actionListener :
283         ruleContext.getPrerequisites(":action_listener", Mode.HOST)) {
284       ExtraActionMapProvider provider = actionListener.getProvider(ExtraActionMapProvider.class);
285       if (provider == null) {
286         ruleContext.ruleError(String.format(
287             "Unable to match experimental_action_listeners to this rule. "
288             + "Specified target %s is not an action_listener rule",
289             actionListener.getLabel().toString()));
290       } else {
291         mnemonicToExtraActionMap.putAll(provider.getExtraActionMap());
292       }
293     }
294     return mnemonicToExtraActionMap;
295   }

297   private <T extends TransitiveInfoProvider> T findProvider(Class<T> clazz) {

309    * Add a specific provider with a given value.
310    */
311   public RuleConfiguredTargetBuilder addProvider(
312       Class<? extends TransitiveInfoProvider> key, TransitiveInfoProvider value) {
313     Preconditions.checkNotNull(key);
314     Preconditions.checkNotNull(value);
315     AnalysisUtils.checkProvider(key);
316     providers.put(key, value);
317     return this;
318   }

339   /**
340    * Add a Skylark transitive info. The provider value must be safe (i.e. a String, a Boolean,
341    * an Integer, an Artifact, a Label, None, a Java TransitiveInfoProvider or something composed
342    * from these in Skylark using lists, sets, structs or dicts). Otherwise an EvalException is
343    * thrown.
344    */
345   public RuleConfiguredTargetBuilder addSkylarkTransitiveInfo(
346       String name, Object value, Location loc) throws EvalException {
347     try {
348       checkSkylarkObjectSafe(value);
349     } catch (IllegalArgumentException e) {
350       throw new EvalException(loc, String.format("Value of provider '%s' is of an illegal type: %s",
351           name, e.getMessage()));
352     }
353     skylarkProviders.put(name, value);
354     return this;
355   }

363     skylarkProviders.put(name, value);
364     return this;

367   /**
368    * Check if the value provided by a Skylark provider is safe (i.e. can be a
369    * TransitiveInfoProvider value).
370    */
371   private void checkSkylarkObjectSafe(Object value) {
372     if (!isSimpleSkylarkObjectSafe(value.getClass())
373         // Java transitive Info Providers are accessible from Skylark.
374         && !(value instanceof TransitiveInfoProvider)) {
375       checkCompositeSkylarkObjectSafe(value);
376     }
377   }
378 
379   private void checkCompositeSkylarkObjectSafe(Object object) {
380     if (object instanceof SkylarkApiProvider) {
381       return;
382     } else if (object instanceof SkylarkList) {
383       SkylarkList list = (SkylarkList) object;
384       if (list == SkylarkList.EMPTY_LIST
385           || isSimpleSkylarkObjectSafe(list.getContentType().getType())) {
386         // Try not to iterate over the list if avoidable.
387         return;
388       }
389       // The list can be a tuple or a list of composite items.
390       for (Object listItem : list) {
391         checkSkylarkObjectSafe(listItem);
392       }
393       return;
394     } else if (object instanceof SkylarkNestedSet) {
395       // SkylarkNestedSets cannot have composite items.
396       Class<?> contentType = ((SkylarkNestedSet) object).getContentType().getType();
397       if (!contentType.equals(Object.class) && !isSimpleSkylarkObjectSafe(contentType)) {
398         throw new IllegalArgumentException(EvalUtils.getDataTypeName(contentType));
399       }
400       return;
401     } else if (object instanceof Map<?, ?>) {
402       for (Map.Entry<?, ?> entry : ((Map<?, ?>) object).entrySet()) {
403         checkSkylarkObjectSafe(entry.getKey());
404         checkSkylarkObjectSafe(entry.getValue());
405       }
406       return;
407     } else if (object instanceof ClassObject) {
408       ClassObject struct = (ClassObject) object;
409       for (String key : struct.getKeys()) {
410         checkSkylarkObjectSafe(struct.getValue(key));
411       }
412       return;
413     }
414     throw new IllegalArgumentException(EvalUtils.getDataTypeName(object));
415   }
416 
417   private boolean isSimpleSkylarkObjectSafe(Class<?> type) {
418     return type.equals(String.class)
419         || type.equals(Integer.class)
420         || type.equals(Boolean.class)
421         || Artifact.class.isAssignableFrom(type)
422         || type.equals(Label.class)
423         || type.equals(Runtime.NoneType.class);
424   }

444   private NestedSetBuilder<Artifact> getOutputGroupBuilder(String name) {
445     NestedSetBuilder<Artifact> result = outputGroupBuilders.get(name);
446     if (result != null) {
447       return result;
448     }
449 
450     result = NestedSetBuilder.<Artifact>stableOrder();
451     outputGroupBuilders.put(name, result);
452     return result;
453   }
