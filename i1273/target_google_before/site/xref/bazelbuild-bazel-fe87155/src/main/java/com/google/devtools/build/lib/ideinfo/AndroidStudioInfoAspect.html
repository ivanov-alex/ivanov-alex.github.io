61  /**
62   * Generates ide-build information for Android Studio.
63   */
64  public class AndroidStudioInfoAspect implements ConfiguredAspectFactory {
65    public static final String NAME = "AndroidStudioInfoAspect";
66  
67    // Output groups.
68    public static final String IDE_RESOLVE = "ide-resolve";
69    public static final String IDE_BUILD = "ide-build";
70  
71    // File suffixes.
72    public static final String ASWB_BUILD_SUFFIX = ".aswb-build";
73    public static final Function<Label, String> LABEL_TO_STRING = new Function<Label, String>() {
74      @Nullable
75      @Override
76      public String apply(Label label) {
77        return label.toString();
78      }
79    };
80  
81    @Override
82    public AspectDefinition getDefinition() {
83      return new AspectDefinition.Builder(NAME)
84          .requireProvider(JavaSourceInfoProvider.class)
85          .attributeAspect("deps", AndroidStudioInfoAspect.class)
86          .build();
87    }
88  
89    @Override
90    public Aspect create(ConfiguredTarget base, RuleContext ruleContext,
91        AspectParameters parameters) {
92      Aspect.Builder builder = new Builder(NAME);
93  
94      // Collect ide build files and calculate dependencies.
95      NestedSetBuilder<Label> transitiveDependenciesBuilder = NestedSetBuilder.stableOrder();
96      NestedSetBuilder<Label> dependenciesBuilder = NestedSetBuilder.stableOrder();
97  
98      NestedSetBuilder<Artifact> ideBuildFilesBuilder = NestedSetBuilder.stableOrder();
99  
100     // todo(dslomov,tomlu): following current build info logic, this code enumerates dependencies
101     // directly by iterating over deps attribute. The more robust way to do this might be
102     // to iterate classpath as provided to build action.
103     if (ruleContext.attributes().has("deps", Type.LABEL_LIST)) {
104       Iterable<AndroidStudioInfoFilesProvider> androidStudioInfoFilesProviders =
105           ruleContext.getPrerequisites("deps", Mode.TARGET, AndroidStudioInfoFilesProvider.class);
106       for (AndroidStudioInfoFilesProvider depProvider : androidStudioInfoFilesProviders) {
107         ideBuildFilesBuilder.addTransitive(depProvider.getIdeBuildFiles());
108         transitiveDependenciesBuilder.addTransitive(depProvider.getTransitiveDependencies());
109       }
110       List<? extends TransitiveInfoCollection> deps =
111           ruleContext.getPrerequisites("deps", Mode.TARGET);
112       for (TransitiveInfoCollection dep : deps) {
113         dependenciesBuilder.add(dep.getLabel());
114       }
115 
116       Iterable<JavaExportsProvider> javaExportsProviders = ruleContext
117           .getPrerequisites("deps", Mode.TARGET, JavaExportsProvider.class);
118       for (JavaExportsProvider javaExportsProvider : javaExportsProviders) {
119         dependenciesBuilder.addTransitive(javaExportsProvider.getTransitiveExports());
120       }
121     }
122 
123     NestedSet<Label> directDependencies = dependenciesBuilder.build();
124     transitiveDependenciesBuilder.addTransitive(directDependencies);
125     NestedSet<Label> transitiveDependencies = transitiveDependenciesBuilder.build();
126 
127     RuleIdeInfo.Kind ruleKind = getRuleKind(ruleContext.getRule(), base);
128 
129     if (ruleKind != RuleIdeInfo.Kind.UNRECOGNIZED) {
130       Artifact ideBuildFile =
131           createIdeBuildArtifact(base, ruleContext, ruleKind,
132               directDependencies,
133               transitiveDependencies);
134       ideBuildFilesBuilder.add(ideBuildFile);
135     }
136 
137     NestedSet<Artifact> ideBuildFiles = ideBuildFilesBuilder.build();
138     builder
139         .addOutputGroup(IDE_BUILD, ideBuildFiles)
140         .addProvider(
141             AndroidStudioInfoFilesProvider.class,
142             new AndroidStudioInfoFilesProvider(ideBuildFiles, transitiveDependencies));
143 
144     return builder.build();
145   }

147   private static AndroidSdkRuleInfo makeAndroidSdkRuleInfo(RuleContext ruleContext,
148       AndroidSdkProvider provider) {
149     AndroidSdkRuleInfo.Builder sdkInfoBuilder = AndroidSdkRuleInfo.newBuilder();
150 
151     Path androidSdkDirectory = provider.getAndroidJar().getPath().getParentDirectory();
152     sdkInfoBuilder.setAndroidSdkPath(androidSdkDirectory.toString());
153 
154     Root genfilesDirectory = ruleContext.getConfiguration().getGenfilesDirectory();
155     sdkInfoBuilder.setGenfilesPath(genfilesDirectory.getPath().toString());
156 
157     Path binfilesPath = ruleContext.getConfiguration().getBinDirectory().getPath();
158     sdkInfoBuilder.setBinPath(binfilesPath.toString());
159 
160     return sdkInfoBuilder.build();
161   }

163   private Artifact createIdeBuildArtifact(
164       ConfiguredTarget base,
165       RuleContext ruleContext,
166       Kind ruleKind,
167       NestedSet<Label> directDependencies, NestedSet<Label> transitiveDependencies) {
168     PathFragment ideBuildFilePath = getOutputFilePath(base, ruleContext);
169     Root genfilesDirectory = ruleContext.getConfiguration().getGenfilesDirectory();
170     Artifact ideBuildFile =
171         ruleContext
172             .getAnalysisEnvironment()
173             .getDerivedArtifact(ideBuildFilePath, genfilesDirectory);
174 
175     RuleIdeInfo.Builder outputBuilder = RuleIdeInfo.newBuilder();
176 
177     outputBuilder.setLabel(base.getLabel().toString());
178 
179     outputBuilder.setBuildFile(
180         ruleContext
181             .getRule()
182             .getPackage()
183             .getBuildFile()
184             .getPath()
185             .toString());
186 
187     outputBuilder.setKind(ruleKind);
188 
189     outputBuilder.addAllDependencies(transform(directDependencies, LABEL_TO_STRING));
190     outputBuilder.addAllTransitiveDependencies(transform(transitiveDependencies, LABEL_TO_STRING));
191 
192     if (ruleKind == Kind.JAVA_LIBRARY
193         || ruleKind == Kind.JAVA_IMPORT
194         || ruleKind == Kind.JAVA_TEST
195         || ruleKind == Kind.JAVA_BINARY) {
196       outputBuilder.setJavaRuleIdeInfo(makeJavaRuleIdeInfo(base));
197     } else if (ruleKind == Kind.ANDROID_SDK) {
198       outputBuilder.setAndroidSdkRuleInfo(
199           makeAndroidSdkRuleInfo(ruleContext, base.getProvider(AndroidSdkProvider.class)));
200     }
201 
202 
203     final RuleIdeInfo ruleIdeInfo = outputBuilder.build();
204     ruleContext.registerAction(
205         makeProtoWriteAction(ruleContext.getActionOwner(), ruleIdeInfo, ideBuildFile));
206 
207     return ideBuildFile;
208   }

210   private static BinaryFileWriteAction makeProtoWriteAction(
211       ActionOwner actionOwner, final MessageLite message, Artifact artifact) {
212     return new BinaryFileWriteAction(
213         actionOwner,
214         artifact,
215         new ByteSource() {
216           @Override
217           public InputStream openStream() throws IOException {
218             return message.toByteString().newInput();
219           }
220         },
221         /*makeExecutable =*/ false);
222   }

224   private static ArtifactLocation makeArtifactLocation(Artifact artifact) {
225     return ArtifactLocation.newBuilder()
226         .setRootPath(artifact.getRoot().getPath().toString())
227         .setRelativePath(artifact.getRootRelativePathString())
228         .build();
229   }

231   private static JavaRuleIdeInfo makeJavaRuleIdeInfo(ConfiguredTarget base) {
232     JavaRuleIdeInfo.Builder builder = JavaRuleIdeInfo.newBuilder();
233     JavaRuleOutputJarsProvider outputJarsProvider =
234         base.getProvider(JavaRuleOutputJarsProvider.class);
235     if (outputJarsProvider != null) {
236       // java_library
237       collectJarsFromOutputJarsProvider(builder, outputJarsProvider);
238     } else {
239       JavaSourceInfoProvider provider = base.getProvider(JavaSourceInfoProvider.class);
240       if (provider != null) {
241         // java_import
242         collectJarsFromSourceInfoProvider(builder, provider);
243       }
244     }
245 
246     Collection<Artifact> sourceFiles = getSources(base);
247 
248     for (Artifact sourceFile : sourceFiles) {
249       builder.addSources(makeArtifactLocation(sourceFile));
250     }
251 
252     return builder.build();
253   }

255   private static void collectJarsFromSourceInfoProvider(
256       JavaRuleIdeInfo.Builder builder, JavaSourceInfoProvider provider) {
257     Collection<Artifact> sourceJarsForJarFiles = provider.getSourceJarsForJarFiles();
258     // For java_import rule, we always have only one source jar specified.
259     // The intent is that that source jar provides sources for all imported jars,
260     // so we reflect that intent, adding that jar to all LibraryArtifacts we produce
261     // for java_import rule. We should consider supporting
262     //    library=<collection of jars>+<collection of srcjars>
263     // mode in our AndroidStudio plugin (Android Studio itself supports that).
264     Artifact sourceJar;
265     if (sourceJarsForJarFiles.size() > 0) {
266       sourceJar = sourceJarsForJarFiles.iterator().next();
267     } else {
268       sourceJar = null;
269     }
270 
271     for (Artifact artifact : provider.getJarFiles()) {
272       LibraryArtifact.Builder libraryBuilder = LibraryArtifact.newBuilder();
273       libraryBuilder.setJar(makeArtifactLocation(artifact));
274       if (sourceJar != null) {
275         libraryBuilder.setSourceJar(makeArtifactLocation(sourceJar));
276       }
277       builder.addJars(libraryBuilder.build());
278     }
279   }
280 
281   private static void collectJarsFromOutputJarsProvider(
282       JavaRuleIdeInfo.Builder builder, JavaRuleOutputJarsProvider outputJarsProvider) {
283     LibraryArtifact.Builder jarsBuilder = LibraryArtifact.newBuilder();
284     Artifact classJar = outputJarsProvider.getClassJar();
285     if (classJar != null) {
286       jarsBuilder.setJar(makeArtifactLocation(classJar));
287     }
288     Artifact srcJar = outputJarsProvider.getSrcJar();
289     if (srcJar != null) {
290       jarsBuilder.setSourceJar(makeArtifactLocation(srcJar));
291     }
292     if (jarsBuilder.hasJar() || jarsBuilder.hasSourceJar()) {
293       builder.addJars(jarsBuilder.build());
294     }
295 
296 
297     LibraryArtifact.Builder genjarsBuilder = LibraryArtifact.newBuilder();
298 
299     Artifact genClassJar = outputJarsProvider.getGenClassJar();
300     if (genClassJar != null) {
301       genjarsBuilder.setJar(makeArtifactLocation(genClassJar));
302     }
303     Artifact gensrcJar = outputJarsProvider.getGensrcJar();
304     if (gensrcJar != null) {
305       genjarsBuilder.setSourceJar(makeArtifactLocation(gensrcJar));
306     }
307     if (genjarsBuilder.hasJar() || genjarsBuilder.hasSourceJar()) {
308       builder.addGeneratedJars(genjarsBuilder.build());
309     }
310   }

312   private static Collection<Artifact> getSources(ConfiguredTarget base) {
313     // Calculate source files.
314     JavaSourceInfoProvider sourceInfoProvider = base.getProvider(JavaSourceInfoProvider.class);
315     return sourceInfoProvider != null
316         ? sourceInfoProvider.getSourceFiles()
317         : ImmutableList.<Artifact>of();
318   }

320   private PathFragment getOutputFilePath(ConfiguredTarget base, RuleContext ruleContext) {
321     PathFragment packagePathFragment =
322         ruleContext.getLabel().getPackageIdentifier().getPathFragment();
323     String name = base.getLabel().getName();
324     return new PathFragment(packagePathFragment, new PathFragment(name + ASWB_BUILD_SUFFIX));
325   }
326 
327   private RuleIdeInfo.Kind getRuleKind(Rule rule, ConfiguredTarget base) {
328     RuleIdeInfo.Kind kind;
329     String ruleClassName = rule.getRuleClassObject().getName();
330     if ("java_library".equals(ruleClassName)) {
331       kind = RuleIdeInfo.Kind.JAVA_LIBRARY;
332     } else if ("java_import".equals(ruleClassName)) {
333       kind = Kind.JAVA_IMPORT;
334     } else if ("java_test".equals(ruleClassName)) {
335       kind = Kind.JAVA_TEST;
336     } else if ("java_binary".equals(ruleClassName)) {
337       kind = Kind.JAVA_BINARY;
338     } else if (base.getProvider(AndroidSdkProvider.class) != null) {
339       kind = RuleIdeInfo.Kind.ANDROID_SDK;
340     } else {
341       kind = RuleIdeInfo.Kind.UNRECOGNIZED;
342     }
343     return kind;
344   }
