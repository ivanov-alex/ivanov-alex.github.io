39  /**
40   * Tests for {@code LinuxSandboxedStrategy}.
41   *
42   * <p>The general idea for each test is to provide a file tree consisting of symlinks, directories
43   * and empty files and then handing that together with an arbitrary number of input files (what
44   * would be specified in the "srcs" attribute, for example) to the LinuxSandboxedStrategy.
45   *
46   * <p>The algorithm that processes the mounts must then always find (and thus mount) the expected
47   * tree of files given only the set of input files.
48   */
49  @RunWith(JUnit4.class)
50  public class LinuxSandboxedStrategyTest extends LinuxSandboxedStrategyTestCase {
51    /**
52     * Strips the working directory (which can be very long) from the file names in the input map, to
53     * make assertion failures easier to read.
54     */
55    private ImmutableMap<String, String> userFriendlyMap(Map<Path, Path> input) {
56      ImmutableMap.Builder<String, String> userFriendlyMap = ImmutableMap.builder();
57      for (Entry<Path, Path> entry : input.entrySet()) {
58        String key = entry.getKey().getPathString().replace(workspaceDir.getPathString(), "");
59        String value = entry.getValue().getPathString().replace(workspaceDir.getPathString(), "");
60        userFriendlyMap.put(key, value);
61      }
62      return userFriendlyMap.build();
63    }
64  
65    /**
66     * Takes a map of file specifications, creates the necessary files / symlinks / dirs,
67     * mounts files listed in customMount at their canonical location in the sandbox and returns the
68     * output of {@code LinuxSandboxedStrategy#fixMounts} for it.
69     */
70    private ImmutableMap<String, String> userFriendlyMounts(
71        Map<String, String> linksAndFiles, List<String> customMounts) throws IOException {
72      return userFriendlyMap(mounts(linksAndFiles, customMounts));
73    }
74  
75    private ImmutableMap<Path, Path> mounts(
76        Map<String, String> linksAndFiles, List<String> customMounts) throws IOException {
77      createTreeStructure(linksAndFiles);
78  
79      ImmutableMap.Builder<Path, Path> mounts = ImmutableMap.builder();
80      for (String customMount : customMounts) {
81        Path customMountPath = workspaceDir.getRelative(customMount);
82        mounts.put(customMountPath, customMountPath);
83      }
84      return LinuxSandboxedStrategy.validateMounts(
85          LinuxSandboxedStrategy.withResolvedSymlinks(
86              LinuxSandboxedStrategy.withRecursedDirs(mounts.build())));
87    }
88  
89    /**
90     * Takes a map of file specifications, creates the necessary files / symlinks / dirs,
91     * mounts the first file of the specification at its canonical location in the sandbox and returns
92     * the output of {@code LinuxSandboxedStrategy#fixMounts} for it.
93     */
94    private Map<String, String> userFriendlyMounts(Map<String, String> linksAndFiles)
95        throws IOException {
96      return userFriendlyMap(mounts(linksAndFiles));
97    }
98  
99    private Map<Path, Path> mounts(Map<String, String> linksAndFiles) throws IOException {
100     return mounts(
101         linksAndFiles, ImmutableList.of(Iterables.getFirst(linksAndFiles.keySet(), null)));
102   }
103 
104   /**
105    * Returns a map of mount entries for a list files, which can be used to assert that all
106    * expected mounts have been made by the LinuxSandboxedStrategy.
107    */
108   private ImmutableMap<String, String> userFriendlyAsserts(List<String> asserts) {
109     return userFriendlyMap(asserts(asserts));
110   }
111 
112   private ImmutableMap<Path, Path> asserts(List<String> asserts) {
113     ImmutableMap.Builder<Path, Path> pathifiedAsserts = ImmutableMap.builder();
114     for (String fileName : asserts) {
115       Path inputPath = workspaceDir.getRelative(fileName);
116       pathifiedAsserts.put(inputPath, inputPath);
117     }
118     return pathifiedAsserts.build();
119   }
120 
121   private void createTreeStructure(Map<String, String> linksAndFiles) throws IOException {
122     for (Entry<String, String> entry : linksAndFiles.entrySet()) {
123       Path filePath = workspaceDir.getRelative(entry.getKey());
124       String linkTarget = entry.getValue();
125 
126       FileSystemUtils.createDirectoryAndParents(filePath.getParentDirectory());
127 
128       if (!linkTarget.isEmpty()) {
129         filePath.createSymbolicLink(new PathFragment(linkTarget));
130       } else if (filePath.getPathString().endsWith("/")) {
131         filePath.createDirectory();
132       } else {
133         FileSystemUtils.createEmptyFile(filePath);
134       }
135     }
136   }

138   @Test
139   public void testResolvesRelativeFileToFileSymlinkInSameDir() throws IOException {
140     Map<String, String> testFiles = new LinkedHashMap<>();
141     testFiles.put("symlink.txt", "goal.txt");
142     testFiles.put("goal.txt", "");
143 
144     List<String> assertMounts = new ArrayList<>();
145     assertMounts.add("symlink.txt");
146     assertMounts.add("goal.txt");
147 
148     assertThat(userFriendlyMounts(testFiles)).isEqualTo(userFriendlyAsserts(assertMounts));
149   }

174   @Test
175   public void testRecursesSubDirs() throws IOException {
176     ImmutableList<String> inputFile = ImmutableList.of("a/b");
177 
178     Map<String, String> testFiles =
179         ImmutableMap.of(
180             "a/b/x.txt", "",
181             "a/b/y.txt", "z.txt",
182             "a/b/z.txt", "");
183 
184     List<String> assertMounts = ImmutableList.of("a/b/x.txt", "a/b/y.txt", "a/b/z.txt");
185 
186     assertThat(userFriendlyMounts(testFiles, inputFile))
187         .isEqualTo(userFriendlyAsserts(assertMounts));
188   }

190   /**
191    * Test that the algorithm correctly identifies and refuses symlink loops.
192    */
193   @Test
194   public void testCatchesSymlinkLoop() throws IOException {
195     try {
196       mounts(
197           ImmutableMap.of(
198               "a", "b",
199               "b", "a"));
200       fail();
201     } catch (IOException e) {
202       assertThat(e)
203           .hasMessage(
204               String.format(
205                   "%s (Too many levels of symbolic links)",
206                   workspaceDir.getRelative("a").getPathString()));
207     }
208   }

230   @Test
231   public void testParseManifestFile() throws IOException {
232     Path targetDir = workspaceDir.getRelative("runfiles");
233     targetDir.createDirectory();
234 
235     Path testFile = workspaceDir.getRelative("testfile");
236     FileSystemUtils.createEmptyFile(testFile);
237 
238     Path manifestFile = workspaceDir.getRelative("MANIFEST");
239     FileSystemUtils.writeContent(
240         manifestFile,
241         Charset.defaultCharset(),
242         String.format("x/testfile %s\nx/emptyfile \n", testFile.getPathString()));
243 
244     MountMap<Path, Path> mounts =
245         LinuxSandboxedStrategy.parseManifestFile(
246             targetDir, manifestFile.getPathFile());
247 
248     assertThat(userFriendlyMap(mounts))
249         .isEqualTo(
250             userFriendlyMap(
251                 ImmutableMap.of(
252                     fileSystem.getPath("/runfiles/x/testfile"),
253                     testFile,
254                     fileSystem.getPath("/runfiles/x/emptyfile"),
255                     fileSystem.getPath("/dev/null"))));
256   }

258   @Test
259   public void testMountMapWithNormalMounts() throws IOException {
260     // Allowed: Just two normal mounts (a -> sandbox/a, b -> sandbox/b)
261     MountMap<Path, Path> mounts = new MountMap<>();
262     mounts.put(fileSystem.getPath("/a"), workspaceDir.getRelative("a"));
263     mounts.put(fileSystem.getPath("/b"), workspaceDir.getRelative("b"));
264     assertThat(mounts)
265         .isEqualTo(
266             ImmutableMap.of(
267                 fileSystem.getPath("/a"), workspaceDir.getRelative("a"),
268                 fileSystem.getPath("/b"), workspaceDir.getRelative("b")));
269   }

298   @Test
299   public void testMountMapWithTwoThingsOneTarget() throws IOException {
300     // Forbidden: Mount two things onto the same target (x -> sandbox/a, y -> sandbox/a)
301     try {
302       MountMap<Path, Path> mounts = new MountMap<>();
303       mounts.put(fileSystem.getPath("/x"), workspaceDir.getRelative("a"));
304       mounts.put(fileSystem.getPath("/x"), workspaceDir.getRelative("b"));
305       fail();
306     } catch (IllegalArgumentException e) {
307       assertThat(e)
308           .hasMessage(
309               String.format(
310                   "Cannot mount both '%s' and '%s' onto '%s'",
311                   workspaceDir.getRelative("a"),
312                   workspaceDir.getRelative("b"),
313                   fileSystem.getPath("/x")));
314     }
