47    /**
48     * Test of viewOf method, of class DirectoryEntry.
49     */
50    @Test
51    public void testViewOf() {
52      ByteBuffer buffer = ByteBuffer.allocate(100).order(ByteOrder.LITTLE_ENDIAN);
53      for (int i = 0; i < 100; i++) {
54        buffer.put((byte) i);
55      }
56      int offset = 20;
57      int filenameLength = 10;
58      int extraLength = 6;
59      int commentLength = 8;
60      int marker = DirectoryEntry.SIGNATURE;
61      buffer.putShort(offset + ZipInputStream.CENNAM, (short) filenameLength); // filename length
62      buffer.putShort(offset + ZipInputStream.CENEXT, (short) extraLength); // extra data length
63      buffer.putShort(offset + ZipInputStream.CENCOM, (short) commentLength); // comment length
64      buffer.putInt(20, marker); // any marker
65      buffer.position(offset);
66      DirectoryEntry view = DirectoryEntry.viewOf(buffer);
67      int expMark = (int) ZipInputStream.CENSIG;
68      int expSize = ZipInputStream.CENHDR + filenameLength + extraLength + commentLength;
69      int expPos = 0;
70      assertEquals("not based at current position", expMark, view.get(CENSIG));
71      assertEquals("Not slice with position 0", expPos, view.buffer.position());
72      assertEquals("Not sized with comment", expSize, view.getSize());
73      assertEquals("Not limited to size", expSize, view.buffer.limit());
74    }

76    /**
77     * Test of view method, of class DirectoryEntry.
78     */
79    @Test
80    public void testView_3Args() {
81      String filename = "pkg/foo.class";
82      String comment = "got milk";
83      byte[] extraData = { 1, 2, 3, 4, 5, 6, 7, 8};
84      int expSize = ZipInputStream.CENHDR + filename.getBytes(UTF_8).length
85          + extraData.length + comment.getBytes(UTF_8).length;
86      int expPos = 0;
87      DirectoryEntry view = DirectoryEntry.allocate(filename, extraData, comment);
88      assertEquals("Incorrect filename", filename, view.getFilename());
89      Assert.assertArrayEquals("Incorrect extra data", extraData, view.getExtraData());
90      assertEquals("Incorrect comment", comment, view.getComment());
91      assertEquals("Not at position 0", expPos, view.buffer.position());
92      assertEquals("Not sized correctly", expSize, view.getSize());
93      assertEquals("Not limited to size", expSize, view.buffer.limit());
94    }

99    @Test
100   public void testView_4Args() {
101     ByteBuffer buffer = ByteBuffer.allocate(100).order(ByteOrder.LITTLE_ENDIAN);

107     String filename = "pkg/foo.class";
108     byte[] extraData = { 1, 2, 3, 4, 5};
109     String comment = "c";
110     int expMark = (int) ZipInputStream.CENSIG;
111     int expSize = 46 + filename.getBytes(UTF_8).length + extraData.length

113     int expPos = 0;
114     DirectoryEntry view = DirectoryEntry.view(buffer, filename, extraData, comment);
115     assertEquals("not based at current position", expMark, view.get(CENSIG));
116     assertEquals("Not slice with position 0", expPos, view.buffer.position());
117     assertEquals("Not sized with filename", expSize, view.getSize());
118     assertEquals("Not limited to size", expSize, view.buffer.limit());
119     assertEquals("Incorrect filename", filename, view.getFilename());
120     Assert.assertArrayEquals("Incorrect extra data", extraData, view.getExtraData());
121     assertEquals("Incorrect comment", comment, view.getComment());
122   }

124   /**
125    * Test of copy method, of class DirectoryEntry.
126    */
127   @Test
128   public void testCopy() {
129     String filename = "pkg/foo.class";
130     byte[] extraData = {};
131     String comment = "always comment!";
132     ByteBuffer buffer = ByteBuffer.allocate(100).order(ByteOrder.LITTLE_ENDIAN);
133     DirectoryEntry view = DirectoryEntry.allocate(filename, extraData, comment);
134     view.copy(buffer);
135     int expSize = view.getSize();
136     assertEquals("buffer not advanced as expected", expSize, buffer.position());
137     buffer.position(0);
138     DirectoryEntry clone = DirectoryEntry.viewOf(buffer);
139     assertEquals("Fail to copy mark", view.get(CENSIG), clone.get(CENSIG));
140     assertEquals("Fail to copy comment", view.getFilename(), clone.getFilename());
141     Assert.assertArrayEquals("Fail to copy comment", view.getExtraData(), clone.getExtraData());
142     assertEquals("Fail to copy comment", view.getComment(), clone.getComment());
143   }

145   /**
146    * Test of with and get methods.
147    */
148   @Test
149   public void testWithAndGetMethods() {
150     int crc = 0x12345678;
151     int compressed = 0x357f1d5;
152     int uncompressed = 0x74813159;
153     short flags = 0x7a61;
154     short method = 0x3b29;
155     int time = 0x12312345;
156     short version = 0x1234;
157     short versionMadeBy = 0x27a1;
158     short disk = 0x5a78;
159     int extAttr = 0x73b27a15;
160     short intAttr = 0x37cc;
161     int offset = 0x74c93ac1;
162     DirectoryEntry view = DirectoryEntry.allocate("pkg/foo.class", null, "")
163         .set(CENCRC, crc)
164         .set(CENSIZ, compressed)
165         .set(CENLEN, uncompressed)
166         .set(CENFLG, flags)
167         .set(CENHOW, method)
168         .set(CENTIM, time)
169         .set(CENVER, version)
170         .set(CENVEM, versionMadeBy)
171         .set(CENDSK, disk)
172         .set(CENATX, extAttr)
173         .set(CENATT, intAttr)
174         .set(CENOFF, offset);
175     assertEquals("CRC", crc, view.get(CENCRC));
176     assertEquals("Compressed size", compressed, view.get(CENSIZ));
177     assertEquals("Uncompressed size", uncompressed, view.get(CENLEN));
178     assertEquals("Flags", flags, view.get(CENFLG));
179     assertEquals("Method", method, view.get(CENHOW));
180     assertEquals("Modified time", time, view.get(CENTIM));
181     assertEquals("Version needed", version, view.get(CENVER));
182     assertEquals("Version made by", versionMadeBy, view.get(CENVEM));
183     assertEquals("Disk", disk, view.get(CENDSK));
184     assertEquals("External attributes", extAttr, view.get(CENATX));
185     assertEquals("Internal attributes", intAttr, view.get(CENATT));
186     assertEquals("Offset", offset, view.get(CENOFF));
187   }
