46      {
47        Label l = Label.parseAbsolute("//foo/bar:baz");
48        assertEquals("foo/bar", l.getPackageName());
49        assertEquals("baz", l.getName());
50      }
51      {
52        Label l = Label.parseAbsolute("//foo/bar");
53        assertEquals("foo/bar", l.getPackageName());
54        assertEquals("bar", l.getName());
55      }
56      {
57        Label l = Label.parseAbsolute("//:bar");
58        assertEquals("", l.getPackageName());
59        assertEquals("bar", l.getName());
60      }

63    private static String parseCommandLine(String label, String prefix) throws LabelSyntaxException {
64      return Label.parseCommandLineLabel(label, new PathFragment(prefix)).toString();
65    }

67    @Test
68    public void testLabelResolution() throws Exception {
69      assertEquals("//absolute:label", parseCommandLine("//absolute:label", ""));
70      assertEquals("//absolute:label", parseCommandLine("//absolute:label", "absolute"));
71      assertEquals("//absolute:label", parseCommandLine(":label", "absolute"));
72      assertEquals("//absolute:label", parseCommandLine("label", "absolute"));
73      assertEquals("//absolute:label", parseCommandLine("absolute:label", ""));
74      assertEquals("//absolute/path:label", parseCommandLine("path:label", "absolute"));
75      assertEquals("//absolute/path:label/path", parseCommandLine("path:label/path", "absolute"));
76      assertEquals("//absolute:label/path", parseCommandLine("label/path", "absolute"));
77    }

181   /**
182    * Asserts that creating a label throws a SyntaxException.
183    * @param label the label to create.
184    */
185   private static void assertSyntaxError(String expectedError, String label) {
186     try {
187       Label.parseAbsolute(label);
188       fail("Label '" + label + "' did not contain a syntax error");
189     } catch (LabelSyntaxException e) {
190       assertContainsRegex(Pattern.quote(expectedError), e.getMessage());
191     }
192   }

194   @Test
195   public void testBadCharacters() throws Exception {
196     assertSyntaxError("package names may contain only",
197                       "//foo/bar baz");
198     assertSyntaxError("target names may not contain ':'",
199                       "//foo:bar:baz");
200     assertSyntaxError("target names may not contain ':'",
201                       "//foo:bar:");
202     assertSyntaxError("target names may not contain ':'",
203                       "//foo/bar::");
204     assertSyntaxError("target names may not contain '&'",
205                       "//foo:bar&");
206     assertSyntaxError("target names may not contain '$'",
207                       "//foo/bar:baz$a");
208     assertSyntaxError("target names may not contain '('",
209                       "//foo/bar:baz(foo)");
210     assertSyntaxError("target names may not contain ')'",
211                       "//foo/bar:bazfoo)");
212   }

214   @Test
215   public void testUplevelReferences() throws Exception {
216     assertSyntaxError(BAD_PACKAGE_CHARS,
217                       "//foo/bar/..:baz");
218     assertSyntaxError(BAD_PACKAGE_CHARS,
219                       "//foo/../baz:baz");
220     assertSyntaxError(BAD_PACKAGE_CHARS,
221                       "//../bar/baz:baz");
222     assertSyntaxError(BAD_PACKAGE_CHARS,
223                       "//..:foo");
224     assertSyntaxError(TARGET_UPLEVEL,
225                       "//foo:bar/../baz");
226     assertSyntaxError(TARGET_UPLEVEL,
227                       "//foo:../bar/baz");
228     assertSyntaxError(TARGET_UPLEVEL,
229                       "//foo:bar/baz/..");
230     assertSyntaxError(TARGET_UPLEVEL,
231                       "//foo:..");
232   }

234   @Test
235   public void testDotAsAPathSegment() throws Exception {
236     assertSyntaxError("package names may contain only A-Z, a-z, 0-9, '/', '-' and '_'",
237                       "//foo/bar/.:baz");
238     assertSyntaxError(BAD_PACKAGE_CHARS,
239                       "//foo/./baz:baz");
240     assertSyntaxError(BAD_PACKAGE_CHARS,
241                       "//./bar/baz:baz");
242     assertSyntaxError("target names may not contain '.' as a path segment",
243                       "//foo:bar/./baz");
244     assertSyntaxError("target names may not contain '.' as a path segment",
245                       "//foo:./bar/baz");
246     // TODO(bazel-team): enable when we have removed the "Workaround" in Label

335   private void checkSerialization(String labelString, int expectedSize) throws Exception {
336     Label a = Label.parseAbsolute(labelString);
337     byte[] sa = TestUtils.serializeObject(a);
338     assertEquals(expectedSize, sa.length);
339 
340     Label a2 = (Label) TestUtils.deserializeObject(sa);
341     assertEquals(a, a2);
342   }
