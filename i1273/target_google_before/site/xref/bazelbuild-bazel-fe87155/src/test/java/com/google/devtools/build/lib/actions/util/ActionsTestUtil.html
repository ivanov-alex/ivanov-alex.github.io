109   /**
110    * {@link SkyFunction.Environment} that internally makes a full Skyframe evaluate call for the
111    * requested keys, blocking until the values are ready.
112    */
113   private static class BlockingSkyFunctionEnvironment extends AbstractSkyFunctionEnvironment {
114     private final BuildDriver driver;
115     private final EventHandler eventHandler;
116 
117     private BlockingSkyFunctionEnvironment(BuildDriver driver, EventHandler eventHandler) {
118       this.driver = driver;
119       this.eventHandler = eventHandler;
120     }
121 
122     @Override
123     protected Map<SkyKey, ValueOrUntypedException> getValueOrUntypedExceptions(
124         Set<SkyKey> depKeys) {
125       EvaluationResult<SkyValue> evaluationResult;
126       Map<SkyKey, ValueOrUntypedException> result = Maps.newHashMapWithExpectedSize(depKeys.size());
127       try {
128         evaluationResult = driver.evaluate(depKeys, /*keepGoing=*/false,
129             ResourceUsage.getAvailableProcessors(), eventHandler);
130       } catch (InterruptedException e) {
131         Thread.currentThread().interrupt();
132         for (SkyKey key : depKeys) {
133           result.put(key, ValueOrExceptionUtils.ofNull());
134         }
135         return result;
136       }
137       for (SkyKey key : depKeys) {
138         SkyValue value = evaluationResult.get(key);
139         if (value != null) {
140           result.put(key, ValueOrExceptionUtils.ofValue(value));
141           continue;
142         }
143         ErrorInfo errorInfo = evaluationResult.getError(key);
144         if (errorInfo == null || errorInfo.getException() == null) {
145           result.put(key, ValueOrExceptionUtils.ofNull());
146           continue;
147         }
148         result.put(key, ValueOrExceptionUtils.ofExn(errorInfo.getException()));
149       }
150       return result;
151     }
152 
153     @Override
154     public EventHandler getListener() {
155       return null;
156     }
157 
158     @Override
159     public boolean inErrorBubblingForTesting() {
160       return false;
161     }
162   }

240   /**
241    * For a bunch of actions, gets the basenames of the paths and accumulates
242    * them in a space separated string, like <code>foo.o bar.o baz.a</code>.
243    */
244   public static String baseNamesOf(Iterable<Artifact> artifacts) {
245     List<String> baseNames = baseArtifactNames(artifacts);
246     return Joiner.on(' ').join(baseNames);
247   }

249   /**
250    * For a bunch of actions, gets the basenames of the paths, sorts them in alphabetical
251    * order and accumulates them in a space separated string, for example
252    * <code>bar.o baz.a foo.o</code>.
253    */
254   public static String sortedBaseNamesOf(Iterable<Artifact> artifacts) {
255     List<String> baseNames = baseArtifactNames(artifacts);
256     Collections.sort(baseNames);
257     return Joiner.on(' ').join(baseNames);
258   }

260   /**
261    * For a bunch of artifacts, gets the basenames and accumulates them in a

265     List<String> baseNames = new ArrayList<>();

367   /**
368    * Returns the closure over the input files of a set of artifacts.
369    */
370   public Set<Artifact> artifactClosureOf(Iterable<Artifact> artifacts) {
371     Set<Artifact> visited = new LinkedHashSet<>();
372     List<Artifact> toVisit = Lists.newArrayList(artifacts);
373     while (!toVisit.isEmpty()) {
374       Artifact current = toVisit.remove(0);
375       if (!visited.add(current)) {
376         continue;
377       }
378       Action generatingAction = actionGraph.getGeneratingAction(current);
379       if (generatingAction != null) {
380         Iterables.addAll(toVisit, generatingAction.getInputs());
381       }
382     }
383     return visited;
384   }

406   /**
407    * Finds all the actions that are instances of <code>actionClass</code>
408    * in the transitive closure of prerequisites.
409    */
410   public <A extends Action> List<A> findTransitivePrerequisitesOf(Artifact artifact,
411       Class<A> actionClass, Predicate<Artifact> allowedArtifacts) {
412     List<A> actions = new ArrayList<>();
413     Set<Artifact> visited = new LinkedHashSet<>();
414     List<Artifact> toVisit = new LinkedList<>();
415     toVisit.add(artifact);
416     while (!toVisit.isEmpty()) {
417       Artifact current = toVisit.remove(0);
418       if (!visited.add(current)) {
419         continue;
420       }
421       Action generatingAction = actionGraph.getGeneratingAction(current);
422       if (generatingAction != null) {
423         Iterables.addAll(toVisit, Iterables.filter(generatingAction.getInputs(), allowedArtifacts));
424         if (actionClass.isInstance(generatingAction)) {
425           actions.add(actionClass.cast(generatingAction));
426         }
427       }
428     }
429     return actions;
430   }
431 
432   public <A extends Action> List<A> findTransitivePrerequisitesOf(
433       Artifact artifact, Class<A> actionClass) {
434     return findTransitivePrerequisitesOf(artifact, actionClass, Predicates.<Artifact>alwaysTrue());
435   }

446   /**
447    * Looks in the given artifacts Iterable for the first Artifact whose path ends with the given
448    * suffix and returns the Artifact.
449    */
450   public static Artifact getFirstArtifactEndingWith(
451       Iterable<Artifact> artifacts, String suffix) {
452     for (Artifact a : artifacts) {
453       if (a.getExecPath().getPathString().endsWith(suffix)) {
454         return a;
455       }
456     }
457     return null;
458   }

460   /**
461    * Returns the first artifact which is an input to "action" and has the
462    * specified basename. An assertion error is raised if none is found.
463    */
464   public static Artifact getInput(Action action, String basename) {
465     for (Artifact artifact : action.getInputs()) {
466       if (artifact.getExecPath().getBaseName().equals(basename)) {
467         return artifact;
468       }
469     }
470     throw new AssertionError("No input with basename '" + basename + "' in action " + action);
471   }

473   /**
474    * Returns true if an artifact that is an input to "action" with the specific
475    * basename exists.
476    */
477   public static boolean hasInput(Action action, String basename) {
478     try {
479       getInput(action, basename);
480       return true;
481     } catch (AssertionError e) {
482       return false;
483     }
484   }

486   /**
487    * Assert that an artifact is the primary output of its generating action.
488    */
489   public void assertPrimaryInputAndOutputArtifacts(Artifact input, Artifact output) {
490     Action generatingAction = actionGraph.getGeneratingAction(output);
491     assertThat(generatingAction).isNotNull();
492     assertThat(generatingAction.getPrimaryOutput()).isEqualTo(output);
493     assertThat(generatingAction.getPrimaryInput()).isEqualTo(input);
494   }
