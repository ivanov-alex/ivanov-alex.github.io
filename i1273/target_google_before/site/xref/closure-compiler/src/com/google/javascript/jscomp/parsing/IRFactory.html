265   private IRFactory(String sourceString,
266                     StaticSourceFile sourceFile,
267                     Config config,
268                     ErrorReporter errorReporter,
269                     ImmutableList<Comment> comments) {
270     this.sourceString = sourceString;
271     this.nextCommentIter = comments.iterator();
272     this.currentComment = nextCommentIter.hasNext() ? nextCommentIter.next() : null;
273     this.newlines = new ArrayList<>();
274     this.sourceFile = sourceFile;
275     this.fileLevelJsDocBuilder = new JSDocInfoBuilder(
276         config.parseJsDocDocumentation);
277 
278     // Pre-generate all the newlines in the file.
279     for (int charNo = 0; true; charNo++) {
280       charNo = sourceString.indexOf('\n', charNo);
281       if (charNo == -1) {
282         break;
283       }
284       newlines.add(Integer.valueOf(charNo));
285     }
286 
287     // Sometimes this will be null in tests.
288     this.sourceName = sourceFile == null ? null : sourceFile.getName();
289 
290     this.config = config;
291     this.errorReporter = errorReporter;
292     this.transformDispatcher = new TransformDispatcher();
293     // The template node properties are applied to all nodes in this transform.
294     this.templateNode = createTemplateNode();
295 
296     switch (config.languageMode) {
297       case ECMASCRIPT3:
298         reservedKeywords = null; // use TokenStream.isKeyword instead
299         break;
300       case ECMASCRIPT5:
301       case ECMASCRIPT6:
302         reservedKeywords = ES5_RESERVED_KEYWORDS;
303         break;
304       case ECMASCRIPT5_STRICT:
305       case ECMASCRIPT6_STRICT:
306       case ECMASCRIPT6_TYPED:
307         reservedKeywords = ES5_STRICT_RESERVED_KEYWORDS;
308         break;
309       default:
310         throw new IllegalStateException("unknown language mode: " + config.languageMode);
311     }
312   }

324   public static Node transformTree(ProgramTree tree,
325                                    StaticSourceFile sourceFile,
326                                    String sourceString,
327                                    Config config,
328                                    ErrorReporter errorReporter) {
329     IRFactory irFactory = new IRFactory(sourceString, sourceFile,
330         config, errorReporter, tree.sourceComments);
331 
332     // don't call transform as we don't want standard jsdoc handling.
333     Node n = irFactory.justTransform(tree);
334     irFactory.setSourceInfo(n, tree);
335 
336     if (tree.sourceComments != null) {
337       for (Comment comment : tree.sourceComments) {
338         if (comment.type == Comment.Type.JSDOC &&
339             !irFactory.parsedComments.contains(comment)) {
340           irFactory.handlePossibleFileOverviewJsDoc(comment);
341         } else if (comment.type == Comment.Type.BLOCK) {
342           irFactory.handleBlockComment(comment);
343         }
344       }
345     }
346 
347     irFactory.setFileOverviewJsDoc(n);
348 
349     irFactory.validateAll(n);
350 
351     return n;
352   }

368   private void validateBreakContinue(Node n) {
369     if (n.isBreak() || n.isContinue()) {
370       Node labelName = n.getFirstChild();
371       if (labelName != null) {
372         Node parent = n.getParent();
373         while (!parent.isLabel() || !labelsMatch(parent, labelName)) {
374           if (parent.isFunction() || parent.isScript()) {
375             // report missing label
376             errorReporter.error(
377                 SimpleFormat.format(UNDEFINED_LABEL, labelName.getString()),
378                 sourceName,
379                 n.getLineno(), n.getCharno());
380             break;
381           }
382           parent = parent.getParent();
383         }
384         if (parent.isLabel() && labelsMatch(parent, labelName)) {
385           if (n.isContinue() && !isContinueTarget(parent.getLastChild())) {
386             // report invalid continue target
387             errorReporter.error(
388                 UNEXPECTED_LABLED_CONTINUE,
389                 sourceName,
390                 n.getLineno(), n.getCharno());
391           }
392         }
393       } else {
394         if (n.isContinue()) {
395           Node parent = n.getParent();
396           while (!isContinueTarget(parent)) {
397             if (parent.isFunction() || parent.isScript()) {
398               // report invalid continue
399               errorReporter.error(
400                   UNEXPECTED_CONTINUE,
401                   sourceName,
402                   n.getLineno(), n.getCharno());
403               break;
404             }
405             parent = parent.getParent();
406           }
407         } else {
408           Node parent = n.getParent();
409           while (!isBreakTarget(parent)) {
410             if (parent.isFunction() || parent.isScript()) {
411               // report invalid break
412               errorReporter.error(
413                   UNLABELED_BREAK,
414                   sourceName,
415                   n.getLineno(), n.getCharno());
416               break;
417             }
418             parent = parent.getParent();
419           }
420         }
421       }
422     }
423   }

425   private void validateReturn(Node n) {
426     if (n.isReturn()) {
427       Node parent = n;
428       while ((parent = parent.getParent()) != null) {
429         if (parent.isFunction()) {
430           return;
431         }
432       }
433       errorReporter.error(UNEXPECTED_RETURN,
434           sourceName, n.getLineno(), n.getCharno());
435     }
436   }

466   private void validateParameters(Node n) {
467     if (n.isParamList()) {
468       Node c = n.getFirstChild();
469       for (; c != null; c = c.getNext()) {
470         if (!c.isName()) {
471           continue;
472         }
473         Node sibling = c.getNext();
474         for (; sibling != null; sibling = sibling.getNext()) {
475           if (sibling.isName() && c.getString().equals(sibling.getString())) {
476             errorReporter.warning(
477                 SimpleFormat.format(DUPLICATE_PARAMETER, c.getString()),
478                 sourceName,
479                 n.getLineno(), n.getCharno());
480           }
481         }
482       }
483     }
484   }

505   private void validateLabel(Node n) {
506     if (n.isLabel()) {
507       Node labelName = n.getFirstChild();
508       for (Node parent = n.getParent();
509            parent != null && !parent.isFunction(); parent = parent.getParent()) {
510         if (parent.isLabel() && labelsMatch(parent, labelName)) {
511           errorReporter.error(
512               SimpleFormat.format(DUPLICATE_LABEL, labelName.getString()),
513               sourceName,
514               n.getLineno(), n.getCharno());
515           break;
516         }
517       }
518     }
519   }

521   private void setFileOverviewJsDoc(Node irNode) {
522     // Only after we've seen all @fileoverview entries, attach the
523     // last one to the root node, and copy the found license strings
524     // to that node.
525     JSDocInfo rootNodeJsDoc = fileLevelJsDocBuilder.build();
526     if (rootNodeJsDoc != null) {
527       irNode.setJSDocInfo(rootNodeJsDoc);
528     }
529 
530     if (fileOverviewInfo != null) {
531       if ((irNode.getJSDocInfo() != null) &&
532           (irNode.getJSDocInfo().getLicense() != null)) {
533         JSDocInfoBuilder builder = JSDocInfoBuilder.copyFrom(fileOverviewInfo);
534         builder.recordLicense(irNode.getJSDocInfo().getLicense());
535         fileOverviewInfo = builder.build();
536       }
537       irNode.setJSDocInfo(fileOverviewInfo);
538     }
539   }

541   private Node transformBlock(ParseTree node) {
542     Node irNode = transform(node);
543     if (!irNode.isBlock()) {
544       if (irNode.isEmpty()) {
545         irNode.setType(Token.BLOCK);
546       } else {
547         Node newBlock = newNode(Token.BLOCK, irNode);
548         setSourceInfo(newBlock, irNode);
549         irNode = newBlock;
550       }
551       irNode.setIsAddedBlock(true);
552     }
553     return irNode;
554   }

556   /**
557    * Check to see if the given block comment looks like it should be JSDoc.
558    */
559   private void handleBlockComment(Comment comment) {
560     Pattern p = Pattern.compile("(/|(\n[ \t]*))\\*[ \t]*@[a-zA-Z]+[ \t\n{]");
561     if (p.matcher(comment.value).find()) {
562       errorReporter.warning(
563           SUSPICIOUS_COMMENT_WARNING,
564           sourceName,
565           lineno(comment.location.start),
566           charno(comment.location.start));
567     }

636   private boolean shouldAttachJSDocHere(ParseTree tree) {
637     switch (tree.type) {
638       case EXPRESSION_STATEMENT:
639       case LABELLED_STATEMENT:
640         return false;
641       case CALL_EXPRESSION:
642       case CONDITIONAL_EXPRESSION:
643       case BINARY_OPERATOR:
644       case MEMBER_EXPRESSION:
645       case MEMBER_LOOKUP_EXPRESSION:
646       case POSTFIX_EXPRESSION:
647         ParseTree nearest = findNearestNode(tree);
648         if (nearest.type == ParseTreeType.PAREN_EXPRESSION) {
649           return false;
650         }
651         return true;
652       default:
653         return true;
654     }
655   }

657   private static ParseTree findNearestNode(ParseTree tree) {
658     while (true) {
659       switch (tree.type) {
660         case EXPRESSION_STATEMENT:
661           tree = tree.asExpressionStatement().expression;
662           continue;
663         case CALL_EXPRESSION:
664           tree = tree.asCallExpression().operand;
665           continue;
666         case BINARY_OPERATOR:
667           tree = tree.asBinaryOperator().left;
668           continue;
669         case CONDITIONAL_EXPRESSION:
670           tree = tree.asConditionalExpression().condition;
671           continue;
672         case MEMBER_EXPRESSION:
673           tree = tree.asMemberExpression().operand;
674           continue;
675         case MEMBER_LOOKUP_EXPRESSION:
676           tree = tree.asMemberLookupExpression().operand;
677           continue;
678         case POSTFIX_EXPRESSION:
679           tree = tree.asPostfixExpression().operand;
680           continue;
681         default:
682           return tree;
683       }
684     }
685   }

1151     Node processFunction(FunctionDeclarationTree functionTree) {
1152       boolean isDeclaration = (functionTree.kind == FunctionDeclarationTree.Kind.DECLARATION);
1153       boolean isMember = (functionTree.kind == FunctionDeclarationTree.Kind.MEMBER);
1154       boolean isArrow = (functionTree.kind == FunctionDeclarationTree.Kind.ARROW);
1155       boolean isGenerator = functionTree.isGenerator;
1156       boolean isSignature = (functionTree.functionBody.type == ParseTreeType.EMPTY_STATEMENT);
1157 
1158       if (!isEs6Mode()) {
1159         if (isGenerator) {
1160           maybeWarnEs6Feature(functionTree, "generators");
1161         }
1162 
1163         if (isMember) {
1164           maybeWarnEs6Feature(functionTree, "member declarations");
1165         }
1166 
1167         if (isArrow) {
1168           maybeWarnEs6Feature(functionTree, "short function syntax");
1169         }
1170       }
1171 
1172       IdentifierToken name = functionTree.name;
1173       Node newName;
1174       if (name != null) {
1175         newName = processNameWithInlineJSDoc(name);
1176       } else {
1177         if (isDeclaration || isMember) {
1178           errorReporter.error(
1179             "unnamed function statement",
1180             sourceName,
1181             lineno(functionTree), charno(functionTree));
1182 
1183           // Return the bare minimum to put the AST in a valid state.
1184           newName = createMissingNameNode();
1185         } else {
1186           newName = newStringNode(Token.NAME, "");
1187         }
1188 
1189         // Old Rhino tagged the empty name node with the line number of the
1190         // declaration.
1191         setSourceInfo(newName, functionTree);
1192       }
1193 
1194       Node node = newNode(Token.FUNCTION);
1195       if (!isMember) {
1196         node.addChildToBack(newName);
1197       } else {
1198         Node emptyName = newStringNode(Token.NAME, "");
1199         setSourceInfo(emptyName, functionTree);
1200         node.addChildToBack(emptyName);
1201       }
1202 
1203       maybeProcessGenerics(node.getFirstChild(), functionTree.generics);
1204       node.addChildToBack(transform(functionTree.formalParameterList));
1205       maybeProcessType(node, functionTree.returnType);
1206 
1207       Node bodyNode = transform(functionTree.functionBody);
1208       if (!isArrow && !isSignature && !bodyNode.isBlock()) {
1209         // When in ideMode the parser tries to parse some constructs the
1210         // compiler doesn't support, repair it here.
1211         Preconditions.checkState(config.isIdeMode);
1212         bodyNode = IR.block();
1213       }
1214       parseDirectives(bodyNode);
1215       node.addChildToBack(bodyNode);
1216 
1217       node.setIsGeneratorFunction(isGenerator);
1218       node.setIsArrowFunction(isArrow);
1219       node.putBooleanProp(Node.OPT_ES6_TYPED, functionTree.isOptional);
1220 
1221       Node result;
1222 
1223       if (functionTree.kind == FunctionDeclarationTree.Kind.MEMBER) {
1224         setSourceInfo(node, functionTree);
1225         Node member = newStringNode(Token.MEMBER_FUNCTION_DEF, name.value);
1226         member.addChildToBack(node);
1227         member.setStaticMember(functionTree.isStatic);
1228         maybeProcessAccessibilityModifier(member, functionTree, functionTree.access);
1229         node.setDeclaredTypeExpression(node.getDeclaredTypeExpression());
1230         result = member;
1231       } else {
1232         result = node;
1233       }
1234 
1235       return result;
1236     }

1613     Node processStringLiteral(LiteralExpressionTree literalTree) {
1614       LiteralToken token = literalTree.literalToken.asLiteral();
1615 
1616       Node n = processString(token);
1617       String value = n.getString();
1618       if (value.indexOf('\u000B') != -1) {
1619         // NOTE(nicksantos): In JavaScript, there are 3 ways to
1620         // represent a vertical tab: \v, \x0B, \u000B.
1621         // The \v notation was added later, and is not understood
1622         // on IE. So we need to preserve it as-is. This is really
1623         // obnoxious, because we do not have a good way to represent
1624         // how the original string was encoded without making the
1625         // representation of strings much more complicated.
1626         //
1627         // To handle this, we look at the original source test, and
1628         // mark the string as \v-encoded or not. If a string is
1629         // \v encoded, then all the vertical tabs in that string
1630         // will be encoded with a \v.
1631         int start = token.location.start.offset;
1632         int end = token.location.end.offset;
1633         if (start < sourceString.length() &&
1634             (sourceString.substring(
1635                 start, Math.min(sourceString.length(), end)).contains("\\v"))) {
1636           n.putBooleanProp(Node.SLASH_V, true);
1637         }
1638       }
1639       return n;
1640     }

2220     private boolean checkParameters(ImmutableList<ParseTree> params) {
2221       boolean seenOptional = false;

2223       for (int i = 0; i < params.size(); i++) {
2224         ParseTree param = params.get(i);
2225         Node type = null;
2226         if (param.type == ParseTreeType.TYPED_PARAMETER) {
2227           TypedParameterTree typedParam = param.asTypedParameter();
2228           type = transform(typedParam.typeAnnotation);
2229           param = typedParam.param;
2230         }
2231         switch (param.type) {
2232           case IDENTIFIER_EXPRESSION:
2233             if (seenOptional) {
2234               errorReporter.error(
2235                   "A required parameter cannot follow an optional parameter.",
2236                   sourceName,
2237                   lineno(param),
2238                   charno(param));
2239               good = false;
2240             }
2241             break;
2242           case OPTIONAL_PARAMETER:
2243             seenOptional = true;
2244             break;
2245           case REST_PARAMETER:
2246             if (i != params.size() - 1) {
2247               errorReporter.error(
2248                   "A rest parameter must be last in a parameter list.",
2249                   sourceName,
2250                   lineno(param),
2251                   charno(param));
2252               good = false;
2253             }
2254             if (type != null && type.getType() != Token.ARRAY_TYPE) {
2255               errorReporter.error(
2256                   "A rest parameter must be of an array type.",
2257                   sourceName,
2258                   lineno(param),
2259                   charno(param));
2260               good = false;
2261             }
2262             break;
2263           default:
2264         }
2265       }
2266       return good;

2635   private String normalizeString(LiteralToken token, boolean templateLiteral) {
2636     String value = token.value;
2637     if (templateLiteral) {
2638       // <CR><LF> and <CR> are normalized as <LF> for raw string value
2639       value = value.replaceAll("(?<!\\\\)\r(\n)?", "\n");
2640     }
2641     int start = templateLiteral ? 0 : 1; // skip the leading quote
2642     int cur = value.indexOf('\\');
2643     if (cur == -1) {
2644       // short circuit no escapes.
2645       return templateLiteral ? value : value.substring(1, value.length() - 1);
2646     }
2647     StringBuilder result = new StringBuilder();
2648     while (cur != -1) {
2649       if (cur - start > 0) {
2650         result.append(value, start, cur);
2651       }
2652       cur += 1; // skip the escape char.
2653       char c = value.charAt(cur);
2654       switch (c) {
2655         case '\'':
2656         case '"':
2657         case '\\':
2658           result.append(c);
2659           break;
2660         case 'b':
2661           result.append('\b');
2662           break;
2663         case 'f':
2664           result.append('\f');
2665           break;
2666         case 'n':
2667           result.append('\n');
2668           break;
2669         case 'r':
2670           result.append('\r');
2671           break;
2672         case 't':
2673           result.append('\t');
2674           break;
2675         case 'v':
2676           result.append('\u000B');
2677           break;
2678         case '\n':
2679           if (isEs5OrBetterMode()) {
2680             errorReporter.warning(STRING_CONTINUATION_WARNING,
2681                 sourceName,
2682                 lineno(token.location.start), charno(token.location.start));
2683           } else {
2684             errorReporter.error(STRING_CONTINUATION_ERROR,
2685                 sourceName,
2686                 lineno(token.location.start), charno(token.location.start));
2687           }
2688           // line continuation, skip the line break
2689           break;
2690         case '0': case '1': case '2': case '3': case '4': case '5': case '6': case '7':
2691           char next1 = value.charAt(cur + 1);
2692 
2693           if (inStrictContext()) {
2694             if (c == '0' && !isOctalDigit(next1)) {
2695               // No warning: "\0" followed by a character which is not an octal digit
2696               // is allowed in strict mode.
2697             } else {
2698               errorReporter.warning(OCTAL_STRING_LITERAL_WARNING,
2699                   sourceName,
2700                   lineno(token.location.start), charno(token.location.start));
2701             }
2702           }
2703 
2704           if (!isOctalDigit(next1)) {
2705             result.append((char) octaldigit(c));
2706           } else {
2707             char next2 = value.charAt(cur + 2);
2708             if (!isOctalDigit(next2)) {
2709               result.append((char) (8 * octaldigit(c) + octaldigit(next1)));
2710               cur += 1;
2711             } else {
2712               result.append((char)
2713                   (8 * 8 * octaldigit(c) + 8 * octaldigit(next1) + octaldigit(next2)));
2714               cur += 2;
2715             }
2716           }
2717 
2718           break;
2719         case 'x':
2720           result.append((char) (
2721               hexdigit(value.charAt(cur + 1)) * 0x10
2722               + hexdigit(value.charAt(cur + 2))));
2723           cur += 2;
2724           break;
2725         case 'u':
2726           int escapeEnd;
2727           String hexDigits;
2728           if (value.charAt(cur + 1) != '{') {
2729             // Simple escape with exactly four hex digits: \\uXXXX
2730             escapeEnd = cur + 5;
2731             hexDigits = value.substring(cur + 1, escapeEnd);
2732           } else {
2733             // Escape with braces can have any number of hex digits: \\u{XXXXXXX}
2734             escapeEnd = cur + 2;
2735             while (Character.digit(value.charAt(escapeEnd), 0x10) >= 0) {
2736               escapeEnd++;
2737             }
2738             hexDigits = value.substring(cur + 2, escapeEnd);
2739             escapeEnd++;
2740           }
2741           result.append(Character.toChars(Integer.parseInt(hexDigits, 0x10)));
2742           cur = escapeEnd - 1;
2743           break;
2744         default:
2745           // TODO(tbreisacher): Add a warning because the user probably
2746           // intended to type an escape sequence.
2747           result.append(c);
2748           break;
2749       }
2750       start = cur + 1;
2751       cur = value.indexOf('\\', start);
2752     }
2753     // skip the trailing quote.
2754     result.append(value, start, templateLiteral ? value.length() : value.length() - 1);
2755 
2756     return result.toString();
2757   }

2777   double normalizeNumber(LiteralToken token) {
2778     String value = token.value;
2779     SourceRange location = token.location;
2780     int length = value.length();
2781     Preconditions.checkState(length > 0);
2782     Preconditions.checkState(value.charAt(0) != '-'
2783         && value.charAt(0) != '+');
2784     if (value.charAt(0) == '.') {
2785       return Double.valueOf('0' + value);
2786     } else if (value.charAt(0) == '0' && length > 1) {
2787       // TODO(johnlenz): accept octal numbers in es3 etc.
2788       switch (value.charAt(1)) {
2789         case '.':
2790         case 'e':
2791         case 'E':
2792           return Double.valueOf(value);
2793         case 'b':
2794         case 'B': {
2795           if (!isEs6Mode()) {
2796             errorReporter.warning(BINARY_NUMBER_LITERAL_WARNING,
2797                 sourceName,
2798                 lineno(token.location.start), charno(token.location.start));
2799           }
2800           double v = 0;
2801           int c = 1;
2802           while (++c < length) {
2803             v = (v * 2) + binarydigit(value.charAt(c));
2804           }
2805           return v;
2806         }
2807         case 'o':
2808         case 'O': {
2809           if (!isEs6Mode()) {
2810             errorReporter.warning(OCTAL_NUMBER_LITERAL_WARNING,
2811                 sourceName,
2812                 lineno(token.location.start), charno(token.location.start));
2813           }
2814           double v = 0;
2815           int c = 1;
2816           while (++c < length) {
2817             v = (v * 8) + octaldigit(value.charAt(c));
2818           }
2819           return v;
2820         }
2821         case 'x':
2822         case 'X': {
2823           double v = 0;
2824           int c = 1;
2825           while (++c < length) {
2826             v = (v * 0x10) + hexdigit(value.charAt(c));
2827           }
2828           return v;
2829         }
2830         case '0': case '1': case '2': case '3':
2831         case '4': case '5': case '6': case '7':
2832           errorReporter.warning(INVALID_ES5_STRICT_OCTAL, sourceName,
2833               lineno(location.start), charno(location.start));
2834           if (!inStrictContext()) {
2835             double v = 0;
2836             int c = 0;
2837             while (++c < length) {
2838               v = (v * 8) + octaldigit(value.charAt(c));
2839             }
2840             return v;
2841           } else {
2842             return Double.valueOf(value);
2843           }
2844         default:
2845           errorReporter.error(INVALID_NUMBER_LITERAL, sourceName,
2846               lineno(location.start), charno(location.start));
2847           return 0;
2848       }
2849     } else {
2850       return Double.valueOf(value);
2851     }
2852   }
