42  /**
43   * Process aliases in goog.scope blocks.
44   *
45   * <pre>
46   * goog.scope(function() {
47   *   var dom = goog.dom;
48   *   var DIV = dom.TagName.DIV;
49   *
50   *   dom.createElement(DIV);
51   * });
52   * </pre>
53   *
54   * should become
55   *
56   * <pre>
57   * goog.dom.createElement(goog.dom.TagName.DIV);
58   * </pre>
59   *
60   * The advantage of using goog.scope is that the compiler will *guarantee*
61   * the anonymous function will be inlined, even if it can't prove
62   * that it's semantically correct to do so. For example, consider this case:
63   *
64   * <pre>
65   * goog.scope(function() {
66   *   goog.getBar = function () { return alias; };
67   *   ...
68   *   var alias = foo.bar;
69   * })
70   * </pre>
71   *
72   * <p>In theory, the compiler can't inline 'alias' unless it can prove that
73   * goog.getBar is called only after 'alias' is defined. In practice, the
74   * compiler will inline 'alias' anyway, at the risk of 'fixing' bad code.
75   *
76   * @author robbyw@google.com (Robby Walker)
77   */
78  class ScopedAliases implements HotSwapCompilerPass {
79    /** Name used to denote an scoped function block used for aliasing. */
80    static final String SCOPING_METHOD_NAME = "goog.scope";

144   @Override
145   public void hotSwapScript(Node root, Node originalRoot) {
146     Traversal traversal = new Traversal();
147     NodeTraversal.traverseEs6(compiler, root, traversal);
148 
149     if (!traversal.hasErrors()) {
150 
151       // Apply the aliases.
152       List<AliasUsage> aliasWorkQueue =
153            new ArrayList<>(traversal.getAliasUsages());
154       while (!aliasWorkQueue.isEmpty()) {
155         List<AliasUsage> newQueue = new ArrayList<>();
156         for (AliasUsage aliasUsage : aliasWorkQueue) {
157           if (aliasUsage.referencesOtherAlias()) {
158             newQueue.add(aliasUsage);
159           } else {
160             aliasUsage.applyAlias();
161           }
162         }
163 
164         // Prevent an infinite loop.
165         if (newQueue.size() == aliasWorkQueue.size()) {
166           Var cycleVar = newQueue.get(0).aliasVar;
167           compiler.report(JSError.make(
168               cycleVar.getNode(), GOOG_SCOPE_ALIAS_CYCLE, cycleVar.getName()));
169           break;
170         } else {
171           aliasWorkQueue = newQueue;
172         }
173       }
174 
175       // Remove the alias definitions.
176       for (Node aliasDefinition : traversal.getAliasDefinitionsInOrder()) {
177         if (NodeUtil.isNameDeclaration(aliasDefinition.getParent())
178             && aliasDefinition.getParent().hasOneChild()) {
179           aliasDefinition.getParent().detachFromParent();
180         } else {
181           aliasDefinition.detachFromParent();
182         }
183       }
184 
185       // Collapse the scopes.
186       for (Node scopeCall : traversal.getScopeCalls()) {
187         Node expressionWithScopeCall = scopeCall.getParent();
188         Node scopeClosureBlock = scopeCall.getLastChild().getLastChild();
189         scopeClosureBlock.detachFromParent();
190         expressionWithScopeCall.getParent().replaceChild(
191             expressionWithScopeCall,
192             scopeClosureBlock);
193         NodeUtil.tryMergeBlock(scopeClosureBlock);
194       }
195 
196       if (!traversal.getAliasUsages().isEmpty() || !traversal.getAliasDefinitionsInOrder().isEmpty()
197           || !traversal.getScopeCalls().isEmpty()) {
198         compiler.reportCodeChange();
199       }
200     }
201   }

203   private abstract static class AliasUsage {
204     final Var aliasVar;
205     final Node aliasReference;
206 
207     AliasUsage(Var aliasVar, Node aliasReference) {
208       this.aliasVar = aliasVar;
209       this.aliasReference = aliasReference;
210     }
211 
212     /** Checks to see if this references another alias. */
213     public boolean referencesOtherAlias() {
214       Node aliasDefinition = aliasVar.getInitialValue();
215       Node root = NodeUtil.getRootOfQualifiedName(aliasDefinition);
216       Var otherAliasVar = aliasVar.getScope().getOwnSlot(root.getString());
217       return otherAliasVar != null;
218     }
219 
220     public abstract void applyAlias();
221   }
222 
223   private static class AliasedNode extends AliasUsage {
224     AliasedNode(Var aliasVar, Node aliasReference) {
225       super(aliasVar, aliasReference);
226     }
227 
228     @Override
229     public void applyAlias() {
230       Node aliasDefinition = aliasVar.getInitialValue();
231       Node replacement = aliasDefinition.cloneTree();
232       replacement.useSourceInfoFromForTree(aliasReference);
233       if (aliasReference.getType() == Token.STRING_KEY) {
234         Preconditions.checkState(!aliasReference.hasChildren());
235         aliasReference.addChildToFront(replacement);
236       } else {
237         aliasReference.getParent().replaceChild(aliasReference, replacement);
238       }
239     }
240   }
241 
242   private static class AliasedTypeNode extends AliasUsage {
243     AliasedTypeNode(Var aliasVar, Node aliasReference) {
244       super(aliasVar, aliasReference);
245     }
246 
247     @Override
248     public void applyAlias() {
249       Node aliasDefinition = aliasVar.getInitialValue();
250       String aliasName = aliasVar.getName();
251       String typeName = aliasReference.getString();
252       if (typeName.startsWith("$jscomp.scope.")) {
253         // Already visited.
254         return;
255       }
256       String aliasExpanded =
257           Preconditions.checkNotNull(aliasDefinition.getQualifiedName());
258       Preconditions.checkState(typeName.startsWith(aliasName),
259           "%s must start with %s", typeName, aliasName);
260       String replacement =
261           aliasExpanded + typeName.substring(aliasName.length());
262       aliasReference.setString(replacement);
263     }
264   }

284     // Suppose you create an alias.
285     // var x = goog.x;
286     // As a side-effect, this means you can shadow the namespace 'goog'
287     // in inner scopes. When we inline the namespaces, we have to rename
288     // these shadows.
289     //
290     // Fortunately, we already have a name uniquifier that runs during tree
291     // normalization (before optimizations). We run it here on a limited
292     // set of variables, but only as a last resort (because this will screw
293     // up warning messages downstream).
294     private final Set<String> forbiddenLocals = new HashSet<>(
295         ImmutableSet.of("$jscomp"));

324      * Returns true if this NodeTraversal is currently within a goog.scope function body

327       return scopeFunctionBody != null;

333     private boolean isGoogScopeFunctionBody(Node n) {
334       return inGoogScopeBody() && n == scopeFunctionBody;

337     private boolean isCallToScopeMethod(Node n) {
338       return n.isCall() && n.getFirstChild().matchesQualifiedName(SCOPING_METHOD_NAME);
339     }

341     /**
342      * @param scopeRoot the Node which is the root of the current scope
343      * @return the goog.scope() CALL node containing the scopeRoot, or null if scopeRoot is not
344      *     in a goog.scope() call.
345      */
346     private Node findScopeMethodCall(Node scopeRoot) {
347       Node n = scopeRoot.getParent().getParent();
348       if (isCallToScopeMethod(n)) {
349         return n;
350       }
351       return null;
352     }

354     @Override
355     public void enterScope(NodeTraversal t) {
356       if (t.inGlobalHoistScope()) {
357         return;
358       }
359       if (inGoogScopeBody()) {
360         Scope hoistedScope = t.getClosestHoistScope();
361         if (isGoogScopeFunctionBody(hoistedScope.getRootNode())) {
362           findAliases(t, hoistedScope);
363         }
364       }
365       Node scopeMethodCall = findScopeMethodCall(t.getScope().getRootNode());
366       if (scopeMethodCall != null) {
367         transformation = transformationHandler.logAliasTransformation(
368             scopeMethodCall.getSourceFileName(), getSourceRegion(scopeMethodCall));
369         findAliases(t, t.getScope());
370         scopeFunctionBody = scopeMethodCall.getLastChild().getLastChild();
371       }
372     }
373 
374     @Override
375     public void exitScope(NodeTraversal t) {
376       if (isGoogScopeFunctionBody(t.getScopeRoot())) {
377         scopeFunctionBody = null;
378         renameNamespaceShadows(t);
379         injectedDecls.clear();
380         aliases.clear();
381         forbiddenLocals.clear();
382         transformation = null;
383         hasNamespaceShadows = false;
384       } else if (inGoogScopeBody()) {
385         findNamespaceShadows(t);
386         reportInvalidVariables(t);
387       }
388     }

390     private void reportInvalidVariables(NodeTraversal t) {
391       Node scopeRoot = t.getScopeRoot();
392       Node enclosingFunctionBody = t.getEnclosingFunction().getLastChild();
393       if (isGoogScopeFunctionBody(enclosingFunctionBody) && scopeRoot.isBlock()
394           && !scopeRoot.getParent().isFunction()) {
395         for (Var v : t.getScope().getVarIterable()) {
396           Node parent = v.getNameNode().getParent();
397           if (NodeUtil.isFunctionDeclaration(parent)) {
398             // Disallow block-scoped function declarations that leak into the goog.scope
399             // function body. Technically they shouldn't leak in ES6 but the browsers don't agree
400             // on that yet.
401             report(t, v.getNode(), GOOG_SCOPE_INVALID_VARIABLE, v.getName());
402           }
403         }
404         if (scopeRoot.hasChildren() && scopeRoot.getFirstChild().isCatch()) {
405           for (Var v : t.getClosestHoistScope().getVarIterable()) {
406             // Disallow variables declared in catch blocks that leak into the goog.scope
407             // function body.
408             report(t, v.getNode(), GOOG_SCOPE_INVALID_VARIABLE, v.getName());
409           }
410         }
411       }
412     }

414     private SourcePosition<AliasTransformation> getSourceRegion(Node n) {
415       Node testNode = n;
416       Node next = null;
417       for (; next != null || testNode.isScript();) {
418         next = testNode.getNext();
419         testNode = testNode.getParent();
420       }
421 
422       int endLine = next == null ? Integer.MAX_VALUE : next.getLineno();
423       int endChar = next == null ? Integer.MAX_VALUE : next.getCharno();
424       SourcePosition<AliasTransformation> pos =
425           new SourcePosition<AliasTransformation>() {};
426       pos.setPositionInformation(
427           n.getLineno(), n.getCharno(), endLine, endChar);
428       return pos;
429     }

437     private void findAliases(NodeTraversal t, Scope scope) {
438       for (Var v : scope.getVarIterable()) {
439         Node n = v.getNode();
440         Node parent = n.getParent();
441         // We use isBlock to avoid variables declared in loop headers.
442         boolean isVar = NodeUtil.isNameDeclaration(parent) && parent.getParent().isBlock();
443         boolean isFunctionDecl = NodeUtil.isFunctionDeclaration(parent);
444         if (isVar && n.getFirstChild() != null && n.getFirstChild().isQualifiedName()) {
445           recordAlias(v);
446         } else if (v.isBleedingFunction()) {
447           // Bleeding functions already get a BAD_PARAMETERS error, so just
448           // do nothing.
449         } else if (parent.getType() == Token.PARAM_LIST) {
450           // Parameters of the scope function also get a BAD_PARAMETERS
451           // error.
452         } else if (isVar || isFunctionDecl || NodeUtil.isClassDeclaration(parent)) {
453           boolean isHoisted = NodeUtil.isHoistedFunctionDeclaration(parent);
454           Node grandparent = parent.getParent();
455           Node value = v.getInitialValue();
456           Node varNode = null;
457 
458           // Pull out inline type declaration if present.
459           if (n.getJSDocInfo() != null) {
460             JSDocInfoBuilder builder = JSDocInfoBuilder.maybeCopyFrom(parent.getJSDocInfo());
461             if (isFunctionDecl) { // Fix inline return type.
462               builder.recordReturnType(n.getJSDocInfo().getType());
463             } else {
464               builder.recordType(n.getJSDocInfo().getType());
465             }
466             parent.setJSDocInfo(builder.build());
467             n.setJSDocInfo(null);
468           }
469           // Grab the docinfo before we do any AST manipulation.
470           JSDocInfo varDocInfo = v.getJSDocInfo();
471 
472           String name = n.getString();
473           int nameCount = scopedAliasNames.count(name);
474           scopedAliasNames.add(name);
475           String globalName =
476               "$jscomp.scope." + name + (nameCount == 0 ? "" : ("$" + nameCount));
477 
478           compiler.ensureLibraryInjected("base", true);
479 
480           // First, we need to free up the function expression (EXPR)
481           // to be used in another expression.
482           if (isFunctionDecl || NodeUtil.isClassDeclaration(parent)) {
483             // Replace "function NAME() { ... }" with "var NAME;".
484             // Replace "class NAME { ... }" with "var NAME;".
485 
486             // We can't keep the local name on the function expression,
487             // because IE is buggy and will leak the name into the global
488             // scope. This is covered in more detail here:
489             // http://wiki.ecmascript.org/lib/exe/fetch.php?id=resources:resources&cache=cache&media=resources:jscriptdeviationsfromes3.pdf
490             //
491             // This will only cause problems if this is a hoisted, recursive
492             // function, and the programmer is using the hoisting.
493             Node newName;
494             if (isFunctionDecl) {
495               newName = IR.name("");
496             } else {
497               newName = IR.empty();
498             }
499             newName.useSourceInfoFrom(n);
500             value.replaceChild(n, newName);
501 
502             varNode = IR.var(n).useSourceInfoFrom(n);
503             grandparent.replaceChild(parent, varNode);
504           } else {
505             if (value != null) {
506               // If this is a VAR, we can just detach the expression and
507               // the tree will still be valid.
508               value.detachFromParent();
509             }
510             varNode = parent;
511           }
512 
513           // Add $jscomp.scope.name = EXPR;
514           // Make sure we copy over all the jsdoc and debug info.
515           if (value != null || varDocInfo != null) {
516             Node newDecl = NodeUtil.newQNameDeclaration(
517                 compiler,
518                 globalName,
519                 value,
520                 varDocInfo)
521                 .useSourceInfoIfMissingFromForTree(n);
522             NodeUtil.setDebugInformation(
523                 newDecl.getFirstChild().getFirstChild(), n, name);
524 
525             if (isHoisted) {
526               grandparent.addChildToFront(newDecl);
527             } else {
528               grandparent.addChildBefore(newDecl, varNode);
529             }
530             injectedDecls.add(newDecl.getFirstChild());
531           }
532 
533           // Rewrite "var name = EXPR;" to "var name = $jscomp.scope.name;"
534           v.getNameNode().addChildToFront(
535               NodeUtil.newQName(
536                   compiler, globalName, n, name));
537 
538           recordAlias(v);
539         } else {
540           // Do not other kinds of local symbols, like catch params.
541           report(t, n, GOOG_SCOPE_NON_ALIAS_LOCAL, n.getString());
542         }
543       }
544     }

546     private void recordAlias(Var aliasVar) {
547       String name = aliasVar.getName();
548       aliases.put(name, aliasVar);
549 
550       String qualifiedName =
551         aliasVar.getInitialValue().getQualifiedName();
552       transformation.addAlias(name, qualifiedName);
553 
554       int rootIndex = qualifiedName.indexOf('.');
555       if (rootIndex != -1) {
556         String qNameRoot = qualifiedName.substring(0, rootIndex);
557         if (!aliases.containsKey(qNameRoot)) {
558           forbiddenLocals.add(qNameRoot);
559         }
560       }
561     }

563     /** Find out if there are any local shadows of namespaces. */
564     private void findNamespaceShadows(NodeTraversal t) {
565       if (hasNamespaceShadows) {
566         return;
567       }
568 
569       Scope scope = t.getScope();
570       for (Var v : scope.getVarIterable()) {
571         if (forbiddenLocals.contains(v.getName())) {
572           hasNamespaceShadows = true;
573           return;
574         }
575       }
576     }
577 
578     /**
579      * Rename any local shadows of namespaces.
580      * This should be a very rare occurrence, so only do this traversal
581      * if we know that we need it.
582      */
583     private void renameNamespaceShadows(NodeTraversal t) {
584       if (hasNamespaceShadows) {
585         MakeDeclaredNamesUnique.Renamer renamer =
586             new MakeDeclaredNamesUnique.WhitelistedRenamer(
587                 new MakeDeclaredNamesUnique.ContextualRenamer(),
588                 forbiddenLocals);
589         for (String s : forbiddenLocals) {
590           renamer.addDeclaredName(s, false);
591         }
592         MakeDeclaredNamesUnique uniquifier =
593             new MakeDeclaredNamesUnique(renamer);
594         Node parent = t.getScopeRoot().getParent();
595         NodeTraversal.traverseEs6(compiler, parent, uniquifier);
596       }
597     }

599     private void validateScopeCall(NodeTraversal t, Node n, Node parent) {
600       if (preprocessorSymbolTable != null) {
601         preprocessorSymbolTable.addReference(n.getFirstChild());
602       }
603       if (!parent.isExprResult()) {
604         report(t, n, GOOG_SCOPE_MUST_BE_ALONE);
605       }
606       if (t.getEnclosingFunction() != null) {
607         report(t, n, GOOG_SCOPE_MUST_BE_IN_GLOBAL_SCOPE);
608       }
609       if (n.getChildCount() != 2) {
610         // The goog.scope call should have exactly 1 parameter.  The first
611         // child is the "goog.scope" and the second should be the parameter.
612         report(t, n, GOOG_SCOPE_HAS_BAD_PARAMETERS);
613       } else {
614         Node anonymousFnNode = n.getChildAtIndex(1);
615         if (!anonymousFnNode.isFunction()
616             || NodeUtil.getFunctionName(anonymousFnNode) != null
617             || NodeUtil.getFunctionParameters(anonymousFnNode).hasChildren()) {
618           report(t, anonymousFnNode, GOOG_SCOPE_HAS_BAD_PARAMETERS);
619         } else {
620           scopeCalls.add(n);
621         }
622       }
623     }

625     @Override
626     public void visit(NodeTraversal t, Node n, Node parent) {
627       if (isCallToScopeMethod(n)) {
628         validateScopeCall(t, n, n.getParent());
629       }
630 
631       if (!inGoogScopeBody()) {
632         return;
633       }
634 
635       int type = n.getType();
636       boolean isObjLitShorthand = type == Token.STRING_KEY && !n.hasChildren();
637       Var aliasVar = null;
638       if (type == Token.NAME || isObjLitShorthand) {
639         String name = n.getString();
640         Var lexicalVar = t.getScope().getVar(name);
641         if (lexicalVar != null && lexicalVar == aliases.get(name)) {
642           aliasVar = lexicalVar;
643         }
644       }
645 
646       if (isGoogScopeFunctionBody(t.getEnclosingFunction().getLastChild())) {
647         if (aliasVar != null && !isObjLitShorthand && NodeUtil.isLValue(n)) {
648           if (aliasVar.getNode() == n) {
649             aliasDefinitionsInOrder.add(n);
650 
651             // Return early, to ensure that we don't record a definition
652             // twice.
653             return;
654           } else {
655             report(t, n, GOOG_SCOPE_ALIAS_REDEFINED, n.getString());
656           }
657         }
658 
659         if (type == Token.RETURN) {
660           report(t, n, GOOG_SCOPE_USES_RETURN);
661         } else if (type == Token.THIS) {
662           report(t, n, GOOG_SCOPE_REFERENCES_THIS);
663         } else if (type == Token.THROW) {
664           report(t, n, GOOG_SCOPE_USES_THROW);
665         }
666       }
667 
668       // Validate all descendent scopes of the goog.scope block.
669       if (inGoogScopeBody()) {
670         // Check if this name points to an alias.
671         if (aliasVar != null) {
672           // Note, to support the transitive case, it's important we don't
673           // clone aliasedNode here.  For example,
674           // var g = goog; var d = g.dom; d.createElement('DIV');
675           // The node in aliasedNode (which is "g") will be replaced in the
676           // changes pass above with "goog".  If we cloned here, we'd end up
677           // with <code>g.dom.createElement('DIV')</code>.
678           aliasUsages.add(new AliasedNode(aliasVar, n));
679         }
680 
681         // When we inject declarations, we duplicate jsdoc. Make sure
682         // we only process that jsdoc once.
683         JSDocInfo info = n.getJSDocInfo();
684         if (info != null && !injectedDecls.contains(n)) {
685           for (Node node : info.getTypeNodes()) {
686             fixTypeNode(node);
687           }
688         }
689       }
690     }
691 
692     private void fixTypeNode(Node typeNode) {
693       if (typeNode.isString()) {
694         String name = typeNode.getString();
695         int endIndex = name.indexOf('.');
696         if (endIndex == -1) {
697           endIndex = name.length();
698         }
699         String baseName = name.substring(0, endIndex);
700         Var aliasVar = aliases.get(baseName);
701         if (aliasVar != null) {
702           aliasUsages.add(new AliasedTypeNode(aliasVar, typeNode));
703         }
704       }
705 
706       for (Node child = typeNode.getFirstChild(); child != null;
707            child = child.getNext()) {
708         fixTypeNode(child);
709       }
710     }
711   }
