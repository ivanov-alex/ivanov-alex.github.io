38  /**
39   * Creates an externs file containing all exported symbols and properties
40   * for later consumption.
41   *
42   * @author dcc@google.com (Devin Coughlin)
43   */
44  final class ExternExportsPass extends NodeTraversal.AbstractPostOrderCallback
45      implements CompilerPass {

59    /** A mapping of internal paths to exported paths. */

63    private final Set<String> alreadyExportedPaths;

71    private abstract class Export {
72      protected final String symbolName;
73      protected final Node value;
74  
75      Export(String symbolName, Node value) {
76        this.symbolName = Preconditions.checkNotNull(symbolName);
77        this.value = Preconditions.checkNotNull(value);
78      }
79  
80      /**
81       * Generates the externs representation of this export and appends
82       * it to the externsRoot AST.
83       */
84      void generateExterns() {
85        appendExtern(getExportedPath(), getValue(value));
86      }
87  
88      /**
89       * Returns the path exported by this export.
90       */
91      abstract String getExportedPath();
92  
93      /**
94       * Appends the exported function and all paths necessary for the path to be
95       * declared. For example, for a property "a.b.c", the initializers for
96       * paths "a", "a.b" will be appended (if they have not already) and a.b.c
97       * will be initialized with the exported version of the function:
98       * <pre>
99       * var a = {};
100      * a.b = {};
101      * a.b.c = function(x,y) { }
102      * </pre>
103      */
104     void appendExtern(String path, Node valueToExport) {
105       List<String> pathPrefixes = computePathPrefixes(path);
106 
107       for (int i = 0; i < pathPrefixes.size(); ++i) {
108         String pathPrefix = pathPrefixes.get(i);
109 
110         /* The complete path (the last path prefix) must be emitted and
111          * it gets initialized to the externed version of the value.
112          */
113         boolean isCompletePathPrefix = (i == pathPrefixes.size() - 1);
114 
115         boolean skipPathPrefix = pathPrefix.endsWith(".prototype")
116             || (alreadyExportedPaths.contains(pathPrefix)
117                 && !isCompletePathPrefix);
118 
119         if (!skipPathPrefix) {
120            Node initializer;
121 
122           /* Namespaces get initialized to {}, functions to
123            * externed versions of their value, and if we can't
124            * figure out where the value came from we initialize
125            * it to {}.
126            *
127            * Since externs are always exported in sorted order,
128            * we know that if we export a.b = function() {} and later
129            * a.b.c = function then a.b will always be in alreadyExportedPaths
130            * when we emit a.b.c and thus we will never overwrite the function
131            * exported for a.b with a namespace.
132            */
133 
134           if (isCompletePathPrefix && valueToExport != null) {
135             if (valueToExport.isFunction()) {
136               initializer = createExternFunction(valueToExport);
137             } else {
138               Preconditions.checkState(valueToExport.isObjectLit());
139               initializer = createExternObjectLit(valueToExport);
140             }
141           } else {
142             initializer = IR.empty();
143           }
144 
145           appendPathDefinition(pathPrefix, initializer);
146         }
147       }
148     }
149 
150     /**
151      * Computes a list of the path prefixes constructed from the components
152      * of the path.
153      * <pre>
154      * E.g., if the path is:
155      *      "a.b.c"
156      * then then path prefixes will be
157      *    ["a","a.b","a.b.c"]:
158      * </pre>
159      */
160     private List<String> computePathPrefixes(String path) {
161       List<String> pieces = Splitter.on('.').splitToList(path);
162       List<String> pathPrefixes = new ArrayList<>();
163 
164       for (int i = 0; i < pieces.size(); i++) {
165         pathPrefixes.add(Joiner.on(".").join(Iterables.limit(pieces, i + 1)));
166       }
167 
168       return pathPrefixes;
169     }
170 
171     private void appendPathDefinition(String path, Node initializer) {
172       Node pathDefinition;
173 
174       if (!path.contains(".")) {
175         if (initializer.isEmpty()) {
176           pathDefinition = IR.var(IR.name(path));
177         } else {
178           pathDefinition = NodeUtil.newVarNode(path, initializer);
179         }
180       } else {
181         Node qualifiedPath = NodeUtil.newQName(compiler, path);
182         if (initializer.isEmpty()) {
183           pathDefinition = NodeUtil.newExpr(qualifiedPath);
184         } else {
185           pathDefinition = NodeUtil.newExpr(
186               IR.assign(qualifiedPath, initializer));
187         }
188       }
189 
190       externsRoot.addChildToBack(pathDefinition);
191 
192       alreadyExportedPaths.add(path);
193     }
194 
195     /**
196      * Given a function to export, create the empty function that
197      * will be put in the externs file. This extern function should have
198      * the same type as the original function and the same parameter
199      * name but no function body.
200      *
201      * We create a warning here if the the function to export is missing
202      * parameter or return types.
203      */
204     private Node createExternFunction(Node exportedFunction) {
205       Node paramList = NodeUtil.getFunctionParameters(exportedFunction)
206           .cloneTree();
207       // Use the original parameter names so that the externs look pretty.
208       Node param = paramList.getFirstChild();
209       while (param != null && param.isName()) {
210         String originalName = (String) param.getProp(Node.ORIGINALNAME_PROP);
211         if (originalName != null) {
212           param.setString(originalName);
213         }
214         param = param.getNext();
215       }
216       Node externFunction = IR.function(IR.name(""), paramList, IR.block());
217 
218       externFunction.setJSType(exportedFunction.getJSType());
219 
220       return externFunction;
221     }
222 
223     private JSDocInfo buildEmptyJSDoc() {
224       // TODO(johnlenz): share the JSDocInfo here rather than building
225       // a new one each time.
226       return new JSDocInfoBuilder(false).build(true);
227     }
228 
229     /**
230      * Given an object literal to export, create an object lit with all its
231      * string properties. We don't care what the values of those properties
232      * are because they are not checked.
233      */
234     private Node createExternObjectLit(Node exportedObjectLit) {
235       Node lit = IR.objectlit();
236       lit.setJSType(exportedObjectLit.getJSType());
237 
238       // This is an indirect way of telling the typed code generator
239       // "print the type of this"
240       lit.setJSDocInfo(buildEmptyJSDoc());
241 
242       int index = 1;
243       for (Node child = exportedObjectLit.getFirstChild();
244            child != null;
245            child = child.getNext()) {
246         // TODO(dimvar): handle getters or setters?
247         if (child.isStringKey()) {
248           lit.addChildToBack(
249               IR.propdef(
250                   IR.stringKey(child.getString()),
251                   IR.number(index++)));
252         }
253       }
254       return lit;
255     }
256 
257     /**
258      * If the given value is a qualified name which refers
259      * a function or object literal, the node is returned. Otherwise,
260      * {@code null} is returned.
261      */
262     protected Node getValue(Node qualifiedNameNode) {
263       String qualifiedName = value.getQualifiedName();
264 
265       if (qualifiedName == null) {
266         return null;
267       }
268 
269       Node definitionParent = definitionMap.get(qualifiedName);
270       if (definitionParent == null) {
271         return null;
272       }
273 
274       Node definition;
275 
276       switch (definitionParent.getType()) {
277         case Token.ASSIGN:
278           definition = definitionParent.getLastChild();
279           break;
280         case Token.VAR:
281           definition = definitionParent.getLastChild().getLastChild();
282           break;
283         case Token.FUNCTION:
284           if (NodeUtil.isFunctionDeclaration(definitionParent)) {
285             definition = definitionParent;
286           } else {
287             return null;
288           }
289           break;
290         default:
291             return null;
292       }
293 
294       if (!definition.isFunction() && !definition.isObjectLit()) {
295         return null;
296       }
297 
298       return definition;
299     }
300   }

302   /**
303    * A symbol export.
304    */
305   private class SymbolExport extends Export {
306 
307     public SymbolExport(String symbolName, Node value) {
308       super(symbolName, value);
309 
310       String qualifiedName = value.getQualifiedName();
311 
312       if (qualifiedName != null) {
313         mappedPaths.put(qualifiedName, symbolName);
314       }
315     }
316 
317     @Override
318     String getExportedPath() {
319       return symbolName;
320     }
321   }

323   /**
324    * A property export.
325    */
326   private class PropertyExport extends Export {
327     private final String exportPath;
328 
329     public PropertyExport(String exportPath, String symbolName, Node value) {
330       super(symbolName, value);
331 
332       this.exportPath = Preconditions.checkNotNull(exportPath);
333     }
334 
335     @Override
336     String getExportedPath() {
337 
338       // Find the longest path that has been mapped (if any).
339       List<String> pieces = Splitter.on('.').splitToList(exportPath);
340 
341       for (int i = pieces.size(); i > 0; i--) {
342         // Find the path of the current length.
343         String cPath = Joiner.on(".").join(Iterables.limit(pieces, i));
344 
345         // If this path is mapped, return the mapped path plus any remaining
346         // pieces.
347         if (mappedPaths.containsKey(cPath)) {
348           String newPath = mappedPaths.get(cPath);
349 
350           if (i < pieces.size()) {
351             newPath += "." + Joiner.on(".").join(Iterables.skip(pieces, i));
352           }
353 
354           return newPath + "." + symbolName;
355         }
356       }
357 
358       return exportPath + "." + symbolName;
359     }
360   }

362   /**
363    * Creates an instance.
364    */
365   ExternExportsPass(AbstractCompiler compiler) {
366     this.exports = new ArrayList<>();
367     this.compiler = compiler;
368     this.definitionMap = new HashMap<>();
369     this.externsRoot = IR.block();
370     this.externsRoot.setIsSyntheticBlock(true);
371     this.alreadyExportedPaths = new HashSet<>();
372     this.mappedPaths = new HashMap<>();
373 
374     initExportMethods();
375   }
376 
377   private void initExportMethods() {
378     exportSymbolFunctionNames = new ArrayList<>();
379     exportPropertyFunctionNames = new ArrayList<>();
380 
381     // From Closure:
382     // goog.exportSymbol = function(publicName, symbol)
383     // goog.exportProperty = function(object, publicName, symbol)
384     CodingConvention convention = compiler.getCodingConvention();
385     exportSymbolFunctionNames.add(convention.getExportSymbolFunction());
386     exportPropertyFunctionNames.add(convention.getExportPropertyFunction());
387 
388     // Another common one used inside google:
389     exportSymbolFunctionNames.add("google_exportSymbol");
390     exportPropertyFunctionNames.add("google_exportProperty");
391   }
392 
393   @Override
394   public void process(Node externs, Node root) {
395     NodeTraversal.traverseEs6(compiler, root, this);
396 
397     // Sort by path length to ensure that the longer
398     // paths (which may depend on the shorter ones)
399     // come later.
400     Set<Export> sorted =
401         new TreeSet<>(new Comparator<Export>() {
402           @Override
403           public int compare(Export e1, Export e2) {
404             return e1.getExportedPath().compareTo(e2.getExportedPath());
405           }
406         });
407 
408     sorted.addAll(exports);
409 
410     for (Export export : sorted) {
411       export.generateExterns();
412     }
413   }

415   /**
416    * Returns the generated externs.
417    */
418   public String getGeneratedExterns() {
419     CodePrinter.Builder builder = new CodePrinter.Builder(externsRoot)
420       .setPrettyPrint(true)
421       .setOutputTypes(true)
422       .setTypeRegistry(compiler.getTypeIRegistry());
423 
424     return builder.build();
425   }

427   @Override
428   public void visit(NodeTraversal t, Node n, Node parent) {
429     switch (n.getType()) {
430 
431       case Token.NAME:
432       case Token.GETPROP:
433         String name = n.getQualifiedName();
434         if (name == null) {
435           return;
436         }
437 
438         if (parent.isAssign() || parent.isVar() || parent.isFunction()) {
439           definitionMap.put(name, parent);
440         }
441 
442         JSDocInfo jsdoc = NodeUtil.getBestJSDocInfo(n);
443         if (jsdoc != null && jsdoc.isExport()) {
444           handleExportDefinition(t, n);
445         }
446 
447         // Only handle function calls. This avoids assignments
448         // that do not export items directly.
449         if (!parent.isCall()) {
450           return;
451         }
452 
453         if (exportPropertyFunctionNames.contains(name)) {
454           handlePropertyExportCall(parent);
455         }
456 
457         if (exportSymbolFunctionNames.contains(name)) {
458           handleSymbolExportCall(parent);
459         }
460 
461     }
462   }
463 
464   private void handleSymbolExportCall(Node parent) {
465     // Ensure that we only check valid calls with the 2 arguments
466     // (plus the GETPROP node itself).
467     if (parent.getChildCount() != 3) {
468       return;
469     }
470 
471     Node thisNode = parent.getFirstChild();
472     Node nameArg = thisNode.getNext();
473     Node valueArg = nameArg.getNext();
474 
475     // Confirm the arguments are the expected types. If they are not,
476     // then we have an export that we cannot statically identify.
477     if (!nameArg.isString()) {
478       return;
479     }
480 
481     // Add the export to the list.
482     this.exports.add(new SymbolExport(nameArg.getString(), valueArg));
483   }
484 
485   private void handlePropertyExportCall(Node parent) {
486     // Ensure that we only check valid calls with the 3 arguments
487     // (plus the GETPROP node itself).
488     if (parent.getChildCount() != 4) {
489       return;
490     }
491 
492     Node thisNode = parent.getFirstChild();
493     Node objectArg = thisNode.getNext();
494     Node nameArg = objectArg.getNext();
495     Node valueArg = nameArg.getNext();
496 
497     // Confirm the arguments are the expected types. If they are not,
498     // then we have an export that we cannot statically identify.
499     if (!objectArg.isQualifiedName()) {
500       return;
501     }
502 
503     if (!nameArg.isString()) {
504       return;
505     }
506 
507     // Add the export to the list.
508     this.exports.add(
509         new PropertyExport(objectArg.getQualifiedName(),
510                            nameArg.getString(),
511                            valueArg));
512   }
513 
514   private void handleExportDefinition(NodeTraversal t, Node definitionNode) {
515     // For now, only handle properties defined on this inside of a constructor
516     if (!definitionNode.isGetProp()
517         || !definitionNode.getFirstChild().isThis()) {
518       // Not a property on THIS
519       return;
520     }
521 
522     Node constructorNode = t.getEnclosingFunction();
523     JSDocInfo constructorJsdoc = NodeUtil.getBestJSDocInfo(constructorNode);
524     if (constructorJsdoc == null || !constructorJsdoc.isConstructor()) {
525       // Not inside a constructor
526       return;
527     }
528 
529     String constructorName = NodeUtil.getFunctionName(constructorNode);
530     String propertyName = definitionNode.getLastChild().getString();
531     String prototypeName = constructorName + ".prototype";
532     Node propertyNameNode = NodeUtil.newQName(compiler, "this." + propertyName);
533 
534     // Add the export to the list.
535     this.exports.add(new PropertyExport(prototypeName, propertyName, propertyNameNode));
536   }
