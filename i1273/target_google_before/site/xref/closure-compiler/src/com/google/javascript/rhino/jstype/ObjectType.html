57  /**
58   * Object type.
59   *
60   * In JavaScript, all object types have properties, and each of those
61   * properties has a type. Property types may be DECLARED, INFERRED, or
62   * UNKNOWN.
63   *
64   * DECLARED properties have an explicit type annotation, as in:
65   * <code>
66   * /xx @type {number} x/
67   * Foo.prototype.bar = 1;
68   * </code>
69   * This property may only hold number values, and an assignment to any
70   * other type of value is an error.
71   *
72   * INFERRED properties do not have an explicit type annotation. Rather,
73   * we try to find all the possible types that this property can hold.
74   * <code>
75   * Foo.prototype.bar = 1;
76   * </code>
77   * If the programmer assigns other types of values to this property,
78   * the property will take on the union of all these types.
79   *
80   * UNKNOWN properties are properties on the UNKNOWN type. The UNKNOWN
81   * type has all properties, but we do not know whether they are
82   * declared or inferred.
83   *
84   */
85  public abstract class ObjectType
86      extends JSType
87      implements ObjectTypeI {

153   /**
154    * Detects a cycle in the implicit prototype chain. This method accesses
155    * the {@link #getImplicitPrototype()} method and must therefore be
156    * invoked only after the object is sufficiently initialized to respond to
157    * calls to this method.<p>
158    *
159    * @return True iff an implicit prototype cycle was detected.
160    */
161   final boolean detectImplicitPrototypeCycle() {
162     // detecting cycle
163     this.visited = true;
164     ObjectType p = getImplicitPrototype();
165     while (p != null) {
166       if (p.visited) {
167         return true;
168       } else {
169         p.visited = true;
170       }
171       p = p.getImplicitPrototype();
172     }
173 
174     // clean up
175     p = this;
176     do {
177       p.visited = false;
178       p = p.getImplicitPrototype();
179     } while (p != null);
180     return false;
181   }

183   /**
184    * Detects cycles in either the implicit prototype chain, or the implemented/extended
185    * interfaces.<p>
186    *
187    * @return True iff a cycle was detected.
188    */
189   final boolean detectInheritanceCycle() {
190     // TODO(dimvar): This should get moved to preventing cycles in FunctionTypeBuilder
191     // rather than removing them here after they have been created.
192     // Also, this doesn't do the right thing for extended interfaces, though that is
193     // masked by another bug.
194     return detectImplicitPrototypeCycle()
195         || Iterables.contains(this.getCtorImplementedInterfaces(), this)
196         || Iterables.contains(this.getCtorExtendedInterfaces(), this);
197   }

201    * like constructors, prototypes, and enums. It notably does not include

208   /**
209    * Due to the complexity of some of our internal type systems, sometimes
210    * we have different types constructed by the same constructor.
211    * In other parts of the type system, these are called delegates.
212    * We construct these types by appending suffixes to the constructor name.
213    *
214    * The normalized reference name does not have these suffixes, and as such,
215    * recollapses these implicit types back to their real type.
216    */
217   public String getNormalizedReferenceName() {
218     String name = getReferenceName();
219     if (name != null) {
220       int pos = name.indexOf('(');
221       if (pos != -1) {
222         return name.substring(0, pos);
223       }
224     }
225     return name;
226   }

249   @Override
250   public TernaryValue testForEquality(JSType that) {
251     // super
252     TernaryValue result = super.testForEquality(that);
253     if (result != null) {
254       return result;
255     }
256     // objects are comparable to everything but null/undefined
257     if (that.isSubtype(
258             getNativeType(JSTypeNative.OBJECT_NUMBER_STRING_BOOLEAN))) {
259       return UNKNOWN;
260     } else {
261       return FALSE;
262     }
263   }

278   /**
279    * Defines a property whose type is explicitly declared by the programmer.
280    * @param propertyName the property's name
281    * @param type the type
282    * @param propertyNode the node corresponding to the declaration of property
283    *        which might later be accessed using {@code getPropertyNode}.
284    */
285   public final boolean defineDeclaredProperty(String propertyName,
286       JSType type, Node propertyNode) {
287     boolean result = defineProperty(propertyName, type, false, propertyNode);
288     // All property definitions go through this method
289     // or defineInferredProperty. Because the properties defined an an
290     // object can affect subtyping, it's slightly more efficient
291     // to register this after defining the property.
292     registry.registerPropertyOnType(propertyName, this);
293     return result;
294   }

299    * bookkeeping in the type system.

302       JSType type, Node propertyNode) {
303     return defineProperty(propertyName, type, false, propertyNode);
304   }

306   /**
307    * Defines a property whose type is inferred.
308    * @param propertyName the property's name
309    * @param type the type
310    * @param propertyNode the node corresponding to the inferred definition of
311    *        property that might later be accessed using {@code getPropertyNode}.
312    */
313   public final boolean defineInferredProperty(String propertyName,
314       JSType type, Node propertyNode) {
315     if (hasProperty(propertyName)) {
316       if (isPropertyTypeDeclared(propertyName)) {
317         // We never want to hide a declared property with an inferred property.
318         return true;
319       }
320       JSType originalType = getPropertyType(propertyName);
321       type = originalType == null ? type :
322           originalType.getLeastSupertype(type);
323     }
324 
325     boolean result = defineProperty(propertyName, type, true,
326         propertyNode);
327 
328     // All property definitions go through this method
329     // or defineDeclaredProperty. Because the properties defined an an
330     // object can affect subtyping, it's slightly more efficient
331     // to register this after defining the property.
332     registry.registerPropertyOnType(propertyName, this);
333 
334     return result;
335   }

415   /**
416    * Gets the property type of the property whose name is given. If the
417    * underlying object does not have this property, the Unknown type is
418    * returned to indicate that no information is available on this property.
419    *
420    * This gets overridden by FunctionType for lazily-resolved call() and
421    * bind() functions.
422    *
423    * @return the property's type or {@link UnknownType}. This method never
424    *         returns {@code null}.
425    */
426   public JSType getPropertyType(String propertyName) {
427     StaticTypedSlot<JSType> slot = getSlot(propertyName);
428     if (slot == null) {
429       if (isNoResolvedType() || isCheckedUnknownType()) {
430         return getNativeType(JSTypeNative.CHECKED_UNKNOWN_TYPE);
431       } else if (isEmptyType()) {
432         return getNativeType(JSTypeNative.NO_TYPE);
433       }
434       return getNativeType(JSTypeNative.UNKNOWN_TYPE);
435     }
436     return slot.getType();
437   }

530   /**
531    * Checks that the prototype is an implicit prototype of this object. Since
532    * each object has an implicit prototype, an implicit prototype's
533    * implicit prototype is also this implicit prototype's.
534    *
535    * @param prototype any prototype based object
536    *
537    * @return {@code true} if {@code prototype} is {@code equal} to any
538    *         object in this object's implicit prototype chain.
539    */
540   final boolean isImplicitPrototype(ObjectType prototype) {
541     for (ObjectType current = this;
542          current != null;
543          current = current.getImplicitPrototype()) {
544       if (current.isTemplatizedType()) {
545         current = current.toMaybeTemplatizedType().getReferencedType();
546       }
547       if (current.isEquivalentTo(prototype)) {
548         return true;
549       }
550     }
551     return false;
552   }

559   /**
560    * We treat this as the unknown type if any of its implicit prototype
561    * properties is unknown.
562    */
563   @Override
564   public boolean isUnknownType() {
565     // If the object is unknown now, check the supertype again,
566     // because it might have been resolved since the last check.
567     if (unknown) {
568       ObjectType implicitProto = getImplicitPrototype();
569       if (implicitProto == null ||
570           implicitProto.isNativeObjectType()) {
571         unknown = false;
572         for (ObjectType interfaceType : getCtorExtendedInterfaces()) {
573           if (interfaceType.isUnknownType()) {
574             unknown = true;
575             break;
576           }
577         }
578       } else {
579         unknown = implicitProto.isUnknownType();
580       }
581     }
582     return unknown;
583   }
