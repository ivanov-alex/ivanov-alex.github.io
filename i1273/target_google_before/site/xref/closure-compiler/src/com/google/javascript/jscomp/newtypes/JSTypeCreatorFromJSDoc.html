183   private JSType getTypeFromCommentHelper(Node n, DeclaredTypeRegistry registry,
184       ImmutableList<String> typeParameters) throws UnknownTypeException {
185     Preconditions.checkNotNull(n);
186     if (typeParameters == null) {
187       typeParameters = ImmutableList.of();
188     }
189     switch (n.getType()) {
190       case Token.LC:
191         return getRecordTypeHelper(n, registry, typeParameters);
192       case Token.EMPTY: // for function types that don't declare a return type
193         return JSType.UNKNOWN;
194       case Token.VOID:
195         return JSType.UNDEFINED;
196       case Token.LB:
197         warn("The [] type syntax is no longer supported." +
198              " Please use Array.<T> instead.", n);
199         return JSType.UNKNOWN;
200       case Token.STRING:
201         return getNamedTypeHelper(n, registry, typeParameters);
202       case Token.PIPE: {
203         // The way JSType.join works, Subtype|Supertype is equal to Supertype,
204         // so when programmers write un-normalized unions, we normalize them
205         // silently. We may also want to warn.
206         JSType union = JSType.BOTTOM;
207         for (Node child = n.getFirstChild(); child != null;
208              child = child.getNext()) {
209           // TODO(dimvar): When the union has many things, we join and throw
210           // away types, except the result of the last join. Very inefficient.
211           // Consider optimizing.
212           JSType nextType =
213               getTypeFromCommentHelper(child, registry, typeParameters);
214           if (nextType.isUnknown()) {
215             warn("This union type is equivalent to '?'.", n);
216             return JSType.UNKNOWN;
217           }
218           JSType nextUnion = JSType.join(union, nextType);
219           if (nextUnion.isBottom()) {
220             warnings.add(JSError.make(n, UNION_IS_UNINHABITABLE,
221                     nextType.toString(), union.toString()));
222             return JSType.UNKNOWN;
223           }
224           union = nextUnion;
225         }
226         return union;
227       }
228       case Token.BANG: {
229         JSType nullableType = getTypeFromCommentHelper(
230             n.getFirstChild(), registry, typeParameters);
231         if (nullableType.isTypeVariable()) {
232           warn("Cannot use ! to restrict type variable type.\n"
233               + "Prefer to make type argument non-nullable and add "
234               + "null explicitly where needed (e.g. through ?T or T|null)", n);
235         }
236         return nullableType.removeType(JSType.NULL);
237       }
238       case Token.QMARK: {
239         Node child = n.getFirstChild();
240         if (child == null) {
241           return JSType.UNKNOWN;
242         } else {
243           return JSType.join(JSType.NULL,
244               getTypeFromCommentHelper(child, registry, typeParameters));
245         }
246       }
247       case Token.STAR:
248         return JSType.TOP;
249       case Token.FUNCTION:
250         return getFunTypeHelper(n, registry, typeParameters);
251       default:
252         throw new IllegalArgumentException("Unsupported type exp: " +
253             Token.name(n.getType()) + " " + n.toStringTree());
254     }
255   }

257   private JSType getRecordTypeHelper(Node n, DeclaredTypeRegistry registry,
258       ImmutableList<String> typeParameters)
259       throws UnknownTypeException {
260     Map<String, Property> props = new LinkedHashMap<>();
261     for (Node propNode = n.getFirstChild().getFirstChild();
262          propNode != null;
263          propNode = propNode.getNext()) {
264       boolean isPropDeclared = propNode.getType() == Token.COLON;
265       Node propNameNode = isPropDeclared ? propNode.getFirstChild() : propNode;
266       String propName = propNameNode.getString();
267       if (propName.startsWith("'") || propName.startsWith("\"")) {
268         propName = propName.substring(1, propName.length() - 1);
269       }
270       JSType propType = !isPropDeclared
271           ? JSType.UNKNOWN
272           : getTypeFromCommentHelper(propNode.getLastChild(), registry, typeParameters);
273       Property prop;
274       if (!propType.isUnknown() && !propType.isTop()
275           && JSType.UNDEFINED.isSubtypeOf(propType)) {
276         prop = Property.makeOptional(null, propType, propType);
277       } else {
278         prop = Property.make(propType, propType);
279       }
280       props.put(propName, prop);
281     }
282     return JSType.fromObjectType(ObjectType.fromProperties(props));
283   }

285   private JSType getNamedTypeHelper(Node n, DeclaredTypeRegistry registry,
286       ImmutableList<String> outerTypeParameters)
287       throws UnknownTypeException {
288     String typeName = n.getString();
289     switch (typeName) {
290       case "boolean":
291         return JSType.BOOLEAN;
292       case "null":
293         return JSType.NULL;
294       case "number":
295         return JSType.NUMBER;
296       case "string":
297         return JSType.STRING;
298       case "undefined":
299       case "void":
300         return JSType.UNDEFINED;
301       case "Function":
302         return maybeMakeNullable(registry.getCommonTypes().qmarkFunction());
303       case "Object":
304         // We don't generally handle parameterized Object<...>, but we want to
305         // at least not warn about inexistent properties on it, so we type it
306         // as @dict.
307         return maybeMakeNullable(n.hasChildren() ? JSType.TOP_DICT : JSType.TOP_OBJECT);
308       default:
309         return lookupTypeByName(typeName, n, registry, outerTypeParameters);
310     }
311   }

313   private JSType lookupTypeByName(String name,
314       Node n, DeclaredTypeRegistry registry, ImmutableList<String> outerTypeParameters)
315       throws UnknownTypeException {
316     if (outerTypeParameters.contains(name)) {
317       return JSType.fromTypeVar(name);
318     }
319     Declaration decl = registry.getDeclaration(QualifiedName.fromQualifiedString(name), true);
320     if (decl == null) {
321       unknownTypeNames.put(n, name);
322       throw new UnknownTypeException("Unhandled type: " + name);
323     }
324     // It's either a typedef, an enum, a type variable or a nominal type
325     if (decl.getTypedef() != null) {
326       return getTypedefType(decl.getTypedef(), registry);
327     }
328     if (decl.getEnum() != null) {
329       return getEnumPropType(decl.getEnum(), registry);
330     }
331     if (decl.isTypeVar()) {
332       howmanyTypeVars++;
333       return decl.getTypeOfSimpleDecl();
334     }
335     if (decl.getNominal() != null) {
336       return getNominalTypeHelper(decl.getNominal(), n, registry, outerTypeParameters);
337     }
338     return JSType.UNKNOWN;
339   }

346   public void resolveTypedef(Typedef td, DeclaredTypeRegistry registry) {
347     Preconditions.checkState(td != null, "getTypedef should only be " +
348         "called when we know that the typedef is defined");
349     if (td.isResolved()) {
350       return;
351     }
352     JSTypeExpression texp = td.getTypeExpr();
353     JSType tdType;
354     if (texp == null) {
355       warn("Circular type definitions are not allowed.",
356           td.getTypeExprForErrorReporting().getRoot());
357       tdType = JSType.UNKNOWN;
358     } else {
359       tdType = getTypeFromJSTypeExpression(texp, registry, null);
360     }
361     td.resolveTypedef(tdType);
362   }

369   public void resolveEnum(EnumType e, DeclaredTypeRegistry registry) {
370     Preconditions.checkState(e != null, "getEnum should only be " +
371         "called when we know that the enum is defined");
372     if (e.isResolved()) {
373       return;
374     }
375     JSTypeExpression texp = e.getTypeExpr();
376     JSType enumeratedType;
377     if (texp == null) {
378       warn("Circular type definitions are not allowed.",
379           e.getTypeExprForErrorReporting().getRoot());
380       enumeratedType = JSType.UNKNOWN;
381     } else {
382       int numTypeVars = howmanyTypeVars;
383       enumeratedType = getTypeFromJSTypeExpression(texp, registry, null);
384       if (howmanyTypeVars > numTypeVars) {
385         warn("An enum type cannot include type variables.", texp.getRoot());
386         enumeratedType = JSType.UNKNOWN;
387         howmanyTypeVars = numTypeVars;
388       } else if (enumeratedType.isTop()) {
389         warn("An enum type cannot be *. " +
390             "Use ? if you do not want the elements checked.",
391             texp.getRoot());
392         enumeratedType = JSType.UNKNOWN;
393       } else if (enumeratedType.isUnion()) {
394         warn("An enum type cannot be a union type.", texp.getRoot());
395         enumeratedType = JSType.UNKNOWN;
396       }
397     }
398     e.resolveEnum(enumeratedType);
399   }

401   private JSType getNominalTypeHelper(RawNominalType rawType, Node n,
402       DeclaredTypeRegistry registry, ImmutableList<String> outerTypeParameters)
403       throws UnknownTypeException {
404     NominalType uninstantiated = rawType.getAsNominalType();
405     if (!rawType.isGeneric() && !n.hasChildren()) {
406       return rawType.getInstanceWithNullability(NULLABLE_TYPES_BY_DEFAULT);
407     }
408     ImmutableList.Builder<JSType> typeList = ImmutableList.builder();
409     if (n.hasChildren()) {
410       // Compute instantiation of polymorphic class/interface.
411       Preconditions.checkState(n.getFirstChild().isBlock());
412       for (Node child : n.getFirstChild().children()) {
413         typeList.add(
414             getTypeFromCommentHelper(child, registry, outerTypeParameters));
415       }
416     }
417     ImmutableList<JSType> typeArguments = typeList.build();
418     ImmutableList<String> typeParameters = rawType.getTypeParameters();
419     int typeArgsSize = typeArguments.size();
420     int typeParamsSize = typeParameters.size();
421     if (typeArgsSize != typeParamsSize) {
422       // We used to also warn when (typeArgsSize < typeParamsSize), but it
423       // happens so often that we stopped. Array, Object and goog.Promise are
424       // common culprits, but many other types as well.
425       if (typeArgsSize > typeParamsSize) {
426         warnings.add(JSError.make(
427             n, INVALID_GENERICS_INSTANTIATION,
428             uninstantiated.getName(), String.valueOf(typeParamsSize),
429             String.valueOf(typeArgsSize)));
430       }
431       return maybeMakeNullable(JSType.fromObjectType(ObjectType.fromNominalType(
432               uninstantiated.instantiateGenerics(
433                   fixLengthOfTypeList(typeParameters.size(), typeArguments)))));
434     }
435     return maybeMakeNullable(JSType.fromObjectType(ObjectType.fromNominalType(
436             uninstantiated.instantiateGenerics(typeArguments))));
437   }

439   private static List<JSType> fixLengthOfTypeList(
440       int desiredLength, List<JSType> typeList) {
441     int length = typeList.size();
442     if (length == desiredLength) {
443       return typeList;
444     }
445     ImmutableList.Builder<JSType> builder = ImmutableList.builder();
446     for (int i = 0; i < desiredLength; i++) {
447       builder.add(i < length ? typeList.get(i) : JSType.UNKNOWN);
448     }
449     return builder.build();
450   }

452   // Don't confuse with getFunTypeFromAtTypeJsdoc; the function below computes a
453   // type that doesn't have an associated AST node.
454   private JSType getFunTypeHelper(Node jsdocNode, DeclaredTypeRegistry registry,
455       ImmutableList<String> typeParameters) throws UnknownTypeException {
456     FunctionTypeBuilder builder = new FunctionTypeBuilder();
457     fillInFunTypeBuilder(jsdocNode, null, registry, typeParameters, builder);
458     return registry.getCommonTypes().fromFunctionType(builder.buildFunction());
459   }

461   private void fillInFunTypeBuilder(
462       Node jsdocNode, RawNominalType ownerType, DeclaredTypeRegistry registry,
463       ImmutableList<String> typeParameters, FunctionTypeBuilder builder)
464       throws UnknownTypeException {
465     Node child = jsdocNode.getFirstChild();
466     NominalType builtinObject = registry.getCommonTypes().getObjectType();
467     if (child.getType() == Token.THIS) {
468       if (ownerType == null) {
469         NominalType nt = getNominalType(child.getFirstChild(), registry, typeParameters);
470         builder.addReceiverType(nt == null ? builtinObject : nt);
471       }
472       child = child.getNext();
473     } else if (child.getType() == Token.NEW) {
474       NominalType nt = getNominalType(child.getFirstChild(), registry, typeParameters);
475       builder.addNominalType(nt == null ? builtinObject : nt);
476       child = child.getNext();
477     }
478     if (child.getType() == Token.PARAM_LIST) {
479       for (Node arg = child.getFirstChild(); arg != null; arg = arg.getNext()) {
480         try {
481           switch (arg.getType()) {
482             case Token.EQUALS:
483               builder.addOptFormal(getTypeFromCommentHelper(
484                   arg.getFirstChild(), registry, typeParameters));
485               break;
486             case Token.ELLIPSIS:
487               Node restNode = arg.getFirstChild();
488               builder.addRestFormals(restNode == null ? JSType.UNKNOWN :
489                   getTypeFromCommentHelper(restNode, registry, typeParameters));
490               break;
491             default:
492               builder.addReqFormal(
493                   getTypeFromCommentHelper(arg, registry, typeParameters));
494               break;
495           }
496         } catch (FunctionTypeBuilder.WrongParameterOrderException e) {
497           warn("Wrong parameter order: required parameters are first, " +
498               "then optional, then varargs", jsdocNode);
499           builder.addPlaceholderFormal();
500         }
501       }
502       child = child.getNext();
503     }
504     builder.addRetType(
505         getTypeFromCommentHelper(child, registry, typeParameters));
506   }

527   private ImmutableSet<NominalType> getInterfacesHelper(
528       JSDocInfo jsdoc, DeclaredTypeRegistry registry,
529       ImmutableList<String> typeParameters, boolean implementedIntfs) {
530     ImmutableSet.Builder<NominalType> builder = ImmutableSet.builder();
531     for (JSTypeExpression texp : (implementedIntfs ?
532           jsdoc.getImplementedInterfaces() :
533           jsdoc.getExtendedInterfaces())) {
534       Node expRoot = texp.getRoot();
535       JSType interfaceType =
536           getMaybeTypeFromComment(expRoot, registry, typeParameters);
537       if (interfaceType != null) {
538         NominalType nt = interfaceType.getNominalTypeIfSingletonObj();
539         if (nt != null && nt.isInterface()) {
540           builder.add(nt);
541         } else {
542           String errorMsg = implementedIntfs ?
543               "Cannot implement non-interface" :
544               "Cannot extend non-interface";
545           warn(errorMsg, expRoot);
546         }
547       }
548     }
549     return builder.build();
550   }

562   /**
563    * Consumes either a "classic" function jsdoc with @param, @return, etc,
564    * or a jsdoc with @type {function ...} and finds the types of the formal
565    * parameters and the return value. It returns a builder because the callers
566    * of this function must separately handle @constructor, @interface, etc.
567    *
568    * constructorType is non-null iff this function is a constructor or
569    * interface declaration.
570    */
571   public FunctionAndSlotType getFunctionType(
572       JSDocInfo jsdoc, String functionName, Node declNode,
573       RawNominalType constructorType, RawNominalType ownerType,
574       DeclaredTypeRegistry registry) {
575     FunctionTypeBuilder builder = new FunctionTypeBuilder();
576     if (ownerType != null) {
577       builder.addReceiverType(ownerType.getAsNominalType());
578     }
579     try {
580       if (jsdoc != null && jsdoc.getType() != null) {
581         JSType simpleType = getDeclaredTypeOfNode(jsdoc, ownerType, registry);
582         if (simpleType.isUnknown() || simpleType.isTop()) {
583           return qmarkFunctionDeclared;
584         }
585         FunctionType funType = simpleType.getFunType();
586         if (funType != null) {
587           JSType slotType = simpleType.isFunctionType() ? null : simpleType;
588           DeclaredFunctionType declType = funType.toDeclaredFunctionType();
589           if (ownerType != null) {
590             declType = declType.withReceiverType(ownerType.getAsNominalType());
591           }
592           return new FunctionAndSlotType(slotType, declType);
593         } else {
594           warn("The function is annotated with a non-function jsdoc. " +
595               "Ignoring jsdoc.", declNode);
596           jsdoc = null;
597         }
598       }
599       DeclaredFunctionType declType = getFunTypeFromTypicalFunctionJsdoc(
600           jsdoc, functionName, declNode,
601           constructorType, ownerType, registry, builder, false);
602       return new FunctionAndSlotType(null, declType);
603     } catch (FunctionTypeBuilder.WrongParameterOrderException e) {
604       warn("Wrong parameter order: required parameters are first, " +
605           "then optional, then varargs. Ignoring jsdoc.", declNode);
606       return qmarkFunctionDeclared;
607     }
608   }

610   private static class ParamIterator {
611     /** The parameter names from the JSDocInfo. Only set if 'params' is null. */
612     Iterator<String> paramNames;
613 
614     /**
615      * The PARAM_LIST node containing the function parameters. Only set if
616      * 'paramNames' is null.
617      */
618     Node params;
619 
620     int index = -1;
621 
622     ParamIterator(Node params, JSDocInfo jsdoc) {
623       Preconditions.checkArgument(params != null || jsdoc != null);
624       if (params != null) {
625         this.params = params;
626         this.paramNames = null;
627       } else {
628         this.params = null;
629         this.paramNames = jsdoc.getParameterNames().iterator();
630       }
631     }
632 
633     boolean hasNext() {
634       if (paramNames != null) {
635         return paramNames.hasNext();
636       }
637       return index + 1 < params.getChildCount();
638     }
639 
640     String nextString() {
641       if (paramNames != null) {
642         return paramNames.next();
643       }
644       index++;
645       return params.getChildAtIndex(index).getString();
646     }
647 
648     Node getNode() {
649       if (paramNames != null) {
650         return null;
651       }
652       return params.getChildAtIndex(index);
653     }
654   }

656   private DeclaredFunctionType getFunTypeFromTypicalFunctionJsdoc(
657       JSDocInfo jsdoc, String functionName, Node funNode,
658       RawNominalType constructorType, RawNominalType ownerType,
659       DeclaredTypeRegistry registry, FunctionTypeBuilder builder,
660       boolean ignoreJsdoc /* for when the jsdoc is malformed */) {
661     Preconditions.checkArgument(!ignoreJsdoc || jsdoc == null);
662     Preconditions.checkArgument(!ignoreJsdoc || funNode.isFunction());
663     ImmutableList.Builder<String> typeParamsBuilder = new ImmutableList.Builder<>();;
664     ImmutableList<String> typeParameters = ImmutableList.of();
665     Node parent = funNode.getParent();
666 
667     // TODO(dimvar): need more @template warnings
668     // - warn for multiple @template annotations
669     // - warn for @template annotation w/out usage
670 
671     if (jsdoc != null) {
672       typeParamsBuilder.addAll(jsdoc.getTemplateTypeNames());
673       // We don't properly support the type transformation language; we treat
674       // its type variables as ordinary type variables.
675       typeParamsBuilder.addAll(jsdoc.getTypeTransformations().keySet());
676       typeParameters = typeParamsBuilder.build();
677       if (!typeParameters.isEmpty()) {
678         if (parent.isSetterDef() || parent.isGetterDef()) {
679           ignoreJsdoc = true;
680           jsdoc = null;
681           warn("@template can't be used with getters/setters", funNode);
682         } else {
683           builder.addTypeParameters(typeParameters);
684         }
685       }
686     }
687     if (ownerType != null) {
688       typeParamsBuilder.addAll(ownerType.getTypeParameters());
689       typeParameters = typeParamsBuilder.build();
690     }
691 
692     fillInFormalParameterTypes(
693         jsdoc, funNode, typeParameters, registry, builder, ignoreJsdoc);
694     fillInReturnType(
695         jsdoc, funNode, parent, typeParameters, registry, builder, ignoreJsdoc);
696     if (jsdoc == null) {
697       return builder.buildDeclaration();
698     }
699 
700     // Look at other annotations, eg, @constructor
701     NominalType parentClass = getMaybeParentClass(
702         jsdoc, functionName, funNode, typeParameters, registry);
703     ImmutableSet<NominalType> implementedIntfs = getImplementedInterfaces(
704         jsdoc, registry, typeParameters);
705     if (constructorType == null && jsdoc.isConstructorOrInterface()) {
706       // Anonymous type, don't register it.
707       return builder.buildDeclaration();
708     } else if (jsdoc.isConstructor()) {
709       handleConstructorAnnotation(functionName, funNode, constructorType,
710           parentClass, implementedIntfs, registry, builder);
711     } else if (jsdoc.isInterface()) {
712       handleInterfaceAnnotation(jsdoc, functionName, funNode, constructorType,
713           implementedIntfs, typeParameters, registry, builder);
714     } else if (!implementedIntfs.isEmpty()) {
715       warnings.add(JSError.make(
716           funNode, IMPLEMENTS_WITHOUT_CONSTRUCTOR, functionName));
717     }
718 
719     if (jsdoc.hasThisType() && ownerType == null) {
720       Node thisRoot = jsdoc.getThisType().getRoot();
721       Preconditions.checkState(thisRoot.getType() == Token.BANG);
722       Node thisNode = thisRoot.getFirstChild();
723       // JsDocInfoParser wraps @this types with !. But we warn when we see !T,
724       // and we don't want to warn for a ! that was automatically inserted.
725       // So, we bypass the ! here.
726       JSType thisType = getMaybeTypeFromComment(thisNode, registry, typeParameters);
727       if (thisType != null) {
728         thisType = thisType.removeType(JSType.NULL);
729       }
730       // TODO(dimvar): thisType may be non-null but have a null
731       // thisTypeAsNominal.
732       // We currently only support nominal types for the receiver type, but
733       // people use other types as well: unions, records, etc.
734       // For now, we just use the generic Object for these.
735       NominalType nt = thisType == null ? null : thisType.getNominalTypeIfSingletonObj();
736       NominalType builtinObject = registry.getCommonTypes().getObjectType();
737       builder.addReceiverType(nt == null ? builtinObject : nt);
738     }
739 
740     return builder.buildDeclaration();
741   }

743   private void fillInFormalParameterTypes(
744       JSDocInfo jsdoc, Node funNode,
745       ImmutableList<String> typeParameters,
746       DeclaredTypeRegistry registry, FunctionTypeBuilder builder,
747       boolean ignoreJsdoc /* for when the jsdoc is malformed */) {
748     boolean ignoreFunNode  = !funNode.isFunction();
749     Node params = ignoreFunNode ? null : funNode.getFirstChild().getNext();
750     ParamIterator iterator = new ParamIterator(params, jsdoc);
751     while (iterator.hasNext()) {
752       String pname = iterator.nextString();
753       Node param = iterator.getNode();
754       ParameterKind p = ParameterKind.REQUIRED;
755       if (param != null && convention.isOptionalParameter(param)) {
756         p = ParameterKind.OPTIONAL;
757       } else if (param != null && convention.isVarArgsParameter(param)) {
758         p = ParameterKind.REST;
759       }
760       ParameterType inlineParamType = (ignoreJsdoc || ignoreFunNode || param.getJSDocInfo() == null)
761           ? null : parseParameter(param.getJSDocInfo().getType(), p, registry, typeParameters);
762       ParameterType fnParamType = inlineParamType;
763       JSTypeExpression jsdocExp = jsdoc == null ? null : jsdoc.getParameterType(pname);
764       if (jsdocExp != null) {
765         if (inlineParamType == null) {
766           fnParamType = parseParameter(jsdocExp, p, registry, typeParameters);
767         } else {
768           warn("Found two JsDoc comments for formal parameter " + pname, param);
769         }
770       }
771       JSType t  = null;
772       if (fnParamType != null) {
773         p = fnParamType.kind;
774         t = fnParamType.type;
775       }
776       switch (p) {
777           case REQUIRED:
778             builder.addReqFormal(t);
779             break;
780           case OPTIONAL:
781             builder.addOptFormal(t);
782             break;
783           case REST:
784             builder.addRestFormals(t != null ? t : JSType.UNKNOWN);
785             break;
786       }
787     }
788   }

790   private void fillInReturnType(
791       JSDocInfo jsdoc, Node funNode, Node parent,
792       ImmutableList<String> typeParameters,
793       DeclaredTypeRegistry registry, FunctionTypeBuilder builder,
794       boolean ignoreJsdoc /* for when the jsdoc is malformed */) {
795     JSDocInfo inlineRetJsdoc =
796         ignoreJsdoc ? null : funNode.getFirstChild().getJSDocInfo();
797     JSTypeExpression retTypeExp = jsdoc == null ? null : jsdoc.getReturnType();
798     if (parent.isSetterDef() && retTypeExp == null) {
799       // inline returns for getters/setters are not parsed
800       builder.addRetType(JSType.UNDEFINED);
801     } else if (inlineRetJsdoc != null) {
802       builder.addRetType(
803           getDeclaredTypeOfNode(inlineRetJsdoc, registry, typeParameters));
804       if (retTypeExp != null) {
805         warn("Found two JsDoc comments for the return type", funNode);
806       }
807     } else {
808       builder.addRetType(
809           getTypeFromJSTypeExpression(retTypeExp, registry, typeParameters));
810     }
811   }

813   private NominalType getMaybeParentClass(
814       JSDocInfo jsdoc, String functionName, Node funNode,
815       ImmutableList<String> typeParameters, DeclaredTypeRegistry registry) {
816     if (!jsdoc.hasBaseType()) {
817       return null;
818     }
819     if (!jsdoc.isConstructor()) {
820       warnings.add(JSError.make(
821           funNode, EXTENDS_NOT_ON_CTOR_OR_INTERF, functionName));
822       return null;
823     }
824     Node docNode = jsdoc.getBaseType().getRoot();
825     JSType extendedType =
826         getMaybeTypeFromComment(docNode, registry, typeParameters);
827     if (extendedType == null) {
828       return null;
829     }
830     NominalType parentClass = extendedType.getNominalTypeIfSingletonObj();
831     if (parentClass != null && parentClass.isClass()) {
832       return parentClass;
833     }
834     if (parentClass == null) {
835       warnings.add(JSError.make(funNode, EXTENDS_NON_OBJECT,
836               functionName, extendedType.toString()));
837     } else {
838       Preconditions.checkState(parentClass.isInterface());
839       warnings.add(JSError.make(funNode, CONFLICTING_EXTENDED_TYPE,
840               "constructor", functionName));
841     }
842     return null;
843   }

845   private void handleConstructorAnnotation(
846       String functionName, Node funNode, RawNominalType constructorType,
847       NominalType parentClass, ImmutableSet<NominalType> implementedIntfs,
848       DeclaredTypeRegistry registry, FunctionTypeBuilder builder) {
849     String className = constructorType.toString();
850     NominalType builtinObject = registry.getCommonTypes().getObjectType();
851     if (parentClass == null && !functionName.equals("Object")) {
852       parentClass = builtinObject;
853     }
854     if (parentClass != null) {
855       if (!constructorType.addSuperClass(parentClass)) {
856         warnings.add(JSError.make(funNode, INHERITANCE_CYCLE, className));
857       } else if (parentClass != builtinObject) {
858         if (constructorType.isStruct() && !parentClass.isStruct()) {
859           warnings.add(JSError.make(
860               funNode, CONFLICTING_SHAPE_TYPE, "struct", className));
861         } else if (constructorType.isDict() && !parentClass.isDict()) {
862           warnings.add(JSError.make(
863               funNode, CONFLICTING_SHAPE_TYPE, "dict", className));
864         }
865       }
866     }
867     if (constructorType.isDict() && !implementedIntfs.isEmpty()) {
868       warnings.add(JSError.make(funNode, DICT_IMPLEMENTS_INTERF, className));
869     }
870     boolean noCycles = constructorType.addInterfaces(implementedIntfs);
871     Preconditions.checkState(noCycles);
872     builder.addNominalType(constructorType.getAsNominalType());
873   }
874 
875   private void handleInterfaceAnnotation(
876       JSDocInfo jsdoc, String functionName, Node funNode,
877       RawNominalType constructorType,
878       ImmutableSet<NominalType> implementedIntfs,
879       ImmutableList<String> typeParameters,
880       DeclaredTypeRegistry registry, FunctionTypeBuilder builder) {
881     if (!implementedIntfs.isEmpty()) {
882       warnings.add(JSError.make(
883           funNode, CONFLICTING_IMPLEMENTED_TYPE, functionName));
884     }
885     boolean noCycles = constructorType.addInterfaces(
886         getExtendedInterfaces(jsdoc, registry, typeParameters));
887     if (!noCycles) {
888       warnings.add(JSError.make(
889           funNode, INHERITANCE_CYCLE, constructorType.toString()));
890     }
891     builder.addNominalType(constructorType.getAsNominalType());
892   }

897     if (funJsdoc == null) {
898       return false;

901     Node jsdocNode = texp == null ? null : texp.getRoot();
902     return jsdocNode != null && jsdocNode.getType() == Token.ELLIPSIS;

910   private ParameterType parseParameter(
911       JSTypeExpression jsdoc, ParameterKind p,
912       DeclaredTypeRegistry registry, ImmutableList<String> typeParameters) {
913     if (jsdoc == null) {
914       return null;
915     }
916     return parseParameter(jsdoc.getRoot(), p, registry, typeParameters);
917   }

919   private ParameterType parseParameter(
920       Node jsdoc, ParameterKind p,
921       DeclaredTypeRegistry registry, ImmutableList<String> typeParameters) {
922     if (jsdoc == null) {
923       return null;
924     }
925     switch (jsdoc.getType()) {
926       case Token.EQUALS:
927         p = ParameterKind.OPTIONAL;
928         jsdoc = jsdoc.getFirstChild();
929         break;
930       case Token.ELLIPSIS:
931         p = ParameterKind.REST;
932         jsdoc = jsdoc.getFirstChild();
933         break;
934     }
935     JSType t = getMaybeTypeFromComment(jsdoc, registry, typeParameters);
936     return new ParameterType(t, p);
937   }
