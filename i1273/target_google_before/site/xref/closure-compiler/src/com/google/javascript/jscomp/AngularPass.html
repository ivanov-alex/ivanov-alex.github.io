29  /**
30   * Compiler pass for AngularJS-specific needs. Generates {@code $inject} \
31   * properties for functions (class constructors, wrappers, etc) annotated with
32   * @ngInject. Without this pass, AngularJS will not work properly if variable
33   * renaming is enabled, because the function arguments will be renamed.
34   * @see http://docs.angularjs.org/tutorial/step_05#a-note-on-minification
35   *
36   * <p>For example, the following code:</p>
37   * <pre>{@code
38   *
39   * /** @ngInject * /
40   * function Controller(dependency1, dependency2) {
41   *   // do something
42   * }
43   *
44   * }</pre>
45   *
46   * <p>will be transformed into:
47   * <pre>{@code
48   *
49   * function Controller(dependency1, dependency2) {
50   *   // do something
51   * }
52   * Controller.$inject = ['dependency1', 'dependency2'];
53   *
54   * }</pre>
55   *
56   * <p> This pass also supports assignments of function expressions to variables
57   * like:
58   * <pre>{@code
59   *
60   * /** @ngInject * /
61   * var filter = function(a, b) {};
62   *
63   * var ns = {};
64   * /** @ngInject * /
65   * ns.method = function(a,b,c) {};
66   *
67   * /** @ngInject * /
68   * var shorthand = ns.method2 = function(a,b,c,) {}
69   *
70   * }</pre>
71   */
72  class AngularPass extends AbstractPostOrderCallback
73      implements HotSwapCompilerPass {
74    final AbstractCompiler compiler;
75  
76    /** Nodes annotated with @ngInject */
77    private final List<NodeContext> injectables = new ArrayList<>();
78  
79    public AngularPass(AbstractCompiler compiler) {
80      this.compiler = compiler;
81    }
82  
83    public static final String INJECT_PROPERTY_NAME = "$inject";
84  
85    static final DiagnosticType INJECT_IN_NON_GLOBAL_OR_BLOCK_ERROR =
86        DiagnosticType.error("JSC_INJECT_IN_NON_GLOBAL_OR_BLOCK_ERROR",
87            "@ngInject only applies to functions defined in blocks or " +
88            "global scope.");
89  
90    static final DiagnosticType INJECT_NON_FUNCTION_ERROR =
91        DiagnosticType.error("JSC_INJECT_NON_FUNCTION_ERROR",
92            "@ngInject can only be used when defining a function or " +
93            "assigning a function expression.");
94  
95    static final DiagnosticType INJECTED_FUNCTION_HAS_DESTRUCTURED_PARAM =
96        DiagnosticType.error("JSC_INJECTED_FUNCTION_HAS_DESTRUCTURED_PARAM",
97            "@ngInject cannot be used on functions containing "
98            + "destructured parameter.");
99  
100   static final DiagnosticType INJECTED_FUNCTION_HAS_DEFAULT_VALUE =
101       DiagnosticType.error("JSC_INJECTED_FUNCTION_HAS_DEFAULT_VALUE",
102           "@ngInject cannot be used on functions containing default value.");
103 
104   @Override
105   public void process(Node externs, Node root) {
106     hotSwapScript(root, null);
107   }
108 
109   @Override
110   public void hotSwapScript(Node scriptRoot, Node originalRoot) {
111     // Traverses AST looking for nodes annotated with @ngInject.
112     NodeTraversal.traverseEs6(compiler, scriptRoot, this);
113     boolean codeChanged = false;
114     // iterates through annotated nodes adding $inject property to elements.
115     for (NodeContext entry : injectables) {
116       String name = entry.getName();
117       Node fn = entry.getFunctionNode();
118       List<Node> dependencies = createDependenciesList(fn);
119       // skips entry if it does not have any dependencies.
120       if (dependencies.isEmpty()) {
121         continue;
122       }
123       Node dependenciesArray = IR.arraylit(dependencies.toArray(
124           new Node[dependencies.size()]));
125       // creates `something.$inject = ['param1', 'param2']` node.
126       Node statement = IR.exprResult(
127           IR.assign(
128               IR.getelem(
129                   NodeUtil.newQName(compiler, name),
130                   IR.string(INJECT_PROPERTY_NAME)),
131               dependenciesArray
132           )
133       );
134       NodeUtil.setDebugInformation(statement, entry.getNode(), name);
135 
136       // adds `something.$inject = [...]` node after the annotated node or the following
137       // goog.inherits call.
138       Node insertionPoint = entry.getTarget();
139       Node next = insertionPoint.getNext();
140       while (next != null
141              && NodeUtil.isExprCall(next)
142              && compiler.getCodingConvention().getClassesDefinedByCall(
143                  next.getFirstChild()) != null) {
144         insertionPoint = next;
145         next = insertionPoint.getNext();
146       }
147 
148       insertionPoint.getParent().addChildAfter(statement, insertionPoint);
149       codeChanged = true;
150     }
151     if (codeChanged) {
152       compiler.reportCodeChange();
153     }
154   }

156   /**
157    * Given a FUNCTION node returns array of STRING nodes representing function
158    * parameters.
159    * @param n the FUNCTION node.
160    * @return STRING nodes.
161    */
162   private List<Node> createDependenciesList(Node n) {
163     Preconditions.checkArgument(n.isFunction());
164     Node params = NodeUtil.getFunctionParameters(n);
165     if (params != null) {
166       return createStringsFromParamList(params);
167     }
168     return new ArrayList<>();
169   }
170 
171   /**
172    * Given a PARAM_LIST node creates an array of corresponding STRING nodes.
173    * @param params PARAM_LIST node.
174    * @return array of STRING nodes.
175    */
176   private List<Node> createStringsFromParamList(Node params) {
177     Node param = params.getFirstChild();
178     ArrayList<Node> names = new ArrayList<>();
179     while (param != null) {
180       if (param.isName()) {
181         names.add(IR.string(param.getString()).srcref(param));
182       } else if (param.isDestructuringPattern()) {
183         compiler.report(JSError.make(param,
184             INJECTED_FUNCTION_HAS_DESTRUCTURED_PARAM));
185         return new ArrayList<>();
186       } else if (param.isDefaultValue()) {
187         compiler.report(JSError.make(param,
188             INJECTED_FUNCTION_HAS_DEFAULT_VALUE));
189         return new ArrayList<>();
190       }
191       param = param.getNext();
192     }
193     return names;
194   }

196   @Override
197   public void visit(NodeTraversal t, Node n, Node parent) {
198     JSDocInfo docInfo = n.getJSDocInfo();
199     if (docInfo != null && docInfo.isNgInject()) {
200       addNode(n, t);
201     }
202   }
203 
204   /**
205    * Add node to the list of injectables.
206    * @param n node to add.
207    * @param t node traversal instance.
208    */
209   private void addNode(Node n, NodeTraversal t) {
210     Node target = null;
211     Node fn = null;
212     String name = null;
213 
214     switch (n.getType()) {
215       // handles assignment cases like:
216       // a = function() {}
217       // a = b = c = function() {}
218       case Token.ASSIGN:
219         name = n.getFirstChild().getQualifiedName();
220         // last node of chained assignment.
221         fn = n;
222         while (fn.isAssign()) {
223           fn = fn.getLastChild();
224         }
225         target = n.getParent();
226         break;
227 
228       // handles function case:
229       // function fnName() {}
230       case Token.FUNCTION:
231         name = NodeUtil.getFunctionName(n);
232         fn = n;
233         target = n;
234         break;
235 
236       // handles var declaration cases like:
237       // var a = function() {}
238       // var a = b = function() {}
239       case Token.VAR:
240       case Token.LET:
241       case Token.CONST:
242         name = n.getFirstChild().getString();
243         // looks for a function node.
244         fn = getDeclarationRValue(n);
245         target = n;
246         break;
247 
248       // handles class method case:
249       // class clName(){
250       //   constructor(){}
251       //   someMethod(){} <===
252       // }
253       case Token.MEMBER_FUNCTION_DEF:
254         Node parent = n.getParent();
255         if (parent.isClassMembers()){
256           Node classNode = parent.getParent();
257           String midPart = n.isStaticMember() ? "." : ".prototype.";
258           name = NodeUtil.getClassName(classNode) + midPart + n.getString();
259           if (n.getString().equals("constructor")) {
260             name = NodeUtil.getClassName(classNode);
261           }
262           fn = n.getFirstChild();
263           if (classNode.getParent().isAssign() || classNode.getParent().isName()) {
264             target = classNode.getParent().getParent();
265           } else {
266             target = classNode;
267           }
268         }
269         break;
270     }
271 
272     if (fn == null || !fn.isFunction()) {
273       compiler.report(t.makeError(n, INJECT_NON_FUNCTION_ERROR));
274       return;
275     }
276     // report an error if the function declaration did not take place in a block or global scope
277     if (!target.getParent().isScript() && !target.getParent().isBlock()) {
278       compiler.report(t.makeError(n, INJECT_IN_NON_GLOBAL_OR_BLOCK_ERROR));
279       return;
280     }
281     // checks that name is present, which must always be the case unless the
282     // compiler allowed a syntax error or a dangling anonymous function
283     // expression.
284     Preconditions.checkNotNull(name);
285     // registers the node.
286     injectables.add(new NodeContext(name, n, fn, target));
287   }

289   /**
290    * Given a VAR node (variable declaration) returns the node of initial value.
291    *
292    * <pre>{@code
293    * var x;  // null
294    * var y = "value"; // STRING "value" node
295    * var z = x = y = function() {}; // FUNCTION node
296    * }</pre>
297    * @param n VAR node.
298    * @return the assigned initial value, or the rightmost rvalue of an assignment
299    * chain, or null.
300    */
301   private static Node getDeclarationRValue(Node n) {
302     Preconditions.checkNotNull(n);
303     Preconditions.checkArgument(NodeUtil.isNameDeclaration(n));
304     n = n.getFirstChild().getFirstChild();
305     if (n == null) {
306       return null;
307     }
308     while (n.isAssign()) {
309       n = n.getLastChild();
310     }
311     return n;
312   }

314   static class NodeContext {
315     /** Name of the function/object. */
316     private final String name;
317     /** Node jsDoc is attached to. */
318     private final Node node;
319     /** Function node */
320     private final Node functionNode;
321     /** Node after which to inject the new code */
322     private final Node target;
323 
324     public NodeContext(String name, Node node, Node functionNode, Node target) {
325       this.name = name;
326       this.node = node;
327       this.functionNode = functionNode;
328       this.target = target;
329     }
330 
331     /**
332      * @return the name.
333      */
334     public String getName() {
335       return name;
336     }
337 
338     /**
339      * @return the node.
340      */
341     public Node getNode() {
342       return node;
343     }
344 
345     /**
346      * @return the context.
347      */
348     public Node getFunctionNode() {
349       return functionNode;
350     }
351 
352     /**
353      * @return the context.
354      */
355     public Node getTarget() {
356       return target;
357     }
358   }
