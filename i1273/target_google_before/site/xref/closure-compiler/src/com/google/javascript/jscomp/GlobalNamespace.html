172    * A simple data class that contains the information necessary to inspect
173    * a node for changes to the global namespace.

175   static class AstChange {
176     final JSModule module;
177     final Scope scope;
178     final Node node;
179 
180     AstChange(JSModule module, Scope scope, Node node) {
181       this.module = module;
182       this.scope = scope;
183       this.node = node;
184     }
185 
186     @Override
187     public boolean equals(Object obj) {
188       Preconditions.checkState(obj instanceof AstChange);
189       AstChange other = (AstChange) obj;
190       return Objects.equals(this.module, other.module)
191           && Objects.equals(this.scope, other.scope)
192           && Objects.equals(this.node, other.node);
193     }
194 
195     @Override
196     public int hashCode() {
197       return Objects.hash(this.module, this.scope, this.node);
198     }

201   /**
202    * If the client adds new nodes to the AST, scan these new nodes
203    * to see if they've added any references to the global namespace.
204    * @param newNodes New nodes to check.
205    */
206   void scanNewNodes(Set<AstChange> newNodes) {
207     BuildGlobalNamespace builder = new BuildGlobalNamespace();
208 
209     for (AstChange info : newNodes) {
210       if (!info.node.isQualifiedName() && !NodeUtil.isObjectLitKey(info.node)) {
211         continue;
212       }
213       scanFromNode(builder, info.module, info.scope, info.node);
214     }
215   }
216 
217   private void scanFromNode(
218     BuildGlobalNamespace builder, JSModule module, Scope scope, Node n) {
219     // Check affected parent nodes first.
220     if (n.isName() || n.isGetProp()) {
221       scanFromNode(builder, module, scope, n.getParent());
222     }
223     builder.collect(module, scope, n);
224   }
225 
226   /**
227    * Builds the namespace lazily.
228    */
229   private void process() {
230     if (externsRoot != null) {
231       inExterns = true;
232       NodeTraversal.traverseEs6(compiler, externsRoot, new BuildGlobalNamespace());
233     }
234     inExterns = false;
235 
236     NodeTraversal.traverseEs6(compiler, root, new BuildGlobalNamespace());
237     generated = true;
238   }

240   /**
241    * Determines whether a name reference in a particular scope is a global name
242    * reference.
243    *
244    * @param name A variable or property name (e.g. "a" or "a.b.c.d")
245    * @param s The scope in which the name is referenced
246    * @return Whether the name reference is a global name reference
247    */
248   private boolean isGlobalNameReference(String name, Scope s) {
249     String topVarName = getTopVarName(name);
250     return isGlobalVarReference(topVarName, s);
251   }
252 
253   /**
254    * Gets the top variable name from a possibly namespaced name.
255    *
256    * @param name A variable or qualified property name (e.g. "a" or "a.b.c.d")
257    * @return The top variable name (e.g. "a")
258    */
259   private static String getTopVarName(String name) {
260     int firstDotIndex = name.indexOf('.');
261     return firstDotIndex == -1 ? name : name.substring(0, firstDotIndex);
262   }
263 
264   /**
265    * Determines whether a variable name reference in a particular scope is a
266    * global variable reference.
267    *
268    * @param name A variable name (e.g. "a")
269    * @param s The scope in which the name is referenced
270    * @return Whether the name reference is a global variable reference
271    */
272   private boolean isGlobalVarReference(String name, Scope s) {
273     Var v = s.getVar(name);
274     if (v == null && externsScope != null) {
275       v = externsScope.getVar(name);
276     }
277     return v != null && !v.isLocal();
278   }

283    * Builds a tree representation of the global namespace. Omits prototypes.
284    */
285   private class BuildGlobalNamespace implements NodeTraversal.Callback {
286 
287     BuildGlobalNamespace() {
288     }
289 
290     @Override
291     public void visit(NodeTraversal t, Node n, Node parent) {}
292 
293     /** Collect the references in pre-order. */
294     @Override
295     public boolean shouldTraverse(NodeTraversal t, Node n, Node parent) {
296       // If we are traversing the externs, then we save a pointer to the scope
297       // generated by them, so that we can do lookups in it later.
298       if (externsRoot != null && n == externsRoot) {
299         externsScope = t.getScope();
300       }
301 
302       collect(t.getModule(), t.getScope(), n);
303 
304       return true;
305     }
306 
307     public void collect(JSModule module, Scope scope, Node n) {
308       Node parent = n.getParent();
309 
310       String name;
311       boolean isSet = false;
312       Name.Type type = Name.Type.OTHER;
313       boolean isPropAssign = false;
314 
315       switch (n.getType()) {
316         case Token.GETTER_DEF:
317         case Token.SETTER_DEF:
318         case Token.STRING_KEY:
319         case Token.MEMBER_FUNCTION_DEF:
320           // This may be a key in an object literal declaration.
321           name = null;
322           if (parent != null && parent.isObjectLit()) {
323             name = getNameForObjLitKey(n);
324           } else if (parent != null && parent.isClassMembers()) {
325             name = getNameForClassMembers(n);
326           }
327           if (name == null) {
328             return;
329           }
330           isSet = true;
331           switch (n.getType()) {
332             case Token.MEMBER_FUNCTION_DEF:
333             case Token.STRING_KEY:
334               type = getValueType(n.getFirstChild());
335               break;
336             case Token.GETTER_DEF:
337               type = Name.Type.GET;
338               break;
339             case Token.SETTER_DEF:
340               type = Name.Type.SET;
341               break;
342             default:
343               throw new IllegalStateException("unexpected:" + n);
344           }
345           break;
346         case Token.NAME:
347           // This may be a variable get or set.
348           if (parent != null) {
349             switch (parent.getType()) {
350               case Token.VAR:
351               case Token.LET:
352               case Token.CONST:
353                 isSet = true;
354                 Node rvalue = n.getFirstChild();
355                 type = rvalue == null ? Name.Type.OTHER : getValueType(rvalue);
356                 break;
357               case Token.ASSIGN:
358                 if (parent.getFirstChild() == n) {
359                   isSet = true;
360                   type = getValueType(n.getNext());
361                 }
362                 break;
363               case Token.GETPROP:
364                 return;
365               case Token.FUNCTION:
366                 Node grandparent = parent.getParent();
367                 if (grandparent == null || NodeUtil.isFunctionExpression(parent)) {
368                   return;
369                 }
370                 isSet = true;
371                 type = Name.Type.FUNCTION;
372                 break;
373               case Token.CATCH:
374               case Token.INC:
375               case Token.DEC:
376                 isSet = true;
377                 type = Name.Type.OTHER;
378                 break;
379               case Token.CLASS:
380                 isSet = true;
381                 type = Name.Type.CLASS;
382                 break;
383               default:
384                 if (NodeUtil.isAssignmentOp(parent) &&
385                     parent.getFirstChild() == n) {
386                   isSet = true;
387                   type = Name.Type.OTHER;
388                 }
389             }
390           }
391           name = n.getString();
392           break;
393         case Token.GETPROP:
394           // This may be a namespaced name get or set.
395           if (parent != null) {
396             switch (parent.getType()) {
397               case Token.ASSIGN:
398                 if (parent.getFirstChild() == n) {
399                   isSet = true;
400                   type = getValueType(n.getNext());
401                   isPropAssign = true;
402                 }
403                 break;
404               case Token.INC:
405               case Token.DEC:
406                 isSet = true;
407                 type = Name.Type.OTHER;
408                 break;
409               case Token.GETPROP:
410                 return;
411               default:
412                 if (NodeUtil.isAssignmentOp(parent) &&
413                     parent.getFirstChild() == n) {
414                   isSet = true;
415                   type = Name.Type.OTHER;
416                 }
417             }
418           }
419           name = n.getQualifiedName();
420           if (name == null) {
421             return;
422           }
423           break;
424         default:
425           return;
426       }
427 
428       // We are only interested in global names.
429       if (!isGlobalNameReference(name, scope)) {
430         return;
431       }
432 
433       scope = scope.getClosestHoistScope();
434       if (isSet) {
435         if (scope.isGlobal()) {
436           handleSetFromGlobal(module, scope, n, parent, name, isPropAssign, type);
437         } else {
438           handleSetFromLocal(module, scope, n, parent, name);
439         }
440       } else {
441         handleGet(module, scope, n, parent, name);
442       }
443     }

445     /**
446      * Gets the fully qualified name corresponding to an object literal key,
447      * as long as it and its prefix property names are valid JavaScript
448      * identifiers. The object literal may be nested inside of other object
449      * literals.
450      *
451      * For example, if called with node {@code n} representing "z" in any of
452      * the following expressions, the result would be "w.x.y.z":
453      * <code> var w = {x: {y: {z: 0}}}; </code>
454      * <code> w.x = {y: {z: 0}}; </code>
455      * <code> w.x.y = {'a': 0, 'z': 0}; </code>
456      *
457      * @param n A child of an OBJLIT node
458      * @return The global name, or null if {@code n} doesn't correspond to the
459      *   key of an object literal that can be named
460      */
461     String getNameForObjLitKey(Node n) {
462       Node parent = n.getParent();
463       Preconditions.checkState(parent.isObjectLit());
464 
465       Node grandparent = parent.getParent();
466       if (grandparent == null) {
467         return null;
468       }
469 
470       Node greatGrandparent = grandparent.getParent();
471       String name;
472       switch (grandparent.getType()) {
473         case Token.NAME:
474           // VAR
475           //   NAME (grandparent)
476           //     OBJLIT (parent)
477           //       STRING (n)
478           if (greatGrandparent == null || !NodeUtil.isNameDeclaration(greatGrandparent)) {
479             return null;
480           }
481           name = grandparent.getString();
482           break;
483         case Token.ASSIGN:
484           // ASSIGN (grandparent)
485           //   NAME|GETPROP
486           //   OBJLIT (parent)
487           //     STRING (n)
488           Node lvalue = grandparent.getFirstChild();
489           name = lvalue.getQualifiedName();
490           break;
491         case Token.STRING_KEY:
492           // OBJLIT
493           //   STRING (grandparent)
494           //     OBJLIT (parent)
495           //       STRING (n)
496           if (greatGrandparent != null &&
497               greatGrandparent.isObjectLit()) {
498             name = getNameForObjLitKey(grandparent);
499           } else {
500             return null;
501           }
502           break;
503         default:
504           return null;
505       }
506       if (name != null) {
507         String key = n.getString();
508         if (TokenStream.isJSIdentifier(key)) {
509           return name + '.' + key;
510         }
511       }
512       return null;
513     }

515     /**
516      * Gets the fully qualified name corresponding to an class member function,
517      * as long as it and its prefix property names are valid JavaScript
518      * identifiers.
519      *
520      * For example, if called with node {@code n} representing "y" in any of
521      * the following expressions, the result would be "x.y":
522      * <code> class x{y(){}}; </code>
523      * <code> var x = class{y(){}}; </code>
524      * <code> var x; x = class{y(){}}; </code>
525      *
526      * @param n A child of an CLASS_MEMBERS node
527      * @return The global name, or null if {@code n} doesn't correspond to
528      *   a class member function that can be named
529      */
530     String getNameForClassMembers(Node n) {
531       Node parent = n.getParent();
532       Preconditions.checkState(parent.isClassMembers());
533       String className = NodeUtil.getClassName(parent.getParent());
534       return className == null ? null : className + '.' + n.getString();
535     }

537     /**
538      * Gets the type of a value or simple expression.
539      *
540      * @param n An r-value in an assignment or variable declaration (not null)
541      * @return A {@link Name.Type}
542      */
543     Name.Type getValueType(Node n) {
544       // Shorthand assignment of extended object literal
545       if (n == null) {
546         return Name.Type.OTHER;
547       }
548       switch (n.getType()) {
549         case Token.CLASS:
550           return Name.Type.CLASS;
551         case Token.OBJECTLIT:
552           return Name.Type.OBJECTLIT;
553         case Token.FUNCTION:
554           return Name.Type.FUNCTION;
555         case Token.OR:
556           // Recurse on the second value. If the first value were an object
557           // literal or function, then the OR would be meaningless and the
558           // second value would be dead code. Assume that if the second value
559           // is an object literal or function, then the first value will also
560           // evaluate to one when it doesn't evaluate to false.
561           return getValueType(n.getLastChild());
562         case Token.HOOK:
563           // The same line of reasoning used for the OR case applies here.
564           Node second = n.getFirstChild().getNext();
565           Name.Type t = getValueType(second);
566           if (t != Name.Type.OTHER) {
567             return t;
568           }
569           Node third = second.getNext();
570           return getValueType(third);
571       }
572       return Name.Type.OTHER;
573     }

575     /**
576      * Updates our representation of the global namespace to reflect an
577      * assignment to a global name in global scope.
578      *
579      * @param module the current module
580      * @param scope the current scope
581      * @param n The node currently being visited
582      * @param parent {@code n}'s parent
583      * @param name The global name (e.g. "a" or "a.b.c.d")
584      * @param isPropAssign Whether this set corresponds to a property
585      *     assignment of the form <code>a.b.c = ...;</code>
586      * @param type The type of the value that the name is being assigned
587      */
588     void handleSetFromGlobal(JSModule module, Scope scope,
589         Node n, Node parent, String name,
590         boolean isPropAssign, Name.Type type) {
591       if (maybeHandlePrototypePrefix(module, scope, n, parent, name)) {
592         return;
593       }
594 
595       Name nameObj = getOrCreateName(name);
596       nameObj.type = type;
597 
598       Ref set = new Ref(module, scope, n, nameObj, Ref.Type.SET_FROM_GLOBAL,
599           currentPreOrderIndex++);
600       nameObj.addRef(set);
601 
602       if (isNestedAssign(parent)) {
603         // This assignment is both a set and a get that creates an alias.
604         Ref get = new Ref(module, scope, n, nameObj, Ref.Type.ALIASING_GET,
605             currentPreOrderIndex++);
606         nameObj.addRef(get);
607         Ref.markTwins(set, get);
608       } else if (isTypeDeclaration(n)) {
609         // Names with a @constructor or @enum annotation are always collapsed
610         nameObj.setDeclaredType();
611       }
612     }

614     /**
615      * Determines whether a set operation is a constructor or enumeration
616      * or interface declaration. The set operation may either be an assignment
617      * to a name, a variable declaration, or an object literal key mapping.
618      *
619      * @param n The node that represents the name being set
620      * @param parent Parent node of {@code n} (an ASSIGN, VAR, or OBJLIT node)
621      * @return Whether the set operation is either a constructor or enum
622      *     declaration
623      */
624     private boolean isTypeDeclaration(Node n) {
625       Node valueNode = NodeUtil.getRValueOfLValue(n);
626       JSDocInfo info = NodeUtil.getBestJSDocInfo(n);
627       // Heed the annotations only if they're sensibly used.
628       return info != null && valueNode != null &&
629              (info.isConstructor() && valueNode.isFunction() ||
630               info.isInterface() && valueNode.isFunction() ||
631               info.hasEnumParameterType() && valueNode.isObjectLit());
632     }

634     /**
635      * Updates our representation of the global namespace to reflect an
636      * assignment to a global name in a local scope.
637      *
638      * @param module The current module
639      * @param scope The current scope
640      * @param n The node currently being visited
641      * @param parent {@code n}'s parent
642      * @param name The global name (e.g. "a" or "a.b.c.d")
643      */
644     void handleSetFromLocal(JSModule module, Scope scope, Node n, Node parent,
645                             String name) {
646       if (maybeHandlePrototypePrefix(module, scope, n, parent, name)) {
647         return;
648       }
649 
650       Name nameObj = getOrCreateName(name);
651       Ref set = new Ref(module, scope, n, nameObj,
652           Ref.Type.SET_FROM_LOCAL, currentPreOrderIndex++);
653       nameObj.addRef(set);
654 
655       if (isNestedAssign(parent)) {
656         // This assignment is both a set and a get that creates an alias.
657         Ref get = new Ref(module, scope, n, nameObj,
658             Ref.Type.ALIASING_GET, currentPreOrderIndex++);
659         nameObj.addRef(get);
660         Ref.markTwins(set, get);
661       }
662     }

664     /**
665      * Updates our representation of the global namespace to reflect a read
666      * of a global name.
667      *
668      * @param module The current module
669      * @param scope The current scope
670      * @param n The node currently being visited
671      * @param parent {@code n}'s parent
672      * @param name The global name (e.g. "a" or "a.b.c.d")
673      */
674     void handleGet(JSModule module, Scope scope,
675         Node n, Node parent, String name) {
676       if (maybeHandlePrototypePrefix(module, scope, n, parent, name)) {
677         return;
678       }
679 
680       Ref.Type type = Ref.Type.DIRECT_GET;
681       if (parent != null) {
682         switch (parent.getType()) {
683           case Token.EXPR_RESULT:
684           case Token.IF:
685           case Token.INSTANCEOF:
686           case Token.TYPEOF:
687           case Token.VOID:
688           case Token.NOT:
689           case Token.BITNOT:
690           case Token.POS:
691           case Token.NEG:
692             break;
693           case Token.CALL:
694             if (n == parent.getFirstChild()) {
695               // It is a call target
696               type = Ref.Type.CALL_GET;
697             } else if (isClassDefiningCall(parent)) {
698               type = Ref.Type.DIRECT_GET;
699             } else {
700               type = Ref.Type.ALIASING_GET;
701             }
702             break;
703           case Token.NEW:
704             type = n == parent.getFirstChild() ? Ref.Type.DIRECT_GET : Ref.Type.ALIASING_GET;
705             break;
706           case Token.OR:
707           case Token.AND:
708             // This node is x or y in (x||y) or (x&&y). We only know that an
709             // alias is not getting created for this name if the result is used
710             // in a boolean context or assigned to the same name
711             // (e.g. var a = a || {}).
712             type = determineGetTypeForHookOrBooleanExpr(module, scope, parent, name);
713             break;
714           case Token.HOOK:
715             if (n != parent.getFirstChild()) {
716               // This node is y or z in (x?y:z). We only know that an alias is
717               // not getting created for this name if the result is assigned to
718               // the same name (e.g. var a = a ? a : {}).
719               type = determineGetTypeForHookOrBooleanExpr(module, scope, parent, name);
720             }
721             break;
722           case Token.DELPROP:
723             type = Ref.Type.DELETE_PROP;
724             break;
725           default:

734     private boolean isClassDefiningCall(Node callNode) {

748     /**
749      * Determines whether the result of a hook (x?y:z) or boolean expression
750      * (x||y) or (x&&y) is assigned to a specific global name.
751      *
752      * @param module The current module
753      * @param scope The current scope
754      * @param parent The parent of the current node in the traversal. This node
755      *     should already be known to be a HOOK, AND, or OR node.
756      * @param name A name that is already known to be global in the current
757      *     scope (e.g. "a" or "a.b.c.d")
758      * @return The expression's get type, either {@link Ref.Type#DIRECT_GET} or
759      *     {@link Ref.Type#ALIASING_GET}
760      */
761     Ref.Type determineGetTypeForHookOrBooleanExpr(
762         JSModule module, Scope scope, Node parent, String name) {
763       Node prev = parent;
764       for (Node anc : parent.getAncestors()) {
765         switch (anc.getType()) {
766           case Token.INSTANCEOF:
767           case Token.EXPR_RESULT:
768           case Token.VAR:
769           case Token.LET:
770           case Token.CONST:
771           case Token.IF:
772           case Token.WHILE:
773           case Token.FOR:
774           case Token.TYPEOF:
775           case Token.VOID:
776           case Token.NOT:
777           case Token.BITNOT:
778           case Token.POS:
779           case Token.NEG:
780             return Ref.Type.DIRECT_GET;
781           case Token.HOOK:
782             if (anc.getFirstChild() == prev) {
783               return Ref.Type.DIRECT_GET;
784             }
785             break;
786           case Token.ASSIGN:
787             if (!name.equals(anc.getFirstChild().getQualifiedName())) {
788               return Ref.Type.ALIASING_GET;
789             }
790             break;
791           case Token.NAME:  // a variable declaration
792             if (!name.equals(anc.getString())) {
793               return Ref.Type.ALIASING_GET;
794             }
795             break;
796           case Token.CALL:
797             if (anc.getFirstChild() != prev) {
798               return Ref.Type.ALIASING_GET;
799             }
800             break;
801           case Token.DELPROP:
802             return Ref.Type.DELETE_PROP;
803         }
804         prev = anc;
805       }
806       return Ref.Type.ALIASING_GET;
807     }

829     /**
830      * Updates our representation of the global namespace to reflect a read
831      * of a global name's longest prefix before the "prototype" property if the
832      * name includes the "prototype" property. Does nothing otherwise.
833      *
834      * @param module The current module
835      * @param scope The current scope
836      * @param n The node currently being visited
837      * @param parent {@code n}'s parent
838      * @param name The global name (e.g. "a" or "a.b.c.d")
839      * @return Whether the name was handled
840      */
841     boolean maybeHandlePrototypePrefix(JSModule module, Scope scope,
842         Node n, Node parent, String name) {
843       // We use a string-based approach instead of inspecting the parse tree
844       // to avoid complexities with object literals, possibly nested, beneath
845       // assignments.
846 
847       int numLevelsToRemove;
848       String prefix;
849       if (name.endsWith(".prototype")) {
850         numLevelsToRemove = 1;
851         prefix = name.substring(0, name.length() - 10);
852       } else {
853         int i = name.indexOf(".prototype.");
854         if (i == -1) {
855           return false;
856         }
857         prefix = name.substring(0, i);
858         numLevelsToRemove = 2;
859         i = name.indexOf('.', i + 11);
860         while (i >= 0) {
861           numLevelsToRemove++;
862           i = name.indexOf('.', i + 1);
863         }
864       }
865 
866       if (parent != null && NodeUtil.isObjectLitKey(n)) {
867         // Object literal keys have no prefix that's referenced directly per
868         // key, so we're done.
869         return true;
870       }
871 
872       for (int i = 0; i < numLevelsToRemove; i++) {
873         parent = n;
874         n = n.getFirstChild();
875       }
876 
877       handleGet(module, scope, n, parent, prefix, Ref.Type.PROTOTYPE_GET);
878       return true;
879     }

895      * Gets a {@link Name} instance for a global name. Creates it if necessary,
896      * as well as instances for any of its prefixes that are not yet defined.
897      *
898      * @param name A global name (e.g. "a", "a.b.c.d")
899      * @return The {@link Name} instance for {@code name}
900      */
901     Name getOrCreateName(String name) {
902       Name node = nameMap.get(name);
903       if (node == null) {
904         int i = name.lastIndexOf('.');
905         if (i >= 0) {
906           String parentName = name.substring(0, i);
907           Name parent = getOrCreateName(parentName);
908           node = parent.addProperty(name.substring(i + 1), inExterns);
909         } else {
910           node = new Name(name, null, inExterns);
911           globalNames.add(node);
912         }
913         nameMap.put(name, node);
914       }
915       return node;
916     }

1005     void addRef(Ref ref) {
1006       addRefInternal(ref);
1007       switch (ref.type) {
1008         case SET_FROM_GLOBAL:
1009           if (declaration == null) {
1010             declaration = ref;
1011             docInfo = getDocInfoForDeclaration(ref);
1012           }
1013           globalSets++;
1014           break;
1015         case SET_FROM_LOCAL:
1016           localSets++;
1017           JSDocInfo info = ref.getNode() == null ? null :
1018               NodeUtil.getBestJSDocInfo(ref.getNode());
1019           if (info != null && info.isNoCollapse()) {
1020             localSetsWithNoCollapse++;
1021           }
1022           break;
1023         case PROTOTYPE_GET:
1024         case DIRECT_GET:
1025           Node node = ref.getNode();
1026           if (node != null && node.isGetProp() && node.getParent().isExprResult()) {
1027             docInfo = node.getJSDocInfo();
1028           }
1029           totalGets++;
1030           break;
1031         case ALIASING_GET:
1032           aliasingGets++;
1033           totalGets++;
1034           break;
1035         case CALL_GET:
1036           callGets++;
1037           totalGets++;
1038           break;
1039         case DELETE_PROP:
1040           deleteProps++;
1041           break;
1042         default:
1043           throw new IllegalStateException();
1044       }
1045     }

1047     void removeRef(Ref ref) {
1048       if (refs != null && refs.remove(ref)) {
1049         if (ref == declaration) {
1050           declaration = null;
1051           if (refs != null) {
1052             for (Ref maybeNewDecl : refs) {
1053               if (maybeNewDecl.type == Ref.Type.SET_FROM_GLOBAL) {
1054                 declaration = maybeNewDecl;
1055                 break;
1056               }
1057             }
1058           }
1059         }
1060 
1061         JSDocInfo info;
1062         switch (ref.type) {
1063           case SET_FROM_GLOBAL:
1064             globalSets--;
1065             break;
1066           case SET_FROM_LOCAL:
1067             localSets--;
1068             info = ref.getNode() == null ? null :
1069                 NodeUtil.getBestJSDocInfo(ref.getNode());
1070             if (info != null && info.isNoCollapse()) {
1071               localSetsWithNoCollapse--;
1072             }
1073             break;
1074           case PROTOTYPE_GET:
1075           case DIRECT_GET:
1076             totalGets--;
1077             break;
1078           case ALIASING_GET:
1079             aliasingGets--;
1080             totalGets--;
1081             break;
1082           case CALL_GET:
1083             callGets--;
1084             totalGets--;
1085             break;
1086           case DELETE_PROP:
1087             deleteProps--;
1088             break;
1089           default:
1090             throw new IllegalStateException();
1091         }
1092       }
1093     }

1143     boolean isInlinableGlobalAlias() {
1144       // Only simple aliases with direct usage are inlinable.
1145       if (inExterns || globalSets != 1 || localSets != 0 || !canCollapse()) {
1146         return false;
1147       }
1148 
1149       // Only allow inlining of simple references.
1150       for (Ref ref : getRefs()) {
1151         switch (ref.type) {
1152           case SET_FROM_GLOBAL:
1153             // Expect one global set
1154             continue;

1157           case ALIASING_GET:

1171     boolean canCollapse() {
1172       return !inExterns && !isGetOrSetDefinition() &&
1173           !isCollapsingExplicitlyDenied() &&
1174           (declaredType ||
1175           (parent == null || parent.canCollapseUnannotatedChildNames()) &&
1176           (globalSets > 0 || localSets > 0) && localSetsWithNoCollapse == 0 &&
1177           deleteProps == 0);
1178     }

1180     boolean isGetOrSetDefinition() {
1181       return this.type == Type.GET || this.type == Type.SET;
1182     }

1184     boolean canCollapseUnannotatedChildNames() {
1185       if (type == Type.OTHER || isGetOrSetDefinition()
1186           || globalSets != 1 || localSets != 0 || deleteProps != 0) {
1187         return false;
1188       }
1189 
1190       // Don't try to collapse if the one global set is a twin reference.
1191       // We could theoretically handle this case in CollapseProperties, but
1192       // it's probably not worth the effort.
1193       Preconditions.checkNotNull(declaration);
1194       if (declaration.getTwin() != null) {
1195         return false;
1196       }
1197 
1198       if (isCollapsingExplicitlyDenied()) {
1199         return false;
1200       }
1201 
1202       if (declaredType) {
1203         return true;
1204       }
1205 
1206       // If this is a key of an aliased object literal, then it will be aliased
1207       // later. So we won't be able to collapse its properties.
1208       if (parent != null && parent.shouldKeepKeys()) {
1209         return false;
1210       }
1211 
1212       // If this is aliased, then its properties can't be collapsed either.
1213       if (aliasingGets > 0) {
1214         return false;
1215       }
1216 
1217       return (parent == null || parent.canCollapseUnannotatedChildNames());
1218     }

1222       return type == Type.OBJECTLIT &&
1223           (aliasingGets > 0 || isCollapsingExplicitlyDenied());
1224     }

1290           return null;
