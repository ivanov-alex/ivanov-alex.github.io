49  /**
50   * A record (structural) type.
51   *
52   * Subtyping: The subtyping of a record type is defined via structural
53   * comparison of a record type's properties. For example, a record
54   * type of the form { a : TYPE_1 } is a supertype of a record type
55   * of the form { b : TYPE_2, a : TYPE_1 } because B can be assigned to
56   * A and matches all constraints. Similarly, a defined type can be assigned
57   * to a record type so long as that defined type matches all property
58   * constraints of the record type. A record type of the form { a : A, b : B }
59   * can be assigned to a record of type { a : A }.
60   *
61   */
62  public class RecordType extends PrototypeObjectType {
63    private static final long serialVersionUID = 1L;
64  
65    private final boolean declared;
66    private boolean isFrozen = false;
67  
68    RecordType(JSTypeRegistry registry, Map<String, RecordProperty> properties) {
69      this(registry, properties, true);
70    }
71  
72    /**
73     * Creates a record type.
74     *
75     * @param registry The type registry under which this type lives.
76     * @param properties A map of all the properties of this record type.
77     * @param declared Whether this is a declared or synthesized type.
78     *     A synthesized record type is just used for bookkeeping
79     *     in the type system. A declared record type was actually used in the
80     *     user's program.
81     * @throws IllegalStateException if the {@code RecordProperty} associated
82     *         with a property is null.
83     */
84    RecordType(JSTypeRegistry registry, Map<String, RecordProperty> properties,
85        boolean declared) {
86      super(registry, null, null);
87      setPrettyPrint(true);
88      this.declared = declared;
89  
90      for (String property : properties.keySet()) {
91        RecordProperty prop = properties.get(property);
92        if (prop == null) {
93          throw new IllegalStateException(
94              "RecordProperty associated with a property should not be null!");
95        }
96        if (declared) {
97          defineDeclaredProperty(
98              property, prop.getType(), prop.getPropertyNode());
99        } else {
100         defineSynthesizedProperty(
101             property, prop.getType(), prop.getPropertyNode());
102       }
103     }
104     // Freeze the record type.
105     isFrozen = true;
106   }
107 
108   /** @return Is this synthesized for internal bookkeeping? */
109   boolean isSynthetic() {
110     return !declared;
111   }
112 
113   boolean checkRecordEquivalenceHelper(RecordType otherRecord,
114       EquivalenceMethod eqMethod, EqCache eqCache) {
115 
116     MatchStatus result = eqCache.checkCache(this, otherRecord);
117     if (result != null) {
118       return result.subtypeValue();
119     }
120     Set<String> keySet = getOwnPropertyNames();
121     Set<String> otherKeySet = otherRecord.getOwnPropertyNames();
122     if (!otherKeySet.equals(keySet)) {
123       return false;
124     }
125     for (String key : keySet) {
126       if (!otherRecord.getPropertyType(key).checkEquivalenceHelper(
127               getPropertyType(key), eqMethod, eqCache)) {
128         eqCache.updateCache(this, otherRecord, MatchStatus.NOT_MATCH);
129         return false;
130       }
131     }
132     eqCache.updateCache(this, otherRecord, MatchStatus.MATCH);
133     return true;
134   }

141   @Override
142   boolean defineProperty(String propertyName, JSType type,
143       boolean inferred, Node propertyNode) {
144     if (isFrozen) {
145       return false;
146     }
147 
148     return super.defineProperty(propertyName, type, inferred,
149         propertyNode);

152   JSType getGreatestSubtypeHelper(JSType that) {
153     if (that.isRecordType()) {
154       RecordType thatRecord = that.toMaybeRecordType();
155       RecordTypeBuilder builder = new RecordTypeBuilder(registry);
156       builder.setSynthesized(true);
157 
158       JSType noType = registry.getNativeObjectType(JSTypeNative.NO_TYPE);
159 
160       // The greatest subtype consists of those *unique* properties of both
161       // record types. If any property conflicts, then the NO_TYPE type
162       // is returned.
163       for (String property : getOwnPropertyNames()) {
164         JSType thisPropertyType = getPropertyType(property);
165         JSType propType = null;
166         if (thatRecord.hasProperty(property)) {
167           JSType thatPropertyType = thatRecord.getPropertyType(property);
168           propType = thisPropertyType.getGreatestSubtype(thatPropertyType);
169           if (propType.isEquivalentTo(noType)) { return noType; }
170         } else {
171           propType = thisPropertyType;
172         }
173         builder.addProperty(property, propType, getPropertyNode(property));
174       }
175 
176       for (String property : thatRecord.getOwnPropertyNames()) {
177         if (!hasProperty(property)) {
178           builder.addProperty(property, thatRecord.getPropertyType(property),
179               thatRecord.getPropertyNode(property));
180         }
181       }
182       return builder.build();
183     }
184 
185     JSType greatestSubtype = registry.getNativeType(
186         JSTypeNative.NO_OBJECT_TYPE);
187     JSType thatRestrictedToObj =
188         registry.getNativeType(JSTypeNative.OBJECT_TYPE)
189         .getGreatestSubtype(that);
190     if (!thatRestrictedToObj.isEmptyType()) {
191       // In this branch, the other type is some object type. We find
192       // the greatest subtype with the following algorithm:
193       // 1) For each property "x" of this record type, take the union
194       //    of all classes with a property "x" with a compatible property type.
195       //    and which are a subtype of {@code that}.
196       // 2) Take the intersection of all of these unions.
197       for (String propName : getOwnPropertyNames()) {
198         JSType propType = getPropertyType(propName);
199         UnionTypeBuilder builder = new UnionTypeBuilder(registry);
200         for (ObjectType alt :
201           registry.getEachReferenceTypeWithProperty(propName)) {
202           JSType altPropType = alt.getPropertyType(propName);
203           if (altPropType != null && !alt.isEquivalentTo(this)
204               && alt.isSubtype(that) && altPropType.isSubtype(propType)) {
205             builder.addAlternate(alt);
206           }
207         }
208         greatestSubtype = greatestSubtype.getLeastSupertype(builder.build());
209       }
210     }
211     return greatestSubtype;
212   }

214   /**
215    * get the map of properties to types covered in an object type
216    * @return a Map that maps the property's name to the property's type
217    */
218   public Map<String, JSType> getOwnPropertyTypeMap() {
219     Map<String, JSType> propTypeMap = new HashMap<String, JSType>();
220     for (String name : this.getOwnPropertyNames()) {
221       propTypeMap.put(name, this.getPropertyType(name));
222     }
223     return propTypeMap;
224   }

241   @Override
242   protected boolean isSubtype(JSType that,
243       ImplCache implicitImplCache) {
244     if (JSType.isSubtypeHelper(this, that, implicitImplCache)) {
245       return true;
246     }
247 
248     // Top of the record types is the empty record, or OBJECT_TYPE.
249     if (registry.getNativeObjectType(
250             JSTypeNative.OBJECT_TYPE).isSubtype(that, implicitImplCache)) {
251       return true;
252     }
253 
254     // A type is a subtype of a record type if it itself is a record
255     // type and it has at least the same members as the parent record type
256     // with the same types.
257     if (!that.isRecordType()) {
258       return false;
259     }
260 
261     return RecordType.isSubtype(
262         this, that.toMaybeRecordType(), implicitImplCache);
263   }
264 
265   /** Determines if typeA is a subtype of typeB */
266   static boolean isSubtype(ObjectType typeA, RecordType typeB) {
267     return isSubtype(typeA, typeB, ImplCache.create());
268   }
269 
270   /** Determines if typeA is a subtype of typeB */
271   static boolean isSubtype(ObjectType typeA, RecordType typeB,
272       ImplCache implicitImplCache) {
273     // typeA is a subtype of record type typeB iff:
274     // 1) typeA has all the properties declared in typeB.
275     // 2) And for each property of typeB, its type must be
276     //    a super type of the corresponding property of typeA.
277     for (String property : typeB.getOwnPropertyNames()) {
278       if (!typeA.hasProperty(property)) {
279         return false;
280       }
281       JSType propA = typeA.getPropertyType(property);
282       JSType propB = typeB.getPropertyType(property);
283       if (!propA.isSubtype(propB, implicitImplCache)) {
284         return false;
285       }
286     }
287     return true;
288   }
