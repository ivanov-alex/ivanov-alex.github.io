26  /**
27   * Rewrites an AMD module https://github.com/amdjs/amdjs-api/wiki/AMD to a
28   * CommonJS module. See {@link ProcessCommonJSModules} for follow up processing
29   * step.
30   */
31  public final class TransformAMDToCJSModule implements CompilerPass {
32  
33    @VisibleForTesting
34    static final DiagnosticType UNSUPPORTED_DEFINE_SIGNATURE_ERROR =
35        DiagnosticType.error(
36            "UNSUPPORTED_DEFINE_SIGNATURE",
37            "Only define(function() ...), define(OBJECT_LITERAL) and define("
38                + "['dep', 'dep1'], function(d0, d2, [exports, module]) ...) forms "
39                + "are currently supported.");
40    static final DiagnosticType NON_TOP_LEVEL_STATEMENT_DEFINE_ERROR =
41        DiagnosticType.error(
42              "NON_TOP_LEVEL_STATEMENT_DEFINE",
43              "The define function must be called as a top-level statement.");
44    static final DiagnosticType REQUIREJS_PLUGINS_NOT_SUPPORTED_WARNING =
45      DiagnosticType.warning(
46            "REQUIREJS_PLUGINS_NOT_SUPPORTED",
47            "Plugins in define requirements are not supported: {0}");
48  
49    static final String VAR_RENAME_SUFFIX = "__alias";
50  
51  
52    private final AbstractCompiler compiler;
53    private int renameIndex = 0;
54  
55    public TransformAMDToCJSModule(AbstractCompiler compiler) {
56      this.compiler = compiler;
57    }
58  
59    @Override
60    public void process(Node externs, Node root) {
61      NodeTraversal.traverseEs6(compiler, root, new TransformAMDModulesCallback());
62    }
63  
64    private static void unsupportedDefineError(NodeTraversal t, Node n) {
65      t.report(n, UNSUPPORTED_DEFINE_SIGNATURE_ERROR);
66    }
67  
68    /**
69     * The modules "exports", "require" and "module" are virtual in terms of
70     * existing implicitly in CommonJS.
71     */
72    private static boolean isVirtualModuleName(String moduleName) {
73      return "exports".equals(moduleName) || "require".equals(moduleName) ||
74          "module".equals(moduleName);
75    }
76  
77    /**
78     * Rewrites calls to define which has to be in void context just below the
79     * current script node.
80     */
81    private class TransformAMDModulesCallback extends AbstractPostOrderCallback {
82  
83      @Override
84      public void visit(NodeTraversal t, Node n, Node parent) {
85        if (n.isCall() && n.getFirstChild() != null &&
86            n.getFirstChild().isName() &&
87            "define".equals(n.getFirstChild().getString())) {
88          Var define = t.getScope().getVar(n.getFirstChild().
89              getString());
90          if (define != null && !define.isGlobal()) {
91            // Ignore non-global define.
92            return;
93          }
94          if (!(parent.isExprResult() && parent.getParent().isScript())) {
95            t.report(n, NON_TOP_LEVEL_STATEMENT_DEFINE_ERROR);
96            return;
97          }
98          Node script = parent.getParent();
99          Node requiresNode = null;
100         Node callback = null;
101         int defineArity = n.getChildCount() - 1;
102         if (defineArity == 0) {
103           unsupportedDefineError(t, n);
104           return;
105         } else if (defineArity == 1) {
106           callback = n.getChildAtIndex(1);
107           if (callback.isObjectLit()) {
108             handleDefineObjectLiteral(parent, callback, script);
109             return;
110           }
111         } else if (defineArity == 2) {
112           requiresNode = n.getChildAtIndex(1);
113           callback = n.getChildAtIndex(2);
114         } else if (defineArity >= 3) {
115           unsupportedDefineError(t, n);
116           return;
117         }
118 
119         if (!callback.isFunction() ||
120             (requiresNode != null && !requiresNode.isArrayLit())) {
121           unsupportedDefineError(t, n);
122           return;
123         }
124 
125         handleRequiresAndParamList(t, n, script, requiresNode, callback);
126 
127         Node callbackBlock = callback.getChildAtIndex(2);
128         NodeTraversal.traverseEs6(compiler, callbackBlock,
129             new DefineCallbackReturnCallback());
130 
131         moveCallbackContentToTopLevel(parent, script, callbackBlock);
132         compiler.reportCodeChange();
133       }
134     }
135 
136     /**
137      * When define is called with an object literal, assign it to module.exports and
138      * we're done.
139      */
140     private void handleDefineObjectLiteral(Node parent, Node onlyExport,
141         Node script) {
142       onlyExport.getParent().removeChild(onlyExport);
143       script.replaceChild(parent,
144           IR.exprResult(
145               IR.assign(
146                   NodeUtil.newQName(compiler, "module.exports"),
147                   onlyExport))
148           .useSourceInfoIfMissingFromForTree(onlyExport));
149       compiler.reportCodeChange();
150     }
151 
152     /**
153      * Rewrites the required modules to
154      * <code>var nameInParamList = require("nameFromRequireList");</code>
155      */
156     private void handleRequiresAndParamList(NodeTraversal t, Node defineNode,
157         Node script, Node requiresNode, Node callback) {
158       Iterator<Node> paramList = callback.getChildAtIndex(1).children().
159           iterator();
160       Iterator<Node> requires = requiresNode != null ?
161           requiresNode.children().iterator() : Collections.<Node>emptyIterator();
162       while (paramList.hasNext() || requires.hasNext()) {
163         Node aliasNode = paramList.hasNext() ? paramList.next() : null;
164         Node modNode = requires.hasNext() ? requires.next() : null;
165         handleRequire(t, defineNode, script, callback, aliasNode, modNode);
166       }
167     }
168 
169     /**
170      * Rewrite a single require call.
171      */
172     private void handleRequire(NodeTraversal t, Node defineNode, Node script,
173         Node callback, Node aliasNode, Node modNode) {
174       String moduleName = null;
175       if (modNode != null) {
176         moduleName = handlePlugins(t, script, modNode.getString(), modNode);
177       }
178 
179       if (isVirtualModuleName(moduleName)) {
180         return;
181       }
182 
183       String aliasName = aliasNode != null ? aliasNode.getString() : null;
184       Scope globalScope = t.getScope();
185       if (aliasName != null &&
186           globalScope.isDeclared(aliasName, true)) {
187         while (true) {
188           String renamed = aliasName + VAR_RENAME_SUFFIX + renameIndex;
189           if (!globalScope.isDeclared(renamed, true)) {
190             NodeTraversal.traverseEs6(compiler, callback,
191                 new RenameCallback(aliasName, renamed));
192             aliasName = renamed;
193             break;
194           }
195           renameIndex++;
196         }
197       }
198 
199       Node requireNode;
200       if (moduleName != null) {
201         Node call = IR.call(IR.name("require"), IR.string(moduleName));
202         call.putBooleanProp(Node.FREE_CALL, true);
203         if (aliasName != null) {
204           requireNode = IR.var(IR.name(aliasName), call)
205               .useSourceInfoIfMissingFromForTree(aliasNode);
206         } else {
207           requireNode = IR.exprResult(call).
208               useSourceInfoIfMissingFromForTree(modNode);
209         }
210       } else {
211         // ignore exports, require and module (because they are implicit
212         // in CommonJS);
213         if (isVirtualModuleName(aliasName)) {
214           return;
215         }
216         requireNode = IR.var(IR.name(aliasName), IR.nullNode())
217             .useSourceInfoIfMissingFromForTree(aliasNode);
218       }
219 
220       script.addChildBefore(requireNode,
221           defineNode.getParent());
222     }
223 
224     /**
225      * Require.js supports a range of plugins that are hard to support
226      * statically. Generally none are supported right now with the
227      * exception of a simple hack to support condition loading. This
228      * was added to make compilation of Dojo work better but will
229      * probably break, so just don't use them :)
230      */
231     private String handlePlugins(NodeTraversal t, Node script,
232         String moduleName, Node modNode) {
233       if (moduleName.contains("!")) {
234         t.report(modNode, REQUIREJS_PLUGINS_NOT_SUPPORTED_WARNING, moduleName);
235         int condition = moduleName.indexOf('?');
236         if (condition > 0) {
237           if (moduleName.contains(":")) {
238             return null;
239           }
240           return handlePlugins(t, script, moduleName.substring(condition + 1),
241               modNode);
242         }
243         moduleName = null;
244       }
245       return moduleName;
246     }
247 
248     /**
249      * Moves the statements in the callback to be direct children of the
250      * current script.
251      */
252     private void moveCallbackContentToTopLevel(Node defineParent, Node script,
253         Node callbackBlock) {
254       int curIndex = script.getIndexOfChild(defineParent);
255       script.removeChild(defineParent);
256       callbackBlock.getParent().removeChild(callbackBlock);
257       Node before = script.getChildAtIndex(curIndex);
258       if (before != null) {
259         script.addChildBefore(callbackBlock, before);
260       }
261       script.addChildToBack(callbackBlock);
262       NodeUtil.tryMergeBlock(callbackBlock);
263     }
264   }

266   /**
267    * Rewrites the return statement of the callback to be an assignment to
268    * module.exports.
269    */
270   private static class DefineCallbackReturnCallback extends
271       NodeTraversal.AbstractShallowStatementCallback {
272     @Override
273     public void visit(NodeTraversal t, Node n, Node parent) {
274       if (n.isReturn() && n.hasChildren()) {
275         Node retVal = n.getFirstChild();
276         n.removeChild(retVal);
277         parent.replaceChild(n, IR.exprResult(
278             IR.assign(
279                 IR.getprop(IR.name("module"), IR.string("exports")), retVal))
280                     .useSourceInfoFromForTree(n));
281       }
282     }
283   }

285   /**
286    * Renames names;
287    */
288   private static class RenameCallback extends AbstractPostOrderCallback {
289 
290     private final String from;
291     private final String to;
292 
293     public RenameCallback(String from, String to) {
294       this.from = from;
295       this.to = to;
296     }
297 
298     @Override
299     public void visit(NodeTraversal t, Node n, Node parent) {
300       if (n.isName() && from.equals(n.getString())) {
301         n.setString(to);
302         n.putProp(Node.ORIGINALNAME_PROP, from);
303       }
304     }
305   }
