43  /**
44   * Reuse variable names if possible.
45   *
46   * <p>For example, from <code>var x = 1; print(x); var y = 2; print(y); </code>
47   * to <code>var x = 1; print(x); x = 2; print(x)</code>. The benefits are
48   * slightly shorter code because of the removed <code>var<code> declaration,
49   * less unique variables in hope for better renaming, and finally better gzip
50   * compression.
51   *
52   * <p>The pass operates similar to a typical register allocator found in an
53   * optimizing compiler by first computing live ranges with
54   * {@link LiveVariablesAnalysis} and a variable interference graph. Then it uses
55   * graph coloring in {@link GraphColoring} to determine which two variables can
56   * be merge together safely.
57   *
58   */
59  class CoalesceVariableNames extends AbstractPostOrderCallback implements
60      CompilerPass, ScopedCallback {
61  
62    private final AbstractCompiler compiler;
63    private final Deque<GraphColoring<Var, Void>> colorings;
64    private final boolean usePseudoNames;
65  
66    private static final Comparator<Var> coloringTieBreaker =
67        new Comparator<Var>() {
68      @Override
69      public int compare(Var v1, Var v2) {
70        return v1.index - v2.index;
71      }
72    };
73  
74    /**
75     * @param usePseudoNames For debug purposes, when merging variable foo and bar
76     * to foo, rename both variable to foo_bar.
77     */
78    CoalesceVariableNames(AbstractCompiler compiler, boolean usePseudoNames) {
79      Preconditions.checkState(!compiler.getLifeCycleStage().isNormalized());
80  
81      this.compiler = compiler;
82      colorings = new LinkedList<>();
83      this.usePseudoNames = usePseudoNames;
84    }
85  
86    @Override
87    public void process(Node externs, Node root) {
88      NodeTraversal.traverse(compiler, root, this);
89    }

91    private static boolean shouldOptimizeScope(Scope scope) {
92      // TODO(user): We CAN do this in the global scope, just need to be
93      // careful when something is exported. Liveness uses bit-vector for live
94      // sets so I don't see compilation time will be a problem for running this
95      // pass in the global scope.
96      if (scope.isGlobal()) {
97        return false;
98      }
99  
100     return LiveVariablesAnalysis.MAX_VARIABLES_TO_ANALYZE >= scope.getVarCount();
101   }
102 
103   @Override
104   public void enterScope(NodeTraversal t) {
105     Scope scope = t.getScope();
106     if (!shouldOptimizeScope(scope)) {
107       return;
108     }
109 
110     ControlFlowGraph<Node> cfg = t.getControlFlowGraph();
111     LiveVariablesAnalysis liveness =
112         new LiveVariablesAnalysis(cfg, scope, compiler);
113     // If the function has exactly 2 params, mark them as escaped. This is
114     // a work-around for an IE bug where it throws an exception if you
115     // write to the parameters of the callback in a sort(). See:
116     // http://code.google.com/p/closure-compiler/issues/detail?id=58
117     if (scope.getRootNode().getFirstChild().getNext().getChildCount() == 2) {
118       liveness.markAllParametersEscaped();
119     }
120     liveness.analyze();
121 
122     UndiGraph<Var, Void> interferenceGraph =
123         computeVariableNamesInterferenceGraph(
124             t, cfg, (Set<Var>) liveness.getEscapedLocals());
125 
126     GraphColoring<Var, Void> coloring =
127         new GreedyGraphColoring<>(interferenceGraph,
128             coloringTieBreaker);
129 
130     coloring.color();
131     colorings.push(coloring);
132   }
133 
134   @Override
135   public void exitScope(NodeTraversal t) {
136     if (!shouldOptimizeScope(t.getScope())) {
137       return;
138     }
139     colorings.pop();
140   }
141 
142   @Override
143   public void visit(NodeTraversal t, Node n, Node parent) {
144     if (colorings.isEmpty() || !n.isName() ||
145         parent.isFunction()) {
146       // Don't rename named functions.
147       return;
148     }
149     Var var = t.getScope().getVar(n.getString());
150     GraphNode<Var, ?> vNode = colorings.peek().getGraph().getNode(var);
151     if (vNode == null) {
152       // This is not a local.
153       return;
154     }
155     Var coalescedVar = colorings.peek().getPartitionSuperNode(var);
156 
157     if (!usePseudoNames) {
158       if (vNode.getValue().equals(coalescedVar)) {
159         // The coalesced name is itself, nothing to do.
160         return;
161       }
162 
163       // Rename.
164       n.setString(coalescedVar.name);
165       compiler.reportCodeChange();
166 
167       if (parent.isVar()) {
168         removeVarDeclaration(n);
169       }
170     } else {
171       // This code block is slow but since usePseudoName is for debugging,
172       // we should not sacrifice performance for non-debugging compilation to
173       // make this fast.
174       String pseudoName = null;
175       Set<String> allMergedNames = new TreeSet<>();
176       for (Iterator<Var> i = t.getScope().getVars(); i.hasNext();) {
177         Var iVar = i.next();
178 
179         // Look for all the variables that can be merged (in the graph by now)
180         // and it is merged with the current coalescedVar.
181         if (colorings.peek().getGraph().getNode(iVar) != null &&
182             coalescedVar.equals(colorings.peek().getPartitionSuperNode(iVar))) {
183           allMergedNames.add(iVar.name);
184         }
185       }
186 
187       // Keep its original name.
188       if (allMergedNames.size() == 1) {
189         return;
190       }
191 
192       pseudoName = Joiner.on("_").join(allMergedNames);
193 
194       while (t.getScope().isDeclared(pseudoName, true)) {
195         pseudoName += "$";
196       }
197 
198       n.setString(pseudoName);
199       compiler.reportCodeChange();
200 
201       if (!vNode.getValue().equals(coalescedVar) && parent.isVar()) {
202         removeVarDeclaration(n);
203       }
204     }
205   }

207   private UndiGraph<Var, Void> computeVariableNamesInterferenceGraph(
208       NodeTraversal t, ControlFlowGraph<Node> cfg, Set<Var> escaped) {
209     UndiGraph<Var, Void> interferenceGraph =
210         LinkedUndirectedGraph.create();
211     Scope scope = t.getScope();
212 
213     // First create a node for each non-escaped variable.
214     for (Iterator<Var> i = scope.getVars(); i.hasNext();) {
215       Var v = i.next();
216       if (!escaped.contains(v)) {
217 
218         // TODO(user): In theory, we CAN coalesce function names just like
219         // any variables. Our Liveness analysis captures this just like it as
220         // described in the specification. However, we saw some zipped and
221         // and unzipped size increase after this. We are not totally sure why
222         // that is but, for now, we will respect the dead functions and not play
223         // around with it.
224         if (!v.getParentNode().isFunction()) {
225           interferenceGraph.createNode(v);
226         }
227       }
228     }
229 
230     // Go through each variable and try to connect them.
231     for (Iterator<Var> i1 = scope.getVars(); i1.hasNext();) {
232       Var v1 = i1.next();
233 
234       NEXT_VAR_PAIR:
235       for (Iterator<Var> i2 = scope.getVars(); i2.hasNext();) {
236         Var v2 = i2.next();
237 
238         // Skip duplicate pairs.
239         if (v1.index >= v2.index) {
240           continue;
241         }
242 
243         if (!interferenceGraph.hasNode(v1) ||
244             !interferenceGraph.hasNode(v2)) {
245           // Skip nodes that were not added. They are globals and escaped
246           // locals. Also avoid merging a variable with itself.
247           continue NEXT_VAR_PAIR;
248         }
249 
250         if (v1.getParentNode().isParamList() &&
251             v2.getParentNode().isParamList()) {
252           interferenceGraph.connectIfNotFound(v1, null, v2);
253           continue NEXT_VAR_PAIR;
254         }
255 
256         // Go through every CFG node in the program and look at
257         // this variable pair. If they are both live at the same
258         // time, add an edge between them and continue to the next pair.
259         NEXT_CROSS_CFG_NODE:
260         for (DiGraphNode<Node, Branch> cfgNode : cfg.getDirectedGraphNodes()) {
261           if (cfg.isImplicitReturn(cfgNode)) {
262             continue NEXT_CROSS_CFG_NODE;
263           }
264 
265           FlowState<LiveVariableLattice> state = cfgNode.getAnnotation();
266           // Check the live states and add edge when possible.
267           if ((state.getIn().isLive(v1) && state.getIn().isLive(v2)) ||
268               (state.getOut().isLive(v1) && state.getOut().isLive(v2))) {
269             interferenceGraph.connectIfNotFound(v1, null, v2);
270             continue NEXT_VAR_PAIR;
271           }
272         }
273 
274         // v1 and v2 might not have an edge between them! woohoo. there's
275         // one last sanity check that we have to do: we have to check
276         // if there's a collision *within* the cfg node.
277         NEXT_INTRA_CFG_NODE:
278         for (DiGraphNode<Node, Branch> cfgNode : cfg.getDirectedGraphNodes()) {
279           if (cfg.isImplicitReturn(cfgNode)) {
280             continue NEXT_INTRA_CFG_NODE;
281           }
282 
283           FlowState<LiveVariableLattice> state = cfgNode.getAnnotation();
284           boolean v1OutLive = state.getOut().isLive(v1);
285           boolean v2OutLive = state.getOut().isLive(v2);
286           CombinedLiveRangeChecker checker = new CombinedLiveRangeChecker(
287               new LiveRangeChecker(v1, v2OutLive ? null : v2),
288               new LiveRangeChecker(v2, v1OutLive ? null : v1));
289           NodeTraversal.traverse(
290               compiler,
291               cfgNode.getValue(),
292               checker);
293           if (checker.connectIfCrossed(interferenceGraph)) {
294             continue NEXT_VAR_PAIR;
295           }
296         }
297       }
298     }
299     return interferenceGraph;
300   }

302   /**
303    * A simple wrapper calls to call two AbstractCfgNodeTraversalCallback
304    * callback during the same traversal.  Both traversals must have the same
305    * "shouldTraverse" conditions.
306    */
307   private static class CombinedLiveRangeChecker
308       extends AbstractCfgNodeTraversalCallback {
309 
310     private final LiveRangeChecker callback1;
311     private final LiveRangeChecker callback2;
312 
313     CombinedLiveRangeChecker(
314         LiveRangeChecker callback1,
315         LiveRangeChecker callback2) {
316       this.callback1 = callback1;
317       this.callback2 = callback2;
318     }
319 
320     @Override
321     public void visit(NodeTraversal t, Node n, Node parent) {
322       if (LiveRangeChecker.shouldVisit(n)) {
323         callback1.visit(t, n, parent);
324         callback2.visit(t, n, parent);
325       }
326     }
327 
328     boolean connectIfCrossed(UndiGraph<Var, Void> interferenceGraph) {
329       if (callback1.crossed || callback2.crossed) {
330         Var v1 = callback1.getDef();
331         Var v2 = callback2.getDef();
332         interferenceGraph.connectIfNotFound(v1, null, v2);
333         return true;
334       }
335       return false;
336     }
337   }

339   /**
340    * Tries to remove variable declaration if the variable has been coalesced
341    * with another variable that has already been declared.
342    */
343   private static void removeVarDeclaration(Node name) {
344     Node var = name.getParent();
345     Node parent = var.getParent();
346 
347     // Special case when we are in FOR-IN loop.
348     if (NodeUtil.isForIn(parent)) {
349       var.removeChild(name);
350       parent.replaceChild(var, name);
351     } else if (var.hasOneChild()) {
352       // The removal is easy when there is only one variable in the VAR node.
353       if (name.hasChildren()) {
354         Node value = name.removeFirstChild();
355         var.removeChild(name);
356         Node assign = IR.assign(name, value).srcref(name);
357 
358         // We don't need to wrapped it with EXPR node if it is within a FOR.
359         if (!parent.isFor()) {
360           assign = NodeUtil.newExpr(assign);
361         }
362         parent.replaceChild(var, assign);
363 
364       } else {
365         // In a FOR( ; ; ) node, we must replace it with an EMPTY or else it
366         // becomes a FOR-IN node.
367         NodeUtil.removeChild(parent, var);
368       }
369     } else {
370       if (!name.hasChildren()) {
371         var.removeChild(name);
372       }
373       // We are going to leave duplicated declaration otherwise.
374     }
375   }

377   private static class LiveRangeChecker
378       extends AbstractCfgNodeTraversalCallback {
379     boolean defFound = false;
380     boolean crossed = false;
381     private final Var def;
382     private final Var use;
383 
384     public LiveRangeChecker(Var def, Var use) {
385       this.def = def;
386       this.use = use;
387     }
388 
389     Var getDef() {
390       return def;
391     }
392 
393     /**
394      * @return Whether any LiveRangeChecker would be interested in the node.
395      */
396     public static boolean shouldVisit(Node n) {
397       return (n.isName()
398         || (n.hasChildren() && n.getFirstChild().isName()));
399     }
400 
401     @Override
402     public void visit(NodeTraversal t, Node n, Node parent) {
403       if (!defFound && isAssignTo(def, n, parent)) {
404         defFound = true;
405       }
406 
407       if (defFound && (use == null || isReadFrom(use, n))) {
408         crossed = true;
409       }
410     }
411 
412     private static boolean isAssignTo(Var var, Node n, Node parent) {
413       if (n.isName() && var.getName().equals(n.getString()) &&
414           parent != null) {
415         if (parent.isParamList()) {
416           // In a function declaration, the formal parameters are assigned.
417           return true;
418         } else if (parent.isVar()) {
419           // If this is a VAR declaration, if the name node has a child, we are
420           // assigning to that name.
421           return n.hasChildren();
422         }
423         return false; // Definitely a read.
424       } else {
425         // Lastly, any assignmentOP is also an assign.
426         Node name = n.getFirstChild();
427         return name != null && name.isName() &&
428           var.getName().equals(name.getString()) &&
429           NodeUtil.isAssignmentOp(n);
430       }
431     }
432 
433     private static boolean isReadFrom(Var var, Node name) {
434       return name != null && name.isName() &&
435           var.getName().equals(name.getString()) &&
436           !NodeUtil.isVarOrSimpleAssignLhs(name, name.getParent());
437     }
