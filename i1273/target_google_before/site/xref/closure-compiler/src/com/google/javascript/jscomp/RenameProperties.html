42  /**
43   * RenameProperties renames properties (including methods) of all JavaScript
44   * objects. This includes prototypes, functions, object literals, etc.
45   *
46   * <p> If provided a VariableMap of previously used names, it tries to reuse
47   * those names.
48   *
49   * <p> To prevent a property from getting renamed you may extern it (add it to
50   * your externs file) or put it in quotes.
51   *
52   * <p> To avoid run-time JavaScript errors, use quotes when accessing properties
53   * that are defined using quotes.
54   *
55   * <pre>
56   *   var a = {'myprop': 0}, b = a['myprop'];  // correct
57   *   var x = {'myprop': 0}, y = x.myprop;     // incorrect
58   * </pre>
59   *
60   */
61  class RenameProperties implements CompilerPass {
62  
63    private final AbstractCompiler compiler;
64    private final boolean generatePseudoNames;
65  
66    /** Property renaming map from a previous compilation. */
67    private final VariableMap prevUsedPropertyMap;
68  
69    private final List<Node> stringNodesToRename = new ArrayList<>();
70    private final Map<Node, Node> callNodeToParentMap =
71        new HashMap<>();
72    private final char[] reservedCharacters;
73  
74    // Map from property name to Property object
75    private final Map<String, Property> propertyMap = new HashMap<>();
76  
77    // Property names that don't get renamed
78    private final Set<String> externedNames = new HashSet<>(
79        Arrays.asList("prototype"));
80  
81    // Names to which properties shouldn't be renamed, to avoid name conflicts
82    private final Set<String> quotedNames = new HashSet<>();
83  
84    private static final Comparator<Property> FREQUENCY_COMPARATOR =
85      new Comparator<Property>() {
86        @Override
87        public int compare(Property p1, Property p2) {
88  
89          /**
90           * First a frequently used names would always be picked first.
91           */
92          if (p1.numOccurrences != p2.numOccurrences) {
93            return p2.numOccurrences - p1.numOccurrences;
94          }
95  
96          /**
97           * Finally, for determinism, we compare them based on the old name.
98           */
99          return p1.oldName.compareTo(p2.oldName);
100        }
101     };
102 
103   /**
104    * The name of a special function that this pass replaces. It takes one
105    * argument: a string literal containing one or more dot-separated JS
106    * identifiers. This pass will replace them as though they were JS property
107    * references. Here are two examples:
108    *    JSCompiler_renameProperty('propertyName') -> 'jYq'
109    *    JSCompiler_renameProperty('myProp.nestedProp.innerProp') -> 'e4.sW.C$'
110    */
111   static final String RENAME_PROPERTY_FUNCTION_NAME =
112       "JSCompiler_renameProperty";
113 
114   static final DiagnosticType BAD_CALL = DiagnosticType.error(
115       "JSC_BAD_RENAME_PROPERTY_FUNCTION_NAME_CALL",
116       "Bad " + RENAME_PROPERTY_FUNCTION_NAME + " call - " +
117       "argument must be a string literal");
118 
119   static final DiagnosticType BAD_ARG = DiagnosticType.error(
120       "JSC_BAD_RENAME_PROPERTY_FUNCTION_NAME_ARG",
121       "Bad " + RENAME_PROPERTY_FUNCTION_NAME + " argument - " +
122       "'{0}' is not a valid JavaScript identifier");
123

130    */
131   RenameProperties(AbstractCompiler compiler, boolean generatePseudoNames) {
132     this(compiler, generatePseudoNames, null, null);
133   }

135   /**
136    * Creates an instance.
137    *
138    * @param compiler The JSCompiler.
139    * @param generatePseudoNames Generate pseudo names. e.g foo -> $foo$ instead
140    *        of compact obfuscated names. This is used for debugging.
141    * @param prevUsedPropertyMap The property renaming map used in a previous
142    *        compilation.
143    */
144   RenameProperties(AbstractCompiler compiler,
145       boolean generatePseudoNames, VariableMap prevUsedPropertyMap) {
146     this(compiler, generatePseudoNames, prevUsedPropertyMap, null);
147   }
148 
149   /**
150    * Creates an instance.
151    *
152    * @param compiler The JSCompiler.
153    * @param generatePseudoNames Generate pseudo names. e.g foo -> $foo$ instead
154    *        of compact obfuscated names. This is used for debugging.
155    * @param prevUsedPropertyMap The property renaming map used in a previous
156    *        compilation.
157    * @param reservedCharacters If specified these characters won't be used in
158    *   generated names
159    */
160   RenameProperties(AbstractCompiler compiler,
161       boolean generatePseudoNames,
162       VariableMap prevUsedPropertyMap,
163       @Nullable char[] reservedCharacters) {
164     this.compiler = compiler;
165     this.generatePseudoNames = generatePseudoNames;
166     this.prevUsedPropertyMap = prevUsedPropertyMap;
167     this.reservedCharacters = reservedCharacters;
168     externedNames.addAll(compiler.getExternProperties());
169   }
170 
171   @Override
172   public void process(Node externs, Node root) {
173     Preconditions.checkState(compiler.getLifeCycleStage().isNormalized());
174 
175     NodeTraversal.traverseEs6(compiler, root, new ProcessProperties());
176 
177     Set<String> reservedNames =
178         new HashSet<>(externedNames.size() + quotedNames.size());
179     reservedNames.addAll(externedNames);
180     reservedNames.addAll(quotedNames);
181 
182     // First, try and reuse as many property names from the previous compilation
183     // as possible.
184     if (prevUsedPropertyMap != null) {
185       reusePropertyNames(reservedNames, propertyMap.values());
186     }
187 
188     compiler.addToDebugLog("JS property assignments:");
189 
190     // Assign names, sorted by descending frequency to minimize code size.
191     Set<Property> propsByFreq = new TreeSet<>(FREQUENCY_COMPARATOR);
192     propsByFreq.addAll(propertyMap.values());
193     generateNames(propsByFreq, reservedNames);
194 
195     // Update the string nodes.
196     boolean changed = false;
197     for (Node n : stringNodesToRename) {
198       String oldName = n.getString();
199       Property p = propertyMap.get(oldName);
200       if (p != null && p.newName != null) {
201         Preconditions.checkState(oldName.equals(p.oldName));
202         n.setString(p.newName);
203         changed = changed || !p.newName.equals(oldName);
204       }
205     }
206 
207     // Update the call nodes.
208     for (Map.Entry<Node, Node> nodeEntry : callNodeToParentMap.entrySet()) {
209       Node parent = nodeEntry.getValue();
210       Node firstArg = nodeEntry.getKey().getFirstChild().getNext();
211       StringBuilder sb = new StringBuilder();
212       for (String oldName : Splitter.on('.').split(firstArg.getString())) {
213         Property p = propertyMap.get(oldName);
214         String replacement;
215         if (p != null && p.newName != null) {
216           Preconditions.checkState(oldName.equals(p.oldName));
217           replacement = p.newName;
218         } else {
219           replacement = oldName;
220         }
221         if (sb.length() > 0) {
222           sb.append('.');
223         }
224         sb.append(replacement);
225       }
226       parent.replaceChild(nodeEntry.getKey(), IR.string(sb.toString()));
227       changed = true;
228     }
229 
230     if (changed) {
231       compiler.reportCodeChange();
232     }
233 
234     compiler.setLifeCycleStage(LifeCycleStage.NORMALIZED_OBFUSCATED);
235   }
236 
237   /**
238    * Runs through the list of properties and renames as many as possible with
239    * names from the previous compilation. Also, updates reservedNames with the
240    * set of reused names.
241    * @param reservedNames Reserved names to use during renaming.
242    * @param allProps Properties to rename.
243    */
244   private void reusePropertyNames(Set<String> reservedNames,
245                                   Collection<Property> allProps) {
246     for (Property prop : allProps) {
247       // Check if this node can reuse a name from a previous compilation - if
248       // it can set the newName for the property too.
249       String prevName = prevUsedPropertyMap.lookupNewName(prop.oldName);
250       if (!generatePseudoNames && prevName != null) {
251         // We can reuse prevName if it's not reserved.
252         if (reservedNames.contains(prevName)) {
253           continue;
254         }
255 
256         prop.newName = prevName;
257         reservedNames.add(prevName);
258       }
259     }
260   }
261 
262   /**
263    * Generates new names for properties.
264    *
265    * @param props Properties to generate new names for
266    * @param reservedNames A set of names to which properties should not be
267    *     renamed
268    */
269   private void generateNames(Set<Property> props, Set<String> reservedNames) {
270     NameGenerator nameGen = new NameGenerator(
271         reservedNames, "", reservedCharacters);
272     for (Property p : props) {
273       if (generatePseudoNames) {
274         p.newName = "$" + p.oldName + "$";
275       } else {
276         // If we haven't already given this property a reusable name.
277         if (p.newName == null) {
278           p.newName = nameGen.generateNextName();
279         }
280       }
281       reservedNames.add(p.newName);
282       compiler.addToDebugLog(p.oldName + " => " + p.newName);
283     }
284   }
285 
286   /**
287    * Gets the property renaming map (the "answer key").
288    *
289    * @return A mapping from original names to new names
290    */
291   VariableMap getPropertyMap() {
292     ImmutableMap.Builder<String, String> map = ImmutableMap.builder();
293     for (Property p : propertyMap.values()) {
294       if (p.newName != null) {
295         map.put(p.oldName, p.newName);
296       }
297     }
298     return new VariableMap(map.build());
299   }

304   /**
305    * A traversal callback that collects property names and counts how
306    * frequently each property name occurs.
307    */
308   private class ProcessProperties extends AbstractPostOrderCallback {
309 
310     @Override
311     public void visit(NodeTraversal t, Node n, Node parent) {
312       switch (n.getType()) {
313         case Token.GETPROP:
314           Node propNode = n.getFirstChild().getNext();
315           if (propNode.isString()) {
316             maybeMarkCandidate(propNode);
317           }
318           break;
319         case Token.OBJECTLIT:
320           for (Node key = n.getFirstChild(); key != null; key = key.getNext()) {
321             if (!key.isQuotedString()) {
322               maybeMarkCandidate(key);
323             } else {
324               // Ensure that we never rename some other property in a way
325               // that could conflict with this quoted key.
326               quotedNames.add(key.getString());
327             }
328           }
329           break;
330         case Token.GETELEM:
331           // If this is a quoted property access (e.g. x['myprop']), we need to
332           // ensure that we never rename some other property in a way that
333           // could conflict with this quoted name.
334           Node child = n.getLastChild();
335           if (child != null && child.isString()) {
336             quotedNames.add(child.getString());
337           }
338           break;
339         case Token.CALL:
340           // We replace a JSCompiler_renameProperty function call with a string
341           // containing the renamed property.
342           Node fnName = n.getFirstChild();
343           if (fnName.isName() &&
344               RENAME_PROPERTY_FUNCTION_NAME.equals(fnName.getString())) {
345             callNodeToParentMap.put(n, parent);
346             countCallCandidates(t, n);
347           }
348           break;
349         case Token.FUNCTION:
350           // We eliminate any stub implementations of JSCompiler_renameProperty
351           // that we encounter.
352           if (NodeUtil.isFunctionDeclaration(n)) {
353             String name = n.getFirstChild().getString();
354             if (RENAME_PROPERTY_FUNCTION_NAME.equals(name)) {
355               if (parent.isExprResult()) {
356                 parent.detachFromParent();
357               } else {
358                 parent.removeChild(n);
359               }
360               compiler.reportCodeChange();
361             }
362           } else if (parent.isName() &&
363                      RENAME_PROPERTY_FUNCTION_NAME.equals(parent.getString())) {
364             Node varNode = parent.getParent();
365             if (varNode.isVar()) {
366               varNode.removeChild(parent);
367               if (!varNode.hasChildren()) {
368                 varNode.detachFromParent();
369               }
370               compiler.reportCodeChange();
371             }
372           }
373           break;
374       }
375     }
376 
377     /**
378      * If a property node is eligible for renaming, stashes a reference to it
379      * and increments the property name's access count.
380      *
381      * @param n The STRING node for a property
382      */
383     private void maybeMarkCandidate(Node n) {
384       String name = n.getString();
385       if (!externedNames.contains(name)) {
386         stringNodesToRename.add(n);
387         countPropertyOccurrence(name);
388       }
389     }
390 
391     /**
392      * Counts references to property names that occur in a special function
393      * call.
394      *
395      * @param callNode The CALL node for a property
396      * @param t The traversal
397      */
398     private void countCallCandidates(NodeTraversal t, Node callNode) {
399       Node firstArg = callNode.getFirstChild().getNext();
400       if (!firstArg.isString()) {
401         t.report(callNode, BAD_CALL);
402         return;
403       }
404 
405       for (String name : firstArg.getString().split("[.]")) {
406         if (!TokenStream.isJSIdentifier(name)) {
407           t.report(callNode, BAD_ARG, name);
408           continue;
409         }
410         if (!externedNames.contains(name)) {
411           countPropertyOccurrence(name);
412         }
413       }
414     }
415 
416     /**
417      * Increments the occurrence count for a property name.
418      *
419      * @param name The property name
420      */
421     private void countPropertyOccurrence(String name) {
422       Property prop = propertyMap.get(name);
423       if (prop == null) {
424         prop = new Property(name);
425         propertyMap.put(name, prop);
426       }
427       prop.numOccurrences++;
428     }
429   }

433   /**
434    * Encapsulates the information needed for renaming a property.
435    */
436   private static class Property {
437     final String oldName;
438     String newName;
439     int numOccurrences;
440 
441     Property(String name) {
442       this.oldName = name;
443     }
