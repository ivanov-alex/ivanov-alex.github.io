200   /**
201    * Creates a scope with all types declared. Declares newly discovered types
202    * and type properties in the type registry.
203    */
204   @Override
205   public TypedScope createScope(Node root, Scope parent) {
206     Preconditions.checkArgument(parent == null || parent instanceof TypedScope);
207     TypedScope typedParent = (TypedScope) parent;
208     // Constructing the global scope is very different than constructing
209     // inner scopes, because only global scopes can contain named classes that
210     // show up in the type registry.
211     TypedScope newScope = null;
212     AbstractScopeBuilder scopeBuilder = null;
213     if (typedParent == null) {
214       JSType globalThis =
215           typeRegistry.getNativeObjectType(JSTypeNative.GLOBAL_THIS);
216 
217       // Mark the main root, the externs root, and the src root
218       // with the global this type.
219       root.setJSType(globalThis);
220       root.getFirstChild().setJSType(globalThis);
221       root.getLastChild().setJSType(globalThis);
222 
223       // Run a first-order analysis over the syntax tree.
224       (new FirstOrderFunctionAnalyzer(compiler, functionAnalysisResults))
225           .process(root.getFirstChild(), root.getLastChild());
226 
227       // Find all the classes in the global scope.
228       newScope = createInitialScope(root);
229 
230       GlobalScopeBuilder globalScopeBuilder = new GlobalScopeBuilder(newScope);
231       scopeBuilder = globalScopeBuilder;
232       NodeTraversal.traverseTyped(compiler, root, scopeBuilder);
233     } else {
234       newScope = new TypedScope(typedParent, root);
235       LocalScopeBuilder localScopeBuilder = new LocalScopeBuilder(newScope);
236       scopeBuilder = localScopeBuilder;
237       localScopeBuilder.build();
238     }
239 
240     scopeBuilder.resolveStubDeclarations();
241 
242     if (typedParent == null) {
243       codingConvention.defineDelegateProxyPrototypeProperties(
244           typeRegistry, newScope, delegateProxyPrototypes,
245           delegateCallingConventions);
246     }
247 
248     newScope.setTypeResolver(scopeBuilder);
249     return newScope;
250   }

252   /**
253    * Patches a given global scope by removing variables previously declared in
254    * a script and re-traversing a new version of that script.
255    *
256    * @param globalScope The global scope generated by {@code createScope}.
257    * @param scriptRoot The script that is modified.
258    */
259   void patchGlobalScope(TypedScope globalScope, Node scriptRoot) {
260     // Preconditions: This is supposed to be called only on (named) SCRIPT nodes
261     // and a global typed scope should have been generated already.
262     Preconditions.checkState(scriptRoot.isScript());
263     Preconditions.checkNotNull(globalScope);
264     Preconditions.checkState(globalScope.isGlobal());
265 
266     String scriptName = NodeUtil.getSourceName(scriptRoot);
267     Preconditions.checkNotNull(scriptName);
268     for (Node node : ImmutableList.copyOf(functionAnalysisResults.keySet())) {
269       if (scriptName.equals(NodeUtil.getSourceName(node))) {
270         functionAnalysisResults.remove(node);
271       }
272     }
273 
274     (new FirstOrderFunctionAnalyzer(
275         compiler, functionAnalysisResults)).process(null, scriptRoot);
276 
277     // TODO(bashir): Variable declaration is not the only side effect of last
278     // global scope generation but here we only wipe that part off!
279 
280     // Remove all variables that were previously declared in this scripts.
281     // First find all vars to remove then remove them because of iterator!
282     Iterator<TypedVar> varIter = globalScope.getVars();
283     List<TypedVar> varsToRemove = new ArrayList<>();
284     while (varIter.hasNext()) {
285       TypedVar oldVar = varIter.next();
286       if (scriptName.equals(oldVar.getInputName())) {
287         varsToRemove.add(oldVar);
288       }
289     }
290     for (TypedVar var : varsToRemove) {
291       globalScope.undeclare(var);
292       globalScope.getTypeOfThis().toObjectType().removeProperty(var.getName());
293     }
294 
295     // Now re-traverse the given script.
296     GlobalScopeBuilder scopeBuilder = new GlobalScopeBuilder(globalScope);
297     NodeTraversal.traverseTyped(compiler, scriptRoot, scopeBuilder);
298   }

352   private static class DiscoverEnumsAndTypedefs
353       extends AbstractShallowStatementCallback {
354     private final JSTypeRegistry registry;
355 
356     DiscoverEnumsAndTypedefs(JSTypeRegistry registry) {
357       this.registry = registry;
358     }

360     @Override
361     public void visit(NodeTraversal t, Node node, Node parent) {
362       switch (node.getType()) {
363         case Token.VAR:
364           for (Node child = node.getFirstChild();
365                child != null; child = child.getNext()) {
366             identifyNameNode(
367                 child, NodeUtil.getBestJSDocInfo(child));
368           }
369           break;
370         case Token.EXPR_RESULT:
371           Node firstChild = node.getFirstChild();
372           if (firstChild.isAssign()) {
373             identifyNameNode(
374                 firstChild.getFirstChild(), firstChild.getJSDocInfo());
375           } else {
376             identifyNameNode(
377                 firstChild, firstChild.getJSDocInfo());
378           }
379           break;
380       }
381     }

383     private void identifyNameNode(
384         Node nameNode, JSDocInfo info) {
385       if (info != null && nameNode.isQualifiedName()) {
386         if (info.hasEnumParameterType()) {
387           registry.identifyNonNullableName(nameNode.getQualifiedName());
388         } else if (info.hasTypedefType()) {
389           registry.identifyNonNullableName(nameNode.getQualifiedName());
390         }
391       }
392     }

399   private abstract class AbstractScopeBuilder
400       implements NodeTraversal.Callback, TypedScope.TypeResolver {
401 
402     /**
403      * The scope that we're building.
404      */
405     final TypedScope scope;
406 
407     private final List<DeferredSetType> deferredSetTypes =
408          new ArrayList<>();
409 
410     /**
411      * Functions that we found in the global scope and not in externs.
412      */
413     private final List<Node> nonExternFunctions = new ArrayList<>();
414 
415     /**
416      * Object literals with a @lends annotation aren't analyzed until we
417      * reach the root of the statement they're defined in.
418      *
419      * This ensures that if there are any @lends annotations on the object
420      * literals, the type on the @lends annotation resolves correctly.
421      *
422      * For more information, see
423      * http://code.google.com/p/closure-compiler/issues/detail?id=314
424      */
425     private List<Node> lentObjectLiterals = null;
426 
427     /**
428      * Type-less stubs.
429      *
430      * If at the end of traversal, we still don't have types for these
431      * stubs, then we should declare UNKNOWN types.
432      */
433     private final List<StubDeclaration> stubDeclarations =
434          new ArrayList<>();
435 
436     /**
437      * The current source file that we're in.
438      */
439     private String sourceName = null;
440 
441     /**
442      * The InputId of the current node.
443      */
444     private InputId inputId;
445 
446     private AbstractScopeBuilder(TypedScope scope) {
447       this.scope = scope;
448     }
449 
450     void setDeferredType(Node node, JSType type) {
451       deferredSetTypes.add(new DeferredSetType(node, type));
452     }

454     @Override
455     public void resolveTypes() {
456       // Resolve types and attach them to nodes.
457       for (DeferredSetType deferred : deferredSetTypes) {
458         deferred.resolve(scope);
459       }
460 
461       // Resolve types and attach them to scope slots.
462       Iterator<TypedVar> vars = scope.getVars();
463       while (vars.hasNext()) {
464         vars.next().resolveType(typeParsingErrorReporter);
465       }
466 
467       // Tell the type registry that any remaining types
468       // are unknown.
469       typeRegistry.resolveTypesInScope(scope);
470     }

472     @Override
473     public final boolean shouldTraverse(NodeTraversal t, Node n, Node parent) {
474       inputId = t.getInputId();
475       if (n.isFunction() ||
476           n.isScript()) {
477         Preconditions.checkNotNull(inputId);
478         sourceName = NodeUtil.getSourceName(n);
479       }
480 
481       // We do want to traverse the name of a named function, but we don't
482       // want to traverse the arguments or body.
483       boolean descend = parent == null || !parent.isFunction() ||
484           n == parent.getFirstChild() || parent == scope.getRootNode();
485 
486       if (descend) {
487         // Handle hoisted functions on pre-order traversal, so that they
488         // get hit before other things in the scope.
489         if (NodeUtil.isStatementParent(n)) {
490           for (Node child = n.getFirstChild();
491                child != null;
492                child = child.getNext()) {
493             if (NodeUtil.isHoistedFunctionDeclaration(child)) {
494               defineFunctionLiteral(child);
495             }
496           }
497         }
498       }
499 
500       return descend;
501     }
502 
503     @Override
504     public void visit(NodeTraversal t, Node n, Node parent) {
505       inputId = t.getInputId();
506       attachLiteralTypes(n);
507 
508       switch (n.getType()) {
509         case Token.CALL:
510           checkForClassDefiningCalls(n);
511           checkForCallingConventionDefiningCalls(n, delegateCallingConventions);
512           break;
513 
514         case Token.FUNCTION:
515           if (t.getInput() == null || !t.getInput().isExtern()) {
516             nonExternFunctions.add(n);
517           }
518 
519           // Hoisted functions are handled during pre-traversal.
520           if (!NodeUtil.isHoistedFunctionDeclaration(n)) {
521             defineFunctionLiteral(n);
522           }
523           break;
524 
525         case Token.ASSIGN:
526           // Handle initialization of properties.
527           Node firstChild = n.getFirstChild();
528           if (firstChild.isGetProp() &&
529               firstChild.isQualifiedName()) {
530             maybeDeclareQualifiedName(t, n.getJSDocInfo(),
531                 firstChild, n, firstChild.getNext());
532           }
533           break;
534 
535         case Token.CATCH:
536           defineCatch(n);
537           break;
538 
539         case Token.VAR:
540           defineVar(n);
541           break;
542 
543         case Token.GETPROP:
544           // Handle stubbed properties.
545           if (parent.isExprResult() &&
546               n.isQualifiedName()) {
547             maybeDeclareQualifiedName(t, n.getJSDocInfo(), n, parent, null);
548           }
549           break;
550       }
551 
552       // Analyze any @lends object literals in this statement.
553       if (n.getParent() != null && NodeUtil.isStatement(n) &&
554           lentObjectLiterals != null) {
555         for (Node objLit : lentObjectLiterals) {
556           defineObjectLiteral(objLit);
557         }
558         lentObjectLiterals.clear();
559       }
560     }

562     private void attachLiteralTypes(Node n) {
563       switch (n.getType()) {
564         case Token.NULL:
565           n.setJSType(getNativeType(NULL_TYPE));
566           break;
567 
568         case Token.VOID:
569           n.setJSType(getNativeType(VOID_TYPE));
570           break;
571 
572         case Token.STRING:
573           n.setJSType(getNativeType(STRING_TYPE));
574           break;
575 
576         case Token.NUMBER:
577           n.setJSType(getNativeType(NUMBER_TYPE));
578           break;
579 
580         case Token.TRUE:
581         case Token.FALSE:
582           n.setJSType(getNativeType(BOOLEAN_TYPE));
583           break;
584 
585         case Token.REGEXP:
586           n.setJSType(getNativeType(REGEXP_TYPE));
587           break;
588 
589         case Token.OBJECTLIT:
590           JSDocInfo info = n.getJSDocInfo();
591           if (info != null &&
592               info.getLendsName() != null) {
593             if (lentObjectLiterals == null) {
594               lentObjectLiterals = new ArrayList<>();
595             }
596             lentObjectLiterals.add(n);
597           } else {
598             defineObjectLiteral(n);
599           }
600           break;
601 
602         // NOTE(johnlenz): If we ever support Array tuples,
603         // we will need to handle them here as we do object literals
604         // above.
605         case Token.ARRAYLIT:
606           n.setJSType(getNativeType(ARRAY_TYPE));
607           break;
608       }
609     }

611     private void defineObjectLiteral(Node objectLit) {
612       // Handle the @lends annotation.
613       JSType type = null;
614       JSDocInfo info = objectLit.getJSDocInfo();
615       if (info != null && info.getLendsName() != null) {
616         String lendsName = info.getLendsName();
617         TypedVar lendsVar = scope.getVar(lendsName);
618         if (lendsVar == null) {
619           compiler.report(
620               JSError.make(objectLit, UNKNOWN_LENDS, lendsName));
621         } else {
622           type = lendsVar.getType();
623           if (type == null) {
624             type = unknownType;
625           }
626           if (!type.isSubtype(typeRegistry.getNativeType(OBJECT_TYPE))) {
627             compiler.report(
628                 JSError.make(objectLit, LENDS_ON_NON_OBJECT,
629                     lendsName, type.toString()));
630             type = null;
631           } else {
632             objectLit.setJSType(type);
633           }
634         }
635       }
636 
637       info = NodeUtil.getBestJSDocInfo(objectLit);
638       Node lValue = NodeUtil.getBestLValue(objectLit);
639       String lValueName = NodeUtil.getBestLValueName(lValue);
640       boolean createdEnumType = false;
641       if (info != null && info.hasEnumParameterType()) {
642         type = createEnumTypeFromNodes(objectLit, lValueName, info, lValue);
643         createdEnumType = true;
644       }
645 
646       if (type == null) {

844     /**
845      * Creates a new function type, based on the given nodes.
846      *
847      * This handles two cases that are semantically very different, but
848      * are not mutually exclusive:
849      * - A function literal that needs a type attached to it.
850      * - An assignment expression with function-type info in the JsDoc.
851      *
852      * All parameters are optional, and we will do the best we can to create
853      * a function type.
854      *
855      * This function will always create a function type, so only call it if
856      * you're sure that's what you want.
857      *
858      * @param rValue The function node.
859      * @param name the function's name
860      * @param info the {@link JSDocInfo} attached to the function definition
861      * @param lvalueNode The node where this function is being
862      *     assigned. For example, {@code A.prototype.foo = ...} would be used to
863      *     determine that this function is a method of A.prototype. May be
864      *     null to indicate that this is not being assigned to a qualified name.
865      */
866     private FunctionType createFunctionTypeFromNodes(
867         @Nullable Node rValue,
868         @Nullable String name,
869         @Nullable JSDocInfo info,
870         @Nullable Node lvalueNode) {
871       FunctionType functionType = null;
872       if (rValue != null && rValue.isQualifiedName() && scope.isGlobal()) {
873         TypedVar var = scope.getVar(rValue.getQualifiedName());
874         if (var != null && var.getType() != null &&
875             var.getType().isFunctionType()) {
876           FunctionType aliasedType  = var.getType().toMaybeFunctionType();
877           if ((aliasedType.isConstructor() || aliasedType.isInterface())
878               && !isGoogAbstractMethod(rValue)) {
879             functionType = aliasedType;
880 
881             // TODO(nick): Remove this. This should already be handled by
882             // normal type resolution.
883             if (name != null && scope.isGlobal()) {
884               typeRegistry.declareType(name, functionType.getInstanceType());
885             }
886           }
887         }
888       }
889 
890       if (functionType == null) {
891         Node errorRoot = rValue == null ? lvalueNode : rValue;
892         boolean isFnLiteral =
893             rValue != null && rValue.isFunction();
894         Node fnRoot = isFnLiteral ? rValue : null;
895         Node parametersNode = isFnLiteral ?
896             rValue.getFirstChild().getNext() : null;
897 
898         if (info != null && info.hasType()) {
899           JSType type = info.getType().evaluate(scope, typeRegistry);
900 
901           // Known to be not null since we have the FUNCTION token there.
902           type = type.restrictByNotNullOrUndefined();
903           if (type.isFunctionType()) {
904             functionType = type.toMaybeFunctionType();
905             functionType.setJSDocInfo(info);
906           }
907         }
908 
909         if (functionType == null) {
910           // Find the type of any overridden function.
911           Node ownerNode = NodeUtil.getBestLValueOwner(lvalueNode);
912           String ownerName = NodeUtil.getBestLValueName(ownerNode);
913           TypedVar ownerVar = null;
914           String propName = null;
915           ObjectType ownerType = null;
916           if (ownerName != null) {
917             ownerVar = scope.getVar(ownerName);
918             if (ownerVar != null) {
919               ownerType = ObjectType.cast(ownerVar.getType());
920             }
921             if (name != null) {
922               propName = name.substring(ownerName.length() + 1);
923             }
924           }
925 
926           ObjectType prototypeOwner = getPrototypeOwnerType(ownerType);
927           TemplateTypeMap prototypeOwnerTypeMap = null;
928           if (prototypeOwner != null &&
929               prototypeOwner.getTypeOfThis() != null) {
930             prototypeOwnerTypeMap =
931                 prototypeOwner.getTypeOfThis().getTemplateTypeMap();
932           }
933 
934           FunctionType overriddenType = null;
935           if (ownerType != null && propName != null) {
936             overriddenType = findOverriddenFunction(
937                 ownerType, propName, prototypeOwnerTypeMap);
938           }
939 
940           FunctionTypeBuilder builder =
941               new FunctionTypeBuilder(name, compiler, errorRoot,
942                   scope)
943               .setContents(getFunctionAnalysisResults(fnRoot))
944               .inferFromOverriddenFunction(overriddenType, parametersNode)
945               .inferTemplateTypeName(info, prototypeOwner)
946               .inferInheritance(info);
947 
948           if (info == null || !info.hasReturnType()) {
949             /**
950              * when there is no {@code @return} annotation, look for inline
951              * return type declaration
952              */
953             if (rValue != null && rValue.isFunction() &&
954                 rValue.getFirstChild() != null) {
955               JSDocInfo nameDocInfo = rValue.getFirstChild().getJSDocInfo();
956               builder.inferReturnType(nameDocInfo, true);
957             }
958           } else {
959             builder.inferReturnType(info, false);
960           }
961 
962           // Infer the context type.
963           boolean searchedForThisType = false;
964           if (ownerType != null && ownerType.isFunctionPrototypeType() &&
965               ownerType.getOwnerFunction().hasInstanceType()) {
966             builder.inferThisType(
967                 info, ownerType.getOwnerFunction().getInstanceType());
968             searchedForThisType = true;
969           } else if (ownerNode != null && ownerNode.isThis()) {
970             // If we have a 'this' node, use the scope type.
971             builder.inferThisType(info, scope.getTypeOfThis());
972             searchedForThisType = true;
973           }
974 
975           if (!searchedForThisType) {
976             builder.inferThisType(info);
977           }
978 
979           functionType = builder
980               .inferParameterTypes(parametersNode, info)
981               .buildAndRegister();
982         }
983       }
984       // set structural interface matching flag
985       if (info != null && info.isInterface() && info.usesImplicitMatch()) {
986         functionType.setImplicitMatch(true);
987       }
988       // all done
989       return functionType;
990     }

1010     /**
1011      * Find the function that's being overridden on this type, if any.
1012      */
1013     private FunctionType findOverriddenFunction(
1014         ObjectType ownerType, String propName, TemplateTypeMap typeMap) {
1015       FunctionType result = null;
1016 
1017       // First, check to see if the property is implemented
1018       // on a superclass.
1019       JSType propType = ownerType.getPropertyType(propName);
1020       if (propType != null && propType.isFunctionType()) {
1021         result =  propType.toMaybeFunctionType();
1022       } else {
1023         // If it's not, then check to see if it's implemented
1024         // on an implemented interface.
1025         for (ObjectType iface :
1026                  ownerType.getCtorImplementedInterfaces()) {
1027           propType = iface.getPropertyType(propName);
1028           if (propType != null && propType.isFunctionType()) {
1029             result = propType.toMaybeFunctionType();
1030             break;
1031           }
1032         }
1033       }
1034 
1035       if (result != null && typeMap != null && !typeMap.isEmpty()) {
1036         result = result.visit(
1037             new TemplateTypeMapReplacer(typeRegistry, typeMap))
1038             .toMaybeFunctionType();
1039       }
1040 
1041       return result;
1042     }

1045      * Creates a new enum type, based on the given nodes.
1046      *
1047      * This handles two cases that are semantically very different, but
1048      * are not mutually exclusive:
1049      * - An object literal that needs an enum type attached to it.
1050      * - An assignment expression with an enum tag in the JsDoc.
1051      *
1052      * This function will always create an enum type, so only call it if
1053      * you're sure that's what you want.
1054      *
1055      * @param rValue The node of the enum.
1056      * @param name The enum's name
1057      * @param info The {@link JSDocInfo} attached to the enum definition.
1058      * @param lValueNode The node where this function is being
1059      *     assigned.
1060      */
1061     private EnumType createEnumTypeFromNodes(Node rValue, String name,
1062         JSDocInfo info, Node lValueNode) {
1063       Preconditions.checkNotNull(info);
1064       Preconditions.checkState(info.hasEnumParameterType());
1065 
1066       EnumType enumType = null;
1067       if (rValue != null && rValue.isQualifiedName()) {
1068         // Handle an aliased enum.
1069         TypedVar var = scope.getVar(rValue.getQualifiedName());
1070         if (var != null && var.getType() instanceof EnumType) {
1071           enumType = (EnumType) var.getType();
1072         }
1073       }
1074 
1075       if (enumType == null) {
1076         JSType elementsType =
1077             info.getEnumParameterType().evaluate(scope, typeRegistry);
1078         enumType = typeRegistry.createEnumType(name, rValue, elementsType);
1079 
1080         if (rValue != null && rValue.isObjectLit()) {
1081           // collect enum elements
1082           Node key = rValue.getFirstChild();

1084             String keyName = NodeUtil.getStringValue(key);
1085             if (keyName == null) {
1086               // GET and SET don't have a String value;

1089             } else if (!codingConvention.isValidEnumKey(keyName)) {
1090               compiler.report(
1091                   JSError.make(key, ENUM_NOT_CONSTANT, keyName));
1092             } else {
1093               enumType.defineElement(keyName, key);
1094             }
1095             key = key.getNext();
1096           }
1097         }
1098       }
1099 
1100       if (name != null && scope.isGlobal()) {
1101         typeRegistry.declareType(name, enumType.getElementsType());
1102       }
1103 
1104       return enumType;
1105     }

1152     /**
1153      * Defines a symbol in the current scope.
1154      *
1155      * @param n the defining NAME or GETPROP or object literal key node.
1156      * @param parent the {@code n}'s parent.
1157      * @param variableName The name that this should be known by.
1158      * @param type the variable's type. It may be {@code null} if
1159      *     {@code inferred} is {@code true}.
1160      * @param inferred Whether the type is inferred or declared.
1161      */
1162     void defineSlot(Node n, Node parent, String variableName,
1163         JSType type, boolean inferred) {
1164       Preconditions.checkArgument(!variableName.isEmpty());
1165 
1166       boolean isGlobalVar = n.isName() && scope.isGlobal();
1167       boolean shouldDeclareOnGlobalThis =
1168           isGlobalVar &&
1169           (parent.isVar() ||
1170            parent.isFunction());
1171 
1172       // If n is a property, then we should really declare it in the
1173       // scope where the root object appears. This helps out people
1174       // who declare "global" names in an anonymous namespace.
1175       TypedScope scopeToDeclareIn = scope;
1176       if (n.isGetProp() && !scope.isGlobal() &&
1177           isQnameRootedInGlobalScope(n)) {
1178         TypedScope globalScope = scope.getGlobalScope();
1179 
1180         // don't try to declare in the global scope if there's
1181         // already a symbol there with this name.
1182         if (!globalScope.isDeclared(variableName, false)) {
1183           scopeToDeclareIn = scope.getGlobalScope();
1184         }
1185       }
1186 
1187       // The input may be null if we are working with a AST snippet. So read
1188       // the extern info from the node.
1189       TypedVar newVar = null;
1190 
1191       // declared in closest scope?
1192       CompilerInput input = compiler.getInput(inputId);
1193       if (scopeToDeclareIn.isDeclared(variableName, false)) {
1194         TypedVar oldVar = scopeToDeclareIn.getVar(variableName);
1195         newVar = validator.expectUndeclaredVariable(
1196             sourceName, input, n, parent, oldVar, variableName, type);
1197       } else {
1198         if (type != null) {
1199           setDeferredType(n, type);
1200         }
1201 
1202         newVar =
1203           scopeToDeclareIn.declare(variableName, n, type, input, inferred);
1204 
1205         if (type instanceof EnumType) {
1206           Node initialValue = newVar.getInitialValue();
1207           boolean isValidValue = initialValue != null &&
1208               (initialValue.isObjectLit() ||
1209                initialValue.isQualifiedName());
1210           if (!isValidValue) {
1211             compiler.report(JSError.make(n, ENUM_INITIALIZER));
1212           }
1213         }
1214       }
1215 
1216       // We need to do some additional work for constructors and interfaces.
1217       FunctionType fnType = JSType.toMaybeFunctionType(type);
1218       if (fnType != null &&
1219           // We don't want to look at empty function types.
1220           !type.isEmptyType()) {
1221 
1222         // We want to make sure that when we declare a new instance type
1223         // (with @constructor) that there's actually a ctor for it.
1224         // This doesn't apply to structural constructors (like
1225         // function(new:Array). Checking the constructed type against
1226         // the variable name is a sufficient check for this.
1227         if ((fnType.isConstructor() || fnType.isInterface()) &&
1228             variableName.equals(fnType.getReferenceName())) {
1229           finishConstructorDefinition(n, variableName, fnType, scopeToDeclareIn,
1230                                       input, newVar);
1231         }
1232       }
1233 
1234       if (shouldDeclareOnGlobalThis) {
1235         ObjectType globalThis =
1236             typeRegistry.getNativeObjectType(GLOBAL_THIS);
1237         if (inferred) {
1238           globalThis.defineInferredProperty(variableName,
1239               type == null ?
1240               getNativeType(JSTypeNative.NO_TYPE) :
1241               type,
1242               n);
1243         } else {
1244           globalThis.defineDeclaredProperty(variableName, type, n);
1245         }
1246       }
1247 
1248       if (isGlobalVar && "Window".equals(variableName)
1249           && type != null
1250           && type.isFunctionType()
1251           && type.isConstructor()) {
1252         FunctionType globalThisCtor =
1253             typeRegistry.getNativeObjectType(GLOBAL_THIS).getConstructor();
1254         globalThisCtor.getInstanceType().clearCachedValues();
1255         globalThisCtor.getPrototype().clearCachedValues();
1256         globalThisCtor
1257             .setPrototypeBasedOn((type.toMaybeFunctionType()).getInstanceType());
1258       }
1259     }

1261     private void finishConstructorDefinition(
1262         Node n, String variableName, FunctionType fnType,
1263         TypedScope scopeToDeclareIn, CompilerInput input, TypedVar newVar) {
1264       // Declare var.prototype in the scope chain.
1265       FunctionType superClassCtor = fnType.getSuperClassConstructor();
1266       Property prototypeSlot = fnType.getSlot("prototype");
1267 
1268       // When we declare the function prototype implicitly, we
1269       // want to make sure that the function and its prototype
1270       // are declared at the same node. We also want to make sure
1271       // that the if a symbol has both a TypedVar and a JSType, they have
1272       // the same node.
1273       //
1274       // This consistency is helpful to users of SymbolTable,
1275       // because everything gets declared at the same place.
1276       prototypeSlot.setNode(n);
1277 
1278       String prototypeName = variableName + ".prototype";
1279 
1280       // There are some rare cases where the prototype will already
1281       // be declared. See TypedScopeCreatorTest#testBogusPrototypeInit.
1282       // Fortunately, other warnings will complain if this happens.
1283       TypedVar prototypeVar = scopeToDeclareIn.getVar(prototypeName);
1284       if (prototypeVar != null && prototypeVar.scope == scopeToDeclareIn) {
1285         scopeToDeclareIn.undeclare(prototypeVar);
1286       }
1287 
1288       scopeToDeclareIn.declare(prototypeName,
1289           n, prototypeSlot.getType(), input,
1290           /* declared iff there's an explicit supertype */
1291           superClassCtor == null ||

1329     /**
1330      * Look for a type declaration on a property assignment
1331      * (in an ASSIGN or an object literal key).
1332      * @param info The doc info for this property.
1333      * @param lValue The l-value node.
1334      * @param rValue The node that {@code n} is being initialized to,
1335      *     or {@code null} if this is a stub declaration.
1336      */
1337     JSType getDeclaredType(JSDocInfo info, Node lValue,
1338         @Nullable Node rValue) {
1339       if (info != null && info.hasType()) {
1340         return getDeclaredTypeInAnnotation(lValue, info);
1341       } else if (rValue != null && rValue.isFunction() &&
1342           shouldUseFunctionLiteralType(
1343               JSType.toMaybeFunctionType(rValue.getJSType()), info, lValue)) {
1344         return rValue.getJSType();
1345       } else if (info != null) {
1346         if (info.hasEnumParameterType()) {
1347           if (rValue != null && rValue.isObjectLit()) {
1348             return rValue.getJSType();
1349           } else {
1350             return createEnumTypeFromNodes(
1351                 rValue, lValue.getQualifiedName(), info, lValue);
1352           }
1353         } else if (info.isConstructorOrInterface()) {
1354           return createFunctionTypeFromNodes(
1355               rValue, lValue.getQualifiedName(), info, lValue);
1356         }
1357       }
1358 
1359       // Check if this is constant, and if it has a known type.
1360       if (NodeUtil.isConstantDeclaration(
1361               compiler.getCodingConvention(), info, lValue)) {
1362         if (rValue != null) {
1363           JSType rValueType = getDeclaredRValueType(lValue, rValue);
1364           if (rValueType == null) {
1365             // Only warn if the user has explicitly declared a value as
1366             // const and they have explicitly not provided a type.
1367             boolean isTypelessConstDecl = info != null
1368                 && info.isConstant()
1369                 && !info.hasType();
1370             if (isTypelessConstDecl) {
1371               compiler.report(JSError.make(lValue, CANNOT_INFER_CONST_TYPE));
1372             }
1373           } else {
1374             return rValueType;
1375           }
1376         }
1377       }
1378 
1379       return getDeclaredTypeInAnnotation(lValue, info);
1380     }

1382     /**
1383      * Check for common idioms of a typed R-value assigned to a const L-value.
1384      *
1385      * Normally, we would only want this sort of propagation to happen under
1386      * type inference. But we want a declared const to be nameable in a type
1387      * annotation, so we need to figure out the type before we try to resolve
1388      * the annotation.
1389      */
1390     private JSType getDeclaredRValueType(Node lValue, Node rValue) {
1391       // If rValue has a type-cast, we use the type in the type-cast.
1392       JSDocInfo rValueInfo = rValue.getJSDocInfo();
1393       if (rValue.isCast() && rValueInfo != null && rValueInfo.hasType()) {
1394         return rValueInfo.getType().evaluate(scope, typeRegistry);
1395       }
1396 
1397       // Check if the type has already been computed during scope-creation.
1398       // This is mostly useful for literals like BOOLEAN, NUMBER, STRING, and
1399       // OBJECT_LITERAL
1400       JSType type = rValue.getJSType();
1401       if (type != null && !type.isUnknownType()) {
1402         return type;
1403       }
1404 
1405       // If rValue is a name, try looking it up in the current scope.
1406       if (rValue.isQualifiedName()) {
1407         return lookupQualifiedName(rValue);
1408       }
1409 
1410       // Check for simple invariant operations, such as "!x" or "+x" or "''+x"
1411       if (NodeUtil.isBooleanResult(rValue)) {
1412         return getNativeType(BOOLEAN_TYPE);
1413       }
1414 
1415       if (NodeUtil.isNumericResult(rValue)) {
1416         return getNativeType(NUMBER_TYPE);
1417       }
1418 
1419       if (NodeUtil.isStringResult(rValue)) {
1420         return getNativeType(STRING_TYPE);
1421       }
1422 
1423       if (rValue.isNew() && rValue.getFirstChild().isQualifiedName()) {
1424         JSType targetType = lookupQualifiedName(rValue.getFirstChild());
1425         if (targetType != null) {
1426           FunctionType fnType = targetType
1427               .restrictByNotNullOrUndefined()
1428               .toMaybeFunctionType();
1429           if (fnType != null && fnType.hasInstanceType()) {
1430             return fnType.getInstanceType();
1431           }
1432         }
1433       }
1434 
1435       // Check for a very specific JS idiom:
1436       // var x = x || TYPE;
1437       // This is used by Closure's base namespace for esoteric
1438       // reasons, so we only really care about that case.
1439       if (rValue.isOr()) {
1440         Node firstClause = rValue.getFirstChild();
1441         Node secondClause = firstClause.getNext();
1442         boolean namesMatch = firstClause.isName()
1443             && lValue.isName()
1444             && firstClause.getString().equals(lValue.getString());
1445         if (namesMatch) {
1446           type = secondClause.getJSType();
1447           if (type != null && !type.isUnknownType()) {
1448             return type;
1449           }
1450         }
1451       }
1452 
1453       return null;
1454     }

1456     private JSType lookupQualifiedName(Node n) {
1457       String name = n.getQualifiedName();
1458       TypedVar slot = scope.getVar(name);
1459       if (slot != null && !slot.isTypeInferred()) {
1460         JSType type = slot.getType();
1461         if (type != null && !type.isUnknownType()) {
1462           return type;
1463         }
1464       } else if (n.isGetProp()) {
1465         JSType type = lookupQualifiedName(n.getFirstChild());
1466         if (type != null && type.isRecordType()) {
1467           JSType propType = type.findPropertyType(
1468              n.getLastChild().getString());
1469           return propType;
1470         }
1471       }
1472       return null;
1473     }

1485      * Look for class-defining calls.
1486      * Because JS has no 'native' syntax for defining classes,
1487      * this is often very coding-convention dependent and business-logic heavy.
1488      */
1489     private void checkForClassDefiningCalls(Node n) {
1490       SubclassRelationship relationship =
1491           codingConvention.getClassesDefinedByCall(n);
1492       if (relationship != null) {
1493         ObjectType superClass = TypeValidator.getInstanceOfCtor(
1494             scope.getVar(relationship.superclassName));
1495         ObjectType subClass = TypeValidator.getInstanceOfCtor(
1496             scope.getVar(relationship.subclassName));
1497         if (superClass != null && subClass != null) {
1498           // superCtor and subCtor might be structural constructors
1499           // (like {function(new:Object)}) so we need to resolve them back
1500           // to the original ctor objects.
1501           FunctionType superCtor = superClass.getConstructor();
1502           FunctionType subCtor = subClass.getConstructor();
1503           if (superCtor != null && subCtor != null) {
1504             codingConvention.applySubclassRelationship(
1505                 superCtor, subCtor, relationship.type);
1506           }
1507         }
1508       }
1509 
1510       String singletonGetterClassName =
1511           codingConvention.getSingletonGetterClassName(n);
1512       if (singletonGetterClassName != null) {
1513         ObjectType objectType = ObjectType.cast(
1514             typeRegistry.getType(singletonGetterClassName));
1515         if (objectType != null) {
1516           FunctionType functionType = objectType.getConstructor();
1517 
1518           if (functionType != null) {
1519             FunctionType getterType = typeRegistry.createFunctionType(objectType);
1520             codingConvention.applySingletonGetterOld(functionType, getterType, objectType);
1521           }
1522         }
1523       }
1524 
1525       DelegateRelationship delegateRelationship =
1526           codingConvention.getDelegateRelationship(n);
1527       if (delegateRelationship != null) {
1528         applyDelegateRelationship(delegateRelationship);
1529       }

1532           codingConvention.getObjectLiteralCast(n);
1533       if (objectLiteralCast != null) {
1534         if (objectLiteralCast.diagnosticType == null) {
1535           ObjectType type = ObjectType.cast(
1536               typeRegistry.getType(objectLiteralCast.typeName));
1537           if (type != null && type.getConstructor() != null) {
1538             setDeferredType(objectLiteralCast.objectNode, type);
1539             objectLiteralCast.objectNode.putBooleanProp(
1540                 Node.REFLECTED_OBJECT, true);
1541           } else {
1542             compiler.report(JSError.make(n, CONSTRUCTOR_EXPECTED));
1543           }
1544         } else {
1545           compiler.report(JSError.make(n, objectLiteralCast.diagnosticType));
1546         }
1547       }

1591     /**
1592      * Declare the symbol for a qualified name in the global scope.
1593      *
1594      * @param info The doc info for this property.
1595      * @param n A top-level GETPROP node (it should not be contained inside
1596      *     another GETPROP).
1597      * @param parent The parent of {@code n}.
1598      * @param rhsValue The node that {@code n} is being initialized to,
1599      *     or {@code null} if this is a stub declaration.
1600      */
1601     void maybeDeclareQualifiedName(NodeTraversal t, JSDocInfo info,
1602         Node n, Node parent, Node rhsValue) {
1603       Node ownerNode = n.getFirstChild();
1604       String ownerName = ownerNode.getQualifiedName();
1605       String qName = n.getQualifiedName();
1606       String propName = n.getLastChild().getString();
1607       Preconditions.checkArgument(qName != null && ownerName != null);
1608 
1609       // Precedence of type information on GETPROPs:
1610       // 1) @type annotation / @enum annotation
1611       // 2) ASSIGN to FUNCTION literal
1612       // 3) @param/@return annotation (with no function literal)
1613       // 4) ASSIGN to something marked @const
1614       // 5) ASSIGN to anything else
1615       //
1616       // 1, 3, and 4 are declarations, 5 is inferred, and 2 is a declaration iff
1617       // the function has JsDoc or has not been declared before.
1618       //
1619       // FUNCTION literals are special because TypedScopeCreator is very smart
1620       // about getting as much type information as possible for them.
1621 
1622       // Determining type for #1 + #2 + #3 + #4
1623       JSType valueType = getDeclaredType(info, n, rhsValue);
1624       if (valueType == null && rhsValue != null) {
1625         // Determining type for #5
1626         valueType = rhsValue.getJSType();
1627       }
1628 
1629       // Function prototypes are special.
1630       // It's a common JS idiom to do:
1631       // F.prototype = { ... };
1632       // So if F does not have an explicitly declared super type,
1633       // allow F.prototype to be redefined arbitrarily.
1634       if ("prototype".equals(propName)) {
1635         TypedVar qVar = scope.getVar(qName);
1636         if (qVar != null) {
1637           // If the programmer has declared that F inherits from Super,
1638           // and they assign F.prototype to an object literal,
1639           // then they are responsible for making sure that the object literal's
1640           // implicit prototype is set up appropriately. We just obey
1641           // the @extends tag.
1642           ObjectType qVarType = ObjectType.cast(qVar.getType());
1643           if (qVarType != null &&
1644               rhsValue != null &&
1645               rhsValue.isObjectLit()) {
1646             typeRegistry.resetImplicitPrototype(
1647                 rhsValue.getJSType(), qVarType.getImplicitPrototype());
1648           } else if (!qVar.isTypeInferred()) {
1649             // If the programmer has declared that F inherits from Super,
1650             // and they assign F.prototype to some arbitrary expression,
1651             // there's not much we can do. We just ignore the expression,
1652             // and hope they've annotated their code in a way to tell us
1653             // what props are going to be on that prototype.
1654             return;
1655           }
1656 
1657           qVar.getScope().undeclare(qVar);
1658         }
1659       }
1660 
1661       if (valueType == null) {
1662         if (parent.isExprResult()) {
1663           stubDeclarations.add(new StubDeclaration(
1664               n,
1665               t.getInput() != null && t.getInput().isExtern(),
1666               ownerName));
1667         }
1668 
1669         return;
1670       }
1671 
1672       boolean inferred = isQualifiedNameInferred(
1673           qName, n, info, rhsValue, valueType);
1674       if (!inferred) {
1675         ObjectType ownerType = getObjectSlot(ownerName);
1676         if (ownerType != null) {
1677           // Only declare this as an official property if it has not been
1678           // declared yet.
1679           boolean isExtern = t.getInput() != null && t.getInput().isExtern();
1680           if ((!ownerType.hasOwnProperty(propName) ||
1681                ownerType.isPropertyTypeInferred(propName)) &&
1682               ((isExtern && !ownerType.isNativeObjectType()) ||
1683                !ownerType.isInstanceType())) {
1684             // If the property is undeclared or inferred, declare it now.
1685             ownerType.defineDeclaredProperty(propName, valueType, n);
1686           }
1687         }
1688 
1689         // If the property is already declared, the error will be
1690         // caught when we try to declare it in the current scope.
1691         defineSlot(n, parent, valueType, inferred);
1692       } else if (rhsValue != null && rhsValue.isTrue()) {
1693         // We declare these for delegate proxy method properties.
1694         ObjectType ownerType = getObjectSlot(ownerName);
1695         FunctionType ownerFnType = JSType.toMaybeFunctionType(ownerType);
1696         if (ownerFnType != null) {
1697           JSType ownerTypeOfThis = ownerFnType.getTypeOfThis();
1698           String delegateName = codingConvention.getDelegateSuperclassName();
1699           JSType delegateType = delegateName == null ?
1700               null : typeRegistry.getType(delegateName);
1701           if (delegateType != null &&
1702               ownerTypeOfThis.isSubtype(delegateType)) {
1703             defineSlot(n, parent, getNativeType(BOOLEAN_TYPE), true);
1704           }
1705         }
1706       }
1707     }

1709     /**
1710      * Determines whether a qualified name is inferred.
1711      * NOTE(nicksantos): Determining whether a property is declared or not
1712      * is really really obnoxious.
1713      *
1714      * The problem is that there are two (equally valid) coding styles:
1715      *
1716      * (function() {
1717      *   /* The authoritative definition of goog.bar. /
1718      *   goog.bar = function() {};
1719      * })();
1720      *
1721      * function f() {
1722      *   goog.bar();
1723      *   /* Reset goog.bar to a no-op. /
1724      *   goog.bar = function() {};
1725      * }
1726      *
1727      * In a dynamic language with first-class functions, it's very difficult
1728      * to know which one the user intended without looking at lots of
1729      * contextual information (the second example demonstrates a small case
1730      * of this, but there are some really pathological cases as well).
1731      *
1732      * The current algorithm checks if either the declaration has
1733      * JsDoc type information, or @const with a known type,
1734      * or a function literal with a name we haven't seen before.
1735      */
1736     private boolean isQualifiedNameInferred(
1737         String qName, Node n, JSDocInfo info,
1738         Node rhsValue, JSType valueType) {
1739       if (valueType == null) {
1740         return true;
1741       }
1742 
1743       // Prototypes of constructors and interfaces are always declared.
1744       if (qName != null && qName.endsWith(".prototype")) {
1745         String className = qName.substring(0, qName.lastIndexOf(".prototype"));
1746         TypedVar slot = scope.getSlot(className);
1747         JSType classType = slot == null ? null : slot.getType();
1748         if (classType != null
1749             && (classType.isConstructor() || classType.isInterface())) {
1750           return false;
1751         }
1752       }
1753 
1754       boolean inferred = true;
1755       if (info != null) {
1756         inferred = !(info.hasType()
1757             || info.hasEnumParameterType()
1758             || (NodeUtil.isConstantDeclaration(
1759                     compiler.getCodingConvention(), info, n)
1760                 && valueType != null
1761                 && !valueType.isUnknownType())
1762             || FunctionTypeBuilder.isFunctionTypeDeclaration(info));
1763       }
1764 
1765       if (inferred && rhsValue != null && rhsValue.isFunction()) {
1766         if (info != null) {
1767           return false;
1768         } else if (!scope.isDeclared(qName, false) &&
1769             n.isUnscopedQualifiedName()) {
1770 
1771           // Check if this is in a conditional block.
1772           // Functions assigned in conditional blocks are inferred.
1773           for (Node current = n.getParent();
1774                !(current.isScript() || current.isFunction());
1775                current = current.getParent()) {
1776             if (NodeUtil.isControlStructure(current)) {
1777               return true;
1778             }
1779           }
1780 
1781           // Check if this is assigned in an inner scope.
1782           // Functions assigned in inner scopes are inferred.
1783           AstFunctionContents contents =
1784               getFunctionAnalysisResults(scope.getRootNode());
1785           if (contents == null ||
1786               !contents.getEscapedQualifiedNames().contains(qName)) {
1787             return false;
1788           }
1789         }
1790       }
1791       return inferred;
1792     }

1900     /**
1901      * Handle typedefs.
1902      * @param candidate A qualified name node.
1903      * @param info JSDoc comments.
1904      */
1905     private void checkForTypedef(Node candidate, JSDocInfo info) {
1906       if (info == null || !info.hasTypedefType()) {
1907         return;
1908       }
1909 
1910       String typedef = candidate.getQualifiedName();
1911       if (typedef == null) {
1912         return;
1913       }
1914 
1915       // TODO(nicksantos|user): This is a terrible, terrible hack
1916       // to bail out on recursive typedefs. We'll eventually need
1917       // to handle these properly.
1918       typeRegistry.declareType(typedef, unknownType);

1920       JSType realType = info.getTypedefType().evaluate(scope, typeRegistry);
1921       if (realType == null) {
1922         compiler.report(
1923             JSError.make(candidate, MALFORMED_TYPEDEF, typedef));
1924       }
1925 
1926       typeRegistry.overwriteDeclaredType(typedef, realType);
1927       if (candidate.isGetProp()) {
1928         defineSlot(candidate, candidate.getParent(),
1929             getNativeType(NO_TYPE), false);
1930       }
1931     }

1934   /**
1935    * A shallow traversal of a local scope to find all arguments and
1936    * local variables.
1937    */
1938   private final class LocalScopeBuilder extends AbstractScopeBuilder {
1939     private final ObjectType thisTypeForProperties;
1940 
1941     /**
1942      * @param scope The scope that we're building.
1943      */
1944     private LocalScopeBuilder(TypedScope scope) {
1945       super(scope);
1946       thisTypeForProperties = getThisTypeForCollectingProperties();
1947     }
1948 
1949     /**
1950      * Traverse the scope root and build it.
1951      */
1952     void build() {
1953       NodeTraversal.traverseTyped(compiler, scope.getRootNode(), this);
1954 
1955       AstFunctionContents contents =
1956           getFunctionAnalysisResults(scope.getRootNode());
1957       if (contents != null) {
1958         for (String varName : contents.getEscapedVarNames()) {
1959           TypedVar v = scope.getVar(varName);
1960           Preconditions.checkState(v.getScope() == scope);
1961           v.markEscaped();
1962         }
1963 
1964         for (Multiset.Entry<String> entry :
1965                  contents.getAssignedNameCounts().entrySet()) {
1966           TypedVar v = scope.getVar(entry.getElement());
1967           Preconditions.checkState(v.getScope() == scope);
1968           if (entry.getCount() == 1) {
1969             v.markAssignedExactlyOnce();
1970           }
1971         }
1972       }
1973     }
1974 
1975     /**
1976      * Visit a node in a local scope, and add any local variables or catch
1977      * parameters into the local symbol table.
1978      *
1979      * @param t The node traversal.
1980      * @param n The node being visited.
1981      * @param parent The parent of n
1982      */
1983     @Override public void visit(NodeTraversal t, Node n, Node parent) {
1984       if (n == scope.getRootNode()) {
1985         return;
1986       }
1987 
1988       if (n.isParamList() && parent == scope.getRootNode()) {
1989         handleFunctionInputs(parent);
1990         return;
1991       }
1992 
1993       // Gather the properties declared in the function,
1994       // if that function has a @this type that we can
1995       // build properties on.
1996       // TODO(nick): It's not clear to me why this is necessary;
1997       // it appears to be papering over bugs in the main analyzer.
1998       if (thisTypeForProperties != null && n.getParent().isExprResult()) {
1999         if (n.isAssign()) {
2000           maybeCollectMember(n.getFirstChild(), n, n.getLastChild());
2001         } else if (n.isGetProp()) {
2002           maybeCollectMember(n, n, null);
2003         }
2004       }
2005 
2006       super.visit(t, n, parent);
2007     }
2008 
2009     private ObjectType getThisTypeForCollectingProperties() {
2010       Node rootNode = scope.getRootNode();
2011       if (rootNode.isFromExterns()) return null;
2012 
2013       JSType type = rootNode.getJSType();
2014       if (type == null || !type.isFunctionType()) return null;
2015 
2016       FunctionType fnType = type.toMaybeFunctionType();
2017       JSType fnThisType = fnType.getTypeOfThis();
2018       return fnThisType.isUnknownType() ? null : fnThisType.toObjectType();
2019     }

2021     private void maybeCollectMember(Node member,
2022         Node nodeWithJsDocInfo, @Nullable Node value) {
2023       JSDocInfo info = nodeWithJsDocInfo.getJSDocInfo();
2024 
2025       // Do nothing if there is no JSDoc type info, or
2026       // if the node is not a member expression, or
2027       // if the member expression is not of the form: this.someProperty.
2028       if (info == null ||
2029           !member.isGetProp() ||
2030           !member.getFirstChild().isThis()) {
2031         return;
2032       }
2033 
2034       JSType jsType = getDeclaredType(info, member, value);
2035       Node name = member.getLastChild();
2036       if (jsType != null) {
2037         thisTypeForProperties.defineDeclaredProperty(
2038             name.getString(),
2039             jsType,
2040             member);
2041       }
2042     }

2044     /** Handle bleeding functions and function parameters. */
2045     private void handleFunctionInputs(Node fnNode) {
2046       // Handle bleeding functions.
2047       Node fnNameNode = fnNode.getFirstChild();
2048       String fnName = fnNameNode.getString();
2049       if (!fnName.isEmpty()) {
2050         TypedVar fnVar = scope.getVar(fnName);
2051         if (fnVar == null ||
2052             // Make sure we're not touching a native function. Native
2053             // functions aren't bleeding, but may not have a declaration
2054             // node.
2055             (fnVar.getNameNode() != null &&
2056                 // Make sure that the function is actually bleeding by checking
2057                 // if has already been declared.
2058                 fnVar.getInitialValue() != fnNode)) {
2059           defineSlot(fnNameNode, fnNode, fnNode.getJSType(), false);
2060         }
2061       }
2062 
2063       declareArguments(fnNode);
2064     }

2066     /**
2067      * Declares all of a function's arguments.
2068      */
2069     private void declareArguments(Node functionNode) {
2070       Node astParameters = functionNode.getFirstChild().getNext();
2071       Node iifeArgumentNode = null;
2072 
2073       if (NodeUtil.isCallOrNewTarget(functionNode)) {
2074         iifeArgumentNode = functionNode.getNext();
2075       }
2076 
2077       FunctionType functionType =
2078           JSType.toMaybeFunctionType(functionNode.getJSType());
2079       if (functionType != null) {
2080         Node jsDocParameters = functionType.getParametersNode();
2081         if (jsDocParameters != null) {
2082           Node jsDocParameter = jsDocParameters.getFirstChild();
2083           for (Node astParameter : astParameters.children()) {
2084             JSType paramType = jsDocParameter == null ?
2085                 unknownType : jsDocParameter.getJSType();
2086             boolean inferred = paramType == null || paramType == unknownType;
2087 
2088             if (iifeArgumentNode != null && inferred) {
2089               String argumentName = iifeArgumentNode.getQualifiedName();
2090               TypedVar argumentVar =
2091                   argumentName == null || scope.getParent() == null
2092                   ? null : scope.getParent().getVar(argumentName);
2093               if (argumentVar != null && !argumentVar.isTypeInferred()) {
2094                 paramType = argumentVar.getType();
2095               }
2096             }
2097 
2098             if (paramType == null) {
2099               paramType = unknownType;
2100             }
2101 
2102             defineSlot(astParameter, functionNode, paramType, inferred);
2103 
2104             if (jsDocParameter != null) {
2105               jsDocParameter = jsDocParameter.getNext();
2106             }
2107             if (iifeArgumentNode != null) {
2108               iifeArgumentNode = iifeArgumentNode.getNext();
2109             }
2110           }
2111         }
2112       }
2113     } // end declareArguments

2117    * Does a first-order function analysis that just looks at simple things
2118    * like what variables are escaped, and whether 'this' is used.
2119    */
2120   private static class FirstOrderFunctionAnalyzer
2121       extends AbstractScopedCallback implements CompilerPass {
2122     private final AbstractCompiler compiler;
2123     private final Map<Node, AstFunctionContents> data;
2124 
2125     FirstOrderFunctionAnalyzer(
2126         AbstractCompiler compiler, Map<Node, AstFunctionContents> outParam) {
2127       this.compiler = compiler;
2128       this.data = outParam;
2129     }
2130 
2131     @Override public void process(Node externs, Node root) {
2132       if (externs == null) {
2133         NodeTraversal.traverseTyped(compiler, root, this);
2134       } else {
2135         NodeTraversal.traverseRootsTyped(compiler, this, externs, root);
2136       }
2137     }
2138 
2139     @Override public void enterScope(NodeTraversal t) {
2140       if (!t.inGlobalScope()) {
2141         Node n = t.getScopeRoot();
2142         data.put(n, new AstFunctionContents(n));
2143       }
2144     }
2145 
2146     @Override public void visit(NodeTraversal t, Node n, Node parent) {
2147       if (t.inGlobalScope()) {
2148         // The first-order function analyzer looks at two types of variables:
2149         //
2150         // 1) Local variables that are assigned in inner scopes ("escaped vars")
2151         //
2152         // 2) Local variables that are assigned more than once.
2153         //
2154         // We treat all global variables as escaped by default, so there's
2155         // no reason to do this extra computation for them.
2156         return;
2157       }
2158 
2159       if (n.isReturn() && n.getFirstChild() != null) {
2160         data.get(t.getScopeRoot()).recordNonEmptyReturn();
2161       }
2162 
2163       if (n.isName() && NodeUtil.isLValue(n) &&
2164           // Be careful of bleeding functions, which create variables
2165           // in the inner scope, not the scope where the name appears.
2166           !NodeUtil.isBleedingFunctionName(n)) {
2167         String name = n.getString();
2168         TypedScope scope = t.getTypedScope();
2169         TypedVar var = scope.getVar(name);
2170         if (var != null) {
2171           TypedScope ownerScope = var.getScope();
2172           if (ownerScope.isLocal()) {
2173             data.get(ownerScope.getRootNode()).recordAssignedName(name);
2174           }
2175 
2176           if (scope != ownerScope && ownerScope.isLocal()) {
2177             data.get(ownerScope.getRootNode()).recordEscapedVarName(name);
2178           }
2179         }
2180       } else if (n.isGetProp() && n.isUnscopedQualifiedName() &&
2181           NodeUtil.isLValue(n)) {
2182         String name = NodeUtil.getRootOfQualifiedName(n).getString();
2183         TypedScope scope = t.getTypedScope();
2184         TypedVar var = scope.getVar(name);
2185         if (var != null) {
2186           TypedScope ownerScope = var.getScope();
2187           if (scope != ownerScope && ownerScope.isLocal()) {
2188             data.get(ownerScope.getRootNode())
2189                 .recordEscapedQualifiedName(n.getQualifiedName());
2190           }
2191         }
2192       }
2193     }
2194   }
