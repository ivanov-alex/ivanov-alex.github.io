38  /**
39   * An abstract compiler, to help remove the circular dependency of
40   * passes on JSCompiler.
41   *
42   * This is an abstract class, so that we can make the methods package-private.
43   *
44   * @author nicksantos@google.com (Nick Santos)
45   */
46  public abstract class AbstractCompiler implements SourceExcerptProvider {

341   static enum LifeCycleStage {
342     RAW,
343 
344     // See constraints put on the tree by Normalize.java
345     NORMALIZED,
346 
347     // The normalize pass has put constraints on the tree,
348     // but variables and properties have been renamed so
349     // coding conventions no longer apply.
350     NORMALIZED_OBFUSCATED;
351 
352     boolean isNormalized() {
353       return this == NORMALIZED || this == NORMALIZED_OBFUSCATED;
354     }
355 
356     boolean isNormalizedUnobfuscated() {
357       return this == NORMALIZED;
358     }
359 
360     boolean isNormalizedObfuscated() {
361       return this == NORMALIZED_OBFUSCATED;
362     }
363   }
