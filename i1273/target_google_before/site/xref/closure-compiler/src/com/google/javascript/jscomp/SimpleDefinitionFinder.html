36  /**
37   * Simple name-based definition gatherer that implements
38   * {@link DefinitionProvider}.
39   *
40   * It treats all variable writes as happening in the global scope and
41   * treats all objects as capable of having the same set of properties.
42   * The current implementation only handles definitions whose right
43   * hand side is an immutable value or function expression.  All
44   * complex definitions are treated as unknowns.
45   *
46   */
47  class SimpleDefinitionFinder implements CompilerPass, DefinitionProvider {
48    private final AbstractCompiler compiler;
49    private final Map<Node, DefinitionSite> definitionSiteMap;
50    private final Multimap<String, Definition> nameDefinitionMultimap;
51    private final Multimap<String, UseSite> nameUseSiteMultimap;
52  
53    public SimpleDefinitionFinder(AbstractCompiler compiler) {
54      this.compiler = compiler;
55      this.definitionSiteMap = new LinkedHashMap<>();
56      this.nameDefinitionMultimap = LinkedHashMultimap.create();
57      this.nameUseSiteMultimap = LinkedHashMultimap.create();
58    }
59  
60    /**
61     * Returns the collection of definition sites found during traversal.
62     *
63     * @return definition site collection.
64     */
65    public Collection<DefinitionSite> getDefinitionSites() {
66      return definitionSiteMap.values();
67    }
68  
69    private DefinitionSite getDefinitionAt(Node node) {
70      return definitionSiteMap.get(node);
71    }
72  
73    DefinitionSite getDefinitionForFunction(Node function) {
74      Preconditions.checkState(function.isFunction());
75      return getDefinitionAt(getNameNodeFromFunctionNode(function));
76    }
77  
78    @Override
79    public Collection<Definition> getDefinitionsReferencedAt(Node useSite) {
80      if (definitionSiteMap.containsKey(useSite)) {
81        return null;
82      }
83  
84      if (useSite.isGetProp()) {
85        String propName = useSite.getLastChild().getString();
86        if (propName.equals("apply") || propName.equals("call")) {
87          useSite = useSite.getFirstChild();
88        }
89      }
90  
91      String name = getSimplifiedName(useSite);
92      if (name != null) {
93        Collection<Definition> defs = nameDefinitionMultimap.get(name);
94        if (!defs.isEmpty()) {
95          return defs;
96        } else {
97          return null;
98        }
99      } else {
100       return null;
101     }
102   }

104   @Override
105   public void process(Node externs, Node source) {
106     NodeTraversal.traverseEs6(
107         compiler, externs, new DefinitionGatheringCallback(true));
108     NodeTraversal.traverseEs6(
109         compiler, source, new DefinitionGatheringCallback(false));
110     NodeTraversal.traverseEs6(
111         compiler, source, new UseSiteGatheringCallback());
112   }

114   /**
115    * Returns a collection of use sites that may refer to provided
116    * definition.  Returns an empty collection if the definition is not
117    * used anywhere.
118    *
119    * @param definition Definition of interest.
120    * @return use site collection.
121    */
122   Collection<UseSite> getUseSites(Definition definition) {
123     String name = getSimplifiedName(definition.getLValue());
124     return nameUseSiteMultimap.get(name);
125   }

127   /**
128    * Extract a name from a node.  In the case of GETPROP nodes,
129    * replace the namespace or object expression with "this" for
130    * simplicity and correctness at the expense of inefficiencies due
131    * to higher chances of name collisions.
132    *
133    * TODO(user) revisit.  it would be helpful to at least use fully
134    * qualified names in the case of namespaces.  Might not matter as
135    * much if this pass runs after "collapsing properties".
136    */
137   private static String getSimplifiedName(Node node) {
138     if (node.isName()) {
139       String name = node.getString();
140       if (name != null && !name.isEmpty()) {
141         return name;
142       } else {
143         return null;
144       }
145     } else if (node.isGetProp()) {
146       return "this." + node.getLastChild().getString();
147     }
148     return null;
149   }

151   private class DefinitionGatheringCallback implements Callback {
152     private boolean inExterns;
153 
154     DefinitionGatheringCallback(boolean inExterns) {
155       this.inExterns = inExterns;
156     }
157 
158     @Override
159     public boolean shouldTraverse(NodeTraversal t, Node n, Node parent) {
160       if (inExterns) {
161         if (n.isFunction() && !n.getFirstChild().isName()) {
162           // No need to crawl functions in JSDoc
163           return false;
164         }
165         if (parent != null
166             && parent.isFunction() && n != parent.getFirstChild()) {
167           // Arguments of external functions should not count as name
168           // definitions.  They are placeholder names for documentation
169           // purposes only which are not reachable from anywhere.
170           return false;
171         }
172       }
173       return true;
174     }
175 
176 
177     @Override
178     public void visit(NodeTraversal traversal, Node node, Node parent) {
179       if (inExterns && node.getJSDocInfo() != null) {
180         for (Node typeRoot : node.getJSDocInfo().getTypeNodes()) {
181           traversal.traverse(typeRoot);
182         }
183       }
184 
185       Definition def =
186           DefinitionsRemover.getDefinition(node, inExterns);
187       if (def != null) {
188         String name = getSimplifiedName(def.getLValue());
189         if (name != null) {
190           Node rValue = def.getRValue();
191           if ((rValue != null) &&
192               !NodeUtil.isImmutableValue(rValue) &&
193               !rValue.isFunction()) {
194 
195             // Unhandled complex expression
196             Definition unknownDef =
197                 new UnknownDefinition(def.getLValue(), inExterns);
198             def = unknownDef;
199           }
200 
201           // TODO(johnlenz) : remove this stub dropping code if it becomes
202           // illegal to have untyped stubs in the externs definitions.
203           if (inExterns) {
204             // We need special handling of untyped externs stubs here:
205             //    the stub should be dropped if the name is provided elsewhere.
206 
207             List<Definition> stubsToRemove = new ArrayList<>();
208 
209             // If there is no qualified name for this, then there will be
210             // no stubs to remove. This will happen if node is an object
211             // literal key.
212             if (node.isQualifiedName()) {
213               for (Definition prevDef : nameDefinitionMultimap.get(name)) {
214                 if (prevDef instanceof ExternalNameOnlyDefinition
215                     && !jsdocContainsDeclarations(node)) {
216                   if (node.matchesQualifiedName(prevDef.getLValue())) {
217                     // Drop this stub, there is a real definition.
218                     stubsToRemove.add(prevDef);
219                   }
220                 }
221               }
222 
223               for (Definition prevDef : stubsToRemove) {
224                 nameDefinitionMultimap.remove(name, prevDef);
225               }
226             }
227           }
228 
229           nameDefinitionMultimap.put(name, def);
230           definitionSiteMap.put(node,
231                                 new DefinitionSite(node,
232                                                    def,
233                                                    traversal.getModule(),
234                                                    traversal.inGlobalScope(),
235                                                    inExterns));
236         }
237       }
238 
239       if (inExterns && (parent != null) && parent.isExprResult()) {
240         String name = getSimplifiedName(node);
241         if (name != null) {
242 
243           // TODO(johnlenz) : remove this code if it becomes illegal to have
244           // stubs in the externs definitions.
245 
246           // We need special handling of untyped externs stubs here:
247           //    the stub should be dropped if the name is provided elsewhere.
248           // We can't just drop the stub now as it needs to be used as the
249           //    externs definition if no other definition is provided.
250 
251           boolean dropStub = false;
252           if (!jsdocContainsDeclarations(node) && node.isQualifiedName()) {
253             for (Definition prevDef : nameDefinitionMultimap.get(name)) {
254               if (node.matchesQualifiedName(prevDef.getLValue())) {
255                 dropStub = true;
256                 break;
257               }
258             }
259           }
260 
261           if (!dropStub) {
262             // Incomplete definition
263             Definition definition = new ExternalNameOnlyDefinition(node);
264             nameDefinitionMultimap.put(name, definition);
265             definitionSiteMap.put(node,
266                                   new DefinitionSite(node,
267                                                      definition,
268                                                      traversal.getModule(),
269                                                      traversal.inGlobalScope(),
270                                                      inExterns));
271           }
272         }
273       }
274     }

276     /**
277      * @return Whether the node has a JSDoc that actually declares something.
278      */
279     private boolean jsdocContainsDeclarations(Node node) {
280       JSDocInfo info = node.getJSDocInfo();
281       return (info != null && info.containsDeclaration());
282     }

285   private class UseSiteGatheringCallback extends AbstractPostOrderCallback {
286     @Override
287     public void visit(NodeTraversal traversal, Node node, Node parent) {
288 
289       Collection<Definition> defs = getDefinitionsReferencedAt(node);
290       if (defs == null) {
291         return;
292       }
293 
294       Definition first = defs.iterator().next();
295 
296       String name = getSimplifiedName(first.getLValue());
297       Preconditions.checkNotNull(name);
298       nameUseSiteMultimap.put(
299           name,
300           new UseSite(node, traversal.getScope(), traversal.getModule()));
301     }
302   }

304   /**
305    * @param use A use site to check.
306    * @return Whether the use is a call or new.
307    */
308   static boolean isCallOrNewSite(UseSite use) {
309     Node call = use.node.getParent();
310     if (call == null) {
311       // The node has been removed from the AST.
312       return false;
313     }
314     // We need to make sure we're dealing with a call to the function we're
315     // optimizing. If the the first child of the parent is not the site, this
316     // is a nested call and it's a call to another function.
317     return NodeUtil.isCallOrNew(call) && call.getFirstChild() == use.node;
318   }

320   boolean canModifyDefinition(Definition definition) {
321     if (isExported(definition)) {
322       return false;
323     }
324 
325     // Don't modify unused definitions for two reasons:
326     // 1) It causes unnecessary churn
327     // 2) Other definitions might be used to reflect on this one using
328     //    goog.reflect.object (the check for definitions with uses is below).
329     Collection<UseSite> useSites = getUseSites(definition);
330     if (useSites.isEmpty()) {
331       return false;
332     }
333 
334     for (UseSite site : useSites) {
335       // This catches the case where an object literal in goog.reflect.object
336       // and a prototype method have the same property name.
337 
338       // NOTE(nicksantos): Maps and trogedit both do this by different
339       // mechanisms.
340 
341       Node nameNode = site.node;
342       Collection<Definition> singleSiteDefinitions =
343           getDefinitionsReferencedAt(nameNode);
344       if (singleSiteDefinitions.size() > 1) {
345         return false;
346       }
347 
348       Preconditions.checkState(!singleSiteDefinitions.isEmpty());
349       Preconditions.checkState(singleSiteDefinitions.contains(definition));
350     }
351 
352     return true;
353   }

355   /**
356    * @return Whether the definition is directly exported.
357    */
358   private boolean isExported(Definition definition) {
359     // Assume an exported method result is used.
360     Node lValue = definition.getLValue();
361     if (lValue == null) {
362       return true;
363     }
364 
365     String partialName;
366     if (lValue.isGetProp()) {
367       partialName = lValue.getLastChild().getString();
368     } else if (lValue.isName()) {
369       partialName = lValue.getString();
370     } else {
371       // GETELEM is assumed to be an export or other expression are unknown
372       // uses.
373       return true;
374     }
375 
376     CodingConvention codingConvention = compiler.getCodingConvention();
377     return codingConvention.isExported(partialName);
378   }

380   /**
381    * @return Whether the function is defined in a non-aliasing expression.
382    */
383   static boolean isSimpleFunctionDeclaration(Node fn) {
384     Node parent = fn.getParent();
385     Node grandparent = parent.getParent();
386 
387     // Simple definition finder doesn't provide useful results in some
388     // cases, specifically:
389     //  - functions with recursive definitions
390     //  - functions defined in object literals
391     //  - functions defined in array literals
392     // Here we defined a set of known function declaration that are 'ok'.
393 
394     // Some projects seem to actually define "JSCompiler_renameProperty"
395     // rather than simply having an extern definition.  Don't mess with it.
396     Node nameNode = SimpleDefinitionFinder.getNameNodeFromFunctionNode(fn);
397     if (nameNode != null
398         && nameNode.isName()) {
399       String name = nameNode.getString();
400       if (name.equals(NodeUtil.JSC_PROPERTY_NAME_FN) ||
401              name.equals(
402                 ObjectPropertyStringPreprocess.EXTERN_OBJECT_PROPERTY_STRING)) {
403         return false;
404       }
405     }
406 
407     // example: function a(){};
408     if (NodeUtil.isFunctionDeclaration(fn)) {
409       return true;
410     }
411 
412     // example: a = function(){};
413     // example: var a = function(){};
414     return fn.getFirstChild().getString().isEmpty()
415         && (NodeUtil.isExprAssign(grandparent) || parent.isName());
416   }

418   /**
419    * @return the node defining the name for this function (if any).
420    */
421   static Node getNameNodeFromFunctionNode(Node function) {
422     Preconditions.checkState(function.isFunction());
423     if (NodeUtil.isFunctionDeclaration(function)) {
424       return function.getFirstChild();
425     } else {
426       Node parent = function.getParent();
427       if (NodeUtil.isVarDeclaration(parent)) {
428         return parent;
429       } else if (parent.isAssign()) {
430         return parent.getFirstChild();
431       } else if (NodeUtil.isObjectLitKey(parent)) {
432         return parent;
433       }
434     }
435     return null;
436   }

438   /**
439    * Traverse a node and its children and remove any references to from
440    * the structures.
441    */
442   void removeReferences(Node node) {
443     if (DefinitionsRemover.isDefinitionNode(node)) {
444       DefinitionSite defSite = definitionSiteMap.get(node);
445       if (defSite != null) {
446         Definition def = defSite.definition;
447         String name = getSimplifiedName(def.getLValue());
448         if (name != null) {
449           this.definitionSiteMap.remove(node);
450           this.nameDefinitionMultimap.remove(name, node);
451         }
452       }
453     } else {
454       Node useSite = node;
455       if (useSite.isGetProp()) {
456         String propName = useSite.getLastChild().getString();
457         if (propName.equals("apply") || propName.equals("call")) {
458           useSite = useSite.getFirstChild();
459         }
460       }
461       String name = getSimplifiedName(useSite);
462       if (name != null) {
463         this.nameUseSiteMultimap.remove(name, new UseSite(useSite, null, null));
464       }
465     }
466 
467     for (Node child : node.children()) {
468       removeReferences(child);
469     }
470   }
