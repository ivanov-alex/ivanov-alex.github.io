125   /**
126    * Set the language to which input sources conform.
127    * @param value The name of the language.
128    *     (ECMASCRIPT3, ECMASCRIPT5, ECMASCRIPT5_STRICT).
129    */
130   public void setLanguageIn(String value) {
131     switch (value) {
132       case "ECMASCRIPT6_STRICT":
133       case "ES6_STRICT":
134         this.languageIn = CompilerOptions.LanguageMode.ECMASCRIPT6_STRICT;
135         break;
136       case "ECMASCRIPT6":
137       case "ES6":
138         this.languageIn = CompilerOptions.LanguageMode.ECMASCRIPT6;
139         break;
140       case "ECMASCRIPT5_STRICT":
141       case "ES5_STRICT":
142         this.languageIn = CompilerOptions.LanguageMode.ECMASCRIPT5_STRICT;
143         break;
144       case "ECMASCRIPT5":
145       case "ES5":
146         this.languageIn = CompilerOptions.LanguageMode.ECMASCRIPT5;
147         break;
148       case "ECMASCRIPT3":
149       case "ES3":
150         this.languageIn = CompilerOptions.LanguageMode.ECMASCRIPT3;
151         break;
152       default:
153         throw new BuildException(
154             "Unrecognized 'languageIn' option value (" + value + ")");
155     }
156   }

356   @Override
357   public void execute() {
358     if (this.outputFile == null) {
359       throw new BuildException("outputFile attribute must be set");
360     }
361 
362     Compiler.setLoggingLevel(Level.OFF);
363 
364     CompilerOptions options = createCompilerOptions();
365     Compiler compiler = createCompiler(options);
366 
367     List<SourceFile> externs = findExternFiles(options);
368     List<SourceFile> sources = findSourceFiles();
369 
370     if (isStale() || forceRecompile) {
371       log("Compiling " + sources.size() + " file(s) with " +
372           externs.size() + " extern(s)");
373 
374       Result result = compiler.compile(externs, sources, options);
375 
376       if (result.success) {
377         StringBuilder source = new StringBuilder(compiler.toSource());
378 
379         if (this.outputWrapperFile != null) {
380           try {
381             this.outputWrapper = Files.toString(this.outputWrapperFile, UTF_8);
382           } catch (Exception e) {
383             throw new BuildException("Invalid output_wrapper_file specified.");
384           }
385         }
386 
387         if (this.outputWrapper != null) {
388           int pos = -1;
389           pos = this.outputWrapper.indexOf(CommandLineRunner.OUTPUT_MARKER);
390           if (pos > -1) {
391             String prefix = this.outputWrapper.substring(0, pos);
392             source.insert(0, prefix);
393 
394             // end of outputWrapper
395             int suffixStart = pos + CommandLineRunner.OUTPUT_MARKER.length();
396             String suffix = this.outputWrapper.substring(suffixStart);
397             source.append(suffix);
398           } else {
399             throw new BuildException("Invalid output_wrapper specified. " +
400                 "Missing '" + CommandLineRunner.OUTPUT_MARKER + "'.");
401           }
402         }
403 
404         if (result.sourceMap != null) {
405           flushSourceMap(result.sourceMap);
406           source.append(System.getProperty("line.separator"));
407           source.append("//# sourceMappingURL=" + sourceMapOutputFile.getName());
408         }
409         writeResult(source.toString());
410       } else {
411         throw new BuildException("Compilation failed.");
412       }
413     } else {
414       log("None of the files changed. Compilation skipped.");
415     }
416   }

418   private void flushSourceMap(SourceMap sourceMap) {
419     try {
420       FileWriter out = new FileWriter(sourceMapOutputFile);
421       sourceMap.appendTo(out, sourceMapOutputFile.getName());
422       out.close();
423     } catch (IOException e) {
424       throw new BuildException("Cannot write sourcemap to file.", e);
425     }
426   }

428   private CompilerOptions createCompilerOptions() {
429     CompilerOptions options = new CompilerOptions();
430 
431     this.compilationLevel.setOptionsForCompilationLevel(options);
432     if (this.debugOptions) {
433       this.compilationLevel.setDebugOptionsForCompilationLevel(options);
434     }
435 
436     options.setEnvironment(this.environment);
437 
438     options.setPrettyPrint(this.prettyPrint);
439     options.setPrintInputDelimiter(this.printInputDelimiter);
440     options.setPreferSingleQuotes(this.preferSingleQuotes);
441     options.setGenerateExports(this.generateExports);
442 
443     options.setLanguageIn(this.languageIn);
444     options.setOutputCharset(this.outputEncoding);
445 
446     this.warningLevel.setOptionsForWarningLevel(options);
447     options.setManageClosureDependencies(manageDependencies);
448     convertEntryPointParameters(options);
449     options.setTrustedStrings(true);
450     options.setAngularPass(angularPass);
451 
452     if (replaceProperties) {
453       convertPropertiesMap(options);
454     }
455 
456     convertDefineParameters(options);
457 
458     for (Warning warning : warnings) {
459       CheckLevel level = warning.getLevel();
460       String groupName = warning.getGroup();
461       DiagnosticGroup group = new DiagnosticGroups().forName(groupName);
462       if (group == null) {
463         throw new BuildException(
464             "Unrecognized 'warning' option value (" + groupName + ")");
465       }
466       options.setWarningLevel(group, level);
467     }
468 
469     if (!Strings.isNullOrEmpty(sourceMapFormat)) {
470       options.setSourceMapFormat(Format.valueOf(sourceMapFormat));
471     }
472 
473     if (!Strings.isNullOrEmpty(sourceMapLocationMapping)) {
474       String tokens[] = sourceMapLocationMapping.split("\\|", -1);
475       LocationMapping lm = new LocationMapping(tokens[0], tokens[1]);
476       options.setSourceMapLocationMappings(Arrays.asList(lm));
477     }
478 
479     if (sourceMapOutputFile != null) {
480       File parentFile = sourceMapOutputFile.getParentFile();
481       if (parentFile.mkdirs()) {
482         log("Created missing parent directory " + parentFile, Project.MSG_DEBUG);
483       }
484       options.setSourceMapOutputPath(parentFile.getAbsolutePath());
485     }
486     return options;
487   }

509   /**
510    * Converts {@code <define/>} nested elements into Compiler {@code @define}
511    * replacements. Note: unlike project properties, {@code <define/>} elements
512    * do not need to be named starting with the replacement prefix.
513    */
514   private void convertDefineParameters(CompilerOptions options) {
515     for (Parameter p : defineParams) {
516       String key = p.getName();
517       Object value = p.getValue();
518 
519       if (!setDefine(options, key, value)) {
520         log("Unexpected @define value for name=" + key + "; value=" + value);
521       }
522     }
523   }
524 
525   /**
526    * Converts {@code <entrypoint/>} nested elements into Compiler entrypoint
527    * replacements.
528    */
529   private void convertEntryPointParameters(CompilerOptions options) {
530     List<String> entryPoints = new LinkedList<>();
531     for (Parameter p : entryPointParams) {
532       String key = p.getName();
533       entryPoints.add(key);
534     }
535     if (this.manageDependencies) {
536       options.setManageClosureDependencies(entryPoints);
537     }
538   }

540   /**
541    * Converts project properties beginning with the replacement prefix
542    * into Compiler {@code @define} replacements.
543    *
544    * @param options
545    */
546   private void convertPropertiesMap(CompilerOptions options) {
547     @SuppressWarnings("unchecked")
548     Map<String, Object> props = getProject().getProperties();
549     for (Map.Entry<String, Object> entry : props.entrySet()) {
550       String key = entry.getKey();
551       Object value = entry.getValue();
552 
553       if (key.startsWith(replacePropertiesPrefix)) {
554         key = key.substring(replacePropertiesPrefix.length());
555 
556         if (!setDefine(options, key, value)) {
557           log("Unexpected property value for key=" + key + "; value=" + value);
558         }
559       }
560     }
561   }

563   /**
564    * Maps Ant-style values (e.g., from Properties) into expected
565    * Closure {@code @define} literals
566    *
567    * @return True if the {@code @define} replacement succeeded, false if
568    *         the variable's value could not be mapped properly.
569    */
570   private boolean setDefine(CompilerOptions options,
571       String key, Object value) {
572     boolean success = false;
573 
574     if (value instanceof String) {
575       final boolean isTrue = "true".equals(value);
576       final boolean isFalse = "false".equals(value);
577 
578       if (isTrue || isFalse) {
579         options.setDefineToBooleanLiteral(key, isTrue);
580       } else {
581         try {
582           double dblTemp = Double.parseDouble((String) value);
583           options.setDefineToDoubleLiteral(key, dblTemp);
584         } catch (NumberFormatException nfe) {
585           // Not a number, assume string
586           options.setDefineToStringLiteral(key, (String) value);
587         }
588       }
589 
590       success = true;
591     } else if (value instanceof Boolean) {
592       options.setDefineToBooleanLiteral(key, (Boolean) value);
593       success = true;
594     } else if (value instanceof Integer) {
595       options.setDefineToNumberLiteral(key, (Integer) value);
596       success = true;
597     } else if (value instanceof Double) {
598       options.setDefineToDoubleLiteral(key, (Double) value);
599       success = true;
600     }
601 
602     return success;
603   }

605   private Compiler createCompiler(CompilerOptions options) {
606     Compiler compiler = new Compiler();
607     MessageFormatter formatter =
608         options.getErrorFormat().toFormatter(compiler, false);
609     AntErrorManager errorManager = new AntErrorManager(formatter, this);
610     compiler.setErrorManager(errorManager);
611     return compiler;
612   }

614   private List<SourceFile> findExternFiles(CompilerOptions options) {
615     List<SourceFile> files = new LinkedList<>();
616     files.addAll(getBuiltinExterns(options));
617 
618     for (FileList list : this.externFileLists) {
619       files.addAll(findJavaScriptFiles(list));
620     }
621 
622     return files;
623   }
624 
625   private List<SourceFile> findSourceFiles() {
626     List<SourceFile> files = new LinkedList<>();
627 
628     for (FileList list : this.sourceFileLists) {
629       files.addAll(findJavaScriptFiles(list));
630     }
631 
632     for (Path list : this.sourcePaths) {
633       files.addAll(findJavaScriptFiles(list));
634     }
635 
636     return files;
637   }

639   /**
640    * Translates an Ant resource collection into the file list format that
641    * the compiler expects.
642    */
643   private List<SourceFile> findJavaScriptFiles(ResourceCollection rc) {
644     List<SourceFile> files = new LinkedList<>();
645     Iterator<Resource> iter = rc.iterator();
646     while (iter.hasNext()) {
647       FileResource fr = (FileResource) iter.next();
648       // Construct path to file, relative to current working directory.
649       File file = Paths.get("")
650           .toAbsolutePath()
651           .relativize(fr.getFile().toPath())
652           .toFile();
653       files.add(SourceFile.fromFile(file, Charset.forName(encoding)));
654     }
655     return files;
656   }

671   private void writeResult(String source) {
672     if (this.outputFile.getParentFile().mkdirs()) {
673       log("Created missing parent directory " +
674           this.outputFile.getParentFile(), Project.MSG_DEBUG);
675     }
676 
677     try {
678       OutputStreamWriter out = new OutputStreamWriter(
679           new FileOutputStream(this.outputFile), outputEncoding);
680       out.append(source);
681       out.flush();
682       out.close();
683     } catch (IOException e) {
684       throw new BuildException(e);
685     }
686 
687     log("Compiled JavaScript written to " + this.outputFile.getAbsolutePath(),
688         Project.MSG_DEBUG);
689   }

691   /**
692    * Determine if compilation must actually happen, i.e. if any input file
693    * (extern or source) has changed after the outputFile was last modified.
694    *
695    * @return true if compilation should happen
696    */
697   private boolean isStale() {
698     long lastRun = outputFile.lastModified();
699     long sourcesLastModified = Math.max(
700         getLastModifiedTime(this.sourceFileLists),
701         getLastModifiedTime(this.sourcePaths));
702     long externsLastModified = getLastModifiedTime(this.externFileLists);
703 
704     return lastRun <= sourcesLastModified || lastRun <= externsLastModified;
705   }

707   /**
708    * Returns the most recent modified timestamp of the file collection.
709    *
710    * Note: this must be combined into one method to account for both
711    * Path and FileList erasure types.
712    *
713    * @param fileLists Collection of FileList or Path
714    * @return Most recent modified timestamp
715    */
716   private long getLastModifiedTime(List<?> fileLists) {
717     long lastModified = 0;
718 
719     for (Object entry : fileLists) {
720       if (entry instanceof FileList) {
721         FileList list = (FileList) entry;
722 
723         for (String fileName : list.getFiles(this.getProject())) {
724           File path = list.getDir(this.getProject());
725           File file = new File(path, fileName);
726           lastModified = Math.max(getLastModifiedTime(file), lastModified);
727         }
728       } else if (entry instanceof Path) {
729         Path path = (Path) entry;
730         for (String src : path.list()) {
731           File file = new File(src);
732           lastModified = Math.max(getLastModifiedTime(file), lastModified);
733         }
734       }
735     }
736 
737     return lastModified;
738   }

740   /**
741    * Returns the last modified timestamp of the given File.
742    */
743   private long getLastModifiedTime(File file) {
744     long fileLastModified = file.lastModified();
745     // If the file is absent, we don't know if it changed (maybe was deleted),
746     // so assume it has just changed.
747     if (fileLastModified == 0) {
748       fileLastModified = new Date().getTime();
749     }
750     return fileLastModified;
751   }
