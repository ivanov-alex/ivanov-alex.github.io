40  package com.google.javascript.rhino.jstype;
41  
42  /**
43   * Represents different ways for comparing equality among types.
44   * @author nicksantos@google.com (Nick Santos)
45   */
46  enum EquivalenceMethod {
47    /**
48     * Indicates that the two types should behave exactly the same under
49     * all type operations.
50     *
51     * Thus, {string} != {?} and {Unresolved} != {?}
52     */
53    IDENTITY,
54  
55    /**
56     * Indicates that the two types are almost exactly the same, and that a
57     * data flow analysis algorithm comparing them should consider them equal.
58     *
59     * In traditional type inference, the types form a finite lattice, and this
60     * ensures that type inference will terminate.
61     *
62     * In our type system, the unknown types do not obey the lattice rules. So
63     * if we continue to perform inference over the unknown types, we may
64     * never terminate.
65     *
66     * By treating all unknown types as equivalent for the purposes of data
67     * flow analysis, we ensure that the algorithm will terminate.
68     *
69     * Thus, {string} != {?} and {Unresolved} ~= {?}
70     */
71    DATA_FLOW,
72  
73    /**
74     * Indicates that two types are invariant.
75     *
76     * In a type system without unknown types, this would be the same
77     * as IDENTITY. But we always want to consider type A invariant with type B
78     * if B is unknown.
79     *
80     * Thus, {string} ~= {?} and {Unresolved} ~= {?}
81     */
82    INVARIANT
83  }
