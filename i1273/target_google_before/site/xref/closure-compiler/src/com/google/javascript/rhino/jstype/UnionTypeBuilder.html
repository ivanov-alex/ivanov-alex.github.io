57  /**
58   * A builder for union types.
59   *
60   * @author nicksantos@google.com (Nick Santos)
61   */
62  public class UnionTypeBuilder implements Serializable {
63    private static final long serialVersionUID = 1L;
64  
65    // If the best we can do is say "this object is one of thirty things",
66    // then we should just give up and admit that we have no clue.
67    private static final int DEFAULT_MAX_UNION_SIZE = 30;
68  
69    private final JSTypeRegistry registry;
70    private final List<JSType> alternates = new ArrayList<>();
71    private boolean isAllType = false;
72    private boolean isNativeUnknownType = false;
73    private boolean areAllUnknownsChecked = true;
74    private final int maxUnionSize;
75  
76    // Every UnionType may have at most one structural function in it.
77    //
78    // NOTE(nicksantos): I've read some literature that says that type-inferenced
79    // languages are fundamentally incompatible with union types. I refuse
80    // to believe this. But they do make the type lattice much more complicated.
81    //
82    // For this reason, when we deal with function types, we actually merge some
83    // nodes on the lattice, and treat them as fundamentally equivalent.
84    // For example, we treat
85    // function(): string | function(): number
86    // as equivalent to
87    // function(): (string|number)
88    // and normalize the first type into the second type.
89    //
90    // To perform this normalization, we've modified UnionTypeBuilder to disallow
91    // multiple structural functions in a union. We always delegate to
92    // FunctionType::getLeastSupertype, which either merges the functions into
93    // one structural function, or just bails out and uses the top function type.
94    private int functionTypePosition = -1;
95  
96    // Memoize the result, in case build() is called multiple times.
97    private JSType result = null;
98  
99    public UnionTypeBuilder(JSTypeRegistry registry) {
100     this(registry, DEFAULT_MAX_UNION_SIZE);
101   }
102 
103   UnionTypeBuilder(JSTypeRegistry registry, int maxUnionSize) {
104     this.registry = registry;
105     this.maxUnionSize = maxUnionSize;
106   }
107 
108   Collection<JSType> getAlternates() {
109     JSType specialCaseType = reduceAlternatesWithoutUnion();
110     if (specialCaseType != null) {
111       return ImmutableList.of(specialCaseType);
112     }
113     return Collections.unmodifiableList(alternates);
114   }
115 
116   private boolean isSubtype(
117       JSType rightType, JSType leftType, boolean isStructural) {
118     // if thisType or thatType is an unresolved templatized type,
119     // then there is no structural interface matching
120     boolean thisUnresolved = rightType.isTemplatizedType()
121         && !rightType.toMaybeTemplatizedType().isResolved();
122     boolean thatUnresolved = leftType.isTemplatizedType()
123         && !leftType.toMaybeTemplatizedType().isResolved();
124     if (isStructural && !thisUnresolved && !thatUnresolved) {
125       return rightType.isSubtype(leftType);
126     } else {
127       return rightType.isSubtypeWithoutStructuralTyping(leftType);
128     }
129   }

131   /**
132    * Adds an alternate to the union type under construction. Returns this
133    * for easy chaining.
134    */
135   public UnionTypeBuilder addAlternate(JSType alternate, boolean isStructural) {
136     // build() returns the bottom type by default, so we can
137     // just bail out early here.
138     if (alternate.isNoType()) {
139       return this;
140     }
141 
142     isAllType = isAllType || alternate.isAllType();
143 
144     boolean isAlternateUnknown = alternate instanceof UnknownType;
145     isNativeUnknownType = isNativeUnknownType || isAlternateUnknown;
146     if (isAlternateUnknown) {
147       areAllUnknownsChecked = areAllUnknownsChecked &&
148           alternate.isCheckedUnknownType();
149     }
150     if (!isAllType && !isNativeUnknownType) {
151       if (alternate.isUnionType()) {
152         UnionType union = alternate.toMaybeUnionType();
153         for (JSType unionAlt : union.getAlternatesWithoutStructuralTyping()) {
154           addAlternate(unionAlt);
155         }
156       } else {
157         if (alternates.size() > maxUnionSize) {
158           return this;
159         }
160 
161         // Function types are special, because they have their
162         // own bizarre sub-lattice. See the comments on
163         // FunctionType#supAndInf helper and above at functionTypePosition.
164         if (alternate.isFunctionType() && functionTypePosition != -1) {
165           // See the comments on functionTypePosition above.
166           FunctionType other =
167               alternates.get(functionTypePosition).toMaybeFunctionType();
168           FunctionType supremum =
169               alternate.toMaybeFunctionType().supAndInfHelper(other, true);
170           alternates.set(functionTypePosition, supremum);
171           result = null;
172           return this;
173         }
174 
175         // Look through the alternates we've got so far,
176         // and check if any of them are duplicates of
177         // one another.
178         int currentIndex = 0;
179         Iterator<JSType> it = alternates.iterator();
180         while (it.hasNext()) {
181           boolean removeCurrent = false;
182           JSType current = it.next();
183 
184           // Unknown and NoResolved types may just be names that haven't
185           // been resolved yet. So keep these in the union, and just use
186           // equality checking for simple de-duping.
187           if (alternate.isUnknownType() ||
188               current.isUnknownType() ||
189               alternate.isNoResolvedType() ||
190               current.isNoResolvedType() ||
191               alternate.hasAnyTemplateTypes() ||
192               current.hasAnyTemplateTypes()) {
193             if (alternate.isEquivalentTo(current, isStructural)) {
194               // Alternate is unnecessary.
195               return this;
196             }
197           } else {
198 
199             // Because "Foo" and "Foo.<?>" are roughly equivalent
200             // templatized types, special care is needed when building the
201             // union. For example:
202             //   Object is consider a subtype of Object.<string>
203             // but we want to leave "Object" not "Object.<string>" when
204             // building the subtype.
205             //
206 
207             if (alternate.isTemplatizedType() || current.isTemplatizedType()) {
208               // Cases:
209               // 1) alternate:Array.<string> and current:Object ==> Object
210               // 2) alternate:Array.<string> and current:Array ==> Array
211               // 3) alternate:Object.<string> and
212               //    current:Array ==> Array|Object.<string>
213               // 4) alternate:Object and current:Array.<string> ==> Object
214               // 5) alternate:Array and current:Array.<string> ==> Array
215               // 6) alternate:Array and
216               //    current:Object.<string> ==> Array|Object.<string>
217               // 7) alternate:Array.<string> and
218               //    current:Array.<number> ==> Array.<?>
219               // 8) alternate:Array.<string> and
220               //    current:Array.<string> ==> Array.<string>
221               // 9) alternate:Array.<string> and
222               //    current:Object.<string> ==> Object.<string>|Array.<string>
223 
224               if (!current.isTemplatizedType()) {
225                 if (isSubtype(alternate, current, isStructural)) {
226                   // case 1, 2
227                   return this;
228                 }
229                 // case 3: leave current, add alternate
230               } else if (!alternate.isTemplatizedType()) {
231                 if (isSubtype(current, alternate, isStructural)) {
232                   // case 4, 5
233                   removeCurrent = true;
234                 }
235                 // case 6: leave current, add alternate
236               } else {
237                 Preconditions.checkState(current.isTemplatizedType()
238                     && alternate.isTemplatizedType());
239                 TemplatizedType templatizedAlternate = alternate.toMaybeTemplatizedType();
240                 TemplatizedType templatizedCurrent = current.toMaybeTemplatizedType();
241 
242                 if (templatizedCurrent.wrapsSameRawType(templatizedAlternate)) {
243                   if (alternate.getTemplateTypeMap().checkEquivalenceHelper(
244                       current.getTemplateTypeMap(),
245                       EquivalenceMethod.IDENTITY)) {
246                     // case 8
247                     return this;
248                   } else {
249                     // TODO(johnlenz): should we leave both types?
250                     // case 7: add a merged alternate
251                     // We currently merge to the templatized types to "unknown"
252                     // which is equivalent to the raw type.
253                     JSType merged = templatizedCurrent
254                         .getReferencedObjTypeInternal();
255                     return addAlternate(merged);
256                   }
257                 }
258                 // case 9: leave current, add alternate
259               }
260               // Otherwise leave both templatized types.
261             } else if (isSubtype(alternate, current, isStructural)) {
262               // Alternate is unnecessary.
263               return this;
264             } else if (isSubtype(current, alternate, isStructural)) {
265               // Alternate makes current obsolete
266               removeCurrent = true;
267             }
268           }
269 
270           if (removeCurrent) {
271             it.remove();
272 
273             if (currentIndex == functionTypePosition) {
274               functionTypePosition = -1;
275             } else if (currentIndex < functionTypePosition) {
276               functionTypePosition--;
277               currentIndex--;
278             }
279           }
280           currentIndex++;
281         }
282 
283         if (alternate.isFunctionType()) {
284           // See the comments on functionTypePosition above.
285           Preconditions.checkState(functionTypePosition == -1);
286           functionTypePosition = alternates.size();
287         }
288 
289         alternates.add(alternate);
290         result = null; // invalidate the memoized result
291       }
292     } else {
293       result = null;
294     }
295     return this;
296   }
297 
298   /**
299    * Adds an alternate to the union type under construction. Returns this
300    * for easy chaining.
301    */
302   public UnionTypeBuilder addAlternate(JSType alternate) {
303     return addAlternate(alternate, false);
304   }

306   /**
307    * Reduce the alternates into a non-union type.
308    * If the alternates can't be accurately represented with a non-union
309    * type, return null.
310    */
311   private JSType reduceAlternatesWithoutUnion() {
312     if (isAllType) {
313       return registry.getNativeType(ALL_TYPE);
314     } else if (isNativeUnknownType) {
315       if (areAllUnknownsChecked) {
316         return registry.getNativeType(CHECKED_UNKNOWN_TYPE);
317       } else {
318         return registry.getNativeType(UNKNOWN_TYPE);
319       }
320     } else {
321       int size = alternates.size();
322       if (size > maxUnionSize) {
323         return registry.getNativeType(UNKNOWN_TYPE);
324       } else if (size > 1) {
325         return null;
326       } else if (size == 1) {
327         return alternates.get(0);
328       } else {
329         return registry.getNativeType(NO_TYPE);
330       }
331     }
332   }
333 
334   /**
335    * Creates a union.
336    * @return A UnionType if it has two or more alternates, the
337    *    only alternate if it has one and otherwise {@code NO_TYPE}.
338    */
339   public JSType build() {
340     if (result == null) {
341       result = reduceAlternatesWithoutUnion();
342       if (result == null) {
343         result = new UnionType(registry, getAlternateListCopy());
344       }
345     }
346     return result;
347   }
348 
349   private Collection<JSType> getAlternateListCopy() {
350     return ImmutableList.copyOf(alternates);
351   }
