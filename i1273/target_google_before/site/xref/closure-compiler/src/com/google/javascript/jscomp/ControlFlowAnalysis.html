36  /**
37   * This is a compiler pass that computes a control flow graph.
38   *
39   */
40  final class ControlFlowAnalysis implements Callback, CompilerPass {
41  
42    /**
43     * Based roughly on the first few pages of
44     *
45     * "Declarative Intraprocedural Flow Analysis of Java Source Code by
46     * Nilsson-Nyman, Hedin, Magnusson & Ekman",
47     *
48     * this pass computes the control flow graph from the AST. However, a full
49     * attribute grammar is not necessary. We will compute the flow edges with a
50     * single post order traversal. The "follow()" of a given node will be
51     * computed recursively in a demand driven fashion.
52     *
53     * As of this moment, we are not performing any inter-procedural analysis
54     * within our framework.
55     */
56

64    private Map<DiGraphNode<Node, Branch>, Integer> nodePriorities;

66    // We order CFG nodes by by looking at the AST positions.
67    // CFG nodes that come first lexically should be visited first, because
68    // they will often be executed first in the source program.
69    private final Comparator<DiGraphNode<Node, Branch>> priorityComparator =
70        new Comparator<DiGraphNode<Node, Branch>>() {
71      @Override
72      public int compare(
73          DiGraphNode<Node, Branch> a, DiGraphNode<Node, Branch> b) {
74        return astPosition.get(a.getValue()) - astPosition.get(b.getValue());
75      }
76    };

89    /*
90     * This stack captures the structure of nested TRY blocks. The top of the
91     * stack is the inner most TRY block. A FUNCTION node in this stack implies
92     * that the handler is determined by the caller of the function at runtime.
93     */
94    private final Deque<Node> exceptionHandler = new ArrayDeque<>();

96    /*
97     * This map is used to handle the follow of FINALLY. For example:
98     *
99     * while(x) {
100    *  try {
101    *    try {
102    *      break;
103    *    } catch (a) {
104    *    } finally {
105    *      foo();
106    *    }
107    *    fooFollow();
108    *  } catch (b) {
109    *  } finally {
110    *    bar();
111    *  }
112    *  barFollow();
113    * }
114    * END();
115    *
116    * In this case finallyMap will contain a map from:
117    *    first FINALLY -> bar()
118    *    second FINALLY -> END()
119    *
120    * When we are connecting foo() and bar() to to their respective follow, we
121    * must also look up this map and connect:
122    *   foo() -> bar()
123    *   bar() -> END
124    */
125   private final Multimap<Node, Node> finallyMap = HashMultimap.create();

145   @Override
146   public void process(Node externs, Node root) {
147     Preconditions.checkArgument(
148         NodeUtil.isValidCfgRoot(root), "Unexpected control flow graph root %s", root);
149     this.root = root;
150     astPositionCounter = 0;
151     astPosition = new HashMap<>();
152     nodePriorities = new HashMap<>();
153     cfg = new AstControlFlowGraph(computeFallThrough(root), nodePriorities,
154                                   edgeAnnotations);
155     NodeTraversal.traverseEs6(compiler, root, this);
156     astPosition.put(null, ++astPositionCounter); // the implicit return is last.
157 
158     // Now, generate the priority of nodes by doing a depth-first
159     // search on the CFG.
160     priorityCounter = 0;
161     DiGraphNode<Node, Branch> entry = cfg.getEntry();
162     prioritizeFromEntryNode(entry);
163 
164     if (shouldTraverseFunctionsAndClasses) {
165       // If we're traversing inner functions, we need to rank the
166       // priority of them too.
167       for (DiGraphNode<Node, Branch> candidate : cfg.getDirectedGraphNodes()) {
168         Node value = candidate.getValue();
169         if (value != null && value.isFunction()) {
170           Preconditions.checkState(
171               !nodePriorities.containsKey(candidate) || candidate == entry);
172           prioritizeFromEntryNode(candidate);
173         }
174       }
175     }
176 
177     // At this point, all reachable nodes have been given a priority, but
178     // unreachable nodes have not been given a priority. Put them last.
179     // Presumably, it doesn't really matter what priority they get, since
180     // this shouldn't happen in real code.
181     for (DiGraphNode<Node, Branch> candidate : cfg.getDirectedGraphNodes()) {
182       if (!nodePriorities.containsKey(candidate)) {
183         nodePriorities.put(candidate, ++priorityCounter);
184       }
185     }
186 
187     // Again, the implicit return node is always last.
188     nodePriorities.put(cfg.getImplicitReturn(), ++priorityCounter);
189   }
190 
191   /**
192    * Given an entry node, find all the nodes reachable from that node
193    * and prioritize them.
194    */
195   private void prioritizeFromEntryNode(DiGraphNode<Node, Branch> entry) {
196     PriorityQueue<DiGraphNode<Node, Branch>> worklist =
197         new PriorityQueue<>(10, priorityComparator);
198     worklist.add(entry);
199 
200     while (!worklist.isEmpty()) {
201       DiGraphNode<Node, Branch> current = worklist.remove();
202       if (nodePriorities.containsKey(current)) {
203         continue;
204       }
205 
206       nodePriorities.put(current, ++priorityCounter);
207 
208       List<DiGraphNode<Node, Branch>> successors =
209           cfg.getDirectedSuccNodes(current);
210       worklist.addAll(successors);
211     }
212   }

214   @Override
215   public boolean shouldTraverse(
216       NodeTraversal nodeTraversal, Node n, Node parent) {
217     astPosition.put(n, astPositionCounter++);
218 
219     switch (n.getType()) {
220       case Token.CLASS:
221         return shouldTraverseFunctionsAndClasses;
222       case Token.FUNCTION:
223         if (shouldTraverseFunctionsAndClasses || n == cfg.getEntry().getValue()) {
224           exceptionHandler.push(n);
225           return true;
226         }
227         return false;
228       case Token.TRY:
229         exceptionHandler.push(n);
230         return true;
231     }
232 
233     /*
234      * We are going to stop the traversal depending on what the node's parent
235      * is.
236      *
237      * We are only interested in adding edges between nodes that change control
238      * flow. The most obvious ones are loops and IF-ELSE's. A statement
239      * transfers control to its next sibling.
240      *
241      * In case of an expression tree, there is no control flow within the tree
242      * even when there are short circuited operators and conditionals. When we
243      * are doing data flow analysis, we will simply synthesize lattices up the
244      * expression tree by finding the meet at each expression node.
245      *
246      * For example: within a Token.SWITCH, the expression in question does not
247      * change the control flow and need not to be considered.
248      */
249     if (parent != null) {
250       switch (parent.getType()) {
251         case Token.FOR:
252         case Token.FOR_OF:
253           // Only traverse the body of the for loop.
254           return n == parent.getLastChild();
255 
256         case Token.DO:
257           // Only traverse the body of the do-while.
258           return n != parent.getFirstChild().getNext();
259 
260         // Skip conditions, and only traverse the body of the cases
261         case Token.IF:
262         case Token.WHILE:
263         case Token.WITH:
264         case Token.SWITCH:
265         case Token.CASE:
266         case Token.CATCH:
267         case Token.LABEL:
268           return n != parent.getFirstChild();
269         case Token.FUNCTION:
270           return n == parent.getLastChild();
271         case Token.CONTINUE:
272         case Token.BREAK:
273         case Token.EXPR_RESULT:
274         case Token.VAR:
275         case Token.LET:
276         case Token.CONST:
277         case Token.RETURN:
278         case Token.THROW:
279           return false;
280         case Token.TRY:
281           /* When we are done with the TRY block and there is no FINALLY block,
282            * or done with both the TRY and CATCH block, then no more exceptions
283            * can be handled at this TRY statement, so it can be taken out of the
284            * stack.
285            */
286           if ((!NodeUtil.hasFinally(parent) && n == NodeUtil.getCatchBlock(parent))
287               || NodeUtil.isTryFinallyNode(parent, n)) {
288             Preconditions.checkState(exceptionHandler.peek() == parent);
289             exceptionHandler.pop();
290           }
291       }
292     }
293     return true;
294   }

296   @Override
297   public void visit(NodeTraversal t, Node n, Node parent) {
298     switch (n.getType()) {
299       case Token.IF:
300         handleIf(n);
301         return;
302       case Token.WHILE:
303         handleWhile(n);
304         return;
305       case Token.DO:
306         handleDo(n);
307         return;
308       case Token.FOR_OF:
309       case Token.FOR:
310         handleFor(n);
311         return;
312       case Token.SWITCH:
313         handleSwitch(n);
314         return;
315       case Token.CASE:
316         handleCase(n);
317         return;
318       case Token.DEFAULT_CASE:
319         handleDefault(n);
320         return;
321       case Token.BLOCK:
322       case Token.SCRIPT:
323         handleStmtList(n);
324         return;
325       case Token.FUNCTION:
326         handleFunction(n);
327         return;
328       case Token.EXPR_RESULT:
329         handleExpr(n);
330         return;
331       case Token.THROW:
332         handleThrow(n);
333         return;
334       case Token.TRY:
335         handleTry(n);
336         return;
337       case Token.CATCH:
338         handleCatch(n);
339         return;
340       case Token.BREAK:
341         handleBreak(n);
342         return;
343       case Token.CONTINUE:
344         handleContinue(n);
345         return;
346       case Token.RETURN:
347         handleReturn(n);
348         return;
349       case Token.WITH:
350         handleWith(n);
351         return;
352       case Token.LABEL:
353         return;
354       default:
355         handleStmt(n);
356         return;
357     }
358   }

360   private void handleIf(Node node) {
361     Node thenBlock = node.getFirstChild().getNext();
362     Node elseBlock = thenBlock.getNext();
363     createEdge(node, Branch.ON_TRUE, computeFallThrough(thenBlock));
364 
365     if (elseBlock == null) {
366       createEdge(node, Branch.ON_FALSE,
367           computeFollowNode(node, this)); // not taken branch
368     } else {
369       createEdge(node, Branch.ON_FALSE, computeFallThrough(elseBlock));
370     }
371     connectToPossibleExceptionHandler(
372         node, NodeUtil.getConditionExpression(node));
373   }
374 
375   private void handleWhile(Node node) {
376     // Control goes to the first statement if the condition evaluates to true.
377     createEdge(node, Branch.ON_TRUE,
378         computeFallThrough(node.getFirstChild().getNext()));
379 
380     // Control goes to the follow() if the condition evaluates to false.
381     createEdge(node, Branch.ON_FALSE,
382         computeFollowNode(node, this));
383     connectToPossibleExceptionHandler(
384         node, NodeUtil.getConditionExpression(node));
385   }
386 
387   private void handleDo(Node node) {
388     // The first edge can be the initial iteration as well as the iterations
389     // after.
390     createEdge(node, Branch.ON_TRUE, computeFallThrough(node.getFirstChild()));
391     // The edge that leaves the do loop if the condition fails.
392     createEdge(node, Branch.ON_FALSE,
393         computeFollowNode(node, this));
394     connectToPossibleExceptionHandler(
395         node, NodeUtil.getConditionExpression(node));
396   }

398   private void handleFor(Node forNode) {
399     if (NodeUtil.isForIn(forNode) || forNode.isForOf()) {
400       // We have:  for (index in collection) { body }
401       // or:       for (item of collection) { body }
402       Node item = forNode.getFirstChild();
403       Node collection = item.getNext();
404       Node body = collection.getNext();
405       // The collection behaves like init.
406       createEdge(collection, Branch.UNCOND, forNode);
407       // The edge that transfer control to the beginning of the loop body.
408       createEdge(forNode, Branch.ON_TRUE, computeFallThrough(body));
409       // The edge to end of the loop.
410       createEdge(forNode, Branch.ON_FALSE,
411           computeFollowNode(forNode, this));
412       connectToPossibleExceptionHandler(forNode, collection);
413     } else {
414       // We have for (init; cond; iter) { body }
415       Node init = forNode.getFirstChild();
416       Node cond = init.getNext();
417       Node iter = cond.getNext();
418       Node body = iter.getNext();
419       // After initialization, we transfer to the FOR which is in charge of
420       // checking the condition (for the first time).
421       createEdge(init, Branch.UNCOND, forNode);
422       // The edge that transfer control to the beginning of the loop body.
423       createEdge(forNode, Branch.ON_TRUE, computeFallThrough(body));
424       // The edge to end of the loop.
425       if (!cond.isEmpty()) {
426         createEdge(forNode, Branch.ON_FALSE,
427             computeFollowNode(forNode, this));
428       }
429       // The end of the body will have a unconditional branch to our iter
430       // (handled by calling computeFollowNode of the last instruction of the
431       // body. Our iter will jump to the forNode again to another condition
432       // check.
433       createEdge(iter, Branch.UNCOND, forNode);
434       connectToPossibleExceptionHandler(init, init);
435       connectToPossibleExceptionHandler(forNode, cond);
436       connectToPossibleExceptionHandler(iter, iter);
437     }
438   }

440   private void handleSwitch(Node node) {
441     // Transfer to the first non-DEFAULT CASE. if there are none, transfer
442     // to the DEFAULT or the EMPTY node.
443     Node next = getNextSiblingOfType(
444         node.getFirstChild().getNext(), Token.CASE, Token.EMPTY);
445     if (next != null) { // Has at least one CASE or EMPTY
446       createEdge(node, Branch.UNCOND, next);
447     } else { // Has no CASE but possibly a DEFAULT
448       if (node.getFirstChild().getNext() != null) {
449         createEdge(node, Branch.UNCOND, node.getFirstChild().getNext());
450       } else { // No CASE, no DEFAULT
451         createEdge(node, Branch.UNCOND, computeFollowNode(node, this));
452       }
453     }
454     connectToPossibleExceptionHandler(node, node.getFirstChild());
455   }
456 
457   private void handleCase(Node node) {
458     // Case is a bit tricky....First it goes into the body if condition is true.
459     createEdge(node, Branch.ON_TRUE,
460         node.getFirstChild().getNext());
461     // Look for the next CASE, skipping over DEFAULT.
462     Node next = getNextSiblingOfType(node.getNext(), Token.CASE);
463     if (next != null) { // Found a CASE
464       Preconditions.checkState(next.isCase());
465       createEdge(node, Branch.ON_FALSE, next);
466     } else { // No more CASE found, go back and search for a DEFAULT.
467       Node parent = node.getParent();
468       Node deflt = getNextSiblingOfType(
469         parent.getFirstChild().getNext(), Token.DEFAULT_CASE);
470       if (deflt != null) { // Has a DEFAULT
471         createEdge(node, Branch.ON_FALSE, deflt);
472       } else { // No DEFAULT found, go to the follow of the SWITCH.
473         createEdge(node, Branch.ON_FALSE, computeFollowNode(node, this));
474       }
475     }
476     connectToPossibleExceptionHandler(node, node.getFirstChild());
477   }
478 
479   private void handleDefault(Node node) {
480     // Directly goes to the body. It should not transfer to the next case.
481     createEdge(node, Branch.UNCOND, node.getFirstChild());
482   }
483 
484   private void handleWith(Node node) {
485     // Directly goes to the body. It should not transfer to the next case.
486     createEdge(node, Branch.UNCOND, node.getLastChild());
487     connectToPossibleExceptionHandler(node, node.getFirstChild());
488   }

490   private void handleStmtList(Node node) {
491     Node parent = node.getParent();
492     // Special case, don't add a block of empty CATCH block to the graph.
493     if (node.isBlock() && parent != null &&
494         parent.isTry() &&
495         NodeUtil.getCatchBlock(parent) == node &&
496         !NodeUtil.hasCatchHandler(node)) {
497       return;
498     }
499 
500     // A block transfer control to its first child if it is not empty.
501     Node child = node.getFirstChild();
502 
503     // Function declarations are skipped since control doesn't go into that
504     // function (unless it is called)
505     while (child != null && child.isFunction()) {
506       child = child.getNext();
507     }
508 
509     if (child != null) {
510       createEdge(node, Branch.UNCOND, computeFallThrough(child));
511     } else {
512       createEdge(node, Branch.UNCOND, computeFollowNode(node, this));
513     }
514 
515     // Synthetic blocks
516     if (parent != null) {
517       switch (parent.getType()) {
518         case Token.DEFAULT_CASE:
519         case Token.CASE:
520         case Token.TRY:
521           break;
522         default:
523           if (node.isBlock() && node.isSyntheticBlock()) {
524             createEdge(node, Branch.SYN_BLOCK, computeFollowNode(node, this));
525           }
526           break;
527       }
528     }
529   }

531   private void handleFunction(Node node) {
532     // A block transfer control to its first child if it is not empty.
533     Preconditions.checkState(node.isFunction());
534     Preconditions.checkState(node.getChildCount() == 3);
535     createEdge(node, Branch.UNCOND,
536         computeFallThrough(node.getLastChild()));
537     Preconditions.checkState(exceptionHandler.peek() == node);
538     exceptionHandler.pop();
539   }

541   private void handleExpr(Node node) {
542     createEdge(node, Branch.UNCOND, computeFollowNode(node, this));
543     connectToPossibleExceptionHandler(node, node);
544   }

546   private void handleThrow(Node node) {
547     connectToPossibleExceptionHandler(node, node);
548   }

550   private void handleTry(Node node) {
551     createEdge(node, Branch.UNCOND, node.getFirstChild());
552   }

554   private void handleCatch(Node node) {
555     createEdge(node, Branch.UNCOND, node.getLastChild());
556   }

558   private void handleBreak(Node node) {
559     String label = null;
560     // See if it is a break with label.
561     if (node.hasChildren()) {
562       label = node.getFirstChild().getString();
563     }
564     Node cur;
565     Node previous = null;
566     Node lastJump;
567     Node parent = node.getParent();
568     /*
569      * Continuously look up the ancestor tree for the BREAK target or the target
570      * with the corresponding label and connect to it. If along the path we
571      * discover a FINALLY, we will connect the BREAK to that FINALLY. From then
572      * on, we will just record the control flow changes in the finallyMap. This
573      * is due to the fact that we need to connect any node that leaves its own
574      * FINALLY block to the outer FINALLY or the BREAK's target but those nodes
575      * are not known yet due to the way we traverse the nodes.
576      */
577     for (cur = node, lastJump = node;
578         !isBreakTarget(cur, label);
579         cur = parent, parent = parent.getParent()) {
580       if (cur.isTry() && NodeUtil.hasFinally(cur)
581           && cur.getLastChild() != previous) {
582         if (lastJump == node) {
583           createEdge(lastJump, Branch.UNCOND, computeFallThrough(
584               cur.getLastChild()));
585         } else {
586           finallyMap.put(lastJump, computeFallThrough(cur.getLastChild()));
587         }
588         lastJump = cur;
589       }
590       if (parent == null) {
591         if (compiler.isIdeMode()) {
592           // In IDE mode, we expect that the data flow graph may
593           // not be well-formed.
594           return;
595         } else {
596           throw new IllegalStateException("Cannot find break target.");
597         }
598       }
599       previous = cur;
600     }
601     if (lastJump == node) {
602       createEdge(lastJump, Branch.UNCOND, computeFollowNode(cur, this));
603     } else {
604       finallyMap.put(lastJump, computeFollowNode(cur, this));
605     }
606   }

608   private void handleContinue(Node node) {
609     String label = null;
610     if (node.hasChildren()) {
611       label = node.getFirstChild().getString();
612     }
613     Node cur;
614     Node previous = null;
615     Node lastJump;
616 
617     // Similar to handBreak's logic with a few minor variation.
618     Node parent = node.getParent();
619     for (cur = node, lastJump = node;
620         !isContinueTarget(cur, parent, label);
621         cur = parent, parent = parent.getParent()) {
622       if (cur.isTry() && NodeUtil.hasFinally(cur)
623           && cur.getLastChild() != previous) {
624         if (lastJump == node) {
625           createEdge(lastJump, Branch.UNCOND, cur.getLastChild());
626         } else {
627           finallyMap.put(lastJump, computeFallThrough(cur.getLastChild()));
628         }
629         lastJump = cur;
630       }
631       Preconditions.checkState(parent != null, "Cannot find continue target.");
632       previous = cur;
633     }
634     Node iter = cur;
635     if (cur.getChildCount() == 4) {
636       iter = cur.getFirstChild().getNext().getNext();
637     }
638 
639     if (lastJump == node) {
640       createEdge(node, Branch.UNCOND, iter);
641     } else {
642       finallyMap.put(lastJump, iter);
643     }
644   }
645 
646   private void handleReturn(Node node) {
647     Node lastJump = null;
648     for (Node curHandler : exceptionHandler) {
649       if (curHandler.isFunction()) {
650         break;
651       }
652       if (NodeUtil.hasFinally(curHandler)) {
653         if (lastJump == null) {
654           createEdge(node, Branch.UNCOND, curHandler.getLastChild());
655         } else {
656           finallyMap.put(lastJump,
657               computeFallThrough(curHandler.getLastChild()));
658         }
659         lastJump = curHandler;
660       }
661     }
662 
663     if (node.hasChildren()) {
664       connectToPossibleExceptionHandler(node, node.getFirstChild());
665     }
666 
667     if (lastJump == null) {
668       createEdge(node, Branch.UNCOND, null);
669     } else {
670       finallyMap.put(lastJump, null);
671     }
672   }

674   private void handleStmt(Node node) {
675     // Simply transfer to the next line.
676     createEdge(node, Branch.UNCOND, computeFollowNode(node, this));
677     connectToPossibleExceptionHandler(node, node);
678   }

684   static Node computeFollowNode(Node node) {
685     return computeFollowNode(node, node, null);

688   /**
689    * Computes the follow() node of a given node and its parent. There is a side
690    * effect when calling this function. If this function computed an edge that
691    * exists a FINALLY, it'll attempt to connect the fromNode to the outer
692    * FINALLY according to the finallyMap.
693    *
694    * @param fromNode The original source node since {@code node} is changed
695    *        during recursion.
696    * @param node The node that follow() should compute.
697    */
698   private static Node computeFollowNode(
699       Node fromNode, Node node, ControlFlowAnalysis cfa) {
700     /*
701      * This is the case where:
702      *
703      * 1. Parent is null implies that we are transferring control to the end of
704      * the script.
705      *
706      * 2. Parent is a function implies that we are transferring control back to
707      * the caller of the function.
708      *
709      * 3. If the node is a return statement, we should also transfer control
710      * back to the caller of the function.
711      *
712      * 4. If the node is root then we have reached the end of what we have been
713      * asked to traverse.
714      *
715      * In all cases we should transfer control to a "symbolic return" node.
716      * This will make life easier for DFAs.
717      */
718     Node parent = node.getParent();
719     if (parent == null || parent.isFunction() ||
720         (cfa != null && node == cfa.root)) {
721       return null;
722     }
723 
724     // If we are just before a IF/WHILE/DO/FOR:
725     switch (parent.getType()) {
726       // The follow() of any of the path from IF would be what follows IF.
727       case Token.IF:
728         return computeFollowNode(fromNode, parent, cfa);
729       case Token.CASE:
730       case Token.DEFAULT_CASE:
731         // After the body of a CASE, the control goes to the body of the next
732         // case, without having to go to the case condition.
733         if (parent.getNext() != null) {
734           if (parent.getNext().isCase()) {
735             return parent.getNext().getFirstChild().getNext();
736           } else if (parent.getNext().isDefaultCase()) {
737             return parent.getNext().getFirstChild();
738           } else {
739             throw new IllegalStateException("Not reachable");
740           }
741         } else {
742           return computeFollowNode(fromNode, parent, cfa);
743         }
744       case Token.FOR:
745       case Token.FOR_OF:
746         if (parent.isForOf() || NodeUtil.isForIn(parent)) {
747           return parent;
748         } else {
749           return parent.getFirstChild().getNext().getNext();
750         }
751       case Token.WHILE:
752       case Token.DO:
753         return parent;
754       case Token.TRY:
755         // If we are coming out of the TRY block...
756         if (parent.getFirstChild() == node) {
757           if (NodeUtil.hasFinally(parent)) { // and have FINALLY block.
758             return computeFallThrough(parent.getLastChild());
759           } else { // and have no FINALLY.
760             return computeFollowNode(fromNode, parent, cfa);
761           }
762         // CATCH block.
763         } else if (NodeUtil.getCatchBlock(parent) == node){
764           if (NodeUtil.hasFinally(parent)) { // and have FINALLY block.
765             return computeFallThrough(node.getNext());
766           } else {
767             return computeFollowNode(fromNode, parent, cfa);
768           }
769         // If we are coming out of the FINALLY block...
770         } else if (parent.getLastChild() == node){
771           if (cfa != null) {
772             for (Node finallyNode : cfa.finallyMap.get(parent)) {
773               cfa.createEdge(fromNode, Branch.ON_EX, finallyNode);
774             }
775           }
776           return computeFollowNode(fromNode, parent, cfa);
777         }
778     }
779 
780     // Now that we are done with the special cases follow should be its
781     // immediate sibling, unless its sibling is a function
782     Node nextSibling = node.getNext();
783 
784     // Skip function declarations because control doesn't get pass into it.
785     while (nextSibling != null && nextSibling.isFunction()) {
786       nextSibling = nextSibling.getNext();
787     }
788 
789     if (nextSibling != null) {
790       return computeFallThrough(nextSibling);
791     } else {
792       // If there are no more siblings, control is transferred up the AST.
793       return computeFollowNode(fromNode, parent, cfa);
794     }
795   }

797   /**
798    * Computes the destination node of n when we want to fallthrough into the
799    * subtree of n. We don't always create a CFG edge into n itself because of
800    * DOs and FORs.
801    */
802   static Node computeFallThrough(Node n) {
803     switch (n.getType()) {
804       case Token.DO:
805         return computeFallThrough(n.getFirstChild());
806       case Token.FOR:
807       case Token.FOR_OF:
808         if (n.isForOf() || NodeUtil.isForIn(n)) {
809           return n.getFirstChild().getNext();
810         }
811         return computeFallThrough(n.getFirstChild());
812       case Token.LABEL:
813         return computeFallThrough(n.getLastChild());
814       default:
815         return n;
816     }
817   }

819   /**
820    * Connects the two nodes in the control flow graph.
821    *
822    * @param fromNode Source.
823    * @param toNode Destination.
824    */
825   private void createEdge(Node fromNode, ControlFlowGraph.Branch branch,
826       Node toNode) {
827     cfg.createNode(fromNode);
828     cfg.createNode(toNode);
829     cfg.connectIfNotFound(fromNode, branch, toNode);
830   }

832   /**
833    * Connects cfgNode to the proper CATCH block if target subtree might throw
834    * an exception. If there are FINALLY blocks reached before a CATCH, it will
835    * make the corresponding entry in finallyMap.
836    */
837   private void connectToPossibleExceptionHandler(Node cfgNode, Node target) {
838     if (mayThrowException(target) && !exceptionHandler.isEmpty()) {
839       Node lastJump = cfgNode;
840       for (Node handler : exceptionHandler) {
841         if (handler.isFunction()) {
842           return;
843         }
844         Preconditions.checkState(handler.isTry());
845         Node catchBlock = NodeUtil.getCatchBlock(handler);
846 
847         boolean lastJumpInCatchBlock = false;
848         for (Node ancestor : lastJump.getAncestors()) {
849           if (ancestor == handler) {
850             break;
851           } else if (ancestor == catchBlock) {
852             lastJumpInCatchBlock = true;
853             break;
854           }
855         }
856 
857         // No catch but a FINALLY, or lastJump is inside the catch block.
858         if (!NodeUtil.hasCatchHandler(catchBlock) || lastJumpInCatchBlock) {
859           if (lastJump == cfgNode) {
860             createEdge(cfgNode, Branch.ON_EX, handler.getLastChild());
861           } else {
862             finallyMap.put(lastJump, handler.getLastChild());
863           }
864         } else { // Has a catch.
865           if (lastJump == cfgNode) {
866             createEdge(cfgNode, Branch.ON_EX, catchBlock);
867             return;
868           } else {
869             finallyMap.put(lastJump, catchBlock);
870           }
871         }
872         lastJump = handler;
873       }
874     }
875   }

881     for (Node c = first; c != null; c = c.getNext()) {
882       for (int type : types) {

892    * Checks if target is actually the break target of labeled continue. The
893    * label can be null if it is an unlabeled break.

897       matchLabel(target.getParent(), label);

900   /**
901    * Checks if target is actually the continue target of labeled continue. The
902    * label can be null if it is an unlabeled continue.
903    */
904   private static boolean isContinueTarget(
905       Node target, Node parent, String label) {
906     return isContinueStructure(target) && matchLabel(parent, label);
907   }
908   /**
909    * Check if label is actually referencing the target control structure. If
910    * label is null, it always returns true.
911    */
912   private static boolean matchLabel(Node target, String label) {
913     if (label == null) {
914       return true;
915     }
916     while (target.isLabel()) {
917       if (target.getFirstChild().getString().equals(label)) {
918         return true;
919       }
920       target = target.getParent();
921     }
922     return false;
923   }

925   /**
926    * Determines if the subtree might throw an exception.
927    */
928   public static boolean mayThrowException(Node n) {
929     switch (n.getType()) {
930       case Token.CALL:
931       case Token.TAGGED_TEMPLATELIT:
932       case Token.GETPROP:
933       case Token.GETELEM:
934       case Token.THROW:
935       case Token.NEW:
936       case Token.ASSIGN:
937       case Token.INC:
938       case Token.DEC:
939       case Token.INSTANCEOF:
940       case Token.IN:
941         return true;
942       case Token.FUNCTION:
943         return false;
944     }
945     for (Node c = n.getFirstChild(); c != null; c = c.getNext()) {
946       if (!ControlFlowGraph.isEnteringNewCfgNode(c) && mayThrowException(c)) {
947         return true;
948       }
949     }
950     return false;
951   }

953   /**
954    * Determines whether the given node can be terminated with a BREAK node.
955    */
956   static boolean isBreakStructure(Node n, boolean labeled) {
957     switch (n.getType()) {
958       case Token.FOR:
959       case Token.FOR_OF:
960       case Token.DO:
961       case Token.WHILE:
962       case Token.SWITCH:
963         return true;
964       case Token.BLOCK:
965       case Token.IF:
966       case Token.TRY:
967         return labeled;
968       default:
969         return false;
970     }
971   }

974    * Determines whether the given node can be advanced with a CONTINUE node.
975    */
976   static boolean isContinueStructure(Node n) {
977     switch (n.getType()) {
978       case Token.FOR:
979       case Token.FOR_OF:
980       case Token.DO:
981       case Token.WHILE:
982         return true;
983       default:
984         return false;
985     }

988   /**
989    * Get the TRY block with a CATCH that would be run if n throws an exception.
990    * @return The CATCH node or null if it there isn't a CATCH before the
991    *     the function terminates.
992    */
993   static Node getExceptionHandler(Node n) {
994     for (Node cur = n;
995         !cur.isScript() && !cur.isFunction();
996         cur = cur.getParent()) {
997       Node catchNode = getCatchHandlerForBlock(cur);
998       if (catchNode != null) {
999         return catchNode;
1000       }
1001     }
1002     return null;
1003   }
1004 
1005   /**
1006    * Locate the catch BLOCK given the first block in a TRY.
1007    * @return The CATCH node or null there is no catch handler.
1008    */
1009   static Node getCatchHandlerForBlock(Node block) {
1010     if (block.isBlock() &&
1011         block.getParent().isTry() &&
1012         block.getParent().getFirstChild() == block) {
1013       for (Node s = block.getNext(); s != null; s = s.getNext()) {
1014         if (NodeUtil.hasCatchHandler(s)) {
1015           return s.getFirstChild();
1016         }
1017       }
1018     }
1019     return null;
1020   }

1022   /**
1023    * A {@link ControlFlowGraph} which provides a node comparator based on the
1024    * pre-order traversal of the AST.
1025    */
1026   private static class AstControlFlowGraph extends ControlFlowGraph<Node> {
1027     private final Map<DiGraphNode<Node, Branch>, Integer> priorities;
1028 
1029     /**
1030      * Constructor.
1031      * @param entry The entry node.
1032      * @param priorities The map from nodes to position in the AST (to be
1033      *    filled by the {@link ControlFlowAnalysis#shouldTraverse}).
1034      */
1035     private AstControlFlowGraph(Node entry,
1036         Map<DiGraphNode<Node, Branch>, Integer> priorities,
1037         boolean edgeAnnotations) {
1038       super(entry,
1039           true /* node annotations */, edgeAnnotations);
1040       this.priorities = priorities;
1041     }
1042 
1043     @Override
1044     /**
1045      * Returns a node comparator based on the pre-order traversal of the AST.
1046      * @param isForward x 'before' y in the pre-order traversal implies
1047      * x 'less than' y (if true) and x 'greater than' y (if false).
1048      */
1049     public Comparator<DiGraphNode<Node, Branch>> getOptionalNodeComparator(
1050         boolean isForward) {
1051       if (isForward) {
1052         return new Comparator<DiGraphNode<Node, Branch>>() {
1053           @Override
1054           public int compare(
1055               DiGraphNode<Node, Branch> n1, DiGraphNode<Node, Branch> n2) {
1056             return getPosition(n1) - getPosition(n2);
1057           }
1058         };
1059       } else {
1060         return new Comparator<DiGraphNode<Node, Branch>>() {
1061           @Override
1062           public int compare(
1063               DiGraphNode<Node, Branch> n1, DiGraphNode<Node, Branch> n2) {
1064             return getPosition(n2) - getPosition(n1);
1065           }
1066         };
1067       }
1068     }
1069 
1070     /**
1071      * Gets the pre-order traversal position of the given node.
1072      * @return An arbitrary counter used for comparing positions.
1073      */
1074     private int getPosition(DiGraphNode<Node, Branch> n) {
1075       Integer priority = priorities.get(n);
1076       Preconditions.checkNotNull(priority);
1077       return priority;
1078     }
1079   }
