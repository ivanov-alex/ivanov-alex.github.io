30  /**
31   * CodeGenerator generates codes from a parse tree, sending it to the specified
32   * CodeConsumer.
33   *
34   */
35  class CodeGenerator {
36    private static final String LT_ESCAPED = "\\x3c";
37    private static final String GT_ESCAPED = "\\x3e";
38  
39    // A memoizer for formatting strings as JS strings.
40    private final Map<String, String> escapedJsStrings = new HashMap<>();
41  
42    private final CodeConsumer cc;
43  
44    private final OutputCharsetEncoder outputCharsetEncoder;
45  
46    private final boolean preferSingleQuotes;
47    private final boolean preserveTypeAnnotations;
48    private final boolean trustedStrings;
49    private final LanguageMode languageMode;
50  
51    private CodeGenerator(CodeConsumer consumer) {
52      cc = consumer;
53      outputCharsetEncoder = null;
54      preferSingleQuotes = false;
55      trustedStrings = true;
56      languageMode = LanguageMode.ECMASCRIPT5;
57      preserveTypeAnnotations = false;
58    }
59  
60    static CodeGenerator forCostEstimation(CodeConsumer consumer) {
61      return new CodeGenerator(consumer);
62    }
63  
64    CodeGenerator(
65        CodeConsumer consumer,
66        CompilerOptions options) {
67      cc = consumer;
68  
69      this.outputCharsetEncoder = new OutputCharsetEncoder(options.getOutputCharset());
70      this.preferSingleQuotes = options.preferSingleQuotes;
71      this.trustedStrings = options.trustedStrings;
72      this.languageMode = options.getLanguageOut();
73      this.preserveTypeAnnotations = options.preserveTypeAnnotations;
74    }
75  
76    /**
77     * Insert a ECMASCRIPT 5 strict annotation.
78     */
79    public void tagAsStrict() {
80      add("'use strict';");
81    }
82  
83    void add(String str) {
84      cc.add(str);
85    }
86  
87    private void addIdentifier(String identifier) {
88      cc.addIdentifier(identifierEscape(identifier));
89    }
90  
91    void add(Node n) {
92      add(n, Context.OTHER);
93    }
94  
95    void add(Node n, Context context) {
96      if (!cc.continueProcessing()) {
97        return;
98      }
99  
100     if (preserveTypeAnnotations && n.getJSDocInfo() != null) {
101       add(JSDocInfoPrinter.print(n.getJSDocInfo()));
102     }
103 
104     int type = n.getType();
105     String opstr = NodeUtil.opToStr(type);
106     int childCount = n.getChildCount();
107     Node first = n.getFirstChild();
108     Node last = n.getLastChild();
109 
110     // Handle all binary operators
111     if (opstr != null && first != last) {
112       Preconditions.checkState(
113           childCount == 2,
114           "Bad binary operator \"%s\": expected 2 arguments but got %s",
115           opstr, childCount);
116       int p = NodeUtil.precedence(type);
117 
118       // For right-hand-side of operations, only pass context if it's
119       // the IN_FOR_INIT_CLAUSE one.
120       Context rhsContext = getContextForNoInOperator(context);
121 
122       boolean needsParens = (context == Context.START_OF_EXPR) && first.isObjectPattern();
123       if (n.isAssign() && needsParens) {
124         add("(");
125       }
126 
127       if (NodeUtil.isAssignmentOp(n) && NodeUtil.isAssignmentOp(last)) {
128         // Assignments are the only right-associative binary operators
129         addExpr(first, p, context);
130         cc.addOp(opstr, true);
131         addExpr(last, p, rhsContext);
132       } else {
133         unrollBinaryOperator(n, type, opstr, context, rhsContext, p, p + 1);
134       }
135 
136       if (n.isAssign() && needsParens) {
137         add(")");
138       }
139       return;
140     }
141 
142     cc.startSourceMapping(n);
143 
144     switch (type) {
145       case Token.TRY:
146         {
147           Preconditions.checkState(
148               first.getNext().isBlock() && !first.getNext().hasMoreThanOneChild());
149           Preconditions.checkState(childCount >= 2 && childCount <= 3);
150 
151           add("try");
152           add(first, Context.PRESERVE_BLOCK);
153 
154           // second child contains the catch block, or nothing if there
155           // isn't a catch block
156           Node catchblock = first.getNext().getFirstChild();
157           if (catchblock != null) {
158             add(catchblock);
159           }
160 
161           if (childCount == 3) {
162             cc.maybeInsertSpace();
163             add("finally");
164             add(last, Context.PRESERVE_BLOCK);
165           }
166           break;
167         }
168 
169       case Token.CATCH:
170         Preconditions.checkState(childCount == 2);
171         cc.maybeInsertSpace();
172         add("catch");
173         cc.maybeInsertSpace();
174         add("(");
175         add(first);
176         add(")");
177         add(last, Context.PRESERVE_BLOCK);
178         break;
179 
180       case Token.THROW:
181         Preconditions.checkState(childCount == 1);
182         add("throw");
183         cc.maybeInsertSpace();
184         add(first);
185 
186         // Must have a ';' after a throw statement, otherwise safari can't
187         // parse this.
188         cc.endStatement(true);
189         break;
190 
191       case Token.RETURN:
192         add("return");
193         if (childCount == 1) {
194           cc.maybeInsertSpace();
195           add(first);
196         } else {
197           Preconditions.checkState(childCount == 0);
198         }
199         cc.endStatement();
200         break;
201 
202       case Token.VAR:
203         if (first != null) {
204           add("var ");
205           addList(first, false, getContextForNoInOperator(context), ",");
206         }
207         break;
208 
209       case Token.CONST:
210         add("const ");
211         addList(first, false, getContextForNoInOperator(context), ",");
212         break;
213 
214       case Token.LET:
215         add("let ");
216         addList(first, false, getContextForNoInOperator(context), ",");
217         break;
218 
219       case Token.LABEL_NAME:
220         Preconditions.checkState(!n.getString().isEmpty());
221         addIdentifier(n.getString());
222         break;
223 
224       case Token.NAME:
225         addIdentifier(n.getString());
226         maybeAddOptional(n);
227         maybeAddTypeDecl(n);
228 
229         if (first != null && !first.isEmpty()) {
230           Preconditions.checkState(childCount == 1);
231           cc.addOp("=", true);
232           if (first.isComma()) {
233             addExpr(first, NodeUtil.precedence(Token.ASSIGN), Context.OTHER);
234           } else {
235             // Add expression, consider nearby code at lowest level of
236             // precedence.
237             addExpr(first, 0, getContextForNoInOperator(context));
238           }
239         }
240         break;
241 
242       case Token.ARRAYLIT:
243         add("[");
244         addArrayList(first);
245         add("]");
246         break;
247 
248       case Token.ARRAY_PATTERN:
249         addArrayPattern(n);
250         maybeAddTypeDecl(n);
251         break;
252 
253       case Token.PARAM_LIST:
254         add("(");
255         addList(first);
256         add(")");
257         break;
258 
259       case Token.DEFAULT_VALUE:
260         add(first);
261         maybeAddTypeDecl(n);
262         cc.addOp("=", true);
263         add(first.getNext());
264         break;
265 
266       case Token.COMMA:
267         Preconditions.checkState(childCount == 2);
268         unrollBinaryOperator(
269             n, Token.COMMA, ",", context, getContextForNoInOperator(context), 0, 0);
270         break;
271 
272       case Token.NUMBER:
273         Preconditions.checkState(childCount == 0);
274         cc.addNumber(n.getDouble());
275         break;
276 
277       case Token.TYPEOF:
278       case Token.VOID:
279       case Token.NOT:
280       case Token.BITNOT:
281       case Token.POS:
282         {
283           // All of these unary operators are right-associative
284           Preconditions.checkState(childCount == 1);
285           cc.addOp(NodeUtil.opToStrNoFail(type), false);
286           addExpr(first, NodeUtil.precedence(type), Context.OTHER);
287           break;
288         }
289 
290       case Token.NEG:
291         {
292           Preconditions.checkState(childCount == 1);
293 
294           // It's important to our sanity checker that the code
295           // we print produces the same AST as the code we parse back.
296           // NEG is a weird case because Rhino parses "- -2" as "2".
297           if (n.getFirstChild().isNumber()) {
298             cc.addNumber(-n.getFirstChild().getDouble());
299           } else {
300             cc.addOp(NodeUtil.opToStrNoFail(type), false);
301             addExpr(first, NodeUtil.precedence(type), Context.OTHER);
302           }
303 
304           break;
305         }
306 
307       case Token.HOOK:
308         {
309           Preconditions.checkState(childCount == 3);
310           int p = NodeUtil.precedence(type);
311           Context rhsContext = getContextForNoInOperator(context);
312           addExpr(first, p + 1, context);
313           cc.addOp("?", true);
314           addExpr(first.getNext(), 1, rhsContext);
315           cc.addOp(":", true);
316           addExpr(last, 1, rhsContext);
317           break;
318         }
319 
320       case Token.REGEXP:
321         if (!first.isString() || !last.isString()) {
322           throw new Error("Expected children to be strings");
323         }
324 
325         String regexp = regexpEscape(first.getString(), outputCharsetEncoder);
326 
327         // I only use one .add because whitespace matters
328         if (childCount == 2) {
329           add(regexp + last.getString());
330         } else {
331           Preconditions.checkState(childCount == 1);
332           add(regexp);
333         }
334         break;
335 
336       case Token.FUNCTION: {
337         if (n.getClass() != Node.class) {
338           throw new Error("Unexpected Node subclass.");
339         }
340         Preconditions.checkState(childCount == 3);
341         if (n.isArrowFunction()) {
342           addArrowFunction(n, first, last, context);
343         } else {
344           addFunction(n, first, last, context);
345         }
346         break;
347       }
348       case Token.REST:
349         add("...");
350         add(n.getString());
351         maybeAddTypeDecl(n);
352         break;
353 
354       case Token.SPREAD:
355         add("...");
356         add(n.getFirstChild());
357         break;
358 
359       case Token.EXPORT:
360         add("export");
361         if (n.getBooleanProp(Node.EXPORT_DEFAULT)) {
362           add("default");
363         }
364         if (n.getBooleanProp(Node.EXPORT_ALL_FROM)) {
365           add("*");
366           Preconditions.checkState(first != null && first.isEmpty());
367         } else {
368           add(first);
369         }
370         if (childCount == 2) {
371           add("from");
372           add(last);
373         }
374         processEnd(first, context);
375         break;
376 
377       case Token.IMPORT:
378         add("import");
379 
380         Node second = first.getNext();
381         if (!first.isEmpty()) {
382           add(first);
383           if (!second.isEmpty()) {
384             cc.listSeparator();
385           }
386         }
387         if (!second.isEmpty()) {
388           add(second);
389         }
390         if (!first.isEmpty() || !second.isEmpty()) {
391           add("from");
392         }
393         add(last);
394         cc.endStatement();
395         break;
396 
397       case Token.EXPORT_SPECS:
398       case Token.IMPORT_SPECS:
399         add("{");
400         for (Node c = first; c != null; c = c.getNext()) {
401           if (c != first) {
402             cc.listSeparator();
403           }
404           add(c);
405         }
406         add("}");
407         break;
408 
409       case Token.EXPORT_SPEC:
410       case Token.IMPORT_SPEC:
411         add(first);
412         if (first != last) {
413           add("as");
414           add(last);
415         }
416         break;
417 
418       case Token.IMPORT_STAR:
419         add("*");
420         add("as");
421         add(n.getString());
422         break;
423 
424         // CLASS -> NAME,EXPR|EMPTY,BLOCK
425       case Token.CLASS:
426         {
427           Preconditions.checkState(childCount == 3);
428           boolean classNeedsParens = (context == Context.START_OF_EXPR);
429           if (classNeedsParens) {
430             add("(");
431           }
432 
433           Node name = first;
434           Node superClass = first.getNext();
435           Node members = last;
436 
437           add("class");
438           if (!name.isEmpty()) {
439             add(name);
440           }
441 
442           maybeAddGenericTypes(first);
443 
444           if (!superClass.isEmpty()) {
445             add("extends");
446             add(superClass);
447           }
448 
449           Node interfaces = (Node) n.getProp(Node.IMPLEMENTS);
450           if (interfaces != null) {
451             add("implements");
452             Node child = interfaces.getFirstChild();
453             add(child);
454             while ((child = child.getNext()) != null) {
455               add(",");
456               cc.maybeInsertSpace();
457               add(child);
458             }
459           }
460           add(members);
461           cc.endClass(context == Context.STATEMENT);
462 
463           if (classNeedsParens) {
464             add(")");
465           }
466         }
467         break;
468 
469       case Token.CLASS_MEMBERS:
470       case Token.INTERFACE_MEMBERS:
471       case Token.NAMESPACE_ELEMENTS:
472         cc.beginBlock();
473         for (Node c = first; c != null; c = c.getNext()) {
474           add(c);
475           processEnd(c, context);
476           cc.endLine();
477         }
478         cc.endBlock(false);
479         break;
480       case Token.ENUM_MEMBERS:
481         cc.beginBlock();
482         for (Node c = first; c != null; c = c.getNext()) {
483           add(c);
484           if (c.getNext() != null) {
485             add(",");
486           }
487           cc.endLine();
488         }
489         cc.endBlock(false);
490         break;
491       case Token.GETTER_DEF:
492       case Token.SETTER_DEF:
493       case Token.MEMBER_FUNCTION_DEF:
494       case Token.MEMBER_VARIABLE_DEF:
495         {
496           n.getParent().toStringTree();
497           Preconditions.checkState(
498               n.getParent().isObjectLit()
499                   || n.getParent().isClassMembers()
500                   || n.getParent().isInterfaceMembers()
501                   || n.getParent().isRecordType()
502                   || n.getParent().isIndexSignature());
503 
504           maybeAddAccessibilityModifier(n);
505           if (n.isStaticMember()) {
506             add("static ");
507           }
508 
509           if (!n.isMemberVariableDef() && n.getFirstChild().isGeneratorFunction()) {
510             Preconditions.checkState(type == Token.MEMBER_FUNCTION_DEF);
511             add("*");
512           }
513 
514           switch (type) {
515             case Token.GETTER_DEF:
516               // Get methods have no parameters.
517               Preconditions.checkState(!first.getChildAtIndex(1).hasChildren());
518               add("get ");
519               break;
520             case Token.SETTER_DEF:
521               // Set methods have one parameter.
522               Preconditions.checkState(first.getChildAtIndex(1).hasOneChild());
523               add("set ");
524               break;
525             case Token.MEMBER_FUNCTION_DEF:
526             case Token.MEMBER_VARIABLE_DEF:
527               // nothing to do.
528               break;
529           }
530 
531           // The name is on the GET or SET node.
532           String name = n.getString();
533           if (n.isMemberVariableDef()) {
534             add(n.getString());
535             maybeAddOptional(n);
536             maybeAddTypeDecl(n);
537           } else {
538             Preconditions.checkState(childCount == 1);
539             Preconditions.checkState(first.isFunction());
540 
541             // The function referenced by the definition should always be unnamed.
542             Preconditions.checkState(first.getFirstChild().getString().isEmpty());
543 
544             Node fn = first;
545             Node parameters = fn.getChildAtIndex(1);
546             Node body = fn.getLastChild();
547 
548             // Add the property name.
549             if (!n.isQuotedString()
550                 && TokenStream.isJSIdentifier(name)
551                 &&
552                 // do not encode literally any non-literal characters that were
553                 // Unicode escaped.
554                 NodeUtil.isLatin(name)) {
555               add(name);
556               maybeAddGenericTypes(fn.getFirstChild());
557             } else {
558               // Determine if the string is a simple number.
559               double d = getSimpleNumber(name);
560               if (!Double.isNaN(d)) {
561                 cc.addNumber(d);
562               } else {
563                 addJsString(n);
564               }
565             }
566             maybeAddOptional(fn);
567             add(parameters);
568             maybeAddTypeDecl(fn);
569             add(body, Context.PRESERVE_BLOCK);
570           }
571           break;
572         }
573 
574       case Token.SCRIPT:
575       case Token.BLOCK:
576         {
577           if (n.getClass() != Node.class) {
578             throw new Error("Unexpected Node subclass.");
579           }
580           boolean preserveBlock = context == Context.PRESERVE_BLOCK;
581           if (preserveBlock) {
582             cc.beginBlock();
583           }
584 
585           boolean preferLineBreaks =
586               type == Token.SCRIPT
587                   || (type == Token.BLOCK
588                       && !preserveBlock
589                       && n.getParent() != null
590                       && n.getParent().isScript());
591           for (Node c = first; c != null; c = c.getNext()) {
592             add(c, Context.STATEMENT);
593 
594             // VAR doesn't include ';' since it gets used in expressions
595             if (NodeUtil.isNameDeclaration(c)) {
596               cc.endStatement();
597             }
598 
599             if (c.isFunction() || c.isClass()) {
600               cc.maybeLineBreak();
601             }
602 
603             // Prefer to break lines in between top-level statements
604             // because top-level statements are more homogeneous.
605             if (preferLineBreaks) {
606               cc.notePreferredLineBreak();
607             }
608           }
609           if (preserveBlock) {
610             cc.endBlock(cc.breakAfterBlockFor(n, context == Context.STATEMENT));
611           }
612           break;
613         }
614 
615       case Token.FOR:
616         if (childCount == 4) {
617           add("for");
618           cc.maybeInsertSpace();
619           add("(");
620           if (NodeUtil.isNameDeclaration(first)) {
621             add(first, Context.IN_FOR_INIT_CLAUSE);
622           } else {
623             addExpr(first, 0, Context.IN_FOR_INIT_CLAUSE);
624           }
625           add(";");
626           add(first.getNext());
627           add(";");
628           add(first.getNext().getNext());
629           add(")");
630           addNonEmptyStatement(last, getContextForNonEmptyExpression(context), false);
631         } else {
632           Preconditions.checkState(childCount == 3);
633           add("for");
634           cc.maybeInsertSpace();
635           add("(");
636           add(first);
637           add("in");
638           add(first.getNext());
639           add(")");
640           addNonEmptyStatement(last, getContextForNonEmptyExpression(context), false);
641         }
642         break;
643 
644       case Token.FOR_OF:
645         // A "for-of" inside an array comprehension only has two children.
646         Preconditions.checkState(childCount == 3);
647         add("for");
648         cc.maybeInsertSpace();
649         add("(");
650         add(first);
651         add("of");
652         add(first.getNext());
653         add(")");
654         addNonEmptyStatement(last, getContextForNonEmptyExpression(context), false);
655         break;
656 
657       case Token.DO:
658         Preconditions.checkState(childCount == 2);
659         add("do");
660         addNonEmptyStatement(first, Context.OTHER, false);
661         cc.maybeInsertSpace();
662         add("while");
663         cc.maybeInsertSpace();
664         add("(");
665         add(last);
666         add(")");
667         cc.endStatement();
668         break;
669 
670       case Token.WHILE:
671         Preconditions.checkState(childCount == 2);
672         add("while");
673         cc.maybeInsertSpace();
674         add("(");
675         add(first);
676         add(")");
677         addNonEmptyStatement(last, getContextForNonEmptyExpression(context), false);
678         break;
679 
680       case Token.EMPTY:
681         Preconditions.checkState(childCount == 0);
682         break;
683 
684       case Token.GETPROP:
685         {
686           Preconditions.checkState(
687               childCount == 2, "Bad GETPROP: expected 2 children, but got %s", childCount);
688           Preconditions.checkState(last.isString(), "Bad GETPROP: RHS should be STRING");
689           boolean needsParens = (first.isNumber());
690           if (needsParens) {
691             add("(");
692           }
693           addExpr(first, NodeUtil.precedence(type), context);
694           if (needsParens) {
695             add(")");
696           }
697           if (this.languageMode == LanguageMode.ECMASCRIPT3
698               && TokenStream.isKeyword(last.getString())) {
699             // Check for ECMASCRIPT3 keywords.
700             add("[");
701             add(last);
702             add("]");
703           } else {
704             add(".");
705             addIdentifier(last.getString());
706           }
707           break;
708         }
709 
710       case Token.GETELEM:
711         Preconditions.checkState(
712             childCount == 2, "Bad GETELEM: expected 2 children but got %s", childCount);
713         addExpr(first, NodeUtil.precedence(type), context);
714         add("[");
715         add(first.getNext());
716         add("]");
717         break;
718 
719       case Token.WITH:
720         Preconditions.checkState(childCount == 2);
721         add("with(");
722         add(first);
723         add(")");
724         addNonEmptyStatement(last, getContextForNonEmptyExpression(context), false);
725         break;
726 
727       case Token.INC:
728       case Token.DEC:
729         {
730           Preconditions.checkState(childCount == 1);
731           String o = type == Token.INC ? "++" : "--";
732           boolean postProp = n.getBooleanProp(Node.INCRDECR_PROP);
733           if (postProp) {
734             addExpr(first, NodeUtil.precedence(type), context);
735             cc.addOp(o, false);
736           } else {
737             cc.addOp(o, false);
738             add(first);
739           }
740           break;
741         }
742 
743       case Token.CALL:
744         // We have two special cases here:
745         // 1) If the left hand side of the call is a direct reference to eval,
746         // then it must have a DIRECT_EVAL annotation. If it does not, then
747         // that means it was originally an indirect call to eval, and that
748         // indirectness must be preserved.
749         // 2) If the left hand side of the call is a property reference,
750         // then the call must not a FREE_CALL annotation. If it does, then
751         // that means it was originally an call without an explicit this and
752         // that must be preserved.
753         if (isIndirectEval(first) || n.getBooleanProp(Node.FREE_CALL) && NodeUtil.isGet(first)) {
754           add("(0,");
755           addExpr(first, NodeUtil.precedence(Token.COMMA), Context.OTHER);
756           add(")");
757         } else {
758           addExpr(first, NodeUtil.precedence(type), context);
759         }
760         Node args = first.getNext();
761         add("(");
762         addList(args);
763         add(")");
764         break;
765 
766       case Token.IF:
767         boolean hasElse = childCount == 3;
768         boolean ambiguousElseClause = context == Context.BEFORE_DANGLING_ELSE && !hasElse;
769         if (ambiguousElseClause) {
770           cc.beginBlock();
771         }
772 
773         add("if");
774         cc.maybeInsertSpace();
775         add("(");
776         add(first);
777         add(")");
778 
779         // An "if" node inside an array comprehension only has one child.
780         if (childCount == 1) {
781           break;
782         }
783 
784         if (hasElse) {
785           addNonEmptyStatement(first.getNext(), Context.BEFORE_DANGLING_ELSE, false);
786           cc.maybeInsertSpace();
787           add("else");
788           addNonEmptyStatement(last, getContextForNonEmptyExpression(context), false);
789         } else {
790           addNonEmptyStatement(first.getNext(), Context.OTHER, false);
791           Preconditions.checkState(childCount == 2);
792         }
793 
794         if (ambiguousElseClause) {
795           cc.endBlock();
796         }
797         break;
798 
799       case Token.NULL:
800         Preconditions.checkState(childCount == 0);
801         cc.addConstant("null");
802         break;
803 
804       case Token.THIS:
805         Preconditions.checkState(childCount == 0);
806         add("this");
807         break;
808 
809       case Token.SUPER:
810         Preconditions.checkState(childCount == 0);
811         add("super");
812         break;
813 
814       case Token.YIELD:
815         Preconditions.checkState(childCount == 1);
816         add("yield");
817         cc.maybeInsertSpace();
818         if (n.isYieldFor()) {
819           add("*");
820         }
821         addExpr(first, NodeUtil.precedence(type), Context.OTHER);
822         break;
823 
824       case Token.FALSE:
825         Preconditions.checkState(childCount == 0);
826         cc.addConstant("false");
827         break;
828 
829       case Token.TRUE:
830         Preconditions.checkState(childCount == 0);
831         cc.addConstant("true");
832         break;
833 
834       case Token.CONTINUE:
835         Preconditions.checkState(childCount <= 1);
836         add("continue");
837         if (childCount == 1) {
838           if (!first.isLabelName()) {
839             throw new Error("Unexpected token type. Should be LABEL_NAME.");
840           }
841           add(" ");
842           add(first);
843         }
844         cc.endStatement();
845         break;
846 
847       case Token.DEBUGGER:
848         Preconditions.checkState(childCount == 0);
849         add("debugger");
850         cc.endStatement();
851         break;
852 
853       case Token.BREAK:
854         Preconditions.checkState(childCount <= 1);
855         add("break");
856         if (childCount == 1) {
857           if (!first.isLabelName()) {
858             throw new Error("Unexpected token type. Should be LABEL_NAME.");
859           }
860           add(" ");
861           add(first);
862         }
863         cc.endStatement();
864         break;
865 
866       case Token.EXPR_RESULT:
867         Preconditions.checkState(childCount == 1);
868         add(first, Context.START_OF_EXPR);
869         cc.endStatement();
870         break;
871 
872       case Token.NEW:
873         add("new ");
874         int precedence = NodeUtil.precedence(type);
875 
876         // If the first child contains a CALL, then claim higher precedence
877         // to force parentheses. Otherwise, when parsed, NEW will bind to the
878         // first viable parentheses (don't traverse into functions).
879         if (NodeUtil.containsType(first, Token.CALL, NodeUtil.MATCH_NOT_FUNCTION)) {
880           precedence = NodeUtil.precedence(first.getType()) + 1;
881         }
882         addExpr(first, precedence, Context.OTHER);
883 
884         // '()' is optional when no arguments are present
885         Node next = first.getNext();
886         if (next != null) {
887           add("(");
888           addList(next);
889           add(")");
890         }
891         break;
892 
893       case Token.STRING_KEY:
894         addStringKey(n);
895         break;
896 
897       case Token.STRING:
898         Preconditions.checkState(childCount == 0, "A string may not have children");
899         // The string is already processed, don't escape it.
900         if (n.getBooleanProp(Node.COOKED_STRING)) {
901           add("\"" + n.getString() + "\"");
902         } else {
903           addJsString(n);
904         }
905         break;
906 
907       case Token.DELPROP:
908         Preconditions.checkState(childCount == 1);
909         add("delete ");
910         add(first);
911         break;
912 
913       case Token.OBJECTLIT:
914         {
915           boolean needsParens = (context == Context.START_OF_EXPR);
916           if (needsParens) {
917             add("(");
918           }
919           add("{");
920           for (Node c = first; c != null; c = c.getNext()) {
921             if (c != first) {
922               cc.listSeparator();
923             }
924 
925             Preconditions.checkState(
926                 c.isComputedProp()
927                     || c.isGetterDef()
928                     || c.isSetterDef()
929                     || c.isStringKey()
930                     || c.isMemberFunctionDef());
931             add(c);
932           }
933           add("}");
934           if (needsParens) {
935             add(")");
936           }
937           break;
938         }
939 
940       case Token.COMPUTED_PROP:
941         maybeAddAccessibilityModifier(n);
942         if (n.getBooleanProp(Node.COMPUTED_PROP_GETTER)) {
943           add("get ");
944         } else if (n.getBooleanProp(Node.COMPUTED_PROP_SETTER)) {
945           add("set ");
946         } else if (last.getBooleanProp(Node.GENERATOR_FN)) {
947           add("*");
948         }
949         add("[");
950         add(first);
951         add("]");
952         // TODO(martinprobst): There's currently no syntax for properties in object literals that
953         // have type declarations on them (a la `{foo: number: 12}`). This comes up for, e.g.,
954         // function parameters with default values. Support when figured out.
955         maybeAddTypeDecl(n);
956         if (n.getBooleanProp(Node.COMPUTED_PROP_METHOD)
957             || n.getBooleanProp(Node.COMPUTED_PROP_GETTER)
958             || n.getBooleanProp(Node.COMPUTED_PROP_SETTER)) {
959           Node function = first.getNext();
960           Node params = function.getFirstChild().getNext();
961           Node body = function.getLastChild();
962 
963           add(params);
964           add(body, Context.PRESERVE_BLOCK);
965         } else {
966           // This is a field or object literal property.
967           boolean isInClass = n.getParent().getType() == Token.CLASS_MEMBERS;
968           Node initializer = first.getNext();
969           if (initializer != null) {
970             // Object literal value.
971             Preconditions.checkState(
972                 !isInClass, "initializers should only exist in object literals, not classes");
973             cc.addOp(":", false);
974             add(initializer);
975           } else {
976             // Computed properties must either have an initializer or be computed member-variable
977             // properties that exist for their type declaration.
978             Preconditions.checkState(n.getBooleanProp(Node.COMPUTED_PROP_VARIABLE));
979           }
980         }
981         break;
982 
983       case Token.OBJECT_PATTERN:
984         addObjectPattern(n);
985         maybeAddTypeDecl(n);
986         break;
987 
988       case Token.SWITCH:
989         add("switch(");
990         add(first);
991         add(")");
992         cc.beginBlock();
993         addAllSiblings(first.getNext());
994         cc.endBlock(context == Context.STATEMENT);
995         break;
996 
997       case Token.CASE:
998         Preconditions.checkState(childCount == 2);
999         add("case ");
1000         add(first);
1001         addCaseBody(last);
1002         break;
1003 
1004       case Token.DEFAULT_CASE:
1005         Preconditions.checkState(childCount == 1);
1006         add("default");
1007         addCaseBody(first);
1008         break;
1009 
1010       case Token.LABEL:
1011         Preconditions.checkState(childCount == 2);
1012         if (!first.isLabelName()) {
1013           throw new Error("Unexpected token type. Should be LABEL_NAME.");
1014         }
1015         add(first);
1016         add(":");
1017         if (!last.isBlock()) {
1018           cc.maybeInsertSpace();
1019         }
1020         addNonEmptyStatement(last, getContextForNonEmptyExpression(context), true);
1021         break;
1022 
1023       case Token.CAST:
1024         add("(");
1025         add(first);
1026         add(")");
1027         break;
1028 
1029       case Token.TAGGED_TEMPLATELIT:
1030         add(first, Context.START_OF_EXPR);
1031         add(first.getNext());
1032         break;
1033 
1034       case Token.TEMPLATELIT:
1035         add("`");
1036         for (Node c = first; c != null; c = c.getNext()) {
1037           if (c.isString()) {
1038             add(c.getString());
1039           } else {
1040             // Can't use add() since isWordChar('$') == true and cc would add
1041             // an extra space.
1042             cc.append("${");
1043             add(c.getFirstChild(), Context.START_OF_EXPR);
1044             add("}");
1045           }
1046         }
1047         add("`");
1048         break;
1049 
1050         // Type Declaration ASTs.
1051       case Token.STRING_TYPE:
1052         add("string");
1053         break;
1054       case Token.BOOLEAN_TYPE:
1055         add("boolean");
1056         break;
1057       case Token.NUMBER_TYPE:
1058         add("number");
1059         break;
1060       case Token.ANY_TYPE:
1061         add("any");
1062         break;
1063       case Token.VOID_TYPE:
1064         add("void");
1065         break;
1066       case Token.NAMED_TYPE:
1067         // Children are a chain of getprop nodes.
1068         add(first);
1069         break;
1070       case Token.ARRAY_TYPE:
1071         addExpr(first, NodeUtil.precedence(Token.ARRAY_TYPE), context);
1072         add("[]");
1073         break;
1074       case Token.FUNCTION_TYPE:
1075         Node returnType = first;
1076         add("(");
1077         addList(first.getNext());
1078         add(")");
1079         cc.addOp("=>", true);
1080         add(returnType);
1081         break;
1082       case Token.UNION_TYPE:
1083         addList(first, "|");
1084         break;
1085       case Token.RECORD_TYPE:
1086         add("{");
1087         addList(first, false, Context.OTHER, ",");
1088         add("}");
1089         break;
1090       case Token.PARAMETERIZED_TYPE:
1091         // First child is the type that's parameterized, later children are the arguments.
1092         add(first);
1093         add("<");
1094         addList(first.getNext());
1095         add(">");
1096         break;
1097         // CLASS -> NAME,EXPR|EMPTY,BLOCK
1098       case Token.GENERIC_TYPE_LIST:
1099         add("<");
1100         addList(first, false, Context.STATEMENT, ",");
1101         add(">");
1102         break;
1103       case Token.GENERIC_TYPE:
1104         addIdentifier(n.getString());
1105         if (n.hasChildren()) {
1106           add("extends");
1107           cc.maybeInsertSpace();
1108           add(n.getFirstChild());
1109         }
1110         break;
1111       case Token.INTERFACE:
1112         {
1113           Preconditions.checkState(childCount == 3);
1114           Node name = first;
1115           Node superTypes = first.getNext();
1116           Node members = last;
1117 
1118           add("interface");
1119           add(name);
1120           maybeAddGenericTypes(name);
1121           if (!superTypes.isEmpty()) {
1122             add("extends");
1123             Node superType = superTypes.getFirstChild();
1124             add(superType);
1125             while ((superType = superType.getNext()) != null) {
1126               add(",");
1127               cc.maybeInsertSpace();
1128               add(superType);
1129             }
1130           }
1131           add(members);
1132         }
1133         break;
1134       case Token.ENUM:
1135         {
1136           Preconditions.checkState(childCount == 2);
1137           Node name = first;
1138           Node members = last;
1139           add("enum");
1140           add(name);
1141           add(members);
1142           break;
1143         }
1144       case Token.NAMESPACE: {
1145         Preconditions.checkState(childCount == 2);
1146         Node name = first;
1147         Node elements = last;
1148         add("namespace");
1149         add(name);
1150         add(elements);
1151         break;
1152       }
1153       case Token.TYPE_ALIAS:
1154         add("type");
1155         add(n.getString());
1156         cc.addOp("=", true);
1157         add(last);
1158         cc.endStatement(true);
1159         break;
1160       case Token.DECLARE:
1161         add("declare");
1162         add(first);
1163         processEnd(n, context);
1164         break;
1165       case Token.INDEX_SIGNATURE:
1166         add("[");
1167         add(first);
1168         add("]");
1169         maybeAddTypeDecl(n);
1170         cc.endStatement(true);
1171         break;
1172       case Token.CALL_SIGNATURE:
1173         if (n.getBooleanProp(Node.CONSTRUCT_SIGNATURE)) {
1174           add("new ");
1175         }
1176         maybeAddGenericTypes(n);
1177         add(first);
1178         maybeAddTypeDecl(n);
1179         cc.endStatement(true);
1180         break;
1181       default:
1182         throw new RuntimeException("Unknown type " + Token.name(type) + "\n" + n.toStringTree());
1183     }
1184 
1185     cc.endSourceMapping(n);
1186   }

1188   private boolean arrowFunctionNeedsParens(Node parent, Context context) {
1189     if (parent == null) {
1190       return false;
1191     }
1192     switch (parent.getType()) {
1193       case Token.EXPR_RESULT:
1194       case Token.COMMA:
1195         // Arrow function bodies bind more tightly than commas, and need no parens in that case.
1196         return false;
1197       default:
1198         return (context == Context.START_OF_EXPR);
1199     }

1286   /**
1287    * We could use addList recursively here, but sometimes we produce
1288    * very deeply nested operators and run out of stack space, so we
1289    * just unroll the recursion when possible.
1290    *
1291    * We assume nodes are left-recursive.
1292    */
1293   private void unrollBinaryOperator(
1294       Node n, int op, String opStr, Context context,
1295       Context rhsContext, int leftPrecedence, int rightPrecedence) {
1296     Node firstNonOperator = n.getFirstChild();
1297     while (firstNonOperator.getType() == op) {
1298       firstNonOperator = firstNonOperator.getFirstChild();
1299     }
1300 
1301     addExpr(firstNonOperator, leftPrecedence, context);
1302 
1303     Node current = firstNonOperator;
1304     do {
1305       current = current.getParent();
1306       cc.addOp(opStr, true);
1307       addExpr(current.getFirstChild().getNext(), rightPrecedence, rhsContext);
1308     } while (current != n);
1309   }

1311   static boolean isSimpleNumber(String s) {
1312     int len = s.length();
1313     if (len == 0) {
1314       return false;
1315     }
1316     for (int index = 0; index < len; index++) {
1317       char c = s.charAt(index);
1318       if (c < '0' || c > '9') {
1319         return false;
1320       }
1321     }
1322     return len == 1 || s.charAt(0) != '0';
1323   }

1325   static double getSimpleNumber(String s) {
1326     if (isSimpleNumber(s)) {
1327       try {
1328         long l = Long.parseLong(s);
1329         if (l < NodeUtil.MAX_POSITIVE_INTEGER_NUMBER) {
1330           return l;
1331         }
1332       } catch (NumberFormatException e) {
1333         // The number was too long to parse. Fall through to NaN.
1334       }
1335     }
1336     return Double.NaN;
1337   }

1347   /**
1348    * Adds a block or expression, substituting a VOID with an empty statement.
1349    * This is used for "for (...);" and "if (...);" type statements.
1350    *
1351    * @param n The node to print.
1352    * @param context The context to determine how the node should be printed.
1353    */
1354   private void addNonEmptyStatement(
1355       Node n, Context context, boolean allowNonBlockChild) {
1356     Node nodeToProcess = n;
1357 
1358     if (!allowNonBlockChild && !n.isBlock()) {
1359       throw new Error("Missing BLOCK child.");
1360     }
1361 
1362     // Strip unneeded blocks, that is blocks with <2 children unless
1363     // the CodePrinter specifically wants to keep them.
1364     if (n.isBlock()) {
1365       int count = getNonEmptyChildCount(n, 2);
1366       if (count == 0) {
1367         if (cc.shouldPreserveExtraBlocks()) {
1368           cc.beginBlock();
1369           cc.endBlock(cc.breakAfterBlockFor(n, context == Context.STATEMENT));
1370         } else {
1371           cc.endStatement(true);
1372         }
1373         return;
1374       }
1375 
1376       if (count == 1) {
1377         // Hack around a couple of browser bugs:
1378         //   Safari needs a block around function declarations.
1379         //   IE6/7 needs a block around DOs.
1380         Node firstAndOnlyChild = getFirstNonEmptyChild(n);
1381         boolean alwaysWrapInBlock = cc.shouldPreserveExtraBlocks();
1382         if (alwaysWrapInBlock || isBlockDeclOrDo(firstAndOnlyChild)) {
1383           cc.beginBlock();
1384           add(firstAndOnlyChild, Context.STATEMENT);
1385           cc.maybeLineBreak();
1386           cc.endBlock(cc.breakAfterBlockFor(n, context == Context.STATEMENT));
1387           return;
1388         } else {
1389           // Continue with the only child.
1390           nodeToProcess = firstAndOnlyChild;
1391         }
1392       }
1393 
1394       if (count > 1) {
1395         context = Context.PRESERVE_BLOCK;
1396       }
1397     }
1398 
1399     if (nodeToProcess.isEmpty()) {
1400       cc.endStatement(true);
1401     } else {
1402       add(nodeToProcess, context);
1403 
1404       // VAR doesn't include ';' since it gets used in expressions - so any
1405       // VAR in a statement context needs a call to endStatement() here.
1406       if (nodeToProcess.isVar()) {
1407         cc.endStatement();
1408       }
1409     }
1410   }

1412   /**
1413    * @return Whether the Node is a DO or a declaration that is only allowed
1414    * in restricted contexts.
1415    */
1416   private static boolean isBlockDeclOrDo(Node n) {
1417     if (n.isLabel()) {
1418       Node labeledStatement = n.getLastChild();
1419       if (!labeledStatement.isBlock()) {
1420         return isBlockDeclOrDo(labeledStatement);
1421       } else {
1422         // For labels with block children, we need to ensure that a
1423         // labeled FUNCTION or DO isn't generated when extraneous BLOCKs
1424         // are skipped.
1425         if (getNonEmptyChildCount(n, 2) == 1) {
1426           return isBlockDeclOrDo(getFirstNonEmptyChild(n));
1427         } else {
1428           // Either a empty statement or an block with more than one child,
1429           // way it isn't a FUNCTION or DO.
1430           return false;
1431         }
1432       }
1433     } else {
1434       switch (n.getType()){
1435         case Token.LET:
1436         case Token.CONST:
1437         case Token.FUNCTION:
1438         case Token.CLASS:
1439         case Token.DO:
1440           return true;
1441         default:
1442           return false;
1443       }
1444     }
1445   }

1450       add("(");

1452       add(")");
1453     } else {
1454       add(n, context);

1480   void addStringKey(Node n) {
1481     String key = n.getString();
1482     // Object literal property names don't have to be quoted if they
1483     // are not JavaScript keywords
1484     if (!n.isQuotedString()
1485         && !(languageMode == LanguageMode.ECMASCRIPT3
1486             && TokenStream.isKeyword(key))
1487         && TokenStream.isJSIdentifier(key)
1488         // do not encode literally any non-literal characters that
1489         // were Unicode escaped.
1490         && NodeUtil.isLatin(key)) {
1491       add(key);
1492     } else {
1493       // Determine if the string is a simple number.
1494       double d = getSimpleNumber(key);
1495       if (!Double.isNaN(d)) {
1496         cc.addNumber(d);
1497       } else {
1498         addJsString(n);
1499       }
1500     }
1501     if (n.hasChildren()) {
1502       add(":");
1503       addExpr(n.getFirstChild(), 1, Context.OTHER);
1504     }
1505   }

1507   /**
1508    * Determines whether the given child of a destructuring pattern is the initializer for
1509    * that pattern. If the pattern is in a var/let/const statement, then the last child of the
1510    * pattern is the initializer, e.g. the tree for
1511    * {@code var {x:y} = z} looks like:
1512    * <pre>
1513    * VAR
1514    *   OBJECT_PATTERN
1515    *     STRING_KEY x
1516    *       NAME y
1517    *     NAME z
1518    * </pre>
1519    * The exception is when the var/let/const is the first child of a for-in or for-of loop, in
1520    * which case all the children belong to the pattern itself, e.g. the VAR node in
1521    * {@code for (var {x: y, z} of []);} looks like
1522    * <pre>
1523    * VAR
1524    *   OBJECT_PATTERN
1525    *     STRING_KEY x
1526    *       NAME y
1527    *     STRING_KEY z
1528    * </pre>
1529    * and the "z" node is *not* an initializer.
1530    */
1531   private boolean isPatternInitializer(Node n) {
1532     Node parent = n.getParent();
1533     Preconditions.checkState(parent.isDestructuringPattern());
1534     if (n != parent.getLastChild()) {
1535       return false;
1536     }
1537     Node decl = parent.getParent();
1538 
1539     if (!NodeUtil.isNameDeclaration(decl)) {
1540       return false;
1541     }
1542     if (NodeUtil.isEnhancedFor(decl.getParent()) && decl == decl.getParent().getFirstChild()) {
1543       return false;
1544     }
1545     return true;
1546   }

1622   /** Outputs a JS string, using the optimal (single/double) quote character */
1623   private void addJsString(Node n) {
1624     String s = n.getString();
1625     boolean useSlashV = n.getBooleanProp(Node.SLASH_V);
1626     if (useSlashV) {
1627       add(jsString(n.getString(), useSlashV));
1628     } else {
1629       String cached = escapedJsStrings.get(s);
1630       if (cached == null) {
1631         cached = jsString(n.getString(), useSlashV);
1632         escapedJsStrings.put(s, cached);
1633       }
1634       add(cached);
1635     }
1636   }

1638   private String jsString(String s, boolean useSlashV) {
1639     int singleq = 0, doubleq = 0;
1640 
1641     // could count the quotes and pick the optimal quote character
1642     for (int i = 0; i < s.length(); i++) {
1643       switch (s.charAt(i)) {
1644         case '"': doubleq++; break;
1645         case '\'': singleq++; break;
1646       }
1647     }
1648 
1649     String doublequote, singlequote;
1650     char quote;
1651     if (preferSingleQuotes ?
1652         (singleq <= doubleq) : (singleq < doubleq)) {
1653       // more double quotes so enclose in single quotes.
1654       quote = '\'';
1655       doublequote = "\"";
1656       singlequote = "\\\'";
1657     } else {
1658       // more single quotes so escape the doubles
1659       quote = '\"';
1660       doublequote = "\\\"";
1661       singlequote = "\'";
1662     }
1663 
1664     return strEscape(s, quote, doublequote, singlequote, "\\\\",
1665         outputCharsetEncoder, useSlashV, false);
1666   }

1680   /** Helper to escape JavaScript string as well as regular expression */
1681   private String strEscape(
1682       String s,
1683       char quote,
1684       String doublequoteEscape,
1685       String singlequoteEscape,
1686       String backslashEscape,
1687       OutputCharsetEncoder outputCharsetEncoder,
1688       boolean useSlashV,
1689       boolean isRegexp) {
1690     StringBuilder sb = new StringBuilder(s.length() + 2);
1691     sb.append(quote);
1692     for (int i = 0; i < s.length(); i++) {
1693       char c = s.charAt(i);
1694       switch (c) {
1695         case '\0': sb.append("\\x00"); break;
1696         case '\u000B':
1697           if (useSlashV) {
1698             sb.append("\\v");
1699           } else {
1700             sb.append("\\x0B");
1701           }
1702           break;
1703         // From the SingleEscapeCharacter grammar production.
1704         case '\b': sb.append("\\b"); break;
1705         case '\f': sb.append("\\f"); break;
1706         case '\n': sb.append("\\n"); break;
1707         case '\r': sb.append("\\r"); break;
1708         case '\t': sb.append("\\t"); break;
1709         case '\\': sb.append(backslashEscape); break;
1710         case '\"': sb.append(doublequoteEscape); break;
1711         case '\'': sb.append(singlequoteEscape); break;
1712 
1713         // From LineTerminators (ES5 Section 7.3, Table 3)
1714         case '\u2028': sb.append("\\u2028"); break;
1715         case '\u2029': sb.append("\\u2029"); break;
1716 
1717         case '=':
1718           // '=' is a syntactically signficant regexp character.
1719           if (trustedStrings || isRegexp) {
1720             sb.append(c);
1721           } else {
1722             sb.append("\\x3d");
1723           }
1724           break;
1725 
1726         case '&':
1727           if (trustedStrings || isRegexp) {
1728             sb.append(c);
1729           } else {
1730             sb.append("\\x26");
1731           }
1732           break;
1733 
1734         case '>':
1735           if (!trustedStrings && !isRegexp) {
1736             sb.append(GT_ESCAPED);
1737             break;
1738           }
1739 
1740           // Break --> into --\> or ]]> into ]]\>
1741           //
1742           // This is just to prevent developers from shooting themselves in the
1743           // foot, and does not provide the level of security that you get
1744           // with trustedString == false.
1745           if (i >= 2 &&
1746               ((s.charAt(i - 1) == '-' && s.charAt(i - 2) == '-') ||
1747                (s.charAt(i - 1) == ']' && s.charAt(i - 2) == ']'))) {
1748             sb.append(GT_ESCAPED);
1749           } else {
1750             sb.append(c);
1751           }
1752           break;
1753         case '<':
1754           if (!trustedStrings && !isRegexp) {
1755             sb.append(LT_ESCAPED);
1756             break;
1757           }
1758 
1759           // Break </script into <\/script
1760           // As above, this is just to prevent developers from doing this
1761           // accidentally.
1762           final String endScript = "/script";
1763 
1764           // Break <!-- into <\!--
1765           final String startComment = "!--";
1766 
1767           if (s.regionMatches(true, i + 1, endScript, 0,
1768                               endScript.length())) {
1769             sb.append(LT_ESCAPED);
1770           } else if (s.regionMatches(false, i + 1, startComment, 0,
1771                                      startComment.length())) {
1772             sb.append(LT_ESCAPED);
1773           } else {
1774             sb.append(c);
1775           }
1776           break;
1777         default:
1778           if (outputCharsetEncoder != null && outputCharsetEncoder.canEncode(c)
1779               || c > 0x1f && c < 0x7f) {
1780             // If we're given an outputCharsetEncoder, then check if the character can be
1781             // represented in this character set. If no charsetEncoder provided - pass straight
1782             // Latin characters through, and escape the rest. Doing the explicit character check is
1783             // measurably faster than using the CharsetEncoder.
1784             sb.append(c);
1785           } else {
1786             // Other characters can be misinterpreted by some JS parsers,
1787             // or perhaps mangled by proxies along the way,
1788             // so we play it safe and Unicode escape them.
1789             Util.appendHexJavaScriptRepresentation(sb, c);
1790           }
1791       }
1792     }
1793     sb.append(quote);
1794     return sb.toString();
1795   }

1797   static String identifierEscape(String s) {
1798     // First check if escaping is needed at all -- in most cases it isn't.
1799     if (NodeUtil.isLatin(s)) {
1800       return s;
1801     }
1802 
1803     // Now going through the string to escape non-Latin characters if needed.
1804     StringBuilder sb = new StringBuilder();
1805     for (int i = 0; i < s.length(); i++) {
1806       char c = s.charAt(i);
1807       // Identifiers should always go to Latin1/ ASCII characters because
1808       // different browser's rules for valid identifier characters are
1809       // crazy.
1810       if (c > 0x1F && c < 0x7F) {
1811         sb.append(c);
1812       } else {
1813         Util.appendHexJavaScriptRepresentation(sb, c);
1814       }
1815     }
1816     return sb.toString();
1817   }

1819    * @param maxCount The maximum number of children to look for.
1820    * @return The number of children of this node that are non empty up to
1821    * maxCount.

1823   private static int getNonEmptyChildCount(Node n, int maxCount) {
1824     int i = 0;
1825     Node c = n.getFirstChild();
1826     for (; c != null && i < maxCount; c = c.getNext()) {
1827       if (c.isBlock()) {
1828         i += getNonEmptyChildCount(c, maxCount - i);
1829       } else if (!c.isEmpty()) {
1830         i++;

1851   // Information on the current context. Used for disambiguating special cases.
1852   // For example, a "{" could indicate the start of an object literal or a
1853   // block, depending on the current context.
1854   enum Context {
1855     STATEMENT,
1856     BEFORE_DANGLING_ELSE, // a hack to resolve the else-clause ambiguity
1857     START_OF_EXPR,
1858     PRESERVE_BLOCK,
1859     // Are we inside the init clause of a for loop?  If so, the containing
1860     // expression can't contain an in operator.  Pass this context flag down
1861     // until we reach expressions which no longer have the limitation.
1862     IN_FOR_INIT_CLAUSE,
1863     OTHER
1864   }
1865 
1866   private static Context getContextForNonEmptyExpression(Context currentContext) {
1867     return currentContext == Context.BEFORE_DANGLING_ELSE ?
1868         Context.BEFORE_DANGLING_ELSE : Context.OTHER;
1869   }

1880   private void processEnd(Node n, Context context) {
1881     switch (n.getType()) {
1882       case Token.CLASS:
1883       case Token.INTERFACE:
1884       case Token.ENUM:
1885       case Token.NAMESPACE:
1886         cc.endClass(context == Context.STATEMENT);
1887         break;
1888       case Token.FUNCTION:
1889         if (n.getLastChild().isEmpty()) {
1890           cc.endStatement(true);
1891         } else {
1892           cc.endFunction(context == Context.STATEMENT);
1893         }
1894         break;
1895       case Token.DECLARE:
1896         if (n.getParent().getType() != Token.NAMESPACE_ELEMENTS) {
1897           processEnd(n.getFirstChild(), context);
1898         }
1899         break;
1900       case Token.EXPORT:
1901         if (n.getParent().getType() != Token.NAMESPACE_ELEMENTS
1902             && n.getFirstChild().getType() != Token.DECLARE) {
1903           processEnd(n.getFirstChild(), context);
1904         }
1905         break;
1906       case Token.COMPUTED_PROP:
1907         if (n.hasOneChild()) {
1908           cc.endStatement(true);
1909         }
1910         break;
1911       case Token.MEMBER_FUNCTION_DEF:
1912       case Token.GETTER_DEF:
1913       case Token.SETTER_DEF:
1914         if (n.getFirstChild().getLastChild().isEmpty()) {
1915           cc.endStatement(true);
1916         }
1917         break;
1918       case Token.MEMBER_VARIABLE_DEF:
1919         cc.endStatement(true);
1920         break;
1921       default:
1922         if (context == Context.STATEMENT) {
1923           cc.endStatement();
1924         }
1925     }
1926   }
