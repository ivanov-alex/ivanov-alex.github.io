21  /**
22   * An abstract syntax tree for JavaScript parse trees.
23   * Immutable.
24   * A plain old data structure. Should include data members and simple accessors only.
25   *
26   * Derived classes should have a 'Tree' suffix. Each concrete derived class should have a
27   * ParseTreeType whose name matches the derived class name.
28   *
29   * A parse tree derived from source should have a non-null location. A parse tree that is
30   * synthesized by the compiler may have a null location.
31   *
32   * When adding a new subclass of ParseTree you must also do the following:
33   *   - add a new entry to ParseTreeType
34   *   - add ParseTree.asXTree()
35   */
36  public class ParseTree {
37    public final ParseTreeType type;
38    public final SourceRange location;
39  
40    protected ParseTree(ParseTreeType type, SourceRange location) {
41      this.type = type;
42      this.location = location;
43    }

160   public boolean isPattern() {
161     ParseTree parseTree = this;
162     while (parseTree.type == ParseTreeType.PAREN_EXPRESSION) {
163       parseTree = parseTree.asParenExpression().expression;
164     }
165 
166     switch (parseTree.type) {
167       case ARRAY_PATTERN:
168       case OBJECT_PATTERN:
169         return true;
170       default:
171         return false;
172     }
173   }
174 
175   public boolean isValidAssignmentTarget() {
176     ParseTree parseTree = this;
177     while (parseTree.type == ParseTreeType.PAREN_EXPRESSION) {
178       parseTree = parseTree.asParenExpression().expression;
179     }
180 
181     switch(parseTree.type) {
182       case IDENTIFIER_EXPRESSION:
183       case MEMBER_EXPRESSION:
184       case MEMBER_LOOKUP_EXPRESSION:
185       case ARRAY_PATTERN:
186       case OBJECT_PATTERN:
187       case DEFAULT_PARAMETER:
188         return true;
189       default:
190         return false;
191     }
192   }

196     switch (this.type) {
197     case FUNCTION_DECLARATION:
198     case BINARY_OPERATOR:
199     case THIS_EXPRESSION:
200     case IDENTIFIER_EXPRESSION:
201     case LITERAL_EXPRESSION:
202     case ARRAY_LITERAL_EXPRESSION:
203     case OBJECT_LITERAL_EXPRESSION:
204     case MISSING_PRIMARY_EXPRESSION:
205     case CONDITIONAL_EXPRESSION:
206     case UNARY_EXPRESSION:
207     case POSTFIX_EXPRESSION:
208     case MEMBER_EXPRESSION:
209     case NEW_EXPRESSION:
210     case CALL_EXPRESSION:
211     case MEMBER_LOOKUP_EXPRESSION:
212     case PAREN_EXPRESSION:
213     case SUPER_EXPRESSION:
214     case TEMPLATE_LITERAL_EXPRESSION:
215       return true;
216     default:
217       return false;
218     }
219   }

229   /**
230    * This function reflects the ECMA standard, or what we would expect to become the ECMA standard.
231    * Most places use isStatement instead which reflects where code on the web diverges from the
232    * standard.
233    */
234   public boolean isStatementStandard() {
235     switch (this.type) {
236     case BLOCK:
237     case VARIABLE_STATEMENT:
238     case EMPTY_STATEMENT:
239     case EXPRESSION_STATEMENT:
240     case IF_STATEMENT:
241     case DO_WHILE_STATEMENT:
242     case WHILE_STATEMENT:
243     case FOR_OF_STATEMENT:
244     case FOR_IN_STATEMENT:
245     case FOR_STATEMENT:
246     case CONTINUE_STATEMENT:
247     case BREAK_STATEMENT:
248     case RETURN_STATEMENT:
249     case YIELD_EXPRESSION:
250     case WITH_STATEMENT:
251     case SWITCH_STATEMENT:
252     case LABELLED_STATEMENT:
253     case THROW_STATEMENT:
254     case TRY_STATEMENT:
255     case DEBUGGER_STATEMENT:
256       return true;
257     default:
258       return false;
259     }
260   }
261 
262   public boolean isSourceElement() {
263     return isStatementStandard() || this.type == ParseTreeType.FUNCTION_DECLARATION;
264   }
