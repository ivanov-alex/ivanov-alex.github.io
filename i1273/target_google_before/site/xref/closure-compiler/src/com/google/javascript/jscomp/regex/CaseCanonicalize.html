21  /**
22   * Implements the ECMAScript 5
23   * <a href="http://es5.github.com/#Canonicalize">Canonicalize</a> operation
24   * used to specify how case-insensitive regular expressions match.
25   *
26   * <p>
27   * From section <a href="http://es5.github.com/#x15.10.2.9">15.10.2.9</a>,
28   * <blockquote>
29   * The abstract operation Canonicalize takes a character parameter ch and
30   * performs the following steps:
31   * <ul>
32   *   <li>If IgnoreCase is false, return ch.
33   *   <li>Let u be ch converted to upper case as if by calling the standard
34   *   built-in method {@code String.prototype.toUpperCase} on the one-character
35   *   String ch.
36   *   <li>If u does not consist of a single character, return ch.
37   *   <li>Let cu be u's character.
38   *   <li>If ch's code unit value is greater than or equal to decimal 128 and
39   *   cu's code unit value is less than decimal 128, then return ch.
40   *   <li>Return cu.
41   * </ul>
42   *
43   * @author Mike Samuel <mikesamuel@gmail.com>
44   */
45  public final class CaseCanonicalize {
46  
47    private CaseCanonicalize() {
48      // Uninstantiable.
49    }

187   /**
188    * Returns the case canonical version of the given string.
189    */
190   public static String caseCanonicalize(String s) {
191     for (int i = 0, n = s.length(); i < n; ++i) {
192       char ch = s.charAt(i);
193       char cu = caseCanonicalize(ch);
194       if (cu != ch) {
195         StringBuilder sb = new StringBuilder(s);
196         sb.setCharAt(i, cu);
197         while (++i < n) {
198           sb.setCharAt(i, caseCanonicalize(s.charAt(i)));
199         }
200         return sb.toString();
201       }
202     }
203     return s;
204   }

206   /**
207    * Returns the case canonical version of the given code-unit.  ECMAScript 5
208    * explicitly says that code-units are to be treated as their code-point
209    * equivalent, even surrogates.
210    */
211   public static char caseCanonicalize(char ch) {
212     if (ch < 0x80) {  // Normal case.
213       return ('A' <= ch && ch <= 'Z') ? (char) (ch | 32) : ch;
214     }
215     // Non-ASCII case.
216     if (CASE_SENSITIVE.contains(ch)) {
217       for (DeltaSet ds : CANON_DELTA_SETS) {
218         if (ds.codeUnits.contains(ch)) {
219           return (char) (ch - ds.delta);
220         }
221       }
222     }
223     return ch;
224   }

226   /**
227    * Given a character range that may include case sensitive code-units,
228    * such as {@code [0-9B-M]}, returns the character range that includes all
229    * the code-units in the input and those that are case-insensitively
230    * equivalent to a code-unit in the input.
231    */
232   public static CharRanges expandToAllMatched(CharRanges ranges) {
233     CharRanges caseSensitive = ranges.intersection(CASE_SENSITIVE);
234     if (caseSensitive.isEmpty()) { return ranges; }
235     CharRanges expanded = CharRanges.EMPTY;
236     for (DeltaSet ds : DELTA_SETS) {
237       expanded = expanded.union(
238           caseSensitive.intersection(ds.codeUnits).shift(-ds.delta));
239     }
240     return ranges.union(expanded);
241   }
242 
243 
244   private static final CharRanges UCASE_ASCII_LETTERS
245       = CharRanges.inclusive('A', 'Z');
246 
247   /**
248    * Given a character range that may include case sensitive code-units,
249    * such as {@code [0-9B-M]}, returns the character range that includes
250    * the minimal set of code units such that for every code unit in the
251    * input there is a case-sensitively equivalent canonical code unit in the
252    * output.
253    */
254   public static CharRanges reduceToMinimum(CharRanges ranges) {
255     CharRanges caseSensitive = ranges.intersection(CASE_SENSITIVE);
256     if (caseSensitive.isEmpty()) { return ranges; }
257     CharRanges expanded = CharRanges.EMPTY;
258     for (DeltaSet ds : CANON_DELTA_SETS) {
259       expanded = expanded.union(
260           caseSensitive.intersection(ds.codeUnits).shift(-ds.delta));
261     }
262     // Letters a-z gzip better than uppercase A-Z since JavaScript keywords
263     // are lower-case, so, even though the definition of Canonicalize is
264     // based on String.prototype.toUpperCase, we use lowercase ASCII characters
265     // in the minimal form.
266     expanded = expanded.difference(UCASE_ASCII_LETTERS).union(
267         expanded.intersection(UCASE_ASCII_LETTERS).shift(32));
268     return ranges.difference(caseSensitive).union(expanded);
269   }
