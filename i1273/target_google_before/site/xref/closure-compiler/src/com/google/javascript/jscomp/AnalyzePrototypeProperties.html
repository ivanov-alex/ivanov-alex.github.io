38  /**
39   * Analyzes properties on prototypes.
40   *
41   * Uses a reference graph to analyze prototype properties. Each unique property
42   * name is represented by a node in this graph. An edge from property A to
43   * property B means that there's a GETPROP access of a property B on some
44   * object inside of a method named A.
45   *
46   * Global functions are also represented by nodes in this graph, with
47   * similar semantics.
48   *
49   * @author nicksantos@google.com (Nick Santos)
50   */
51  class AnalyzePrototypeProperties implements CompilerPass {
52  
53    // Constants for symbol types, for easier readability.
54    private static final SymbolType PROPERTY = SymbolType.PROPERTY;
55    private static final SymbolType VAR = SymbolType.VAR;
56  
57    private final AbstractCompiler compiler;
58    private final boolean canModifyExterns;
59    private final boolean anchorUnusedVars;
60    private final JSModuleGraph moduleGraph;
61    private final JSModule firstModule;
62  
63    // Properties that are implicitly used as part of the JS language.
64    private static final Set<String> IMPLICITLY_USED_PROPERTIES =
65        ImmutableSet.of("length", "toString", "valueOf");
66  
67    // A graph where the nodes are property names or variable names,
68    // and the edges signify the modules where the property is referenced.
69    // For example, if we had the code:
70    //
71    // Foo.prototype.bar = function(x) { x.baz(); }; // in module 2.;
72    //
73    // then this would be represented in the graph by a node representing
74    // "bar", a node representing "baz", and an edge between them representing
75    // module #2.
76    //
77    // Similarly, if we had:
78    //
79    // var scotch = function(f) { return f.age(); };
80    //
81    // then there would be a node for "scotch", a node for "age", and an edge
82    // from scotch to age.
83    private final LinkedDirectedGraph<NameInfo, JSModule> symbolGraph =
84        LinkedDirectedGraph.createWithoutAnnotations();
85  
86    // A dummy node for representing global references.
87    private final NameInfo globalNode = new NameInfo("[global]");
88  
89    // A dummy node for representing extern references.
90    private final NameInfo externNode = new NameInfo("[extern]");
91  
92    // A dummy node for representing all anonymous functions with no names.
93    private final NameInfo anonymousNode = new NameInfo("[anonymous]");
94  
95    // All the real NameInfo for prototype properties, hashed by the name
96    // of the property that they represent.
97    private final Map<String, NameInfo> propertyNameInfo = new LinkedHashMap<>();
98  
99    // All the NameInfo for global functions, hashed by the name of the
100   // global variable that it's assigned to.
101   private final Map<String, NameInfo> varNameInfo = new LinkedHashMap<>();
102 
103   /**
104    * Creates a new pass for analyzing prototype properties.
105    * @param compiler The compiler.
106    * @param moduleGraph The graph for resolving module dependencies. May be
107    *     null if we don't care about module dependencies.
108    * @param canModifyExterns If true, then we can move prototype
109    *     properties that are declared in the externs file.
110    * @param anchorUnusedVars If true, then we must mark all vars as referenced,
111    *     even if they are never used.
112    */
113   AnalyzePrototypeProperties(AbstractCompiler compiler,
114       JSModuleGraph moduleGraph, boolean canModifyExterns,
115       boolean anchorUnusedVars) {
116     this.compiler = compiler;
117     this.moduleGraph = moduleGraph;
118     this.canModifyExterns = canModifyExterns;
119     this.anchorUnusedVars = anchorUnusedVars;
120 
121     if (moduleGraph != null) {
122       firstModule = moduleGraph.getRootModule();
123     } else {
124       firstModule = null;
125     }
126 
127     globalNode.markReference(null);
128     externNode.markReference(null);
129     symbolGraph.createNode(globalNode);
130     symbolGraph.createNode(externNode);
131 
132     for (String property : IMPLICITLY_USED_PROPERTIES) {
133       NameInfo nameInfo = getNameInfoForName(property, PROPERTY);
134       if (moduleGraph == null) {
135         symbolGraph.connect(externNode, null, nameInfo);
136       } else {
137         for (JSModule module : moduleGraph.getAllModules()) {
138           symbolGraph.connect(externNode, module, nameInfo);
139         }
140       }
141     }
142   }
143 
144   @Override
145   public void process(Node externRoot, Node root) {
146     if (!canModifyExterns) {
147       NodeTraversal.traverseEs6(compiler, externRoot,
148           new ProcessExternProperties());
149     }
150 
151     NodeTraversal.traverseEs6(compiler, root, new ProcessProperties());
152 
153     FixedPointGraphTraversal<NameInfo, JSModule> t =
154         FixedPointGraphTraversal.newTraversal(new PropagateReferences());
155     t.computeFixedPoint(symbolGraph,
156         ImmutableSet.of(externNode, globalNode));
157   }

159   /**
160    * Returns information on all prototype properties.
161    */
162   public Collection<NameInfo> getAllNameInfo() {
163     List<NameInfo> result = new ArrayList<>(propertyNameInfo.values());
164     result.addAll(varNameInfo.values());
165     return result;
166   }

168   /**
169    * Gets the name info for the property or variable of a given name,
170    * and creates a new one if necessary.
171    *
172    * @param name The name of the symbol.
173    * @param type The type of symbol.
174    */
175   private NameInfo getNameInfoForName(String name, SymbolType type) {
176     Map<String, NameInfo> map = type == PROPERTY ?
177         propertyNameInfo : varNameInfo;
178     if (map.containsKey(name)) {
179       return map.get(name);
180     } else {
181       NameInfo nameInfo = new NameInfo(name);
182       map.put(name, nameInfo);
183       symbolGraph.createNode(nameInfo);
184       return nameInfo;
185     }
186   }

188   private class ProcessProperties implements NodeTraversal.ScopedCallback {
189     // There are two types of context information on this stack:
190     // 1) Every scope has a NameContext corresponding to its scope.
191     //    Variables are given VAR contexts.
192     //    Prototype properties are given PROPERTY contexts.
193     //    The global scope is given the special [global] context.
194     //    And function expressions that we aren't able to give a reasonable
195     //    name are given a special [anonymous] context.
196     // 2) Every assignment of a prototype property of a non-function is
197     //    given a name context. These contexts do not have scopes.
198     private final Stack<NameContext> symbolStack = new Stack<>();
199 
200     @Override
201     public void enterScope(NodeTraversal t) {
202       Node n = t.getCurrentNode();
203       Scope scope = t.getScope();
204       Node root = scope.getRootNode();
205       if (root.isFunction()) {
206         String propName = getPrototypePropertyNameFromRValue(n);
207         if (propName != null) {
208           symbolStack.push(
209               new NameContext(
210                   getNameInfoForName(propName, PROPERTY),
211                   scope));
212         } else if (isGlobalFunctionDeclaration(t, n)) {
213           Node parent = n.getParent();
214           String name = parent.isName() ?
215               parent.getString() /* VAR */ :
216               n.getFirstChild().getString() /* named function */;
217           symbolStack.push(
218               new NameContext(getNameInfoForName(name, VAR), scope.getClosestHoistScope()));
219         } else {
220           // NOTE(nicksantos): We use the same anonymous node for all
221           // functions that do not have reasonable names. I can't remember
222           // at the moment why we do this. I think it's because anonymous
223           // nodes can never have in-edges. They're just there as a placeholder
224           // for scope information, and do not matter in the edge propagation.
225           symbolStack.push(new NameContext(anonymousNode, scope));
226         }
227       } else if (t.inGlobalScope()) {
228         symbolStack.push(new NameContext(globalNode, scope));
229       } else {
230         // TODO(moz): It's not yet clear if we need another kind of NameContext for block scopes
231         // in ES6, use anonymous node for now and investigate later.
232         Preconditions.checkState(NodeUtil.createsBlockScope(root), scope);
233         symbolStack.push(new NameContext(anonymousNode, scope));
234       }
235     }
236 
237     @Override
238     public void exitScope(NodeTraversal t) {
239       symbolStack.pop();
240     }
241 
242     @Override
243     public boolean shouldTraverse(NodeTraversal t, Node n, Node parent) {
244       // Process prototype assignments to non-functions.
245       String propName = processNonFunctionPrototypeAssign(n, parent);
246       if (propName != null) {
247         symbolStack.push(
248             new NameContext(
249                 getNameInfoForName(propName, PROPERTY), null));
250       }
251       return true;
252     }
253 
254     @Override
255     public void visit(NodeTraversal t, Node n, Node parent) {
256       if (n.isGetProp()) {
257         String propName = n.getFirstChild().getNext().getString();
258 
259         if (n.isQualifiedName()) {
260           if (propName.equals("prototype")) {
261             if (processPrototypeRef(t, n)) {
262               return;
263             }
264           } else if (compiler.getCodingConvention().isExported(propName)) {
265             addGlobalUseOfSymbol(propName, t.getModule(), PROPERTY);
266             return;
267           } else {
268             // Do not mark prototype prop assigns as a 'use' in the global scope.
269             if (n.getParent().isAssign() && n.getNext() != null) {
270               String rValueName = getPrototypePropertyNameFromRValue(n);
271               if (rValueName != null) {
272                 return;
273               }
274             }
275           }
276         }
277 
278         addSymbolUse(propName, t.getModule(), PROPERTY);
279       } else if (n.isObjectLit()) {
280         // Make sure that we're not handling object literals being
281         // assigned to a prototype, as in:
282         // Foo.prototype = {bar: 3, baz: 5};
283         String lValueName = NodeUtil.getBestLValueName(
284             NodeUtil.getBestLValue(n));
285         if (lValueName != null && lValueName.endsWith(".prototype")) {
286           return;
287         }
288 
289         // var x = {a: 1, b: 2}
290         // should count as a use of property a and b.
291         for (Node propNameNode = n.getFirstChild(); propNameNode != null;
292              propNameNode = propNameNode.getNext()) {
293           // May be STRING, GET, or SET, but NUMBER isn't interesting.
294           if (!propNameNode.isQuotedString()) {
295             addSymbolUse(propNameNode.getString(), t.getModule(), PROPERTY);
296           }
297         }
298       } else if (n.isName()) {
299         String name = n.getString();
300 
301         Var var = t.getScope().getVar(name);
302         if (var != null) {
303           // Only process global functions.
304           if (var.isGlobal()) {
305             if (var.getInitialValue() != null &&
306                 var.getInitialValue().isFunction()) {
307               if (t.inGlobalHoistScope()) {
308                 if (!processGlobalFunctionDeclaration(t, n, var)) {
309                   addGlobalUseOfSymbol(name, t.getModule(), VAR);
310                 }
311               } else {
312                 addSymbolUse(name, t.getModule(), VAR);
313               }
314             }
315 
316           // If it is not a global, it might be accessing a local of the outer
317           // scope. If that's the case the functions between the variable's
318           // declaring scope and the variable reference scope cannot be moved.
319           } else if (var.getScope() != t.getScope()) {
320             for (int i = symbolStack.size() - 1; i >= 0; i--) {
321               NameContext context = symbolStack.get(i);
322               if (context.scope == var.getScope()) {
323                 break;
324               }
325 
326               context.name.readClosureVariables = true;
327             }
328           }
329         }
330       }
331 
332       // Process prototype assignments to non-functions.
333       if (processNonFunctionPrototypeAssign(n, parent) != null) {
334         symbolStack.pop();
335       }
336     }
337 
338     private void addSymbolUse(String name, JSModule module, SymbolType type) {
339       NameInfo info = getNameInfoForName(name, type);
340       NameInfo def = null;
341       // Skip all anonymous nodes. We care only about symbols with names.
342       for (int i = symbolStack.size() - 1; i >= 0; i--) {
343         def = symbolStack.get(i).name;
344         if (def != anonymousNode) {
345           break;
346         }
347       }
348       if (!def.equals(info)) {
349         symbolGraph.connect(def, module, info);
350       }
351     }

353     /**
354      * If this is a non-function prototype assign, return the prop name.
355      * Otherwise, return null.
356      */
357     private String processNonFunctionPrototypeAssign(Node n, Node parent) {
358       if (isAssignRValue(n, parent) && !n.isFunction()) {
359         return getPrototypePropertyNameFromRValue(n);
360       }
361       return null;
362     }
363 
364     /**
365      * Determines whether {@code n} is the FUNCTION node in a global function
366      * declaration.
367      */
368     private boolean isGlobalFunctionDeclaration(NodeTraversal t, Node n) {
369       // Make sure we're not in a function scope, or if we are then the function we're looking at
370       // is defined in the global scope.
371       if (!(t.inGlobalHoistScope()
372             || n.isFunction() && t.getScopeRoot() == n
373                && t.getScope().getParent().getClosestHoistScope().isGlobal())) {
374         return false;
375       }
376 
377       return NodeUtil.isFunctionDeclaration(n)
378           || n.isFunction() && n.getParent().isName();
379     }
380 
381     /**
382      * Returns true if this is the r-value of an assignment.
383      */
384     private boolean isAssignRValue(Node n, Node parent) {
385       return parent != null && parent.isAssign() && parent.getFirstChild() != n;
386     }
387 
388     /**
389      * Returns the name of a prototype property being assigned to this r-value.
390      *
391      * Returns null if this is not the R-value of a prototype property, or if
392      * the R-value is used in multiple expressions (i.e., if there's
393      * a prototype property assignment in a more complex expression).
394      */
395     private String getPrototypePropertyNameFromRValue(Node rValue) {
396       Node lValue = NodeUtil.getBestLValue(rValue);
397       if (lValue == null ||
398           lValue.getParent() == null ||
399           lValue.getParent().getParent() == null ||
400           !((NodeUtil.isObjectLitKey(lValue) && !lValue.isQuotedString()) ||
401             NodeUtil.isExprAssign(lValue.getParent().getParent()))) {
402         return null;
403       }
404 
405       String lValueName =
406           NodeUtil.getBestLValueName(NodeUtil.getBestLValue(rValue));
407       if (lValueName == null) {
408         return null;
409       }
410       int lastDot = lValueName.lastIndexOf('.');
411       if (lastDot == -1) {
412         return null;
413       }
414 
415       String firstPart = lValueName.substring(0, lastDot);
416       if (!firstPart.endsWith(".prototype")) {
417         return null;
418       }
419 
420       return lValueName.substring(lastDot + 1);
421     }

423     /**
424      * Processes a NAME node to see if it's a global function declaration.
425      * If it is, record it and return true. Otherwise, return false.
426      */
427     private boolean processGlobalFunctionDeclaration(NodeTraversal t,
428         Node nameNode, Var v) {
429       Node firstChild = nameNode.getFirstChild();
430       Node parent = nameNode.getParent();
431 
432       if (// Check for a named FUNCTION.
433           isGlobalFunctionDeclaration(t, parent) ||
434           // Check for a VAR declaration.
435           firstChild != null &&
436           isGlobalFunctionDeclaration(t, firstChild)) {
437         String name = nameNode.getString();
438         getNameInfoForName(name, VAR).getDeclarations().add(
439             new GlobalFunction(nameNode, v, t.getModule()));
440 
441         // If the function name is exported, we should create an edge here
442         // so that it's never removed.
443         if (compiler.getCodingConvention().isExported(name) ||
444             anchorUnusedVars) {
445           addGlobalUseOfSymbol(name, t.getModule(), VAR);
446         }
447 
448         return true;
449       }
450       return false;
451     }

453     /**
454      * Processes the GETPROP of prototype, which can either be under
455      * another GETPROP (in the case of Foo.prototype.bar), or can be
456      * under an assignment (in the case of Foo.prototype = ...).
457      * @return True if a declaration was added.
458      */
459     private boolean processPrototypeRef(NodeTraversal t, Node ref) {
460       Node root = NodeUtil.getRootOfQualifiedName(ref);
461 
462       Node n = ref.getParent();
463       switch (n.getType()) {
464         // Foo.prototype.getBar = function() { ... }
465         case Token.GETPROP:
466           Node dest = n.getFirstChild().getNext();
467           Node parent = n.getParent();
468           Node grandParent = parent.getParent();
469 
470           if (dest.isString() &&
471               NodeUtil.isExprAssign(grandParent) &&
472               NodeUtil.isVarOrSimpleAssignLhs(n, parent)) {
473             String name = dest.getString();
474             Property prop = new AssignmentProperty(
475                 grandParent,
476                 maybeGetVar(t, root),
477                 t.getModule());
478             getNameInfoForName(name, PROPERTY).getDeclarations().add(prop);
479             return true;
480           }
481           break;
482 
483         // Foo.prototype = { "getBar" : function() { ... } }
484         case Token.ASSIGN:
485           Node map = n.getFirstChild().getNext();
486           if (map.isObjectLit()) {
487             for (Node key = map.getFirstChild();
488                  key != null; key = key.getNext()) {
489               if (!key.isQuotedString()) {
490                 // May be STRING, GETTER_DEF, or SETTER_DEF,
491                 String name = key.getString();
492                 Property prop = new LiteralProperty(
493                     key, key.getFirstChild(), map, n,
494                     maybeGetVar(t, root),
495                     t.getModule());
496                 getNameInfoForName(name, PROPERTY).getDeclarations().add(prop);
497               }
498             }
499             return true;
500           }
501           break;
502       }
503       return false;
504     }
505 
506     private Var maybeGetVar(NodeTraversal t, Node maybeName) {
507       return maybeName.isName()
508           ? t.getScope().getVar(maybeName.getString()) : null;
509     }
510 
511     private void addGlobalUseOfSymbol(String name, JSModule module,
512         SymbolType type) {
513       symbolGraph.connect(globalNode, module, getNameInfoForName(name, type));
514     }
515   }

527   private class PropagateReferences
528       implements EdgeCallback<NameInfo, JSModule> {
529     @Override
530     public boolean traverseEdge(NameInfo start, JSModule edge, NameInfo dest) {
531       if (start.isReferenced()) {
532         JSModule startModule = start.getDeepestCommonModuleRef();
533         if (startModule != null &&
534             moduleGraph.dependsOn(startModule, edge)) {
535           return dest.markReference(startModule);
536         } else {
537           return dest.markReference(edge);
538         }
539       }
540       return false;
541     }
542   }

573   /**
574    * A function initialized as a VAR statement or a function declaration.
575    */
576   static class GlobalFunction implements Symbol {
577     private final Node nameNode;
578     private final Var var;
579     private final JSModule module;
580 
581     GlobalFunction(Node nameNode, Var var, JSModule module) {
582       Node parent = nameNode.getParent();
583       Preconditions.checkState(
584           parent.isVar() ||
585           NodeUtil.isFunctionDeclaration(parent));
586       this.nameNode = nameNode;
587       this.var = var;
588       this.module = module;
589     }
590 
591     @Override
592     public Var getRootVar() {
593       return var;
594     }
595 
596     @Override
597     public void remove(AbstractCompiler compiler) {
598       Node parent = nameNode.getParent();
599       compiler.reportChangeToEnclosingScope(parent);
600       if (parent.isFunction() || parent.hasOneChild()) {
601         NodeUtil.removeChild(parent.getParent(), parent);
602       } else {
603         Preconditions.checkState(parent.isVar());
604         parent.removeChild(nameNode);
605       }
606     }
607 
608     @Override
609     public JSModule getModule() {
610       return module;
611     }
612   }

614   /**
615    * Since there are two ways of assigning properties to prototypes, we hide
616    * then behind this interface so they can both be removed regardless of type.
617    */
618   interface Property extends Symbol {
619 
620     /** Returns the GETPROP node that refers to the prototype. */
621     Node getPrototype();
622 
623     /** Returns the value of this property. */
624     Node getValue();
625   }

627   /**
628    * Properties created via EXPR assignment:
629    *
630    * <pre>function Foo() { ... };
631    * Foo.prototype.bar = function() { ... };</pre>
632    */
633   static class AssignmentProperty implements Property {
634     private final Node exprNode;
635     private final Var rootVar;
636     private final JSModule module;
637 
638     /**
639      * @param node An EXPR node.
640      */
641     AssignmentProperty(Node node, Var rootVar, JSModule module) {
642       this.exprNode = node;
643       this.rootVar = rootVar;
644       this.module = module;
645     }
646 
647     @Override
648     public Var getRootVar() {
649       return rootVar;
650     }
651 
652     @Override
653     public void remove(AbstractCompiler compiler) {
654       compiler.reportChangeToEnclosingScope(exprNode);
655       NodeUtil.removeChild(exprNode.getParent(), exprNode);
656     }
657 
658     @Override
659     public Node getPrototype() {
660       return getAssignNode().getFirstChild().getFirstChild();
661     }
662 
663     @Override
664     public Node getValue() {
665       return getAssignNode().getLastChild();
666     }
667 
668     private Node getAssignNode() {
669       return exprNode.getFirstChild();
670     }
671 
672     @Override
673     public JSModule getModule() {
674       return module;
675     }
676   }

678   /**
679    * Properties created via object literals:
680    *
681    * <pre>function Foo() { ... };
682    * Foo.prototype = {bar: function() { ... };</pre>
683    */
684   static class LiteralProperty implements Property {
685     private final Node key;
686     private final Node value;
687     private final Node map;
688     private final Node assign;
689     private final Var rootVar;
690     private final JSModule module;
691 
692     LiteralProperty(Node key, Node value, Node map, Node assign,
693         Var rootVar, JSModule module) {
694       this.key = key;
695       this.value = value;
696       this.map = map;
697       this.assign = assign;
698       this.rootVar = rootVar;
699       this.module = module;
700     }
701 
702     @Override
703     public Var getRootVar() {
704       return rootVar;
705     }
706 
707     @Override
708     public void remove(AbstractCompiler compiler) {
709       compiler.reportChangeToEnclosingScope(key);
710       map.removeChild(key);
711     }
712 
713     @Override
714     public Node getPrototype() {
715       return assign.getFirstChild();
716     }
717 
718     @Override
719     public Node getValue() {
720       return value;
721     }
722 
723     @Override
724     public JSModule getModule() {
725       return module;
726     }

729   /**
730    * The context of the current name. This includes the NameInfo and the scope
731    * if it is a scope defining name (function).
732    */
733   private class NameContext {
734     final NameInfo name;
735 
736     // If this is a function context, then scope will be the scope of the
737     // corresponding function. Otherwise, it will be null.
738     final Scope scope;
739 
740     NameContext(NameInfo name, Scope scope) {
741       this.name = name;
742       this.scope = scope;
743     }
744   }
745 
746   /**
747    * Information on all properties or global variables of a given name.
748    */
749   class NameInfo {
750 
751     final String name;
752 
753     private boolean referenced = false;
754     private final Deque<Symbol> declarations = new ArrayDeque<>();
755     private JSModule deepestCommonModuleRef = null;
756 
757     // True if this property is a function that reads a variable from an
758     // outer scope which isn't the global scope.
759     private boolean readClosureVariables = false;
760 
761     /**
762      * Constructs a new NameInfo.
763      * @param name The name of the property that this represents. May be null
764      *     to signify dummy nodes in the property graph.
765      */
766     NameInfo(String name) {
767       this.name = name;
768     }
769 
770     @Override public String toString() { return name; }
771 
772     /** Determines whether we've marked a reference to this property name. */
773     boolean isReferenced() {
774       return referenced;
775     }
776 
777     /** Determines whether it reads a closure variable. */
778     boolean readsClosureVariables() {
779       return readClosureVariables;
780     }
781 
782     /**
783      * Mark a reference in a given module to this property name, and record
784      * the deepest common module reference.
785      * @param module The module where it was referenced.
786      * @return Whether the name info has changed.
787      */
788     boolean markReference(JSModule module) {
789       boolean hasChanged = false;
790       if (!referenced) {
791         referenced = true;
792         hasChanged = true;
793       }
794 
795       if (moduleGraph != null) {
796         JSModule originalDeepestCommon = deepestCommonModuleRef;
797 
798         if (deepestCommonModuleRef == null) {
799           deepestCommonModuleRef = module;
800         } else {
801           deepestCommonModuleRef =
802               moduleGraph.getDeepestCommonDependencyInclusive(
803                   deepestCommonModuleRef, module);
804         }
805 
806         if (originalDeepestCommon != deepestCommonModuleRef) {
807           hasChanged = true;
808         }
809       }
810 
811       return hasChanged;
812     }
813 
814     /**
815      * Returns the deepest common module of all the references to this
816      * property.
817      */
818     JSModule getDeepestCommonModuleRef() {
819       return deepestCommonModuleRef;
820     }
821 
822     /**
823      * Returns a mutable collection of all the prototype property declarations
824      * of this property name.
825      */
826     Deque<Symbol> getDeclarations() {
827       return declarations;
828     }
829   }
