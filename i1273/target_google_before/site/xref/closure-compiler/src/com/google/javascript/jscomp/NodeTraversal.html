91    /**
92     * Callback for tree-based traversals
93     */
94    public interface Callback {
95      /**
96       * <p>Visits a node in pre order (before visiting its children) and decides
97       * whether this node's children should be traversed. If children are
98       * traversed, they will be visited by
99       * {@link #visit(NodeTraversal, Node, Node)} in postorder.</p>
100      * <p>Implementations can have side effects (e.g. modifying the parse
101      * tree).</p>
102      * @return whether the children of this node should be visited
103      */
104     boolean shouldTraverse(NodeTraversal nodeTraversal, Node n, Node parent);
105 
106     /**
107      * <p>Visits a node in postorder (after its children have been visited).
108      * A node is visited only if all its parents should be traversed
109      * ({@link #shouldTraverse(NodeTraversal, Node, Node)}).</p>
110      * <p>Implementations can have side effects (e.g. modifying the parse
111      * tree).</p>
112      */
113     void visit(NodeTraversal t, Node n, Node parent);
114   }

169   /**
170    * Abstract callback to visit all nodes but not traverse into function
171    * bodies.
172    */
173   public abstract static class AbstractShallowCallback implements Callback {
174     @Override
175     public final boolean shouldTraverse(NodeTraversal nodeTraversal, Node n,
176         Node parent) {
177       // We do want to traverse the name of a named function, but we don't
178       // want to traverse the arguments or body.
179       return parent == null || !parent.isFunction() ||

258   private void throwUnexpectedException(Exception unexpectedException) {
259     // If there's an unexpected exception, try to get the
260     // line number of the code that caused it.
261     String message = unexpectedException.getMessage();
262 
263     // TODO(user): It is possible to get more information if curNode or
264     // its parent is missing. We still have the scope stack in which it is still
265     // very useful to find out at least which function caused the exception.
266     if (inputId != null) {
267       message =
268           unexpectedException.getMessage() + "\n" +
269           formatNodeContext("Node", curNode) +
270           (curNode == null ?
271               "" :
272               formatNodeContext("Parent", curNode.getParent()));
273     }
274     compiler.throwInternalError(message, unexpectedException);
275   }

277   private String formatNodeContext(String label, Node n) {
278     if (n == null) {
279       return "  " + label + ": NULL";
280     }
281     return "  " + label + "(" + n.toString(false, false, false) + "): "
282         + formatNodePosition(n);
283   }

285   /**
286    * Traverses a parse tree recursively.
287    */
288   public void traverse(Node root) {
289     try {
290       inputId = NodeUtil.getInputId(root);
291       sourceName = "";
292       curNode = root;
293       pushScope(root);
294       // null parent ensures that the shallow callbacks will traverse root
295       traverseBranch(root, null);
296       popScope();
297     } catch (Exception unexpectedException) {
298       throwUnexpectedException(unexpectedException);
299     }
300   }
301 
302   void traverseRoots(Node externs, Node root) {
303     try {
304       Node scopeRoot = externs.getParent();
305       Preconditions.checkState(scopeRoot != null);
306 
307       inputId = NodeUtil.getInputId(scopeRoot);
308       sourceName = "";
309       curNode = scopeRoot;
310       pushScope(scopeRoot);
311 
312       traverseBranch(externs, scopeRoot);
313       Preconditions.checkState(root.getParent() == scopeRoot);
314       traverseBranch(root, scopeRoot);
315 
316       popScope();
317     } catch (Exception unexpectedException) {
318       throwUnexpectedException(unexpectedException);
319     }
320   }

322   private static final String MISSING_SOURCE = "[source unknown]";

324   private String formatNodePosition(Node n) {
325     String sourceFileName = getBestSourceFileName(n);
326     if (sourceFileName == null) {
327       return MISSING_SOURCE + "\n";
328     }
329 
330     int lineNumber = n.getLineno();
331     int columnNumber = n.getCharno();
332     String src = compiler.getSourceLine(sourceFileName, lineNumber);
333     if (src == null) {
334       src = MISSING_SOURCE;
335     }
336     return sourceFileName + ":" + lineNumber + ":" + columnNumber + "\n"
337         + src + "\n";
338   }

341    * Traverses a parse tree recursively with a scope, starting with the given
342    * root. This should only be used in the global scope. Otherwise, use
343    * {@link #traverseAtScope}.
344    */
345   void traverseWithScope(Node root, Scope s) {
346     Preconditions.checkState(s.isGlobal());
347     try {
348       inputId = null;
349       sourceName = "";
350       curNode = root;
351       pushScope(s);
352       traverseBranch(root, null);
353       popScope();
354     } catch (Exception unexpectedException) {
355       throwUnexpectedException(unexpectedException);
356     }

359   /**
360    * Traverses a parse tree recursively with a scope, starting at that scope's
361    * root.
362    */
363   void traverseAtScope(Scope s) {
364     Node n = s.getRootNode();
365     if (n.isFunction()) {
366       // We need to do some extra magic to make sure that the scope doesn't
367       // get re-created when we dive into the function.
368       if (inputId == null) {
369         inputId = NodeUtil.getInputId(n);
370       }
371       sourceName = getSourceName(n);
372       curNode = n;
373       pushScope(s);
374 
375       Node args = n.getFirstChild().getNext();
376       Node body = args.getNext();
377       traverseBranch(args, n);
378       traverseBranch(body, n);
379 
380       popScope();
381     } else if (n.isBlock()) {
382       if (inputId == null) {
383         inputId = NodeUtil.getInputId(n);
384       }
385       sourceName = getSourceName(n);
386       curNode = n;
387       pushScope(s);
388 
389       // traverseBranch is not called here to avoid re-creating the block scope.
390       for (Node child = n.getFirstChild(); child != null; ) {
391         Node next = child.getNext();
392         traverseBranch(child, n);
393         child = next;
394       }
395 
396       popScope();
397     } else {
398       Preconditions.checkState(s.isGlobal(), "Expected global scope. Got:", s);
399       traverseWithScope(n, s);
400     }
401   }

403   /**
404    * Traverse a function out-of-band of normal traversal.
405    *
406    * @param node The function node.
407    * @param scope The scope the function is contained in. Does not fire enter/exit
408    *     callback events for this scope.
409    */
410   public void traverseFunctionOutOfBand(Node node, Scope scope) {
411     Preconditions.checkNotNull(scope);
412     Preconditions.checkState(node.isFunction());
413     Preconditions.checkState(scope.getRootNode() != null);
414     if (inputId == null) {
415       inputId = NodeUtil.getInputId(node);
416     }
417     curNode = node.getParent();
418     pushScope(scope, true /* quietly */);
419     traverseBranch(node, curNode);
420     popScope(true /* quietly */);
421   }

423   /**
424    * Traverses an inner node recursively with a refined scope. An inner node may
425    * be any node with a non {@code null} parent (i.e. all nodes except the
426    * root).
427    *
428    * @param node the node to traverse
429    * @param parent the node's parent, it may not be {@code null}
430    * @param refinedScope the refined scope of the scope currently at the top of
431    *     the scope stack or in trivial cases that very scope or {@code null}
432    */
433   void traverseInnerNode(Node node, Node parent, Scope refinedScope) {
434     Preconditions.checkNotNull(parent);
435     if (inputId == null) {
436       inputId = NodeUtil.getInputId(node);
437     }
438     if (refinedScope != null && getScope() != refinedScope) {
439       curNode = node;
440       pushScope(refinedScope);
441       traverseBranch(node, parent);
442       popScope();
443     } else {
444       traverseBranch(node, parent);
445     }
446   }

453    * Gets the current line number, or zero if it cannot be determined. The line
454    * number is retrieved lazily as a running time optimization.
455    */
456   public int getLineNumber() {
457     Node cur = curNode;
458     while (cur != null) {
459       int line = cur.getLineno();
460       if (line >= 0) {

462       }
463       cur = cur.getParent();
464     }
465     return 0;

513   /**
514    * Traversal for passes that work only on changed functions.
515    * Suppose a loopable pass P1 uses this traversal.
516    * Then, if a function doesn't change between two runs of P1, it won't look at
517    * the function the second time.
518    * (We're assuming that P1 runs to a fixpoint, o/w we may miss optimizations.)
519    *
520    * <p>Most changes are reported with calls to Compiler.reportCodeChange(), which
521    * doesn't know which scope changed. We keep track of the current scope by
522    * calling Compiler.setScope inside pushScope and popScope.
523    * The automatic tracking can be wrong in rare cases when a pass changes scope
524    * w/out causing a call to pushScope or popScope. It's very hard to find the
525    * places where this happens unless a bug is triggered.
526    * Passes that do cross-scope modifications call
527    * Compiler.reportChangeToEnclosingScope(Node n).
528    */
529   public static void traverseChangedFunctions(
530       AbstractCompiler compiler, FunctionCallback callback) {
531     final AbstractCompiler comp = compiler;
532     final FunctionCallback cb = callback;
533     final Node jsRoot = comp.getJsRoot();
534     NodeTraversal.traverseEs6(comp, jsRoot,
535         new AbstractPreOrderCallback() {
536           @Override
537           public final boolean shouldTraverse(NodeTraversal t, Node n, Node p) {
538             if ((n == jsRoot || n.isFunction()) && comp.hasScopeChanged(n)) {
539               cb.enterFunction(comp, n);
540             }
541             return true;
542           }
543         });
544   }

579   /**
580    * Traverses a branch.
581    */
582   private void traverseBranch(Node n, Node parent) {
583     int type = n.getType();
584     if (type == Token.SCRIPT) {
585       inputId = n.getInputId();
586       sourceName = getSourceName(n);
587     }
588 
589     curNode = n;
590     if (!callback.shouldTraverse(this, n, parent)) {
591       return;
592     }
593 
594     if (type == Token.FUNCTION) {
595       traverseFunction(n, parent);
596     } else if (useBlockScope && NodeUtil.createsBlockScope(n)) {
597       traverseBlockScope(n);
598     } else {
599       for (Node child = n.getFirstChild(); child != null; ) {
600         // child could be replaced, in which case our child node
601         // would no longer point to the true next
602         Node next = child.getNext();
603         traverseBranch(child, n);
604         child = next;
605       }
606     }
607 
608     curNode = n;
609     callback.visit(this, n, parent);
610   }

612   /** Traverses a function. */
613   private void traverseFunction(Node n, Node parent) {
614     Preconditions.checkState(n.getChildCount() == 3);
615     Preconditions.checkState(n.isFunction());
616 
617     final Node fnName = n.getFirstChild();
618     boolean isFunctionExpression = (parent != null)
619         && NodeUtil.isFunctionExpression(n);
620 
621     if (!isFunctionExpression) {
622       // Functions declarations are in the scope containing the declaration.
623       traverseBranch(fnName, n);
624     }
625 
626     curNode = n;
627     pushScope(n);
628 
629     if (isFunctionExpression) {
630       // Function expression names are only accessible within the function
631       // scope.
632       traverseBranch(fnName, n);
633     }
634 
635     final Node args = fnName.getNext();
636     final Node body = args.getNext();
637 
638     // Args
639     traverseBranch(args, n);
640 
641     // Body
642     // ES6 "arrow" function may not have a block as a body.
643     traverseBranch(body, n);
644 
645     popScope();
646   }

648   /** Traverses a non-function block. */
649   private void traverseBlockScope(Node n) {
650     pushScope(n);
651     for (Node child : n.children()) {
652       traverseBranch(child, n);
653     }
654     popScope();
655   }

665   /** Creates a new scope (e.g. when entering a function). */
666   private void pushScope(Node node) {
667     Preconditions.checkState(curNode != null);
668     Preconditions.checkState(node != null);
669     compiler.setScope(node);
670     scopeRoots.push(node);
671     if (NodeUtil.isValidCfgRoot(node)) {
672       cfgRoots.push(node);
673       cfgs.push(null);
674     }
675     if (scopeCallback != null) {
676       scopeCallback.enterScope(this);
677     }
678   }

680   /** Creates a new scope (e.g. when entering a function). */
681   private void pushScope(Scope s) {
682     pushScope(s, false);
683   }

685   /**
686    * Creates a new scope (e.g. when entering a function).
687    * @param quietly Don't fire an enterScope callback.
688    */
689   private void pushScope(Scope s, boolean quietly) {
690     Preconditions.checkState(curNode != null);
691     compiler.setScope(s.getRootNode());
692     scopes.push(s);
693     if (NodeUtil.isValidCfgRoot(s.getRootNode())) {
694       cfgRoots.push(s.getRootNode());
695       cfgs.push(null);
696     }
697     if (!quietly && scopeCallback != null) {
698       scopeCallback.enterScope(this);
699     }
700   }

702   private void popScope() {
703     popScope(false);
704   }

706   /**
707    * Pops back to the previous scope (e.g. when leaving a function).
708    * @param quietly Don't fire the exitScope callback.
709    */
710   private void popScope(boolean quietly) {
711     if (!quietly && scopeCallback != null) {
712       scopeCallback.exitScope(this);
713     }
714     Node scopeRoot;
715     if (scopeRoots.isEmpty()) {
716       scopeRoot = scopes.pop().getRootNode();
717     } else {
718       scopeRoot = scopeRoots.pop();
719     }
720     if (NodeUtil.isValidCfgRoot(scopeRoot)) {
721       Preconditions.checkState(!cfgRoots.isEmpty());
722       Preconditions.checkState(cfgRoots.pop() == scopeRoot);
723       cfgs.pop();
724     }
725     if (hasScope()) {
726       compiler.setScope(getScopeRoot());
727     }
728   }

730   /** Gets the current scope. */
731   public Scope getScope() {
732     Scope scope = scopes.isEmpty() ? null : scopes.peek();
733     if (scopeRoots.isEmpty()) {
734       return scope;
735     }
736 
737     Iterator<Node> it = scopeRoots.descendingIterator();
738     while (it.hasNext()) {
739       scope = scopeCreator.createScope(it.next(), scope);
740       scopes.push(scope);
741     }
742     scopeRoots.clear();
743     // No need to call compiler.setScope; the top scopeRoot is now the top scope
744     return scope;
745   }

754   public TypedScope getTypedScope() {
755     Scope s = getScope();
756     Preconditions.checkState(s instanceof TypedScope,
757         "getTypedScope called for untyped traversal");
758     return (TypedScope) s;
759   }

761   /** Gets the control flow graph for the current JS scope. */
762   public ControlFlowGraph<Node> getControlFlowGraph() {
763     if (cfgs.peek() == null) {
764       ControlFlowAnalysis cfa = new ControlFlowAnalysis(compiler, false, true);
765       cfa.process(null, getCfgRoot());
766       cfgs.pop();
767       cfgs.push(cfa.getCfg());
768     }
769     return cfgs.peek();
770   }

772   /** Returns the current scope's root. */
773   public Node getScopeRoot() {
774     if (scopeRoots.isEmpty()) {
775       return scopes.peek().getRootNode();
776     } else {
777       return scopeRoots.peek();
778     }
779   }

781   private Node getCfgRoot() {
782     return cfgRoots.peek();
783   }

785   /**
786    * Determines whether the traversal is currently in the global scope. Note that this returns false
787    * in a global block scope.
788    */
789   boolean inGlobalScope() {

791   }

793   /**
794    * Determines whether the hoist scope of the current traversal is global.
795    */
796   boolean inGlobalHoistScope() {
797     return !getCfgRoot().isFunction();

800   int getScopeDepth() {
801     int sum = scopes.size() + scopeRoots.size();
802     Preconditions.checkState(sum > 0);
803     return sum - 1; // Use 0-based scope depth to be consistent within the compiler
804   }

806   public boolean hasScope() {
807     return !(scopes.isEmpty() && scopeRoots.isEmpty());
