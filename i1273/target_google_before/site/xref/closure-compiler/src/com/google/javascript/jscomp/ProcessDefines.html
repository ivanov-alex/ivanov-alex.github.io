43  /**
44   * Process variables annotated as {@code @define}. A define is
45   * a special constant that may be overridden by later files and
46   * manipulated by the compiler, much like C preprocessor {@code #define}s.
47   *
48   * @author nicksantos@google.com (Nick Santos)
49   */
50  class ProcessDefines implements CompilerPass {
51  
52    /**
53     * Defines in this set will not be flagged with "unknown define" warnings.
54     * There are legacy flags that always set these defines, even when they
55     * might not be in the binary.
56     */
57    private static final Set<String> KNOWN_DEFINES = ImmutableSet.of("COMPILED");
58  
59    private final AbstractCompiler compiler;
60    private final Map<String, Node> dominantReplacements;
61  
62    private GlobalNamespace namespace = null;
63  
64    // Warnings
65    static final DiagnosticType UNKNOWN_DEFINE_WARNING = DiagnosticType.warning(
66        "JSC_UNKNOWN_DEFINE_WARNING",
67        "unknown @define variable {0}");
68  
69    // Errors
70    static final DiagnosticType INVALID_DEFINE_TYPE_ERROR =
71      DiagnosticType.error(
72          "JSC_INVALID_DEFINE_TYPE_ERROR",
73          "@define tag only permits literal types");
74  
75    static final DiagnosticType INVALID_DEFINE_INIT_ERROR =
76        DiagnosticType.error(
77            "JSC_INVALID_DEFINE_INIT_ERROR",
78            "illegal initialization of @define variable {0}");
79  
80    static final DiagnosticType NON_GLOBAL_DEFINE_INIT_ERROR =
81        DiagnosticType.error(
82            "JSC_NON_GLOBAL_DEFINE_INIT_ERROR",
83            "@define variable {0} assignment must be global");
84  
85    static final DiagnosticType DEFINE_NOT_ASSIGNABLE_ERROR =
86        DiagnosticType.error(
87            "JSC_DEFINE_NOT_ASSIGNABLE_ERROR",
88            "@define variable {0} cannot be reassigned due to code at {1}.");
89  
90    private static final MessageFormat REASON_DEFINE_NOT_ASSIGNABLE =
91        new MessageFormat("line {0} of {1}");

93    /**
94     * Create a pass that overrides define constants.
95     *
96     * TODO(nicksantos): Write a builder to help JSCompiler induce
97     *    {@code replacements} from command-line flags
98     *
99     * @param replacements A hash table of names of defines to their replacements.
100    *   All replacements <b>must</b> be literals.
101    */
102   ProcessDefines(AbstractCompiler compiler, Map<String, Node> replacements) {
103     this.compiler = compiler;
104     dominantReplacements = replacements;
105   }
106 
107   /**
108    * Injects a pre-computed global namespace, so that the same namespace
109    * can be re-used for multiple check passes. Returns {@code this} for
110    * easy chaining.
111    */
112   ProcessDefines injectNamespace(GlobalNamespace namespace) {
113     this.namespace = namespace;
114     return this;
115   }

117   @Override
118   public void process(Node externs, Node root) {
119     if (namespace == null) {
120       namespace = new GlobalNamespace(compiler, root);
121     }
122     overrideDefines(collectDefines(root, namespace));
123   }
124 
125   private void overrideDefines(Map<String, DefineInfo> allDefines) {
126     boolean changed = false;
127     for (Map.Entry<String, DefineInfo> def : allDefines.entrySet()) {
128       String defineName = def.getKey();
129       DefineInfo info = def.getValue();
130       Node inputValue = dominantReplacements.get(defineName);
131       Node finalValue = inputValue != null ?
132           inputValue : info.getLastValue();
133       if (finalValue != info.initialValue) {
134         info.initialValueParent.replaceChild(
135             info.initialValue, finalValue.cloneTree());
136         compiler.addToDebugLog("Overriding @define variable " + defineName);
137         changed = changed ||
138             finalValue.getType() != info.initialValue.getType() ||
139             !finalValue.isEquivalentTo(info.initialValue);
140       }
141     }
142 
143     if (changed) {
144       compiler.reportCodeChange();
145     }
146 
147     Set<String> unusedReplacements = Sets.difference(
148         dominantReplacements.keySet(), Sets.union(KNOWN_DEFINES, allDefines.keySet()));
149 
150     for (String unknownDefine : unusedReplacements) {
151       compiler.report(JSError.make(UNKNOWN_DEFINE_WARNING, unknownDefine));
152     }
153   }
154 
155   private static String format(MessageFormat format, Object... params) {
156     return format.format(params);
157   }

159   /**
160    * Only defines of literal number, string, or boolean are supported.
161    */
162   private boolean isValidDefineType(JSTypeExpression expression) {
163     TypeIRegistry registry = compiler.getTypeIRegistry();
164     TypeI type = expression.evaluateInEmptyScope(registry);
165     return !type.isUnknownType()
166         && type.isSubtypeOf(registry.getNativeType(NUMBER_STRING_BOOLEAN));
167   }

169   /**
170    * Finds all defines, and creates a {@link DefineInfo} data structure for
171    * each one.
172    * @return A map of {@link DefineInfo} structures, keyed by name.
173    */
174   private Map<String, DefineInfo> collectDefines(Node root,
175       GlobalNamespace namespace) {
176     // Find all the global names with a @define annotation
177     List<Name> allDefines = new ArrayList<>();
178     for (Name name : namespace.getNameIndex().values()) {
179       Ref decl = name.getDeclaration();
180       if (name.docInfo != null && name.docInfo.isDefine()) {
181         // Process defines should not depend on check types being enabled,
182         // so we look for the JSDoc instead of the inferred type.
183         if (isValidDefineType(name.docInfo.getType())) {
184           allDefines.add(name);
185         } else {
186           JSError error = JSError.make(
187               decl.node, INVALID_DEFINE_TYPE_ERROR);
188           compiler.report(error);
189         }
190       } else {
191         for (Ref ref : name.getRefs()) {
192           if (ref == decl) {
193             // Declarations were handled above.
194             continue;
195           }
196 
197           Node n = ref.node;
198           Node parent = ref.node.getParent();
199           JSDocInfo info = n.getJSDocInfo();
200           if (info == null &&
201               parent.isVar() && parent.hasOneChild()) {
202             info = parent.getJSDocInfo();
203           }
204 
205           if (info != null && info.isDefine()) {
206             allDefines.add(name);
207             break;
208           }
209         }
210       }
211     }
212 
213     CollectDefines pass = new CollectDefines(compiler, allDefines);
214     NodeTraversal.traverseEs6(compiler, root, pass);
215     return pass.getAllDefines();
216   }

218   /**
219    * Finds all assignments to @defines, and figures out the last value of
220    * the @define.
221    */
222   private static final class CollectDefines implements Callback {
223 
224     private final AbstractCompiler compiler;
225     private final Map<String, DefineInfo> assignableDefines;
226     private final Map<String, DefineInfo> allDefines;
227     private final Map<Node, RefInfo> allRefInfo;
228 
229     // A hack that allows us to remove ASSIGN/VAR statements when
230     // we're currently visiting one of the children of the assign.
231     private Node lvalueToRemoveLater = null;
232 
233     // A stack tied to the node traversal, to keep track of whether
234     // we're in a conditional block. If 1 is at the top, assignment to
235     // a define is allowed. Otherwise, it's not allowed.
236     private final Deque<Integer> assignAllowed;
237 
238     CollectDefines(AbstractCompiler compiler, List<Name> listOfDefines) {
239       this.compiler = compiler;
240       this.allDefines = new HashMap<>();
241 
242       assignableDefines = new HashMap<>();
243       assignAllowed = new ArrayDeque<>();
244       assignAllowed.push(1);
245 
246       // Create a map of references to defines keyed by node for easy lookup
247       allRefInfo = new HashMap<>();
248       for (Name name : listOfDefines) {
249         Ref decl = name.getDeclaration();
250         if (decl != null) {
251           allRefInfo.put(decl.node,
252                          new RefInfo(decl, name));
253         }
254         for (Ref ref : name.getRefs()) {
255           if (ref == decl) {
256             // Declarations were handled above.
257             continue;
258           }
259 
260           // If there's a TWIN def, only put one of the twins in.
261           if (ref.getTwin() == null || !ref.getTwin().isSet()) {
262             allRefInfo.put(ref.node, new RefInfo(ref, name));
263           }
264         }
265       }
266     }
267 
268     /**
269      * Get a map of {@link DefineInfo} structures, keyed by the name of
270      * the define.
271      */
272     Map<String, DefineInfo> getAllDefines() {
273       return allDefines;
274     }
275 
276     /**
277      * Keeps track of whether the traversal is in a conditional branch.
278      * We traverse all nodes of the parse tree.
279      */
280     @Override
281     public boolean shouldTraverse(NodeTraversal nodeTraversal, Node n,
282         Node parent) {
283       updateAssignAllowedStack(n, true);
284       return true;
285     }
286 
287     @Override
288     public void visit(NodeTraversal t, Node n, Node parent) {
289       RefInfo refInfo = allRefInfo.get(n);
290       if (refInfo != null) {
291         Ref ref = refInfo.ref;
292         Name name = refInfo.name;
293         String fullName = name.getFullName();
294         switch (ref.type) {
295           case SET_FROM_GLOBAL:
296           case SET_FROM_LOCAL:
297             Node valParent = getValueParent(ref);
298             Node val = valParent.getLastChild();
299             if (valParent.isAssign() && name.isSimpleName() &&
300                 name.getDeclaration() == ref) {
301               // For defines, it's an error if a simple name is assigned
302               // before it's declared
303               compiler.report(
304                   t.makeError(val, INVALID_DEFINE_INIT_ERROR, fullName));
305             } else if (processDefineAssignment(t, fullName, val, valParent)) {
306               // remove the assignment so that the variable is still declared,
307               // but no longer assigned to a value, e.g.,
308               // DEF_FOO = 5; // becomes "5;"
309 
310               // We can't remove the ASSIGN/VAR when we're still visiting its
311               // children, so we'll have to come back later to remove it.
312               refInfo.name.removeRef(ref);
313               lvalueToRemoveLater = valParent;
314             }
315             break;
316           default:
317             if (t.inGlobalHoistScope()) {
318               // Treat this as a reference to a define in the global scope.
319               // After this point, the define must not be reassigned,
320               // or it's an error.
321               DefineInfo info = assignableDefines.get(fullName);
322               if (info != null) {
323                 setDefineInfoNotAssignable(info, t);
324                 assignableDefines.remove(fullName);
325               }
326             }
327             break;
328         }
329       }
330 
331       if (!t.inGlobalScope() &&
332           n.getJSDocInfo() != null && n.getJSDocInfo().isDefine()) {
333         // warn about @define annotations in local scopes
334         compiler.report(
335             t.makeError(n, NON_GLOBAL_DEFINE_INIT_ERROR, ""));
336       }
337 
338       if (lvalueToRemoveLater == n) {
339         lvalueToRemoveLater = null;
340         if (n.isAssign()) {
341           Node last = n.getLastChild();
342           n.removeChild(last);
343           parent.replaceChild(n, last);
344         } else {
345           Preconditions.checkState(n.isName());
346           n.removeChild(n.getFirstChild());
347         }
348         compiler.reportCodeChange();
349       }
350 
351       if (n.isCall()) {
352         if (t.inGlobalScope()) {
353           // If there's a function call in the global scope,
354           // we just say it's unsafe and freeze all the defines.
355           //
356           // NOTE(nicksantos): We could be a lot smarter here. For example,
357           // ReplaceOverriddenVars keeps a call graph of all functions and
358           // which functions/variables that they reference, and tries
359           // to statically determine which functions are "safe" and which
360           // are not. But this would be overkill, especially because
361           // the intended use of defines is with config_files, where
362           // all the defines are at the top of the bundle.
363           for (DefineInfo info : assignableDefines.values()) {
364             setDefineInfoNotAssignable(info, t);
365           }
366 
367           assignableDefines.clear();
368         }
369       }
370 
371       updateAssignAllowedStack(n, false);
372     }
373 
374     /**
375      * Determines whether assignment to a define should be allowed
376      * in the subtree of the given node, and if not, records that fact.
377      *
378      * @param n The node whose subtree we're about to enter or exit.
379      * @param entering True if we're entering the subtree, false otherwise.
380      */
381     private void updateAssignAllowedStack(Node n, boolean entering) {
382       switch (n.getType()) {
383         case Token.CASE:
384         case Token.FOR:
385         case Token.FUNCTION:
386         case Token.HOOK:
387         case Token.IF:
388         case Token.SWITCH:
389         case Token.WHILE:
390           if (entering) {
391             assignAllowed.push(0);
392           } else {
393             assignAllowed.remove();
394           }
395           break;
396       }
397     }
398 
399     /**
400      * Determines whether assignment to a define should be allowed
401      * at the current point of the traversal.
402      */
403     private boolean isAssignAllowed() {
404       return assignAllowed.element() == 1;
405     }
406 
407     /**
408      * Tracks the given define.
409      *
410      * @param t The current traversal, for context.
411      * @param name The full name for this define.
412      * @param value The value assigned to the define.
413      * @param valueParent The parent node of value.
414      * @return Whether we should remove this assignment from the parse tree.
415      */
416     private boolean processDefineAssignment(NodeTraversal t,
417         String name, Node value, Node valueParent) {
418       if (value == null || !NodeUtil.isValidDefineValue(value,
419                                                         allDefines.keySet())) {
420         compiler.report(
421             t.makeError(value, INVALID_DEFINE_INIT_ERROR, name));
422       } else if (!isAssignAllowed()) {
423         compiler.report(
424             t.makeError(valueParent, NON_GLOBAL_DEFINE_INIT_ERROR, name));
425       } else {
426         DefineInfo info = allDefines.get(name);
427         if (info == null) {
428           // First declaration of this define.
429           info = new DefineInfo(value, valueParent);
430           allDefines.put(name, info);
431           assignableDefines.put(name, info);
432         } else if (info.recordAssignment(value)) {
433           // The define was already initialized, but this is a safe
434           // re-assignment.
435           return true;
436         } else {
437           // The define was already initialized, and this is an unsafe
438           // re-assignment.
439           compiler.report(
440               t.makeError(valueParent, DEFINE_NOT_ASSIGNABLE_ERROR,
441                   name, info.getReasonWhyNotAssignable()));
442         }
443       }
444 
445       return false;
446     }

448     /**
449      * Gets the parent node of the value for any assignment to a Name.
450      * For example, in the assignment
451      * {@code var x = 3;}
452      * the parent would be the NAME node.
453      */
454     private static Node getValueParent(Ref ref) {
455       // there are two types of declarations: VARs and ASSIGNs
456       return ref.node.getParent() != null &&
457           ref.node.getParent().isVar() ?
458           ref.node : ref.node.getParent();
459     }

461     /**
462      * Records the fact that because of the current node in the node traversal,
463      * the define can't ever be assigned again.
464      *
465      * @param info Represents the define variable.
466      * @param t The current traversal.
467      */
468     private static void setDefineInfoNotAssignable(DefineInfo info, NodeTraversal t) {
469       info.setNotAssignable(format(REASON_DEFINE_NOT_ASSIGNABLE,
470                                 t.getLineNumber(), t.getSourceName()));
471     }

473     /**
474      * A simple data structure for associating a Ref with the name
475      * that it references.
476      */
477     private static class RefInfo {
478       final Ref ref;
479       final Name name;
480 
481       RefInfo(Ref ref, Name name) {
482         this.ref = ref;
483         this.name = name;
484       }
485     }

488   /**
489    * A simple class for storing information about a define.
490    * Gathers the initial value, the last assigned value, and whether
491    * the define can be safely assigned a new value.
492    */
493   private static final class DefineInfo {
494     public final Node initialValueParent;
495     public final Node initialValue;
496     private Node lastValue;
497     private boolean isAssignable;
498     private String reasonNotAssignable;
499 
500     /**
501      * Initializes a define.
502      */
503     public DefineInfo(Node initialValue, Node initialValueParent) {
504       this.initialValueParent = initialValueParent;
505       this.initialValue = initialValue;
506       lastValue = initialValue;
507       isAssignable = true;
508     }
509 
510     /**
511      * Records the fact that this define can't be assigned a value anymore.
512      *
513      * @param reason A message describing the reason why it can't be assigned.
514      */
515     public void setNotAssignable(String reason) {
516       isAssignable = false;
517       reasonNotAssignable = reason;
518     }
519 
520     /**
521      * Gets the reason why a define is not assignable.
522      */
523     public String getReasonWhyNotAssignable() {
524       return reasonNotAssignable;
525     }
526 
527     /**
528      * Records an assigned value.
529      *
530      * @return False if there was an error.
531      */
532     public boolean recordAssignment(Node value) {
533       lastValue = value;
534       return isAssignable;
535     }
536 
537     /**
538      * Gets the last assigned value.
539      */
540     public Node getLastValue() {
541       return lastValue;
542     }
