62    static void testModules(CompilerTestCase test, String input, String expected) {
63      // Shared with ProcessCommonJSModulesTest.
64      String fileName = test.getFilename() + ".js";
65      ImmutableList<SourceFile> inputs =
66          ImmutableList.of(SourceFile.fromCode("other.js", ""), SourceFile.fromCode(fileName, input));
67      String fileoverview = "/** @fileoverview\n * @suppress {missingProvide|missingRequire}\n */";
68      ImmutableList<SourceFile> expecteds =
69          ImmutableList.of(
70              SourceFile.fromCode("other.js", ""),
71              SourceFile.fromCode(fileName, fileoverview + expected));
72      test.test(inputs, expecteds);
73    }
74  
75    static void testModules(
76        CompilerTestCase test, ImmutableList<SourceFile> inputs, String expected) {
77      ImmutableList<SourceFile> expecteds =
78          ImmutableList.of(
79              SourceFile.fromCode("other.js", ""),
80              SourceFile.fromCode(test.getFilename() + ".js", expected));
81      test.test(inputs, expecteds);
82    }
83  
84    void testModules(String input, String expected) {
85      testModules(this, input, expected);
86    }

88    public void testImport() {
89      testModules(
90          "import name from 'other'; use(name);",
91          LINE_JOINER.join("goog.require('module$other');", "use(module$other.default);"));
92  
93      testModules("import {n as name} from 'other';", "goog.require('module$other');");
94  
95      testModules(
96          "import x, {f as foo, b as bar} from 'other'; use(x);",
97          LINE_JOINER.join("goog.require('module$other');", "use(module$other.default);"));
98    }
