80    @Override
81    protected CompilerOptions getOptions() {
82      CompilerOptions options = super.getOptions();
83      options.setWarningLevel(
84          DiagnosticGroups.MISSING_PROPERTIES, CheckLevel.OFF);
85      options.setCodingConvention(getCodingConvention());
86      return options;
87    }

89    @Override
90    protected void setUp() throws Exception {
91      super.setUp();
92      super.enableTypeCheck(CheckLevel.OFF);
93      super.enableClosurePass();
94      configuration = DEFAULT_CONFORMANCE;
95    }

97    @Override
98    public CompilerPass getProcessor(final Compiler compiler) {
99      ConformanceConfig.Builder builder = ConformanceConfig.newBuilder();
100     try {
101       TextFormat.merge(configuration, builder);
102     } catch (Exception e) {
103       Throwables.propagate(e);
104     }
105     return new CheckConformance(compiler, ImmutableList.of(builder.build()));
106   }
