941   private void testTTL(JSType expectedType, String ttlExp,
942       String... expectedWarnings) {
943     TypeTransformationParser ttlParser = new TypeTransformationParser(ttlExp,
944         SourceFile.fromCode("[testcode]", ttlExp), errorReporter, 0, 0);
945     // Run the test if the parsing was successful
946     if (ttlParser.parseTypeTransformation()) {
947       Node ast = ttlParser.getTypeTransformationAst();
948       // Create the scope using the extra definitions
949       Node extraTypeDefs = compiler.parseTestCode(EXTRA_TYPE_DEFS);
950       TypedScope scope = new TypedScopeCreator(compiler).createScope(
951           extraTypeDefs, null);
952       // Evaluate the type transformation
953       TypeTransformation typeTransformation =
954           new TypeTransformation(compiler, scope);
955       JSType resultType = typeTransformation.eval(ast, typeVars, nameVars);
956       checkReportedWarningsHelper(expectedWarnings);
957       assertTypeEquals(expectedType, resultType);
958     }
