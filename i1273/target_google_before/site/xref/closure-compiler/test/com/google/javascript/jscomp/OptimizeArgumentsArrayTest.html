17  package com.google.javascript.jscomp;
18  
19  /**
20   * Unit tests for {@link OptimizeArgumentsArray}.
21   *
22   */
23  public final class OptimizeArgumentsArrayTest extends CompilerTestCase {
24  
25    public OptimizeArgumentsArrayTest() {
26      /*
27       * arguments is a builtin variable of the javascript language and
28       * OptimizeArgumentsArray does not make any attempt to resolve it. However,
29       * I am leaving "var arguments" in the externs to emulate the current
30       * behavior we have for JS compilation where var arguments in defined in
31       * externs/es3.js as extern.
32       */
33      super("var arguments, alert" /* Externs */);
34    }
35  
36    @Override
37    protected CompilerPass getProcessor(Compiler compiler) {
38      return new OptimizeArgumentsArray(compiler, "p");
39    }
40  
41    public void testSimple() {
42      test("function foo()   { alert(arguments[0]); }",
43           "function foo(p0) { alert(p0); }");
44    }
45  
46    public void testNoVarArgs() {
47      testSame("function f(a,b,c) { alert(a + b + c) }");
48  
49      test("function f(a,b,c) { alert(arguments[0]) }",
50           "function f(a,b,c) { alert(a) }");
51    }
52  
53    public void testMissingVarArgs() {
54      testSame("function f() { alert(arguments[x]) }");
55    }
56  
57    public void testArgumentRefOnNamedParameter() {
58      test("function f(a,b) { alert(arguments[0]) }",
59           "function f(a,b) { alert(a) }");
60    }
61  
62    public void testTwoVarArgs() {
63      test("function foo(a) { alert(arguments[1] + arguments[2]); }",
64           "function foo(a, p0, p1) { alert(p0 + p1); }");
65    }
66  
67    public void testTwoFourArgsTwoUsed() {
68      test("function foo() { alert(arguments[0] + arguments[3]); }",
69           "function foo(p0, p1, p2, p3) { alert(p0 + p3); }");
70    }
71  
72    public void testOneRequired() {
73      test("function foo(req0, var_args) { alert(req0 + arguments[1]); }",
74           "function foo(req0, var_args) { alert(req0 + var_args); }");
75    }
76  
77    public void testTwoRequiredSixthVarArgReferenced() {
78      test("function foo(r0, r1, var_args) {alert(r0 + r1 + arguments[5]);}",
79           "function foo(r0, r1, var_args, p0, p1, p2) { alert(r0 + r1 + p2); }");
80    }
81  
82    public void testTwoRequiredOneOptionalFifthVarArgReferenced() {
83      test("function foo(r0, r1, opt_1)"
84         + "  {alert(r0 + r1 + opt_1 + arguments[4]);}",
85           "function foo(r0, r1, opt_1, p0, p1)"
86         + "  {alert(r0 + r1 + opt_1 + p1); }");
87    }
88  
89    public void testTwoRequiredTwoOptionalSixthVarArgReferenced() {
90      test("function foo(r0, r1, opt_1, opt_2)"
91         + "  {alert(r0 + r1 + opt_1 + opt_2 + arguments[5]);}",
92           "function foo(r0, r1, opt_1, opt_2, p0, p1)"
93         + "  {alert(r0 + r1 + opt_1 + opt_2 + p1); }");
94    }
95  
96    public void testInnerFunctions() {
97      test("function f() { function b(  ) { arguments[0]  }}",
98           "function f() { function b(p0) {            p0 }}");
99  
100     test("function f(  ) { function b() { }  arguments[0] }",
101          "function f(p0) { function b() { }            p0 }");
102 
103     test("function f( )  { arguments[0]; function b(  ) { arguments[0] }}",
104          "function f(p1) {           p1; function b(p0) {           p0 }}");
105   }
106 
107   public void testInnerFunctionsWithNamedArgumentInInnerFunction() {
108     test("function f() { function b(x   ) { arguments[1] }}",
109          "function f() { function b(x,p0) {           p0 }}");
110 
111     test("function f(  ) { function b(x) { }  arguments[0] }",
112          "function f(p0) { function b(x) { }            p0 }");
113 
114     test("function f( )  { arguments[0]; function b(x   ) { arguments[1] }}",
115          "function f(p1) {           p1; function b(x,p0) {           p0 }}");
116   }
117 
118   public void testInnerFunctionsWithNamedArgumentInOutterFunction() {
119     test("function f(x) { function b(  ) { arguments[0] }}",
120          "function f(x) { function b(p0) {           p0 }}");
121 
122     test("function f(x   ) { function b() { }  arguments[1] }",
123          "function f(x,p0) { function b() { }            p0 }");
124 
125     test("function f(x   ) { arguments[1]; function b(  ) { arguments[0] }}",
126          "function f(x,p1) {           p1; function b(p0) {           p0 }}");
127   }
128 
129   public void testInnerFunctionsWithNamedArgumentInInnerAndOutterFunction() {
130     test("function f(x) { function b(x   ) { arguments[1] }}",
131          "function f(x) { function b(x,p0) {           p0 }}");
132 
133     test("function f(x   ) { function b(x) { }  arguments[1] }",
134          "function f(x,p0) { function b(x) { }            p0 }");
135 
136     test("function f(x   ) { arguments[1]; function b(x   ) { arguments[1] }}",
137          "function f(x,p1) {           p1; function b(x,p0) {           p0 }}");
138   }
139 
140   public void testInnerFunctionsAfterArguments() {
141     // This caused a bug earlier due to incorrect push and pop of the arguments
142     // access stack.
143     test("function f(  ) { arguments[0]; function b() { function c() { }} }",
144          "function f(p0) {           p0; function b() { function c() { }} }");
145   }
146 
147   public void testNoOptimizationWhenGetProp() {
148     testSame("function f() { arguments[0]; arguments.size }");
149   }
150 
151   public void testNoOptimizationWhenIndexIsNotNumberConstant() {
152     testSame("function f() { arguments[0]; arguments['callee'].length}");
153     testSame("function f() { arguments[0]; arguments.callee.length}");
154     testSame(
155         "function f() { arguments[0]; var x = 'callee'; arguments[x].length}");
156   }
157 
158   public void testNoOptimizationWhenArgumentIsUsedAsFunctionCall() {
159     testSame("function f() {arguments[0]()}");
160   }
161 
162   public void testUnusualArgumentsUsage() {
163     testSame("function f(x) { x[arguments]; }");
164   }
165 }
