265   public void testDeterministic() {
266     // Make the variable interference graph a pentagon.
267     //         a - b
268     //        /     \
269     //        e     c
270     //         \   /
271     //           d
272     // The coloring partitioning would be:
273     //  a = { a, c }
274     //  b = { b, d }
275     //  e = { e }
276     inFunction("var a,b,c,d,e;" +
277                "  a=1; b=1; a; b;" +
278                "  b=1; c=1; b; c;" +
279                "  c=1; d=1; c; d;" +
280                "  d=1; e=1; d; e;" +
281                "  e=1; a=1; e; a;",
282 
283                "var a,b,    e;" +
284                "  a=1; b=1; a; b;" +
285                "  b=1; a=1; b; a;" +
286                "  a=1; b=1; a; b;" +
287                "  b=1; e=1; b; e;" +
288                "  e=1; a=1; e; a;");
289 
290     // If we favor "d" first by declaring "d" earlier,
291     // the coloring partitioning would be:
292     //  b = { b, e }
293     //  d = { d, a }
294     //  c = { c }
295     inFunction("var d,a,b,c,e;" +
296                "  a=1; b=1; a; b;" +
297                "  b=1; c=1; b; c;" +
298                "  c=1; d=1; c; d;" +
299                "  d=1; e=1; d; e;" +
300                "  e=1; a=1; e; a;",
301 
302                "var d,  b,c  ;" +
303                "  d=1; b=1; d; b;" +
304                "  b=1; c=1; b; c;" +
305                "  c=1; d=1; c; d;" +
306                "  d=1; b=1; d; b;" +
307                "  b=1; d=1; b; d;");
308   }
