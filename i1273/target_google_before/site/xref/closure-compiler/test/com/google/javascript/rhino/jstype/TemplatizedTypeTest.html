39  package com.google.javascript.rhino.jstype;
40  
41  import com.google.common.collect.ImmutableList;
42  import com.google.javascript.rhino.testing.BaseJSTypeTestCase;
43  
44  public class TemplatizedTypeTest extends BaseJSTypeTestCase {
45  
46    @Override
47    public void setUp() throws Exception {
48      super.setUp();
49    }
50  
51    /**
52     * Assert that a type can assign to itself.
53     */
54    private void assertTypeCanAssignToItself(JSType type) {
55      assertTrue(type.isSubtype(type));
56    }
57  
58    /**
59     * Tests the behavior of variants type.
60     */
61    @SuppressWarnings("checked")
62    public void testTemplatizedType() throws Exception {
63      TemplatizedType arrOfString = createTemplatizedType(
64          ARRAY_TYPE, STRING_TYPE);
65      assertTypeCanAssignToItself(arrOfString);
66      assertTrue(arrOfString.isSubtype(ARRAY_TYPE));
67      assertTrue(ARRAY_TYPE.isSubtype(arrOfString));
68  
69      TemplatizedType arrOfNumber = createTemplatizedType(
70          ARRAY_TYPE, NUMBER_TYPE);
71      assertTypeCanAssignToItself(arrOfNumber);
72      assertTrue(arrOfNumber.isSubtype(ARRAY_TYPE));
73      assertTrue(ARRAY_TYPE.isSubtype(arrOfNumber));
74  
75      assertTrue(arrOfString.isEquivalentTo(createTemplatizedType(
76          ARRAY_TYPE, STRING_TYPE)));
77  
78      assertFalse(arrOfString.isEquivalentTo(ARRAY_TYPE));
79      assertFalse(arrOfString.isEquivalentTo(ARRAY_TYPE));
80      assertFalse(arrOfString.isEquivalentTo(arrOfNumber));
81      assertFalse(arrOfNumber.isEquivalentTo(arrOfString));
82    }
83  
84    public void testPrint1() throws Exception {
85      TemplatizedType arrOfString = createTemplatizedType(
86          ARRAY_TYPE, STRING_TYPE);
87      assertEquals("Array<string>", arrOfString.toString());
88    }
89  
90    public void testPrint2() throws Exception {
91      TemplatizedType arrOfTemplateType = createTemplatizedType(
92          ARRAY_TYPE, new TemplateType(registry, "T"));
93      assertEquals("Array<T>", arrOfTemplateType.toString());
94    }
95  
96    public void testPrint3() throws Exception {
97      TemplatizedType arrOfUnknown = createTemplatizedType(
98          ARRAY_TYPE, UNKNOWN_TYPE);
99      assertEquals("Array<?>", arrOfUnknown.toString());
100   }
101 
102   public void testDifferentRawTypes() throws Exception {
103     TemplatizedType arrOfNumber = createTemplatizedType(
104         ARRAY_TYPE, NUMBER_TYPE);
105     TemplatizedType objType = createTemplatizedType(
106         OBJECT_TYPE, UNKNOWN_TYPE);
107     assertTrue(arrOfNumber.isSubtype(objType));
108     assertFalse(objType.isSubtype(arrOfNumber));
109   }
110 
111   public void testCustomTemplatizedType() throws Exception {
112     FunctionType ctor = registry.createConstructorType(
113         "Foo", null, null, null, ImmutableList.of(
114             registry.createTemplateType("T"),
115             registry.createTemplateType("U")));
116     ObjectType baseType = ctor.getInstanceType();
117 
118     JSType templatizedType1 = registry.createTemplatizedType(
119         baseType, ImmutableList.of(STRING_TYPE, NUMBER_TYPE));
120     JSType templatizedType2 = registry.createTemplatizedType(
121         baseType, ImmutableList.of(STRING_TYPE, ALL_TYPE));
122     JSType templatizedType3 = registry.createTemplatizedType(
123         baseType, ImmutableList.of(STRING_TYPE, UNKNOWN_TYPE));
124     JSType templatizedType4 = registry.createTemplatizedType(
125         baseType, ImmutableList.<JSType>of(UNKNOWN_TYPE, UNKNOWN_TYPE));
126 
127     assertTrue(templatizedType1.isSubtype(baseType));
128     assertTrue(templatizedType2.isSubtype(baseType));
129     assertTrue(templatizedType3.isSubtype(baseType));
130     assertTrue(templatizedType4.isSubtype(baseType));
131 
132     assertFalse(templatizedType1.isEquivalentTo(baseType));
133     assertFalse(templatizedType2.isEquivalentTo(baseType));
134     assertFalse(templatizedType3.isEquivalentTo(baseType));
135     assertTrue(templatizedType4.isEquivalentTo(baseType));
136 
137     assertTrue(baseType.isSubtype(templatizedType1));
138     assertTrue(baseType.isSubtype(templatizedType2));
139     assertTrue(baseType.isSubtype(templatizedType3));
140     assertTrue(baseType.isSubtype(templatizedType4));
141 
142     assertFalse(templatizedType1.isSubtype(templatizedType2));
143     assertFalse(templatizedType2.isSubtype(templatizedType1));
144 
145     assertTrue(templatizedType2.isSubtype(templatizedType3));
146     assertTrue(templatizedType3.isSubtype(templatizedType2));
147   }
148 }
