17  package com.google.javascript.jscomp;
18  
19  import static com.google.javascript.jscomp.CheckJSDoc.ANNOTATION_DEPRECATED;
20  import static com.google.javascript.jscomp.CheckJSDoc.ARROW_FUNCTION_AS_CONSTRUCTOR;
21  import static com.google.javascript.jscomp.CheckJSDoc.DEFAULT_PARAM_MUST_BE_MARKED_OPTIONAL;
22  import static com.google.javascript.jscomp.CheckJSDoc.DISALLOWED_MEMBER_JSDOC;
23  import static com.google.javascript.jscomp.CheckJSDoc.MISPLACED_ANNOTATION;
24  import static com.google.javascript.jscomp.CheckJSDoc.MISPLACED_MSG_ANNOTATION;

26  /**
27   * Tests for {@link CheckJSDoc}.
28   *
29   * @author chadkillingsworth@gmail.com (Chad Killingsworth)
30   */
31  
32  public final class CheckJsDocTest extends Es6CompilerTestCase {
33  
34    @Override
35    protected CompilerPass getProcessor(final Compiler compiler) {
36      return new CheckJSDoc(compiler);
37    }
38  
39    @Override
40    protected CompilerOptions getOptions() {
41      CompilerOptions options = super.getOptions();
42      options.setWarningLevel(
43          DiagnosticGroups.MISPLACED_TYPE_ANNOTATION, CheckLevel.WARNING);
44      return options;
45    }
46  
47    public void testInvalidClassJsdoc() {
48      testSameEs6("class Foo { /** @param {number} x */ constructor(x) {}}");
49  
50      testWarningEs6(
51          "class Foo { /** @constructor */ constructor() {}}",
52          DISALLOWED_MEMBER_JSDOC);
53  
54      testWarningEs6(
55          "class Foo { /** @interface */ constructor() {}}",
56          DISALLOWED_MEMBER_JSDOC);
57  
58      testWarningEs6(
59          "class Foo { /** @extends {Foo} */ constructor() {}}",
60          DISALLOWED_MEMBER_JSDOC);
61  
62      testWarningEs6(
63          "class Foo { /** @implements {Foo} */ constructor() {}}",
64          DISALLOWED_MEMBER_JSDOC);
65    }
66  
67    public void testFunctionJSDocOnMethods() {
68      testSameEs6("class Foo { /** @return {?} */ bar() {} }");
69      testSameEs6("class Foo { /** @return {?} */ get bar() {} }");
70      testSameEs6("class Foo { /** @param {?} x */ set bar(x) {} }");
71  
72      testSameEs6("class Foo { /** @return {?} */ [bar]() {} }");
73      testSameEs6("class Foo { /** @return {?} */ get [bar]() {} }");
74      testSameEs6("class Foo { /** @return {?} x */ set [bar](x) {} }");
75    }
76  
77    public void testMethodsOnObjectLiterals() {
78      testSameEs6("var x = { /** @return {?} */ foo() {} };");
79      testSameEs6("var x = { /** @return {?} */ [foo]() {} };");
80      testSameEs6("var x = { /** @return {?} */ foo: someFn };");
81      testSameEs6("var x = { /** @return {?} */ [foo]: someFn };");
82    }
83  
84    public void testExposeDeprecated() {
85      testWarning("/** @expose */ var x = 0;", ANNOTATION_DEPRECATED);
86    }
87  
88    public void testJSDocFunctionNodeAttachment() {
89      testWarning("var a = /** @param {number} index */5;"
90          + "/** @return boolean */function f(index){}", MISPLACED_ANNOTATION);
91    }
92  
93    public void testJSDocDescAttachment() {
94      testWarning(
95          "function f() { return /** @type {string} */ (g(1 /** @desc x */)); };",
96          MISPLACED_MSG_ANNOTATION);
97  
98      testWarning("/** @desc Foo. */ var bar = goog.getMsg('hello');",
99          MISPLACED_MSG_ANNOTATION);
100     testWarning("/** @desc Foo. */ x.y.z.bar = goog.getMsg('hello');",
101         MISPLACED_MSG_ANNOTATION);
102     testWarning("var msgs = {/** @desc x */ x: goog.getMsg('x')}",
103         MISPLACED_MSG_ANNOTATION);
104     testWarning("/** @desc Foo. */ bar = goog.getMsg('x');",
105         MISPLACED_MSG_ANNOTATION);
106   }
107 
108   public void testJSDocTypeAttachment() {
109     testWarning(
110         "function f() {  /** @type {string} */ if (true) return; };", MISPLACED_ANNOTATION);
111 
112     testWarning(
113         "function f() {  /** @type {string} */  return; };", MISPLACED_ANNOTATION);
114   }
115 
116 
117   public void testMisplacedTypeAnnotation1() {
118     // misuse with COMMA
119     testWarning(
120         "var o = {}; /** @type {string} */ o.prop1 = 1, o.prop2 = 2;", MISPLACED_ANNOTATION);
121   }
122 
123   public void testMisplacedTypeAnnotation2() {
124     // missing parentheses for the cast.
125     testWarning(
126         "var o = /** @type {string} */ getValue();",
127         MISPLACED_ANNOTATION);
128   }

131     // missing parentheses for the cast.

135   }

163   public void testAllowedNocollapseAnnotation1() {
164     testSame("var foo = {}; /** @nocollapse */ foo.bar = true;");
165   }
166 
167   public void testAllowedNocollapseAnnotation2() {
168     testSame(
169         "/** @constructor */ function Foo() {};\n"
170         + "var ns = {};\n"
171         + "/** @nocollapse */ ns.bar = Foo.prototype.blah;");
172   }
173 
174   public void testMisplacedNocollapseAnnotation1() {
175     testWarning(
176         "/** @constructor */ function foo() {};"
177             + "/** @nocollapse */ foo.prototype.bar = function() {};",
178         MISPLACED_ANNOTATION);
179   }
180 
181   public void testNocollapseInExterns() {
182     testSame("var foo = {}; /** @nocollapse */ foo.bar = true;",
183         "foo.bar;", MISPLACED_ANNOTATION);
184   }
185 
186   public void testArrowFuncAsConstructor() {
187     testErrorEs6("/** @constructor */ var a = ()=>{}; var b = a();",
188         ARROW_FUNCTION_AS_CONSTRUCTOR);
189     testErrorEs6("var a = /** @constructor */ ()=>{}; var b = a();",
190         ARROW_FUNCTION_AS_CONSTRUCTOR);
191     testErrorEs6("/** @constructor */ let a = ()=>{}; var b = a();",
192         ARROW_FUNCTION_AS_CONSTRUCTOR);
193     testErrorEs6("/** @constructor */ const a = ()=>{}; var b = a();",
194         ARROW_FUNCTION_AS_CONSTRUCTOR);
195     testErrorEs6("var a; /** @constructor */ a = ()=>{}; var b = a();",
196         ARROW_FUNCTION_AS_CONSTRUCTOR);
197   }
198 
199   public void testDefaultParam() {
200     testErrorEs6("function f(/** number */ x=0) {}", DEFAULT_PARAM_MUST_BE_MARKED_OPTIONAL);
201     testSameEs6("function f(/** number= */ x=0) {}");
202   }
