16  package com.google.javascript.jscomp;
17  
18  import com.google.javascript.jscomp.CompilerOptions.LanguageMode;
19  import com.google.javascript.rhino.Node;
20  
21  import java.util.LinkedHashMap;
22  import java.util.Map;
23  
24  /**
25   * Tests for {@link FunctionNames}
26   *
27   */
28  public final class FunctionNamesTest extends CompilerTestCase {
29    private FunctionNames functionNames;
30  
31    public FunctionNamesTest() {
32      this.functionNames = null;
33    }
34  
35    @Override
36    protected CompilerPass getProcessor(Compiler compiler) {
37      functionNames = new FunctionNames(compiler);
38      return functionNames;
39    }
40  
41    public void testFunctionsNamesAndIds() {
42      setAcceptedLanguage(LanguageMode.ECMASCRIPT6);
43      final String jsSource = LINE_JOINER.join(
44          "goog.widget = function(str) {",
45          "  this.member_fn = function() {};",
46          "  local_fn = function() {};",
47          "  (function(a){})(1);",
48          "}",
49          "function foo() {",
50          "  function bar() {}",
51          "}",
52          "literal = {f1 : function(){}, f2 : function(){}};",
53          "goog.array.map(arr, function named(){});",
54          "goog.array.map(arr, function(){});",
55          "named_twice = function quax(){};",
56          "recliteral = {l1 : {l2 : function(){}}};",
57          "namedliteral = {n1 : function litnamed(){}};",
58          "namedrecliteral = {n1 : {n2 : function reclitnamed(){}}};",
59          "numliteral = {1 : function(){}};",
60          "recnumliteral = {1 : {a : function(){}}};",
61          "literalWithShorthand = {shorthandF1(){}, shorthandF2(){}};",
62          "class Klass{ constructor(){} method(){}}",
63          "KlassExpression = class{ constructor(){} method(){}}",
64          "var KlassExpressionToVar = class{ constructor(){} method(){}}",
65          "class KlassWithStaticMethod{ static staticMethod(){}}");
66  
67      testSame(jsSource);
68  
69      final Map<Integer, String> idNameMap = new LinkedHashMap<>();
70      int count = 0;
71      for (Node f : functionNames.getFunctionNodeList()) {
72        int id = functionNames.getFunctionId(f);
73        String name = functionNames.getFunctionName(f);
74        idNameMap.put(id, name);
75        count++;
76      }
77  
78      assertEquals("Unexpected number of functions", 25, count);
79  
80      final Map<Integer, String> expectedMap = new LinkedHashMap<>();
81  
82      expectedMap.put(0, "goog.widget.member_fn");
83      expectedMap.put(1, "goog.widget::local_fn");
84      expectedMap.put(2, "goog.widget::<anonymous>");
85      expectedMap.put(3, "goog.widget");
86      expectedMap.put(4, "foo::bar");
87      expectedMap.put(5, "foo");
88      expectedMap.put(6, "literal.f1");
89      expectedMap.put(7, "literal.f2");
90      expectedMap.put(8, "named");
91      expectedMap.put(9, "<anonymous>");
92      expectedMap.put(10, "quax");
93      expectedMap.put(11, "recliteral.l1.l2");
94      expectedMap.put(12, "litnamed");
95      expectedMap.put(13, "reclitnamed");
96      expectedMap.put(14, "numliteral.__2");
97      expectedMap.put(15, "recnumliteral.__3.a");
98      expectedMap.put(16, "literalWithShorthand.shorthandF1");
99      expectedMap.put(17, "literalWithShorthand.shorthandF2");
100     expectedMap.put(18, "Klass.constructor");
101     expectedMap.put(19, "Klass.method");
102     expectedMap.put(20, "KlassExpression.constructor");
103     expectedMap.put(21, "KlassExpression.method");
104     expectedMap.put(22, "KlassExpressionToVar.constructor");
105     expectedMap.put(23, "KlassExpressionToVar.method");
106     expectedMap.put(24, "KlassWithStaticMethod.staticMethod");
107     assertEquals("Function id/name mismatch",
108                  expectedMap, idNameMap);
109   }
110 }
