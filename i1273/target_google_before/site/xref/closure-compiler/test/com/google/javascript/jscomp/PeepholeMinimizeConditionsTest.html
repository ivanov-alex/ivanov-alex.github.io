49      return peepholePass;

69    // TODO(user): This is same as fold() except it uses string comparison. Any

71    private static void assertResultString(String js, String expected,
72        boolean normalize) {
73      PeepholeMinimizeConditionsTest scTest
74          = new PeepholeMinimizeConditionsTest(false);
75  
76      if (normalize) {
77        scTest.enableNormalize();
78      } else {
79        scTest.disableNormalize();
80      }
81  
82      scTest.test(js, expected);
83    }

85    /** Check that removing blocks with 1 child works */
86    public void testFoldOneChildBlocks() {
87      late = false;
88      fold("function f(){if(x)a();x=3}",
89          "function f(){x&&a();x=3}");
90      fold("function f(){if(x){a()}x=3}",
91          "function f(){x&&a();x=3}");
92      fold("function f(){if(x){return 3}}",
93          "function f(){if(x)return 3}");
94      fold("function f(){if(x){a()}}",
95          "function f(){x&&a()}");
96      fold("function f(){if(x){throw 1}}", "function f(){if(x)throw 1;}");
97  
98      // Try it out with functions
99      fold("function f(){if(x){foo()}}", "function f(){x&&foo()}");
100     fold("function f(){if(x){foo()}else{bar()}}",
101          "function f(){x?foo():bar()}");
102 
103     // Try it out with properties and methods
104     fold("function f(){if(x){a.b=1}}", "function f(){if(x)a.b=1}");
105     fold("function f(){if(x){a.b*=1}}", "function f(){x&&(a.b*=1)}");
106     fold("function f(){if(x){a.b+=1}}", "function f(){x&&(a.b+=1)}");
107     fold("function f(){if(x){++a.b}}", "function f(){x&&++a.b}");
108     fold("function f(){if(x){a.foo()}}", "function f(){x&&a.foo()}");
109 
110     // Try it out with throw/catch/finally [which should not change]
111     fold("function f(){try{foo()}catch(e){bar(e)}finally{baz()}}",
112          "function f(){try{foo()}catch(e){bar(e)}finally{baz()}}");
113 
114     // Try it out with switch statements
115     fold("function f(){switch(x){case 1:break}}",
116          "function f(){switch(x){case 1:break}}");
117 
118     // Do while loops stay in a block if that's where they started
119     fold("function f(){if(e1){do foo();while(e2)}else foo2()}",
120          "function f(){if(e1){do foo();while(e2)}else foo2()}");
121     // Test an obscure case with do and while
122     fold("if(x){do{foo()}while(y)}else bar()",
123          "if(x){do foo();while(y)}else bar()");
124 
125     // Play with nested IFs
126     fold("function f(){if(x){if(y)foo()}}",
127          "function f(){x && (y && foo())}");
128     fold("function f(){if(x){if(y)foo();else bar()}}",
129          "function f(){x&&(y?foo():bar())}");
130     fold("function f(){if(x){if(y)foo()}else bar()}",
131          "function f(){x?y&&foo():bar()}");
132     fold("function f(){if(x){if(y)foo();else bar()}else{baz()}}",
133          "function f(){x?y?foo():bar():baz()}");
134 
135     fold("if(e1){while(e2){if(e3){foo()}}}else{bar()}",
136          "if(e1)while(e2)e3&&foo();else bar()");
137 
138     fold("if(e1){with(e2){if(e3){foo()}}}else{bar()}",
139          "if(e1)with(e2)e3&&foo();else bar()");
140 
141     fold("if(a||b){if(c||d){var x;}}", "if(a||b)if(c||d)var x");
142     fold("if(x){ if(y){var x;}else{var z;} }",
143          "if(x)if(y)var x;else var z");
144 
145     // NOTE - technically we can remove the blocks since both the parent
146     // and child have elses. But we don't since it causes ambiguities in
147     // some cases where not all descendent ifs having elses
148     fold("if(x){ if(y){var x;}else{var z;} }else{var w}",
149          "if(x)if(y)var x;else var z;else var w");
150     fold("if (x) {var x;}else { if (y) { var y;} }",
151          "if(x)var x;else if(y)var y");
152 
153     // Here's some of the ambiguous cases
154     fold("if(a){if(b){f1();f2();}else if(c){f3();}}else {if(d){f4();}}",
155          "if(a)if(b){f1();f2()}else c&&f3();else d&&f4()");
156 
157     fold("function f(){foo()}", "function f(){foo()}");
158     fold("switch(x){case y: foo()}", "switch(x){case y:foo()}");
159     fold("try{foo()}catch(ex){bar()}finally{baz()}",
160          "try{foo()}catch(ex){bar()}finally{baz()}");
161   }

473   public void testSubsituteReturn() {
474 
475     fold("function f() { while(x) { return }}",
476          "function f() { while(x) { break }}");
477 
478     foldSame("function f() { while(x) { return 5 } }");
479 
480     foldSame("function f() { a: { return 5 } }");
481 
482     fold("function f() { while(x) { return 5}  return 5}",
483          "function f() { while(x) { break }    return 5}");
484 
485     fold("function f() { while(x) { return x}  return x}",
486          "function f() { while(x) { break }    return x}");
487 
488     fold("function f() { while(x) { if (y) { return }}}",
489          "function f() { while(x) { if (y) { break  }}}");
490 
491     fold("function f() { while(x) { if (y) { return }} return}",
492          "function f() { while(x) { if (y) { break  }}}");
493 
494     fold("function f() { while(x) { if (y) { return 5 }} return 5}",
495          "function f() { while(x) { if (y) { break    }} return 5}");
496 
497     // It doesn't matter if x is changed between them. We are still returning
498     // x at whatever x value current holds. The whole x = 1 is skipped.
499     fold("function f() { while(x) { if (y) { return x } x = 1} return x}",
500          "function f() { while(x) { if (y) { break    } x = 1} return x}");
501 
502     // RemoveUnreachableCode would take care of the useless breaks.
503     fold("function f() { while(x) { if (y) { return x } return x} return x}",
504          "function f() { while(x) { if (y) {} break }return x}");
505 
506     // A break here only breaks out of the inner loop.
