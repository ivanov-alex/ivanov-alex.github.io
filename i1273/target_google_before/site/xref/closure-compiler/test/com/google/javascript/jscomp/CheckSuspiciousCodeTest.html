42    public void testSuspiciousSemi() {
43      final DiagnosticType e = CheckSuspiciousCode.SUSPICIOUS_SEMICOLON;
44  
45      testSame("if(x()) x = y;");
46      testWarning("if(x()); x = y;", e);  // I've had this bug, damned ;
47      testSame("if(x()){} x = y;");
48  
49      testSame("if(x()) x = y; else y=z;");
50      testWarning("if(x()); else y=z;", e);
51      testSame("if(x()){} else y=z;");
52      testWarning("if(x()) x = y; else;", e);
53      testSame("if(x()) x = y; else {}");
54  
55      testSame("while(x()) x = y;");
56      testWarning("while(x()); x = y;", e);
57      testSame("while(x()){} x = y;");
58      testWarning("while(x()); {x = y}", e);
59      testSame("while(x()){} {x = y}");
60  
61      testSame("for(;;) x = y;");
62      testWarning("for(;;); x = y;", e);
63      testSame("for(;;){} x = y;");
64      testSame("for(x in y) x = y;");
65      testWarning("for(x in y); x = y;", e);
66      testSame("for(x in y){} x = y;");
67  
68      testSameEs6("var y = [1, 2, 3]; for(x of y) console.log(x);");
69      testWarningEs6("var y = [1, 2, 3]; for(x of y); console.log(x);", e);
70      testSameEs6("var y = [1, 2, 3]; for(x of y){} console.log(x);");
71    }

73    public void testSuspiciousIn() {
74      testWarning("'foo' in 1", CheckSuspiciousCode.SUSPICIOUS_IN_OPERATOR);
75      testWarning("'foo' in 'test'", CheckSuspiciousCode.SUSPICIOUS_IN_OPERATOR);
76      testWarning("'foo' in NaN", CheckSuspiciousCode.SUSPICIOUS_IN_OPERATOR);
77      testWarning("'foo' in undefined", CheckSuspiciousCode.SUSPICIOUS_IN_OPERATOR);
78      testWarning("'foo' in Infinity", CheckSuspiciousCode.SUSPICIOUS_IN_OPERATOR);
79      testWarning("'foo' in true", CheckSuspiciousCode.SUSPICIOUS_IN_OPERATOR);
80      testWarning("'foo' in false", CheckSuspiciousCode.SUSPICIOUS_IN_OPERATOR);
81      testWarning("'foo' in null", CheckSuspiciousCode.SUSPICIOUS_IN_OPERATOR);
82      testWarning("'foo' in !Object", CheckSuspiciousCode.SUSPICIOUS_IN_OPERATOR);
83      testSame("'foo' in Object");
84      testSame("'foo' in {}");
85    }

87    public void testForOf() {
88      testSameEs6("var y = [1, 2, 3]; for (var x of y) console.log(x);");
89      testSameEs6("var y = [1, 2, 3]; for (var x of 'test') console.log(x);");
90      testSameEs6("for (var x of 123) console.log(x);");
91      testSameEs6("for (var x of false) console.log(x);");
92      testSameEs6("for (var x of true) console.log(x);");
93      testSameEs6("for (var x of undefined) console.log(x);");
94      testSameEs6("for (var x of NaN) console.log(x);");
95      testSameEs6("for (var x of Infinity) console.log(x);");
96      testSameEs6("for (var x of null) console.log(x);");
97    }

147   public void testInstanceOf() {
148     testReportInstanceOf("''", "String");
149     testReportInstanceOf("4", "Number");
150     testReportInstanceOf("-4", "Number");
151     testReportInstanceOf("null", "Number");
152     testReportInstanceOf("true", "Boolean");
153     testReportInstanceOf("false", "Boolean");
154     testReportInstanceOf("!true", "Boolean");
155     testReportInstanceOf("undefined", "Number");
156     testReportInstanceOf("Infinity", "Number");
157     testReportInstanceOf("NaN", "Number");
158     testReportInstanceOf(
159         "/** @constructor */ function Foo() {}; var foo = new Foo();"
160         + "!foo", "Foo");
161 
162     testReportInstanceOf("(4 + 5)", "Number");
163     testReportInstanceOf("('a' + 'b')", "String");
164     testReportInstanceOf("('a' + 5)", "String");
165 
166     // The following five are from "Wat — Destroy All Software Talks"
167     testReportInstanceOf("([] + [])", "String");
168     testReportInstanceOf("([] + {})", "String");
169     testReportInstanceOf("({} + [])", "String");
170     testReportInstanceOf("({c:1} + [])", "String");
171     testReportInstanceOf("({} + {})", "Number"); // NaN
172 
173     testReportInstanceOf("!''", "String");
174     testReportInstanceOf("!4", "Number");
175     testReportInstanceOf("!(new Boolean(true))", "Boolean");
176     testReportInstanceOf("!(new Object())", "Object");
177     testReportInstanceOf("!Object.prototype", "Object");
178     testReportInstanceOf("!Function", "Object");
179     testReportInstanceOf("!func()", "String");
180     testReportInstanceOf("!({})", "Object");
181     testReportInstanceOf("/** @constructor */ function Foo() {"
182         + "!this", "Foo;"
183         + "}");
184 
185     testSame("new String('') instanceof String");
186     testSame("new Number(4) instanceof Number");
187     testSame("var a = new Number(4); a instanceof Number");
188     testSame("new Boolean(true) instanceof Boolean");
189     testSame("new Object() instanceof Object");
190     testSame("Object.prototype instanceof Object");
191     testSame("Function instanceof Object");
192     testSame("func() instanceof String");
193     testSame("({}) instanceof Object");
194     testSame("/** @constructor */ function Foo() {"
195         + " var a = this instanceof Foo; }");
196 
197     testSameEs6("(()=>42) instanceof Function");
198     testSameEs6("class Person{} Person instanceof Function");
199     testSameEs6(LINE_JOINER.join(
200         "class Person{}",
201         "var peter = new Person();",
202         "peter instanceof Person"));
203     testSameEs6("taggedTemplate`${tagged}Temp` instanceof Function");
204   }

208         CheckSuspiciousCode.SUSPICIOUS_INSTANCEOF_LEFT_OPERAND);
