17  package com.google.javascript.jscomp;
18  
19  import static com.google.javascript.jscomp.CheckRequiresForConstructors.DUPLICATE_REQUIRE_WARNING;
20  import static com.google.javascript.jscomp.CheckRequiresForConstructors.EXTRA_REQUIRE_WARNING;
21  
22  /**
23   * Tests for the "extra requires" check in {@link CheckRequiresForConstructors}.
24   */
25  public final class ExtraRequireTest extends Es6CompilerTestCase {
26    public ExtraRequireTest() {
27      super();
28      enableRewriteClosureCode();
29    }
30  
31    @Override
32    protected CompilerOptions getOptions(CompilerOptions options) {
33      options.setWarningLevel(DiagnosticGroups.EXTRA_REQUIRE, CheckLevel.ERROR);
34      return super.getOptions(options);
35    }
36  
37    @Override
38    protected CompilerPass getProcessor(Compiler compiler) {
39      return new CheckRequiresForConstructors(compiler);
40    }
41  
42    public void testNoWarning() {
43      testSame("goog.require('foo.Bar'); var x = new foo.Bar();");
44      testSameEs6("goog.require('foo.Bar'); let x = new foo.Bar();");
45      testSameEs6("goog.require('foo.Bar'); const x = new foo.Bar();");
46      testSame("goog.require('foo.Bar'); /** @type {foo.Bar} */ var x;");
47      testSame("goog.require('foo.Bar'); /** @type {Array<foo.Bar>} */ var x;");
48      testSame("goog.require('foo.Bar'); var x = new foo.Bar.Baz();");
49      testSame("goog.require('foo.bar'); var x = foo.bar();");
50      testSame("goog.require('foo.bar'); var x = /** @type {foo.bar} */ (null);");
51      testSame("goog.require('foo.bar'); function f(/** foo.bar */ x) {}");
52      testSame("goog.require('foo.bar'); alert(foo.bar.baz);");
53      testSame("/** @suppress {extraRequire} */ goog.require('foo.bar');");
54      test("goog.require('foo.bar'); goog.scope(function() { var bar = foo.bar; alert(bar); });",
55          "goog.require('foo.bar'); alert(foo.bar);");
56      testSame("goog.require('foo'); foo();");
57      testSame("/** @suppress {extraRequire} */ var bar = goog.require('foo.bar');");
58    }
59  
60    public void testNoWarning_InnerClassInExtends() {
61      String js =
62          LINE_JOINER.join(
63              "var goog = {};",
64              "goog.require('goog.foo.Bar');",
65              "",
66              "/** @constructor @extends {goog.foo.Bar.Inner} */",
67              "function SubClass() {}");
68      testSame(js);
69    }
70  
71    public void testWarning() {
72      testError("goog.require('foo.bar');", EXTRA_REQUIRE_WARNING);
73      // The local var "bar" is unused so after goog.scope rewriting, foo.bar is unused.
74      testError("goog.require('foo.bar'); goog.scope(function() { var bar = foo.bar; });",
75          EXTRA_REQUIRE_WARNING);
76  
77      testErrorEs6(LINE_JOINER.join(
78          "goog.require('Bar');",
79          "function func( {a} ){}",
80          "func( {a: 1} );"), EXTRA_REQUIRE_WARNING);
81      testErrorEs6(LINE_JOINER.join(
82          "goog.require('Bar');",
83          "function func( a = 1 ){}",
84          "func(42);"), EXTRA_REQUIRE_WARNING);
85  
86      testError(
87          LINE_JOINER.join(
88              "goog.require('Bar');",
89              "goog.require('Bar');",
90              "var b = new Bar();"),
91          DUPLICATE_REQUIRE_WARNING);
92    }
93  
94    public void testNoWarningMultipleFiles() {
95      String[] js = new String[] {
96        "goog.require('Foo'); var foo = new Foo();",
97        "goog.require('Bar'); var bar = new Bar();"
98      };
99      testSame(js);
100   }
