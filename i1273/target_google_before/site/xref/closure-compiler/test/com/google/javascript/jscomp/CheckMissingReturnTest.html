37    public void testMissingReturn() {
38      // Requires control flow analysis.
39      testMissing("if (a) { return 1; }");
40  
41      // Switch statement.
42      testMissing("switch(1) { case 12: return 5; }");
43  
44      // Test try catch finally.
45      testMissing("try { foo() } catch (e) { return 5; } finally { }");
46  
47      // Nested scope.
48      testMissing("/** @return {number} */ function f() { var x; }; return 1;");
49      testMissing("/** @return {number} */ function f() { return 1; };");
50    }

52    public void testReturnNotMissing()  {
53      // Empty function body. Ignore this case. The remainder of the functions in
54      // this test have non-empty bodies.
55      testNotMissing("");
56  
57      // Simple cases.
58      testSame("function f() { var x; }");
59      testNotMissing("return 1;");
60  
61      // Returning void and undefined.
62      testNotMissing("void", "var x;");
63      testNotMissing("undefined", "var x;");
64  
65      // Returning a union that includes void or undefined.
66      testNotMissing("number|undefined", "var x;");
67      testNotMissing("number|void", "var x;");
68      testNotMissing("(number,void)", "var x;");
69      testNotMissing("(number,undefined)", "var x;");
70      testNotMissing("*", "var x;");
71  
72      // Test try catch finally.
73      testNotMissing("try { return foo() } catch (e) { } finally { }");
74      testNotMissing("try {throw e;} catch (e) { return foo() } finally { }");
75      testNotMissing("try {} catch (e) {} finally {return foo()};");
76  
77      // Nested function.
78      testNotMissing(
79          "/** @return {number} */ function f() { return 1; }; return 1;");
80  
81      // Strange tests that come up when reviewing closure code.
82      testNotMissing("try { return 12; } finally { return 62; }");
83      testNotMissing("try { } finally { return 1; }");
84      testNotMissing("switch(1) { default: return 1; }");
85      testNotMissing("switch(g) { case 1: return 1; default: return 2; }");
86    }

88    public void testFinallyStatements() {
89      // The control flow analysis (CFA) treats finally blocks somewhat strangely.
90      // The CFA might indicate that a finally block implicitly returns. However,
91      // if entry into the finally block is normally caused by an explicit return
92      // statement, then a return statement isn't missing:
93      //
94      // try {
95      //   return 1;
96      // } finally {
97      //   // CFA determines implicit return. However, return not missing
98      //   // because of return statement in try block.
99      // }
100     //
101     // Hence extra tests are warranted for various cases involving finally
102     // blocks.
103 
104     // Simple finally case.
105     testNotMissing("try { return 1; } finally { }");
106     testNotMissing("try { } finally { return 1; }");
107     testMissing("try { } finally { }");
108 
109     // Cycles in the CFG within the finally block were causing problems before.
110     testNotMissing("try { return 1; } finally { while (true) { } }");
111     testMissing("try { } finally { while (x) { } }");
112     testMissing("try { } finally { while (x) { if (x) { break; } } }");
113     testNotMissing(
114         "try { return 2; } finally { while (x) { if (x) { break; } } }");
115 
116     // Test various cases with nested try statements.
117     testMissing("try { } finally { try { } finally { } }");
118     testNotMissing("try { } finally { try { return 1; } finally { } }");
119     testNotMissing("try { return 1; } finally { try { } finally { } }");
120 
121     // Calling a function potentially causes control flow to transfer to finally
122     // block. However, the function will not return in this case as the
123     // exception will unwind the stack. Hence this function isn't missing a
124     // return statement (i.e., the running program will not expect a return
125     // value from the function if an exception is thrown).
126     testNotMissing("try { g(); return 1; } finally { }");
127 
128     // Closures within try ... finally affect missing return statement analysis
129     // because of the nested scopes. The following tests check for missing
130     // return statements in the three possible configurations: both scopes
131     // return; enclosed doesn't return; enclosing doesn't return.
132     testNotMissing(
133         "try {" +
134         "   /** @return {number} */ function f() {" +
135         "       try { return 1; }" +
136         "       finally { }" +
137         "   };" +
138         "   return 1;" +
139         "}" +
140         "finally { }");
141     testMissing(
142         "try {" +
143         "   /** @return {number} */ function f() {" +
144         "       try { }" +
145         "       finally { }" +
146         "   };" +
147         "   return 1;" +
148         "}" +
149         "finally { }");
150     testMissing(
151         "try {" +
152         "   /** @return {number} */ function f() {" +
153         "       try { return 1; }" +
154         "       finally { }" +
155         "   };" +
156         "}" +
157         "finally { }");
158   }

160   public void testKnownConditions() {
161     testNotMissing("if (true) return 1");
162     testMissing("if (true) {} else {return 1}");
163 
164     testMissing("if (false) return 1");
165     testNotMissing("if (false) {} else {return 1}");
166 
167     testNotMissing("if (1) return 1");
168     testMissing("if (1) {} else {return 1}");
169 
170     testMissing("if (0) return 1");
171     testNotMissing("if (0) {} else {return 1}");
172 
173     testNotMissing("if (3) return 1");
174     testMissing("if (3) {} else {return 1}");
175   }

177   public void testKnownWhileLoop() {
178     testNotMissing("while (1) return 1");
179     testNotMissing("while (1) { if (x) {return 1} else {return 1}}");
180     testNotMissing("while (0) {} return 1");
181 
182     // TODO(user): The current algorithm will not detect this case. It is
183     // still computable in most cases.
184     testNotMissing("while (1) {} return 0");
185     testMissing("while (false) return 1");
186 
187     // Not known.
188     testMissing("while(x) { return 1 }");
189   }

192     testMissing("if (a) { } else { while (1) {return 1} }");

211   public void testClosureAsserts() {
212     String closureDefs =
213         "/** @const */ var goog = {};\n" +
214         "goog.asserts = {};\n" +
215         "goog.asserts.fail = function(x) {};";

217     testNotMissing(closureDefs + "goog.asserts.fail('');");

223   public void testInfiniteLoops() {
224     testNotMissing("while (true) { x = y; if (x === 0) { return 1; } }");
225     testNotMissing("for (;true;) { x = y; if (x === 0) { return 1; } }");
226     testNotMissing("for (;;) { x = y; if (x === 0) { return 1; } }");
227   }

229   private static String createFunction(String returnType, String body) {
230     return "/** @return {" + returnType + "} */ function foo() {" + body + "}";
231   }
232 
233   private static String createShorthandFunctionInObjLit(
234       String returnType, String body) {
235     return LINE_JOINER.join(
236         "var obj = {",
237         "  /** @return {" + returnType + "} */",
238         "  foo() {", body, "}",
239         "}");
240   }
241 
242   private void testMissingInTraditionalFunction(String returnType, String body) {
243     String js = createFunction(returnType, body);
244     testError(js, CheckMissingReturn.MISSING_RETURN_STATEMENT);
245   }
246 
247   private void testNotMissingInTraditionalFunction(String returnType, String body) {
248     testSame(createFunction(returnType, body));
249   }
250 
251   private void testMissingInShorthandFunction(String returnType, String body) {
252     setAcceptedLanguage(LanguageMode.ECMASCRIPT6);
253     String js = createShorthandFunctionInObjLit(returnType, body);
254     testError(js, CheckMissingReturn.MISSING_RETURN_STATEMENT);
255   }
256 
257   private void testNotMissingInShorthandFunction(String returnType, String body) {
258     setAcceptedLanguage(LanguageMode.ECMASCRIPT6);
259     testSame(createShorthandFunctionInObjLit(returnType, body));
260   }
261 
262   private void testMissing(String returnType, String body) {
263     testMissingInTraditionalFunction(returnType, body);
264     testMissingInShorthandFunction(returnType, body);
265   }
266 
267   private void testNotMissing(String returnType, String body) {
268     testNotMissingInTraditionalFunction(returnType, body);
269     testNotMissingInShorthandFunction(returnType, body);
270   }

272   /** Creates function with return type {number} */
273   private void testNotMissing(String body) {
274     testNotMissing("number", body);
275   }

277   /** Creates function with return type {number} */
278   private void testMissing(String body) {
279     testMissing("number", body);
