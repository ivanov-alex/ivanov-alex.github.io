48    public void testBreakOptimization() throws Exception {
49      fold("f:{if(true){a();break f;}else;b();}",
50           "f:{if(true){a()}else{b()}}");
51      fold("f:{if(false){a();break f;}else;b();break f;}",
52           "f:{if(false){a()}else{b()}}");
53      fold("f:{if(a()){b();break f;}else;c();}",
54           "f:{if(a()){b();}else{c();}}");
55      fold("f:{if(a()){b()}else{c();break f;}}",
56           "f:{if(a()){b()}else{c();}}");
57      fold("f:{if(a()){b();break f;}else;}",
58           "f:{if(a()){b();}else;}");
59      fold("f:{if(a()){break f;}else;}",
60           "f:{if(a()){}else;}");
61  
62      fold("f:while(a())break f;",
63           "f:while(a())break f");
64      foldSame("f:for(x in a())break f");
65  
66      fold("f:{while(a())break;}",
67           "f:{while(a())break;}");
68      foldSame("f:{for(x in a())break}");
69  
70      fold("f:try{break f;}catch(e){break f;}",
71           "f:try{}catch(e){}");
72      fold("f:try{if(a()){break f;}else{break f;} break f;}catch(e){}",
73           "f:try{if(a()){}else{}}catch(e){}");
74  
75      fold("f:g:break f",
76           "");
77      fold("f:g:{if(a()){break f;}else{break f;} break f;}",
78           "f:g:{if(a()){}else{}}");
79    }

81    public void testFunctionReturnOptimization() throws Exception {
82      fold("function f(){if(a()){b();if(c())return;}}",
83           "function f(){if(a()){b();if(c());}}");
84      fold("function f(){if(x)return; x=3; return; }",
85           "function f(){if(x); else x=3}");
86      fold("function f(){if(true){a();return;}else;b();}",
87           "function f(){if(true){a();}else{b();}}");
88      fold("function f(){if(false){a();return;}else;b();return;}",
89           "function f(){if(false){a();}else{b();}}");
90      fold("function f(){if(a()){b();return;}else;c();}",
91           "function f(){if(a()){b();}else{c();}}");
92      fold("function f(){if(a()){b()}else{c();return;}}",
93           "function f(){if(a()){b()}else{c();}}");
94      fold("function f(){if(a()){b();return;}else;}",
95           "function f(){if(a()){b();}else;}");
96      fold("function f(){if(a()){return;}else{return;} return;}",
97           "function f(){if(a()){}else{}}");
98      fold("function f(){if(a()){return;}else{return;} b();}",
99           "function f(){if(a()){}else{return;b()}}");
100     fold("function f(){ if (x) return; if (y) return; if (z) return; w(); }",
101         " function f() {" +
102         "   if (x) {} else { if (y) {} else { if (z) {} else w(); }}" +
103         " }");
104 
105     fold("function f(){while(a())return;}",
106          "function f(){while(a())return}");
107     foldSame("function f(){for(x in a())return}");
108 
109     fold("function f(){while(a())break;}",
110          "function f(){while(a())break}");
111     foldSame("function f(){for(x in a())break}");
112 
113     fold("function f(){try{return;}catch(e){throw 9;}finally{return}}",
114          "function f(){try{}catch(e){throw 9;}finally{return}}");
115     foldSame("function f(){try{throw 9;}finally{return;}}");
116 
117     fold("function f(){try{return;}catch(e){return;}}",
118          "function f(){try{}catch(e){}}");
119     fold("function f(){try{if(a()){return;}else{return;} return;}catch(e){}}",
120          "function f(){try{if(a()){}else{}}catch(e){}}");
121 
122     fold("function f(){g:return}",
123          "function f(){}");
124     fold("function f(){g:if(a()){return;}else{return;} return;}",
125          "function f(){g:if(a()){}else{}}");
126     fold("function f(){try{g:if(a()){throw 9;} return;}finally{return}}",
127          "function f(){try{g:if(a()){throw 9;}}finally{return}}");
128   }

130   public void testWhileContinueOptimization() throws Exception {
131     fold("while(true){if(x)continue; x=3; continue; }",
132          "while(true)if(x);else x=3");
133     foldSame("while(true){a();continue;b();}");
134     fold("while(true){if(true){a();continue;}else;b();}",
135          "while(true){if(true){a();}else{b()}}");
136     fold("while(true){if(false){a();continue;}else;b();continue;}",
137          "while(true){if(false){a()}else{b();}}");
138     fold("while(true){if(a()){b();continue;}else;c();}",
139          "while(true){if(a()){b();}else{c();}}");
140     fold("while(true){if(a()){b();}else{c();continue;}}",
141          "while(true){if(a()){b();}else{c();}}");
142     fold("while(true){if(a()){b();continue;}else;}",
143          "while(true){if(a()){b();}else;}");
144     fold("while(true){if(a()){continue;}else{continue;} continue;}",
145          "while(true){if(a()){}else{}}");
146     fold("while(true){if(a()){continue;}else{continue;} b();}",
147          "while(true){if(a()){}else{continue;b();}}");
148 
149     fold("while(true)while(a())continue;",
150          "while(true)while(a());");
151     fold("while(true)for(x in a())continue",
152          "while(true)for(x in a());");
153 
154     fold("while(true)while(a())break;",
155          "while(true)while(a())break");
156     fold("while(true)for(x in a())break",
157          "while(true)for(x in a())break");
158 
159     fold("while(true){try{continue;}catch(e){continue;}}",
160          "while(true){try{}catch(e){}}");
161     fold("while(true){try{if(a()){continue;}else{continue;}" +
162          "continue;}catch(e){}}",
163          "while(true){try{if(a()){}else{}}catch(e){}}");
164 
165     fold("while(true){g:continue}",
166          "while(true){}");
167     // This case could be improved.
168     fold("while(true){g:if(a()){continue;}else{continue;} continue;}",
169          "while(true){g:if(a());else;}");
170   }

172   public void testDoContinueOptimization() throws Exception {
173     fold("do{if(x)continue; x=3; continue; }while(true)",
174          "do if(x); else x=3; while(true)");
175     foldSame("do{a();continue;b()}while(true)");
176     fold("do{if(true){a();continue;}else;b();}while(true)",
177          "do{if(true){a();}else{b();}}while(true)");
178     fold("do{if(false){a();continue;}else;b();continue;}while(true)",
179          "do{if(false){a();}else{b();}}while(true)");
180     fold("do{if(a()){b();continue;}else;c();}while(true)",
181          "do{if(a()){b();}else{c()}}while(true)");
182     fold("do{if(a()){b();}else{c();continue;}}while(true)",
183          "do{if(a()){b();}else{c();}}while(true)");
184     fold("do{if(a()){b();continue;}else;}while(true)",
185          "do{if(a()){b();}else;}while(true)");
186     fold("do{if(a()){continue;}else{continue;} continue;}while(true)",
187          "do{if(a()){}else{}}while(true)");
188     fold("do{if(a()){continue;}else{continue;} b();}while(true)",
189          "do{if(a()){}else{continue; b();}}while(true)");
190 
191     fold("do{while(a())continue;}while(true)",
192          "do while(a());while(true)");
193     fold("do{for(x in a())continue}while(true)",
194          "do for(x in a());while(true)");
195 
196     fold("do{while(a())break;}while(true)",
197          "do while(a())break;while(true)");
198     fold("do for(x in a())break;while(true)",
199          "do for(x in a())break;while(true)");
200 
201     fold("do{try{continue;}catch(e){continue;}}while(true)",
202          "do{try{}catch(e){}}while(true)");
203     fold("do{try{if(a()){continue;}else{continue;}" +
204          "continue;}catch(e){}}while(true)",
205          "do{try{if(a()){}else{}}catch(e){}}while(true)");
206 
207     fold("do{g:continue}while(true)",
208          "do{}while(true)");
209     // This case could be improved.
210     fold("do{g:if(a()){continue;}else{continue;} continue;}while(true)",
211          "do{g:if(a());else;}while(true)");
212 
213     fold("do { foo(); continue; } while(false)",
214          "do { foo(); } while(false)");
215     fold("do { foo(); break; } while(false)",
216          "do { foo(); } while(false)");
217   }

219   public void testForContinueOptimization() throws Exception {
220     fold("for(x in y){if(x)continue; x=3; continue; }",
221          "for(x in y)if(x);else x=3");
222     foldSame("for(x in y){a();continue;b()}");
223     fold("for(x in y){if(true){a();continue;}else;b();}",
224          "for(x in y){if(true)a();else b();}");
225     fold("for(x in y){if(false){a();continue;}else;b();continue;}",
226          "for(x in y){if(false){a();}else{b()}}");
227     fold("for(x in y){if(a()){b();continue;}else;c();}",
228          "for(x in y){if(a()){b();}else{c();}}");
229     fold("for(x in y){if(a()){b();}else{c();continue;}}",
230          "for(x in y){if(a()){b();}else{c();}}");
231     fold("for(x=0;x<y;x++){if(a()){b();continue;}else;}",
232          "for(x=0;x<y;x++){if(a()){b();}else;}");
233     fold("for(x=0;x<y;x++){if(a()){continue;}else{continue;} continue;}",
234          "for(x=0;x<y;x++){if(a()){}else{}}");
235     fold("for(x=0;x<y;x++){if(a()){continue;}else{continue;} b();}",
236          "for(x=0;x<y;x++){if(a()){}else{continue; b();}}");
237 
238     fold("for(x=0;x<y;x++)while(a())continue;",
239          "for(x=0;x<y;x++)while(a());");
240     fold("for(x=0;x<y;x++)for(x in a())continue",
241          "for(x=0;x<y;x++)for(x in a());");
242 
243     fold("for(x=0;x<y;x++)while(a())break;",
244          "for(x=0;x<y;x++)while(a())break");
245     foldSame("for(x=0;x<y;x++)for(x in a())break");
246 
247     fold("for(x=0;x<y;x++){try{continue;}catch(e){continue;}}",
248          "for(x=0;x<y;x++){try{}catch(e){}}");
249     fold("for(x=0;x<y;x++){try{if(a()){continue;}else{continue;}" +
250          "continue;}catch(e){}}",
251          "for(x=0;x<y;x++){try{if(a()){}else{}}catch(e){}}");
252 
253     fold("for(x=0;x<y;x++){g:continue}",
254          "for(x=0;x<y;x++){}");
255     // This case could be improved.
256     fold("for(x=0;x<y;x++){g:if(a()){continue;}else{continue;} continue;}",
257          "for(x=0;x<y;x++){g:if(a());else;}");
258   }
