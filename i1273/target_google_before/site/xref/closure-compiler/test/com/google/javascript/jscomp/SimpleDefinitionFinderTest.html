50    public void testDefineNumber() {
51      checkDefinitionsInJs(
52          "var a = 1",
53          ImmutableSet.of("DEF NAME a -> NUMBER"));
54  
55      checkDefinitionsInJs(
56          "a = 1",
57          ImmutableSet.of("DEF NAME a -> NUMBER"));
58  
59      checkDefinitionsInJs(
60          "a.b = 1",
61          ImmutableSet.of("DEF GETPROP a.b -> NUMBER"));
62  
63      // getelem expressions are invisible to the definition gatherer.
64      checkDefinitionsInJs(
65          "a[\"b\"] = 1",
66          ImmutableSet.<String>of());
67  
68      checkDefinitionsInJs(
69          "f().b = 1",
70          ImmutableSet.of("DEF GETPROP null -> NUMBER"));
71  
72      checkDefinitionsInJs(
73          "({a : 1}); o.a",
74          ImmutableSet.of("DEF STRING_KEY null -> NUMBER",
75                          "USE GETPROP o.a -> [NUMBER]"));
76  
77      // TODO(johnlenz): Fix this.
78      checkDefinitionsInJs(
79        "({'a' : 1}); o['a']",
80        ImmutableSet.of("DEF STRING_KEY null -> NUMBER"));
81  
82      checkDefinitionsInJs(
83        "({1 : 1}); o[1]",
84        ImmutableSet.of("DEF STRING_KEY null -> NUMBER"));
85  
86      checkDefinitionsInJs(
87          "var a = {b : 1}; a.b",
88          ImmutableSet.of("DEF NAME a -> <null>",
89                          "DEF STRING_KEY null -> NUMBER",
90                          "USE NAME a -> [<null>]",
91                          "USE GETPROP a.b -> [NUMBER]"));
92    }

368   @Override
369   protected CompilerPass getProcessor(Compiler compiler) {
370     return new SimpleDefinitionEnumerator(compiler);
371   }
372 
373   /**
374    * Run SimpleDefinitionFinder, then gather a list of definitions.
375    */
376   private class SimpleDefinitionEnumerator
377       extends AbstractPostOrderCallback implements CompilerPass {
378     private final SimpleDefinitionFinder passUnderTest;
379     private final Compiler compiler;
380 
381     SimpleDefinitionEnumerator(Compiler compiler) {
382       this.passUnderTest = new SimpleDefinitionFinder(compiler);
383       this.compiler = compiler;
384     }
385 
386     @Override
387     public void process(Node externs, Node root) {
388       passUnderTest.process(externs, root);
389       NodeTraversal.traverseEs6(compiler, externs, this);
390       NodeTraversal.traverseEs6(compiler, root, this);
391 
392       for (DefinitionSite defSite : passUnderTest.getDefinitionSites()) {
393         Node node = defSite.node;
394         Definition definition = defSite.definition;
395         StringBuilder sb = new StringBuilder();
396         sb.append("DEF ");
397         sb.append(Token.name(node.getType()));
398         sb.append(" ");
399         sb.append(node.getQualifiedName());
400         sb.append(" -> ");
401 
402         if (definition.isExtern()) {
403           sb.append("EXTERN ");
404         }
405 
406         Node rValue = definition.getRValue();
407         if (rValue != null) {
408           sb.append(Token.name(rValue.getType()));
409         } else {
410           sb.append("<null>");
411         }
412 
413         found.add(sb.toString());
414       }
415 
416     }
417 
418     @Override
419     public void visit(NodeTraversal traversal, Node node, Node parent) {
420       Collection<Definition> defs =
421           passUnderTest.getDefinitionsReferencedAt(node);
422       if (defs != null) {
423         StringBuilder sb = new StringBuilder();
424         sb.append("USE ");
425         sb.append(Token.name(node.getType()));
426         sb.append(" ");
427         sb.append(node.getQualifiedName());
428         sb.append(" -> ");
429         Multiset<String> defstrs = TreeMultiset.create();
430         for (Definition def : defs) {
431           String defstr;
432 
433           Node rValue = def.getRValue();
434           if (rValue != null) {
435             defstr = Token.name(rValue.getType());
436           } else {
437             defstr = "<null>";
438           }
439 
440           if (def.isExtern()) {
441             defstr = "EXTERN " + defstr;
442           }
443 
444           defstrs.add(defstr);
445         }
446 
447         sb.append(defstrs);
448         found.add(sb.toString());
449       }
450     }
