25    public static class Sample {
26  
27      public final String name;
28      public final String contentId;
29      public final String uri;
30      public final int type;
31  
32      public Sample(String name, String uri, int type) {
33        this(name, name.toLowerCase(Locale.US).replaceAll("\\s", ""), uri, type);
34      }
35  
36      public Sample(String name, String contentId, String uri, int type) {
37        this.name = name;
38        this.contentId = contentId;
39        this.uri = uri;
40        this.type = type;
41      }
42  
43    }
