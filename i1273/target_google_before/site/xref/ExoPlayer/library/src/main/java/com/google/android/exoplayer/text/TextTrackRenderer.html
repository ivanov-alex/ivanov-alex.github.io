38  /**
39   * A {@link TrackRenderer} for subtitles. Text is parsed from sample data using a
40   * {@link SubtitleParser}. The actual rendering of each line of text is delegated to a
41   * {@link TextRenderer}.
42   * <p>
43   * If no {@link SubtitleParser} instances are passed to the constructor, the subtitle type will be
44   * detected automatically for the following supported formats:
45   *
46   * <ul>
47   * <li>WebVTT ({@link com.google.android.exoplayer.text.webvtt.WebvttParser})</li>
48   * <li>TTML
49   * ({@link com.google.android.exoplayer.text.ttml.TtmlParser})</li>
50   * <li>SubRip
51   * ({@link com.google.android.exoplayer.text.subrip.SubripParser})</li>
52   * <li>TX3G
53   * ({@link com.google.android.exoplayer.text.tx3g.Tx3gParser})</li>
54   * </ul>
55   *
56   * <p>To override the default parsers, pass one or more {@link SubtitleParser} instances to the
57   * constructor. The first {@link SubtitleParser} that returns {@code true} from
58   * {@link SubtitleParser#canParse(String)} will be used.
59   */
60  @TargetApi(16)
61  public final class TextTrackRenderer extends TrackRenderer implements Callback {
62  
63    private static final int MSG_UPDATE_OVERLAY = 0;
64  
65    /**
66     * Default parser classes in priority order. They are referred to indirectly so that it is
67     * possible to remove unused parsers.
68     */
69    private static final List<Class<? extends SubtitleParser>> DEFAULT_PARSER_CLASSES;
70    static {
71      DEFAULT_PARSER_CLASSES = new ArrayList<>();
72      // Load parsers using reflection so that they can be deleted cleanly.
73      // Class.forName(<class name>) appears for each parser so that automated tools like proguard
74      // can detect the use of reflection (see http://proguard.sourceforge.net/FAQ.html#forname).
75      try {
76        DEFAULT_PARSER_CLASSES.add(
77            Class.forName("com.google.android.exoplayer.text.webvtt.WebvttParser")
78                .asSubclass(SubtitleParser.class));
79      } catch (ClassNotFoundException e) {
80        // Parser not found.
81      }
82      try {
83        DEFAULT_PARSER_CLASSES.add(
84            Class.forName("com.google.android.exoplayer.text.ttml.TtmlParser")
85                .asSubclass(SubtitleParser.class));
86      } catch (ClassNotFoundException e) {
87        // Parser not found.
88      }
89      try {
90        DEFAULT_PARSER_CLASSES.add(
91            Class.forName("com.google.android.exoplayer.text.subrip.SubripParser")
92                .asSubclass(SubtitleParser.class));
93      } catch (ClassNotFoundException e) {
94        // Parser not found.
95      }
96      try {
97        DEFAULT_PARSER_CLASSES.add(
98            Class.forName("com.google.android.exoplayer.text.tx3g.Tx3gParser")
99                .asSubclass(SubtitleParser.class));
100     } catch (ClassNotFoundException e) {
101       // Parser not found.
102     }
103   }

105   private final Handler textRendererHandler;
106   private final TextRenderer textRenderer;
107   private final SampleSourceReader source;
108   private final MediaFormatHolder formatHolder;
109   private final SubtitleParser[] subtitleParsers;
110 
111   private int parserIndex;
112   private int trackIndex;
113 
114   private boolean inputStreamEnded;
115 
116   private Subtitle subtitle;
117   private Subtitle nextSubtitle;
118   private SubtitleParserHelper parserHelper;
119   private HandlerThread parserThread;
120   private int nextSubtitleEventIndex;
121 
122   /**
123    * @param source A source from which samples containing subtitle data can be read.
124    * @param textRenderer The text renderer.
125    * @param textRendererLooper The looper associated with the thread on which textRenderer should be
126    *     invoked. If the renderer makes use of standard Android UI components, then this should
127    *     normally be the looper associated with the applications' main thread, which can be
128    *     obtained using {@link android.app.Activity#getMainLooper()}. Null may be passed if the
129    *     renderer should be invoked directly on the player's internal rendering thread.
130    * @param subtitleParsers {@link SubtitleParser}s to parse text samples, in order of decreasing
131    *     priority. If omitted, the default parsers will be used.
132    */
133   public TextTrackRenderer(SampleSource source, TextRenderer textRenderer,
134       Looper textRendererLooper, SubtitleParser... subtitleParsers) {
135     this.source = source.register();
136     this.textRenderer = Assertions.checkNotNull(textRenderer);
137     this.textRendererHandler = textRendererLooper == null ? null
138         : new Handler(textRendererLooper, this);
139     if (subtitleParsers == null || subtitleParsers.length == 0) {
140       subtitleParsers = new SubtitleParser[DEFAULT_PARSER_CLASSES.size()];
141       for (int i = 0; i < subtitleParsers.length; i++) {
142         try {
143           subtitleParsers[i] = DEFAULT_PARSER_CLASSES.get(i).newInstance();
144         } catch (InstantiationException e) {
145           throw new IllegalStateException("Unexpected error creating default parser", e);
146         } catch (IllegalAccessException e) {
147           throw new IllegalStateException("Unexpected error creating default parser", e);
148         }
149       }
150     }
151     this.subtitleParsers = subtitleParsers;
152     formatHolder = new MediaFormatHolder();
153   }
154 
155   @Override
156   protected int doPrepare(long positionUs) {
157     boolean sourcePrepared = source.prepare(positionUs);
158     if (!sourcePrepared) {
159       return TrackRenderer.STATE_UNPREPARED;
160     }
161     int trackCount = source.getTrackCount();
162     for (int i = 0; i < subtitleParsers.length; i++) {
163       for (int j = 0; j < trackCount; j++) {
164         if (subtitleParsers[i].canParse(source.getTrackInfo(j).mimeType)) {
165           parserIndex = i;
166           trackIndex = j;
167           return TrackRenderer.STATE_PREPARED;
168         }
169       }
170     }
171     return TrackRenderer.STATE_IGNORE;
172   }
173 
174   @Override
175   protected void onEnabled(long positionUs, boolean joining) {
176     source.enable(trackIndex, positionUs);
177     parserThread = new HandlerThread("textParser");
178     parserThread.start();
179     parserHelper = new SubtitleParserHelper(parserThread.getLooper(), subtitleParsers[parserIndex]);
180     seekToInternal();
181   }
182 
183   @Override
184   protected void seekTo(long positionUs) {
185     source.seekToUs(positionUs);
186     seekToInternal();
187   }
188 
189   private void seekToInternal() {
190     inputStreamEnded = false;
191     subtitle = null;
192     nextSubtitle = null;
193     parserHelper.flush();
194     clearTextRenderer();
195   }
196 
197   @Override
198   protected void doSomeWork(long positionUs, long elapsedRealtimeUs) throws ExoPlaybackException {
199     source.continueBuffering(trackIndex, positionUs);
200 
201     if (nextSubtitle == null) {
202       try {
203         nextSubtitle = parserHelper.getAndClearResult();
204       } catch (IOException e) {
205         throw new ExoPlaybackException(e);
206       }
207     }
208 
209     if (getState() != TrackRenderer.STATE_STARTED) {
210       return;
211     }
212 
213     boolean textRendererNeedsUpdate = false;
214     long subtitleNextEventTimeUs = Long.MAX_VALUE;
215     if (subtitle != null) {
216       // We're iterating through the events in a subtitle. Set textRendererNeedsUpdate if we
217       // advance to the next event.
218       subtitleNextEventTimeUs = getNextEventTime();
219       while (subtitleNextEventTimeUs <= positionUs) {
220         nextSubtitleEventIndex++;
221         subtitleNextEventTimeUs = getNextEventTime();
222         textRendererNeedsUpdate = true;
223       }
224     }
225 
226     if (subtitleNextEventTimeUs == Long.MAX_VALUE && nextSubtitle != null
227         && nextSubtitle.getStartTime() <= positionUs) {
228       // Advance to the next subtitle. Sync the next event index and trigger an update.
229       subtitle = nextSubtitle;
230       nextSubtitle = null;
231       nextSubtitleEventIndex = subtitle.getNextEventTimeIndex(positionUs);
232       textRendererNeedsUpdate = true;
233     }
234 
235     if (textRendererNeedsUpdate) {
236       // textRendererNeedsUpdate is set and we're playing. Update the renderer.
237       updateTextRenderer(subtitle.getCues(positionUs));
238     }
239 
240     if (!inputStreamEnded && nextSubtitle == null && !parserHelper.isParsing()) {
241       // Try and read the next subtitle from the source.
242       SampleHolder sampleHolder = parserHelper.getSampleHolder();
243       sampleHolder.clearData();
244       int result = source.readData(trackIndex, positionUs, formatHolder, sampleHolder, false);
245       if (result == SampleSource.SAMPLE_READ) {
246         parserHelper.startParseOperation();
247       } else if (result == SampleSource.END_OF_STREAM) {
248         inputStreamEnded = true;
249       }
250     }
251   }
252 
253   @Override
254   protected void onDisabled() {
255     subtitle = null;
256     nextSubtitle = null;
257     parserThread.quit();
258     parserThread = null;
259     parserHelper = null;
260     clearTextRenderer();
261     source.disable(trackIndex);
262   }
263 
264   @Override
265   protected void onReleased() {
266     source.release();
267   }
268 
269   @Override
270   protected void maybeThrowError() throws ExoPlaybackException {
271     try {
272       source.maybeThrowError();
273     } catch (IOException e) {
274       throw new ExoPlaybackException(e);
275     }
276   }

278   @Override
279   protected long getDurationUs() {
280     return source.getTrackInfo(trackIndex).durationUs;
281   }
282 
283   @Override
284   protected long getBufferedPositionUs() {
285     // Don't block playback whilst subtitles are loading.
286     return END_OF_TRACK_US;
287   }
288 
289   @Override
290   protected boolean isEnded() {
291     return inputStreamEnded && (subtitle == null || getNextEventTime() == Long.MAX_VALUE);
292   }
293 
294   @Override
295   protected boolean isReady() {
296     // Don't block playback whilst subtitles are loading.
297     // Note: To change this behavior, it will be necessary to consider [Internal: b/12949941].
298     return true;
299   }
300 
301   private long getNextEventTime() {
302     return ((nextSubtitleEventIndex == -1)
303         || (nextSubtitleEventIndex >= subtitle.getEventTimeCount())) ? Long.MAX_VALUE
304         : (subtitle.getEventTime(nextSubtitleEventIndex));
305   }
306 
307   private void updateTextRenderer(List<Cue> cues) {
308     if (textRendererHandler != null) {
309       textRendererHandler.obtainMessage(MSG_UPDATE_OVERLAY, cues).sendToTarget();
310     } else {
311       invokeRendererInternalCues(cues);
312     }
313   }
314 
315   private void clearTextRenderer() {
316     updateTextRenderer(Collections.<Cue>emptyList());
317   }
318 
319   @SuppressWarnings("unchecked")
320   @Override
321   public boolean handleMessage(Message msg) {
322     switch (msg.what) {
323       case MSG_UPDATE_OVERLAY:
324         invokeRendererInternalCues((List<Cue>) msg.obj);
325         return true;
326     }
327     return false;
328   }
329 
330   private void invokeRendererInternalCues(List<Cue> cues) {
331     textRenderer.onCues(cues);
332   }
