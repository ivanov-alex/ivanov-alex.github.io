38    /**
39     * Interface definition for a callback to be notified of {@link MediaCodecVideoTrackRenderer}
40     * events.
41     */
42    public interface EventListener extends MediaCodecTrackRenderer.EventListener {
43  
44      /**
45       * Invoked to report the number of frames dropped by the renderer. Dropped frames are reported
46       * whenever the renderer is stopped having dropped frames, and optionally, whenever the count
47       * reaches a specified threshold whilst the renderer is started.
48       *
49       * @param count The number of dropped frames.
50       * @param elapsed The duration in milliseconds over which the frames were dropped. This
51       *     duration is timed from when the renderer was started or from when dropped frames were
52       *     last reported (whichever was more recent), and not from when the first of the reported
53       *     drops occurred.
54       */
55      void onDroppedFrames(int count, long elapsed);
56  
57      /**
58       * Invoked each time there's a change in the size of the video being rendered.
59       *
60       * @param width The video width in pixels.
61       * @param height The video height in pixels.
62       * @param pixelWidthHeightRatio The width to height ratio of each pixel. For the normal case
63       *     of square pixels this will be equal to 1.0. Different values are indicative of anamorphic
64       *     content.
65       */
66      void onVideoSizeChanged(int width, int height, float pixelWidthHeightRatio);
67  
68      /**
69       * Invoked when a frame is rendered to a surface for the first time following that surface
70       * having been set as the target for the renderer.
71       *
72       * @param surface The surface to which a first frame has been rendered.
73       */
74      void onDrawnToSurface(Surface surface);
75  
76    }

78    /**
79     * An interface for fine-grained adjustment of frame release times.
80     */
81    public interface FrameReleaseTimeHelper {
82  
83      /**
84       * Enables the helper.
85       */
86      void enable();
87  
88      /**
89       * Disables the helper.
90       */
91      void disable();
92  
93      /**
94       * Called to make a fine-grained adjustment to a frame release time.
95       *
96       * @param framePresentationTimeUs The frame's media presentation time, in microseconds.
97       * @param unadjustedReleaseTimeNs The frame's unadjusted release time, in nanoseconds and in
98       *     the same time base as {@link System#nanoTime()}.
99       * @return An adjusted release time for the frame, in nanoseconds and in the same time base as
100      *     {@link System#nanoTime()}.
101      */
102     public long adjustReleaseTime(long framePresentationTimeUs, long unadjustedReleaseTimeNs);

263   @Override
264   protected void onEnabled(long positionUs, boolean joining) {
265     super.onEnabled(positionUs, joining);
266     renderedFirstFrame = false;
267     if (joining && allowedJoiningTimeUs > 0) {
268       joiningDeadlineUs = SystemClock.elapsedRealtime() * 1000L + allowedJoiningTimeUs;
269     }
270     if (frameReleaseTimeHelper != null) {
271       frameReleaseTimeHelper.enable();
272     }

282   @Override
283   protected boolean isReady() {
284     if (super.isReady() && (renderedFirstFrame || !codecInitialized()
285         || getSourceState() == SOURCE_STATE_READY_READ_MAY_FAIL)) {
286       // Ready. If we were joining then we've now joined, so clear the joining deadline.
287       joiningDeadlineUs = -1;
288       return true;
289     } else if (joiningDeadlineUs == -1) {
290       // Not joining.
291       return false;
292     } else if (SystemClock.elapsedRealtime() * 1000 < joiningDeadlineUs) {
293       // Joining and still within the joining deadline.
294       return true;
295     } else {
296       // The joining deadline has been exceeded. Give up and clear the deadline.
297       joiningDeadlineUs = -1;
298       return false;
299     }
300   }

340   /**
341    * @param surface The surface to set.
342    * @throws ExoPlaybackException
343    */
344   private void setSurface(Surface surface) throws ExoPlaybackException {
345     if (this.surface == surface) {
346       return;
347     }
348     this.surface = surface;
349     this.reportedDrawnToSurface = false;
350     int state = getState();
351     if (state == TrackRenderer.STATE_ENABLED || state == TrackRenderer.STATE_STARTED) {
352       releaseCodec();
353       maybeInitCodec();
354     }
355   }

384   @Override
385   protected void onOutputFormatChanged(MediaFormat inputFormat,
386       android.media.MediaFormat outputFormat) {
387     boolean hasCrop = outputFormat.containsKey(KEY_CROP_RIGHT)
388         && outputFormat.containsKey(KEY_CROP_LEFT) && outputFormat.containsKey(KEY_CROP_BOTTOM)
389         && outputFormat.containsKey(KEY_CROP_TOP);
390     currentWidth = hasCrop
391         ? outputFormat.getInteger(KEY_CROP_RIGHT) - outputFormat.getInteger(KEY_CROP_LEFT) + 1
392         : outputFormat.getInteger(android.media.MediaFormat.KEY_WIDTH);
393     currentHeight = hasCrop
394         ? outputFormat.getInteger(KEY_CROP_BOTTOM) - outputFormat.getInteger(KEY_CROP_TOP) + 1
395         : outputFormat.getInteger(android.media.MediaFormat.KEY_HEIGHT);
396     currentPixelWidthHeightRatio = pendingPixelWidthHeightRatio;
397   }

407   @Override
408   protected boolean processOutputBuffer(long positionUs, long elapsedRealtimeUs, MediaCodec codec,
409       ByteBuffer buffer, MediaCodec.BufferInfo bufferInfo, int bufferIndex, boolean shouldSkip) {
410     if (shouldSkip) {
411       skipOutputBuffer(codec, bufferIndex);
412       return true;
413     }
414 
415     // Compute how many microseconds it is until the buffer's presentation time.
416     long elapsedSinceStartOfLoopUs = (SystemClock.elapsedRealtime() * 1000) - elapsedRealtimeUs;
417     long earlyUs = bufferInfo.presentationTimeUs - positionUs - elapsedSinceStartOfLoopUs;
418 
419     // Compute the buffer's desired release time in nanoseconds.
420     long systemTimeNs = System.nanoTime();
421     long unadjustedFrameReleaseTimeNs = systemTimeNs + (earlyUs * 1000);
422 
423     // Apply a timestamp adjustment, if there is one.
424     long adjustedReleaseTimeNs;
425     if (frameReleaseTimeHelper != null) {
426       adjustedReleaseTimeNs = frameReleaseTimeHelper.adjustReleaseTime(
427           bufferInfo.presentationTimeUs, unadjustedFrameReleaseTimeNs);
428       earlyUs = (adjustedReleaseTimeNs - systemTimeNs) / 1000;
429     } else {
430       adjustedReleaseTimeNs = unadjustedFrameReleaseTimeNs;
431     }
432 
433     if (earlyUs < -30000) {
434       // We're more than 30ms late rendering the frame.
435       dropOutputBuffer(codec, bufferIndex);
436       return true;
437     }
438 
439     if (!renderedFirstFrame) {
440       if (Util.SDK_INT >= 21) {
441         renderOutputBufferV21(codec, bufferIndex, System.nanoTime());
442       } else {
443         renderOutputBuffer(codec, bufferIndex);
444       }
445       return true;
446     }
447 
448     if (getState() != TrackRenderer.STATE_STARTED) {
449       return false;
450     }
451 
452     if (Util.SDK_INT >= 21) {
453       // Let the underlying framework time the release.
454       if (earlyUs < 50000) {
455         renderOutputBufferV21(codec, bufferIndex, adjustedReleaseTimeNs);
456         return true;
457       }
458     } else {
459       // We need to time the release ourselves.
460       if (earlyUs < 30000) {
461         if (earlyUs > 11000) {
462           // We're a little too early to render the frame. Sleep until the frame can be rendered.
463           // Note: The 11ms threshold was chosen fairly arbitrarily.
464           try {
465             // Subtracting 10000 rather than 11000 ensures the sleep time will be at least 1ms.
466             Thread.sleep((earlyUs - 10000) / 1000);
467           } catch (InterruptedException e) {
468             Thread.currentThread().interrupt();
469           }
470         }
471         renderOutputBuffer(codec, bufferIndex);
472         return true;
473       }
474     }
475 
476     // We're either not playing, or it's not time to render the frame yet.
477     return false;
478   }

480   protected void skipOutputBuffer(MediaCodec codec, int bufferIndex) {
481     TraceUtil.beginSection("skipVideoBuffer");
482     codec.releaseOutputBuffer(bufferIndex, false);
483     TraceUtil.endSection();
484     codecCounters.skippedOutputBufferCount++;
485   }
486 
487   protected void dropOutputBuffer(MediaCodec codec, int bufferIndex) {
488     TraceUtil.beginSection("dropVideoBuffer");
489     codec.releaseOutputBuffer(bufferIndex, false);
490     TraceUtil.endSection();
491     codecCounters.droppedOutputBufferCount++;
492     droppedFrameCount++;
493     if (droppedFrameCount == maxDroppedFrameCountToNotify) {
494       maybeNotifyDroppedFrameCount();
495     }
496   }
497 
498   protected void renderOutputBuffer(MediaCodec codec, int bufferIndex) {
499     maybeNotifyVideoSizeChanged();
500     TraceUtil.beginSection("releaseOutputBuffer");
501     codec.releaseOutputBuffer(bufferIndex, true);
502     TraceUtil.endSection();
503     codecCounters.renderedOutputBufferCount++;
504     renderedFirstFrame = true;
505     maybeNotifyDrawnToSurface();
506   }
507 
508   @TargetApi(21)
509   protected void renderOutputBufferV21(MediaCodec codec, int bufferIndex, long releaseTimeNs) {
510     maybeNotifyVideoSizeChanged();
511     TraceUtil.beginSection("releaseOutputBuffer");
512     codec.releaseOutputBuffer(bufferIndex, releaseTimeNs);
513     TraceUtil.endSection();
514     codecCounters.renderedOutputBufferCount++;
515     renderedFirstFrame = true;
516     maybeNotifyDrawnToSurface();
517   }

519   private void maybeNotifyVideoSizeChanged() {
520     if (eventHandler == null || eventListener == null
521         || (lastReportedWidth == currentWidth && lastReportedHeight == currentHeight
522         && lastReportedPixelWidthHeightRatio == currentPixelWidthHeightRatio)) {
523       return;
524     }
525     // Make final copies to ensure the runnable reports the correct values.
526     final int currentWidth = this.currentWidth;
527     final int currentHeight = this.currentHeight;
528     final float currentPixelWidthHeightRatio = this.currentPixelWidthHeightRatio;
529     eventHandler.post(new Runnable()  {
530       @Override
531       public void run() {
532         eventListener.onVideoSizeChanged(currentWidth, currentHeight, currentPixelWidthHeightRatio);
533       }
534     });
535     // Update the last reported values.
536     lastReportedWidth = currentWidth;
537     lastReportedHeight = currentHeight;
538     lastReportedPixelWidthHeightRatio = currentPixelWidthHeightRatio;
539   }
540 
541   private void maybeNotifyDrawnToSurface() {
542     if (eventHandler == null || eventListener == null || reportedDrawnToSurface) {
543       return;
544     }
545     // Make a final copy to ensure the runnable reports the correct surface.
546     final Surface surface = this.surface;
547     eventHandler.post(new Runnable()  {
548       @Override
549       public void run() {
550         eventListener.onDrawnToSurface(surface);
551       }
552     });
553     // Record that we have reported that the surface has been drawn to.
554     reportedDrawnToSurface = true;
555   }
556 
557   private void maybeNotifyDroppedFrameCount() {
558     if (eventHandler == null || eventListener == null || droppedFrameCount == 0) {
559       return;
560     }
561     long now = SystemClock.elapsedRealtime();
562     // Make final copies to ensure the runnable reports the correct values.
563     final int countToNotify = droppedFrameCount;
564     final long elapsedToNotify = now - droppedFrameAccumulationStartTimeMs;
565     eventHandler.post(new Runnable()  {
566       @Override
567       public void run() {
568         eventListener.onDroppedFrames(countToNotify, elapsedToNotify);
569       }
570     });
571     // Reset the dropped frame tracking.
572     droppedFrameCount = 0;
573     droppedFrameAccumulationStartTimeMs = now;
574   }
