16  package com.google.android.exoplayer;
17  
18  import android.os.Looper;
19  
20  /**
21   * An extensible media player exposing traditional high-level media player functionality, such as
22   * the ability to prepare, play, pause and seek.
23   *
24   * <p>Topics covered here are:
25   * <ol>
26   * <li><a href="#Assumptions">Assumptions and player composition</a>
27   * <li><a href="#Threading">Threading model</a>
28   * <li><a href="#State">Player state</a>
29   * </ol>
30   *
31   * <a name="Assumptions"></a>
32   * <h3>Assumptions and player construction</h3>
33   *
34   * <p>The implementation is designed make no assumptions about (and hence impose no restrictions
35   * on) the type of the media being played, how and where it is stored, or how it is rendered.
36   * Rather than implementing the loading and rendering of media directly, {@link ExoPlayer} instead
37   * delegates this work to one or more {@link TrackRenderer}s, which are injected when the player
38   * is prepared. Hence {@link ExoPlayer} is capable of loading and playing any media for which a
39   * {@link TrackRenderer} implementation can be provided.
40   *
41   * <p>{@link MediaCodecAudioTrackRenderer} and {@link MediaCodecVideoTrackRenderer} can be used for
42   * the common cases of rendering audio and video. These components in turn require an
43   * <i>upstream</i> {@link SampleSource} to be injected through their constructors, where upstream
44   * is defined to denote a component that is closer to the source of the media. This pattern of
45   * upstream dependency injection is actively encouraged, since it means that the functionality of
46   * the player is built up through the composition of components that can easily be exchanged for
47   * alternate implementations. For example a {@link SampleSource} implementation may require a
48   * further upstream data loading component to be injected through its constructor, with different
49   * implementations enabling the loading of data from various sources.
50   *
51   * <a name="Threading"></a>
52   * <h3>Threading model</h3>
53   *
54   * <p>The figure below shows the {@link ExoPlayer} threading model.</p>
55   * <p align="center"><img src="../../../../../images/exoplayer_threading_model.png"
56   *     alt="MediaPlayer state diagram"
57   *     border="0"/></p>
58   *
59   * <ul>
60   * <li>It is recommended that instances are created and accessed from a single application thread.
61   * An application's main thread is ideal. Accessing an instance from multiple threads is
62   * discouraged, however if an application does wish to do this then it may do so provided that it
63   * ensures accesses are synchronized.
64   * </li>
65   * <li>Registered {@link Listener}s are invoked on the thread that created the {@link ExoPlayer}
66   * instance.</li>
67   * <li>An internal playback thread is responsible for managing playback and invoking the
68   * {@link TrackRenderer}s in order to load and play the media.</li>
69   * <li>{@link TrackRenderer} implementations (or any upstream components that they depend on) may
70   * use additional background threads (e.g. to load data). These are implementation specific.</li>
71   * </ul>
72   *
73   * <a name="State"></a>
74   * <h3>Player state</h3>
75   *
76   * <p>The components of an {@link ExoPlayer}'s state can be divided into two distinct groups. State
77   * accessed by {@link #getRendererEnabled(int)} and {@link #getPlayWhenReady()} are only ever
78   * changed by invoking the player's methods, and are never changed as a result of operations that
79   * have been performed asynchronously by the playback thread. In contrast, the playback state
80   * accessed by {@link #getPlaybackState()} is only ever changed as a result of operations
81   * completing on the playback thread, as illustrated below.</p>
82   * <p align="center"><img src="../../../../../images/exoplayer_state.png"
83   *     alt="ExoPlayer state"
84   *     border="0"/></p>
85   *
86   * <p>The possible playback state transitions are shown below. Transitions can be triggered either
87   * by changes in the state of the {@link TrackRenderer}s being used, or as a result of
88   * {@link #prepare(TrackRenderer[])}, {@link #stop()} or {@link #release()} being invoked.</p>
89   * <p align="center"><img src="../../../../../images/exoplayer_playbackstate.png"
90   *     alt="ExoPlayer playback state transitions"
91   *     border="0"/></p>
92   */
93  public interface ExoPlayer {
94  
95    /**
96     * A factory for instantiating ExoPlayer instances.
97     */
98    public static final class Factory {
99  
100     /**
101      * The default minimum duration of data that must be buffered for playback to start or resume
102      * following a user action such as a seek.
103      */
104     public static final int DEFAULT_MIN_BUFFER_MS = 2500;
105 
106     /**
107      * The default minimum duration of data that must be buffered for playback to resume
108      * after a player invoked rebuffer (i.e. a rebuffer that occurs due to buffer depletion, and
109      * not due to a user action such as starting playback or seeking).
110      */
111     public static final int DEFAULT_MIN_REBUFFER_MS = 5000;
112 
113     private Factory() {}
114 
115     /**
116      * Obtains an {@link ExoPlayer} instance.
117      * <p>
118      * Must be invoked from a thread that has an associated {@link Looper}.
119      *
120      * @param rendererCount The number of {@link TrackRenderer}s that will be passed to
121      *     {@link #prepare(TrackRenderer[])}.
122      * @param minBufferMs A minimum duration of data that must be buffered for playback to start
123      *     or resume following a user action such as a seek.
124      * @param minRebufferMs A minimum duration of data that must be buffered for playback to resume
125      *     after a player invoked rebuffer (i.e. a rebuffer that occurs due to buffer depletion, and
126      *     not due to a user action such as starting playback or seeking).
127      */
128     public static ExoPlayer newInstance(int rendererCount, int minBufferMs, int minRebufferMs) {
129       return new ExoPlayerImpl(rendererCount, minBufferMs, minRebufferMs);
130     }
131 
132     /**
133      * Obtains an {@link ExoPlayer} instance.
134      * <p>
135      * Must be invoked from a thread that has an associated {@link Looper}.
136      *
137      * @param rendererCount The number of {@link TrackRenderer}s that will be passed to
138      *     {@link #prepare(TrackRenderer[])}.
139      */
140     public static ExoPlayer newInstance(int rendererCount) {
141       return new ExoPlayerImpl(rendererCount, DEFAULT_MIN_BUFFER_MS, DEFAULT_MIN_REBUFFER_MS);
142     }
143 
144   }
145 
146   /**
147    * Interface definition for a callback to be notified of changes in player state.
148    */
149   public interface Listener {
150     /**
151      * Invoked when the value returned from either {@link ExoPlayer#getPlayWhenReady()} or
152      * {@link ExoPlayer#getPlaybackState()} changes.
153      *
154      * @param playWhenReady Whether playback will proceed when ready.
155      * @param playbackState One of the {@code STATE} constants defined in the {@link ExoPlayer}
156      *     interface.
157      */
158     void onPlayerStateChanged(boolean playWhenReady, int playbackState);
159     /**
160      * Invoked when the current value of {@link ExoPlayer#getPlayWhenReady()} has been reflected
161      * by the internal playback thread.
162      * <p>
163      * An invocation of this method will shortly follow any call to
164      * {@link ExoPlayer#setPlayWhenReady(boolean)} that changes the state. If multiple calls are
165      * made in rapid succession, then this method will be invoked only once, after the final state
166      * has been reflected.
167      */
168     void onPlayWhenReadyCommitted();
169     /**
170      * Invoked when an error occurs. The playback state will transition to
171      * {@link ExoPlayer#STATE_IDLE} immediately after this method is invoked. The player instance
172      * can still be used, and {@link ExoPlayer#release()} must still be called on the player should
173      * it no longer be required.
174      *
175      * @param error The error.
176      */
177     void onPlayerError(ExoPlaybackException error);
178   }
179 
180   /**
181    * A component of an {@link ExoPlayer} that can receive messages on the playback thread.
182    * <p>
183    * Messages can be delivered to a component via {@link ExoPlayer#sendMessage} and
184    * {@link ExoPlayer#blockingSendMessage}.
185    */
186   public interface ExoPlayerComponent {
187 
188     /**
189      * Handles a message delivered to the component. Invoked on the playback thread.
190      *
191      * @param messageType An integer identifying the type of message.
192      * @param message The message object.
193      * @throws ExoPlaybackException If an error occurred whilst handling the message.
194      */
195     void handleMessage(int messageType, Object message) throws ExoPlaybackException;
196 
197   }
198 
199   /**
200    * The player is neither prepared or being prepared.
201    */
202   public static final int STATE_IDLE = 1;
203   /**
204    * The player is being prepared.
205    */
206   public static final int STATE_PREPARING = 2;
207   /**
208    * The player is prepared but not able to immediately play from the current position. The cause
209    * is {@link TrackRenderer} specific, but this state typically occurs when more data needs
210    * to be buffered for playback to start.
211    */
212   public static final int STATE_BUFFERING = 3;
213   /**
214    * The player is prepared and able to immediately play from the current position. The player will
215    * be playing if {@link #setPlayWhenReady(boolean)} returns true, and paused otherwise.
216    */
217   public static final int STATE_READY = 4;
218   /**
219    * The player has finished playing the media.
220    */
221   public static final int STATE_ENDED = 5;
222   /**
223    * Represents an unknown time or duration.
224    */
225   public static final long UNKNOWN_TIME = -1;
226 
227   /**
228    * Gets the {@link Looper} associated with the playback thread.
229    *
230    * @return The {@link Looper} associated with the playback thread.
231    */
232   public Looper getPlaybackLooper();
233 
234   /**
235    * Register a listener to receive events from the player. The listener's methods will be invoked
236    * on the thread that was used to construct the player.
237    *
238    * @param listener The listener to register.
239    */
240   public void addListener(Listener listener);
241 
242   /**
243    * Unregister a listener. The listener will no longer receive events from the player.
244    *
245    * @param listener The listener to unregister.
246    */
247   public void removeListener(Listener listener);
248 
249   /**
250    * Returns the current state of the player.
251    *
252    * @return One of the {@code STATE} constants defined in this interface.
253    */
254   public int getPlaybackState();
255 
256   /**
257    * Prepares the player for playback.
258    *
259    * @param renderers The {@link TrackRenderer}s to use. The number of renderers must match the
260    *     value that was passed to the {@link ExoPlayer.Factory#newInstance} method.
261    */
262   public void prepare(TrackRenderer... renderers);
263 
264   /**
265    * Returns whether the renderer at the given index has media to play.
266    * <p>
267    * Always returns false whilst the player is in the {@link #STATE_PREPARING} state.
268    *
269    * @param rendererIndex The index of the renderer.
270    * @return True if the renderer has media to play, false otherwise.
271    */
272   public boolean getRendererHasMedia(int rendererIndex);
273 
274   /**
275    * Sets whether the renderer at the given index is enabled.
276    *
277    * @param rendererIndex The index of the renderer.
278    * @param enabled Whether the renderer at the given index should be enabled.
279    */
280   public void setRendererEnabled(int rendererIndex, boolean enabled);
281 
282   /**
283    * Whether the renderer at the given index is enabled.
284    *
285    * @param rendererIndex The index of the renderer.
286    * @return Whether the renderer is enabled.
287    */
288   public boolean getRendererEnabled(int rendererIndex);
289 
290   /**
291    * Sets whether playback should proceed when {@link #getPlaybackState()} == {@link #STATE_READY}.
292    * If the player is already in this state, then this method can be used to pause and resume
293    * playback.
294    *
295    * @param playWhenReady Whether playback should proceed when ready.
296    */
297   public void setPlayWhenReady(boolean playWhenReady);
298 
299   /**
300    * Whether playback will proceed when {@link #getPlaybackState()} == {@link #STATE_READY}.
301    *
302    * @return Whether playback will proceed when ready.
303    */
304   public boolean getPlayWhenReady();
305 
306   /**
307    * Whether the current value of {@link ExoPlayer#getPlayWhenReady()} has been reflected by the
308    * internal playback thread.
309    *
310    * @return True if the current value has been reflected. False otherwise.
311    */
312   public boolean isPlayWhenReadyCommitted();
313 
314   /**
315    * Seeks to a position specified in milliseconds.
316    *
317    * @param positionMs The seek position.
318    */
319   public void seekTo(long positionMs);
320 
321   /**
322    * Stops playback. Use {@code setPlayWhenReady(false)} rather than this method if the intention
323    * is to pause playback.
324    * <p>
325    * Calling this method will cause the playback state to transition to
326    * {@link ExoPlayer#STATE_IDLE}. The player instance can still be used, and
327    * {@link ExoPlayer#release()} must still be called on the player if it's no longer required.
328    * <p>
329    * Calling this method does not reset the playback position. If this player instance will be used
330    * to play another video from its start, then {@code seekTo(0)} should be called after stopping
331    * the player and before preparing it for the next video.
332    */
333   public void stop();
334 
335   /**
336    * Releases the player. This method must be called when the player is no longer required.
337    * <p>
338    * The player must not be used after calling this method.
339    */
340   public void release();
341 
342   /**
343    * Sends a message to a specified component. The message is delivered to the component on the
344    * playback thread. If the component throws a {@link ExoPlaybackException}, then it is
345    * propagated out of the player as an error.
346    *
347    * @param target The target to which the message should be delivered.
348    * @param messageType An integer that can be used to identify the type of the message.
349    * @param message The message object.
350    */
351   public void sendMessage(ExoPlayerComponent target, int messageType, Object message);
352 
353   /**
354    * Blocking variant of {@link #sendMessage(ExoPlayerComponent, int, Object)} that does not return
355    * until after the message has been delivered.
356    *
357    * @param target The target to which the message should be delivered.
358    * @param messageType An integer that can be used to identify the type of the message.
359    * @param message The message object.
360    */
361   public void blockingSendMessage(ExoPlayerComponent target, int messageType, Object message);
362 
363   /**
364    * Gets the duration of the track in milliseconds.
365    *
366    * @return The duration of the track in milliseconds, or {@link ExoPlayer#UNKNOWN_TIME} if the
367    *     duration is not known.
368    */
369   public long getDuration();
370 
371   /**
372    * Gets the current playback position in milliseconds.
373    *
374    * @return The current playback position in milliseconds.
375    */
376   public long getCurrentPosition();
377 
378   /**
379    * Gets an estimate of the absolute position in milliseconds up to which data is buffered.
380    *
381    * @return An estimate of the absolute position in milliseconds up to which data is buffered,
382    *     or {@link ExoPlayer#UNKNOWN_TIME} if no estimate is available.
383    */
384   public long getBufferedPosition();
385 
386   /**
387    * Gets an estimate of the percentage into the media up to which data is buffered.
388    *
389    * @return An estimate of the percentage into the media up to which data is buffered. 0 if the
390    *     duration of the media is not known or if no estimate is available.
391    */
392   public int getBufferedPercentage();
393
