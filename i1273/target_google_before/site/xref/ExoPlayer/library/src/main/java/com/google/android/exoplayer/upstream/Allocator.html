16  package com.google.android.exoplayer.upstream;
17  
18  /**
19   * A source of allocations.
20   */
21  public interface Allocator {
22  
23    /**
24     * Obtain an {@link Allocation}.
25     * <p>
26     * When the caller has finished with the {@link Allocation}, it should be returned by calling
27     * {@link #release(Allocation)}.
28     *
29     * @return The {@link Allocation}.
30     */
31    Allocation allocate();
32  
33    /**
34     * Return an {@link Allocation}.
35     *
36     * @param allocation The {@link Allocation} being returned.
37     */
38    void release(Allocation allocation);
39  
40    /**
41     * Hints to the {@link Allocator} that it should make a best effort to release any memory that it
42     * has allocated, beyond the specified target number of bytes.
43     *
44     * @param targetSize The target size in bytes.
45     */
46    void trim(int targetSize);
47  
48    /**
49     * Blocks execution until the number of bytes allocated is not greater than the limit, or the
50     * thread is interrupted.
51     *
52     * @param limit The limit in bytes.
53     * @throws InterruptedException If the thread is interrupted.
54     */
55    void blockWhileTotalBytesAllocatedExceeds(int limit) throws InterruptedException;
56  
57    /**
58     * Returns the total number of bytes currently allocated.
59     */
60    int getTotalBytesAllocated();
61  
62    /**
63     * Returns the length of each individual {@link Allocation}.
64     */
65    int getIndividualAllocationLength();
66  
67  }
