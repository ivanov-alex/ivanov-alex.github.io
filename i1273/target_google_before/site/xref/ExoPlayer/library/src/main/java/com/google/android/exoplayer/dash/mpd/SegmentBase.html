24  /**
25   * An approximate representation of a SegmentBase manifest element.
26   */
27  public abstract class SegmentBase {
28  
29    /* package */ final RangedUri initialization;
30    /* package */ final long timescale;
31    /* package */ final long presentationTimeOffset;
32  
33    /**
34     * @param initialization A {@link RangedUri} corresponding to initialization data, if such data
35     *     exists.
36     * @param timescale The timescale in units per second.
37     * @param presentationTimeOffset The presentation time offset. The value in seconds is the
38     *     division of this value and {@code timescale}.
39     */
40    public SegmentBase(RangedUri initialization, long timescale, long presentationTimeOffset) {
41      this.initialization = initialization;
42      this.timescale = timescale;
43      this.presentationTimeOffset = presentationTimeOffset;
44    }
45  
46    /**
47     * Gets the {@link RangedUri} defining the location of initialization data for a given
48     * representation. May be null if no initialization data exists.
49     *
50     * @param representation The {@link Representation} for which initialization data is required.
51     * @return A {@link RangedUri} defining the location of the initialization data, or null.
52     */
53    public RangedUri getInitialization(Representation representation) {
54      return initialization;
55    }
56  
57    /**
58     * Gets the presentation time offset, in microseconds.
59     *
60     * @return The presentation time offset, in microseconds.
61     */
62    public long getPresentationTimeOffsetUs() {
63      return Util.scaleLargeTimestamp(presentationTimeOffset, C.MICROS_PER_SECOND, timescale);
64    }

66    /**
67     * A {@link SegmentBase} that defines a single segment.
68     */
69    public static class SingleSegmentBase extends SegmentBase {
70  
71      /**
72       * The uri of the segment.
73       */
74      public final String uri;
75  
76      /* package */ final long indexStart;
77      /* package */ final long indexLength;
78  
79      /**
80       * @param initialization A {@link RangedUri} corresponding to initialization data, if such data
81       *     exists.
82       * @param timescale The timescale in units per second.
83       * @param presentationTimeOffset The presentation time offset. The value in seconds is the
84       *     division of this value and {@code timescale}.
85       * @param uri The uri of the segment.
86       * @param indexStart The byte offset of the index data in the segment.
87       * @param indexLength The length of the index data in bytes.
88       */
89      public SingleSegmentBase(RangedUri initialization, long timescale, long presentationTimeOffset,
90          String uri, long indexStart, long indexLength) {
91        super(initialization, timescale, presentationTimeOffset);
92        this.uri = uri;
93        this.indexStart = indexStart;
94        this.indexLength = indexLength;
95      }
96  
97      /**
98       * @param uri The uri of the segment.
99       */
100     public SingleSegmentBase(String uri) {
101       this(null, 1, 0, uri, 0, -1);
102     }
103 
104     public RangedUri getIndex() {
105       return indexLength <= 0 ? null : new RangedUri(uri, null, indexStart, indexLength);
106     }
107 
108   }

110   /**
111    * A {@link SegmentBase} that consists of multiple segments.
112    */
113   public abstract static class MultiSegmentBase extends SegmentBase {
114 
115     /* package */ final long periodDurationMs;
116     /* package */ final int startNumber;
117     /* package */ final long duration;
118     /* package */ final List<SegmentTimelineElement> segmentTimeline;
119 
120     /**
121      * @param initialization A {@link RangedUri} corresponding to initialization data, if such data
122      *     exists.
123      * @param timescale The timescale in units per second.
124      * @param presentationTimeOffset The presentation time offset. The value in seconds is the
125      *     division of this value and {@code timescale}.
126      * @param periodDurationMs The duration of the enclosing period in milliseconds.
127      * @param startNumber The sequence number of the first segment.
128      * @param duration The duration of each segment in the case of fixed duration segments. The
129      *     value in seconds is the division of this value and {@code timescale}. If
130      *     {@code segmentTimeline} is non-null then this parameter is ignored.
131      * @param segmentTimeline A segment timeline corresponding to the segments. If null, then
132      *     segments are assumed to be of fixed duration as specified by the {@code duration}
133      *     parameter.
134      */
135     public MultiSegmentBase(RangedUri initialization, long timescale, long presentationTimeOffset,
136         long periodDurationMs, int startNumber, long duration,
137         List<SegmentTimelineElement> segmentTimeline) {
138       super(initialization, timescale, presentationTimeOffset);
139       this.periodDurationMs = periodDurationMs;
140       this.startNumber = startNumber;
141       this.duration = duration;
142       this.segmentTimeline = segmentTimeline;
143     }
144 
145     /**
146      * @see DashSegmentIndex#getSegmentNum(long)
147      */
148     public int getSegmentNum(long timeUs) {
149       final int firstSegmentNum = getFirstSegmentNum();
150       int lowIndex = firstSegmentNum;
151       int highIndex = getLastSegmentNum();
152       if (segmentTimeline == null) {
153         // All segments are of equal duration (with the possible exception of the last one).
154         long durationUs = (duration * C.MICROS_PER_SECOND) / timescale;
155         int segmentNum = startNumber + (int) (timeUs / durationUs);
156         // Ensure we stay within bounds.
157         return segmentNum < lowIndex ? lowIndex
158             : highIndex != DashSegmentIndex.INDEX_UNBOUNDED && segmentNum > highIndex ? highIndex
159             : segmentNum;
160       } else {
161         // The high index cannot be unbounded. Identify the segment using binary search.
162         while (lowIndex <= highIndex) {
163           int midIndex = (lowIndex + highIndex) / 2;
164           long midTimeUs = getSegmentTimeUs(midIndex);
165           if (midTimeUs < timeUs) {
166             lowIndex = midIndex + 1;
167           } else if (midTimeUs > timeUs) {
168             highIndex = midIndex - 1;
169           } else {
170             return midIndex;
171           }
172         }
173         return lowIndex == firstSegmentNum ? lowIndex : highIndex;
174       }
175     }
176 
177     /**
178      * @see DashSegmentIndex#getDurationUs(int)
179      */
180     public final long getSegmentDurationUs(int sequenceNumber) {
181       if (segmentTimeline != null) {
182         long duration = segmentTimeline.get(sequenceNumber - startNumber).duration;
183         return (duration * C.MICROS_PER_SECOND) / timescale;
184       } else {
185         return sequenceNumber == getLastSegmentNum()
186             ? ((periodDurationMs * 1000) - getSegmentTimeUs(sequenceNumber))
187             : ((duration * C.MICROS_PER_SECOND) / timescale);
188       }
189     }
190 
191     /**
192      * @see DashSegmentIndex#getTimeUs(int)
193      */
194     public final long getSegmentTimeUs(int sequenceNumber) {
195       long unscaledSegmentTime;
196       if (segmentTimeline != null) {
197         unscaledSegmentTime = segmentTimeline.get(sequenceNumber - startNumber).startTime
198             - presentationTimeOffset;
199       } else {
200         unscaledSegmentTime = (sequenceNumber - startNumber) * duration;
201       }
202       return Util.scaleLargeTimestamp(unscaledSegmentTime, C.MICROS_PER_SECOND, timescale);
203     }
204 
205     /**
206      * Returns a {@link RangedUri} defining the location of a segment for the given index in the
207      * given representation.
208      *
209      * @see DashSegmentIndex#getSegmentUrl(int)
210      */
211     public abstract RangedUri getSegmentUrl(Representation representation, int index);
212 
213     /**
214      * @see DashSegmentIndex#getFirstSegmentNum()
215      */
216     public int getFirstSegmentNum() {
217       return startNumber;
218     }
219 
220     /**
221      * @see DashSegmentIndex#getLastSegmentNum()
222      */
223     public abstract int getLastSegmentNum();
224 
225     /**
226      * @see DashSegmentIndex#isExplicit()
227      */
228     public boolean isExplicit() {
229       return segmentTimeline != null;
230     }
231 
232   }

234   /**
235    * A {@link MultiSegmentBase} that uses a SegmentList to define its segments.
236    */
237   public static class SegmentList extends MultiSegmentBase {
238 
239     /* package */ final List<RangedUri> mediaSegments;
240 
241     /**
242      * @param initialization A {@link RangedUri} corresponding to initialization data, if such data
243      *     exists.
244      * @param timescale The timescale in units per second.
245      * @param presentationTimeOffset The presentation time offset. The value in seconds is the
246      *     division of this value and {@code timescale}.
247      * @param periodDurationMs The duration of the enclosing period in milliseconds.
248      * @param startNumber The sequence number of the first segment.
249      * @param duration The duration of each segment in the case of fixed duration segments. The
250      *     value in seconds is the division of this value and {@code timescale}. If
251      *     {@code segmentTimeline} is non-null then this parameter is ignored.
252      * @param segmentTimeline A segment timeline corresponding to the segments. If null, then
253      *     segments are assumed to be of fixed duration as specified by the {@code duration}
254      *     parameter.
255      * @param mediaSegments A list of {@link RangedUri}s indicating the locations of the segments.
256      */
257     public SegmentList(RangedUri initialization, long timescale, long presentationTimeOffset,
258         long periodDurationMs, int startNumber, long duration,
259         List<SegmentTimelineElement> segmentTimeline, List<RangedUri> mediaSegments) {
260       super(initialization, timescale, presentationTimeOffset, periodDurationMs, startNumber,
261           duration, segmentTimeline);
262       this.mediaSegments = mediaSegments;
263     }
264 
265     @Override
266     public RangedUri getSegmentUrl(Representation representation, int sequenceNumber) {
267       return mediaSegments.get(sequenceNumber - startNumber);
268     }
269 
270     @Override
271     public int getLastSegmentNum() {
272       return startNumber + mediaSegments.size() - 1;
273     }
274 
275     @Override
276     public boolean isExplicit() {
277       return true;
278     }
279 
280   }

282   /**
283    * A {@link MultiSegmentBase} that uses a SegmentTemplate to define its segments.
284    */
285   public static class SegmentTemplate extends MultiSegmentBase {
286 
287     /* package */ final UrlTemplate initializationTemplate;
288     /* package */ final UrlTemplate mediaTemplate;
289 
290     private final String baseUrl;
291 
292     /**
293      * @param initialization A {@link RangedUri} corresponding to initialization data, if such data
294      *     exists. The value of this parameter is ignored if {@code initializationTemplate} is
295      *     non-null.
296      * @param timescale The timescale in units per second.
297      * @param presentationTimeOffset The presentation time offset. The value in seconds is the
298      *     division of this value and {@code timescale}.
299      * @param periodDurationMs The duration of the enclosing period in milliseconds.
300      * @param startNumber The sequence number of the first segment.
301      * @param duration The duration of each segment in the case of fixed duration segments. The
302      *     value in seconds is the division of this value and {@code timescale}. If
303      *     {@code segmentTimeline} is non-null then this parameter is ignored.
304      * @param segmentTimeline A segment timeline corresponding to the segments. If null, then
305      *     segments are assumed to be of fixed duration as specified by the {@code duration}
306      *     parameter.
307      * @param initializationTemplate A template defining the location of initialization data, if
308      *     such data exists. If non-null then the {@code initialization} parameter is ignored. If
309      *     null then {@code initialization} will be used.
310      * @param mediaTemplate A template defining the location of each media segment.
311      * @param baseUrl A url to use as the base for relative urls generated by the templates.
312      */
313     public SegmentTemplate(RangedUri initialization, long timescale, long presentationTimeOffset,
314         long periodDurationMs, int startNumber, long duration,
315         List<SegmentTimelineElement> segmentTimeline, UrlTemplate initializationTemplate,
316         UrlTemplate mediaTemplate, String baseUrl) {
317       super(initialization, timescale, presentationTimeOffset, periodDurationMs, startNumber,
318           duration, segmentTimeline);
319       this.initializationTemplate = initializationTemplate;
320       this.mediaTemplate = mediaTemplate;
321       this.baseUrl = baseUrl;
322     }
323 
324     @Override
325     public RangedUri getInitialization(Representation representation) {
326       if (initializationTemplate != null) {
327         String urlString = initializationTemplate.buildUri(representation.format.id, 0,
328             representation.format.bitrate, 0);
329         return new RangedUri(baseUrl, urlString, 0, -1);
330       } else {
331         return super.getInitialization(representation);
332       }
333     }
334 
335     @Override
336     public RangedUri getSegmentUrl(Representation representation, int sequenceNumber) {
337       long time = 0;
338       if (segmentTimeline != null) {
339         time = segmentTimeline.get(sequenceNumber - startNumber).startTime;
340       } else {
341         time = (sequenceNumber - startNumber) * duration;
342       }
343       String uriString = mediaTemplate.buildUri(representation.format.id, sequenceNumber,
344           representation.format.bitrate, time);
345       return new RangedUri(baseUrl, uriString, 0, -1);
346     }
347 
348     @Override
349     public int getLastSegmentNum() {
350       if (segmentTimeline != null) {
351         return segmentTimeline.size() + startNumber - 1;
352       } else if (periodDurationMs == -1) {
353         return DashSegmentIndex.INDEX_UNBOUNDED;
354       } else {
355         long durationMs = (duration * 1000) / timescale;
356         return startNumber + (int) Util.ceilDivide(periodDurationMs, durationMs) - 1;
357       }
358     }
359 
360   }

362   /**
363    * Represents a timeline segment from the MPD's SegmentTimeline list.
364    */
365   public static class SegmentTimelineElement {
366 
367     /* package */ long startTime;
368     /* package */ long duration;
369 
370     /**
371      * @param startTime The start time of the element. The value in seconds is the division of this
372      *     value and the {@code timescale} of the enclosing element.
373      * @param duration The duration of the element. The value in seconds is the division of this
374      *     value and the {@code timescale} of the enclosing element.
375      */
376     public SegmentTimelineElement(long startTime, long duration) {
377       this.startTime = startTime;
378       this.duration = duration;
379     }
