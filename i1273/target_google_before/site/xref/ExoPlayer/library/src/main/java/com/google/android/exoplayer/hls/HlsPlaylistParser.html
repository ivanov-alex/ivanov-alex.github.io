39    private static final String VERSION_TAG = "#EXT-X-VERSION";
40    private static final String STREAM_INF_TAG = "#EXT-X-STREAM-INF";
41    private static final String MEDIA_TAG = "#EXT-X-MEDIA";
42    private static final String DISCONTINUITY_TAG = "#EXT-X-DISCONTINUITY";
43    private static final String MEDIA_DURATION_TAG = "#EXTINF";
44    private static final String MEDIA_SEQUENCE_TAG = "#EXT-X-MEDIA-SEQUENCE";
45    private static final String TARGET_DURATION_TAG = "#EXT-X-TARGETDURATION";
46    private static final String ENDLIST_TAG = "#EXT-X-ENDLIST";
47    private static final String KEY_TAG = "#EXT-X-KEY";
48    private static final String BYTERANGE_TAG = "#EXT-X-BYTERANGE";
49  
50    private static final String BANDWIDTH_ATTR = "BANDWIDTH";
51    private static final String CODECS_ATTR = "CODECS";
52    private static final String RESOLUTION_ATTR = "RESOLUTION";
53    private static final String LANGUAGE_ATTR = "LANGUAGE";
54    private static final String NAME_ATTR = "NAME";
55    private static final String AUTOSELECT_ATTR = "AUTOSELECT";
56    private static final String DEFAULT_ATTR = "DEFAULT";
57    private static final String TYPE_ATTR = "TYPE";
58    private static final String METHOD_ATTR = "METHOD";
59    private static final String URI_ATTR = "URI";
60    private static final String IV_ATTR = "IV";
61  
62    private static final String AUDIO_TYPE = "AUDIO";
63    private static final String VIDEO_TYPE = "VIDEO";
64    private static final String SUBTITLES_TYPE = "SUBTITLES";
65    private static final String CLOSED_CAPTIONS_TYPE = "CLOSED-CAPTIONS";
66  
67    private static final String METHOD_NONE = "NONE";
68    private static final String METHOD_AES128 = "AES-128";
69  
70    private static final Pattern BANDWIDTH_ATTR_REGEX =
71        Pattern.compile(BANDWIDTH_ATTR + "=(\\d+)\\b");
72    private static final Pattern CODECS_ATTR_REGEX =
73        Pattern.compile(CODECS_ATTR + "=\"(.+?)\"");
74    private static final Pattern RESOLUTION_ATTR_REGEX =
75        Pattern.compile(RESOLUTION_ATTR + "=(\\d+x\\d+)");
76    private static final Pattern MEDIA_DURATION_REGEX =
77        Pattern.compile(MEDIA_DURATION_TAG + ":([\\d.]+)\\b");
78    private static final Pattern MEDIA_SEQUENCE_REGEX =
79        Pattern.compile(MEDIA_SEQUENCE_TAG + ":(\\d+)\\b");
80    private static final Pattern TARGET_DURATION_REGEX =
81        Pattern.compile(TARGET_DURATION_TAG + ":(\\d+)\\b");
82    private static final Pattern VERSION_REGEX =
83        Pattern.compile(VERSION_TAG + ":(\\d+)\\b");
84    private static final Pattern BYTERANGE_REGEX =
85        Pattern.compile(BYTERANGE_TAG + ":(\\d+(?:@\\d+)?)\\b");
86  
87    private static final Pattern METHOD_ATTR_REGEX =
88        Pattern.compile(METHOD_ATTR + "=(" + METHOD_NONE + "|" + METHOD_AES128 + ")");
89    private static final Pattern URI_ATTR_REGEX =
90        Pattern.compile(URI_ATTR + "=\"(.+)\"");
91    private static final Pattern IV_ATTR_REGEX =
92        Pattern.compile(IV_ATTR + "=([^,.*]+)");
93    private static final Pattern TYPE_ATTR_REGEX =
94        Pattern.compile(TYPE_ATTR + "=(" + AUDIO_TYPE + "|" + VIDEO_TYPE + "|" + SUBTITLES_TYPE + "|"
95            + CLOSED_CAPTIONS_TYPE + ")");
96    private static final Pattern LANGUAGE_ATTR_REGEX =
97        Pattern.compile(LANGUAGE_ATTR + "=\"(.+?)\"");
98    private static final Pattern NAME_ATTR_REGEX =
99        Pattern.compile(NAME_ATTR + "=\"(.+?)\"");
100   private static final Pattern AUTOSELECT_ATTR_REGEX =
101       HlsParserUtil.compileBooleanAttrPattern(AUTOSELECT_ATTR);
102   private static final Pattern DEFAULT_ATTR_REGEX =
103       HlsParserUtil.compileBooleanAttrPattern(DEFAULT_ATTR);
104 
105   @Override
106   public HlsPlaylist parse(String connectionUrl, InputStream inputStream)
107       throws IOException, ParserException {
108     BufferedReader reader = new BufferedReader(new InputStreamReader(inputStream));
109     Queue<String> extraLines = new LinkedList<>();
110     String line;
111     try {
112       while ((line = reader.readLine()) != null) {
113         line = line.trim();
114         if (line.isEmpty()) {
115           // Do nothing.
116         } else if (line.startsWith(STREAM_INF_TAG)) {
117           extraLines.add(line);
118           return parseMasterPlaylist(new LineIterator(extraLines, reader), connectionUrl);
119         } else if (line.startsWith(TARGET_DURATION_TAG)
120             || line.startsWith(MEDIA_SEQUENCE_TAG)
121             || line.startsWith(MEDIA_DURATION_TAG)
122             || line.startsWith(KEY_TAG)
123             || line.startsWith(BYTERANGE_TAG)
124             || line.equals(DISCONTINUITY_TAG)
125             || line.equals(ENDLIST_TAG)) {
126           extraLines.add(line);
127           return parseMediaPlaylist(new LineIterator(extraLines, reader), connectionUrl);
128         } else {
129           extraLines.add(line);
130         }
131       }
132     } finally {
133       reader.close();
134     }
135     throw new ParserException("Failed to parse the playlist, could not identify any tags.");
136   }

138   private static HlsMasterPlaylist parseMasterPlaylist(LineIterator iterator, String baseUri)
139       throws IOException {
140     ArrayList<Variant> variants = new ArrayList<>();
141     ArrayList<Subtitle> subtitles = new ArrayList<>();
142     int bitrate = 0;
143     String codecs = null;
144     int width = -1;
145     int height = -1;
146 
147     boolean expectingStreamInfUrl = false;
148     String line;
149     while (iterator.hasNext()) {
150       line = iterator.next();
151       if (line.startsWith(MEDIA_TAG)) {
152         String type = HlsParserUtil.parseStringAttr(line, TYPE_ATTR_REGEX, TYPE_ATTR);
153         if (SUBTITLES_TYPE.equals(type)) {
154           // We assume all subtitles belong to the same group.
155           String name = HlsParserUtil.parseStringAttr(line, NAME_ATTR_REGEX, NAME_ATTR);
156           String uri = HlsParserUtil.parseStringAttr(line, URI_ATTR_REGEX, URI_ATTR);
157           String language = HlsParserUtil.parseOptionalStringAttr(line, LANGUAGE_ATTR_REGEX);
158           boolean isDefault = HlsParserUtil.parseOptionalBooleanAttr(line, DEFAULT_ATTR_REGEX);
159           boolean autoSelect = HlsParserUtil.parseOptionalBooleanAttr(line, AUTOSELECT_ATTR_REGEX);
160           subtitles.add(new Subtitle(name, uri, language, isDefault, autoSelect));
161         } else {
162           // TODO: Support other types of media tag.
163         }
164       } else if (line.startsWith(STREAM_INF_TAG)) {
165         bitrate = HlsParserUtil.parseIntAttr(line, BANDWIDTH_ATTR_REGEX, BANDWIDTH_ATTR);
166         codecs = HlsParserUtil.parseOptionalStringAttr(line, CODECS_ATTR_REGEX);
167         String resolutionString = HlsParserUtil.parseOptionalStringAttr(line,
168             RESOLUTION_ATTR_REGEX);
169         if (resolutionString != null) {
170           String[] widthAndHeight = resolutionString.split("x");
171           width = Integer.parseInt(widthAndHeight[0]);
172           if (width <= 0) {
173             // Width was invalid.
174             width = -1;
175           }
176           height = Integer.parseInt(widthAndHeight[1]);
177           if (height <= 0) {
178             // Height was invalid.
179             height = -1;
180           }
181         } else {
182           width = -1;
183           height = -1;
184         }
185         expectingStreamInfUrl = true;
186       } else if (!line.startsWith("#") && expectingStreamInfUrl) {
187         variants.add(new Variant(variants.size(), line, bitrate, codecs, width, height));
188         bitrate = 0;
189         codecs = null;
190         width = -1;
191         height = -1;
192         expectingStreamInfUrl = false;
193       }
194     }
195     return new HlsMasterPlaylist(baseUri, Collections.unmodifiableList(variants),
196         Collections.unmodifiableList(subtitles));
197   }

199   private static HlsMediaPlaylist parseMediaPlaylist(LineIterator iterator, String baseUri)
200       throws IOException {
201     int mediaSequence = 0;
202     int targetDurationSecs = 0;
203     int version = 1; // Default version == 1.
204     boolean live = true;
205     List<Segment> segments = new ArrayList<>();
206 
207     double segmentDurationSecs = 0.0;
208     boolean segmentDiscontinuity = false;
209     long segmentStartTimeUs = 0;
210     int segmentByterangeOffset = 0;
211     int segmentByterangeLength = C.LENGTH_UNBOUNDED;
212     int segmentMediaSequence = 0;
213 
214     boolean isEncrypted = false;
215     String encryptionKeyUri = null;
216     String encryptionIV = null;
217 
218     String line;
219     while (iterator.hasNext()) {
220       line = iterator.next();
221       if (line.startsWith(TARGET_DURATION_TAG)) {
222         targetDurationSecs = HlsParserUtil.parseIntAttr(line, TARGET_DURATION_REGEX,
223             TARGET_DURATION_TAG);
224       } else if (line.startsWith(MEDIA_SEQUENCE_TAG)) {
225         mediaSequence = HlsParserUtil.parseIntAttr(line, MEDIA_SEQUENCE_REGEX, MEDIA_SEQUENCE_TAG);
226         segmentMediaSequence = mediaSequence;
227       } else if (line.startsWith(VERSION_TAG)) {
228         version = HlsParserUtil.parseIntAttr(line, VERSION_REGEX, VERSION_TAG);
229       } else if (line.startsWith(MEDIA_DURATION_TAG)) {
230         segmentDurationSecs = HlsParserUtil.parseDoubleAttr(line, MEDIA_DURATION_REGEX,
231             MEDIA_DURATION_TAG);
232       } else if (line.startsWith(KEY_TAG)) {
233         String method = HlsParserUtil.parseStringAttr(line, METHOD_ATTR_REGEX, METHOD_ATTR);
234         isEncrypted = METHOD_AES128.equals(method);
235         if (isEncrypted) {
236           encryptionKeyUri = HlsParserUtil.parseStringAttr(line, URI_ATTR_REGEX, URI_ATTR);
237           encryptionIV = HlsParserUtil.parseOptionalStringAttr(line, IV_ATTR_REGEX);
238         } else {
239           encryptionKeyUri = null;
240           encryptionIV = null;
241         }
242       } else if (line.startsWith(BYTERANGE_TAG)) {
243         String byteRange = HlsParserUtil.parseStringAttr(line, BYTERANGE_REGEX, BYTERANGE_TAG);
244         String[] splitByteRange = byteRange.split("@");
245         segmentByterangeLength = Integer.parseInt(splitByteRange[0]);
246         if (splitByteRange.length > 1) {
247           segmentByterangeOffset = Integer.parseInt(splitByteRange[1]);
248         }
249       } else if (line.equals(DISCONTINUITY_TAG)) {
250         segmentDiscontinuity = true;
251       } else if (!line.startsWith("#")) {
252         String segmentEncryptionIV;
253         if (!isEncrypted) {
254           segmentEncryptionIV = null;
255         } else if (encryptionIV != null) {
256           segmentEncryptionIV = encryptionIV;
257         } else {
258           segmentEncryptionIV = Integer.toHexString(segmentMediaSequence);
259         }
260         segmentMediaSequence++;
261         if (segmentByterangeLength == C.LENGTH_UNBOUNDED) {
262           segmentByterangeOffset = 0;
263         }
264         segments.add(new Segment(line, segmentDurationSecs, segmentDiscontinuity,
265             segmentStartTimeUs, isEncrypted, encryptionKeyUri, segmentEncryptionIV,
266             segmentByterangeOffset, segmentByterangeLength));
267         segmentStartTimeUs += (long) (segmentDurationSecs * C.MICROS_PER_SECOND);
268         segmentDiscontinuity = false;
269         segmentDurationSecs = 0.0;
270         if (segmentByterangeLength != C.LENGTH_UNBOUNDED) {
271           segmentByterangeOffset += segmentByterangeLength;
272         }
273         segmentByterangeLength = C.LENGTH_UNBOUNDED;
274       } else if (line.equals(ENDLIST_TAG)) {
275         live = false;
276         break;
277       }
278     }
279     return new HlsMediaPlaylist(baseUri, mediaSequence, targetDurationSecs, version, live,
280         Collections.unmodifiableList(segments));
281   }

283   private static class LineIterator {
284 
285     private final BufferedReader reader;
286     private final Queue<String> extraLines;
287 
288     private String next;
289 
290     public LineIterator(Queue<String> extraLines, BufferedReader reader) {
291       this.extraLines = extraLines;
292       this.reader = reader;
293     }
294 
295     public boolean hasNext() throws IOException {
296       if (next != null) {
297         return true;
298       }
299       if (!extraLines.isEmpty()) {
300         next = extraLines.poll();
301         return true;
302       }
303       while ((next = reader.readLine()) != null) {
304         next = next.trim();
305         if (!next.isEmpty()) {
306           return true;
307         }
308       }
309       return false;
310     }
311 
312     public String next() throws IOException {
313       String result = null;
314       if (hasNext()) {
315         result = next;
316         next = null;
317       }
318       return result;
319     }
