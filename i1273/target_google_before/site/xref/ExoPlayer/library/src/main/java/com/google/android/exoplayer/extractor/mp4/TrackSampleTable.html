16  package com.google.android.exoplayer.extractor.mp4;
17  
18  import com.google.android.exoplayer.C;
19  import com.google.android.exoplayer.util.Assertions;
20  import com.google.android.exoplayer.util.Util;
21  
22  /** Sample table for a track in an MP4 file. */
23  /* package */ final class TrackSampleTable {
24  
25    /** Sample index when no sample is available. */
26    public static final int NO_SAMPLE = -1;
27  
28    /** Number of samples. */
29    public final int sampleCount;
30    /** Sample offsets in bytes. */
31    public final long[] offsets;
32    /** Sample sizes in bytes. */
33    public final int[] sizes;
34    /** Sample timestamps in microseconds. */
35    public final long[] timestampsUs;
36    /** Sample flags. */
37    public final int[] flags;
38  
39    TrackSampleTable(
40        long[] offsets, int[] sizes, long[] timestampsUs, int[] flags) {
41      Assertions.checkArgument(sizes.length == timestampsUs.length);
42      Assertions.checkArgument(offsets.length == timestampsUs.length);
43      Assertions.checkArgument(flags.length == timestampsUs.length);
44  
45      this.offsets = offsets;
46      this.sizes = sizes;
47      this.timestampsUs = timestampsUs;
48      this.flags = flags;
49      sampleCount = offsets.length;
50    }
51  
52    /**
53     * Returns the sample index of the closest synchronization sample at or before the given
54     * timestamp, if one is available.
55     *
56     * @param timeUs Timestamp adjacent to which to find a synchronization sample.
57     * @return Index of the synchronization sample, or {@link #NO_SAMPLE} if none.
58     */
59    public int getIndexOfEarlierOrEqualSynchronizationSample(long timeUs) {
60      int startIndex = Util.binarySearchFloor(timestampsUs, timeUs, true, false);
61      for (int i = startIndex; i >= 0; i--) {
62        if (timestampsUs[i] <= timeUs && (flags[i] & C.SAMPLE_FLAG_SYNC) != 0) {
63          return i;
64        }
65      }
66      return NO_SAMPLE;
67    }
68  
69    /**
70     * Returns the sample index of the closest synchronization sample at or after the given timestamp,
71     * if one is available.
72     *
73     * @param timeUs Timestamp adjacent to which to find a synchronization sample.
74     * @return index Index of the synchronization sample, or {@link #NO_SAMPLE} if none.
75     */
76    public int getIndexOfLaterOrEqualSynchronizationSample(long timeUs) {
77      int startIndex = Util.binarySearchCeil(timestampsUs, timeUs, true, false);
78      for (int i = startIndex; i < timestampsUs.length; i++) {
79        if (timestampsUs[i] >= timeUs && (flags[i] & C.SAMPLE_FLAG_SYNC) != 0) {
80          return i;
81        }
82      }
83      return NO_SAMPLE;
84    }
85  
86  }
