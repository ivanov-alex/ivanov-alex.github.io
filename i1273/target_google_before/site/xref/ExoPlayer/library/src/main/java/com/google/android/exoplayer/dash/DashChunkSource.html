243   /* package */ DashChunkSource(ManifestFetcher<MediaPresentationDescription> manifestFetcher,
244       MediaPresentationDescription initialManifest, int adaptationSetIndex,
245       int[] representationIndices, DataSource dataSource, FormatEvaluator formatEvaluator,
246       Clock systemClock, long liveEdgeLatencyUs, long elapsedRealtimeOffsetUs,
247       boolean startAtLiveEdge, Handler eventHandler, EventListener eventListener) {
248     this.manifestFetcher = manifestFetcher;
249     this.currentManifest = initialManifest;
250     this.adaptationSetIndex = adaptationSetIndex;
251     this.representationIndices = representationIndices;
252     this.dataSource = dataSource;
253     this.formatEvaluator = formatEvaluator;
254     this.systemClock = systemClock;
255     this.liveEdgeLatencyUs = liveEdgeLatencyUs;
256     this.elapsedRealtimeOffsetUs = elapsedRealtimeOffsetUs;
257     this.startAtLiveEdge = startAtLiveEdge;
258     this.eventHandler = eventHandler;
259     this.eventListener = eventListener;
260     this.evaluation = new Evaluation();
261     this.headerBuilder = new StringBuilder();
262     this.seekRangeValues = new long[2];
263 
264     drmInitData = getDrmInitData(currentManifest, adaptationSetIndex);
265     Representation[] representations = getFilteredRepresentations(currentManifest,
266         adaptationSetIndex, representationIndices);
267     long periodDurationUs = (representations[0].periodDurationMs == TrackRenderer.UNKNOWN_TIME_US)
268         ? TrackRenderer.UNKNOWN_TIME_US : representations[0].periodDurationMs * 1000;
269     this.trackInfo = new TrackInfo(representations[0].format.mimeType, periodDurationUs);
270 
271     this.formats = new Format[representations.length];
272     this.representationHolders = new HashMap<>();
273     int maxWidth = 0;
274     int maxHeight = 0;
275     for (int i = 0; i < representations.length; i++) {
276       formats[i] = representations[i].format;
277       maxWidth = Math.max(formats[i].width, maxWidth);
278       maxHeight = Math.max(formats[i].height, maxHeight);
279       Extractor extractor = mimeTypeIsWebm(formats[i].mimeType) ? new WebmExtractor()
280           : new FragmentedMp4Extractor();
281       representationHolders.put(formats[i].id,
282           new RepresentationHolder(representations[i], new ChunkExtractorWrapper(extractor)));
283     }
284     this.maxWidth = maxWidth;
285     this.maxHeight = maxHeight;
286     Arrays.sort(formats, new DecreasingBandwidthComparator());
287   }

306   @Override
307   public void enable() {
308     fatalError = null;
309     formatEvaluator.enable();
310     if (manifestFetcher != null) {
311       manifestFetcher.enable();
312     }
313     DashSegmentIndex segmentIndex =
314         representationHolders.get(formats[0].id).representation.getIndex();
315     if (segmentIndex == null) {
316       seekRange = new TimeRange(TimeRange.TYPE_SNAPSHOT, 0, currentManifest.duration * 1000);
317       notifySeekRangeChanged(seekRange);
318     } else {
319       long nowUs = getNowUs();
320       updateAvailableSegmentBounds(segmentIndex, nowUs);
321       updateSeekRange(segmentIndex, nowUs);
322     }
323   }

334   @Override
335   public void continueBuffering(long playbackPositionUs) {
336     if (manifestFetcher == null || !currentManifest.dynamic || fatalError != null) {
337       return;
338     }
339 
340     MediaPresentationDescription newManifest = manifestFetcher.getManifest();
341     if (currentManifest != newManifest && newManifest != null) {
342       Representation[] newRepresentations = DashChunkSource.getFilteredRepresentations(newManifest,
343           adaptationSetIndex, representationIndices);
344       for (Representation representation : newRepresentations) {
345         RepresentationHolder representationHolder =
346             representationHolders.get(representation.format.id);
347         DashSegmentIndex oldIndex = representationHolder.segmentIndex;
348         int oldIndexLastSegmentNum = oldIndex.getLastSegmentNum();
349         long oldIndexEndTimeUs = oldIndex.getTimeUs(oldIndexLastSegmentNum)
350             + oldIndex.getDurationUs(oldIndexLastSegmentNum);
351         DashSegmentIndex newIndex = representation.getIndex();
352         int newIndexFirstSegmentNum = newIndex.getFirstSegmentNum();
353         long newIndexStartTimeUs = newIndex.getTimeUs(newIndexFirstSegmentNum);
354         if (oldIndexEndTimeUs < newIndexStartTimeUs) {
355           // There's a gap between the old manifest and the new one which means we've slipped behind
356           // the live window and can't proceed.
357           fatalError = new BehindLiveWindowException();
358           return;
359         }
360         int segmentNumShift;
361         if (oldIndexEndTimeUs == newIndexStartTimeUs) {
362           // The new manifest continues where the old one ended, with no overlap.
363           segmentNumShift = oldIndex.getLastSegmentNum() + 1 - newIndexFirstSegmentNum;
364         } else {
365           // The new manifest overlaps with the old one.
366           segmentNumShift = oldIndex.getSegmentNum(newIndexStartTimeUs) - newIndexFirstSegmentNum;
367         }
368         representationHolder.segmentNumShift += segmentNumShift;
369         representationHolder.segmentIndex = newIndex;
370       }
371       currentManifest = newManifest;
372       finishedCurrentManifest = false;
373 
374       long nowUs = getNowUs();
375       updateAvailableSegmentBounds(newRepresentations[0].getIndex(), nowUs);
376       updateSeekRange(newRepresentations[0].getIndex(), nowUs);
377     }
378 
379     // TODO: This is a temporary hack to avoid constantly refreshing the MPD in cases where
380     // minUpdatePeriod is set to 0. In such cases we shouldn't refresh unless there is explicit
381     // signaling in the stream, according to:
382     // http://azure.microsoft.com/blog/2014/09/13/dash-live-streaming-with-azure-media-service/
383     long minUpdatePeriod = currentManifest.minUpdatePeriod;
384     if (minUpdatePeriod == 0) {
385       minUpdatePeriod = 5000;
386     }
387 
388     if (finishedCurrentManifest && (android.os.SystemClock.elapsedRealtime()
389         > manifestFetcher.getManifestLoadStartTimestamp() + minUpdatePeriod)) {
390       manifestFetcher.requestRefresh();
391     }
392   }

394   @Override
395   public final void getChunkOperation(List<? extends MediaChunk> queue, long seekPositionUs,
396       long playbackPositionUs, ChunkOperationHolder out) {
397     if (fatalError != null) {
398       out.chunk = null;
399       return;
400     }
401 
402     evaluation.queueSize = queue.size();
403     if (evaluation.format == null || !lastChunkWasInitialization) {
404       formatEvaluator.evaluate(queue, playbackPositionUs, formats, evaluation);
405     }
406     Format selectedFormat = evaluation.format;
407     out.queueSize = evaluation.queueSize;
408 
409     if (selectedFormat == null) {
410       out.chunk = null;
411       return;
412     } else if (out.queueSize == queue.size() && out.chunk != null
413         && out.chunk.format.equals(selectedFormat)) {
414       // We already have a chunk, and the evaluation hasn't changed either the format or the size
415       // of the queue. Leave unchanged.
416       return;
417     }
418 
419     // In all cases where we return before instantiating a new chunk, we want out.chunk to be null.
420     out.chunk = null;
421 
422     RepresentationHolder representationHolder = representationHolders.get(selectedFormat.id);
423     Representation selectedRepresentation = representationHolder.representation;
424     DashSegmentIndex segmentIndex = representationHolder.segmentIndex;
425     ChunkExtractorWrapper extractorWrapper = representationHolder.extractorWrapper;
426 
427     RangedUri pendingInitializationUri = null;
428     RangedUri pendingIndexUri = null;
429 
430     if (representationHolder.format == null) {
431       pendingInitializationUri = selectedRepresentation.getInitializationUri();
432     }
433     if (segmentIndex == null) {
434       pendingIndexUri = selectedRepresentation.getIndexUri();
435     }
436 
437     if (pendingInitializationUri != null || pendingIndexUri != null) {
438       // We have initialization and/or index requests to make.
439       Chunk initializationChunk = newInitializationChunk(pendingInitializationUri, pendingIndexUri,
440           selectedRepresentation, extractorWrapper, dataSource, evaluation.trigger);
441       lastChunkWasInitialization = true;
442       out.chunk = initializationChunk;
443       return;
444     }
445 
446     int segmentNum;
447     boolean indexUnbounded = segmentIndex.getLastSegmentNum() == DashSegmentIndex.INDEX_UNBOUNDED;
448     if (indexUnbounded) {
449       // Manifests with unbounded indexes aren't updated regularly, so we need to update the
450       // segment bounds before use to ensure that they are accurate to the current time; also if
451       // the bounds have changed, we should update the seek range
452       long nowUs = getNowUs();
453       int oldFirstAvailableSegmentNum = firstAvailableSegmentNum;
454       int oldLastAvailableSegmentNum = lastAvailableSegmentNum;
455       updateAvailableSegmentBounds(segmentIndex, nowUs);
456       if (oldFirstAvailableSegmentNum != firstAvailableSegmentNum
457           || oldLastAvailableSegmentNum != lastAvailableSegmentNum) {
458         updateSeekRange(segmentIndex, nowUs);
459       }
460     }
461     if (queue.isEmpty()) {
462       if (currentManifest.dynamic) {
463         seekRangeValues = seekRange.getCurrentBoundsUs(seekRangeValues);
464         if (startAtLiveEdge) {
465           // We want live streams to start at the live edge instead of the beginning of the
466           // manifest
467           startAtLiveEdge = false;
468           seekPositionUs = seekRangeValues[1];
469         } else {
470           seekPositionUs = Math.max(seekPositionUs, seekRangeValues[0]);
471           seekPositionUs = Math.min(seekPositionUs, seekRangeValues[1]);
472         }
473       }
474       segmentNum = segmentIndex.getSegmentNum(seekPositionUs);
475 
476       // if the index is unbounded then the result of getSegmentNum isn't clamped to ensure that
477       // it doesn't exceed the last available segment. Clamp it here.
478       if (indexUnbounded) {
479         segmentNum = Math.min(segmentNum, lastAvailableSegmentNum);
480       }
481     } else {
482       MediaChunk previous = queue.get(out.queueSize - 1);
483       segmentNum = previous.isLastChunk ? -1
484           : previous.chunkIndex + 1 - representationHolder.segmentNumShift;
485     }
486 
487     if (currentManifest.dynamic) {
488       if (segmentNum < firstAvailableSegmentNum) {
489         // This is before the first chunk in the current manifest.
490         fatalError = new BehindLiveWindowException();
491         return;
492       } else if (segmentNum > lastAvailableSegmentNum) {
493         // This chunk is beyond the last chunk in the current manifest. If the index is bounded
494         // we'll need to refresh it. If it's unbounded we just need to wait for a while before
495         // attempting to load the chunk.
496         finishedCurrentManifest = !indexUnbounded;
497         return;
498       } else if (!indexUnbounded && segmentNum == lastAvailableSegmentNum) {
499         // This is the last chunk in a dynamic bounded manifest. We'll need to refresh the manifest
500         // to obtain the next chunk.
501         finishedCurrentManifest = true;
502       }
503     }
504 
505     if (segmentNum == -1) {
506       // We've reached the end of the stream.
507       return;
508     }
509 
510     Chunk nextMediaChunk = newMediaChunk(representationHolder, dataSource, segmentNum,
511         evaluation.trigger);
512     lastChunkWasInitialization = false;
513     out.chunk = nextMediaChunk;
514   }

525   @Override
526   public void onChunkLoadCompleted(Chunk chunk) {
527     if (chunk instanceof InitializationChunk) {
528       InitializationChunk initializationChunk = (InitializationChunk) chunk;
529       String formatId = initializationChunk.format.id;
530       RepresentationHolder representationHolder = representationHolders.get(formatId);
531       if (initializationChunk.hasFormat()) {
532         representationHolder.format = initializationChunk.getFormat();
533       }
534       if (initializationChunk.hasSeekMap()) {
535         representationHolder.segmentIndex = new DashWrappingSegmentIndex(
536             (ChunkIndex) initializationChunk.getSeekMap(),
537             initializationChunk.dataSpec.uri.toString(),
538             representationHolder.representation.periodStartMs * 1000);
539       }
540       // The null check avoids overwriting drmInitData obtained from the manifest with drmInitData
541       // obtained from the stream, as per DASH IF Interoperability Recommendations V3.0, 7.5.3.
542       if (drmInitData == null && initializationChunk.hasDrmInitData()) {
543         drmInitData = initializationChunk.getDrmInitData();
544       }
545     }
546   }

553   private void updateAvailableSegmentBounds(DashSegmentIndex segmentIndex, long nowUs) {
554     int indexFirstAvailableSegmentNum = segmentIndex.getFirstSegmentNum();
555     int indexLastAvailableSegmentNum = segmentIndex.getLastSegmentNum();
556     if (indexLastAvailableSegmentNum == DashSegmentIndex.INDEX_UNBOUNDED) {
557       // The index is itself unbounded. We need to use the current time to calculate the range of
558       // available segments.
559       long liveEdgeTimestampUs = nowUs - currentManifest.availabilityStartTime * 1000;
560       if (currentManifest.timeShiftBufferDepth != -1) {
561         long bufferDepthUs = currentManifest.timeShiftBufferDepth * 1000;
562         indexFirstAvailableSegmentNum = Math.max(indexFirstAvailableSegmentNum,
563             segmentIndex.getSegmentNum(liveEdgeTimestampUs - bufferDepthUs));
564       }
565       // getSegmentNum(liveEdgeTimestampUs) will not be completed yet, so subtract one to get the
566       // index of the last completed segment.
567       indexLastAvailableSegmentNum = segmentIndex.getSegmentNum(liveEdgeTimestampUs) - 1;
568     }
569     firstAvailableSegmentNum = indexFirstAvailableSegmentNum;
570     lastAvailableSegmentNum = indexLastAvailableSegmentNum;
571   }

573   private void updateSeekRange(DashSegmentIndex segmentIndex, long nowUs) {
574     long earliestSeekPosition = segmentIndex.getTimeUs(firstAvailableSegmentNum);
575     long latestSeekPosition = segmentIndex.getTimeUs(lastAvailableSegmentNum)
576         + segmentIndex.getDurationUs(lastAvailableSegmentNum);
577     if (currentManifest.dynamic) {
578       long liveEdgeTimestampUs;
579       if (segmentIndex.getLastSegmentNum() == DashSegmentIndex.INDEX_UNBOUNDED) {
580         liveEdgeTimestampUs = nowUs - currentManifest.availabilityStartTime * 1000;
581       } else {
582         liveEdgeTimestampUs = segmentIndex.getTimeUs(segmentIndex.getLastSegmentNum())
583             + segmentIndex.getDurationUs(segmentIndex.getLastSegmentNum());
584         if (!segmentIndex.isExplicit()) {
585           // Some segments defined by the index may not be available yet. Bound the calculated live
586           // edge based on the elapsed time since the manifest became available.
587           liveEdgeTimestampUs = Math.min(liveEdgeTimestampUs,
588               nowUs - currentManifest.availabilityStartTime * 1000);
589         }
590       }
591 
592       // it's possible that the live edge latency actually puts our latest position before
593       // the earliest position in the case of a DVR-like stream that's just starting up, so
594       // in that case just return the earliest position instead
595       latestSeekPosition = Math.max(earliestSeekPosition, liveEdgeTimestampUs - liveEdgeLatencyUs);
596     }
597 
598     TimeRange newSeekRange = new TimeRange(TimeRange.TYPE_SNAPSHOT, earliestSeekPosition,
599         latestSeekPosition);
600     if (seekRange == null || !seekRange.equals(newSeekRange)) {
601       seekRange = newSeekRange;
602       notifySeekRangeChanged(seekRange);
603     }
604   }

606   private static boolean mimeTypeIsWebm(String mimeType) {
607     return mimeType.startsWith(MimeTypes.VIDEO_WEBM) || mimeType.startsWith(MimeTypes.AUDIO_WEBM);
608   }

610   private Chunk newInitializationChunk(RangedUri initializationUri, RangedUri indexUri,
611       Representation representation, ChunkExtractorWrapper extractor, DataSource dataSource,
612       int trigger) {
613     RangedUri requestUri;
614     if (initializationUri != null) {
615       // It's common for initialization and index data to be stored adjacently. Attempt to merge
616       // the two requests together to request both at once.
617       requestUri = initializationUri.attemptMerge(indexUri);
618       if (requestUri == null) {
619         requestUri = initializationUri;
620       }
621     } else {
622       requestUri = indexUri;
623     }
624     DataSpec dataSpec = new DataSpec(requestUri.getUri(), requestUri.start, requestUri.length,
625         representation.getCacheKey());
626     return new InitializationChunk(dataSource, dataSpec, trigger, representation.format, extractor);
627   }

629   private Chunk newMediaChunk(RepresentationHolder representationHolder, DataSource dataSource,
630       int segmentNum, int trigger) {
631     Representation representation = representationHolder.representation;
632     DashSegmentIndex segmentIndex = representationHolder.segmentIndex;
633 
634     long startTimeUs = segmentIndex.getTimeUs(segmentNum);
635     long endTimeUs = startTimeUs + segmentIndex.getDurationUs(segmentNum);
636 
637     int absoluteSegmentNum = segmentNum + representationHolder.segmentNumShift;
638     boolean isLastSegment = !currentManifest.dynamic
639         && segmentNum == segmentIndex.getLastSegmentNum();
640 
641     RangedUri segmentUri = segmentIndex.getSegmentUrl(segmentNum);
642     DataSpec dataSpec = new DataSpec(segmentUri.getUri(), segmentUri.start, segmentUri.length,
643         representation.getCacheKey());
644 
645     long sampleOffsetUs = representation.periodStartMs * 1000
646         - representation.presentationTimeOffsetUs;
647     if (representation.format.mimeType.equals(MimeTypes.TEXT_VTT)) {
648       if (representationHolder.vttHeaderOffsetUs != sampleOffsetUs) {
649         // Update the VTT header.
650         headerBuilder.setLength(0);
651         headerBuilder.append(C.WEBVTT_EXO_HEADER).append("=")
652             .append(C.WEBVTT_EXO_HEADER_OFFSET).append(sampleOffsetUs)
653             .append("\n");
654         representationHolder.vttHeader = headerBuilder.toString().getBytes();
655         representationHolder.vttHeaderOffsetUs = sampleOffsetUs;
656       }
657       return new SingleSampleMediaChunk(dataSource, dataSpec, Chunk.TRIGGER_INITIAL,
658           representation.format, startTimeUs, endTimeUs, absoluteSegmentNum, isLastSegment,
659           MediaFormat.createTextFormat(MimeTypes.TEXT_VTT), null, representationHolder.vttHeader);
660     } else {
661       return new ContainerMediaChunk(dataSource, dataSpec, trigger, representation.format,
662           startTimeUs, endTimeUs, absoluteSegmentNum, isLastSegment, sampleOffsetUs,
663           representationHolder.extractorWrapper, representationHolder.format, drmInitData, true);
664     }
665   }

667   private long getNowUs() {
668     if (elapsedRealtimeOffsetUs != 0) {
669       return (systemClock.elapsedRealtime() * 1000) + elapsedRealtimeOffsetUs;
670     } else {
671       return System.currentTimeMillis() * 1000;
672     }

675   private static Representation[] getFilteredRepresentations(MediaPresentationDescription manifest,
676       int adaptationSetIndex, int[] representationIndices) {
677     AdaptationSet adaptationSet = manifest.periods.get(0).adaptationSets.get(adaptationSetIndex);
678     List<Representation> representations = adaptationSet.representations;
679     if (representationIndices == null) {
680       Representation[] filteredRepresentations = new Representation[representations.size()];
681       representations.toArray(filteredRepresentations);
682       return filteredRepresentations;
683     } else {
684       Representation[] filteredRepresentations = new Representation[representationIndices.length];
685       for (int i = 0; i < representationIndices.length; i++) {
686         filteredRepresentations[i] = representations.get(representationIndices[i]);
687       }
688       return filteredRepresentations;
689     }
690   }

692   private static DrmInitData getDrmInitData(MediaPresentationDescription manifest,
693       int adaptationSetIndex) {
694     AdaptationSet adaptationSet = manifest.periods.get(0).adaptationSets.get(adaptationSetIndex);
695     String drmInitMimeType = mimeTypeIsWebm(adaptationSet.representations.get(0).format.mimeType)
696         ? MimeTypes.VIDEO_WEBM : MimeTypes.VIDEO_MP4;
697     if (adaptationSet.contentProtections.isEmpty()) {
698       return null;
699     } else {
700       DrmInitData.Mapped drmInitData = null;
701       for (ContentProtection contentProtection : adaptationSet.contentProtections) {
702         if (contentProtection.uuid != null && contentProtection.data != null) {
703           if (drmInitData == null) {
704             drmInitData = new DrmInitData.Mapped(drmInitMimeType);
705           }
706           drmInitData.put(contentProtection.uuid, contentProtection.data);
707         }
708       }
709       return drmInitData;
710     }
711   }

713   private static MediaPresentationDescription buildManifest(List<Representation> representations) {
714     Representation firstRepresentation = representations.get(0);
715     AdaptationSet adaptationSet = new AdaptationSet(0, AdaptationSet.TYPE_UNKNOWN, representations);
716     Period period = new Period(null, firstRepresentation.periodStartMs,
717         firstRepresentation.periodDurationMs, Collections.singletonList(adaptationSet));
718     long duration = firstRepresentation.periodDurationMs - firstRepresentation.periodStartMs;
719     return new MediaPresentationDescription(-1, duration, -1, false, -1, -1, null, null,
720         Collections.singletonList(period));
721   }
