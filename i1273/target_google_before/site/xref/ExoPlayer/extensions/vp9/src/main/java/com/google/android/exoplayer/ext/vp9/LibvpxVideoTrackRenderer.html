37  /**
38   * Decodes and renders video using the native VP9 decoder.
39   */
40  public class LibvpxVideoTrackRenderer extends TrackRenderer {
41  
42    /**
43     * Interface definition for a callback to be notified of {@link LibvpxVideoTrackRenderer} events.
44     */
45    public interface EventListener {
46  
47      /**
48       * Invoked to report the number of frames dropped by the renderer. Dropped frames are reported
49       * whenever the renderer is stopped having dropped frames, and optionally, whenever the count
50       * reaches a specified threshold whilst the renderer is started.
51       *
52       * @param count The number of dropped frames.
53       * @param elapsed The duration in milliseconds over which the frames were dropped. This
54       *     duration is timed from when the renderer was started or from when dropped frames were
55       *     last reported (whichever was more recent), and not from when the first of the reported
56       *     drops occurred.
57       */
58      void onDroppedFrames(int count, long elapsed);
59  
60      /**
61       * Invoked each time there's a change in the size of the video being rendered.
62       *
63       * @param width The video width in pixels.
64       * @param height The video height in pixels.
65       */
66      void onVideoSizeChanged(int width, int height);
67  
68      /**
69       * Invoked when a frame is rendered to a surface for the first time following that surface
70       * having been set as the target for the renderer.
71       *
72       * @param surface The surface to which a first frame has been rendered.
73       */
74      void onDrawnToSurface(Surface surface);
75  
76      /**
77       * Invoked when one of the following happens: libvpx initialization failure, decoder error,
78       * renderer error.
79       *
80       * @param e The corresponding exception.
81       */
82      void onDecoderError(VpxDecoderException e);
83  
84    }
85  
86    /**
87     * The type of a message that can be passed to an instance of this class via
88     * {@link ExoPlayer#sendMessage} or {@link ExoPlayer#blockingSendMessage}. The message object
89     * should be the target {@link Surface}, or null.
90     */
91    public static final int MSG_SET_SURFACE = 1;
92    public static final int MSG_SET_VPX_SURFACE_VIEW = 2;

128   public LibvpxVideoTrackRenderer(SampleSource source, boolean scaleToFit) {
129     this(source, scaleToFit, null, null, 0);

132   /**
133    * @param source The upstream source from which the renderer obtains samples.
134    * @param scaleToFit Boolean that indicates if video frames should be scaled to fit when
135    *     rendering.
136    * @param eventHandler A handler to use when delivering events to {@code eventListener}. May be
137    *     null if delivery of events is not required.
138    * @param eventListener A listener of events. May be null if delivery of events is not required.
139    * @param maxDroppedFrameCountToNotify The maximum number of frames that can be dropped between
140    *     invocations of {@link EventListener#onDroppedFrames(int, long)}.
141    */
142   public LibvpxVideoTrackRenderer(SampleSource source, boolean scaleToFit,
143       Handler eventHandler, EventListener eventListener, int maxDroppedFrameCountToNotify) {
144     this.source = source.register();
145     this.scaleToFit = scaleToFit;
146     this.eventHandler = eventHandler;
147     this.eventListener = eventListener;
148     this.maxDroppedFrameCountToNotify = maxDroppedFrameCountToNotify;
149     previousWidth = -1;
150     previousHeight = -1;
151     formatHolder = new MediaFormatHolder();
152   }

154   @Override
155   protected int doPrepare(long positionUs) throws ExoPlaybackException {
156     boolean sourcePrepared = source.prepare(positionUs);
157     if (!sourcePrepared) {
158       return TrackRenderer.STATE_UNPREPARED;
159     }
160     int trackCount = source.getTrackCount();
161     for (int i = 0; i < trackCount; i++) {
162       if (source.getTrackInfo(i).mimeType.equalsIgnoreCase(MimeTypes.VIDEO_VP9)
163           || source.getTrackInfo(i).mimeType.equalsIgnoreCase(MimeTypes.VIDEO_WEBM)) {
164         trackIndex = i;
165         return TrackRenderer.STATE_PREPARED;
166       }
167     }
168     return TrackRenderer.STATE_IGNORE;
169   }
170 
171   @Override
172   protected void doSomeWork(long positionUs, long elapsedRealtimeUs) throws ExoPlaybackException {
173     if (outputStreamEnded) {
174       return;
175     }
176     sourceIsReady = source.continueBuffering(trackIndex, positionUs);
177     checkForDiscontinuity(positionUs);
178 
179     // Try and read a format if we don't have one already.
180     if (format == null && !readFormat(positionUs)) {
181       // We can't make progress without one.
182       return;
183     }
184 
185     // If we don't have a decoder yet, we need to instantiate one.
186     // TODO: Add support for dynamic switching between one type of surface to another.
187     if (decoder == null) {
188       decoder = new VpxDecoderWrapper(outputRgb);
189       decoder.start();
190     }
191 
192     // Rendering loop.
193     try {
194       processOutputBuffer(positionUs, elapsedRealtimeUs);
195       while (feedInputBuffer(positionUs)) {}
196     } catch (VpxDecoderException e) {
197       notifyDecoderError(e);
198       throw new ExoPlaybackException(e);
199     }
200   }
201 
202   private void processOutputBuffer(long positionUs, long elapsedRealtimeUs)
203       throws VpxDecoderException {
204     if (outputStreamEnded) {
205       return;
206     }
207 
208     if (outputBuffer == null) {
209       outputBuffer = decoder.dequeueOutputBuffer();
210       if (outputBuffer == null) {
211         return;
212       }
213     }
214 
215     if (outputBuffer.flags == VpxDecoderWrapper.FLAG_END_OF_STREAM) {
216       outputStreamEnded = true;
217       releaseOutputBuffer();
218       return;
219     }
220 
221     long elapsedSinceStartOfLoop = SystemClock.elapsedRealtime() * 1000 - elapsedRealtimeUs;
222     long timeToRenderUs = outputBuffer.timestampUs - positionUs - elapsedSinceStartOfLoop;
223 
224     if (timeToRenderUs < -30000 || outputBuffer.timestampUs < positionUs) {
225       // Drop frame if we are too late.
226       droppedFrameCount++;
227       if (droppedFrameCount == maxDroppedFrameCountToNotify) {
228         notifyAndResetDroppedFrameCount();
229       }
230       releaseOutputBuffer();
231       return;
232     }
233 
234     // If we have not renderered any frame so far (either initially or immediately following a
235     // seek), render one frame irresepective of the state.
236     if (!renderedFirstFrame) {
237       renderBuffer();
238       renderedFirstFrame = true;
239       return;
240     }
241 
242     // Do nothing if we are not playing or if we are too early to render the next frame.
243     if (getState() != TrackRenderer.STATE_STARTED || timeToRenderUs > 30000) {
244       return;
245     }
246 
247     if (timeToRenderUs > 11000) {
248       try {
249         // Subtracting 10000 rather than 11000 ensures that the sleep time
250         // will be at least 1ms.
251         Thread.sleep((timeToRenderUs - 10000) / 1000);
252       } catch (InterruptedException e) {
253         Thread.currentThread().interrupt();
254       }
255     }
256     renderBuffer();
257   }
258 
259   private void renderBuffer() throws VpxDecoderException {
260     notifyIfVideoSizeChanged(outputBuffer);
261     if (outputRgb) {
262       renderRgbFrame(outputBuffer, scaleToFit);
263     } else {
264       vpxVideoSurfaceView.renderFrame(outputBuffer);
265     }
266     if (!drawnToSurface) {
267       drawnToSurface = true;
268       notifyDrawnToSurface(surface);
269     }
270     releaseOutputBuffer();
271   }
272 
273   private void releaseOutputBuffer() throws VpxDecoderException {
274     decoder.releaseOutputBuffer(outputBuffer);
275     outputBuffer = null;
276   }
277 
278   private void renderRgbFrame(OutputBuffer outputBuffer, boolean scale) {
279     if (bitmap == null || bitmap.getWidth() != outputBuffer.width
280         || bitmap.getHeight() != outputBuffer.height) {
281       bitmap = Bitmap.createBitmap(outputBuffer.width, outputBuffer.height, Bitmap.Config.RGB_565);
282     }
283     bitmap.copyPixelsFromBuffer(outputBuffer.data);
284     Canvas canvas = surface.lockCanvas(null);
285     if (scale) {
286       canvas.scale(((float) canvas.getWidth()) / outputBuffer.width,
287           ((float) canvas.getHeight()) / outputBuffer.height);
288     }
289     canvas.drawBitmap(bitmap, 0, 0, null);
290     surface.unlockCanvasAndPost(canvas);
291   }

293   private boolean feedInputBuffer(long positionUs) throws VpxDecoderException {
294     if (inputStreamEnded) {
295       return false;
296     }
297 
298     if (inputBuffer == null) {
299       inputBuffer = decoder.getInputBuffer();
300       if (inputBuffer == null) {
301         return false;
302       }
303     }
304 
305     int result = source.readData(trackIndex, positionUs, formatHolder, inputBuffer.sampleHolder,
306         false);
307     if (result == SampleSource.NOTHING_READ) {
308       return false;
309     }
310     if (result == SampleSource.DISCONTINUITY_READ) {
311       flushDecoder();
312       return true;
313     }
314     if (result == SampleSource.FORMAT_READ) {
315       format = formatHolder.format;
316       return true;
317     }
318     if (result == SampleSource.END_OF_STREAM) {
319       inputBuffer.flags = VpxDecoderWrapper.FLAG_END_OF_STREAM;
320       decoder.queueInputBuffer(inputBuffer);
321       inputBuffer = null;
322       inputStreamEnded = true;
323       return false;
324     }
325 
326     inputBuffer.width = format.width;
327     inputBuffer.height = format.height;
328     decoder.queueInputBuffer(inputBuffer);
329     inputBuffer = null;
330     return true;
331   }
332 
333   private void checkForDiscontinuity(long positionUs) {
334     if (decoder == null) {
335       return;
336     }
337     int result = source.readData(trackIndex, positionUs, formatHolder, null, true);
338     if (result == SampleSource.DISCONTINUITY_READ) {
339       flushDecoder();
340     }
341   }
342 
343   private void flushDecoder() {
344     inputBuffer = null;
345     outputBuffer = null;
346     decoder.flush();
347   }

381   private void seekToInternal() {
382     sourceIsReady = false;
383     inputStreamEnded = false;
384     outputStreamEnded = false;
385     renderedFirstFrame = false;
386   }
387 
388   @Override
389   protected void onStarted() {
390     droppedFrameCount = 0;
391     droppedFrameAccumulationStartTimeMs = SystemClock.elapsedRealtime();
392   }
393 
394   @Override
395   protected void onStopped() {
396     notifyAndResetDroppedFrameCount();
397   }
398 
399   @Override
400   protected void onReleased() {
401     source.release();
402   }
403 
404   @Override
405   protected void onDisabled() {
406     if (decoder != null) {
407       decoder.release();
408       decoder = null;
409     }
410     inputBuffer = null;
411     outputBuffer = null;
412     format = null;
413     source.disable(trackIndex);
414   }
415 
416   @Override
417   protected void maybeThrowError() throws ExoPlaybackException {
418     try {
419       source.maybeThrowError();
420     } catch (IOException e) {
421       throw new ExoPlaybackException(e);
422     }
423   }

425   private boolean readFormat(long positionUs) {
426     int result = source.readData(trackIndex, positionUs, formatHolder, null, false);
427     if (result == SampleSource.FORMAT_READ) {
428       format = formatHolder.format;
429       return true;
430     }
431     return false;
432   }
433 
434   @Override
435   public void handleMessage(int messageType, Object message) throws ExoPlaybackException {
436     if (messageType == MSG_SET_SURFACE) {
437       surface = (Surface) message;
438       vpxVideoSurfaceView = null;
439       outputRgb = true;
440     } else if (messageType == MSG_SET_VPX_SURFACE_VIEW) {
441       vpxVideoSurfaceView = (VpxVideoSurfaceView) message;
442       surface = null;
443       outputRgb = false;
444     } else {
445       super.handleMessage(messageType, message);
446     }
447   }

449   private void notifyIfVideoSizeChanged(final OutputBuffer outputBuffer) {
450     if (previousWidth == -1 || previousHeight == -1
451         || previousWidth != outputBuffer.width || previousHeight != outputBuffer.height) {
452       previousWidth = outputBuffer.width;
453       previousHeight = outputBuffer.height;
454       if (eventHandler != null && eventListener != null) {
455         eventHandler.post(new Runnable()  {
456           @Override
457           public void run() {
458             eventListener.onVideoSizeChanged(outputBuffer.width, outputBuffer.height);
459           }
460         });
461       }
462     }
463   }
464 
465   private void notifyAndResetDroppedFrameCount() {
466     if (eventHandler != null && eventListener != null && droppedFrameCount > 0) {
467       long now = SystemClock.elapsedRealtime();
468       final int countToNotify = droppedFrameCount;
469       final long elapsedToNotify = now - droppedFrameAccumulationStartTimeMs;
470       droppedFrameCount = 0;
471       droppedFrameAccumulationStartTimeMs = now;
472       eventHandler.post(new Runnable()  {
473         @Override
474         public void run() {
475           eventListener.onDroppedFrames(countToNotify, elapsedToNotify);
476         }
477       });
478     }
479   }
480 
481   private void notifyDrawnToSurface(final Surface surface) {
482     if (eventHandler != null && eventListener != null) {
483       eventHandler.post(new Runnable()  {
484         @Override
485         public void run() {
486           eventListener.onDrawnToSurface(surface);
487         }
488       });
489     }
490   }
491 
492   private void notifyDecoderError(final VpxDecoderException e) {
493     if (eventHandler != null && eventListener != null) {
494       eventHandler.post(new Runnable()  {
495         @Override
496         public void run() {
497           eventListener.onDecoderError(e);
498         }
499       });
500     }
