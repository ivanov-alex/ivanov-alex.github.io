24      @BeforeClass

29      }
30  
31      @Test
32      public void testClassDefault() throws IOException, Exception {
33  
34          final String[] expected = {
35              "5:31: " + getCheckMessage(checkConfig.getMessages(), msgKey, "t", format),
36              "13:14: " + getCheckMessage(checkConfig.getMessages(), msgKey, "foo", format),
37              "27:24: " + getCheckMessage(checkConfig.getMessages(), msgKey, "$foo", format),
38          };
39  
40          String filePath = builder.getFilePath("ClassTypeParameterNameInput");
41          
42          Integer[] warnList = builder.getLinesWithWarn(filePath);

44      }
