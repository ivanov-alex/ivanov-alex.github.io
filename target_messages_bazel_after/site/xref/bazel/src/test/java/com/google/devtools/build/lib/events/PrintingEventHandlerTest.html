14  package com.google.devtools.build.lib.events;
15  
16  import static com.google.common.truth.Truth.assertThat;
17  
18  import com.google.devtools.build.lib.testutil.MoreAsserts;
19  import com.google.devtools.build.lib.util.io.RecordingOutErr;
20  
21  import org.junit.Test;
22  import org.junit.runner.RunWith;
23  import org.junit.runners.JUnit4;
24  
25  /**
26   * Tests the {@link PrintingEventHandler}.
27   */
28  @RunWith(JUnit4.class)
29  public class PrintingEventHandlerTest extends EventTestTemplate {
30  
31    @Test
32    public void collectsEvents() {
33      RecordingOutErr recordingOutErr = new RecordingOutErr();
34      PrintingEventHandler handler = new PrintingEventHandler(EventKind.ERRORS_AND_WARNINGS);
35      handler.setOutErr(recordingOutErr);
36      handler.handle(event);
37      MoreAsserts.assertEqualsUnifyingLineEnds("WARNING: /path/to/workspace/my/sample/path.txt:3:4: "
38                   + "This is not an error message.\n",
39                   recordingOutErr.errAsLatin1());
40      assertThat(recordingOutErr.outAsLatin1()).isEmpty();
41    }
42  
43  }
