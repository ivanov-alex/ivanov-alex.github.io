355    * Creates BuildView using current hostConfig/targetConfig values.
356    * Ensures that hostConfig is either identical to the targetConfig or has
357    * 'host' short name.
358    */
359   protected final void createBuildView() throws Exception {
360     Preconditions.checkNotNull(masterConfig);
361     Preconditions.checkState(getHostConfiguration() == getTargetConfiguration()
362         || getHostConfiguration().isHostConfiguration(),
363         "Host configuration %s is not a host configuration' "
364         + "and does not match target configuration %s",
365         getHostConfiguration(), getTargetConfiguration());
366 
367     String defaultsPackageContent = ruleClassProvider.getDefaultsPackageContent(optionsParser);
368     skyframeExecutor.setupDefaultPackage(defaultsPackageContent);
369     skyframeExecutor.dropConfiguredTargets();
370 
371     view = new BuildView(directories, getPackageManager(), ruleClassProvider, skyframeExecutor,
372         binTools, null);
373     view.setConfigurationsForTesting(masterConfig);
374 
375     view.setArtifactRoots(
376         ImmutableMap.of(PackageIdentifier.createInDefaultRepo(""), rootDirectory));
377     simulateLoadingPhase();
378   }

396   /**
397    * Asserts that a target's prerequisites contain the given dependency.
398    */
399   // TODO(bazel-team): replace this method with assertThat(iterable).contains(target).
400   // That doesn't work now because dynamic configurations aren't yet applied to top-level targets.
401   // This means that getConfiguredTarget("//go:two") returns a different configuration than
402   // requesting "//go:two" as a dependency. So the configured targets aren't considered "equal".
403   // Once we apply dynamic configs to top-level targets this discrepancy will go away.
404   protected void assertDirectPrerequisitesContain(ConfiguredTarget target, ConfiguredTarget dep) {
405     Iterable<ConfiguredTarget> prereqs = getDirectPrerequisites(target);
406     BuildConfiguration depConfig = dep.getConfiguration();
407     for (ConfiguredTarget contained : prereqs) {
408       if (contained.getLabel().equals(dep.getLabel())) {
409         BuildConfiguration containedConfig = contained.getConfiguration();
410         if (containedConfig == null && depConfig == null) {
411           return;
412         } else if (containedConfig != null
413             && depConfig != null
414             && containedConfig.cloneOptions().equals(depConfig.cloneOptions())) {
415           return;
416         }
417       }
418     }
419     fail("Cannot find " + target.toString() + " in " + prereqs.toString());
420   }

1005    * Verifies whether the rule checks the 'srcs' attribute validity.
1006    *
1007    * <p>At the call site it expects the {@code packageName} to contain:
1008    * <ol>
1009    *   <li>{@code :gvalid} - genrule that outputs a valid file</li>
1010    *   <li>{@code :ginvalid} - genrule that outputs an invalid file</li>
1011    *   <li>{@code :gmix} - genrule that outputs a mix of valid and invalid
1012    *       files</li>
1013    *   <li>{@code :valid} - rule of type {@code ruleType} that has a valid
1014    *       file, {@code :gvalid} and {@code :gmix} in the srcs</li>
1015    *   <li>{@code :invalid} - rule of type {@code ruleType} that has an invalid
1016    *       file, {@code :ginvalid} in the srcs</li>
1017    *   <li>{@code :mix} - rule of type {@code ruleType} that has a valid and an
1018    *       invalid file in the srcs</li>
1019    * </ol>
1020    *
1021    * @param packageName the package where the rules under test are located
1022    * @param ruleType rules under test types
1023    * @param expectedTypes expected file types
1024    */
1025   protected void assertSrcsValidityForRuleType(String packageName, String ruleType,
1026       String expectedTypes) throws Exception {
1027     reporter.removeHandler(failFastHandler);
1028     String descriptionSingle = ruleType + " srcs file (expected " + expectedTypes + ")";
1029     String descriptionPlural = ruleType + " srcs files (expected " + expectedTypes + ")";

1035     assertSrcsValidity(ruleType, packageName + ":invalid", true,
1036         "file '" + packageName + ":a.foo' is misplaced here " + descriptionPluralFile,
1037         "'" + packageName + ":ginvalid' does not produce any " + descriptionPlural);
1038     assertSrcsValidity(ruleType, packageName + ":mix", true,
1039         "'" + packageName + ":a.foo' does not produce any " + descriptionPlural);
