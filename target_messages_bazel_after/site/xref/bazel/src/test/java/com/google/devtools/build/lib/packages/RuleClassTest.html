261   public void testCreateRule() throws Exception {
262     RuleClass ruleClassA = createRuleClassA();
263 
264     // LinkedHashMap -> predictable iteration order for testing
265     Map<String, Object> attributeValues = new LinkedHashMap<>();
266     attributeValues.put("my-labellist-attr", "foobar"); // wrong type
267     attributeValues.put("bogus-attr", "foobar"); // no such attr
268     attributeValues.put("my-stringlist-attr", Arrays.asList("foo", "bar"));
269 
270     reporter.removeHandler(failFastHandler);
271     EventCollector collector = new EventCollector(EventKind.ERRORS);
272     reporter.addHandler(collector);
273 
274     Rule rule = createRule(ruleClassA, TEST_RULE_NAME, attributeValues, testRuleLocation);
275 
276     // TODO(blaze-team): (2009) refactor to use assertContainsEvent
277     Iterator<String> expectedMessages = Arrays.asList(
278         "expected value of type 'list(label)' for attribute 'my-labellist-attr' "
279         + "in 'ruleA' rule, but got \"foobar\" (string)",
280         "no such attribute 'bogus-attr' in 'ruleA' rule",
281         "missing value for mandatory "
282         + "attribute 'my-string-attr' in 'ruleA' rule",
283         "missing value for mandatory attribute 'my-label-attr' in 'ruleA' rule",
284         "missing value for mandatory "
285         + "attribute 'my-labellist-attr' in 'ruleA' rule",
286         "missing value for mandatory "
287         + "attribute 'my-string-attr2' in 'ruleA' rule"
288     ).iterator();
289 
290     for (Event event : collector) {
291       assertEquals(TEST_RULE_DEFINED_AT_LINE,
292           event.getLocation().getStartLineAndColumn().getLine());
293       assertEquals(testBuildfilePath.asFragment(), event.getLocation().getPath());
294       assertEquals(TEST_RULE_LABEL + ": " + expectedMessages.next(), event.getMessage());
295     }
296 
297     // Test basic rule properties:
298     assertEquals("ruleA",
299                  rule.getRuleClass());
300     assertEquals(TEST_RULE_NAME,
301                  rule.getName());
302     assertEquals(TEST_RULE_LABEL,
303                  rule.getLabel().toString());
304 
305     // Test attribute access:
306     AttributeMap attributes = RawAttributeMapper.of(rule);
307     assertEquals("//default:label",
308                  attributes.get("my-label-attr", Type.LABEL).toString());
309     assertEquals(42,
310                  attributes.get("my-integer-attr", Type.INTEGER).intValue());
311     assertEquals("",  // missing attribute -> default chosen based on type
312                  attributes.get("my-string-attr", Type.STRING));
313     assertThat(attributes.get("my-labellist-attr", Type.LABEL_LIST)).isEmpty();
314     assertEquals(Arrays.asList("foo", "bar"),
315                  attributes.get("my-stringlist-attr", Type.STRING_LIST));
316     try {
317       attributes.get("my-labellist-attr", Type.STRING); // wrong type
318       fail();
319     } catch (IllegalArgumentException e) {
320       assertThat(e).hasMessage("Attribute my-labellist-attr is not of type string "
321           + "in rule //testpackage:my-rule-A");
322     }
323   }

783    * under --compile_one_dependency.

785   public void testPreferredDependencyChecker() throws Exception {
786     final String cppFile = "file.cc";
787     final String textFile = "file.txt";
788 
789     // Default: not preferred for anything.
790     RuleClass defaultClass = new RuleClass.Builder("defaultClass", RuleClassType.NORMAL, false)
791         .factory(DUMMY_CONFIGURED_TARGET_FACTORY)
792         .add(attr("tags", STRING_LIST))
793         .build();
794     final Rule defaultRule = createRule(defaultClass, "defaultRule",
795         Collections.<String, Object>emptyMap(), testRuleLocation);
796     assertFalse(defaultRule.getRuleClassObject().isPreferredDependency(cppFile));
797     assertFalse(defaultRule.getRuleClassObject().isPreferredDependency(textFile));
798 
799     // Make a rule that's preferred for C++ sources.
800     RuleClass cppClass = new RuleClass.Builder("cppClass", RuleClassType.NORMAL, false)
801         .factory(DUMMY_CONFIGURED_TARGET_FACTORY)
802         .add(attr("tags", STRING_LIST))
803         .setPreferredDependencyPredicate(new Predicate<String>() {
804           @Override
805           public boolean apply(String filename) {
806             return filename.endsWith(".cc");
807           }
808         })
809         .build();
810     final Rule cppRule = createRule(cppClass, "cppRule",
811         Collections.<String, Object>emptyMap(), testRuleLocation);
812     assertTrue(cppRule.getRuleClassObject().isPreferredDependency(cppFile));
813     assertFalse(cppRule.getRuleClassObject().isPreferredDependency(textFile));
814   }
