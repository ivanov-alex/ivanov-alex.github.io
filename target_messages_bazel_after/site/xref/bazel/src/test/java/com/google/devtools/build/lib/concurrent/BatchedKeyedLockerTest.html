37  /** Base class for tests for {@link BatchedKeyedLocker} implementations. */
38  public abstract class BatchedKeyedLockerTest extends KeyedLockerTest {
39    private BatchedKeyedLocker<String> batchLocker;
40  
41    protected abstract BatchedKeyedLocker.Factory<String> getLockerFactory();
42  
43    @Override
44    protected BatchedKeyedLocker<String> makeFreshLocker() {
45      return getLockerFactory().create(Ordering.<String>natural());
46    }
47  
48    @Before
49    public void setUp_BatchedKeyedLockerTest() {
50      batchLocker = makeFreshLocker();
51    }
52  
53    private Supplier<AutoUnlocker> makeBatchLockInvoker(Set<String> keys) {
54      // The collection returned by Collections2#permutations isn't thread-safe.
55      ImmutableList<List<String>> perms = ImmutableList.copyOf(Collections2.permutations(keys));
56      final Iterator<List<String>> permutationsIter = Iterables.cycle(perms).iterator();
57      return new Supplier<KeyedLocker.AutoUnlocker>() {
58        @Override
59        public AutoUnlocker get() {
60          Set<String> keys;
61          synchronized (this) {
62            // The Iterable returned by Iterables#cycle isn't threadsafe.
63            keys = ImmutableSet.copyOf(permutationsIter.next());
64          }
65          // Each time, we call lockBatch with a set whose iteration order is different, for the sake
66          // of trying to tickle hypothetical concurrency bugs resulting from bad KeyedLocker
67          // implementations not being careful about ordering.
68          return batchLocker.lockBatch(keys);
69        }
70      };
71    }
72  
73    private Supplier<AutoUnlocker> makeBatchLockFn1() {
74      return makeBatchLockInvoker(ImmutableSet.of("1a", "1b", "1c", "1d", "1e"));
75    }
76  
77    private Supplier<AutoUnlocker> makeBatchLockFn2() {
78      return makeBatchLockInvoker(ImmutableSet.of("2a", "2b", "2c", "2d", "2e"));
79    }

121   @Test
122   public void testMixOfLockAndLockBatch_MutualExclusion() throws Exception {
123     final AtomicInteger count = new AtomicInteger(0);
124     final AtomicLongMap<String> mutexCounters = AtomicLongMap.create();
125     Set<Set<String>> powerSet = Sets.powerSet(
126         ImmutableSet.of("k1", "k2", "k3", "k4", "k5", "k6", "k8", "k9", "k10"));
127     for (final Set<String> keys : powerSet) {
128       executorService.submit(wrapper.wrap(new Runnable() {
129         @Override
130         public void run() {
131           if (keys.size() == 1) {
132             String key = Iterables.getOnlyElement(keys);
133             try (AutoUnlocker unlocker = batchLocker.lock(key)) {
134               long newCount = mutexCounters.addAndGet(key, 1);
135               assertEquals(1, newCount);
136               mutexCounters.decrementAndGet(key);
137             }
138           } else if (keys.size() > 1) {
139             try (AutoUnlocker unlocker = batchLocker.lockBatch(keys)) {
140               for (String key : keys) {
141                 long newCount = mutexCounters.addAndGet(key, 1);
142                 assertEquals(1, newCount);
143                 mutexCounters.decrementAndGet(key);
144               }
145             }
146           }
147           count.incrementAndGet();
148         }
149       }));
150     }
151     boolean interrupted = ExecutorUtil.interruptibleShutdown(executorService);
152     Throwables.propagateIfPossible(wrapper.getFirstThrownError());
153     if (interrupted) {
154       Thread.currentThread().interrupt();
155       throw new InterruptedException();
156     }
157     assertEquals(powerSet.size(), count.get());
158   }
