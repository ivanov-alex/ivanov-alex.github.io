48    @Test
49    public void testDuplicatedElementsOf() {
50      assertDups(ImmutableList.<Integer>of(), ImmutableSet.<Integer>of());
51      assertDups(ImmutableList.of(0), ImmutableSet.<Integer>of());
52      assertDups(ImmutableList.of(0, 0, 0), ImmutableSet.of(0));
53      assertDups(ImmutableList.of(1, 2, 3, 1, 2, 3), ImmutableSet.of(1, 2, 3));
54      assertDups(ImmutableList.of(1, 2, 3, 1, 2, 3, 4), ImmutableSet.of(1, 2, 3));
55      assertDups(ImmutableList.of(1, 2, 3, 4), ImmutableSet.<Integer>of());
56    }

58    private static void assertDups(List<Integer> collection, Set<Integer> dups) {
59      assertEquals(dups, CollectionUtils.duplicatedElementsOf(collection));
60    }

62    @Test
63    public void testIsImmutable() throws Exception {
64      assertTrue(CollectionUtils.isImmutable(ImmutableList.of(1, 2, 3)));
65      assertTrue(CollectionUtils.isImmutable(ImmutableSet.of(1, 2, 3)));
66  
67      NestedSet<Integer> ns = NestedSetBuilder.<Integer>compileOrder()
68          .add(1).add(2).add(3).build();
69      assertTrue(CollectionUtils.isImmutable(ns));
70  
71      NestedSet<Integer> ns2 = NestedSetBuilder.<Integer>linkOrder().add(1).add(2).add(3).build();
72      assertTrue(CollectionUtils.isImmutable(ns2));
73  
74      IterablesChain<Integer> chain = IterablesChain.<Integer>builder().addElement(1).build();
75  
76      assertTrue(CollectionUtils.isImmutable(chain));
77  
78      assertFalse(CollectionUtils.isImmutable(Lists.newArrayList()));
79      assertFalse(CollectionUtils.isImmutable(Lists.newLinkedList()));
80      assertFalse(CollectionUtils.isImmutable(Sets.newHashSet()));
81      assertFalse(CollectionUtils.isImmutable(Sets.newLinkedHashSet()));
82  
83      // The result of Iterables.concat() actually is immutable, but we have no way of checking if
84      // a given Iterable comes from concat().
85      assertFalse(CollectionUtils.isImmutable(Iterables.concat(ns, ns2)));
86  
87      // We can override the check by using the ImmutableIterable wrapper.
88      assertTrue(CollectionUtils.isImmutable(
89          ImmutableIterable.from(Iterables.concat(ns, ns2))));
90    }

105   @Test
106   public void testMakeImmutable() throws Exception {
107     Iterable<Integer> immutableList = ImmutableList.of(1, 2, 3);
108     assertSame(immutableList, CollectionUtils.makeImmutable(immutableList));
109 
110     Iterable<Integer> mutableList = Lists.newArrayList(1, 2, 3);
111     Iterable<Integer> converted = CollectionUtils.makeImmutable(mutableList);
112     assertNotSame(mutableList, converted);
113     assertEquals(mutableList, ImmutableList.copyOf(converted));
114   }

135   private <T extends Enum<T>> void assertAllDifferent(Class<T> clazz) throws Exception {
136     Set<EnumSet<T>> allSets = new HashSet<>();
137 
138     int maxBits = 1 << clazz.getEnumConstants().length;
139     for (int i = 0; i < maxBits; i++) {
140       EnumSet<T> set = CollectionUtils.fromBits(i, clazz);
141       int back = CollectionUtils.toBits(set);
142       assertEquals(back, i);  // Assert that a roundtrip is idempotent
143       allSets.add(set);
144     }
145 
146     assertThat(allSets).hasSize(maxBits); // Assert that every decoded value is different
147   }

149   @Test
150   public void testEnumBitfields() throws Exception {
151     assertEquals(0, CollectionUtils.<Small>toBits());
152     assertEquals(EnumSet.noneOf(Small.class), CollectionUtils.fromBits(0, Small.class));
153     assertEquals(3, CollectionUtils.toBits(Small.ALPHA, Small.BRAVO));
154     assertEquals(10, CollectionUtils.toBits(Medium.TWO, Medium.FOUR));
155     assertEquals(EnumSet.of(Medium.SEVEN, Medium.EIGHT),
156         CollectionUtils.fromBits(192, Medium.class));
157 
158     assertAllDifferent(Small.class);
159     assertAllDifferent(Medium.class);
160     assertAllDifferent(Large.class);
161 
162     try {
163       CollectionUtils.toBits(TooLarge.T32);
164       fail();
165     } catch (IllegalArgumentException e) {
166       // good
167     }
168 
169     try {
170       CollectionUtils.fromBits(0, TooLarge.class);
171       fail();
172     } catch (IllegalArgumentException e) {
173       // good
174     }
175   }
