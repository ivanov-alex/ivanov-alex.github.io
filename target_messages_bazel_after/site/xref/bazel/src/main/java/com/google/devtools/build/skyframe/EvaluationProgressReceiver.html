14  package com.google.devtools.build.skyframe;
15  
16  import com.google.devtools.build.lib.concurrent.ThreadSafety;
17  
18  import javax.annotation.Nullable;
19  
20  /**
21   * Receiver to inform callers which values have been invalidated. Values may be invalidated and then
22   * re-validated if they have been found not to be changed.
23   */
24  public interface EvaluationProgressReceiver {
25    /**
26     * New state of the value entry after evaluation.
27     */
28    enum EvaluationState {
29      /** The value was successfully re-evaluated. */
30      BUILT,
31      /** The value is clean or re-validated. */
32      CLEAN,
33    }
34  
35    /**
36     * New state of the value entry after invalidation.
37     */
38    enum InvalidationState {
39      /** The value is dirty, although it might get re-validated again. */
40      DIRTY,
41      /** The value is dirty and got deleted, cannot get re-validated again. */
42      DELETED,
43    }
44  
45    /**
46     * Notifies that {@code value} has been invalidated.
47     *
48     * <p>{@code state} indicates the new state of the value.
49     *
50     * <p>This method is not called on invalidation of values which do not have a value (usually
51     * because they are in error).
52     *
53     * <p>May be called concurrently from multiple threads, possibly with the same {@code value}
54     * object.
55     */
56    @ThreadSafety.ThreadSafe
57    void invalidated(SkyValue value, InvalidationState state);
58  
59    /**
60     * Notifies that {@code skyKey} is about to get queued for evaluation.
61     *
62     * <p>Note that we don't guarantee that it actually got enqueued or will, only that if
63     * everything "goes well" (e.g. no interrupts happen) it will.
64     *
65     * <p>This guarantee is intentionally vague to encourage writing robust implementations.
66     */
67    @ThreadSafety.ThreadSafe
68    void enqueueing(SkyKey skyKey);
69  
70    /**
71     * Notifies that the node for {@code skyKey} has been evaluated.
72     *
73     * <p>{@code state} indicates the new state of the node.
74     *
75     * <p>If the value builder threw an error when building this node, then {@code value} is null.
76     */
77    @ThreadSafety.ThreadSafe
78    void evaluated(SkyKey skyKey, @Nullable SkyValue value, EvaluationState state);
79  }
