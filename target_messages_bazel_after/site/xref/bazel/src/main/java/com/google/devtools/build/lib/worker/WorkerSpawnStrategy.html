37  /**
38   * A spawn action context that launches Spawns the first time they are used in a persistent mode and
39   * then shards work over all the processes.
40   */
41  @ExecutionStrategy(name = { "worker" }, contextType = SpawnActionContext.class)
42  final class WorkerSpawnStrategy implements SpawnActionContext {
43    private final WorkerPool workers;
44    private final IncrementalHeuristic incrementalHeuristic;
45  
46    public WorkerSpawnStrategy(
47        OptionsClassProvider optionsProvider, WorkerPool workers, EventBus eventBus) {
48      Preconditions.checkNotNull(optionsProvider);
49      WorkerOptions options = optionsProvider.getOptions(WorkerOptions.class);
50      workers.setMaxTotalPerKey(options.workerMaxInstances);
51      workers.setMaxIdlePerKey(options.workerMaxInstances);
52      workers.setMinIdlePerKey(options.workerMaxInstances);
53      this.workers = workers;
54      this.incrementalHeuristic = new IncrementalHeuristic(options.workerMaxChangedFiles);
55      eventBus.register(incrementalHeuristic);
56    }
57  
58    @Override
59    public void exec(Spawn spawn, ActionExecutionContext actionExecutionContext)
60        throws ExecException, InterruptedException {
61      if (!incrementalHeuristic.shouldUseWorkers()) {
62        SpawnActionContext context = actionExecutionContext.getExecutor().getSpawnActionContext("");
63        if (context != this) {
64          context.exec(spawn, actionExecutionContext);
65          return;
66        }
67      }
68  
69      String paramFile = Iterables.getLast(spawn.getArguments());
70  
71      // We assume that the spawn to be executed always gets a single argument, which is a flagfile
72      // prefixed with @ and that it will start in persistent mode when we don't pass it one.
73      // Thus, we can extract the last element from its args (which will be the flagfile) to start the
74      // persistent mode and then pass it the flagfile via a WorkRequest to make it actually do the
75      // work.
76      if (!Iterables.getLast(spawn.getArguments()).startsWith("@")) {
77        Preconditions.checkState(
78            false, "Must have parameter file as last arg, got args: " + spawn.getArguments());
79      }
80  
81      ImmutableList<String> args = ImmutableList.<String>builder()
82          .addAll(spawn.getArguments().subList(0, spawn.getArguments().size() - 1))
83          .add("--persistent_worker")
84          .build();
85      ImmutableMap<String, String> env = spawn.getEnvironment();
86      Path workDir = actionExecutionContext.getExecutor().getExecRoot();
87      WorkerKey key = new WorkerKey(args, env, workDir);
88  
89      try {
90        Worker worker = workers.borrowObject(key);
91        try {
92          WorkRequest.newBuilder()
93              .addArguments(paramFile)
94              .build()
95              .writeDelimitedTo(worker.getOutputStream());
96          worker.getOutputStream().flush();
97  
98          WorkResponse response = WorkResponse.parseDelimitedFrom(worker.getInputStream());
99  
100         if (response == null) {
101           throw new UserExecException("Worker did not return a correct WorkResponse");
102         }
103 
104         String trimmedOutput = response.getOutput().trim();
105         if (!trimmedOutput.isEmpty()) {
106           System.err.println(trimmedOutput);
107         }
108 
109         if (response.getExitCode() != 0) {
110           throw new UserExecException(String.format("Failed with exit code: %d.",
111               response.getExitCode()));
112         }
113       } finally {
114         if (worker != null) {
115           workers.returnObject(key, worker);
116         }
117       }
118     } catch (Exception e) {
119       throw new UserExecException(e.getMessage(), e);
120     }
121   }
122 
123   @Override
124   public String strategyLocality(String mnemonic, boolean remotable) {
125     return "worker";
126   }
127 
128   @Override
129   public boolean isRemotable(String mnemonic, boolean remotable) {
130     return false;
131   }

133   /**
134    * For installation with remote execution, non-incremental builds may be slowed down by the
135    * persistent worker system. To avoid this we only use workers for builds where few files
136    * changed.
137    */
138   @ThreadSafety.ThreadSafe
139   private static class IncrementalHeuristic {
140     private final AtomicBoolean fewFilesChanged = new AtomicBoolean(false);
141     private int limit = 0;
142 
143     public boolean shouldUseWorkers() {
144       return limit == 0 || fewFilesChanged.get();
145     }
146 
147     IncrementalHeuristic(int limit) {
148       this.limit = limit;
149     }
150 
151     @Subscribe
152     public void changedFiles(ChangedFilesMessage msg) {
153       fewFilesChanged.set(msg.getChangedFiles().size() <= limit);
154     }
155   }
