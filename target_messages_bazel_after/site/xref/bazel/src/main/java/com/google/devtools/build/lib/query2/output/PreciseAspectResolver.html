39  /**
40   * An aspect resolver that returns only those aspects that are possibly active given the rule
41   * classes of direct dependencies.
42   *
43   * <p>Needs to load the packages that contain dependencies through attributes with aspects.
44   */
45  public class PreciseAspectResolver implements AspectResolver {
46    private final PackageProvider packageProvider;
47    private final EventHandler eventHandler;
48  
49    public PreciseAspectResolver(PackageProvider packageProvider, EventHandler eventHandler) {
50      this.packageProvider = packageProvider;
51      this.eventHandler = eventHandler;
52    }
53  
54    @Override
55    public ImmutableMultimap<Attribute, Label> computeAspectDependencies(Target target)
56        throws InterruptedException {
57      Multimap<Attribute, Label> result = LinkedListMultimap.create();
58      if (target instanceof Rule) {
59        Multimap<Attribute, Label> transitions =
60            ((Rule) target).getTransitions(Rule.NO_NODEP_ATTRIBUTES);
61        for (Entry<Attribute, Label> entry : transitions.entries()) {
62          Target toTarget;
63          try {
64            toTarget = packageProvider.getTarget(eventHandler, entry.getValue());
65            result.putAll(AspectDefinition.visitAspectsIfRequired(target, entry.getKey(), toTarget));
66          } catch (NoSuchThingException e) {
67            // Do nothing. One of target direct deps has an error. The dependency on the BUILD file
68            // (or one of the files included in it) will be reported in the query result of :BUILD.
69          }
70        }
71      }
72      return ImmutableMultimap.copyOf(result);
73    }
74  
75    @Override
76    public Set<Label> computeBuildFileDependencies(Package pkg, BuildFileDependencyMode mode)
77        throws InterruptedException {
78      Set<Label> result = new LinkedHashSet<>();
79      result.addAll(mode.getDependencies(pkg));
80  
81      Set<PackageIdentifier> dependentPackages = new LinkedHashSet<>();
82      // First compute with packages can possibly affect the aspect attributes of this package:
83      // Iterate over all rules...
84      for (Target target : pkg.getTargets()) {
85  
86        if (!(target instanceof Rule)) {
87          continue;
88        }
89  
90        // ...figure out which direct dependencies can possibly have aspects attached to them...
91        Multimap<Attribute, Label> depsWithPossibleAspects = ((Rule) target).getTransitions(
92            new BinaryPredicate<Rule, Attribute>() {
93              @Override
94              public boolean apply(@Nullable Rule rule, @Nullable Attribute attribute) {
95                for (Class<? extends AspectFactory<?, ?, ?>> aspectFactory : attribute.getAspects()) {
96                  if (!AspectFactory.Util.create(aspectFactory).getDefinition()
97                      .getAttributes().isEmpty()) {
98                    return true;
99                  }
100               }
101 
102               return false;
103             }
104           });
105 
106       // ...and add the package of the aspect.
107       for (Label depLabel : depsWithPossibleAspects.values()) {
108         dependentPackages.add(depLabel.getPackageIdentifier());
109       }
110     }
111 
112     // Then add all the subinclude labels of the packages thus found to the result.
113     for (PackageIdentifier packageIdentifier : dependentPackages) {
114       try {
115         result.add(Label.create(packageIdentifier, "BUILD"));
116         Package dependentPackage = packageProvider.getPackage(eventHandler, packageIdentifier);
117         result.addAll(mode.getDependencies(dependentPackage));
118       } catch (NoSuchPackageException e) {
119         // If the package is not found, just add its BUILD file, which is already done above.
120         // Hopefully this error is not raised when there is a syntax error in a subincluded file
121         // or something.
122       } catch (Label.SyntaxException e) {
123         throw new IllegalStateException(e);
124       }
125     }
126 
127     return result;
128   }
129 }
