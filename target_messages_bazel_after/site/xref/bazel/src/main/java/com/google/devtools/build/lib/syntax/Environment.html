145   // Sets up the global environment
146   private void setupGlobal() {
147     // In Python 2.x, True and False are global values and can be redefined by the user.
148     // In Python 3.x, they are keywords. We implement them as values, for the sake of
149     // simplicity. We define them as Boolean objects.
150     update("False", FALSE);
151     update("True", TRUE);
152     update("None", NONE);
153   }

159   protected boolean hasVariable(String varname) {
160     return env.containsKey(varname);
161   }

163   /**
164    * @return the value from the environment whose name is "varname".
165    * @throws NoSuchVariableException if the variable is not defined in the Environment.
166    *
167    */
168   public Object lookup(String varname) throws NoSuchVariableException {
169     if (disabledVariables.contains(varname)) {
170       throw new NoSuchVariableException(varname);
171     }
172     Object value = env.get(varname);
173     if (value == null) {
174       if (parent != null) {
175         return parent.lookup(varname);
176       }
177       throw new NoSuchVariableException(varname);
178     }
179     return value;
180   }
181 
182   /**
183    * Like <code>lookup(String)</code>, but instead of throwing an exception in
184    * the case where "varname" is not defined, "defaultValue" is returned instead.
185    *
186    */
187   public Object lookup(String varname, Object defaultValue) {
188     Object value = env.get(varname);
189     if (value == null) {
190       if (parent != null) {
191         return parent.lookup(varname, defaultValue);
192       }
193       return defaultValue;
194     }
195     return value;
196   }

198   /**
199    * Updates the value of variable "varname" in the environment, corresponding
200    * to an {@link AssignmentStatement}.
201    */
202   public void update(String varname, Object value) {
203     Preconditions.checkNotNull(value, "update(value == null)");
204     env.put(varname, value);
205   }
206 
207   /**
208    * Same as {@link #update}, but also marks the variable propagating, meaning it will
209    * be present in the execution environment of a UserDefinedFunction called from this
210    * Environment. Using this method is discouraged.
211    */
212   public void updateAndPropagate(String varname, Object value) {
213     update(varname, value);
214     propagatingVariables.add(varname);
215   }

232   /**
233    * Returns the (immutable) set of names of all variables defined in this
234    * environment. Exposed for testing; not very efficient!
235    */
236   @VisibleForTesting
237   public Set<String> getVariableNames() {
238     if (parent == null) {
239       return env.keySet();
240     } else {
241       Set<String> vars = new HashSet<>();
242       vars.addAll(env.keySet());
243       vars.addAll(parent.getVariableNames());
244       return vars;
245     }

258   @Override
259   public String toString() {
260     StringBuilder out = new StringBuilder();
261     out.append("Environment{");
262     List<String> keys = new ArrayList<>(env.keySet());
263     Collections.sort(keys);
264     for (String key : keys) {
265       out.append(key).append(" -> ").append(env.get(key)).append(", ");
266     }
267     out.append("}");
268     if (parent != null) {
269       out.append("=>");
270       out.append(parent);
271     }
272     return out.toString();
273   }

300   public void importSymbol(PathFragment extension, Identifier symbol, String nameInLoadedFile)
301       throws NoSuchVariableException, LoadFailedException {
302     if (!importedExtensions.containsKey(extension)) {
303       throw new LoadFailedException(extension.toString());
304     }
305 
306     Object value = importedExtensions.get(extension).lookup(nameInLoadedFile);
307     if (!isSkylarkEnabled()) {
308       value = SkylarkType.convertFromSkylark(value);
309     }
310 
311     update(symbol.getName(), value);
312   }

314   /**
315    * Registers a function with namespace to this global environment.
316    */
317   public void registerFunction(Class<?> nameSpace, String name, BaseFunction function) {
318     nameSpace = getCanonicalRepresentation(nameSpace);
319     Preconditions.checkArgument(parent == null);
320     if (!functions.containsKey(nameSpace)) {
321       functions.put(nameSpace, new HashMap<String, BaseFunction>());
322     }
323     functions.get(nameSpace).put(name, function);
324   }
325 
326   private Map<String, BaseFunction> getNamespaceFunctions(Class<?> nameSpace) {
327     nameSpace = getCanonicalRepresentation(nameSpace);
328     if (disabledNameSpaces.contains(nameSpace)
329         || (parent != null && parent.disabledNameSpaces.contains(nameSpace))) {
330       return null;
331     }
332     Environment topLevel = this;
333     while (topLevel.parent != null) {
334       topLevel = topLevel.parent;
335     }
336     return topLevel.functions.get(nameSpace);
337   }
338 
339   /**
340    * Returns the canonical representation of the given class, i.e. the super class for which any
341    * functions were registered.
342    *
343    * <p>Currently, this is only necessary for mapping the different subclasses of {@link
344    * java.util.Map} to the interface.
345    */
346   private Class<?> getCanonicalRepresentation(Class<?> clazz) {
347     return Map.class.isAssignableFrom(clazz) ? Map.class : clazz;
348   }
349 
350   /**
351    * Returns the function of the namespace of the given name or null of it does not exists.
352    */
353   public BaseFunction getFunction(Class<?> nameSpace, String name) {
354     Map<String, BaseFunction> nameSpaceFunctions = getNamespaceFunctions(nameSpace);
355     return nameSpaceFunctions != null ? nameSpaceFunctions.get(name) : null;
356   }
357 
358   /**
359    * Returns the function names registered with the namespace.
360    */
361   public Set<String> getFunctionNames(Class<?> nameSpace) {
362     Map<String, BaseFunction> nameSpaceFunctions = getNamespaceFunctions(nameSpace);
363     return nameSpaceFunctions != null ? nameSpaceFunctions.keySet() : ImmutableSet.<String>of();
364   }
