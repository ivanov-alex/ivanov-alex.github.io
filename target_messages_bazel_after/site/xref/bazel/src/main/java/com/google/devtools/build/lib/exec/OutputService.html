15  package com.google.devtools.build.lib.exec;
16  
17  import com.google.devtools.build.lib.actions.BuildFailedException;
18  import com.google.devtools.build.lib.actions.ExecException;
19  import com.google.devtools.build.lib.packages.Target;
20  import com.google.devtools.build.lib.util.AbruptExitException;
21  import com.google.devtools.build.lib.vfs.BatchStat;
22  import com.google.devtools.build.lib.vfs.Path;
23  import com.google.devtools.build.lib.vfs.PathFragment;
24  
25  import java.io.IOException;
26  import java.util.UUID;
27  
28  /**
29   * An OutputService retains control over the Blaze output tree, and provides a higher level of
30   * abstraction compared to the VFS layer.
31   *
32   * <p>Higher-level facilities include batch statting, cleaning the output tree, creating symlink
33   * trees, and out-of-band insertion of metadata into the tree.
34   */
35  public interface OutputService {
36  
37    /**
38     * @return the name of filesystem, akin to what you might see in /proc/mounts
39     */
40    String getFilesSystemName();
41  
42    /**
43     * @return true if the output service uses FUSE
44     */
45    boolean usesFuse();
46  
47    /**
48     * @return a human-readable, one word name for the service
49     */
50    String getName();
51  
52    /**
53     * Start the build.
54     *
55     * @param buildId the UUID build identifier
56     * @throws BuildFailedException if build preparation failed
57     * @throws InterruptedException
58     */
59    void startBuild(UUID buildId)
60        throws BuildFailedException, AbruptExitException, InterruptedException;
61  
62    /**
63     * Finish the build.
64     *
65     * @param buildSuccessful iff build was successful
66     * @throws BuildFailedException on failure
67     */
68    void finalizeBuild(boolean buildSuccessful) throws BuildFailedException, AbruptExitException;
69  
70    /**
71     * Stages the given tool from the package path, possibly copying it to local disk.
72     *
73     * @param tool target representing the tool to stage
74     * @return a Path pointing to the staged target
75     */
76    Path stageTool(Target tool) throws IOException;
77  
78    /**
79     * @return the name of the workspace this output service controls.
80     */
81    String getWorkspace();
82  
83    /**
84     * @return the BatchStat instance or null.
85     */
86    BatchStat getBatchStatter();
87  
88    /**
89     * @return true iff createSymlinkTree() is available.
90     */
91    boolean canCreateSymlinkTree();
92  
93    /**
94     * Creates the symlink tree
95     *
96     * @param inputPath the input manifest
97     * @param outputPath the output manifest
98     * @param filesetTree is true iff we're constructing a Fileset
99     * @param symlinkTreeRoot the symlink tree root, relative to the execRoot
100    * @throws ExecException on failure
101    * @throws InterruptedException
102    */
103   void createSymlinkTree(Path inputPath, Path outputPath, boolean filesetTree,
104       PathFragment symlinkTreeRoot) throws ExecException, InterruptedException;
105 
106   /**
107    * Cleans the entire output tree.
108    *
109    * @throws ExecException on failure
110    * @throws InterruptedException
111    */
112   void clean() throws ExecException, InterruptedException;
113 
114   /**
115    * @param file the File
116    * @return true iff the file actually lives on a remote server
117    */
118   boolean isRemoteFile(Path file);
119 
120   /**
121    * @param path a fully-resolved path
122    * @return true iff path is under this output service's control
123    */
124   boolean resolvedPathUnderTree(Path path);
125 }
