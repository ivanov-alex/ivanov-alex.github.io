30  /**
31   * Used to keep track of resources consumed by the Blaze action execution threads and throttle them
32   * when necessary.
33   *
34   * <p>Threads which are known to consume a significant amount of resources should call
35   * {@link #acquireResources} method. This method will check whether requested resources are
36   * available and will either mark them as used and allow the thread to proceed or will block the
37   * thread until requested resources will become available. When the thread completes its task, it
38   * must release allocated resources by calling {@link #releaseResources} method.
39   *
40   * <p>Available resources can be calculated using one of three ways:
41   * <ol>
42   * <li>They can be preset using {@link #setAvailableResources(ResourceSet)} method. This is used
43   *     mainly by the unit tests (however it is possible to provide a future option that would
44   *     artificially limit amount of CPU/RAM consumed by the Blaze).
45   * <li>They can be preset based on the /proc/cpuinfo and /proc/meminfo information. Blaze will
46   *     calculate amount of available CPU cores (adjusting for hyperthreading logical cores) and
47   *     amount of the total available memory and will limit itself to the number of effective cores
48   *     and 2/3 of the available memory. For details, please look at the {@link
49   *     LocalHostCapacity#getLocalHostCapacity} method.
50   * </ol>
51   *
52   * <p>The resource manager also allows a slight overallocation of the resources to account for the
53   * fact that requested resources are usually estimated using a pessimistic approximation. It also
54   * guarantees that at least one thread will always be able to acquire any amount of requested
55   * resources (even if it is greater than amount of available resources). Therefore, assuming that
56   * threads correctly release acquired resources, Blaze will never be fully blocked.
57   */
58  @ThreadSafe
59  public class ResourceManager {
60  
61    private EventBus eventBus;
62  
63    private final ThreadLocal<Boolean> threadLocked = new ThreadLocal<Boolean>() {
64      @Override
65      protected Boolean initialValue() {
66        return false;
67      }
68    };
69  
70    /**
71     * Singleton reference defined in a separate class to ensure thread-safe lazy
72     * initialization.
73     */
74    private static class Singleton {
75      static ResourceManager instance = new ResourceManager();
76    }

78    /**
79     * Returns singleton instance of the resource manager.
80     */
81    public static ResourceManager instance() {
82      return Singleton.instance;

85    // Allocated resources are allowed to go "negative", but at least
86    // MIN_AVAILABLE_CPU_RATIO portion of CPU and MIN_AVAILABLE_RAM_RATIO portion
87    // of RAM should be available.
88    // Please note that this value is purely empirical - we assume that generally
89    // requested resources are somewhat pessimistic and thread would end up
90    // using less than requested amount.
91    private static final double MIN_NECESSARY_CPU_RATIO = 0.6;
92    private static final double MIN_NECESSARY_RAM_RATIO = 1.0;
93    private static final double MIN_NECESSARY_IO_RATIO = 1.0;
94  
95    // List of blocked threads. Associated CountDownLatch object will always
96    // be initialized to 1 during creation in the acquire() method.
97    private final List<Pair<ResourceSet, CountDownLatch>> requestList;
98  
99    // The total amount of resources on the local host. Must be set by
100   // an explicit call to setAvailableResources(), often using
101   // LocalHostCapacity.getLocalHostCapacity() as an argument.
102   private ResourceSet staticResources = null;
103 
104   private ResourceSet availableResources = null;
105 
106   // Used amount of CPU capacity (where 1.0 corresponds to the one fully
107   // occupied CPU core. Corresponds to the CPU resource definition in the
108   // ResourceSet class.
109   private double usedCpu;
110 
111   // Used amount of RAM capacity in MB. Corresponds to the RAM resource
112   // definition in the ResourceSet class.
113   private double usedRam;
114 
115   // Used amount of I/O resources. Corresponds to the I/O resource
116   // definition in the ResourceSet class.
117   private double usedIo;
118 
119   // Used local test count. Corresponds to the local test count definition in the ResourceSet class.
120   private int usedLocalTestCount;
121 
122   // Specifies how much of the RAM in staticResources we should allow to be used.
123   public static final int DEFAULT_RAM_UTILIZATION_PERCENTAGE = 67;
124   private int ramUtilizationPercentage = DEFAULT_RAM_UTILIZATION_PERCENTAGE;
125 
126   private ResourceManager() {
127     requestList = new LinkedList<>();
128   }
129 
130   @VisibleForTesting public static ResourceManager instanceForTestingOnly() {
131     return new ResourceManager();
132   }

134   /**
135    * Resets resource manager state and releases all thread locks.
136    * Note - it does not reset available resources. Use
137    * separate call to setAvailableResoures().
138    */
139   public synchronized void resetResourceUsage() {
140     usedCpu = 0;
141     usedRam = 0;
142     usedIo = 0;
143     usedLocalTestCount = 0;
144     for (Pair<ResourceSet, CountDownLatch> request : requestList) {
145       // CountDownLatch can be set only to 0 or 1.
146       request.second.countDown();
147     }
148     requestList.clear();
149   }

151   /**
152    * Sets available resources using given resource set. Must be called
153    * at least once before using resource manager.
154    */
155   public synchronized void setAvailableResources(ResourceSet resources) {
156     Preconditions.checkNotNull(resources);
157     staticResources = resources;
158     availableResources = ResourceSet.create(
159         staticResources.getMemoryMb() * this.ramUtilizationPercentage / 100.0,
160         staticResources.getCpuUsage(),
161         staticResources.getIoUsage(),
162         staticResources.getLocalTestCount());
163     processWaitingThreads();
164   }
165 
166   /**
167    * Specify how much of the available RAM we should allow to be used.
168    * This has no effect if autosensing is enabled.
169    */
170   public synchronized void setRamUtilizationPercentage(int percentage) {
171     ramUtilizationPercentage = percentage;
172   }

174   /**
175    * Acquires requested resource set. Will block if resource is not available.
176    * NB! This method must be thread-safe!
177    */
178   public void acquireResources(ActionMetadata owner, ResourceSet resources)
179       throws InterruptedException {
180     Preconditions.checkNotNull(resources);
181     long startTime = Profiler.nanoTimeMaybe();
182     CountDownLatch latch = null;
183     try {
184       waiting(owner);
185       latch = acquire(resources);
186       if (latch != null) {
187         latch.await();
188       }
189     } finally {
190       threadLocked.set(resources.getCpuUsage() != 0 || resources.getMemoryMb() != 0
191           || resources.getIoUsage() != 0 || resources.getLocalTestCount() != 0);
192       acquired(owner);
193 
194       // Profile acquisition only if it waited for resource to become available.
195       if (latch != null) {
196         Profiler.instance().logSimpleTask(startTime, ProfilerTask.ACTION_LOCK, owner);
197       }
198     }
199   }
200 
201   /**
202    * Acquires the given resources if available immediately. Does not block.
203    * @return true iff the given resources were locked (all or nothing).
204    */
205   public boolean tryAcquire(ActionMetadata owner, ResourceSet resources) {
206     boolean acquired = false;
207     synchronized (this) {
208       if (areResourcesAvailable(resources)) {
209         incrementResources(resources);
210         acquired = true;
211       }
212     }
213 
214     if (acquired) {
215       threadLocked.set(resources.getCpuUsage() != 0 || resources.getMemoryMb() != 0
216           || resources.getIoUsage() != 0 || resources.getLocalTestCount() != 0);
217       acquired(owner);
218     }
219 
220     return acquired;
221   }
222 
223   private void incrementResources(ResourceSet resources) {
224     usedCpu += resources.getCpuUsage();
225     usedRam += resources.getMemoryMb();
226     usedIo += resources.getIoUsage();
227     usedLocalTestCount += resources.getLocalTestCount();
228   }
229 
230   /**
231    * Return true if any resources have been claimed through this manager.
232    */
233   public synchronized boolean inUse() {
234     return usedCpu != 0.0 || usedRam != 0.0 || usedIo != 0.0 || usedLocalTestCount != 0
235         || !requestList.isEmpty();
236   }

240    * Return true iff this thread has a lock on non-zero resources.

243     return threadLocked.get();
244   }

256   private void waiting(ActionMetadata owner) {
257     if (eventBus != null) {
258       // Null only in tests.
259       eventBus.post(ActionStatusMessage.schedulingStrategy(owner));
260     }
261   }
262 
263   private void acquired(ActionMetadata owner) {
264     if (eventBus != null) {
265       // Null only in tests.
266       eventBus.post(ActionStatusMessage.runningStrategy(owner));
267     }
268   }
269 
270   /**
271    * Releases previously requested resource =.
272    *
273    * <p>NB! This method must be thread-safe!
274    */
275   public void releaseResources(ActionMetadata owner, ResourceSet resources) {
276     boolean isConflict = false;
277     long startTime = Profiler.nanoTimeMaybe();
278     try {
279       isConflict = release(resources);
280     } finally {
281       threadLocked.set(false);
282 
283       // Profile resource release only if it resolved at least one allocation request.
284       if (isConflict) {
285         Profiler.instance().logSimpleTask(startTime, ProfilerTask.ACTION_RELEASE, owner);
286       }
287     }
288   }
289 
290   private synchronized CountDownLatch acquire(ResourceSet resources) {
291     if (areResourcesAvailable(resources)) {
292       incrementResources(resources);
293       return null;
294     }
295     Pair<ResourceSet, CountDownLatch> request =
296       new Pair<>(resources, new CountDownLatch(1));
297     requestList.add(request);
298     return request.second;
299   }
300 
301   private synchronized boolean release(ResourceSet resources) {
302     usedCpu -= resources.getCpuUsage();
303     usedRam -= resources.getMemoryMb();
304     usedIo -= resources.getIoUsage();
305     usedLocalTestCount -= resources.getLocalTestCount();
306 
307     // TODO(bazel-team): (2010) rounding error can accumulate and value below can end up being
308     // e.g. 1E-15. So if it is small enough, we set it to 0. But maybe there is a better solution.
309     double epsilon = 0.0001;
310     if (usedCpu < epsilon) {
311       usedCpu = 0;
312     }
313     if (usedRam < epsilon) {
314       usedRam = 0;
315     }
316     if (usedIo < epsilon) {
317       usedIo = 0;
318     }
319     if (!requestList.isEmpty()) {
320       processWaitingThreads();
321       return true;
322     }
323     return false;
324   }
325 
326 
327   /**
328    * Tries to unblock one or more waiting threads if there are sufficient resources available.
329    */
330   private synchronized void processWaitingThreads() {
331     Iterator<Pair<ResourceSet, CountDownLatch>> iterator = requestList.iterator();
332     while (iterator.hasNext()) {
333       Pair<ResourceSet, CountDownLatch> request = iterator.next();
334       if (areResourcesAvailable(request.first)) {
335         incrementResources(request.first);
336         request.second.countDown();
337         iterator.remove();
338       }
339     }
340   }

342   // Method will return true if all requested resources are considered to be available.
343   private boolean areResourcesAvailable(ResourceSet resources) {
344     Preconditions.checkNotNull(availableResources);
345     // Comparison below is robust, since any calculation errors will be fixed
346     // by the release() method.
347     if (usedCpu == 0.0 && usedRam == 0.0 && usedIo == 0.0 && usedLocalTestCount == 0) {
348       return true;
349     }
350     // Use only MIN_NECESSARY_???_RATIO of the resource value to check for
351     // allocation. This is necessary to account for the fact that most of the
352     // requested resource sets use pessimistic estimations. Note that this
353     // ratio is used only during comparison - for tracking we will actually
354     // mark whole requested amount as used.
355     double cpu = resources.getCpuUsage() * MIN_NECESSARY_CPU_RATIO;
356     double ram = resources.getMemoryMb() * MIN_NECESSARY_RAM_RATIO;
357     double io = resources.getIoUsage() * MIN_NECESSARY_IO_RATIO;
358     int localTestCount = resources.getLocalTestCount();
359 
360     double availableCpu = availableResources.getCpuUsage();
361     double availableRam = availableResources.getMemoryMb();
362     double availableIo = availableResources.getIoUsage();
363     int availableLocalTestCount = availableResources.getLocalTestCount();
364 
365     // Resources are considered available if any one of the conditions below is true:
366     // 1) If resource is not requested at all, it is available.
367     // 2) If resource is not used at the moment, it is considered to be
368     // available regardless of how much is requested. This is necessary to
369     // ensure that at any given time, at least one thread is able to acquire
370     // resources even if it requests more than available.
371     // 3) If used resource amount is less than total available resource amount.
372     boolean cpuIsAvailable = cpu == 0.0 || usedCpu == 0.0 || usedCpu + cpu <= availableCpu;
373     boolean ramIsAvailable = ram == 0.0 || usedRam == 0.0 || usedRam + ram <= availableRam;
374     boolean ioIsAvailable = io == 0.0 || usedIo == 0.0 || usedIo + io <= availableIo;
375     boolean localTestCountIsAvailable = localTestCount == 0 || usedLocalTestCount == 0
376         || usedLocalTestCount + localTestCount <= availableLocalTestCount;
377     return cpuIsAvailable && ramIsAvailable && ioIsAvailable && localTestCountIsAvailable;
378   }

386   @VisibleForTesting
387   synchronized boolean isAvailable(double ram, double cpu, double io, int localTestCount) {
