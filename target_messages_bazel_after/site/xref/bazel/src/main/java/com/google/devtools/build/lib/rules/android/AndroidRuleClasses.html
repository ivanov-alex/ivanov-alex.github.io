173   public static final SplitTransition<BuildOptions> ANDROID_SPLIT_TRANSITION =
174       new SplitTransition<BuildOptions>() {
175     @Override
176     public boolean defaultsToSelf() {
177       return true;
178     }
179 
180     private void setCrosstoolToAndroid(BuildOptions output, BuildOptions input) {
181       AndroidConfiguration.Options inputAndroidOptions =
182           input.get(AndroidConfiguration.Options.class);
183       AndroidConfiguration.Options outputAndroidOptions =
184           output.get(AndroidConfiguration.Options.class);
185 
186       CppOptions cppOptions = output.get(CppOptions.class);
187       if (inputAndroidOptions.realAndroidCrosstoolTop() != null
188           && !cppOptions.crosstoolTop.equals(inputAndroidOptions.realAndroidCrosstoolTop())) {
189         if (cppOptions.hostCrosstoolTop == null) {
190           cppOptions.hostCrosstoolTop = cppOptions.crosstoolTop;
191         }
192         cppOptions.crosstoolTop = inputAndroidOptions.realAndroidCrosstoolTop();
193       }
194 
195       outputAndroidOptions.configurationDistinguisher = ConfigurationDistinguisher.ANDROID;
196     }
197 
198     @Override
199     public List<BuildOptions> split(BuildOptions buildOptions) {
200       AndroidConfiguration.Options androidOptions =
201           buildOptions.get(AndroidConfiguration.Options.class);
202       CppOptions cppOptions = buildOptions.get(CppOptions.class);
203       Label androidCrosstoolTop = androidOptions.realAndroidCrosstoolTop();
204       if (androidOptions.realFatApkCpus().isEmpty()
205           && (androidCrosstoolTop == null || androidCrosstoolTop.equals(cppOptions.crosstoolTop))) {
206         return ImmutableList.of();
207       }
208 
209       if (androidOptions.realFatApkCpus().isEmpty()) {
210         BuildOptions splitOptions = buildOptions.clone();
211         setCrosstoolToAndroid(splitOptions, buildOptions);
212         return ImmutableList.of(splitOptions);
213       }
214 
215       List<BuildOptions> result = new ArrayList<>();
216       for (String cpu : ImmutableSortedSet.copyOf(androidOptions.realFatApkCpus())) {
217         BuildOptions splitOptions = buildOptions.clone();
218         // Disable fat APKs for the child configurations.
219         splitOptions.get(AndroidConfiguration.Options.class).fatApkCpus = ImmutableList.of();
220 
221         // Set the cpu & android_cpu.
222         // TODO(bazel-team): --android_cpu doesn't follow --cpu right now; it should.
223         splitOptions.get(AndroidConfiguration.Options.class).cpu = cpu;
224         splitOptions.get(BuildConfiguration.Options.class).cpu = cpu;
225         setCrosstoolToAndroid(splitOptions, buildOptions);
226         result.add(splitOptions);
227       }
228       return result;
229     }

241   public static final SafeImplicitOutputsFunction ANDROID_BINARY_IMPLICIT_OUTPUTS =
242       new SafeImplicitOutputsFunction() {
243 
244         @Override
245         public Iterable<String> getImplicitOutputs(AttributeMap rule) {
246           boolean mapping = rule.get("proguard_generate_mapping", Type.BOOLEAN);
247           List<SafeImplicitOutputsFunction> functions = Lists.newArrayList();
248           functions.add(AndroidRuleClasses.ANDROID_BINARY_APK);
249           functions.add(AndroidRuleClasses.ANDROID_BINARY_UNSIGNED_APK);
250           functions.add(AndroidRuleClasses.ANDROID_BINARY_DEPLOY_JAR);
251           functions.add(AndroidRuleClasses.ANDROID_BINARY_GEN_JAR);
252 
253           // The below is a hack to support configurable attributes (proguard_specs seems like
254           // too valuable an attribute to make nonconfigurable, and we don't currently
255           // have the ability to know the configuration when determining implicit outputs).
256           // An IllegalArgumentException gets triggered if the attribute instance is configurable.
257           // We assume, heuristically, that means every configurable value is a non-empty list.
258           //
259           // TODO(bazel-team): find a stronger approach for this. One simple approach is to somehow
260           // receive 'rule' as an AggregatingAttributeMapper instead of a RawAttributeMapper,
261           // check that all possible values are non-empty, and simply don't support configurable
262           // instances that mix empty and non-empty lists. A more ambitious approach would be
263           // to somehow determine implicit outputs after the configuration is known. A third
264           // approach is to refactor the Android rule logic to avoid these dependencies in the
265           // first place.
266           boolean hasProguardSpecs;
267           try {
268             hasProguardSpecs = !rule.get("proguard_specs", LABEL_LIST).isEmpty();
269           } catch (IllegalArgumentException e) {
270             // We assume at this point the attribute instance is configurable.
271             hasProguardSpecs = true;
272           }
273 
274           if (hasProguardSpecs) {
275             functions.add(AndroidRuleClasses.ANDROID_BINARY_PROGUARD_JAR);
276             if (mapping) {
277               functions.add(AndroidRuleClasses.ANDROID_BINARY_PROGUARD_MAP);
278             }
279           }
280           return fromFunctions(functions).getImplicitOutputs(rule);
281         }
282       };

425   public static final class AndroidResourceSupportRule implements RuleDefinition {
426     @Override
427     public RuleClass build(RuleClass.Builder builder, RuleDefinitionEnvironment env) {
428       return builder.setUndocumented()
429           /* <!-- #BLAZE_RULE($android_resource_support).ATTRIBUTE(manifest) -->
430           The name of the Android manifest file, normally <code>AndroidManifest.xml</code>.
431           Must be defined if resource_files or assets are defined.
432           ${SYNOPSIS}
433           <!-- #END_BLAZE_RULE.ATTRIBUTE --> */
434           .add(attr("manifest", LABEL).legacyAllowAnyFileType())
435           /* <!-- #BLAZE_RULE($android_resource_support).ATTRIBUTE(exports_manifest) -->
436           Whether to export manifest entries to <code>android_binary</code> targets
437           that depend on this target. <code>uses-permissions</code> attributes are never exported.
438           ${SYNOPSIS}
439           <!-- #END_BLAZE_RULE.ATTRIBUTE --> */
440           .add(attr("exports_manifest", BOOLEAN).value(false))
441           /* <!-- #BLAZE_RULE($android_resource_support).ATTRIBUTE(resource_files) -->
442           The list of resources to be packaged.
443           ${SYNOPSIS}
444           This is typically a <code>glob</code> of all files under the
445           <code>res</code> directory.
446           <br/>
447           Generated files (from genrules) can be referenced by
448           <a href="build-ref.html#labels">Label</a> here as well. The only restriction is that the
449           generated outputs must be under the same "<code>res</code>" directory as any other
450           resource files that are included.
451           <!-- #END_BLAZE_RULE.ATTRIBUTE --> */
452           .add(attr("resource_files", LABEL_LIST).legacyAllowAnyFileType())
453           /* <!-- #BLAZE_RULE($android_resource_support).ATTRIBUTE(assets_dir) -->
454           The string giving the path to the files in <code>assets</code>.
455           ${SYNOPSIS}
456           The pair <code>assets</code> and <code>assets_dir</code> describe packaged
457           assets and either both attributes should be provided or none of them.
458           <!-- #END_BLAZE_RULE.ATTRIBUTE --> */
459           .add(attr("assets_dir", STRING))
460           /* <!-- #BLAZE_RULE($android_resource_support).ATTRIBUTE(assets) -->
461           The list of assets to be packaged.
462           ${SYNOPSIS}
463           This is typically a <code>glob</code> of all files under the
464           <code>assets</code> directory. You can also reference other rules (any rule that produces
465           files) or exported files in the other packages, as long as all those files are under the
466           <code>assets_dir</code> directory in the corresponding package.
467           <!-- #END_BLAZE_RULE.ATTRIBUTE --> */
468           .add(attr("assets", LABEL_LIST).legacyAllowAnyFileType())
469           /* <!-- #BLAZE_RULE($android_resource_support).ATTRIBUTE(inline_constants) -->
470           Let the compiler inline the constants defined in the generated java sources.
471           ${SYNOPSIS}
472           This attribute must be set to 0 for all <code>android_library</code> rules
473           used directly by an <code>android_binary</code>,
474           and for any <code>android_binary</code> that has an <code>android_library</code>
475           in its transitive closure.
476           <!-- #END_BLAZE_RULE.ATTRIBUTE --> */
477           .add(attr("inline_constants", BOOLEAN).value(false))
478           /* <!-- #BLAZE_RULE($android_resource_support).ATTRIBUTE(custom_package) -->
479           Java package for which java sources will be generated.
480           ${SYNOPSIS}
481           By default the package is inferred from the directory where the BUILD file
482           containing the rule is. You can specify a different package but this is
483           highly discouraged since it can introduce classpath conflicts with other
484           libraries that will only be detected at runtime.
485           <!-- #END_BLAZE_RULE.ATTRIBUTE --> */
486           .add(attr("custom_package", STRING))
487         .build();
488     }

710   /**
711    * Semantic options for the dexer's multidex behavior.
712    */
713   public static enum MultidexMode {
714     // Build dexes with multidex, assuming native platform support for multidex.
715     NATIVE("native"),
716     // Build dexes with multidex and implement support at the application level.
717     LEGACY("legacy"),
718     // Build dexes with multidex, main dex list needs to be manually specified.
719     MANUAL_MAIN_DEX("legacy"),
720     // Build all dex code into a single classes.dex file.
721     OFF("none");
722 
723     @Nullable private final String jackFlagValue;
724 
725     private MultidexMode(String jackFlagValue) {
726       this.jackFlagValue = jackFlagValue;
727     }
728 
729     /**
730      * Returns the attribute value that specifies this mode.
731      */
732     public String getAttributeValue() {
733       return toString().toLowerCase();
734     }
735 
736     /**
737      * Returns whether or not this multidex mode can be passed to Jack.
738      */
739     public boolean isSupportedByJack() {
740       return jackFlagValue != null;
741     }
742 
743     /**
744      * Returns the value that should be passed to Jack's --multi-dex flag.
745      *
746      * @throws UnsupportedOperationException if the dex mode is not supported by Jack
747      *     ({@link #isSupportedByJack()} returns false)
748      */
749     public String getJackFlagValue() {
750       if (!isSupportedByJack()) {
751         throw new UnsupportedOperationException();
752       }
753       return jackFlagValue;
754     }
755 
756     /**
757      * Returns the name of the output dex classes file. In multidex mode, this is an archive
758      * of (possibly) multiple files.
759      */
760     public String getOutputDexFilename() {
761       return this == OFF ? "classes.dex" : "classes.dex.zip";
762     }
763 
764     /**
765      * Converts an attribute value to a corresponding mode. Returns null on no match.
766      */
767     public static MultidexMode fromValue(String value) {
768       for (MultidexMode mode : values()) {
769         if (mode.getAttributeValue().equals(value)) {
770           return mode;
771         }
772       }
773       return null;
774     }
775 
776     /**
777      * Enumerates valid values for the "multidex" attribute.
778      */
779     public static List<String> getValidValues() {
780       List<String> ans = Lists.newArrayList();
781       for (MultidexMode mode : MultidexMode.values()) {
782         ans.add(mode.getAttributeValue());
783       }
784       return ans;
785     }
