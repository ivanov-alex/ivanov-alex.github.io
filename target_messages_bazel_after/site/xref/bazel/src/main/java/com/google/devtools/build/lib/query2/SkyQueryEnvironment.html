77  /**
78   * {@link AbstractBlazeQueryEnvironment} that introspects the Skyframe graph to find forward and
79   * reverse edges. Results obtained by calling {@link #evaluateQuery} are not guaranteed to be in
80   * any particular order. As well, this class eagerly loads the full transitive closure of targets,
81   * even if the full closure isn't needed.
82   */
83  public class SkyQueryEnvironment extends AbstractBlazeQueryEnvironment<Target> {
84    private WalkableGraph graph;
85  
86    private ImmutableList<TargetPatternKey> universeTargetPatternKeys;
87  
88    private final BlazeTargetAccessor accessor = new BlazeTargetAccessor(this);
89    private final int loadingPhaseThreads;
90    private final WalkableGraphFactory graphFactory;
91    private final List<String> universeScope;
92    private final String parserPrefix;
93    private final PathPackageLocator pkgPath;

95    private static final Logger LOG = Logger.getLogger(SkyQueryEnvironment.class.getName());

97    public SkyQueryEnvironment(boolean keepGoing, boolean strictScope, int loadingPhaseThreads,
98        Predicate<Label> labelFilter,
99        EventHandler eventHandler,
100       Set<Setting> settings,
101       Iterable<QueryFunction> extraFunctions, String parserPrefix,
102       WalkableGraphFactory graphFactory,
103       List<String> universeScope, PathPackageLocator pkgPath) {
104     super(keepGoing, strictScope, labelFilter,
105         eventHandler,
106         settings,
107         extraFunctions);
108     this.loadingPhaseThreads = loadingPhaseThreads;
109     this.graphFactory = graphFactory;
110     this.pkgPath = pkgPath;
111     this.universeScope = Preconditions.checkNotNull(universeScope);
112     this.parserPrefix = parserPrefix;
113     Preconditions.checkState(!universeScope.isEmpty(),
114         "No queries can be performed with an empty universe");
115   }

117   private void init() throws InterruptedException {
118     long startTime = Profiler.nanoTimeMaybe();
119     EvaluationResult<SkyValue> result =
120         graphFactory.prepareAndGet(universeScope, loadingPhaseThreads, eventHandler);
121     graph = result.getWalkableGraph();
122     Collection<SkyValue> values = result.values();
123     long duration = Profiler.nanoTimeMaybe() - startTime;
124     if (duration > 0) {
125       LOG.info("Spent " + (duration / 1000 / 1000) + " ms on evaluation and walkable graph");
126     }
127 
128     // The universe query may fail if there are errors during its evaluation, e.g. because of
129     // cycles in the target graph.
130     boolean singleValueEvaluated = values.size() == 1;
131     boolean foundError = !result.errorMap().isEmpty();
132     boolean evaluationFoundCycle =
133         foundError && !Iterables.isEmpty(result.getError().getCycleInfo());
134     Preconditions.checkState(singleValueEvaluated || evaluationFoundCycle,
135         "Universe query \"%s\" unexpectedly did not result in a single value as expected (%s"
136             + " values in result) and it did not fail because of a cycle.%s",
137         universeScope, values.size(), foundError ? " Error: " + result.getError().toString() : "");
138     if (singleValueEvaluated) {
139       PrepareDepsOfPatternsValue prepareDepsOfPatternsValue =
140           (PrepareDepsOfPatternsValue) Iterables.getOnlyElement(values);
141       universeTargetPatternKeys = prepareDepsOfPatternsValue.getTargetPatternKeys();
142     } else {
143       // The error is because of a cycle, so keep going with the graph we managed to load.
144       universeTargetPatternKeys = ImmutableList.of();
145     }
146   }

148   @Override
149   public QueryEvalResult<Target> evaluateQuery(QueryExpression expr)
150       throws QueryException, InterruptedException {
151     // Some errors are reported as QueryExceptions and others as ERROR events (if --keep_going). The
152     // result is set to have an error iff there were errors emitted during the query, so we reset
153     // errors here.
154     eventHandler.resetErrors();
155     init();
156     return super.evaluateQuery(expr);
157   }

159   private Map<Target, Collection<Target>> makeTargetsMap(Map<SkyKey, Iterable<SkyKey>> input) {
160     ImmutableMap.Builder<Target, Collection<Target>> result = ImmutableMap.builder();
161 
162     for (Map.Entry<SkyKey, Target> entry : makeTargetsWithAssociations(input.keySet()).entrySet()) {
163       result.put(entry.getValue(), makeTargets(input.get(entry.getKey())));
164     }
165     return result.build();
166   }

168   private Map<Target, Collection<Target>> getRawFwdDeps(Iterable<Target> targets) {

174   }

176   private Set<Label> getAllowedDeps(Rule rule) {
177     Set<Label> allowedLabels = new HashSet<>(rule.getLabels(dependencyFilter));
178     allowedLabels.addAll(rule.getVisibility().getDependencyLabels());
179     // We should add deps from aspects, otherwise they are going to be filtered out.
180     allowedLabels.addAll(rule.getAspectLabelsSuperset(dependencyFilter));
181     return allowedLabels;
182   }

184   private Collection<Target> filterFwdDeps(Target target, Collection<Target> rawFwdDeps) {
185     if (!(target instanceof Rule)) {
186       return rawFwdDeps;
187     }
188     final Set<Label> allowedLabels = getAllowedDeps((Rule) target);
189     return Collections2.filter(rawFwdDeps,
190         new Predicate<Target>() {
191           @Override
192           public boolean apply(Target target) {
193             return allowedLabels.contains(target.getLabel());
194           }
195         });
196   }

198   @Override
199   public Collection<Target> getFwdDeps(Iterable<Target> targets) {
200     Set<Target> result = new HashSet<>();
201     for (Map.Entry<Target, Collection<Target>> entry : getRawFwdDeps(targets).entrySet()) {
202       result.addAll(filterFwdDeps(entry.getKey(), entry.getValue()));
203     }
204     return result;
205   }
206 
207   private Collection<Target> filterReverseDeps(final Target target,
208       Collection<Target> rawReverseDeps) {
209     return Collections2.filter(rawReverseDeps, new Predicate<Target>() {
210       @Override
211       public boolean apply(Target parent) {
212         return !(parent instanceof Rule)
213             || getAllowedDeps((Rule) parent).contains(target.getLabel());
214       }
215     });
216 
217   }

219   @Override

221     Set<Target> result = new HashSet<>();
222     for (Map.Entry<Target, Collection<Target>> entry : getRawReverseDeps(targets).entrySet()) {
223       result.addAll(filterReverseDeps(entry.getKey(), entry.getValue()));

228   @Override
229   public Set<Target> getTransitiveClosure(Set<Target> targets) {
230     Set<Target> visited = new HashSet<>();
231     Collection<Target> current = targets;
232     while (!current.isEmpty()) {
233       Collection<Target> toVisit = Collections2.filter(current,
234           Predicates.not(Predicates.in(visited)));
235       current = getFwdDeps(toVisit);
236       visited.addAll(toVisit);
237     }
238     return ImmutableSet.copyOf(visited);
239   }

241   // Implemented with a breadth-first search.
242   @Override
243   public Set<Target> getNodesOnPath(Target from, Target to) {
244     // Tree of nodes visited so far.
245     Map<Target, Target> nodeToParent = new HashMap<>();
246     // Contains all nodes left to visit in a (LIFO) stack.
247     Deque<Target> toVisit = new ArrayDeque<>();
248     toVisit.add(from);
249     nodeToParent.put(from, null);
250     while (!toVisit.isEmpty()) {
251       Target current = toVisit.removeFirst();
252       if (to.equals(current)) {
253         return ImmutableSet.copyOf(Digraph.getPathToTreeNode(nodeToParent, to));
254       }
255       for (Target dep : getFwdDeps(ImmutableList.of(current))) {
256         if (!nodeToParent.containsKey(dep)) {
257           nodeToParent.put(dep, current);
258           toVisit.addFirst(dep);
259         }
260       }
261     }
262     // Note that the only current caller of this method checks first to see if there is a path
263     // before calling this method. It is not clear what the return value should be here.
264     return null;
265   }

267   @Override
268   public Set<Target> getTargetsMatchingPattern(QueryExpression owner, String pattern)
269       throws QueryException {
270     Set<Target> targets = new LinkedHashSet<>(resolvedTargetPatterns.get(pattern).getTargets());
271 
272     // Sets.filter would be more convenient here, but can't deal with exceptions.
273     Iterator<Target> targetIterator = targets.iterator();
274     while (targetIterator.hasNext()) {
275       Target target = targetIterator.next();
276       if (!validateScope(target.getLabel(), strictScope)) {
277         targetIterator.remove();
278       }
279     }
280     return targets;
281   }

283   @Override
284   public Set<Target> getBuildFiles(QueryExpression caller, Set<Target> nodes)
285       throws QueryException {
286     Set<Target> dependentFiles = new LinkedHashSet<>();
287     Set<Package> seenPackages = new HashSet<>();
288     // Keep track of seen labels, to avoid adding a fake subinclude label that also exists as a
289     // real target.
290     Set<Label> seenLabels = new HashSet<>();
291 
292     // Adds all the package definition files (BUILD files and build
293     // extensions) for package "pkg", to "buildfiles".
294     for (Target x : nodes) {
295       Package pkg = x.getPackage();
296       if (seenPackages.add(pkg)) {
297         addIfUniqueLabel(pkg.getBuildFile(), seenLabels, dependentFiles);
298         for (Label subinclude
299             : Iterables.concat(pkg.getSubincludeLabels(), pkg.getSkylarkFileDependencies())) {
300           addIfUniqueLabel(getSubincludeTarget(subinclude, pkg), seenLabels, dependentFiles);
301 
302           // Also add the BUILD file of the subinclude.
303           try {
304             addIfUniqueLabel(getSubincludeTarget(
305                 subinclude.getLocalTargetLabel("BUILD"), pkg), seenLabels, dependentFiles);
306           } catch (Label.SyntaxException e) {
307             throw new AssertionError("BUILD should always parse as a target name", e);
308           }
309         }
310       }
311     }
312     return dependentFiles;
313   }

315   private static void addIfUniqueLabel(Target node, Set<Label> labels, Set<Target> nodes) {
316     if (labels.add(node.getLabel())) {
317       nodes.add(node);
318     }
319   }

330   private SkyKey getPackageKeyAndValidateLabel(Label label) throws QueryException {
331     // Can't use strictScope here because we are expecting a target back.
332     validateScope(label, true);
333     return PackageValue.key(label.getPackageIdentifier());
334   }

336   @Override
337   public Target getTarget(Label label) throws TargetNotFoundException, QueryException {
338     SkyKey packageKey = getPackageKeyAndValidateLabel(label);
339     if (!graph.exists(packageKey)) {
340       throw new QueryException(packageKey + " does not exist in graph");
341     }
342     try {
343       PackageValue packageValue = (PackageValue) graph.getValue(packageKey);
344       if (packageValue != null) {
345         return packageValue.getPackage().getTarget(label.getName());
346       } else {
347         throw (NoSuchThingException) Preconditions.checkNotNull(
348             graph.getException(packageKey), label);
349       }
350     } catch (NoSuchThingException e) {
351       throw new TargetNotFoundException(e);
352     }
353   }

355   @Override
356   public void buildTransitiveClosure(QueryExpression caller, Set<Target> targets, int maxDepth)
357       throws QueryException {
358     // Everything has already been loaded, so here we just check for errors so that we can
359     // pre-emptively throw/report if needed.
360     for (Map.Entry<SkyKey, Exception> entry :
361       graph.getMissingAndExceptions(makeKeys(targets)).entrySet()) {
362       if (entry.getValue() == null) {
363         throw new QueryException(entry.getKey().argument() + " does not exist in graph");
364       }
365       reportBuildFileError(caller, entry.getValue().getMessage());
366     }
367   }

369   @Override
370   protected Map<String, ResolvedTargets<Target>> preloadOrThrow(QueryExpression caller,
371       Collection<String> patterns) throws QueryException, TargetParsingException {
372     GraphBackedRecursivePackageProvider provider =
373         new GraphBackedRecursivePackageProvider(graph, universeTargetPatternKeys);
374     Map<String, ResolvedTargets<Target>> result = Maps.newHashMapWithExpectedSize(patterns.size());
375     for (String pattern : patterns) {
376       SkyKey patternKey = TargetPatternValue.key(pattern,
377           TargetPatternEvaluator.DEFAULT_FILTERING_POLICY, parserPrefix);
378 
379       TargetPatternValue.TargetPatternKey targetPatternKey =
380           ((TargetPatternValue.TargetPatternKey) patternKey.argument());
381 
382       TargetParsingException targetParsingException = null;
383       if (graph.exists(patternKey)) {
384         // The graph already contains a value or exception for this target pattern, so we use it.
385         TargetPatternValue value = (TargetPatternValue) graph.getValue(patternKey);
386         if (value != null) {
387           ResolvedTargets.Builder<Target> targetsBuilder = ResolvedTargets.builder();
388           targetsBuilder.addAll(makeTargetsFromLabels(value.getTargets().getTargets()));
389           targetsBuilder.removeAll(makeTargetsFromLabels(value.getTargets().getFilteredTargets()));
390           result.put(pattern, targetsBuilder.build());
391         } else {
392           // Because the graph was always initialized via a keep_going build, we know that the
393           // exception stored here must be a TargetParsingException. Thus the comment in
394           // SkyframeTargetPatternEvaluator#parseTargetPatternKeys describing the situation in which
395           // the exception acceptance must be looser does not apply here.
396           targetParsingException =
397               (TargetParsingException)
398                   Preconditions.checkNotNull(graph.getException(patternKey), pattern);
399         }
400       } else {
401         // If the graph doesn't contain a value for this target pattern, try to directly evaluate
402         // it, by making use of packages already present in the graph.
403         RecursivePackageProviderBackedTargetPatternResolver resolver =
404             new RecursivePackageProviderBackedTargetPatternResolver(provider, eventHandler,
405                 targetPatternKey.getPolicy(), pkgPath);
406         TargetPattern parsedPattern = targetPatternKey.getParsedPattern();
407         try {
408           result.put(pattern, parsedPattern.eval(resolver));
409         } catch (TargetParsingException e) {
410           targetParsingException = e;
411         } catch (InterruptedException e) {
412           throw new QueryException(e.getMessage());
413         }
414       }
415 
416       if (targetParsingException != null) {
417         if (!keepGoing) {
418           throw targetParsingException;
419         } else {
420           eventHandler.handle(Event.error("Evaluation of query \"" + caller + "\" failed: "
421               + targetParsingException.getMessage()));
422           result.put(pattern, ResolvedTargets.<Target>builder().setError().build());
423         }
424       }
425     }
426     return result;
427   }

433   private static final Function<SkyKey, Label> SKYKEY_TO_LABEL = new Function<SkyKey, Label>() {
434     @Nullable
435     @Override
436     public Label apply(SkyKey skyKey) {
437       SkyFunctionName functionName = skyKey.functionName();
438       if (!functionName.equals(SkyFunctions.TRANSITIVE_TRAVERSAL)) {
439         // Skip non-targets.
440         return null;
441       }
442       return (Label) skyKey.argument();
443     }
444   };

446   private Map<SkyKey, Target> makeTargetsWithAssociations(Iterable<SkyKey> keys) {
447     return makeTargetsWithAssociations(keys, SKYKEY_TO_LABEL);
448   }

450   private Collection<Target> makeTargetsFromLabels(Iterable<Label> labels) {
451     return makeTargetsWithAssociations(labels, Functions.<Label>identity()).values();
452   }
453 
454   private <E> Map<E, Target> makeTargetsWithAssociations(Iterable<E> keys,
455       Function<E, Label> toLabel) {
456     Multimap<SkyKey, E> packageKeyToTargetKeyMap = ArrayListMultimap.create();
457     for (E key : keys) {
458       Label label = toLabel.apply(key);
459       if (label == null) {
460         continue;
461       }
462       try {
463         packageKeyToTargetKeyMap.put(getPackageKeyAndValidateLabel(label), key);
464       } catch (QueryException e) {
465         // Skip disallowed labels.
466       }
467     }
468     ImmutableMap.Builder<E, Target> result = ImmutableMap.builder();
469     Map<SkyKey, SkyValue> packageMap = graph.getDoneValues(packageKeyToTargetKeyMap.keySet());
470     for (Map.Entry<SkyKey, SkyValue> entry : packageMap.entrySet()) {
471       for (E targetKey : packageKeyToTargetKeyMap.get(entry.getKey())) {
472         try {
473           result.put(targetKey, ((PackageValue) entry.getValue()).getPackage()
474               .getTarget((toLabel.apply(targetKey)).getName()));
475         } catch (NoSuchTargetException e) {
476           // Skip missing target.
477         }
478       }
479     }
480     return result.build();
481   }

483   private Iterable<SkyKey> makeKeys(Iterable<Target> targets) {
484     return Iterables.transform(targets, new Function<Target, SkyKey>() {
485       @Override
486       public SkyKey apply(Target target) {
487         return TransitiveTraversalValue.key(target.getLabel());
488       }
489     });
490   }
