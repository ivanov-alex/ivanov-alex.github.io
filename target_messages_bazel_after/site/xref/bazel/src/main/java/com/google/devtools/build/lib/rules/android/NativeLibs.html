44  /** Represents the collection of native libraries (.so) to be installed in the APK. */
45  public final class NativeLibs {
46    public static final NativeLibs EMPTY =
47        new NativeLibs(ImmutableMap.<String, Iterable<Artifact>>of(), null);
48  
49    public static NativeLibs fromPrecompiledObjects(
50        RuleContext ruleContext, Multimap<String, TransitiveInfoCollection> depsByArchitecture) {
51      ImmutableMap.Builder<String, Iterable<Artifact>> builder = ImmutableMap.builder();
52      for (Map.Entry<String, Collection<TransitiveInfoCollection>> entry :
53          depsByArchitecture.asMap().entrySet()) {
54        NestedSet<LinkerInput> nativeLibraries =
55            AndroidCommon.collectTransitiveNativeLibraries(entry.getValue());
56        builder.put(entry.getKey(), checkUniqueBaseNames(ruleContext, nativeLibraries));
57      }
58      return new NativeLibs(builder.build(), null);
59    }
60  
61    public static NativeLibs fromLinkedNativeDeps(
62        RuleContext ruleContext,
63        String nativeDepsFileName,
64        Multimap<String, TransitiveInfoCollection> depsByArchitecture,
65        Map<String, CcToolchainProvider> toolchainMap,
66        Map<String, BuildConfiguration> configurationMap) {
67      Map<String, Iterable<Artifact>> result = new LinkedHashMap<>();
68      for (Map.Entry<String, Collection<TransitiveInfoCollection>> entry :
69          depsByArchitecture.asMap().entrySet()) {
70        CcLinkParams linkParams = AndroidCommon.getCcLinkParamsStore(entry.getValue())
71            .get(/* linkingStatically */ true, /* linkShared */ true);
72        Artifact nativeDepsLibrary = NativeDepsHelper.maybeCreateAndroidNativeDepsAction(
73            ruleContext, linkParams, configurationMap.get(entry.getKey()),
74            toolchainMap.get(entry.getKey()));
75        if (nativeDepsLibrary != null) {
76          result.put(entry.getKey(), ImmutableList.of(nativeDepsLibrary));
77        }
78      }
79      if (result.isEmpty()) {
80        return NativeLibs.EMPTY;
81      } else {
82        Artifact anyNativeLibrary =
83            result.entrySet().iterator().next().getValue().iterator().next();
84        // The native deps name file must be the only file in its directory because ApkBuilder does
85        // not have an option to add a particular file to the .apk, only one to add every file in a
86        // particular directory.
87        Artifact nativeDepsName = ruleContext.getUniqueDirectoryArtifact(
88            "nativedeps_filename", nativeDepsFileName,
89            ruleContext.getBinOrGenfilesDirectory());
90        ruleContext.registerAction(new FileWriteAction(ruleContext.getActionOwner(), nativeDepsName,
91            anyNativeLibrary.getExecPath().getBaseName(), false));
92  
93        return new NativeLibs(ImmutableMap.copyOf(result), nativeDepsName);
94      }
95    }

97    // Map from architecture (CPU folder to place the library in) to libraries for that CPU
98    private final Map<String, Iterable<Artifact>> nativeLibs;
99    private final Artifact nativeLibsName;
100 
101   private NativeLibs(Map<String, Iterable<Artifact>> nativeLibs, Artifact nativeLibsName) {
102     this.nativeLibs = nativeLibs;
103     this.nativeLibsName = nativeLibsName;
104   }
105 
106   /**
107    * Returns a map from the name of the architecture (CPU folder to place the library in) to the
108    * set of libraries for that architecture.
109    */
110   public Map<String, Iterable<Artifact>> getMap() {
111     return nativeLibs;
112   }
113 
114   public ImmutableSet<Artifact> getAllNativeLibs() {
115     ImmutableSet.Builder<Artifact> result = ImmutableSet.builder();
116 
117     for (Iterable<Artifact> libs : nativeLibs.values()) {
118       result.addAll(libs);
119     }
120 
121     return result.build();
122   }
123 
124   public Artifact createApkBuilderSymlinks(RuleContext ruleContext) {
125     Map<PathFragment, Artifact> symlinks = new LinkedHashMap<>();
126     for (Map.Entry<String, Iterable<Artifact>> entry : nativeLibs.entrySet()) {
127       String arch = entry.getKey();
128       for (Artifact lib : entry.getValue()) {
129         symlinks.put(new PathFragment(arch + "/" + lib.getExecPath().getBaseName()), lib);
130       }
131     }
132 
133     if (symlinks.isEmpty()) {
134       return null;
135     }
136 
137     Artifact inputManifest = AndroidBinary.getDxArtifact(ruleContext, "native_symlinks.manifest");
138     ruleContext.registerAction(new SourceManifestAction.Builder(
139         ManifestType.SOURCE_SYMLINKS, ruleContext.getActionOwner(), inputManifest)
140         .addRootSymlinks(symlinks)
141         .build());
142     Artifact outputManifest = AndroidBinary.getDxArtifact(ruleContext, "native_symlinks/MANIFEST");
143     ruleContext.registerAction(new SymlinkTreeAction(
144         ruleContext.getActionOwner(), inputManifest, outputManifest, false));
145     return outputManifest;
146   }

148   /**
149    * Returns the artifact containing the names of the native libraries or null if it does not exist.
150    *
151    * <p>This artifact will be put in the root directory of the .apk and can be used to load the
152    * libraries programmatically without knowing their names.
153    */
154   @Nullable
155   public Artifact getName() {
156     return nativeLibsName;
157   }
158 
159   private static Iterable<Artifact> checkUniqueBaseNames(
160       RuleContext ruleContext, NestedSet<LinkerInput> libraries) {
161     Map<String, Artifact> basenames = new HashMap<>();
162     Set<Artifact> artifacts = new HashSet<>();
163     for (LinkerInput linkerInput : libraries) {
164       Artifact artifact = linkerInput.getOriginalLibraryArtifact();
165       if (!artifacts.add(artifact)) {
166         // We have already reached this library, e.g., through a different solib symlink.
167         continue;
168       }
169       String basename = artifact.getExecPath().getBaseName();
170       Artifact oldArtifact = basenames.put(basename, artifact);
171       if (oldArtifact != null) {
172         // Without linking, there may be name collisions in the libraries which were provided, so
173         // check for this at this step.
174         ruleContext.ruleError("Each library in the transitive closure must have a unique "
175             + "basename, but two libraries had the basename '" + basename + "': "
176             + artifact.prettyPrint() + " and " + oldArtifact.prettyPrint());
177       }
178     }
179     return artifacts;
180   }
