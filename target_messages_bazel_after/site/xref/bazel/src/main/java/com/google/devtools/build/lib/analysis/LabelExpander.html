31  /**
32   * Helper class encapsulating string scanning state used during "heuristic"
33   * expansion of labels embedded within rules.
34   */
35  public final class LabelExpander {
36    /**
37     * An exception that is thrown when a label is expanded to zero or multiple
38     * files during expansion.
39     */
40    public static class NotUniqueExpansionException extends Exception {
41      public NotUniqueExpansionException(int sizeOfResultSet, String labelText) {
42        super("heuristic label expansion found '" + labelText + "', which expands to "
43            + sizeOfResultSet + " files"
44            + (sizeOfResultSet > 1
45                ? ", please use $(locations " + labelText + ") instead"
46                : ""));
47      }
48    }
49  
50    // This is a utility class, no need to instantiate.
51    private LabelExpander() {}
52  
53    /**
54     * CharMatcher to determine if a given character is valid for labels.
55     *
56     * <p>The Build Concept Reference additionally allows '=' and ',' to appear in labels,
57     * but for the purposes of the heuristic, this function does not, as it would cause
58     * "--foo=:rule1,:rule2" to scan as a single possible label, instead of three
59     * ("--foo", ":rule1", ":rule2").
60     */
61    private static final CharMatcher LABEL_CHAR_MATCHER =
62        CharMatcher.inRange('a', 'z')
63        .or(CharMatcher.inRange('A', 'Z'))
64        .or(CharMatcher.inRange('0', '9'))
65        .or(CharMatcher.anyOf(":/_.-+" + PathFragment.SEPARATOR_CHAR));
66  
67    /**
68     * Expands all references to labels embedded within a string using the
69     * provided expansion mapping from labels to artifacts.
70     *
71     * <p>Since this pass is heuristic, references to non-existent labels (such
72     * as arbitrary words) or invalid labels are simply ignored and are unchanged
73     * in the output. However, if the heuristic discovers a label, which
74     * identifies an existing target producing zero or multiple files, an error
75     * is reported.
76     *
77     * @param expression the expression to expand.
78     * @param labelMap the mapping from labels to artifacts, whose relative path
79     *     is to be used as the expansion.
80     * @param labelResolver the {@code Label} that can resolve label strings
81     *     to {@code Label} objects. The resolved label is either relative to
82     *     {@code labelResolver} or is a global label (i.e. starts with "//").
83     * @return the expansion of the string.
84     * @throws NotUniqueExpansionException if a label that is present in the
85     *     mapping expands to zero or multiple files.
86     */
87    public static <T extends Iterable<Artifact>> String expand(@Nullable String expression,
88        Map<Label, T> labelMap, Label labelResolver) throws NotUniqueExpansionException {
89      if (Strings.isNullOrEmpty(expression)) {
90        return "";
91      }
92      Preconditions.checkNotNull(labelMap);
93      Preconditions.checkNotNull(labelResolver);
94  
95      int offset = 0;
96      StringBuilder result = new StringBuilder();
97      while (offset < expression.length()) {
98        String labelText = scanLabel(expression, offset);
99        if (labelText != null) {
100         offset += labelText.length();
101         result.append(tryResolvingLabelTextToArtifactPath(labelText, labelMap, labelResolver));
102       } else {
103         result.append(expression.charAt(offset));
104         offset++;
105       }
106     }
107     return result.toString();
108   }
109 
110   /**
111    * Tries resolving a label text to a full label for the associated {@code
112    * Artifact}, using the provided mapping.
113    *
114    * <p>The method succeeds if the label text can be resolved to a {@code
115    * Label} object, which is present in the {@code labelMap} and maps to
116    * exactly one {@code Artifact}.
117    *
118    * @param labelText the text to resolve.
119    * @param labelMap the mapping from labels to artifacts, whose relative path
120    *     is to be used as the expansion.
121    * @param labelResolver the {@code Label} that can resolve label strings
122    *     to {@code Label} objects. The resolved label is either relative to
123    *     {@code labelResolver} or is a global label (i.e. starts with "//").
124    * @return an absolute label to an {@code Artifact} if the resolving was
125    *     successful or the original label text.
126    * @throws NotUniqueExpansionException if a label that is present in the
127    *     mapping expands to zero or multiple files.
128    */
129   private static <T extends Iterable<Artifact>> String tryResolvingLabelTextToArtifactPath(
130       String labelText, Map<Label, T> labelMap, Label labelResolver)
131       throws NotUniqueExpansionException {
132     Label resolvedLabel = resolveLabelText(labelText, labelResolver);
133     if (resolvedLabel != null) {
134       Iterable<Artifact> artifacts = labelMap.get(resolvedLabel);
135       if (artifacts != null) { // resolvedLabel identifies an existing target
136         List<String> locations = new ArrayList<>();
137         Artifact.addExecPaths(artifacts, locations);
138         int resultSetSize = locations.size();
139         if (resultSetSize == 1) {
140           return Iterables.getOnlyElement(locations); // success!
141         } else {
142           throw new NotUniqueExpansionException(resultSetSize, labelText);
143         }
144       }
145     }
146     return labelText;
147   }
148 
149   /**
150    * Resolves a string to a label text. Uses {@code labelResolver} to do so.
151    * The result is either relative to {@code labelResolver} or is an absolute
152    * label. In case of an invalid label text, the return value is null.
153    */
154   private static Label resolveLabelText(String labelText, Label labelResolver) {
155     try {
156       return labelResolver.getRelative(labelText);
157     } catch (Label.SyntaxException e) {
158       // It's a heuristic, so quietly ignore "errors". Because Label.getRelative never
159       // returns null, we can use null to indicate an error.
160       return null;
161     }
162   }
163 
164   /**
165    * Scans the argument string from a given start position until the name of a
166    * potential label has been consumed, then returns the label text. If
167    * the expression contains no possible label starting at the start position,
168    * the return value is null.
169    */
170   private static String scanLabel(String expression, int start) {
171     int offset = start;
172     while (offset < expression.length() && LABEL_CHAR_MATCHER.matches(expression.charAt(offset))) {
173       ++offset;
174     }
175     if (offset > start) {
176       return expression.substring(start, offset);
177     } else {
178       return null;
179     }
180   }
