42  /**
43   * A Skyframe function to look up and import a single Skylark extension.
44   */
45  public class SkylarkImportLookupFunction implements SkyFunction {
46  
47    private final RuleClassProvider ruleClassProvider;
48    private final ImmutableList<BaseFunction> nativeRuleFunctions;
49  
50    public SkylarkImportLookupFunction(
51        RuleClassProvider ruleClassProvider, PackageFactory packageFactory) {
52      this.ruleClassProvider = ruleClassProvider;
53      this.nativeRuleFunctions = packageFactory.collectNativeRuleFunctions();
54    }
55  
56    @Override
57    public SkyValue compute(SkyKey skyKey, Environment env) throws SkyFunctionException,
58        InterruptedException {
59      PackageIdentifier arg = (PackageIdentifier) skyKey.argument();
60      PathFragment file = arg.getPackageFragment();
61      ASTFileLookupValue astLookupValue = null;
62      try {
63        SkyKey astLookupKey = ASTFileLookupValue.key(file);
64        astLookupValue = (ASTFileLookupValue) env.getValueOrThrow(astLookupKey,
65            ErrorReadingSkylarkExtensionException.class, InconsistentFilesystemException.class);
66      } catch (ErrorReadingSkylarkExtensionException e) {
67        throw new SkylarkImportLookupFunctionException(SkylarkImportFailedException.errorReadingFile(
68            file, e.getMessage()));
69      } catch (InconsistentFilesystemException e) {
70        throw new SkylarkImportLookupFunctionException(e, Transience.PERSISTENT);
71      } catch (ASTLookupInputException e) {
72        throw new SkylarkImportLookupFunctionException(e, Transience.PERSISTENT);
73      }
74      if (astLookupValue == null) {
75        return null;
76      }
77      if (astLookupValue.getAST() == null) {
78        // Skylark import files have to exist.
79        throw new SkylarkImportLookupFunctionException(SkylarkImportFailedException.noFile(file));
80      }
81  
82      Map<PathFragment, SkylarkEnvironment> importMap = new HashMap<>();
83      ImmutableList.Builder<SkylarkFileDependency> fileDependencies = ImmutableList.builder();
84      BuildFileAST ast = astLookupValue.getAST();
85      // TODO(bazel-team): Refactor this code and PackageFunction to reduce code duplications.
86      for (PathFragment importFile : ast.getImports()) {
87        try {
88          SkyKey importsLookupKey =
89              SkylarkImportLookupValue.key(arg.getRepository(), file, importFile);
90          SkylarkImportLookupValue importsLookupValue;
91          importsLookupValue = (SkylarkImportLookupValue) env.getValueOrThrow(
92              importsLookupKey, ASTLookupInputException.class);
93          if (importsLookupValue != null) {
94            importMap.put(importFile, importsLookupValue.getImportedEnvironment());
95            fileDependencies.add(importsLookupValue.getDependency());
96          }
97        } catch (ASTLookupInputException e) {
98          throw new SkylarkImportLookupFunctionException(e, Transience.PERSISTENT);
99        }
100     }
101     Label label = pathFragmentToLabel(arg.getRepository(), file, env);
102     if (env.valuesMissing()) {
103       // This means some imports are unavailable.
104       return null;
105     }
106 
107     if (ast.containsErrors()) {
108       throw new SkylarkImportLookupFunctionException(SkylarkImportFailedException.skylarkErrors(
109           file));
110     }
111 
112     SkylarkEnvironment extensionEnv = createEnv(ast, file, importMap, env);
113     // Skylark UserDefinedFunctions are sharing function definition Environments, so it's extremely
114     // important not to modify them from this point. Ideally they should be only used to import
115     // symbols and serve as global Environments of UserDefinedFunctions.
116     return new SkylarkImportLookupValue(
117         extensionEnv, new SkylarkFileDependency(label, fileDependencies.build()));
118   }

120   /**
121    * Converts the PathFragment of the Skylark file to a Label using the BUILD file closest to the
122    * Skylark file in its directory hierarchy - finds the package to which the Skylark file belongs.
123    * Throws an exception if no such BUILD file exists.
124    */
125   private Label pathFragmentToLabel(RepositoryName repo, PathFragment file, Environment env)
126       throws SkylarkImportLookupFunctionException {
127     ContainingPackageLookupValue containingPackageLookupValue = null;
128     try {
129       PackageIdentifier newPkgId = new PackageIdentifier(repo, file.getParentDirectory());
130       containingPackageLookupValue = (ContainingPackageLookupValue) env.getValueOrThrow(
131           ContainingPackageLookupValue.key(newPkgId),
132           BuildFileNotFoundException.class, InconsistentFilesystemException.class);
133     } catch (BuildFileNotFoundException e) {
134       // Thrown when there are IO errors looking for BUILD files.
135       throw new SkylarkImportLookupFunctionException(e, Transience.PERSISTENT);
136     } catch (InconsistentFilesystemException e) {
137       throw new SkylarkImportLookupFunctionException(e, Transience.PERSISTENT);
138     }
139 
140     if (containingPackageLookupValue == null) {
141       return null;
142     }
143 
144     if (!containingPackageLookupValue.hasContainingPackage()) {
145       throw new SkylarkImportLookupFunctionException(
146           SkylarkImportFailedException.noBuildFile(file));
147     }
148 
149     PathFragment pkgName =
150         containingPackageLookupValue.getContainingPackageName().getPackageFragment();
151     PathFragment fileInPkg = file.relativeTo(pkgName);
152 
153     try {
154       // This code relies on PackageIdentifier.RepositoryName.toString()
155       return Label.parseAbsolute(repo + "//" + pkgName.getPathString() + ":" + fileInPkg);
156     } catch (SyntaxException e) {
157       throw new IllegalStateException(e);
158     }
159   }

161   /**
162    * Creates the SkylarkEnvironment to be imported. After it's returned, the Environment
163    * must not be modified.
164    */
165   private SkylarkEnvironment createEnv(BuildFileAST ast, PathFragment file,
166       Map<PathFragment, SkylarkEnvironment> importMap, Environment env)
167           throws InterruptedException, SkylarkImportLookupFunctionException {
168     StoredEventHandler eventHandler = new StoredEventHandler();
169     // TODO(bazel-team): this method overestimates the changes which can affect the
170     // Skylark RuleClass. For example changes to comments or unused functions can modify the hash.
171     // A more accurate - however much more complicated - way would be to calculate a hash based on
172     // the transitive closure of the accessible AST nodes.
173     SkylarkEnvironment extensionEnv = ruleClassProvider
174         .createSkylarkRuleClassEnvironment(eventHandler, ast.getContentHashCode());
175     // Adding native rules module for build extensions.
176     // TODO(bazel-team): this might not be the best place to do this.
177     for (BaseFunction function : nativeRuleFunctions) {
178         extensionEnv.registerFunction(
179             SkylarkNativeModule.class, function.getName(), function);
180     }
181     extensionEnv.setImportedExtensions(importMap);
182     ast.exec(extensionEnv, eventHandler);
183     SkylarkRuleClassFunctions.exportRuleFunctions(extensionEnv, file);
184 
185     Event.replayEventsOn(env.getListener(), eventHandler.getEvents());
186     if (eventHandler.hasErrors()) {
187       throw new SkylarkImportLookupFunctionException(SkylarkImportFailedException.errors(file));
188     }
189     return extensionEnv;
190   }

197   static final class SkylarkImportFailedException extends Exception {
198     private SkylarkImportFailedException(String errorMessage) {
199       super(errorMessage);
200     }
201 
202     static SkylarkImportFailedException errors(PathFragment file) {
203       return new SkylarkImportFailedException(
204           String.format("Extension file '%s' has errors", file));
205     }
206 
207     static SkylarkImportFailedException errorReadingFile(PathFragment file, String error) {
208       return new SkylarkImportFailedException(
209           String.format("Encountered error while reading extension file '%s': %s", file, error));
210     }
211 
212     static SkylarkImportFailedException noFile(PathFragment file) {
213       return new SkylarkImportFailedException(
214           String.format("Extension file not found: '%s'", file));
215     }
216 
217     static SkylarkImportFailedException noBuildFile(PathFragment file) {
218       return new SkylarkImportFailedException(
219           String.format("Every .bzl file must have a corresponding package, but '%s' "
220               + "does not have one. Please create a BUILD file in the same or any parent directory."
221               + " Note that this BUILD file does not need to do anything except exist.", file));
222     }
223 
224     static SkylarkImportFailedException skylarkErrors(PathFragment file) {
225       return new SkylarkImportFailedException(String.format("Extension '%s' has errors", file));
226     }
227   }

229   private static final class SkylarkImportLookupFunctionException extends SkyFunctionException {
230     private SkylarkImportLookupFunctionException(SkylarkImportFailedException cause) {
231       super(cause, Transience.PERSISTENT);
232     }
233 
234     private SkylarkImportLookupFunctionException(InconsistentFilesystemException e,
235         Transience transience) {
236       super(e, transience);
237     }
238 
239     private SkylarkImportLookupFunctionException(ASTLookupInputException e,
240         Transience transience) {
241       super(e, transience);
242     }
243 
244     private SkylarkImportLookupFunctionException(BuildFileNotFoundException e,
245         Transience transience) {
246       super(e, transience);
247     }
248   }
