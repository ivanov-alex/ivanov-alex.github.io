55    private final CppSemantics semantics;
56  
57    protected CcLibrary(CppSemantics semantics) {
58      this.semantics = semantics;

61    // These file extensions don't generate object files.
62    private static final FileTypeSet NO_OBJECT_GENERATING_FILETYPES = FileTypeSet.of(
63        CppFileTypes.CPP_HEADER, CppFileTypes.ARCHIVE, CppFileTypes.PIC_ARCHIVE,
64        CppFileTypes.ALWAYS_LINK_LIBRARY, CppFileTypes.ALWAYS_LINK_PIC_LIBRARY,
65        CppFileTypes.SHARED_LIBRARY);
66  
67    private static final Predicate<LibraryToLink> PIC_STATIC_FILTER = new Predicate<LibraryToLink>() {
68      @Override
69      public boolean apply(LibraryToLink input) {
70        String name = input.getArtifact().getExecPath().getBaseName();
71        return !name.endsWith(".nopic.a") && !name.endsWith(".nopic.lo");
72      }
73    };

75    private static Runfiles collectRunfiles(RuleContext context,
76        CcLinkingOutputs ccLinkingOutputs,
77        boolean neverLink, boolean addDynamicRuntimeInputArtifactsToRunfiles,
78        boolean linkingStatically) {
79      Runfiles.Builder builder = new Runfiles.Builder();
80  
81      // neverlink= true creates a library that will never be linked into any binary that depends on
82      // it, but instead be loaded as an extension. So we need the dynamic library for this in the
83      // runfiles.
84      builder.addArtifacts(ccLinkingOutputs.getLibrariesForRunfiles(linkingStatically && !neverLink));
85      builder.add(context, CppRunfilesProvider.runfilesFunction(linkingStatically));
86  
87      builder.addDataDeps(context);
88  
89      if (addDynamicRuntimeInputArtifactsToRunfiles) {
90        builder.addTransitiveArtifacts(CppHelper.getToolchain(context).getDynamicRuntimeLinkInputs());
91      }
92      return builder.build();
93    }

95    @Override
96    public ConfiguredTarget create(RuleContext context) {
97      RuleConfiguredTargetBuilder builder = new RuleConfiguredTargetBuilder(context);
98      LinkTargetType linkType = getStaticLinkType(context);
99      boolean linkStatic = context.attributes().get("linkstatic", Type.BOOLEAN);
100     init(semantics, context, builder, linkType,
101         /*neverLink =*/ false,
102         linkStatic,
103         /*collectLinkstamp =*/ true,
104         /*addDynamicRuntimeInputArtifactsToRunfiles =*/ false);
105     return builder.build();
106   }

108   public static void init(CppSemantics semantics, RuleContext ruleContext,
109       RuleConfiguredTargetBuilder targetBuilder, LinkTargetType linkType,
110       boolean neverLink,
111       boolean linkStatic,
112       boolean collectLinkstamp,
113       boolean addDynamicRuntimeInputArtifactsToRunfiles) {
114     FeatureConfiguration featureConfiguration = CcCommon.configureFeatures(ruleContext);
115     final CcCommon common = new CcCommon(ruleContext, featureConfiguration);
116     PrecompiledFiles precompiledFiles = new PrecompiledFiles(ruleContext);
117 
118     CcLibraryHelper helper =
119         new CcLibraryHelper(ruleContext, semantics, featureConfiguration)
120             .fromCommon(common)
121 
122             .addLinkopts(common.getLinkopts())
123             .addSources(common.getCAndCppSources())
124             .addPublicHeaders(CcCommon.getHeaders(ruleContext))
125             .enableCcNativeLibrariesProvider()
126             .enableCompileProviders()
127             .enableInterfaceSharedObjects()
128             // Generate .a and .so outputs even without object files to fulfill the rule class contract
129             // wrt. implicit output files, if the contract says so. Behavior here differs between Bazel
130             // and Blaze.
131             .setGenerateLinkActionsIfEmpty(
132                 ruleContext.getRule().getRuleClassObject().getImplicitOutputsFunction()
133                     != ImplicitOutputsFunction.NONE)
134             .setLinkType(linkType)
135             .setNeverLink(neverLink)
136             .addPrecompiledFiles(precompiledFiles);
137 
138     if (collectLinkstamp) {
139       helper.addLinkstamps(ruleContext.getPrerequisites("linkstamp", Mode.TARGET));
140     }
141 
142     Artifact soImplArtifact = null;
143     boolean createDynamicLibrary =
144         !linkStatic && appearsToHaveObjectFiles(ruleContext.attributes());
145     if (ruleContext.getRule().isAttrDefined("outs", Type.STRING_LIST)) {
146       List<String> outs = ruleContext.attributes().get("outs", Type.STRING_LIST);
147       if (outs.size() > 1) {
148         ruleContext.attributeError("outs", "must be a singleton list");
149       } else if (outs.size() == 1) {
150         PathFragment soImplFilename = new PathFragment(ruleContext.getLabel().getName());
151         if (LinkTargetType.DYNAMIC_LIBRARY != LinkTargetType.EXECUTABLE) {
152           soImplFilename = soImplFilename.replaceName(
153               "lib" + soImplFilename.getBaseName() + LinkTargetType.DYNAMIC_LIBRARY.getExtension());
154         }
155         soImplFilename = soImplFilename.replaceName(outs.get(0));
156         if (!soImplFilename.getPathString().endsWith(".so")) { // Sanity check.
157           ruleContext.attributeError("outs", "file name must end in '.so'");
158         }
159 
160         if (createDynamicLibrary) {
161           soImplArtifact = ruleContext.getPackageRelativeArtifact(
162               soImplFilename, ruleContext.getConfiguration().getBinDirectory());
163         }
164       }
165     }
166 
167     if (ruleContext.getRule().isAttrDefined("srcs", Type.LABEL_LIST)) {
168       helper.addPrivateHeaders(FileType.filter(
169           ruleContext.getPrerequisiteArtifacts("srcs", Mode.TARGET).list(),
170           CppFileTypes.CPP_HEADER));
171       ruleContext.checkSrcsSamePackage(true);
172     }
173     if (ruleContext.getRule().isAttrDefined("textual_hdrs", Type.LABEL_LIST)) {
174       helper.addPublicTextualHeaders(
175           ruleContext.getPrerequisiteArtifacts("textual_hdrs", Mode.TARGET).list());
176     }
177 
178     if (common.getLinkopts().contains("-static")) {
179       ruleContext.attributeWarning("linkopts", "Using '-static' here won't work. "
180                                    + "Did you mean to use 'linkstatic=1' instead?");
181     }
182 
183     helper.setCreateDynamicLibrary(createDynamicLibrary);
184     helper.setDynamicLibrary(soImplArtifact);
185 
186     // If "srcs" is configurable, the .so output is always declared because the logic that
187     // determines implicit outs doesn't know which value of "srcs" will ultimately get chosen. Here,
188     // where we *do* have the correct value, it may not contain any source files to generate an
189     // .so with. If that's the case, register a fake generating action to prevent a "no generating
190     // action for this artifact" error.
191     if (!createDynamicLibrary && ruleContext.attributes().isConfigurable("srcs", Type.LABEL_LIST)) {
192       Artifact solibArtifact = CppHelper.getLinkedArtifact(
193           ruleContext, LinkTargetType.DYNAMIC_LIBRARY);
194       ruleContext.registerAction(new FailAction(ruleContext.getActionOwner(),
195           ImmutableList.of(solibArtifact), "configurable \"srcs\" triggers an implicit .so output "
196           + "even though there are no sources to compile in this configuration"));
197     }
198 
199     /*
200      * Add the libraries from srcs, if any. For static/mostly static
201      * linking we setup the dynamic libraries if there are no static libraries
202      * to choose from. Path to the libraries will be mangled to avoid using
203      * absolute path names on the -rpath, but library filenames will be
204      * preserved (since some libraries might have SONAME tag) - symlink will
205      * be created to the parent directory instead.
206      *
207      * For compatibility with existing BUILD files, any ".a" or ".lo" files listed in
208      * srcs are assumed to be position-independent code, or at least suitable for
209      * inclusion in shared libraries, unless they end with ".nopic.a" or ".nopic.lo".
210      *
211      * Note that some target platforms do not require shared library code to be PIC.
212      */
213     Iterable<LibraryToLink> staticLibrariesFromSrcs =
214         LinkerInputs.opaqueLibrariesToLink(precompiledFiles.getStaticLibraries());
215     helper.addStaticLibraries(staticLibrariesFromSrcs);
216     helper.addPicStaticLibraries(Iterables.filter(staticLibrariesFromSrcs, PIC_STATIC_FILTER));
217     helper.addPicStaticLibraries(precompiledFiles.getPicStaticLibraries());
218     helper.addDynamicLibraries(Iterables.transform(precompiledFiles.getSharedLibraries(),
219         new Function<Artifact, LibraryToLink>() {
220       @Override
221       public LibraryToLink apply(Artifact library) {
222         return common.getDynamicLibrarySymlink(library, true);
223       }
224     }));
225     CcLibraryHelper.Info info = helper.build();
226 
227     /*
228      * We always generate a static library, even if there aren't any source files.
229      * This keeps things simpler by avoiding special cases when making use of the library.
230      * For example, this is needed to ensure that building a library with "bazel build"
231      * will also build all of the library's "deps".
232      * However, we only generate a dynamic library if there are source files.
233      */
234     // For now, we don't add the precompiled libraries to the files to build.
235     CcLinkingOutputs linkedLibraries = info.getCcLinkingOutputsExcludingPrecompiledLibraries();
236 
237     NestedSet<Artifact> artifactsToForce =
238         collectHiddenTopLevelArtifacts(ruleContext, common, info.getCcCompilationOutputs());
239 
240     NestedSetBuilder<Artifact> filesBuilder = NestedSetBuilder.stableOrder();
241     filesBuilder.addAll(LinkerInputs.toLibraryArtifacts(linkedLibraries.getStaticLibraries()));
242     filesBuilder.addAll(LinkerInputs.toLibraryArtifacts(linkedLibraries.getPicStaticLibraries()));
243     filesBuilder.addAll(LinkerInputs.toNonSolibArtifacts(linkedLibraries.getDynamicLibraries()));
244     filesBuilder.addAll(
245         LinkerInputs.toNonSolibArtifacts(linkedLibraries.getExecutionDynamicLibraries()));
246 
247     CcLinkingOutputs linkingOutputs = info.getCcLinkingOutputs();
248     warnAboutEmptyLibraries(
249         ruleContext, info.getCcCompilationOutputs(), linkStatic);
250     NestedSet<Artifact> filesToBuild = filesBuilder.build();
251 
252     Runfiles staticRunfiles = collectRunfiles(ruleContext,
253         linkingOutputs, neverLink, addDynamicRuntimeInputArtifactsToRunfiles, true);
254     Runfiles sharedRunfiles = collectRunfiles(ruleContext,
255         linkingOutputs, neverLink, addDynamicRuntimeInputArtifactsToRunfiles, false);
256 
257     List<Artifact> instrumentedObjectFiles = new ArrayList<>();
258     instrumentedObjectFiles.addAll(info.getCcCompilationOutputs().getObjectFiles(false));
259     instrumentedObjectFiles.addAll(info.getCcCompilationOutputs().getObjectFiles(true));
260     InstrumentedFilesProvider instrumentedFilesProvider =
261         common.getInstrumentedFilesProvider(instrumentedObjectFiles);
262     targetBuilder
263         .setFilesToBuild(filesToBuild)
264         .addProviders(info.getProviders())
265         .addSkylarkTransitiveInfo(CcSkylarkApiProvider.NAME, new CcSkylarkApiProvider())
266         .addOutputGroups(info.getOutputGroups())
267         .add(InstrumentedFilesProvider.class, instrumentedFilesProvider)
268         .add(RunfilesProvider.class, RunfilesProvider.withData(staticRunfiles, sharedRunfiles))
269         // Remove this?
270         .add(CppRunfilesProvider.class, new CppRunfilesProvider(staticRunfiles, sharedRunfiles))
271         .addOutputGroup(OutputGroupProvider.HIDDEN_TOP_LEVEL, artifactsToForce)
272         .addOutputGroup(OutputGroupProvider.BASELINE_COVERAGE, BaselineCoverageAction
273                 .getBaselineCoverageArtifacts(ruleContext,
274                     instrumentedFilesProvider.getInstrumentedFiles()));
275 
276   }

278   private static NestedSet<Artifact> collectHiddenTopLevelArtifacts(RuleContext ruleContext,
279       CcCommon common, CcCompilationOutputs ccCompilationOutputs) {
280     // Ensure that we build all the dependencies, otherwise users may get confused.
281     NestedSetBuilder<Artifact> artifactsToForceBuilder = NestedSetBuilder.stableOrder();
282     artifactsToForceBuilder.addTransitive(common.getFilesToCompile(ccCompilationOutputs));
283     for (OutputGroupProvider dep :
284         ruleContext.getPrerequisites("deps", Mode.TARGET, OutputGroupProvider.class)) {
285       artifactsToForceBuilder.addTransitive(
286           dep.getOutputGroup(OutputGroupProvider.HIDDEN_TOP_LEVEL));
287     }
288     return artifactsToForceBuilder.build();
289   }

294   private static LinkTargetType getStaticLinkType(RuleContext context) {
295     return context.attributes().get("alwayslink", Type.BOOLEAN)
296         ? LinkTargetType.ALWAYS_LINK_STATIC_LIBRARY
297         : LinkTargetType.STATIC_LIBRARY;
298   }

300   private static void warnAboutEmptyLibraries(RuleContext ruleContext,
301       CcCompilationOutputs ccCompilationOutputs,
302       boolean linkstaticAttribute) {
303     if (ruleContext.getFragment(CppConfiguration.class).isLipoContextCollector()) {
304       // Do not signal warnings in the lipo context collector configuration. These will be duly
305       // signaled in the target configuration, and there can be spurious warnings since targets in
306       // the LIPO context collector configuration do not compile anything.
307       return;
308     }
309     if (ccCompilationOutputs.getObjectFiles(false).isEmpty()
310         && ccCompilationOutputs.getObjectFiles(true).isEmpty()) {
311       if (!linkstaticAttribute && appearsToHaveObjectFiles(ruleContext.attributes())) {
312         ruleContext.attributeWarning("linkstatic",
313             "setting 'linkstatic=1' is recommended if there are no object files");
314       }
315     } else {
316       if (!linkstaticAttribute && !appearsToHaveObjectFiles(ruleContext.attributes())) {
317         Artifact element = Iterables.getFirst(
318             ccCompilationOutputs.getObjectFiles(false),
319             ccCompilationOutputs.getObjectFiles(true).get(0));
320         ruleContext.attributeWarning("srcs",
321              "this library appears at first glance to have no object files, "
322              + "but on closer inspection it does have something to link, e.g. "
323              + element.prettyPrint() + ". "
324              + "(You may have used some very confusing rule names in srcs? "
325              + "Or the library consists entirely of a linker script?) "
326              + "Bazel assumed linkstatic=1, but this may be inappropriate. "
327              + "You may need to add an explicit '.cc' file to 'srcs'. "
328              + "Alternatively, add 'linkstatic=1' to suppress this warning");
329       }
330     }

333   /**
334    * Returns true if the rule (which must be a cc_library rule) appears to have object files.
335    * This only looks at the rule itself, not at any other rules (from this package or other
336    * packages) that it might reference.
337    *
338    * <p>In some cases, this may return "true" even though the rule actually has no object files.
339    * For example, it will return true for a rule such as
340    * <code>cc_library(name = 'foo', srcs = [':bar'])</code> because we can't tell what ':bar' is;
341    * it might be a genrule that generates a source file, or it might be a genrule that generates a
342    * header file. Likewise,
343    * <code>cc_library(name = 'foo', srcs = select({':a': ['foo.cc'], ':b': []}))</code> returns
344    * "true" even though the sources *may* be empty. This reflects the fact that there's no way
345    * to tell which value "srcs" will take without knowing the rule's configuration.
346    *
347    * <p>In other cases, this may return "false" even though the rule actually does have object
348    * files. For example, it will return false for a rule such as
349    * <code>cc_library(name = 'foo', srcs = ['bar.h'])</code> but as in the other example above,
350    * we can't tell whether 'bar.h' is a file name or a rule name, and 'bar.h' could in fact be the
351    * name of a genrule that generates a source file.
352    */
353   public static boolean appearsToHaveObjectFiles(AttributeMap rule) {
354     if ((rule instanceof RawAttributeMapper) && rule.isConfigurable("srcs", Type.LABEL_LIST)) {
355       // Since this method gets called by loading phase logic (e.g. the cc_library implicit outputs
356       // function), the attribute mapper may not be able to resolve configurable attributes. When
357       // that's the case, there's no way to know which value a configurable "srcs" will take, so
358       // we conservatively assume object files are possible.
359       return true;
360     }
361 
362     List<Label> srcs = rule.get("srcs", Type.LABEL_LIST);
363     if (srcs != null) {
364       for (Label srcfile : srcs) {
365         /*
366          * We cheat a little bit here by looking at the file extension
367          * of the Label treated as file name.  In general that might
368          * not necessarily work, because of the possibility that the
369          * user might give a rule a funky name ending in one of these
370          * extensions, e.g.
371          *    genrule(name = 'foo.h', outs = ['foo.cc'], ...) // Funky rule name!
372          *    cc_library(name = 'bar', srcs = ['foo.h']) // This DOES have object files.
373          */
374         if (!NO_OBJECT_GENERATING_FILETYPES.matches(srcfile.getName())) {
375           return true;
376         }
377       }
378     }
379     return false;
380   }
