129   /**
130    * Validates that the packages listed under "deps" all have the given constraint. If a package
131    * does not have this attribute, an error is generated.
132    */
133   public static final void validateConstraint(RuleContext ruleContext,
134       String constraint, Iterable<? extends TransitiveInfoCollection> targets) {
135     for (JavaConstraintProvider constraintProvider :
136         AnalysisUtils.getProviders(targets, JavaConstraintProvider.class)) {
137       if (!constraintProvider.getJavaConstraints().contains(constraint)) {
138         ruleContext.attributeError("deps",
139             String.format("%s: does not have constraint '%s'",
140                 constraintProvider.getLabel(), constraint));
141       }
142     }

149   public static void createInstrumentedJarAction(RuleContext ruleContext, JavaSemantics semantics,

151     // In Jacoco's setup, metadata artifacts are real jars.
152     new DeployArchiveBuilder(semantics, ruleContext)
153         .setOutputJar(instrumentedJar)
154         // We need to save the original mainClass because we're going to run inside CoverageRunner
155         .setJavaStartClass(mainClass)

157         .addRuntimeJars(ImmutableList.copyOf(metadataArtifacts))
158         .setCompression(DeployArchiveBuilder.Compression.UNCOMPRESSED)
159         .build();
160   }

198   /**
199    * Creates the java.library.path from a list of the native libraries.
200    * Concatenates the parent directories of the shared libraries into a Java
201    * search path. Each relative path entry is prepended with "${JAVA_RUNFILES}/"
202    * so it can be resolved at runtime.
203    *
204    * @param sharedLibraries a collection of native libraries to create the java
205    *        library path from
206    * @return a String containing the ":" separated java library path
207    */
208   public static String javaLibraryPath(
209       Collection<Artifact> sharedLibraries, String runfilePrefix) {
210     StringBuilder buffer = new StringBuilder();
211     Set<PathFragment> entries = new HashSet<>();
212     for (Artifact sharedLibrary : sharedLibraries) {
213       PathFragment entry = sharedLibrary.getRootRelativePath().getParentDirectory();
214       if (entries.add(entry)) {
215         if (buffer.length() > 0) {
216           buffer.append(':');
217         }
218         buffer.append("${JAVA_RUNFILES}/" + runfilePrefix + "/");
219         buffer.append(entry.getPathString());
220       }
221     }
222     return buffer.toString();
223   }

225   /**
226    * Collects Java compilation arguments for this target.
227    *
228    * @param recursive Whether to scan dependencies recursively.
229    * @param isNeverLink Whether the target has the 'neverlink' attr.
230    */
231   JavaCompilationArgs collectJavaCompilationArgs(boolean recursive, boolean isNeverLink,
232       Iterable<SourcesJavaCompilationArgsProvider> compilationArgsFromSources) {
233     ClasspathType type = isNeverLink ? ClasspathType.COMPILE_ONLY : ClasspathType.BOTH;
234     JavaCompilationArgs.Builder builder = JavaCompilationArgs.builder()
235         .merge(getJavaCompilationArtifacts(), isNeverLink)
236         .addTransitiveTargets(getExports(ruleContext), recursive, type);
237     if (recursive) {
238       builder
239           .addTransitiveTargets(targetsTreatedAsDeps(ClasspathType.COMPILE_ONLY), recursive, type)
240           .addTransitiveTargets(getRuntimeDeps(ruleContext), recursive, ClasspathType.RUNTIME_ONLY)
241           .addSourcesTransitiveCompilationArgs(compilationArgsFromSources, recursive, type);
242     }
243     return builder.build();
244   }

274   /**
275    * Sanity checks the given runtime dependencies, and emits errors if there is a problem.
276    * Also called by {@link #initCommon()} for the current target's runtime dependencies.
277    */
278   public void checkRuntimeDeps(List<TransitiveInfoCollection> runtimeDepInfo) {
279     for (TransitiveInfoCollection c : runtimeDepInfo) {
280       JavaNeverlinkInfoProvider neverLinkedness =
281           c.getProvider(JavaNeverlinkInfoProvider.class);
282       if (neverLinkedness == null) {
283         continue;
284       }
285       boolean reportError = !ruleContext.getConfiguration().getAllowRuntimeDepsOnNeverLink();
286       if (neverLinkedness.isNeverlink()) {
287         String msg = String.format("neverlink dep %s not allowed in runtime deps", c.getLabel());
288         if (reportError) {
289           ruleContext.attributeError("runtime_deps", msg);
290         } else {
291           ruleContext.attributeWarning("runtime_deps", msg);
292         }
293       }
294     }
295   }

345    * Collects labels of targets and artifacts reached transitively via the "exports" attribute.
346    */
347   protected NestedSet<Label> collectTransitiveExports() {
348     NestedSetBuilder<Label> builder = NestedSetBuilder.stableOrder();
349     List<TransitiveInfoCollection> currentRuleExports = getExports(ruleContext);
350 
351     builder.addAll(Iterables.transform(currentRuleExports, GET_COLLECTION_LABEL));
352 
353     for (TransitiveInfoCollection dep : currentRuleExports) {
354       JavaExportsProvider exportsProvider = dep.getProvider(JavaExportsProvider.class);
355 
356       if (exportsProvider != null) {
357         builder.addTransitive(exportsProvider.getTransitiveExports());
358       }
359     }
360 
361     return builder.build();
362   }

374   /**
375    * Returns the string that the stub should use to determine the JVM
376    * @param launcher if non-null, the cc_binary used to launch the Java Virtual Machine
377    */
378   public String getJavaBinSubstitution(@Nullable Artifact launcher) {
379     PathFragment javaExecutable;
380 
381     if (launcher != null) {
382       javaExecutable = launcher.getRootRelativePath();
383     } else {
384       javaExecutable = ruleContext.getFragment(Jvm.class).getJavaExecutable();
385     }
386 
387     String pathPrefix = javaExecutable.isAbsolute() ? "" : "${JAVA_RUNFILES}/"
388         + ruleContext.getRule().getWorkspaceName() + "/";
389     return "JAVABIN=${JAVABIN:-" + pathPrefix + javaExecutable.getPathString() + "}";
390   }

392   /**
393    * Heuristically determines the name of the primary Java class for this
394    * executable, based on the rule name and the "srcs" list.
395    *
396    * <p>(This is expected to be the class containing the "main" method for a
397    * java_binary, or a JUnit Test class for a java_test.)
398    *
399    * @param sourceFiles the source files for this rule
400    * @return a fully qualified Java class name, or null if none could be
401    *   determined.
402    */
403   public String determinePrimaryClass(Collection<Artifact> sourceFiles) {
404     if (!sourceFiles.isEmpty()) {
405       String mainSource = ruleContext.getTarget().getName() + ".java";
406       for (Artifact sourceFile : sourceFiles) {
407         PathFragment path = sourceFile.getRootRelativePath();
408         if (path.getBaseName().equals(mainSource)) {
409           return JavaUtil.getJavaFullClassname(FileSystemUtils.removeExtension(path));
410         }
411       }
412     }
413     // Last resort: Use the name and package name of the target.
414     // TODO(bazel-team): this should be fixed to use a source file from the dependencies to
415     // determine the package of the Java class.
416     return JavaUtil.getJavaFullClassname(Util.getWorkspaceRelativePath(ruleContext.getTarget()));
417   }

444   /**
445    * Initialize the common actions and build various collections of artifacts
446    * for the initializationHook() methods of the subclasses.
447    *
448    * <p>Note that not all subclasses call this method.
449    *
450    * @return the processed attributes
451    */
452   public JavaTargetAttributes.Builder initCommon(Collection<Artifact> extraSrcs) {
453     Preconditions.checkState(javacOpts != null);
454     sources = ruleContext.getPrerequisiteArtifacts("srcs", Mode.TARGET).list();
455     activePlugins = collectPlugins();
456 
457     JavaTargetAttributes.Builder javaTargetAttributes = new JavaTargetAttributes.Builder(semantics);
458     processSrcs(javaTargetAttributes, javacOpts);
459     javaTargetAttributes.addSourceArtifacts(extraSrcs);
460     processRuntimeDeps(javaTargetAttributes);
461 
462     semantics.commonDependencyProcessing(ruleContext, javaTargetAttributes,
463         targetsTreatedAsDeps(ClasspathType.COMPILE_ONLY));
464 
465     // Check that we have do not have both sources and jars.
466     if ((javaTargetAttributes.hasSourceFiles() || javaTargetAttributes.hasSourceJars())
467         && javaTargetAttributes.hasJarFiles()) {
468       ruleContext.attributeWarning("srcs", "cannot use both Java sources - source "
469           + "jars or source files - and precompiled jars");
470     }
471 
472     if (disallowDepsWithoutSrcs(ruleContext.getRule().getRuleClass())
473         && ruleContext.attributes().get("srcs", Type.LABEL_LIST).isEmpty()
474         && ruleContext.getRule().isAttributeValueExplicitlySpecified("deps")) {
475       ruleContext.attributeError("deps", "deps not allowed without srcs; move to runtime_deps?");
476     }
477 
478     javaTargetAttributes.addResources(semantics.collectResources(ruleContext));
479     addPlugins(javaTargetAttributes);
480 
481     javaTargetAttributes.setRuleKind(ruleContext.getRule().getRuleClass());
482     javaTargetAttributes.setTargetLabel(ruleContext.getLabel());
483 
484     return javaTargetAttributes;
485   }

487   private boolean disallowDepsWithoutSrcs(String ruleClass) {
488     return ruleClass.equals("java_library")
489         || ruleClass.equals("java_binary")
490         || ruleClass.equals("java_test");

498   private static ImmutableList<TransitiveInfoCollection> collectTargetsTreatedAsDeps(
499       RuleContext ruleContext, JavaSemantics semantics, ClasspathType type) {
500     ImmutableList.Builder<TransitiveInfoCollection> builder = new Builder<>();
501 
502     if (!type.equals(ClasspathType.COMPILE_ONLY)) {
503       builder.addAll(getRuntimeDeps(ruleContext));
504       builder.addAll(getExports(ruleContext));
505     }
506     builder.addAll(ruleContext.getPrerequisites("deps", Mode.TARGET));
507 
508     semantics.collectTargetsTreatedAsDeps(ruleContext, builder);
509 
510     // Implicitly add dependency on java launcher cc_binary when --java_launcher= is enabled,
511     // or when launcher attribute is specified in a build rule.
512     TransitiveInfoCollection launcher = JavaHelper.launcherForTarget(semantics, ruleContext);
513     if (launcher != null) {
514       builder.add(launcher);
515     }
516 
517     return builder.build();
518   }

520   public void addTransitiveInfoProviders(RuleConfiguredTargetBuilder builder,
521       NestedSet<Artifact> filesToBuild, @Nullable Artifact classJar) {
522     InstrumentedFilesCollector instrumentedFilesCollector =
523         new InstrumentedFilesCollector(ruleContext, semantics.getCoverageInstrumentationSpec(),
524             JAVA_METADATA_COLLECTOR, filesToBuild);
525 
526     builder
527         .add(InstrumentedFilesProvider.class, new InstrumentedFilesProviderImpl(
528             instrumentedFilesCollector))
529         .add(JavaExportsProvider.class, new JavaExportsProvider(collectTransitiveExports()))
530         .addOutputGroup(OutputGroupProvider.FILES_TO_COMPILE, getFilesToCompile(classJar));
531 
532     if (!TargetUtils.isTestRule(ruleContext.getTarget())) {
533       builder.addOutputGroup(OutputGroupProvider.BASELINE_COVERAGE,
534           BaselineCoverageAction.getBaselineCoverageArtifacts(ruleContext,
535               instrumentedFilesCollector.getInstrumentedFiles()));
536     }
537   }

539   /**
540    * Processes the sources of this target, adding them as messages, proper
541    * sources or to the list of targets treated as deps as required.
542    */
543   private void processSrcs(JavaTargetAttributes.Builder attributes,
544       ImmutableList<String> javacOpts) {
545     for (MessageBundleProvider srcItem : ruleContext.getPrerequisites(
546         "srcs", Mode.TARGET, MessageBundleProvider.class)) {
547       attributes.addMessages(srcItem.getMessages());
548     }
549 
550     attributes.addSourceArtifacts(sources);
551 
552     addCompileTimeClassPathEntriesMaybeThroughIjar(attributes, javacOpts);
553   }

555   /**
556    * Processes the transitive runtime_deps of this target.
557    */
558   private void processRuntimeDeps(JavaTargetAttributes.Builder attributes) {
559     List<TransitiveInfoCollection> runtimeDepInfo = getRuntimeDeps(ruleContext);
560     checkRuntimeDeps(runtimeDepInfo);
561     JavaCompilationArgs args = JavaCompilationArgs.builder()
562         .addTransitiveTargets(runtimeDepInfo, true, ClasspathType.RUNTIME_ONLY)
563         .build();
564     attributes.addRuntimeClassPathEntries(args.getRuntimeJars());
565     attributes.addInstrumentationMetadataEntries(args.getInstrumentationMetadata());
566   }

573   /**
574    * Adds jars in the given group of entries to the compile time classpath after
575    * using ijar to create jar interfaces for the generated jars.
576    */
577   private void addCompileTimeClassPathEntriesMaybeThroughIjar(
578       JavaTargetAttributes.Builder attributes, ImmutableList<String> javacOpts) {
579     JavaCompilationHelper helper = new JavaCompilationHelper(
580         ruleContext, semantics, javacOpts, attributes);
581     for (FileProvider provider : ruleContext
582         .getPrerequisites("srcs", Mode.TARGET, FileProvider.class)) {
583       Iterable<Artifact> jarFiles = helper.filterGeneratedJarsThroughIjar(
584           FileType.filter(provider.getFilesToBuild(), JavaSemantics.JAR));
585       List<Artifact> jarsWithOwners = Lists.newArrayList(jarFiles);
586       attributes.addDirectCompileTimeClassPathEntries(jarsWithOwners);
587       attributes.addCompileTimeJarFiles(jarsWithOwners);
588     }
589   }

591   /**
592    * Adds information about the annotation processors that should be run for this java target to
593    * the target attributes.
594    */
595   private void addPlugins(JavaTargetAttributes.Builder attributes) {
596     for (JavaPluginInfoProvider plugin : activePlugins) {
597       for (String name : plugin.getProcessorClasses()) {
598         attributes.addProcessorName(name);
599       }
600       // Now get the plugin-libraries runtime classpath.
601       attributes.addProcessorPath(plugin.getProcessorClasspath());
602     }
603   }

656   public ImmutableList<Dependency> computeStrictDepsFromJavaAttributes(
657       JavaTargetAttributes javaTargetAttributes) {
658     Multimap<Label, String> depMap = HashMultimap.<Label, String>create();
659     for (Artifact jar : javaTargetAttributes.getDirectJars()) {
660       depMap.put(Preconditions.checkNotNull(jar.getOwner()),
661           jar.getExecPathString());
662     }
663     ImmutableList.Builder<Dependency> depOuts = ImmutableList.builder();
664     for (Label label : depMap.keySet()) {
665       depOuts.add(new Dependency(label, depMap.get(label)));
666     }
667     return depOuts.build();
668   }
