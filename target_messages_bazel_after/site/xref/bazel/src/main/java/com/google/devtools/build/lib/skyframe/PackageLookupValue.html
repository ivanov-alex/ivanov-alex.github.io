26  /**
27   * A value that represents a package lookup result.
28   *
29   * <p>Package lookups will always produce a value. On success, the {@code #getRoot} returns the
30   * package path root under which the package resides and the package's BUILD file is guaranteed to
31   * exist (unless this is looking up a WORKSPACE file, in which case the underlying file may or may
32   * not exist. On failure, {@code #getErrorReason} and {@code #getErrorMsg} describe why the package
33   * doesn't exist.
34   *
35   * <p>Implementation detail: we use inheritance here to optimize for memory usage.
36   */
37  abstract class PackageLookupValue implements SkyValue {
38  
39    enum ErrorReason {
40      // There is no BUILD file.
41      NO_BUILD_FILE,
42  
43      // The package name is invalid.
44      INVALID_PACKAGE_NAME,
45  
46      // The package is considered deleted because of --deleted_packages.
47      DELETED_PACKAGE,
48  
49      // The //external package could not be loaded, either because the WORKSPACE file could not be
50      // parsed or the packages it references cannot be loaded.
51      NO_EXTERNAL_PACKAGE
52    }
53  
54    protected PackageLookupValue() {
55    }
56  
57    public static PackageLookupValue success(Path root) {
58      return new SuccessfulPackageLookupValue(root);
59    }
60  
61    public static PackageLookupValue overlaidBuildFile(
62        Path root, Optional<FileValue> overlaidBuildFile) {
63      return new OverlaidPackageLookupValue(root, overlaidBuildFile);
64    }
65  
66    public static PackageLookupValue workspace(Path root) {
67      return new WorkspacePackageLookupValue(root);
68    }
69  
70    public static PackageLookupValue noBuildFile() {
71      return NoBuildFilePackageLookupValue.INSTANCE;
72    }
73  
74    public static PackageLookupValue noExternalPackage() {
75      return NoExternalPackageLookupValue.INSTANCE;
76    }
77  
78    public static PackageLookupValue invalidPackageName(String errorMsg) {
79      return new InvalidNamePackageLookupValue(errorMsg);
80    }
81  
82    public static PackageLookupValue deletedPackage() {
83      return DeletedPackageLookupValue.INSTANCE;
84    }
85  
86    public boolean isExternalPackage() {
87      return false;
88    }
89  
90    /**
91     * For a successful package lookup, returns the root (package path entry) that the package
92     * resides in.
93     */
94    public abstract Path getRoot();
95  
96    /**
97     * Returns whether the package lookup was successful.
98     */
99    public abstract boolean packageExists();
100 
101   /**
102    * For an unsuccessful package lookup, gets the reason why {@link #packageExists} returns
103    * {@code false}.
104    */
105   abstract ErrorReason getErrorReason();
106 
107   /**
108    * For an unsuccessful package lookup, gets a detailed error message for {@link #getErrorReason}
109    * that is suitable for reporting to a user.
110    */
111   abstract String getErrorMsg();
112 
113   static SkyKey key(PathFragment directory) {
114     Preconditions.checkArgument(!directory.isAbsolute(), directory);
115     return key(PackageIdentifier.createInDefaultRepo(directory));
116   }
117 
118   static SkyKey key(PackageIdentifier pkgIdentifier) {
119     return new SkyKey(SkyFunctions.PACKAGE_LOOKUP, pkgIdentifier);
120   }
121 
122   private static class SuccessfulPackageLookupValue extends PackageLookupValue {
123 
124     private final Path root;
125 
126     private SuccessfulPackageLookupValue(Path root) {
127       this.root = root;
128     }
129 
130     @Override
131     public boolean packageExists() {
132       return true;
133     }
134 
135     @Override
136     public Path getRoot() {
137       return root;
138     }
139 
140     @Override
141     ErrorReason getErrorReason() {
142       throw new IllegalStateException();
143     }
144 
145     @Override
146     String getErrorMsg() {
147       throw new IllegalStateException();
148     }
149 
150     @Override
151     public boolean equals(Object obj) {
152       if (!(obj instanceof SuccessfulPackageLookupValue)) {
153         return false;
154       }
155       SuccessfulPackageLookupValue other = (SuccessfulPackageLookupValue) obj;
156       return root.equals(other.root);
157     }
158 
159     @Override
160     public int hashCode() {
161       return root.hashCode();
162     }
163   }
164 
165   /**
166    * A package under external/ that has a BUILD file that is not under external/.
167    *
168    * <p>This is kind of a hack to get around our assumption that external/ is immutable.</p>
169    */
170   private static class OverlaidPackageLookupValue extends SuccessfulPackageLookupValue {
171 
172     private final Optional<FileValue> overlaidBuildFile;
173 
174     public OverlaidPackageLookupValue(Path root, Optional<FileValue> overlaidBuildFile) {
175       super(root);
176       this.overlaidBuildFile = overlaidBuildFile;
177     }
178 
179     @Override
180     public boolean equals(Object obj) {
181       if (!(obj instanceof OverlaidPackageLookupValue)) {
182         return false;
183       }
184       OverlaidPackageLookupValue other = (OverlaidPackageLookupValue) obj;
185       return super.equals(other) && overlaidBuildFile.equals(other.overlaidBuildFile);
186     }
187 
188     @Override
189     public int hashCode() {
190       return Objects.hash(super.hashCode(), overlaidBuildFile);
191     }
192   }
193 
194   // TODO(kchodorow): fix these semantics.  This class should not exist, WORKSPACE lookup should
195   // just return success/failure like a "normal" package.
196   private static class WorkspacePackageLookupValue extends SuccessfulPackageLookupValue {
197 
198     private WorkspacePackageLookupValue(Path root) {
199       super(root);
200     }
201 
202     // TODO(kchodorow): get rid of this, the semantics are wrong (successful package lookup should
203     // mean the package exists).
204     @Override
205     public boolean packageExists() {
206       return getRoot().exists();
207     }
208 
209     @Override
210     public boolean isExternalPackage() {
211       return true;
212     }
213   }
214 
215   private abstract static class UnsuccessfulPackageLookupValue extends PackageLookupValue {
216 
217     @Override
218     public boolean packageExists() {
219       return false;
220     }
221 
222     @Override
223     public Path getRoot() {
224       throw new IllegalStateException();
225     }
226   }

228   private static class NoBuildFilePackageLookupValue extends UnsuccessfulPackageLookupValue {
229 
230     public static final NoBuildFilePackageLookupValue INSTANCE =
231         new NoBuildFilePackageLookupValue();
232 
233     private NoBuildFilePackageLookupValue() {
234     }
235 
236     @Override
237     ErrorReason getErrorReason() {
238       return ErrorReason.NO_BUILD_FILE;
239     }
240 
241     @Override
242     String getErrorMsg() {
243       return "BUILD file not found on package path";
244     }
245   }
