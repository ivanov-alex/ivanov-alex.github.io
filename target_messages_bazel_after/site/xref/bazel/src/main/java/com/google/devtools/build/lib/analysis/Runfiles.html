51  /**
52   * An object that encapsulates runfiles. Conceptually, the runfiles are a map of paths to files,
53   * forming a symlink tree.
54   *
55   * <p>In order to reduce memory consumption, this map is not explicitly stored here, but instead as
56   * a combination of four parts: artifacts placed at their root-relative paths, source tree symlinks,
57   * root symlinks (outside of the source tree), and artifacts included as parts of "pruning
58   * manifests" (see {@link PruningManifest}).
59   */
60  @Immutable
61  @SkylarkModule(name = "runfiles", doc = "An interface for a set of runfiles.")
62  public final class Runfiles {
63    private static final Function<SymlinkEntry, Artifact> TO_ARTIFACT =
64        new Function<SymlinkEntry, Artifact>() {
65          @Override
66          public Artifact apply(SymlinkEntry input) {
67            return input.getArtifact();
68          }
69        };

73          @Override
74          public Iterable<PathFragment> getExtraPaths(Set<PathFragment> manifestPaths) {
75            return ImmutableList.of();

79    /**
80     * An entry in the runfiles map.
81     */
82    //
83    // O intrepid fixer or bugs and implementor of features, dare not to add a .equals() method
84    // to this class, lest you condemn yourself, or a fellow other developer to spending two
85    // delightful hours in a fancy hotel on a Chromebook that is utterly unsuitable for Java
86    // development to figure out what went wrong, just like I just did.
87    //
88    // The semantics of the symlinks nested set dictates that later entries overwrite earlier
89    // ones. However, the semantics of nested sets dictate that if there are duplicate entries, they
90    // are only returned once in the iterator.
91    //
92    // These two things, innocent when taken alone, result in the effect that when there are three
93    // entries for the same path, the first one and the last one the same, and the middle one
94    // different, the *middle* one will take effect: the middle one overrides the first one, and the
95    // first one prevents the last one from appearing on the iterator.
96    //
97    // The lack of a .equals() method prevents this by making the first entry in the above case not
98    // equals to the third one if they are not the same instance (which they almost never are)
99    //
100   // Goodnight, prince(ss)?, and sweet dreams.
101   private static final class SymlinkEntry {
102     private final PathFragment path;
103     private final Artifact artifact;
104 
105     private SymlinkEntry(PathFragment path, Artifact artifact) {
106       this.path = path;
107       this.artifact = artifact;
108     }
109 
110     public PathFragment getPath() {
111       return path;
112     }
113 
114     public Artifact getArtifact() {
115       return artifact;
116     }
117   }

168   /**
169    * Defines a set of artifacts that may or may not be included in the runfiles directory and
170    * a manifest file that makes that determination. These are applied on top of any artifacts
171    * specified in {@link #unconditionalArtifacts}.
172    *
173    * <p>The incentive behind this is to enable execution-phase "pruning" of runfiles. Anything
174    * set in unconditionalArtifacts is hard-set in Blaze's analysis phase, and thus unchangeable in
175    * response to execution phase results. This isn't always convenient. For example, say we have an
176    * action that consumes a set of "possible" runtime dependencies for a source file, parses that
177    * file for "import a.b.c" statements, and outputs a manifest of the actual dependencies that are
178    * referenced and thus really needed. This can reduce the size of the runfiles set, but we can't
179    * use this information until the manifest output is available.
180    *
181    * <p>Only artifacts present in the candidate set AND the manifest output make it into the
182    * runfiles tree. The candidate set requirement guarantees that analysis-time dependencies are a
183    * superset of the pruned dependencies, so undeclared inclusions (which can break build
184    * correctness) aren't possible.
185    */
186   public static class PruningManifest {
187     private final NestedSet<Artifact> candidateRunfiles;
188     private final Artifact manifestFile;
189 
190     /**
191      * Creates a new pruning manifest.
192      *
193      * @param candidateRunfiles set of possible artifacts that the manifest file may reference
194      * @param manifestFile the manifest file, expected to be a newline-separated list of
195      *     source tree root-relative paths (i.e. "my/package/myfile.txt"). Anything that can't be
196      *     resolved back to an entry in candidateRunfiles is ignored and will *not* make it into
197      *     the runfiles tree.
198      */
199     public PruningManifest(NestedSet<Artifact> candidateRunfiles, Artifact manifestFile) {
200       this.candidateRunfiles = candidateRunfiles;
201       this.manifestFile = manifestFile;
202     }
203 
204     public NestedSet<Artifact> getCandidateRunfiles() {
205       return candidateRunfiles;
206     }
207 
208     public Artifact getManifestFile() {
209       return manifestFile;
210     }
211   }

297   /**
298    * @param eventHandler Used for throwing an error if we have an obscuring runlink.
299    *                 May be null, in which case obscuring symlinks are silently discarded.
300    * @param location Location for reporter. Ignored if reporter is null.
301    * @param workingManifest Manifest to be checked for obscuring symlinks.
302    * @return map of source file names mapped to their location on disk.
303    */
304   @VisibleForTesting
305   static Map<PathFragment, Artifact> filterListForObscuringSymlinks(
306       EventHandler eventHandler, Location location, Map<PathFragment, Artifact> workingManifest) {
307     Map<PathFragment, Artifact> newManifest = new HashMap<>();
308 
309     outer:
310     for (Iterator<Entry<PathFragment, Artifact>> i = workingManifest.entrySet().iterator();
311          i.hasNext(); ) {
312       Entry<PathFragment, Artifact> entry = i.next();
313       PathFragment source = entry.getKey();
314       Artifact symlink = entry.getValue();
315       // drop nested entries; warn if this changes anything
316       int n = source.segmentCount();
317       for (int j = 1; j < n; ++j) {
318         PathFragment prefix = source.subFragment(0, n - j);
319         Artifact ancestor = workingManifest.get(prefix);
320         if (ancestor != null) {
321           // This is an obscuring symlink, so just drop it and move on if there's no reporter.
322           if (eventHandler == null) {
323             continue outer;
324           }
325           PathFragment suffix = source.subFragment(n - j, n);
326           Path viaAncestor = ancestor.getPath().getRelative(suffix);
327           Path expected = symlink.getPath();
328           if (!viaAncestor.equals(expected)) {
329             eventHandler.handle(Event.warn(location, "runfiles symlink " + source + " -> "
330                 + expected + " obscured by " + prefix + " -> " + ancestor.getPath()));
331           }
332           continue outer;
333         }
334       }
335       newManifest.put(entry.getKey(), entry.getValue());
336     }
337     return newManifest;
338   }

340   /**
341    * Returns the symlinks as a map from PathFragment to Artifact.
342    *
343    * @param eventHandler Used for throwing an error if we have an obscuring runlink within the
344    *    normal source tree entries. May be null, in which case obscuring symlinks are silently
345    *    discarded.
346    * @param location Location for eventHandler warnings. Ignored if eventHandler is null.
347    * @return Map<PathFragment, Artifact> path fragment to artifact, of normal source tree entries
348    *    and elements that live outside the source tree. Null values represent empty input files.
349    */
350   public Map<PathFragment, Artifact> getRunfilesInputs(EventHandler eventHandler,
351       Location location) throws IOException {
352     Map<PathFragment, Artifact> manifest = getSymlinksAsMap();
353     // Add unconditional artifacts (committed to inclusion on construction of runfiles).
354     for (Artifact artifact : getUnconditionalArtifactsWithoutMiddlemen()) {
355       manifest.put(artifact.getRootRelativePath(), artifact);
356     }
357 
358     // Add conditional artifacts (only included if they appear in a pruning manifest).
359     for (Runfiles.PruningManifest pruningManifest : getPruningManifests()) {
360       // This map helps us convert from source tree root-relative paths back to artifacts.
361       Map<String, Artifact> allowedRunfiles = new HashMap<>();
362       for (Artifact artifact : pruningManifest.getCandidateRunfiles()) {
363         allowedRunfiles.put(artifact.getRootRelativePath().getPathString(), artifact);
364       }
365       BufferedReader reader = new BufferedReader(
366           new InputStreamReader(pruningManifest.getManifestFile().getPath().getInputStream()));
367       String line;
368       while ((line = reader.readLine()) != null) {
369         Artifact artifact = allowedRunfiles.get(line);
370         if (artifact != null) {
371           manifest.put(artifact.getRootRelativePath(), artifact);
372         }
373       }
374     }
375 
376     manifest = filterListForObscuringSymlinks(eventHandler, location, manifest);
377 
378     // TODO(bazel-team): Create /dev/null-like Artifact to avoid nulls?
379     for (PathFragment extraPath : emptyFilesSupplier.getExtraPaths(manifest.keySet())) {
380       manifest.put(extraPath, null);
381     }
382 
383     PathFragment path = new PathFragment(suffix);
384     Map<PathFragment, Artifact> result = new HashMap<>();
385     for (Map.Entry<PathFragment, Artifact> entry : manifest.entrySet()) {
386       result.put(path.getRelative(entry.getKey()), entry.getValue());
387     }
388 
389     // Finally add symlinks outside the source tree on top of everything else.
390     for (Map.Entry<PathFragment, Artifact> entry : getRootSymlinksAsMap().entrySet()) {
391       PathFragment mappedPath = entry.getKey();
392       Artifact mappedArtifact = entry.getValue();
393       if (result.put(mappedPath, mappedArtifact) != null) {
394         // Emit warning if we overwrote something and we're capable of emitting warnings.
395         if (eventHandler != null) {
396           eventHandler.handle(Event.warn(location, "overwrote " + mappedPath + " symlink mapping "
397               + "with root symlink to " + mappedArtifact));
398         }
399       }
400     }
401 
402     return result;
403   }

419   /**
420    * Returns the unified map of path fragments to artifacts, taking both artifacts and symlinks into
421    * account.
422    */
423   public Map<PathFragment, Artifact> asMapWithoutRootSymlinks() {
424     Map<PathFragment, Artifact> result = entriesToMap(symlinks);
425     // If multiple artifacts have the same root-relative path, the last one in the list will win.
426     // That is because the runfiles tree cannot contain the same artifact for different
427     // configurations, because it only uses root-relative paths.
428     for (Artifact artifact : Iterables.filter(unconditionalArtifacts, Artifact.MIDDLEMAN_FILTER)) {
429       result.put(artifact.getRootRelativePath(), artifact);
430     }
431     return result;
432   }

448   /**
449    * Returns the unified map of path fragments to artifacts, taking into account artifacts,
450    * symlinks, and pruning manifest candidates. The returned set is guaranteed to be a (not
451    * necessarily strict) superset of the actual runfiles tree created at execution time.
452    */
453   public NestedSet<Artifact> getAllArtifacts() {
454     if (isEmpty()) {
455       return NestedSetBuilder.emptySet(Order.STABLE_ORDER);
456     }
457     NestedSetBuilder<Artifact> allArtifacts = NestedSetBuilder.stableOrder();
458     allArtifacts
459         .addTransitive(unconditionalArtifacts)
460         .addAll(Iterables.transform(symlinks, TO_ARTIFACT))
461         .addAll(Iterables.transform(rootSymlinks, TO_ARTIFACT));
462     for (PruningManifest manifest : getPruningManifests()) {
463       allArtifacts.addTransitive(manifest.getCandidateRunfiles());
464     }
465     return allArtifacts.build();
466   }

476   private static Map<PathFragment, Artifact> entriesToMap(Iterable<SymlinkEntry> entrySet) {
477     Map<PathFragment, Artifact> map = new LinkedHashMap<>();
478     for (SymlinkEntry entry : entrySet) {
479       map.put(entry.getPath(), entry.getArtifact());
480     }
481     return map;
482   }

640     /**
641      * Adds the runfiles for a particular target and visits the transitive closure of "srcs",
642      * "deps" and "data", collecting all of their respective runfiles.
643      */
644     public Builder addRunfiles(RuleContext ruleContext,
645         Function<TransitiveInfoCollection, Runfiles> mapping) {
646       Preconditions.checkNotNull(mapping);
647       Preconditions.checkNotNull(ruleContext);
648       suffix = ruleContext.getWorkspaceName();
649       addDataDeps(ruleContext);
650       addNonDataDeps(ruleContext, mapping);
651       return this;
652     }

654     /**
655      * Adds the files specified by a mapping from the transitive info collection to the runfiles.
656      *
657      * <p>Dependencies in {@code srcs} and {@code deps} are considered.
658      */
659     public Builder add(RuleContext ruleContext,
660         Function<TransitiveInfoCollection, Runfiles> mapping) {
661       Preconditions.checkNotNull(ruleContext);
662       Preconditions.checkNotNull(mapping);
663       suffix = ruleContext.getWorkspaceName();
664       for (TransitiveInfoCollection dep : getNonDataDeps(ruleContext)) {
665         Runfiles runfiles = mapping.apply(dep);
666         if (runfiles != null) {
667           merge(runfiles);
668         }
669       }
670 
671       return this;
672     }
673 
674     /**
675      * Collects runfiles from data dependencies of a target.
676      */
677     public Builder addDataDeps(RuleContext ruleContext) {
678       suffix = ruleContext.getWorkspaceName();
679       addTargets(getPrerequisites(ruleContext, "data", Mode.DATA), RunfilesProvider.DATA_RUNFILES);
680       return this;
681     }

683     /**
684      * Collects runfiles from "srcs" and "deps" of a target.
685      */
686     public Builder addNonDataDeps(RuleContext ruleContext,
687         Function<TransitiveInfoCollection, Runfiles> mapping) {
688       suffix = ruleContext.getWorkspaceName();
689       for (TransitiveInfoCollection target : getNonDataDeps(ruleContext)) {
690         addTargetExceptFileTargets(target, mapping);
691       }
692       return this;
693     }

695     public Builder addTargets(Iterable<? extends TransitiveInfoCollection> targets,
696         Function<TransitiveInfoCollection, Runfiles> mapping) {
697       for (TransitiveInfoCollection target : targets) {
698         addTarget(target, mapping);
699       }
700       return this;
701     }

703     public Builder addTarget(TransitiveInfoCollection target,
704         Function<TransitiveInfoCollection, Runfiles> mapping) {
705       return addTargetIncludingFileTargets(target, mapping);
706     }

708     private Builder addTargetExceptFileTargets(TransitiveInfoCollection target,
709         Function<TransitiveInfoCollection, Runfiles> mapping) {
710       Runfiles runfiles = mapping.apply(target);
711       if (runfiles != null) {
712         merge(runfiles);
713       }
714 
715       return this;
716     }

718     private Builder addTargetIncludingFileTargets(TransitiveInfoCollection target,
719         Function<TransitiveInfoCollection, Runfiles> mapping) {
720       if (target.getProvider(RunfilesProvider.class) == null
721           && mapping == RunfilesProvider.DATA_RUNFILES) {
722         // RuleConfiguredTarget implements RunfilesProvider, so this will only be called on
723         // FileConfiguredTarget instances.
724         // TODO(bazel-team): This is a terrible hack. We should be able to make this go away
725         // by implementing RunfilesProvider on FileConfiguredTarget. We'd need to be mindful
726         // of the memory use, though, since we have a whole lot of FileConfiguredTarget instances.
727         addTransitiveArtifacts(target.getProvider(FileProvider.class).getFilesToBuild());
728         return this;
729       }
730 
731       return addTargetExceptFileTargets(target, mapping);
732     }

768     /**
769      * Add the other {@link Runfiles} object transitively, with the option to include or exclude
770      * pruning manifests in the merge.
771      */
772     private Builder merge(Runfiles runfiles, boolean includePruningManifests) {
773       artifactsBuilder.addTransitive(runfiles.getUnconditionalArtifacts());
774       symlinksBuilder.addTransitive(runfiles.getSymlinks());
775       rootSymlinksBuilder.addTransitive(runfiles.getRootSymlinks());
776       if (suffix == null) {
777         suffix = runfiles.suffix;
778       }
779       if (includePruningManifests) {
780         pruningManifestsBuilder.addTransitive(runfiles.getPruningManifests());
781       }
782       if (emptyFilesSupplier == DUMMY_EMPTY_FILES_SUPPLIER) {
783         emptyFilesSupplier = runfiles.getEmptyFilesProvider();
784       } else {
785         EmptyFilesSupplier otherSupplier = runfiles.getEmptyFilesProvider();
786         Preconditions.checkState((otherSupplier == DUMMY_EMPTY_FILES_SUPPLIER)
787           || emptyFilesSupplier.equals(otherSupplier));
788       }
789       return this;
790     }

792     private static Iterable<TransitiveInfoCollection> getNonDataDeps(RuleContext ruleContext) {
793       return Iterables.concat(
794           // TODO(bazel-team): This line shouldn't be here. Removing it requires that no rules have
795           // dependent rules in srcs (except for filegroups and such), but always in deps.
796           // TODO(bazel-team): DONT_CHECK is not optimal here. Rules that use split configs need to
797           // be changed not to call into here.
798           getPrerequisites(ruleContext, "srcs", Mode.DONT_CHECK),
799           getPrerequisites(ruleContext, "deps", Mode.DONT_CHECK));
800     }

802     /**
803      * For the specified attribute "attributeName" (which must be of type list(label)), resolves all
804      * the labels into ConfiguredTargets (for the same configuration as this one) and returns them
805      * as a list.
806      *
807      * <p>If the rule does not have the specified attribute, returns the empty list.
808      */
809     private static Iterable<? extends TransitiveInfoCollection> getPrerequisites(
810         RuleContext ruleContext, String attributeName, Mode mode) {
811       if (ruleContext.getRule().isAttrDefined(attributeName, Type.LABEL_LIST)) {
812         return ruleContext.getPrerequisites(attributeName, mode);
813       } else {
814         return Collections.emptyList();
815       }
816     }
