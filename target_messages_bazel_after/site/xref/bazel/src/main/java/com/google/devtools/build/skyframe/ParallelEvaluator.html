67  /**
68   * Evaluates a set of given functions ({@code SkyFunction}s) with arguments ({@code SkyKey}s).
69   * Cycles are not allowed and are detected during the traversal.
70   *
71   * <p>This class implements multi-threaded evaluation. This is a fairly complex process that has
72   * strong consistency requirements between the {@link ProcessableGraph}, the nodes in the graph of
73   * type {@link NodeEntry}, the work queue, and the set of in-flight nodes.
74   *
75   * <p>The basic invariants are:
76   *
77   * <p>A node can be in one of three states: ready, waiting, and done. A node is ready if and only
78   * if all of its dependencies have been signaled. A node is done if it has a value. It is waiting
79   * if not all of its dependencies have been signaled.
80   *
81   * <p>A node must be in the work queue if and only if it is ready. It is an error for a node to be
82   * in the work queue twice at the same time.
83   *
84   * <p>A node is considered in-flight if it has been created, and is not done yet. In case of an
85   * interrupt, the work queue is discarded, and the in-flight set is used to remove partially
86   * computed values.
87   *
88   * <p>Each evaluation of the graph takes place at a "version," which is currently given by a
89   * non-negative {@code long}. The version can also be thought of as an "mtime." Each node in the
90   * graph has a version, which is the last version at which its value changed. This version data is
91   * used to avoid unnecessary re-evaluation of values. If a node is re-evaluated and found to have
92   * the same data as before, its version (mtime) remains the same. If all of a node's children's
93   * have the same version as before, its re-evaluation can be skipped.
94   *
95   * <p>This class is not intended for direct use, and is only exposed as public for use in
96   * evaluation implementations outside of this package.
97   */

218     private NestedSet<TaggedEvents> buildEvents(boolean missingChildren) {
219       // Aggregate the nested set of events from the direct deps, also adding the events from
220       // building this value.
221       NestedSetBuilder<TaggedEvents> eventBuilder = NestedSetBuilder.stableOrder();
222       ImmutableList<Event> events = eventHandler.getEvents();
223       if (!events.isEmpty()) {
224         eventBuilder.add(new TaggedEvents(getTagFromKey(), events));
225       }
226       for (SkyKey dep : graph.get(skyKey).getTemporaryDirectDeps()) {
227         ValueWithMetadata value = getValueMaybeFromError(dep, bubbleErrorInfo);
228         if (value != null) {
229           eventBuilder.addTransitive(value.getTransitiveEvents());
230         } else {
231           Preconditions.checkState(missingChildren, "", dep, skyKey);
232         }
233       }
234       return eventBuilder.build();
235     }

280     @Override
281     protected ImmutableMap<SkyKey, ValueOrUntypedException> getValueOrUntypedExceptions(
282         Iterable<SkyKey> depKeys) {
283       checkActive();
284       Set<SkyKey> keys = new LinkedHashSet<>();
285       for (SkyKey depKey : depKeys) {
286         // Canonicalize SkyKeys to save memory.
287         keys.add(KEY_CANONICALIZER.intern(depKey));
288       }
289       depKeys = keys;
290       Map<SkyKey, ValueWithMetadata> values = getValuesMaybeFromError(depKeys, bubbleErrorInfo);
291       ImmutableMap.Builder<SkyKey, ValueOrUntypedException> builder = ImmutableMap.builder();
292       for (SkyKey depKey : depKeys) {
293         ValueWithMetadata value = values.get(depKey);
294         if (value == null) {
295           // If this entry is not yet done then (optionally) record the missing dependency and
296           // return null.
297           valuesMissing = true;
298           if (bubbleErrorInfo != null) {
299             // Values being built just for their errors don't get to request new children.
300             builder.put(depKey, ValueOrExceptionUtils.ofNull());
301             continue;
302           }
303           Preconditions.checkState(!directDeps.contains(depKey), "%s %s %s", skyKey, depKey,
304               value);
305           addDep(depKey);
306           valuesMissing = true;
307           builder.put(depKey, ValueOrExceptionUtils.ofNull());
308           continue;
309         }
310 
311         if (!directDeps.contains(depKey)) {
312           // If this child is done, we will return it, but also record that it was newly requested
313           // so that the dependency can be properly registered in the graph.
314           addDep(depKey);
315         }
316 
317         replayingNestedSetEventVisitor.visit(value.getTransitiveEvents());
318         ErrorInfo errorInfo = value.getErrorInfo();
319 
320         if (errorInfo != null) {
321           childErrorInfos.add(errorInfo);
322         }
323 
324         if (value.getValue() != null && (keepGoing || errorInfo == null)) {
325           // If the dep did compute a value, it is given to the caller if we are in keepGoing mode
326           // or if we are in noKeepGoingMode and there were no errors computing it.
327           builder.put(depKey, ValueOrExceptionUtils.ofValueUntyped(value.getValue()));
328           continue;
329         }
330 
331         // There was an error building the value, which we will either report by throwing an
332         // exception or insulate the caller from by returning null.
333         Preconditions.checkNotNull(errorInfo, "%s %s %s", skyKey, depKey, value);
334 
335         if (!keepGoing && errorInfo.getException() != null && bubbleErrorInfo == null) {
336           // Child errors should not be propagated in noKeepGoing mode (except during error
337           // bubbling). Instead we should fail fast.
338 
339           // We arbitrarily record the first child error.
340           if (depErrorKey == null) {
341             depErrorKey = depKey;
342           }
343           valuesMissing = true;
344           builder.put(depKey, ValueOrExceptionUtils.ofNull());
345           continue;
346         }
347 
348         if (bubbleErrorInfo != null) {
349           // Set interrupted status, to try to prevent the calling SkyFunction from doing anything
350           // fancy after this. SkyFunctions executed during error bubbling are supposed to
351           // (quickly) rethrow errors or return a value/null (but there's currently no way to
352           // enforce this).
353           Thread.currentThread().interrupt();
354         }
355         if (errorInfo.getException() != null) {
356           // Give builder a chance to handle this exception.
357           Exception e = errorInfo.getException();
358           builder.put(depKey, ValueOrExceptionUtils.ofExn(e));
359           continue;
360         }
361         // In a cycle.
362         Preconditions.checkState(!Iterables.isEmpty(errorInfo.getCycleInfo()), "%s %s %s %s",
363             skyKey, depKey, errorInfo, value);
364         valuesMissing = true;
365         builder.put(depKey, ValueOrExceptionUtils.ofNull());
366       }
367       return builder.build();
368     }

408     /**
409      * Apply the change to the graph (mostly) atomically and signal all nodes that are waiting for
410      * this node to complete. Adding nodes and signaling is not atomic, but may need to be changed
411      * for interruptibility.
412      *
413      * <p>Parents are only enqueued if {@code enqueueParents} holds. Parents should be enqueued
414      * unless (1) this node is being built after the main evaluation has aborted, or (2) this node
415      * is being built with --nokeep_going, and so we are about to shut down the main evaluation
416      * anyway.
417      *
418      * <p>The node entry is informed if the node's value and error are definitive via the flag
419      * {@code completeValue}.
420      */
421     void commit(boolean enqueueParents) {
422       NodeEntry primaryEntry = Preconditions.checkNotNull(graph.get(skyKey), skyKey);
423       // Construct the definitive error info, if there is one.
424       finalizeErrorInfo();
425 
426       // We have the following implications:
427       // errorInfo == null => value != null => enqueueParents.
428       // All these implications are strict:
429       // (1) errorInfo != null && value != null happens for values with recoverable errors.
430       // (2) value == null && enqueueParents happens for values that are found to have errors
431       // during a --keep_going build.
432 
433       NestedSet<TaggedEvents> events = buildEvents(/*missingChildren=*/false);
434       Version valueVersion;
435       SkyValue valueWithMetadata;
436       if (value == null) {
437         Preconditions.checkNotNull(errorInfo, "%s %s", skyKey, primaryEntry);
438         valueWithMetadata = ValueWithMetadata.error(errorInfo, events);
439       } else {
440         // We must be enqueueing parents if we have a value.
441         Preconditions.checkState(enqueueParents, "%s %s", skyKey, primaryEntry);
442         valueWithMetadata = ValueWithMetadata.normal(value, errorInfo, events);
443       }
444       // If this entry is dirty, setValue may not actually change it, if it determines that
445       // the data being written now is the same as the data already present in the entry.
446       // We could consider using max(childVersions) here instead of graphVersion. When full
447       // versioning is implemented, this would allow evaluation at a version between
448       // max(childVersions) and graphVersion to re-use this result.
449       Set<SkyKey> reverseDeps = primaryEntry.setValue(valueWithMetadata, graphVersion);
450       // Note that if this update didn't actually change the value entry, this version may not
451       // be the graph version.
452       valueVersion = primaryEntry.getVersion();
453       Preconditions.checkState(valueVersion.atMost(graphVersion),
454           "%s should be at most %s in the version partial ordering",
455           valueVersion, graphVersion);
456       if (progressReceiver != null) {
457         // Tell the receiver that this value was built. If valueVersion.equals(graphVersion), it
458         // was evaluated this run, and so was changed. Otherwise, it is less than graphVersion,
459         // by the Preconditions check above, and was not actually changed this run -- when it was
460         // written above, its version stayed below this update's version, so its value remains the
461         // same as before.
462         progressReceiver.evaluated(skyKey, value,
463             valueVersion.equals(graphVersion) ? EvaluationState.BUILT : EvaluationState.CLEAN);
464       }
465       signalValuesAndEnqueueIfReady(enqueueParents ? visitor : null, reverseDeps, valueVersion);
466 
467       visitor.notifyDone(skyKey);
468       replayingNestedSetEventVisitor.visit(events);
469     }

516     public void enqueueEvaluation(final SkyKey key) {
517       // We unconditionally add the key to the set of in-flight nodes because even if evaluation is
518       // never scheduled we still want to remove the previously created NodeEntry from the graph.
519       // Otherwise we would leave the graph in a weird state (wasteful garbage in the best case and
520       // inconsistent in the worst case).
521       boolean newlyEnqueued = inflightNodes.add(key);
522       // All nodes enqueued for evaluation will be either verified clean, re-evaluated, or cleaned
523       // up after being in-flight when an error happens in nokeep_going mode or in the event of an
524       // interrupt. In any of these cases, they won't be dirty anymore.
525       if (newlyEnqueued) {
526         dirtyKeyTracker.notDirty(key);
527       }
528       if (preventNewEvaluations.get()) {
529         return;
530       }
531       if (newlyEnqueued && progressReceiver != null) {
532         progressReceiver.enqueueing(key);
533       }
534       enqueue(new Evaluate(this, key));
535     }
536 
537     /**
538      * Stop any new evaluations from being enqueued. Returns whether this was the first thread to
539      * request a halt. If true, this thread should proceed to throw an exception. If false, another
540      * thread already requested a halt and will throw an exception, and so this thread can simply
541      * end.
542      */
543     boolean preventNewEvaluations() {
544       return preventNewEvaluations.compareAndSet(false, true);
545     }
546 
547     void notifyDone(SkyKey key) {
548       inflightNodes.remove(key);
549     }

551     private boolean isInflight(SkyKey key) {
552       return inflightNodes.contains(key);

560     private final ValueVisitor visitor;
561     /** The name of the value to be evaluated. */
562     private final SkyKey skyKey;

564     private Evaluate(ValueVisitor visitor, SkyKey skyKey) {
565       this.visitor = visitor;
566       this.skyKey = skyKey;

569     private void enqueueChild(SkyKey skyKey, NodeEntry entry, SkyKey child) {
570       Preconditions.checkState(!entry.isDone(), "%s %s", skyKey, entry);
571       Preconditions.checkState(!ErrorTransienceValue.key().equals(child),
572           "%s cannot request ErrorTransienceValue as a dep: %s", skyKey, entry);
573 
574       NodeEntry depEntry = graph.createIfAbsent(child);
575       switch (depEntry.addReverseDepAndCheckIfDone(skyKey)) {
576         case DONE :
577           if (entry.signalDep(depEntry.getVersion())) {
578             // This can only happen if there are no more children to be added.
579             visitor.enqueueEvaluation(skyKey);
580           }
581           break;
582         case ADDED_DEP :
583           break;
584         case NEEDS_SCHEDULING :
585           visitor.enqueueEvaluation(child);
586           break;
587       }
588     }

590     /**
591      * Returns true if this depGroup consists of the error transience value and the error transience
592      * value is newer than the entry, meaning that the entry must be re-evaluated.
593      */
594     private boolean invalidatedByErrorTransience(Collection<SkyKey> depGroup, NodeEntry entry) {
595       return depGroup.size() == 1
596           && depGroup.contains(ErrorTransienceValue.key())
597           && !graph.get(ErrorTransienceValue.key()).getVersion().atMost(entry.getVersion());
598     }

600     @Override
601     public void run() {
602       NodeEntry state = graph.get(skyKey);
603       Preconditions.checkNotNull(state, "%s %s", skyKey, state);
604       Preconditions.checkState(state.isReady(), "%s %s", skyKey, state);
605 
606       if (state.isDirty()) {
607         switch (state.getDirtyState()) {
608           case CHECK_DEPENDENCIES:
609             // Evaluating a dirty node for the first time, and checking its children to see if any
610             // of them have changed. Note that there must be dirty children for this to happen.
611 
612             // Check the children group by group -- we don't want to evaluate a value that is no
613             // longer needed because an earlier dependency changed. For example, //foo:foo depends
614             // on target //bar:bar and is built. Then foo/BUILD is modified to remove the dependence
615             // on bar, and bar/BUILD is deleted. Reloading //bar:bar would incorrectly throw an
616             // exception. To avoid this, we must reload foo/BUILD first, at which point we will
617             // discover that it has changed, and re-evaluate target //foo:foo from scratch.
618             // On the other hand, when an action requests all of its inputs, we can safely check all
619             // of them in parallel on a subsequent build. So we allow checking an entire group in
620             // parallel here, if the node builder requested a group last build.
621             Collection<SkyKey> directDepsToCheck = state.getNextDirtyDirectDeps();
622 
623             if (invalidatedByErrorTransience(directDepsToCheck, state)) {
624               // If this dep is the ErrorTransienceValue and the ErrorTransienceValue has been
625               // updated then we need to force a rebuild. We would like to just signal the entry as
626               // usual, but we can't, because then the ErrorTransienceValue would remain as a dep,
627               // which would be incorrect if, for instance, the value re-evaluated to a non-error.
628               state.forceRebuild();
629               break; // Fall through to re-evaluation.
630             }
631             if (!keepGoing) {
632               // This check ensures that we maintain the invariant that if a node with an error is
633               // reached during a no-keep-going build, none of its currently building parents
634               // finishes building. If the child isn't done building yet, it will detect on its own
635               // that it has an error (see the VERIFIED_CLEAN case below). On the other hand, if it
636               // is done, then it is the parent's responsibility to notice that, which we do here.
637               // We check the deps for errors so that we don't continue building this node if it has
638               // a child error.
639               for (Map.Entry<SkyKey, NodeEntry> entry :
640                   graph.getBatch(directDepsToCheck).entrySet()) {
641                 if (entry.getValue().isDone() && entry.getValue().getErrorInfo() != null) {
642                   // If any child has an error, we arbitrarily add a dep on the first one (needed
643                   // for error bubbling) and throw an exception coming from it.
644                   if (!visitor.preventNewEvaluations()) {
645                     // An error was already thrown in the evaluator. Don't do anything here.
646                     return;
647                   }
648                   SkyKey errorKey = entry.getKey();
649                   NodeEntry errorEntry = entry.getValue();
650                   state.addTemporaryDirectDeps(
651                       GroupedListHelper.create(ImmutableList.of(errorKey)));
652                   DependencyState errorState = entry.getValue().addReverseDepAndCheckIfDone(skyKey);
653                   Preconditions.checkState(errorState == DependencyState.DONE, "%s %s %s %s",
654                       skyKey, state, errorKey, errorEntry);
655                   throw SchedulerException.ofError(errorEntry.getErrorInfo(), entry.getKey());
656                 }
657               }
658             }
659             // If this isn't the error transience value, it is safe to add these deps back to the
660             // node -- even if one of them has changed, the contract of pruning is that the node
661             // will request these deps again when it rebuilds. We must add these deps before
662             // enqueuing them, so that the node knows that it depends on them.
663             state.addTemporaryDirectDeps(GroupedListHelper.create(directDepsToCheck));
664             for (SkyKey directDep : directDepsToCheck) {
665               enqueueChild(skyKey, state, directDep);
666             }
667             return;
668           case VERIFIED_CLEAN:
669             // No child has a changed value. This node can be marked done and its parents signaled
670             // without any re-evaluation.
671             visitor.notifyDone(skyKey);
672             Set<SkyKey> reverseDeps = state.markClean();
673             SkyValue value = state.getValue();
674             if (progressReceiver != null) {
675               // Tell the receiver that the value was not actually changed this run.
676               progressReceiver.evaluated(skyKey, value, EvaluationState.CLEAN);
677             }
678             if (!keepGoing && state.getErrorInfo() != null) {
679               if (!visitor.preventNewEvaluations()) {
680                 return;
681               }
682               throw SchedulerException.ofError(state.getErrorInfo(), skyKey);
683             }
684             signalValuesAndEnqueueIfReady(visitor, reverseDeps, state.getVersion());
685             return;
686           case REBUILDING:
687             // Nothing to be done if we are already rebuilding.
688         }
689       }
690 
691       // TODO(bazel-team): Once deps are requested in a deterministic order within a group, or the
692       // framework is resilient to rearranging group order, change this so that
693       // SkyFunctionEnvironment "follows along" as the node builder runs, iterating through the
694       // direct deps that were requested on a previous run. This would allow us to avoid the
695       // conversion of the direct deps into a set.
696       Set<SkyKey> directDeps = state.getTemporaryDirectDeps();
697       Preconditions.checkState(!directDeps.contains(ErrorTransienceValue.key()),
698           "%s cannot have a dep on ErrorTransienceValue during building: %s", skyKey, state);
699       // Get the corresponding SkyFunction and call it on this value.
700       SkyFunctionEnvironment env = new SkyFunctionEnvironment(skyKey, directDeps, visitor);
701       SkyFunctionName functionName = skyKey.functionName();
702       SkyFunction factory = skyFunctions.get(functionName);
703       Preconditions.checkState(factory != null, "%s %s", functionName, state);
704 
705       SkyValue value = null;
706       Profiler.instance().startTask(ProfilerTask.SKYFUNCTION, skyKey);
707       try {
708         // TODO(bazel-team): count how many of these calls returns null vs. non-null
709         value = factory.compute(skyKey, env);
710       } catch (final SkyFunctionException builderException) {
711         ReifiedSkyFunctionException reifiedBuilderException =
712             new ReifiedSkyFunctionException(builderException, skyKey);
713         // Propagated transitive errors are treated the same as missing deps.
714         if (reifiedBuilderException.getRootCauseSkyKey().equals(skyKey)) {
715           boolean shouldFailFast = !keepGoing || builderException.isCatastrophic();
716           if (shouldFailFast) {
717             // After we commit this error to the graph but before the eval call completes with the
718             // error there is a race-like opportunity for the error to be used, either by an
719             // in-flight computation or by a future computation.
720             if (!visitor.preventNewEvaluations()) {
721               // This is not the first error encountered, so we ignore it so that we can terminate
722               // with the first error.
723               return;
724             }
725           }
726 
727           registerNewlyDiscoveredDepsForDoneEntry(skyKey, state, env);
728           ErrorInfo errorInfo = new ErrorInfo(reifiedBuilderException);
729           env.setError(errorInfo);
730           env.commit(/*enqueueParents=*/keepGoing);
731           if (!shouldFailFast) {
732             return;
733           }
734           throw SchedulerException.ofError(errorInfo, skyKey);
735         }
736       } catch (InterruptedException ie) {
737         // InterruptedException cannot be thrown by Runnable.run, so we must wrap it.
738         // Interrupts can be caught by both the Evaluator and the AbstractQueueVisitor.
739         // The former will unwrap the IE and propagate it as is; the latter will throw a new IE.
740         throw SchedulerException.ofInterruption(ie, skyKey);
741       } catch (RuntimeException re) {
742         // Programmer error (most likely NPE or a failed precondition in a SkyFunction). Output
743         // some context together with the exception.
744         String msg = prepareCrashMessage(skyKey, state.getInProgressReverseDeps());
745         throw new RuntimeException(msg, re);
746       } finally {
747         env.doneBuilding();
748         Profiler.instance().completeTask(ProfilerTask.SKYFUNCTION);
749       }
750 
751       GroupedListHelper<SkyKey> newDirectDeps = env.newlyRequestedDeps;
752 
753       if (value != null) {
754         Preconditions.checkState(!env.valuesMissing(),
755             "%s -> %s, ValueEntry: %s", skyKey, newDirectDeps, state);
756         env.setValue(value);
757         registerNewlyDiscoveredDepsForDoneEntry(skyKey, state, env);
758         env.commit(/*enqueueParents=*/true);
759         return;
760       }
761 
762       if (env.getDepErrorKey() != null) {
763         Preconditions.checkState(!keepGoing, "%s %s %s", skyKey, state, env.getDepErrorKey());
764         // We encountered a child error in noKeepGoing mode, so we want to fail fast. But we first
765         // need to add the edge between the current node and the child error it requested so that
766         // error bubbling can occur. Note that this edge will subsequently be removed during graph
767         // cleaning (since the current node will never be committed to the graph).
768         SkyKey childErrorKey = env.getDepErrorKey();
769         NodeEntry childErrorEntry = Preconditions.checkNotNull(graph.get(childErrorKey),
770             "skyKey: %s, state: %s childErrorKey: %s", skyKey, state, childErrorKey);
771         if (!state.getTemporaryDirectDeps().contains(childErrorKey)) {
772           // This means the cached error was freshly requested (e.g. the parent has never been
773           // built before).
774           Preconditions.checkState(newDirectDeps.contains(childErrorKey), "%s %s %s", state,
775               childErrorKey, newDirectDeps);
776           state.addTemporaryDirectDeps(GroupedListHelper.create(ImmutableList.of(childErrorKey)));
777           DependencyState childErrorState = childErrorEntry.addReverseDepAndCheckIfDone(skyKey);
778           Preconditions.checkState(childErrorState == DependencyState.DONE,
779               "skyKey: %s, state: %s childErrorKey: %s", skyKey, state, childErrorKey,
780               childErrorEntry);
781         } else {
782           // This means the cached error was previously requested, and was then subsequently (after
783           // a restart) requested along with another sibling dep. This can happen on an incremental
784           // eval call when the parent is dirty and the child error is in a separate dependency
785           // group from the sibling dep.
786           Preconditions.checkState(!newDirectDeps.contains(childErrorKey), "%s %s %s", state,
787               childErrorKey, newDirectDeps);
788           Preconditions.checkState(childErrorEntry.isDone(),
789               "skyKey: %s, state: %s childErrorKey: %s", skyKey, state, childErrorKey,
790               childErrorEntry);
791         }
792         ErrorInfo childErrorInfo = Preconditions.checkNotNull(childErrorEntry.getErrorInfo());
793         throw SchedulerException.ofError(childErrorInfo, childErrorKey);
794       }
795 
796       // TODO(bazel-team): This code is not safe to interrupt, because we would lose the state in
797       // newDirectDeps.
798 
799       // TODO(bazel-team): An ill-behaved SkyFunction can throw us into an infinite loop where we
800       // add more dependencies on every run. [skyframe-core]
801 
802       // Add all new keys to the set of known deps.
803       state.addTemporaryDirectDeps(newDirectDeps);
804 
805       // If there were no newly requested dependencies, at least one of them was in error or there
806       // is a bug in the SkyFunction implementation. The environment has collected its errors, so we
807       // just order it to be built.
808       if (newDirectDeps.isEmpty()) {
809         // TODO(bazel-team): This means a bug in the SkyFunction. What to do?
810         Preconditions.checkState(!env.childErrorInfos.isEmpty(), "%s %s", skyKey, state);
811         env.commit(/*enqueueParents=*/keepGoing);
812         if (!keepGoing) {
813           throw SchedulerException.ofError(state.getErrorInfo(), skyKey);
814         }
815         return;
816       }
817 
818       for (SkyKey newDirectDep : newDirectDeps) {
819         enqueueChild(skyKey, state, newDirectDep);
820       }
821       // It is critical that there is no code below this point.
822     }

824     private String prepareCrashMessage(SkyKey skyKey, Iterable<SkyKey> reverseDeps) {
825       StringBuilder reverseDepDump = new StringBuilder();
826       for (SkyKey key : reverseDeps) {
827         if (reverseDepDump.length() > MAX_REVERSEDEP_DUMP_LENGTH) {
828           reverseDepDump.append(", ...");
829           break;
830         }
831         if (reverseDepDump.length() > 0) {
832           reverseDepDump.append(", ");
833         }
834         reverseDepDump.append("'");
835         reverseDepDump.append(key.toString());
836         reverseDepDump.append("'");
837       }
838 
839       return String.format(
840           "Unrecoverable error while evaluating node '%s' (requested by nodes %s)",
841           skyKey, reverseDepDump);
842     }

847   /**
848    * Signals all parents that this node is finished. If visitor is not null, also enqueues any
849    * parents that are ready. If visitor is null, indicating that we are building this node after
850    * the main build aborted, then skip any parents that are already done (that can happen with
851    * cycles).
852    */
853   private void signalValuesAndEnqueueIfReady(@Nullable ValueVisitor visitor, Iterable<SkyKey> keys,
854       Version version) {
855     if (visitor != null) {
856       for (SkyKey key : keys) {
857         if (graph.get(key).signalDep(version)) {
858           visitor.enqueueEvaluation(key);
859         }
860       }
861     } else {
862       for (SkyKey key : keys) {
863         NodeEntry entry = Preconditions.checkNotNull(graph.get(key), key);
864         if (!entry.isDone()) {
865           // In cycles, we can have parents that are already done.
866           entry.signalDep(version);
867         }
868       }
869     }
870   }

885   /**
886    * Add any additional deps that were registered during the run of a builder that finished by
887    * creating a node or throwing an error. Builders may throw errors even if all their deps were
888    * not provided -- we trust that a SkyFunction may be know it should throw an error even if not
889    * all of its requested deps are done. However, that means we're assuming the SkyFunction would
890    * throw that same error if all of its requested deps were done. Unfortunately, there is no way to
891    * enforce that condition.
892    */
893   private void registerNewlyDiscoveredDepsForDoneEntry(SkyKey skyKey, NodeEntry entry,
894       SkyFunctionEnvironment env) {
895     Set<SkyKey> unfinishedDeps = new HashSet<>();
896     Iterables.addAll(unfinishedDeps,
897         Iterables.filter(env.newlyRequestedDeps, Predicates.not(nodeEntryIsDone)));
898     env.newlyRequestedDeps.remove(unfinishedDeps);
899     entry.addTemporaryDirectDeps(env.newlyRequestedDeps);
900     for (SkyKey newDep : env.newlyRequestedDeps) {
901       NodeEntry depEntry = graph.get(newDep);
902       DependencyState triState = depEntry.addReverseDepAndCheckIfDone(skyKey);
903       Preconditions.checkState(DependencyState.DONE == triState,
904           "new dep %s was not already done for %s. ValueEntry: %s. DepValueEntry: %s",
905           newDep, skyKey, entry, depEntry);
906       entry.signalDep();
907     }
908     Preconditions.checkState(entry.isReady(), "%s %s %s", skyKey, entry, env.newlyRequestedDeps);
909   }

911   private void informProgressReceiverThatValueIsDone(SkyKey key) {
912     if (progressReceiver != null) {
913       NodeEntry entry = graph.get(key);
914       Preconditions.checkState(entry.isDone(), entry);
915       SkyValue value = entry.getValue();
916       Version valueVersion = entry.getVersion();
917       Preconditions.checkState(valueVersion.atMost(graphVersion),
918           "%s should be at most %s in the version partial ordering", valueVersion, graphVersion);
919       // For most nodes we do not inform the progress receiver if they were already done when we
920       // retrieve them, but top-level nodes are presumably of more interest.
921       // If valueVersion is not equal to graphVersion, it must be less than it (by the
922       // Preconditions check above), and so the node is clean.
923       progressReceiver.evaluated(key, value, valueVersion.equals(graphVersion)
924           ? EvaluationState.BUILT
925           : EvaluationState.CLEAN);

931   public <T extends SkyValue> EvaluationResult<T> eval(Iterable<SkyKey> skyKeys)
932       throws InterruptedException {
933     ImmutableSet<SkyKey> skyKeySet = ImmutableSet.copyOf(skyKeys);
934 
935     // Optimization: if all required node values are already present in the cache, return them
936     // directly without launching the heavy machinery, spawning threads, etc.
937     // Inform progressReceiver that these nodes are done to be consistent with the main code path.
938     if (Iterables.all(skyKeySet, nodeEntryIsDone)) {
939       for (SkyKey skyKey : skyKeySet) {
940         informProgressReceiverThatValueIsDone(skyKey);
941       }
942       // Note that the 'catastrophe' parameter doesn't really matter here (it's only used for
943       // sanity checking).
944       return constructResult(null, skyKeySet, null, /*catastrophe=*/false);
945     }
946 
947     if (!keepGoing) {
948       Set<SkyKey> cachedErrorKeys = new HashSet<>();
949       for (SkyKey skyKey : skyKeySet) {
950         NodeEntry entry = graph.get(skyKey);
951         if (entry == null) {
952           continue;
953         }
954         if (entry.isDone() && entry.getErrorInfo() != null) {
955           informProgressReceiverThatValueIsDone(skyKey);
956           cachedErrorKeys.add(skyKey);
957         }
958       }
959 
960       // Errors, even cached ones, should halt evaluations not in keepGoing mode.
961       if (!cachedErrorKeys.isEmpty()) {
962         // Note that the 'catastrophe' parameter doesn't really matter here (it's only used for
963         // sanity checking).
964         return constructResult(null, cachedErrorKeys, null, /*catastrophe=*/false);
965       }
966     }
967 
968     // We delay this check until we know that some kind of evaluation is necessary, since !keepGoing
969     // and !keepsEdges are incompatible only in the case of a failed evaluation -- there is no
970     // need to be overly harsh to callers who are just trying to retrieve a cached result.
971     Preconditions.checkState(keepGoing || !(graph instanceof InMemoryGraph)
972         || ((InMemoryGraph) graph).keepsEdges(),
973         "nokeep_going evaluations are not allowed if graph edges are not kept: %s", skyKeys);
974 
975     Profiler.instance().startTask(ProfilerTask.SKYFRAME_EVAL, skyKeySet);
976     try {
977       return eval(skyKeySet, new ValueVisitor(threadCount));
978     } finally {
979       Profiler.instance().completeTask(ProfilerTask.SKYFRAME_EVAL);
980     }

983   @ThreadCompatible
984   private <T extends SkyValue> EvaluationResult<T> eval(ImmutableSet<SkyKey> skyKeys,
985       ValueVisitor visitor) throws InterruptedException {
986     // We unconditionally add the ErrorTransienceValue here, to ensure that it will be created, and
987     // in the graph, by the time that it is needed. Creating it on demand in a parallel context sets
988     // up a race condition, because there is no way to atomically create a node and set its value.
989     NodeEntry errorTransienceEntry = graph.createIfAbsent(ErrorTransienceValue.key());
990     DependencyState triState = errorTransienceEntry.addReverseDepAndCheckIfDone(null);
991     Preconditions.checkState(triState != DependencyState.ADDED_DEP,
992         "%s %s", errorTransienceEntry, triState);
993     if (triState != DependencyState.DONE) {
994       errorTransienceEntry.setValue(new ErrorTransienceValue(), graphVersion);
995       // The error transience entry is always invalidated by the RecordingDifferencer.
996       // Now that the entry's value is set, it is no longer dirty.
997       dirtyKeyTracker.notDirty(ErrorTransienceValue.key());
998 
999       Preconditions.checkState(
1000           errorTransienceEntry.addReverseDepAndCheckIfDone(null) != DependencyState.ADDED_DEP,
1001           errorTransienceEntry);
1002     }
1003     for (SkyKey skyKey : skyKeys) {
1004       NodeEntry entry = graph.createIfAbsent(skyKey);
1005       // This must be equivalent to the code in enqueueChild above, in order to be thread-safe.
1006       switch (entry.addReverseDepAndCheckIfDone(null)) {
1007         case NEEDS_SCHEDULING:
1008           visitor.enqueueEvaluation(skyKey);
1009           break;
1010         case DONE:
1011           informProgressReceiverThatValueIsDone(skyKey);
1012           break;
1013         case ADDED_DEP:
1014           break;
1015         default:
1016           throw new IllegalStateException(entry + " for " + skyKey + " in unknown state");
1017       }
1018     }
1019     boolean shouldClean = false;
1020     try {
1021       EvaluationResult<T> result = waitForCompletionAndConstructResult(visitor, skyKeys);
1022       shouldClean = true;
1023       return result;
1024     } catch (InterruptedException e) {
1025       shouldClean = true;
1026       throw e;
1027     } finally {
1028       if (shouldClean) {
1029         // TODO(bazel-team): In nokeep_going mode or in case of an interrupt, we need to remove
1030         // partial values from the graph. Find a better way to handle those cases.
1031         clean(visitor.inflightNodes);
1032       }
1033       // If we're going to crash anyway, things might be in a weird state such that 'clean' is
1034       // likely to crash too. So propagating the original exception is much more useful.
1035       // Note that this is safe in the sense that this decision doesn't prevent weird
1036       // SkyFunctions from doing weird things with unchecked exceptions - that's already
1037       // prevented directly by SkyFunctionException#validateExceptionType.
1038     }
1039   }

1041   private void clean(Set<SkyKey> inflightNodes) throws InterruptedException {
1042     boolean alreadyInterrupted = Thread.interrupted();
1043     // This parallel computation is fully cpu-bound, so we use a thread for each processor.
1044     ExecutorService executor = Executors.newFixedThreadPool(
1045         Runtime.getRuntime().availableProcessors(),
1046         new ThreadFactoryBuilder().setNameFormat("ParallelEvaluator#clean %d").build());
1047     ThrowableRecordingRunnableWrapper wrapper =
1048         new ThrowableRecordingRunnableWrapper("ParallelEvaluator#clean");
1049     for (final SkyKey key : inflightNodes) {
1050       final NodeEntry entry = graph.get(key);
1051       Preconditions.checkState(!entry.isDone(), "%s %s", key, entry);
1052       executor.execute(wrapper.wrap(new Runnable() {
1053         @Override
1054         public void run() {
1055           cleanInflightNode(key, entry);
1056         }
1057       }));
1058     }
1059     // We uninterruptibly wait for all nodes to be cleaned because we want to make sure the graph
1060     // is left in a good state.
1061     //
1062     // TODO(bazel-team): Come up with a better design for graph cleaning such that we can respond
1063     // to interrupts in constant time.
1064     boolean newlyInterrupted = ExecutorUtil.uninterruptibleShutdown(executor);
1065     Throwables.propagateIfPossible(wrapper.getFirstThrownError());
1066     if (newlyInterrupted || alreadyInterrupted) {
1067       throw new InterruptedException();
1068     }
1069   }

1071   private void cleanInflightNode(SkyKey key, NodeEntry entry) {
1072     Set<SkyKey> temporaryDeps = entry.getTemporaryDirectDeps();
1073     graph.remove(key);
1074     for (SkyKey dep : temporaryDeps) {
1075       NodeEntry nodeEntry = graph.get(dep);
1076       // The direct dep might have already been cleaned from the graph.
1077       if (nodeEntry != null) {
1078         // Only bother removing the reverse dep on done nodes since other in-flight nodes will be
1079         // cleaned too.
1080         if (nodeEntry.isDone()) {
1081           nodeEntry.removeReverseDep(key);
1082         }
1083       }
1084     }
1085   }

1087   private <T extends SkyValue> EvaluationResult<T> waitForCompletionAndConstructResult(
1088       ValueVisitor visitor, Iterable<SkyKey> skyKeys) throws InterruptedException {
1089     Map<SkyKey, ValueWithMetadata> bubbleErrorInfo = null;
1090     boolean catastrophe = false;
1091     try {
1092       visitor.waitForCompletion();
1093     } catch (final SchedulerException e) {
1094       Throwables.propagateIfPossible(e.getCause(), InterruptedException.class);
1095       if (Thread.interrupted()) {
1096         // As per the contract of AbstractQueueVisitor#work, if an unchecked exception is thrown and
1097         // the build is interrupted, the thrown exception is what will be rethrown. Since the user
1098         // presumably wanted to interrupt the build, we ignore the thrown SchedulerException (which
1099         // doesn't indicate a programming bug) and throw an InterruptedException.
1100         throw new InterruptedException();
1101       }
1102 
1103       SkyKey errorKey = Preconditions.checkNotNull(e.getFailedValue(), e);
1104       // ErrorInfo could only be null if SchedulerException wrapped an InterruptedException, but
1105       // that should have been propagated.
1106       ErrorInfo errorInfo = Preconditions.checkNotNull(e.getErrorInfo(), errorKey);
1107       catastrophe = errorInfo.isCatastrophic();
1108       if (!catastrophe || !keepGoing) {
1109         bubbleErrorInfo = bubbleErrorUp(errorInfo, errorKey, skyKeys, visitor);
1110       } else {
1111         // Bubbling the error up requires that graph edges are present for done nodes. This is not
1112         // always the case in a keepGoing evaluation, since it is assumed that done nodes do not
1113         // need to be traversed. In this case, we hope the caller is tolerant of a possibly empty
1114         // result, and return prematurely.
1115         bubbleErrorInfo = ImmutableMap.of(errorKey, graph.get(errorKey).getValueWithMetadata());
1116       }
1117     }
1118 
1119     // Successful evaluation, either because keepGoing or because we actually did succeed.
1120     // TODO(bazel-team): Maybe report root causes during the build for lower latency.
1121     return constructResult(visitor, skyKeys, bubbleErrorInfo, catastrophe);
1122   }

1124   /**
1125    * Walk up graph to find a top-level node (without parents) that wanted this failure. Store
1126    * the failed nodes along the way in a map, with ErrorInfos that are appropriate for that layer.
1127    * Example:
1128    *                      foo   bar
1129    *                        \   /
1130    *           unrequested   baz
1131    *                     \    |
1132    *                      failed-node
1133    * User requests foo, bar. When failed-node fails, we look at its parents. unrequested is not
1134    * in-flight, so we replace failed-node by baz and repeat. We look at baz's parents. foo is
1135    * in-flight, so we replace baz by foo. Since foo is a top-level node and doesn't have parents,
1136    * we then break, since we know a top-level node, foo, that depended on the failed node.
1137    *
1138    * There's the potential for a weird "track jump" here in the case:
1139    *                        foo
1140    *                       / \
1141    *                   fail1 fail2
1142    * If fail1 and fail2 fail simultaneously, fail2 may start propagating up in the loop below.
1143    * However, foo requests fail1 first, and then throws an exception based on that. This is not
1144    * incorrect, but may be unexpected.
1145    *
1146    * <p>Returns a map of errors that have been constructed during the bubbling up, so that the
1147    * appropriate error can be returned to the caller, even though that error was not written to the
1148    * graph. If a cycle is detected during the bubbling, this method aborts and returns null so that
1149    * the normal cycle detection can handle the cycle.
1150    *
1151    * <p>Note that we are not propagating error to the first top-level node but to the highest one,
1152    * because during this process we can add useful information about error from other nodes.
1153    */
1154   private Map<SkyKey, ValueWithMetadata> bubbleErrorUp(final ErrorInfo leafFailure,
1155       SkyKey errorKey, Iterable<SkyKey> skyKeys, ValueVisitor visitor) {
1156     Set<SkyKey> rootValues = ImmutableSet.copyOf(skyKeys);
1157     ErrorInfo error = leafFailure;
1158     Map<SkyKey, ValueWithMetadata> bubbleErrorInfo = new HashMap<>();
1159     boolean externalInterrupt = false;
1160     while (true) {
1161       NodeEntry errorEntry = graph.get(errorKey);
1162       Iterable<SkyKey> reverseDeps = errorEntry.isDone()
1163           ? errorEntry.getReverseDeps()
1164           : errorEntry.getInProgressReverseDeps();
1165       // We should break from loop only when node doesn't have any parents.
1166       if (Iterables.isEmpty(reverseDeps)) {
1167         Preconditions.checkState(rootValues.contains(errorKey),
1168             "Current key %s has to be a top-level key: %s", errorKey, rootValues);
1169         break;
1170       }
1171       SkyKey parent = null;
1172       NodeEntry parentEntry = null;
1173       for (SkyKey bubbleParent : reverseDeps) {
1174         if (bubbleErrorInfo.containsKey(bubbleParent)) {
1175           // We are in a cycle. Don't try to bubble anything up -- cycle detection will kick in.
1176           return null;
1177         }
1178         NodeEntry bubbleParentEntry = Preconditions.checkNotNull(graph.get(bubbleParent),
1179             "parent %s of %s not in graph", bubbleParent, errorKey);
1180         // Might be the parent that requested the error.
1181         if (bubbleParentEntry.isDone()) {
1182           // This parent is cached from a previous evaluate call. We shouldn't bubble up to it
1183           // since any error message produced won't be meaningful to this evaluate call.
1184           // The child error must also be cached from a previous build.
1185           Preconditions.checkState(errorEntry.isDone(), "%s %s", errorEntry, bubbleParentEntry);
1186           Version parentVersion = bubbleParentEntry.getVersion();
1187           Version childVersion = errorEntry.getVersion();
1188           Preconditions.checkState(childVersion.atMost(graphVersion)
1189               && !childVersion.equals(graphVersion),
1190               "child entry is not older than the current graph version, but had a done parent. "
1191               + "child: %s childEntry: %s, childVersion: %s"
1192               + "bubbleParent: %s bubbleParentEntry: %s, parentVersion: %s, graphVersion: %s",
1193               errorKey, errorEntry, childVersion,
1194               bubbleParent, bubbleParentEntry, parentVersion, graphVersion);
1195           Preconditions.checkState(parentVersion.atMost(graphVersion)
1196               && !parentVersion.equals(graphVersion),
1197               "parent entry is not older than the current graph version. "
1198               + "child: %s childEntry: %s, childVersion: %s"
1199               + "bubbleParent: %s bubbleParentEntry: %s, parentVersion: %s, graphVersion: %s",
1200               errorKey, errorEntry, childVersion,
1201               bubbleParent, bubbleParentEntry, parentVersion, graphVersion);
1202           continue;
1203         }
1204         // Arbitrarily pick the first in-flight parent.
1205         Preconditions.checkState(visitor.isInflight(bubbleParent),
1206             "errorKey: %s, errorEntry: %s, bubbleParent: %s, bubbleParentEntry: %s", errorKey,
1207             errorEntry, bubbleParent, bubbleParentEntry);
1208         parent = bubbleParent;
1209         parentEntry = bubbleParentEntry;
1210         break;
1211       }
1212       Preconditions.checkNotNull(parent, "%s %s", errorKey, bubbleErrorInfo);
1213       errorKey = parent;
1214       SkyFunction factory = skyFunctions.get(parent.functionName());
1215       if (parentEntry.isDirty()) {
1216         switch (parentEntry.getDirtyState()) {
1217           case CHECK_DEPENDENCIES:
1218             // If this value's child was bubbled up to, it did not signal this value, and so we must
1219             // manually make it ready to build.
1220             parentEntry.signalDep();
1221             // Fall through to REBUILDING, since state is now REBUILDING.
1222           case REBUILDING:
1223             // Nothing to be done.
1224             break;
1225           default:
1226             throw new AssertionError(parent + " not in valid dirty state: " + parentEntry);
1227         }
1228       }
1229       SkyFunctionEnvironment env =
1230           new SkyFunctionEnvironment(parent, parentEntry.getTemporaryDirectDeps(),
1231               bubbleErrorInfo, visitor);
1232       externalInterrupt = externalInterrupt || Thread.currentThread().isInterrupted();
1233       try {
1234         // This build is only to check if the parent node can give us a better error. We don't
1235         // care about a return value.
1236         factory.compute(parent, env);
1237       } catch (SkyFunctionException builderException) {
1238         ReifiedSkyFunctionException reifiedBuilderException =
1239             new ReifiedSkyFunctionException(builderException, parent);
1240         if (reifiedBuilderException.getRootCauseSkyKey().equals(parent)) {
1241           error = new ErrorInfo(reifiedBuilderException);
1242           bubbleErrorInfo.put(errorKey,
1243               ValueWithMetadata.error(new ErrorInfo(errorKey, ImmutableSet.of(error)),
1244                   env.buildEvents(/*missingChildren=*/true)));
1245           continue;
1246         }
1247       } catch (InterruptedException interruptedException) {
1248         // Do nothing.
1249         // This throw happens if the builder requested the failed node, and then checked the
1250         // interrupted state later -- getValueOrThrow sets the interrupted bit after the failed
1251         // value is requested, to prevent the builder from doing too much work.
1252       } finally {
1253         // Clear interrupted status. We're not listening to interrupts here.
1254         Thread.interrupted();
1255       }
1256       // Builder didn't throw an exception, so just propagate this one up.
1257       bubbleErrorInfo.put(errorKey,
1258           ValueWithMetadata.error(new ErrorInfo(errorKey, ImmutableSet.of(error)),
1259               env.buildEvents(/*missingChildren=*/true)));
1260     }
1261 
1262     // Reset the interrupt bit if there was an interrupt from outside this evaluator interrupt.
1263     // Note that there are internal interrupts set in the node builder environment if an error
1264     // bubbling node calls getValueOrThrow() on a node in error.
1265     if (externalInterrupt) {
1266       Thread.currentThread().interrupt();
1267     }
1268     return bubbleErrorInfo;
1269   }

1271   /**
1272    * Constructs an {@link EvaluationResult} from the {@link #graph}.  Looks for cycles if there
1273    * are unfinished nodes but no error was already found through bubbling up
1274    * (as indicated by {@code bubbleErrorInfo} being null).
1275    *
1276    * <p>{@code visitor} may be null, but only in the case where all graph entries corresponding to
1277    * {@code skyKeys} are known to be in the DONE state ({@code entry.isDone()} returns true).
1278    */
1279   private <T extends SkyValue> EvaluationResult<T> constructResult(
1280       @Nullable ValueVisitor visitor, Iterable<SkyKey> skyKeys,
1281       Map<SkyKey, ValueWithMetadata> bubbleErrorInfo, boolean catastrophe) {
1282     Preconditions.checkState(!keepGoing || catastrophe || bubbleErrorInfo == null,
1283         "", skyKeys, bubbleErrorInfo);
1284     EvaluationResult.Builder<T> result = EvaluationResult.builder();
1285     List<SkyKey> cycleRoots = new ArrayList<>();
1286     boolean hasError = false;
1287     for (SkyKey skyKey : skyKeys) {
1288       ValueWithMetadata valueWithMetadata = getValueMaybeFromError(skyKey, bubbleErrorInfo);
1289       // Cycle checking: if there is a cycle, evaluation cannot progress, therefore,
1290       // the final values will not be in DONE state when the work runs out.
1291       if (valueWithMetadata == null) {
1292         // Don't look for cycles if the build failed for a known reason.
1293         if (bubbleErrorInfo == null) {
1294           cycleRoots.add(skyKey);
1295         }
1296         hasError = true;
1297         continue;
1298       }
1299       SkyValue value = valueWithMetadata.getValue();
1300       // TODO(bazel-team): Verify that message replay is fast and works in failure
1301       // modes [skyframe-core]
1302       // Note that replaying events here is only necessary on null builds, because otherwise we
1303       // would have already printed the transitive messages after building these values.
1304       replayingNestedSetEventVisitor.visit(valueWithMetadata.getTransitiveEvents());
1305       ErrorInfo errorInfo = valueWithMetadata.getErrorInfo();
1306       Preconditions.checkState(value != null || errorInfo != null, skyKey);
1307       hasError = hasError || (errorInfo != null);
1308       if (!keepGoing && errorInfo != null) {
1309         // value will be null here unless the value was already built on a prior keepGoing build.
1310         result.addError(skyKey, errorInfo);
1311         continue;
1312       }
1313       if (value == null) {
1314         // Note that we must be in the keepGoing case. Only make this value an error if it doesn't
1315         // have a value. The error shouldn't matter to the caller since the value succeeded after a
1316         // fashion.
1317         result.addError(skyKey, errorInfo);
1318       } else {
1319         result.addResult(skyKey, value);
1320       }
1321     }
1322     if (!cycleRoots.isEmpty()) {
1323       Preconditions.checkState(visitor != null, skyKeys);
1324       checkForCycles(cycleRoots, result, visitor, keepGoing);
1325     }
1326     Preconditions.checkState(bubbleErrorInfo == null || hasError,
1327         "If an error bubbled up, some top-level node must be in error", bubbleErrorInfo, skyKeys);
1328     result.setHasError(hasError);
1329     return result.build();
1330   }

1332   private <T extends SkyValue> void checkForCycles(
1333       Iterable<SkyKey> badRoots, EvaluationResult.Builder<T> result, final ValueVisitor visitor,
1334       boolean keepGoing) {
1335     for (SkyKey root : badRoots) {
1336       ErrorInfo errorInfo = checkForCycles(root, visitor, keepGoing);
1337       if (errorInfo == null) {
1338         // This node just wasn't finished when evaluation aborted -- there were no cycles below it.
1339         Preconditions.checkState(!keepGoing, "", root, badRoots);
1340         continue;
1341       }
1342       Preconditions.checkState(!Iterables.isEmpty(errorInfo.getCycleInfo()),
1343           "%s was not evaluated, but was not part of a cycle", root);
1344       result.addError(root, errorInfo);
1345       if (!keepGoing) {
1346         return;
1347       }
1348     }
1349   }

1351   /**
1352    * Marker value that we push onto a stack before we push a node's children on. When the marker
1353    * value is popped, we know that all the children are finished. We would use null instead, but
1354    * ArrayDeque does not permit null elements.
1355    */
1356   private static final SkyKey CHILDREN_FINISHED =
1357       new SkyKey(SkyFunctionName.create("MARKER"), "MARKER");
1358 
1359   /** The max number of cycles we will report to the user for a given root, to avoid OOMing. */
1360   private static final int MAX_CYCLES = 20;
1361 
1362   /**
1363    * The algorithm for this cycle detector is as follows. We visit the graph depth-first, keeping
1364    * track of the path we are currently on. We skip any DONE nodes (they are transitively
1365    * error-free). If we come to a node already on the path, we immediately construct a cycle. If
1366    * we are in the noKeepGoing case, we return ErrorInfo with that cycle to the caller. Otherwise,
1367    * we continue. Once all of a node's children are done, we construct an error value for it, based
1368    * on those children. Finally, when the original root's node is constructed, we return its
1369    * ErrorInfo.
1370    */
1371   private ErrorInfo checkForCycles(SkyKey root, ValueVisitor visitor, boolean keepGoing) {
1372     // The number of cycles found. Do not keep on searching for more cycles after this many were
1373     // found.
1374     int cyclesFound = 0;
1375     // The path through the graph currently being visited.
1376     List<SkyKey> graphPath = new ArrayList<>();
1377     // Set of nodes on the path, to avoid expensive searches through the path for cycles.
1378     Set<SkyKey> pathSet = new HashSet<>();
1379 
1380     // Maintain a stack explicitly instead of recursion to avoid stack overflows
1381     // on extreme graphs (with long dependency chains).
1382     Deque<SkyKey> toVisit = new ArrayDeque<>();
1383 
1384     toVisit.push(root);
1385 
1386     // The procedure for this check is as follows: we visit a node, push it onto the graph stack,
1387     // push a marker value onto the toVisit stack, and then push all of its children onto the
1388     // toVisit stack. Thus, when the marker node comes to the top of the toVisit stack, we have
1389     // visited the downward transitive closure of the value. At that point, all of its children must
1390     // be finished, and so we can build the definitive error info for the node, popping it off the
1391     // graph stack.
1392     while (!toVisit.isEmpty()) {
1393       SkyKey key = toVisit.pop();
1394       NodeEntry entry = graph.get(key);
1395 
1396       if (key == CHILDREN_FINISHED) {
1397         // A marker node means we are done with all children of a node. Since all nodes have
1398         // errors, we must have found errors in the children when that happens.
1399         key = graphPath.remove(graphPath.size() - 1);
1400         entry = Preconditions.checkNotNull(graph.get(key), key);
1401         pathSet.remove(key);
1402         // Skip this node if it was first/last node of a cycle, and so has already been processed.
1403         if (entry.isDone()) {
1404           continue;
1405         }
1406         if (!keepGoing) {
1407           // in the --nokeep_going mode, we would have already returned if we'd found a cycle below
1408           // this node. The fact that we haven't means that there were no cycles below this node
1409           // -- it just hadn't finished evaluating. So skip it.
1410           continue;
1411         }
1412         if (cyclesFound < MAX_CYCLES) {
1413           // Value must be ready, because all of its children have finished, so we can build its
1414           // error.
1415           Preconditions.checkState(entry.isReady(), "%s not ready. ValueEntry: %s", key, entry);
1416         } else if (!entry.isReady()) {
1417           removeIncompleteChildrenForCycle(key, entry, entry.getTemporaryDirectDeps());
1418         }
1419         Set<SkyKey> directDeps = entry.getTemporaryDirectDeps();
1420         // Find out which children have errors. Similar logic to that in Evaluate#run().
1421         List<ErrorInfo> errorDeps = getChildrenErrorsForCycle(directDeps);
1422         Preconditions.checkState(!errorDeps.isEmpty(),
1423             "Value %s was not successfully evaluated, but had no child errors. ValueEntry: %s", key,
1424             entry);
1425         SkyFunctionEnvironment env = new SkyFunctionEnvironment(key, directDeps, visitor);
1426         env.setError(new ErrorInfo(key, errorDeps));
1427         env.commit(/*enqueueParents=*/false);
1428       }
1429 
1430       Preconditions.checkNotNull(entry, key);
1431       // Nothing to be done for this node if it already has an entry.
1432       if (entry.isDone()) {
1433         continue;
1434       }
1435       if (cyclesFound == MAX_CYCLES) {
1436         // Do not keep on searching for cycles indefinitely, to avoid excessive runtime/OOMs.
1437         continue;
1438       }
1439 
1440       if (pathSet.contains(key)) {
1441         int cycleStart = graphPath.indexOf(key);
1442         // Found a cycle!
1443         cyclesFound++;
1444         Iterable<SkyKey> cycle = graphPath.subList(cycleStart, graphPath.size());
1445         // Put this node into a consistent state for building if it is dirty.
1446         if (entry.isDirty() && entry.getDirtyState() == NodeEntry.DirtyState.CHECK_DEPENDENCIES) {
1447           // In the check deps state, entry has exactly one child not done yet. Note that this node
1448           // must be part of the path to the cycle we have found (since done nodes cannot be in
1449           // cycles, and this is the only missing one). Thus, it will not be removed below in
1450           // removeDescendantsOfCycleValue, so it is safe here to signal that it is done.
1451           entry.signalDep();
1452         }
1453         if (keepGoing) {
1454           // Any children of this node that we haven't already visited are not worth visiting,
1455           // since this node is about to be done. Thus, the only child worth visiting is the one in
1456           // this cycle, the cycleChild (which may == key if this cycle is a self-edge).
1457           SkyKey cycleChild = selectCycleChild(key, graphPath, cycleStart);
1458           removeDescendantsOfCycleValue(key, entry, cycleChild, toVisit,
1459                   graphPath.size() - cycleStart);
1460           ValueWithMetadata dummyValue = ValueWithMetadata.wrapWithMetadata(new SkyValue() {});
1461 
1462 
1463           SkyFunctionEnvironment env =
1464               new SkyFunctionEnvironment(key, entry.getTemporaryDirectDeps(),
1465                   ImmutableMap.of(cycleChild, dummyValue), visitor);
1466 
1467           // Construct error info for this node. Get errors from children, which are all done
1468           // except possibly for the cycleChild.
1469           List<ErrorInfo> allErrors =
1470               getChildrenErrors(entry.getTemporaryDirectDeps(), /*unfinishedChild=*/cycleChild);
1471           CycleInfo cycleInfo = new CycleInfo(cycle);
1472           // Add in this cycle.
1473           allErrors.add(new ErrorInfo(cycleInfo));
1474           env.setError(new ErrorInfo(key, allErrors));
1475           env.commit(/*enqueueParents=*/false);
1476           continue;
1477         } else {
1478           // We need to return right away in the noKeepGoing case, so construct the cycle (with the
1479           // path) and return.
1480           Preconditions.checkState(graphPath.get(0).equals(root),
1481               "%s not reached from %s. ValueEntry: %s", key, root, entry);
1482           return new ErrorInfo(new CycleInfo(graphPath.subList(0, cycleStart), cycle));
1483         }
1484       }
1485 
1486       // This node is not yet known to be in a cycle. So process its children.
1487       Iterable<? extends SkyKey> children = graph.get(key).getTemporaryDirectDeps();
1488       if (Iterables.isEmpty(children)) {
1489         continue;
1490       }
1491 
1492       // This marker flag will tell us when all this node's children have been processed.
1493       toVisit.push(CHILDREN_FINISHED);
1494       // This node is now part of the path through the graph.
1495       graphPath.add(key);
1496       pathSet.add(key);
1497       for (SkyKey nextValue : children) {
1498         toVisit.push(nextValue);
1499       }
1500     }
1501     return keepGoing ? getAndCheckDone(root).getErrorInfo() : null;
1502   }

1559   /**
1560    * Removes direct children of key from toVisit and from the entry itself, and makes the entry
1561    * ready if necessary. We must do this because it would not make sense to try to build the
1562    * children after building the entry. It would violate the invariant that a parent can only be
1563    * built after its children are built; See bug "Precondition error while evaluating a Skyframe
1564    * graph with a cycle".
1565    *
1566    * @param key SkyKey of node in a cycle.
1567    * @param entry NodeEntry of node in a cycle.
1568    * @param cycleChild direct child of key in the cycle, or key itself if the cycle is a self-edge.
1569    * @param toVisit list of remaining nodes to visit by the cycle-checker.
1570    * @param cycleLength the length of the cycle found.
1571    */
1572   private void removeDescendantsOfCycleValue(SkyKey key, NodeEntry entry,
1573       @Nullable SkyKey cycleChild, Iterable<SkyKey> toVisit, int cycleLength) {
1574     Set<SkyKey> unvisitedDeps = new HashSet<>(entry.getTemporaryDirectDeps());
1575     unvisitedDeps.remove(cycleChild);
1576     // Remove any children from this node that are not part of the cycle we just found. They are
1577     // irrelevant to the node as it stands, and if they are deleted from the graph because they are
1578     // not built by the end of cycle-checking, we would have dangling references.
1579     removeIncompleteChildrenForCycle(key, entry, unvisitedDeps);
1580     if (!entry.isReady()) {
1581       // The entry has at most one undone dep now, its cycleChild. Signal to make entry ready. Note
1582       // that the entry can conceivably be ready if its cycleChild already found a different cycle
1583       // and was built.
1584       entry.signalDep();
1585     }
1586     Preconditions.checkState(entry.isReady(), "%s %s %s", key, cycleChild, entry);
1587     Iterator<SkyKey> it = toVisit.iterator();
1588     while (it.hasNext()) {
1589       SkyKey descendant = it.next();
1590       if (descendant == CHILDREN_FINISHED) {
1591         // Marker value, delineating the end of a group of children that were enqueued.
1592         cycleLength--;
1593         if (cycleLength == 0) {
1594           // We have seen #cycleLength-1 marker values, and have arrived at the one for this value,
1595           // so we are done.
1596           return;
1597         }
1598         continue; // Don't remove marker values.
1599       }
1600       if (cycleLength == 1) {
1601         // Remove the direct children remaining to visit of the cycle node.
1602         Preconditions.checkState(unvisitedDeps.contains(descendant),
1603             "%s %s %s %s %s", key, descendant, cycleChild, unvisitedDeps, entry);
1604         it.remove();
1605       }
1606     }
1607     throw new IllegalStateException("There were not " + cycleLength + " groups of children in "
1608         + toVisit + " when trying to remove children of " + key + " other than " + cycleChild);
1609   }

1620   }
