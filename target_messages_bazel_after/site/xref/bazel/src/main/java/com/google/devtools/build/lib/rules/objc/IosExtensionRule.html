27  /**
28   * Rule definition for ios_extension.
29   */
30  public class IosExtensionRule implements RuleDefinition {
31    @Override
32    public RuleClass build(Builder builder, RuleDefinitionEnvironment env) {
33      return builder
34          /*<!-- #BLAZE_RULE(ios_extension).IMPLICIT_OUTPUTS -->
35          <ul>
36           <li><code><var>name</var>.ipa</code>: the extension bundle as an <code>.ipa</code>
37               file</li>
38           <li><code><var>name</var>.xcodeproj/project.pbxproj</code>: An Xcode project file which
39               can be used to develop or build on a Mac.</li>
40          </ul>
41          <!-- #END_BLAZE_RULE.IMPLICIT_OUTPUTS -->*/
42          .setImplicitOutputsFunction(
43              ImplicitOutputsFunction.fromFunctions(ReleaseBundlingSupport.IPA, XcodeSupport.PBXPROJ))
44          /* <!-- #BLAZE_RULE(ios_extension).ATTRIBUTE(binary) -->
45          The binary target containing the logic for the extension.
46          ${SYNOPSIS}
47          <!-- #END_BLAZE_RULE.ATTRIBUTE -->*/
48          .add(attr("binary", LABEL)
49              .allowedRuleClasses("ios_extension_binary")
50              .allowedFileTypes()
51              .mandatory()
52              .direct_compile_time_input()
53              .cfg(IosExtension.MINIMUM_OS_AND_SPLIT_ARCH_TRANSITION))
54          .build();
55    }
56  
57    @Override
58    public Metadata getMetadata() {
59      return RuleDefinition.Metadata.builder()
60          .name("ios_extension")
61          .factoryClass(IosExtension.class)
62          .ancestors(BaseRuleClasses.BaseRule.class, ObjcRuleClasses.ReleaseBundlingRule.class,
63              ObjcRuleClasses.XcodegenRule.class)
64          .build();
65    }
66  }

68  /*<!-- #BLAZE_RULE (NAME = ios_extension, TYPE = BINARY, FAMILY = Objective-C) -->
69  
70  ${ATTRIBUTE_SIGNATURE}
71  
72  <p>This rule produces a bundled binary for an iOS app extension from a compiled binary and bundle
73  metadata.</p>
74  
75  <p>An iOS app extension is a nested bundle that is located inside the application bundle and is
76  released with it. An iOS app extension cannot be released alone, although this rule allows you to
77  build an <code>.ipa</code> with only the extension.
78  
79  <p>Bundles generated by this rule use a bundle directory called
80  <code>PlugIns/<var>target-name</var>.appex</code>, while an application bundle uses
81  <code>Payload/<var>target-name</var>.app</code>. For instance, if an application call Foo has an app
82  extension called Bar, the Bar extension bundle files will be stored in
83  <code>Payload/Foo.app/PlugIns/Bar.appex</code> in the final application <code>.ipa</code>.
84  
85  <p>There are many similarities between app extensions and applications with little to no difference
86  between how each thing is processed:
87  <ul>
88    <li>both have entitlements and Info.plist files
89    <li>both are code-signed. Signing and merging happens in this order: the extension is code-signed,
90        bundles are merged, application is code-signed
91    <li>both can have an app icon and launch image, and of course asset catalogs and all kinds of
92        resources
93    <li>both have linked binaries. The app extension binary is different in that it is linked with
94        these additional flags:
95        <ul>
96            <li><code>-e _NSExtensionMain</code> - sets the entry point to a standard function in the
97                iOS runtime rather than <code>main()</code>
98            <li><code>-fapplicationextension</code>
99        </ul>
100 </ul>
101 
102 ${IMPLICIT_OUTPUTS}
103 
104 ${ATTRIBUTE_DEFINITION}
105 
106 <!-- #END_BLAZE_RULE -->*/
