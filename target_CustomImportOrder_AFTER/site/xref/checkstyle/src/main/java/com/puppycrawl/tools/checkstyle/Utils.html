44      /** maps from a token name to value */
45      private static final ImmutableMap<String, Integer> TOKEN_NAME_TO_VALUE;
46      /** maps from a token value to name */
47      private static final String[] TOKEN_VALUE_TO_NAME;
48  
49      /** Array of all token IDs */
50      private static final int[] TOKEN_IDS;
51  
52      // initialise the constants
53      static {
54          final ImmutableMap.Builder<String, Integer> builder =
55                  ImmutableMap.builder();
56          final Field[] fields = TokenTypes.class.getDeclaredFields();
57          String[] tempTokenValueToName = new String[0];
58          for (final Field f : fields) {
59              // Only process the int declarations.
60              if (f.getType() != Integer.TYPE) {
61                  continue;
62              }
63  
64              final String name = f.getName();
65              final int tokenValue = getIntFromField(f, name);
66              builder.put(name, tokenValue);
67              if (tokenValue > tempTokenValueToName.length - 1) {
68                  final String[] temp = new String[tokenValue + 1];
69                  System.arraycopy(tempTokenValueToName, 0,
70                          temp, 0, tempTokenValueToName.length);
71                  tempTokenValueToName = temp;
72              }
73              tempTokenValueToName[tokenValue] = name;
74          }
75  
76          TOKEN_NAME_TO_VALUE = builder.build();
77          TOKEN_VALUE_TO_NAME = tempTokenValueToName;
78  
79          final Integer[] ids = TOKEN_NAME_TO_VALUE.values().toArray(new Integer[0]);
80          TOKEN_IDS = new int[ids.length];
81          for (int i = 0; i < ids.length; i++) {
82              TOKEN_IDS[i] = ids[i].intValue();
83          }
84      }
85  
86  
87      /** stop instances being created **/
88      private Utils() {
89      }

91      /**
92       * Get all token IDs that are available in TokenTypes.
93       * @return array of token IDs
94       */
95      public static int[] getAllTokenIds() {
96          final int[] safeCopy = new int[TOKEN_IDS.length];
97          System.arraycopy(TOKEN_IDS, 0, safeCopy, 0, TOKEN_IDS.length);
98          return safeCopy;
99      }

101     /**
102      * Returns whether the file extension matches what we are meant to
103      * process.
104      * @param file the file to be checked.
105      * @param fileExtensions files extensions, empty property in config makes it matches to all.
106      * @return whether there is a match.
107      */
108     public static boolean fileExtensionMatches(File file, String... fileExtensions) {
109         boolean result = false;
110         if (fileExtensions == null || fileExtensions.length == 0) {
111             result = true;
112         }
113         else {
114             // normalize extensions so all of them have a leading dot
115             final String[] withDotExtensions = new String[fileExtensions.length];
116             for (int i = 0; i < fileExtensions.length; i++) {
117                 final String extension = fileExtensions[i];
118                 if (startsWithChar(extension, '.')) {
119                     withDotExtensions[i] = extension;
120                 }
121                 else {
122                     withDotExtensions[i] = "." + extension;
123                 }
124             }
125 
126             final String fileName = file.getName();
127             for (final String fileExtension : withDotExtensions) {
128                 if (fileName.endsWith(fileExtension)) {
129                     result = true;
130                 }
131             }
132         }
133 
134         return result;
135     }

137     /**
138      * Returns whether the specified string contains only whitespace up to the
139      * specified index.
140      *
141      * @param index index to check up to
142      * @param line the line to check
143      * @return whether there is only whitespace
144      */
145     public static boolean whitespaceBefore(int index, String line) {
146         for (int i = 0; i < index; i++) {
147             if (!Character.isWhitespace(line.charAt(i))) {
148                 return false;
149             }
150         }
151         return true;
152     }
153 
154     /**
155      * Returns the length of a string ignoring all trailing whitespace. It is a
156      * pity that there is not a trim() like method that only removed the
157      * trailing whitespace.
158      * @param line the string to process
159      * @return the length of the string ignoring all trailing whitespace
160      **/
161     public static int lengthMinusTrailingWhitespace(String line) {
162         int len = line.length();
163         for (int i = len - 1; i >= 0; i--) {
164             if (!Character.isWhitespace(line.charAt(i))) {
165                 break;
166             }
167             len--;
168         }
169         return len;
170     }

172     /**
173      * Returns the length of a String prefix with tabs expanded.
174      * Each tab is counted as the number of characters is takes to
175      * jump to the next tab stop.
176      * @param string the input String
177      * @param toIdx index in string (exclusive) where the calculation stops
178      * @param tabWidth the distance between tab stop position.
179      * @return the length of string.substring(0, toIdx) with tabs expanded.
180      */
181     public static int lengthExpandedTabs(String string,
182                                          int toIdx,
183                                          int tabWidth) {
184         int len = 0;
185         for (int idx = 0; idx < toIdx; idx++) {
186             if (string.charAt(idx) == '\t') {
187                 len = (len / tabWidth + 1) * tabWidth;
188             }
189             else {
190                 len++;
191             }
192         }
193         return len;
194     }

196     /**
197      * Validates whether passed string is a valid pattern or not.
198      * @param pattern
199      *        string to validate
200      * @return true if the pattern is valid false otherwise
201      */
202     public static boolean isPatternValid(String pattern) {
203         try {
204             Pattern.compile(pattern);
205         }
206         catch (final PatternSyntaxException e) {
207             return false;
208         }
209         return true;
210     }

212     /**
213      * Helper method to create a regular expression.
214      * @param pattern the pattern to match
215      * @return a created regexp object
216      * @throws ConversionException if unable to create Pattern object.
217      **/
218     public static Pattern createPattern(String pattern)
219         throws ConversionException {
220         try {
221             return Pattern.compile(pattern);
222         }
223         catch (final PatternSyntaxException e) {
224             throw new ConversionException(
225                 "Failed to initialise regular expression " + pattern, e);
226         }
227     }

238     /**
239      * Constructs a normalized relative path between base directory and a given path.
240      * @param baseDirectory the base path to which given path is relativized
241      * @param path the path to relativize against base directory
242      * @return the relative normalized path between base directory and path or path if base
243      * directory is null
244      */
245     public static String relativizeAndNormalizePath(final String baseDirectory, final String path) {
246         if (baseDirectory == null) {
247             return path;
248         }
249         final Path pathAbsolute = Paths.get(path).normalize();
250         final Path pathBase = Paths.get(baseDirectory).normalize();
251         return pathBase.relativize(pathAbsolute).toString();
252     }

254     /**
255      * Tests if this string starts with the specified prefix.
256      * <p/>
257      * It is faster version of {@link String#startsWith(String)} optimized for one-character
258      * prefixes at the expense of some readability. Suggested by SimplifyStartsWith PMD rule:
259      * http://pmd.sourceforge.net/pmd-5.3.1/pmd-java/rules/java/optimizations.html#SimplifyStartsWith
260      *
261      * @param string the <code>String</code> to check
262      * @param prefix the prefix to find
263      * @return <code>true</code> if the <code>char</code> is a prefix of the given
264      * <code>String</code>; <code>false</code> otherwise.
265      */
266     public static boolean startsWithChar(String string, char prefix) {
267         return string.length() > 0 && string.charAt(0) == prefix;
268     }
269 
270     /**
271      * Tests if this string ends with the specified suffix.
272      * <p/>
273      * It is faster version of {@link String#endsWith(String)} optimized for one-character
274      * suffixes at the expense of some readability. Suggested by SimplifyStartsWith PMD rule:
275      * http://pmd.sourceforge.net/pmd-5.3.1/pmd-java/rules/java/optimizations.html#SimplifyStartsWith
276      *
277      * @param string the <code>String</code> to check
278      * @param suffix the suffix to find
279      * @return <code>true</code> if the <code>char</code> is a suffix of the given
280      * <code>String</code>; <code>false</code> otherwise.
281      */
282     public static boolean endsWithChar(String string, char suffix) {
283         return string.length() > 0 && string.charAt(string.length() - 1) == suffix;
284     }

286     /**
287      * Returns the name of a token for a given ID.
288      * @param iD the ID of the token name to get
289      * @return a token name
290      */
291     public static String getTokenName(int iD) {
292         if (iD > TOKEN_VALUE_TO_NAME.length - 1) {
293             throw new IllegalArgumentException("given id " + iD);
294         }
295         final String name = TOKEN_VALUE_TO_NAME[iD];
296         if (name == null) {
297             throw new IllegalArgumentException("given id " + iD);
298         }
299         return name;
300     }

302     /**
303      * Returns the ID of a token for a given name.
304      * @param name the name of the token ID to get
305      * @return a token ID
306      */
307     public static int getTokenId(String name) {
308         final Integer id = TOKEN_NAME_TO_VALUE.get(name);
309         if (id == null) {
310             throw new IllegalArgumentException("given name " + name);
311         }
312         return id.intValue();
313     }

315     /**
316      * Returns the short description of a token for a given name.
317      * @param name the name of the token ID to get
318      * @return a short description
319      */
320     public static String getShortDescription(String name) {
321         if (!TOKEN_NAME_TO_VALUE.containsKey(name)) {
322             throw new IllegalArgumentException("given name " + name);
323         }
324 
325         final String tokentypes =
326             "com.puppycrawl.tools.checkstyle.api.tokentypes";
327         final ResourceBundle bundle = ResourceBundle.getBundle(tokentypes);
328         return bundle.getString(name);
329     }

331     /**
332      * Is argument comment-related type (SINGLE_LINE_COMMENT,
333      * BLOCK_COMMENT_BEGIN, BLOCK_COMMENT_END, COMMENT_CONTENT).
334      * @param type
335      *        token type.
336      * @return true if type is comment-related type.
337      */
338     public static boolean isCommentType(int type) {
339         return type == TokenTypes.SINGLE_LINE_COMMENT
340                 || type == TokenTypes.BLOCK_COMMENT_BEGIN
341                 || type == TokenTypes.BLOCK_COMMENT_END
342                 || type == TokenTypes.COMMENT_CONTENT;
343     }

356     /**
357      * @param targetClass from which constructor is returned
358      * @param parameterTypes of constructor
359      * @return constructor of targetClass or {@link IllegalStateException} if any exception occurs
360      * @see Class#getConstructor(Class[])
361      */
362     public static Constructor<?> getConstructor(Class<?> targetClass, Class<?>... parameterTypes) {
363         try {
364             return targetClass.getConstructor(parameterTypes);
365         }
366         catch (NoSuchMethodException ex) {
367             throw new IllegalStateException(ex);
368         }
369     }

371     /**
372      * @param constructor to invoke
373      * @param parameters to pass to constructor
374      * @param <T> type of constructor
375      * @return new instance of class or {@link IllegalStateException} if any exception occurs
376      * @see Constructor#newInstance(Object...)
377      */
378     public static <T> T invokeConstructor(Constructor<T> constructor, Object... parameters) {
379         try {
380             return constructor.newInstance(parameters);
381         }
382         catch (InstantiationException | IllegalAccessException | InvocationTargetException ex) {
383             throw new IllegalStateException(ex);
384         }
385     }

387     /**
388      * Gets the value of a static or instance field of type int or of another primitive type
389      * convertible to type int via a widening conversion. Does not throw any checked exceptions.
390      * @param field from which the int should be extracted
391      * @param object to extract the int value from
392      * @return the value of the field converted to type int
393      * @throws IllegalStateException if this Field object is enforcing Java language access control
394      *         and the underlying field is inaccessible
395      * @see Field#getInt(Object)
396      */
397     public static int getIntFromField(Field field, Object object) {
398         try {
399             return field.getInt(object);
400         }
401         catch (final IllegalAccessException exception) {
402             throw new IllegalStateException(exception);
403         }
