41      /** maps from a token name to value */
42      private static final ImmutableMap<String, Integer> TOKEN_NAME_TO_VALUE;
43      /** maps from a token value to name */
44      private static final String[] TOKEN_VALUE_TO_NAME;
45  
46      // initialise the constants
47      static {
48          final ImmutableMap.Builder<String, Integer> builder =
49                  ImmutableMap.builder();
50          final Field[] fields = TokenTypes.class.getDeclaredFields();
51          String[] tempTokenValueToName = new String[0];
52          for (final Field f : fields) {
53              // Only process the int declarations.
54              if (f.getType() != Integer.TYPE) {
55                  continue;
56              }
57  
58              final String name = f.getName();
59              try {
60                  final int tokenValue = f.getInt(name);
61                  builder.put(name, tokenValue);
62                  if (tokenValue > tempTokenValueToName.length - 1) {
63                      final String[] temp = new String[tokenValue + 1];
64                      System.arraycopy(tempTokenValueToName, 0,
65                              temp, 0, tempTokenValueToName.length);
66                      tempTokenValueToName = temp;
67                  }
68                  tempTokenValueToName[tokenValue] = name;
69              }
70              catch (final IllegalArgumentException | IllegalAccessException e) {
71                  throw new IllegalStateException(
72                          "Failed to instantiate collection of Java tokens", e);
73              }
74          }
75  
76          TOKEN_NAME_TO_VALUE = builder.build();
77          TOKEN_VALUE_TO_NAME = tempTokenValueToName;
78      }

85      /**
86       * Returns whether the file extension matches what we are meant to
87       * process.
88       * @param file the file to be checked.
89       * @param fileExtensions files extensions, empty property in config makes it matches to all.
90       * @return whether there is a match.
91       */
92      public static boolean fileExtensionMatches(File file, String... fileExtensions) {
93          boolean result = false;
94          if (fileExtensions == null || fileExtensions.length == 0) {
95              result = true;
96          }
97          else {
98              // normalize extensions so all of them have a leading dot
99              final String[] withDotExtensions = new String[fileExtensions.length];
100             for (int i = 0; i < fileExtensions.length; i++) {
101                 final String extension = fileExtensions[i];
102                 if (startsWithChar(extension, '.')) {
103                     withDotExtensions[i] = extension;
104                 }
105                 else {
106                     withDotExtensions[i] = "." + extension;
107                 }
108             }
109 
110             final String fileName = file.getName();
111             for (final String fileExtension : withDotExtensions) {
112                 if (fileName.endsWith(fileExtension)) {
113                     result = true;
114                 }
115             }
116         }
117 
118         return result;
119     }

121     /**
122      * Returns whether the specified string contains only whitespace up to the
123      * specified index.
124      *
125      * @param index index to check up to
126      * @param line the line to check
127      * @return whether there is only whitespace
128      */
129     public static boolean whitespaceBefore(int index, String line) {
130         for (int i = 0; i < index; i++) {
131             if (!Character.isWhitespace(line.charAt(i))) {
132                 return false;
133             }
134         }
135         return true;
136     }
137 
138     /**
139      * Returns the length of a string ignoring all trailing whitespace. It is a
140      * pity that there is not a trim() like method that only removed the
141      * trailing whitespace.
142      * @param line the string to process
143      * @return the length of the string ignoring all trailing whitespace
144      **/
145     public static int lengthMinusTrailingWhitespace(String line) {
146         int len = line.length();
147         for (int i = len - 1; i >= 0; i--) {
148             if (!Character.isWhitespace(line.charAt(i))) {
149                 break;
150             }
151             len--;
152         }
153         return len;
154     }

156     /**
157      * Returns the length of a String prefix with tabs expanded.
158      * Each tab is counted as the number of characters is takes to
159      * jump to the next tab stop.
160      * @param string the input String
161      * @param toIdx index in string (exclusive) where the calculation stops
162      * @param tabWidth the distance between tab stop position.
163      * @return the length of string.substring(0, toIdx) with tabs expanded.
164      */
165     public static int lengthExpandedTabs(String string,
166                                          int toIdx,
167                                          int tabWidth) {
168         int len = 0;
169         for (int idx = 0; idx < toIdx; idx++) {
170             if (string.charAt(idx) == '\t') {
171                 len = (len / tabWidth + 1) * tabWidth;
172             }
173             else {
174                 len++;
175             }
176         }
177         return len;
178     }

180     /**
181      * Validates whether passed string is a valid pattern or not.
182      * @param pattern
183      *        string to validate
184      * @return true if the pattern is valid false otherwise
185      */
186     public static boolean isPatternValid(String pattern) {
187         try {
188             Pattern.compile(pattern);
189         }
190         catch (final PatternSyntaxException e) {
191             return false;
192         }
193         return true;
194     }

196     /**
197      * Helper method to create a regular expression.
198      * @param pattern the pattern to match
199      * @return a created regexp object
200      * @throws ConversionException if unable to create Pattern object.
201      **/
202     public static Pattern createPattern(String pattern)
203         throws ConversionException {
204         try {
205             return Pattern.compile(pattern);
206         }
207         catch (final PatternSyntaxException e) {
208             throw new ConversionException(
209                 "Failed to initialise regular expression " + pattern, e);
210         }
211     }

213     /**
214      * @param type the fully qualified name. Cannot be null
215      * @return the base class name from a fully qualified name
216      */
217     public static String baseClassname(String type) {
218         final int i = type.lastIndexOf('.');
219         return i == -1 ? type : type.substring(i + 1);
220     }

222     /**
223      * Constructs a normalized relative path between base directory and a given path.
224      * @param baseDirectory the base path to which given path is relativized
225      * @param path the path to relativize against base directory
226      * @return the relative normalized path between base directory and path or path if base
227      * directory is null
228      */
229     public static String relativizeAndNormalizePath(final String baseDirectory, final String path) {
230         if (baseDirectory == null) {
231             return path;
232         }
233         final Path pathAbsolute = Paths.get(path).normalize();
234         final Path pathBase = Paths.get(baseDirectory).normalize();
235         return pathBase.relativize(pathAbsolute).toString();
236     }

238     /**
239      * Tests if this string starts with the specified prefix.
240      * <p/>
241      * It is faster version of {@link String#startsWith(String)} optimized for one-character
242      * prefixes at the expense of some readability. Suggested by SimplifyStartsWith PMD rule:
243      * http://pmd.sourceforge.net/pmd-5.3.1/pmd-java/rules/java/optimizations.html#SimplifyStartsWith
244      *
245      * @param string the <code>String</code> to check
246      * @param prefix the prefix to find
247      * @return <code>true</code> if the <code>char</code> is a prefix of the given
248      * <code>String</code>; <code>false</code> otherwise.
249      */
250     public static boolean startsWithChar(String string, char prefix) {
251         return string.length() > 0 && string.charAt(0) == prefix;
252     }
253 
254     /**
255      * Tests if this string ends with the specified suffix.
256      * <p/>
257      * It is faster version of {@link String#endsWith(String)} optimized for one-character
258      * suffixes at the expense of some readability. Suggested by SimplifyStartsWith PMD rule:
259      * http://pmd.sourceforge.net/pmd-5.3.1/pmd-java/rules/java/optimizations.html#SimplifyStartsWith
260      *
261      * @param string the <code>String</code> to check
262      * @param suffix the suffix to find
263      * @return <code>true</code> if the <code>char</code> is a suffix of the given
264      * <code>String</code>; <code>false</code> otherwise.
265      */
266     public static boolean endsWithChar(String string, char suffix) {
267         return string.length() > 0 && string.charAt(string.length() - 1) == suffix;
268     }

270     /**
271      * Returns the name of a token for a given ID.
272      * @param iD the ID of the token name to get
273      * @return a token name
274      */
275     public static String getTokenName(int iD) {
276         if (iD > TOKEN_VALUE_TO_NAME.length - 1) {
277             throw new IllegalArgumentException("given id " + iD);
278         }
279         final String name = TOKEN_VALUE_TO_NAME[iD];
280         if (name == null) {
281             throw new IllegalArgumentException("given id " + iD);
282         }
283         return name;
284     }
285 
286     /**
287      * Returns the ID of a token for a given name.
288      * @param name the name of the token ID to get
289      * @return a token ID
290      */
291     public static int getTokenId(String name) {
292         final Integer id = TOKEN_NAME_TO_VALUE.get(name);
293         if (id == null) {
294             throw new IllegalArgumentException("given name " + name);
295         }
296         return id.intValue();
297     }

299     /**
300      * Returns the short description of a token for a given name.
301      * @param name the name of the token ID to get
302      * @return a short description
303      */
304     public static String getShortDescription(String name) {
305         if (!TOKEN_NAME_TO_VALUE.containsKey(name)) {
306             throw new IllegalArgumentException("given name " + name);
307         }
308 
309         final String tokentypes =
310             "com.puppycrawl.tools.checkstyle.api.tokentypes";
311         final ResourceBundle bundle = ResourceBundle.getBundle(tokentypes);
312         return bundle.getString(name);
313     }

315     /**
316      * Is argument comment-related type (SINGLE_LINE_COMMENT,
317      * BLOCK_COMMENT_BEGIN, BLOCK_COMMENT_END, COMMENT_CONTENT).
318      * @param type
319      *        token type.
320      * @return true if type is comment-related type.
321      */
322     public static boolean isCommentType(int type) {
323         return type == TokenTypes.SINGLE_LINE_COMMENT
324                 || type == TokenTypes.BLOCK_COMMENT_BEGIN
325                 || type == TokenTypes.BLOCK_COMMENT_END
326                 || type == TokenTypes.COMMENT_CONTENT;
327     }

329     /**
330      * Is argument comment-related type name (SINGLE_LINE_COMMENT,
331      * BLOCK_COMMENT_BEGIN, BLOCK_COMMENT_END, COMMENT_CONTENT).
332      * @param type
333      *        token type name.
334      * @return true if type is comment-related type name.
335      */
336     public static boolean isCommentType(String type) {
337         return isCommentType(getTokenId(type));
